def is_palindrome(num: int) -> bool:
    """Check if a number is a palindrome."""
    return str(num) == str(num)[::-1]

def next_smallest_palindrome(num: int) -> int:
    """Find the next smallest palindrome greater than the specified number."""
    # Start searching from the next number
    candidate = num + 1
    
    while True:
        if is_palindrome(candidate):
            return candidate
        candidate += 1

# Example usage
if __name__ == "__main__":
    number = 123
    print(f"The next smallest palindrome after {number} is {next_smallest_palindrome(number)}.")
