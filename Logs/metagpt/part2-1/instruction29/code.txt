def count_substrings(s: str) -> int:
    """
    Count all substrings starting and ending with the same characters.

    Args:
        s (str): The input string.

    Returns:
        int: The count of substrings that start and end with the same character.
    """
    n = len(s)
    count = 0

    # Iterate through each character in the string
    for i in range(n):
        # Initialize a variable to track the number of occurrences of the current character
        char_count = 0
        
        # Check for substrings starting from the current character
        for j in range(i, n):
            if s[j] == s[i]:
                char_count += 1
                count += char_count

    return count

# Example usage
if __name__ == "__main__":
    test_string = "abcab"
    result = count_substrings(test_string)
    print(f"Count of substrings starting and ending with the same characters: {result}")
