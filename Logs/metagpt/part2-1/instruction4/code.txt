
def count_ways_to_fill_dominoes(n):
    """
    Calculate the number of ways to fill a 3 x n board with 2 x 1 dominoes.

    Args:
        n (int): The width of the board (n columns).

    Returns:
        int: The number of ways to fill the board.
    """
    if n < 0:
        return 0
    elif n == 0:
        return 1
    elif n == 1:
        return 0
    elif n == 2:
        return 3
    elif n == 3:
        return 0

    # Initialize a list to store the number of ways for each width
    dp = [0] * (n + 1)
    dp[0] = 1  # Base case: 1 way to fill a 3 x 0 board
    dp[1] = 0  # No way to fill a 3 x 1 board
    dp[2] = 3  # 3 ways to fill a 3 x 2 board
    dp[3] = 0  # No way to fill a 3 x 3 board

    # Fill the dp array for widths greater than 3
    for i in range(4, n + 1):
        dp[i] = 4 * dp[i - 2] + dp[i - 4]

    return dp[n]

# Example usage:
if __name__ == "__main__":
    n = 5  # Change this value to test with different board widths
    print(f"The number of ways to fill a 3 x {n} board is: {count_ways_to_fill_dominoes(n)}")
