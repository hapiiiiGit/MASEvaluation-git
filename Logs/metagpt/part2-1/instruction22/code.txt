def max_sum(list_of_lists):
    """
    Find the maximum sum of elements of a list in a list of lists.

    Args:
        list_of_lists (list of list of int): A list containing sublists of integers.

    Returns:
        int: The maximum sum of elements from any sublist. If the input is empty, returns 0.
    """
    if not list_of_lists:
        return 0

    max_sum_value = float('-inf')

    for sublist in list_of_lists:
        current_sum = sum(sublist)
        if current_sum > max_sum_value:
            max_sum_value = current_sum

    return max_sum_value
