def check_k_elements(tuple_list, k):
    """
    Check if the given list of tuples contains all k elements.

    Args:
        tuple_list (list of tuple): A list where each element is a tuple.
        k (int): The number of unique elements to check for.

    Returns:
        bool: True if all k elements are present, False otherwise.
    """
    # Create a set to store unique elements
    unique_elements = set()

    # Iterate through each tuple in the list
    for tup in tuple_list:
        # Add each element of the tuple to the set
        unique_elements.update(tup)

    # Check if the number of unique elements is at least k
    return len(unique_elements) >= k
