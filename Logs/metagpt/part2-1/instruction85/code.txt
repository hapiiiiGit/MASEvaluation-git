def centered_hexagonal_number(n):
    """
    Calculate the nth centered hexagonal number.

    A centered hexagonal number is given by the formula:
    H(n) = 3n(n - 1) + 1

    Args:
        n (int): The index of the centered hexagonal number to calculate.

    Returns:
        int: The nth centered hexagonal number.
    
    Raises:
        ValueError: If n is not a positive integer.
    """
    if n <= 0:
        raise ValueError("n must be a positive integer.")
    
    return 3 * n * (n - 1) + 1

# Example usage:
if __name__ == "__main__":
    n = 5  # Example input
    print(f"The {n}th centered hexagonal number is: {centered_hexagonal_number(n)}")
