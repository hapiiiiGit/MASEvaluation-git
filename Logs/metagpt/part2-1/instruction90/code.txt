def check_substring(substring, string_list):
    """
    Check if a substring is present in any of the strings in the given list.

    Args:
        substring (str): The substring to search for.
        string_list (list of str): The list of strings to search within.

    Returns:
        bool: True if the substring is found in any of the strings, False otherwise.
    """
    for string in string_list:
        if substring in string:
            return True
    return False

# Example usage
if __name__ == "__main__":
    sample_list = ["hello world", "python programming", "substring search"]
    result = check_substring("python", sample_list)
    print(f"Is the substring present? {result}")
