def count_odd_set_bits(n):
    """
    Count the number of integers from 1 to n (inclusive) that have an odd number of set bits.

    Args:
        n (int): The upper limit of the range (inclusive).

    Returns:
        int: The count of integers with an odd number of set bits.
    """
    def count_set_bits(x):
        """
        Count the number of set bits in the binary representation of x.

        Args:
            x (int): The integer to count set bits for.

        Returns:
            int: The number of set bits in x.
        """
        count = 0
        while x:
            count += x & 1
            x >>= 1
        return count

    odd_count = 0
    for i in range(1, n + 1):
        if count_set_bits(i) % 2 == 1:
            odd_count += 1

    return odd_count

# Example usage:
if __name__ == "__main__":
    n = 10
    result = count_odd_set_bits(n)
    print(f"The number of integers with an odd number of set bits from 1 to {n} is: {result}")
