import unittest
from undulating_number import is_undulating_number

class TestUndulatingNumber(unittest.TestCase):

    def test_undulating_number(self):
        self.assertTrue(is_undulating_number(121))  # 121 is undulating
        self.assertTrue(is_undulating_number(232))  # 232 is undulating
        self.assertTrue(is_undulating_number(12121))  # 12121 is undulating
        self.assertFalse(is_undulating_number(123))  # 123 is not undulating
        self.assertFalse(is_undulating_number(112))  # 112 is not undulating
        self.assertTrue(is_undulating_number(1))  # Single digit number is undulating

if __name__ == '__main__':
    unittest.main()

def is_undulating_number(number):
    """
    Check if the given number is undulating.

    An undulating number is defined as a number where the digits 
    alternate between increasing and decreasing. For example, 
    121, 232, and 12121 are undulating numbers, while 123 and 112 
    are not.

    Args:
        number (int): The number to check.

    Returns:
        bool: True if the number is undulating, False otherwise.
    """
    # Convert the number to a string to easily access each digit
    num_str = str(number)

    # If the number has less than 2 digits, it is undulating by default
    if len(num_str) < 2:
        return True

    # Initialize a variable to track the direction of change
    direction = None

    for i in range(1, len(num_str)):
        # Compare the current digit with the previous one
        if num_str[i] > num_str[i - 1]:
            current_direction = 'up'
        elif num_str[i] < num_str[i - 1]:
            current_direction = 'down'
        else:
            # If two consecutive digits are the same, it's not undulating
            return False

        # If we have a direction set, check if it changes
        if direction is not None and direction == current_direction:
            return False

        # Update the direction for the next iteration
        direction = current_direction

    return True
