
def differ_at_one_bit_position(num1: int, num2: int) -> bool:
    """
    Check whether two numbers differ at exactly one bit position.

    Args:
        num1 (int): The first integer.
        num2 (int): The second integer.

    Returns:
        bool: True if the two numbers differ at exactly one bit position, False otherwise.
    """
    # Calculate the XOR of the two numbers
    xor_result = num1 ^ num2
    
    # Check if xor_result has exactly one bit set
    # This is true if xor_result is not zero and (xor_result & (xor_result - 1)) is zero
    return xor_result != 0 and (xor_result & (xor_result - 1)) == 0

# Example usage
if __name__ == "__main__":
    print(differ_at_one_bit_position(5, 4))  # Output: True
    print(differ_at_one_bit_position(5, 3))  # Output: False
