def get_n_smallest(data, n):
    """
    Get the n smallest items from a dataset.

    Args:
        data (iterable): The dataset from which to retrieve the smallest items.
        n (int): The number of smallest items to retrieve.

    Returns:
        list: A list containing the n smallest items from the dataset.
    
    Raises:
        ValueError: If n is less than 1 or greater than the length of the dataset.
    """
    if n < 1:
        raise ValueError("n must be at least 1.")
    if n > len(data):
        raise ValueError("n must not be greater than the length of the dataset.")

    # Convert the dataset to a list if it's not already
    data_list = list(data)

    # Sort the data and return the first n elements
    return sorted(data_list)[:n]

import unittest
from get_n_smallest import get_n_smallest

class TestGetNSmallest(unittest.TestCase):
    def test_basic_functionality(self):
        self.assertEqual(get_n_smallest([5, 3, 1, 4, 2], 3), [1, 2, 3])
        self.assertEqual(get_n_smallest([10, 20, 30, 40, 50], 2), [10, 20])
        self.assertEqual(get_n_smallest([-1, -2, -3, -4, -5], 2), [-5, -4])

    def test_n_equals_one(self):
        self.assertEqual(get_n_smallest([5, 3, 1, 4, 2], 1), [1])
        self.assertEqual(get_n_smallest([10], 1), [10])

    def test_n_equals_length_of_data(self):
        self.assertEqual(get_n_smallest([5, 3, 1, 4, 2], 5), [1, 2, 3, 4, 5])

    def test_invalid_n(self):
        with self.assertRaises(ValueError):
            get_n_smallest([5, 3, 1, 4, 2], 0)
        with self.assertRaises(ValueError):
            get_n_smallest([5, 3, 1, 4, 2], 6)

    def test_empty_data(self):
        with self.assertRaises(ValueError):
            get_n_smallest([], 1)

if __name__ == '__main__':
    unittest.main()
