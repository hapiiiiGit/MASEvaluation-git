def count_substrings_with_sum_equal_length(s):
    def is_valid_substring(sub):
        return sum(int(digit) for digit in sub) == len(sub)

    count = 0
    for i in range(len(s)):
        for j in range(i+1, len(s)+1):
            if is_valid_substring(s[i:j]):
                count += 1

    return count

# Test the function
s = "122"
print(count_substrings_with_sum_equal_length(s))  # Output: 3
