import re
from typing import List

def find_long_words(input_string: str) -> List[str]:
    """
    Find all words in the input string that are at least 4 characters long.

    Args:
        input_string (str): The string to search for long words.

    Returns:
        List[str]: A list of words that are at least 4 characters long.
    """
    # Use regex to find all words with at least 4 characters
    long_words = re.findall(r'\b\w{4,}\b', input_string)
    return long_words
