def find_tuples(tuples_list, k):
    """
    Find all tuples in the given list where all elements are divisible by k.

    Args:
        tuples_list (list of tuple): A list of tuples containing integers.
        k (int): The divisor to check for divisibility.

    Returns:
        list of tuple: A list of tuples where all elements are divisible by k.
    """
    if k == 0:
        raise ValueError("The divisor k cannot be zero.")

    result = []
    for tup in tuples_list:
        if all(isinstance(x, int) and x % k == 0 for x in tup):
            result.append(tup)
    
    return result

# Example usage:
if __name__ == "__main__":
    sample_tuples = [(2, 4, 6), (3, 5, 7), (10, 20, 30), (1, 2, 3)]
    k = 2
    divisible_tuples = find_tuples(sample_tuples, k)
    print(divisible_tuples)  # Output: [(2, 4, 6), (10, 20, 30)]
