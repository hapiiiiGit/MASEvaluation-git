def find_multiples(n, m):
    """
    Generate a list of the first m multiples of n.

    Args:
        n (int): The number for which multiples are to be found.
        m (int): The number of multiples to find.

    Returns:
        list: A list containing the first m multiples of n.
    """
    multiples = []
    for i in range(1, m + 1):
        multiples.append(n * i)
    return multiples

# Example usage:
if __name__ == "__main__":
    n = 5  # Example number
    m = 10  # Example count of multiples
    result = find_multiples(n, m)
    print(f"The first {m} multiples of {n} are: {result}")
