def comb_sort(arr):
    """
    Sorts a list of elements using the comb sort algorithm.

    Args:
        arr (list): The list of elements to be sorted.

    Returns:
        list: The sorted list of elements.
    """
    # Initialize the gap
    gap = len(arr)
    shrink = 1.3  # Shrink factor
    sorted = False

    while not sorted:
        # Update the gap for the next comb
        gap = int(gap / shrink)
        if gap < 1:
            gap = 1
        
        sorted = True  # Assume the list is sorted

        # Compare elements at the current gap
        for i in range(len(arr) - gap):
            if arr[i] > arr[i + gap]:
                # Swap the elements
                arr[i], arr[i + gap] = arr[i + gap], arr[i]
                sorted = False  # We made a swap, so the list is not sorted yet

    return arr
