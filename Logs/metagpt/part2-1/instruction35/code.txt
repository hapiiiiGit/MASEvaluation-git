def find_nth_digit(numerator, denominator, n):
    if numerator < 0 or denominator <= 0 or n <= 0:
        return "Invalid input"

    fraction = numerator / denominator
    fraction_str = str(fraction).split('.')[1]

    if n > len(fraction_str):
        return "Nth digit is out of range"

    return int(fraction_str[n-1])

# Example Usage
numerator = 22
denominator = 7
n = 3
result = find_nth_digit(numerator, denominator, n)
print(result)
