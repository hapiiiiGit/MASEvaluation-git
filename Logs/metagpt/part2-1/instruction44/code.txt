from math import gcd

def find_gcd(arr):
    if not arr:
        return None
    
    result = arr[0]
    for i in range(1, len(arr)):
        result = gcd(result, arr[i])
    
    return result

# Test the function
print(find_gcd([12, 24, 36]))  # Expected output: 12
print(find_gcd([18, 27, 45]))  # Expected output: 9
print(find_gcd([5, 10, 15, 20]))  # Expected output: 5
print(find_gcd([7, 14, 21, 28]))  # Expected output: 7
print(find_gcd([]))  # Expected output: None
