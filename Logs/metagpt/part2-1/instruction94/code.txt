def min_length_sublist(lst, target_sum):
    """
    Find the minimum length of a contiguous sublist of `lst` that has a sum greater than or equal to `target_sum`.

    Args:
        lst (list of int): The list of integers to search within.
        target_sum (int): The target sum that the sublist must meet or exceed.

    Returns:
        int: The minimum length of the sublist that meets the criteria, or 0 if no such sublist exists.
    """
    n = len(lst)
    min_length = float('inf')
    current_sum = 0
    start = 0

    for end in range(n):
        current_sum += lst[end]

        while current_sum >= target_sum:
            min_length = min(min_length, end - start + 1)
            current_sum -= lst[start]
            start += 1

    return min_length if min_length != float('inf') else 0
