def counting_sort(arr):
    """
    Sorts an array using the counting sort algorithm.

    Args:
        arr (list of int): The array of integers to be sorted.

    Returns:
        list of int: The sorted array.
    """
    if not arr:
        return []

    # Find the maximum and minimum values in the array
    max_val = max(arr)
    min_val = min(arr)

    # Create a count array with a size based on the range of input values
    range_of_elements = max_val - min_val + 1
    count = [0] * range_of_elements

    # Store the count of each number in the count array
    for number in arr:
        count[number - min_val] += 1

    # Build the output array
    output = []
    for i in range(range_of_elements):
        output.extend([i + min_val] * count[i])

    return output
