def find_first_duplicate(arr):
    """
    Finds the first duplicate element in a given array of integers.

    Args:
        arr (list of int): The array of integers to search for duplicates.

    Returns:
        int: The first duplicate element if found, otherwise -1.
    """
    seen = set()
    for number in arr:
        if number in seen:
            return number
        seen.add(number)
    return -1
