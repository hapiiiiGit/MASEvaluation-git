class RectangularNumber:
    def find_nth_rectangular_number(self, n):
        if n <= 0:
            return None
        return n * (n + 1) // 2

# Example Usage
rect_num = RectangularNumber()
n = 5
result = rect_num.find_nth_rectangular_number(n)
print(f"The {n}-th rectangular number is: {result}")
