def split_string_at_lowercase(input_string):
    """
    Splits the input string at each lowercase letter.

    Args:
        input_string (str): The string to be split.

    Returns:
        list: A list of substrings split at lowercase letters.
    """
    result = []
    current_substring = []

    for char in input_string:
        if char.islower():
            if current_substring:
                result.append(''.join(current_substring))
                current_substring = []
        current_substring.append(char)

    if current_substring:
        result.append(''.join(current_substring))

    return result

# Example usage:
if __name__ == "__main__":
    test_string = "HelloWorldThisIsATest"
    print(split_string_at_lowercase(test_string))
