def contains_sublist(main_list, sublist):
    """
    Check if the main_list contains the given sublist.

    Args:
        main_list (list): The list in which to search for the sublist.
        sublist (list): The sublist to search for within the main_list.

    Returns:
        bool: True if the sublist is found within the main_list, False otherwise.
    """
    sublist_length = len(sublist)
    main_list_length = len(main_list)

    if sublist_length == 0:
        return True  # An empty sublist is always considered to be contained

    if sublist_length > main_list_length:
        return False  # A longer sublist cannot be contained in a shorter list

    for i in range(main_list_length - sublist_length + 1):
        if main_list[i:i + sublist_length] == sublist:
            return True

    return False
