import unittest
from set_odd_bits import set_odd_bits

class TestSetOddBits(unittest.TestCase):
    def test_set_odd_bits(self):
        # Test cases to validate the set_odd_bits function
        self.assertEqual(set_odd_bits(0b0000), 0b0101)  # 0b0000 -> 0b0101
        self.assertEqual(set_odd_bits(0b0001), 0b0101)  # 0b0001 -> 0b0101
        self.assertEqual(set_odd_bits(0b0010), 0b0110)  # 0b0010 -> 0b0110
        self.assertEqual(set_odd_bits(0b0011), 0b0111)  # 0b0011 -> 0b0111
        self.assertEqual(set_odd_bits(0b0100), 0b0101)  # 0b0100 -> 0b0101
        self.assertEqual(set_odd_bits(0b1111), 0b1111)  # 0b1111 -> 0b1111
        self.assertEqual(set_odd_bits(0b1010), 0b1110)  # 0b1010 -> 0b1110
        self.assertEqual(set_odd_bits(0b1100), 0b1101)  # 0b1100 -> 0b1101

if __name__ == '__main__':
    unittest.main()
