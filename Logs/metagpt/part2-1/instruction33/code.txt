def find_missing_number(arr):
    if not arr:
        return None
    
    left = 0
    right = len(arr) - 1
    
    while left <= right:
        mid = left + (right - left) // 2
        
        if arr[mid] != mid + 1:
            right = mid - 1
        else:
            left = mid + 1
    
    return left + 1
