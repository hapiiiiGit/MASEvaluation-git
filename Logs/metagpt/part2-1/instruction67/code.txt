def is_monotonic(array):
    """
    Check if the given array is monotonic.

    An array is monotonic if it is either entirely non-increasing or 
    non-decreasing.

    Args:
        array (list): A list of integers.

    Returns:
        bool: True if the array is monotonic, False otherwise.
    """
    if not array:
        return True  # An empty array is considered monotonic

    increasing = decreasing = True

    for i in range(1, len(array)):
        if array[i] > array[i - 1]:
            decreasing = False
        elif array[i] < array[i - 1]:
            increasing = False

    return increasing or decreasing

# Example usage:
if __name__ == "__main__":
    test_array = [1, 2, 2, 3]
    print(is_monotonic(test_array))  # Output: True

    test_array = [6, 5, 4, 4]
    print(is_monotonic(test_array))  # Output: True

    test_array = [1, 3, 2]
    print(is_monotonic(test_array))  # Output: False
