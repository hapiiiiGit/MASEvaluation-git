from typing import List, Union

class TwoDListExtractor:
    def __init__(self, two_d_list: List[List[Union[int, str]]]):
        self.two_d_list = two_d_list

    def extract_first_elements(self) -> List[Union[int, str]]:
        first_elements = [row[0] for row in self.two_d_list]
        return first_elements

    def extract_specified_elements(self, index: int) -> List[Union[int, str]]:
        specified_elements = [row[index] for row in self.two_d_list]
        return specified_elements

# Example Usage
two_d_list = [
    [1, 'a', 3],
    [4, 'b', 6],
    [7, 'c', 9]
]

extractor = TwoDListExtractor(two_d_list)
first_elements = extractor.extract_first_elements()
specified_elements = extractor.extract_specified_elements(1)

print("First Elements:", first_elements)
print("Specified Elements at index 1:", specified_elements)
