def character_sum(input_string: str) -> str:
    """
    Calculate the character resulting from the sum of all characters in the input string.

    Args:
        input_string (str): The string from which to calculate the character sum.

    Returns:
        str: A single character that represents the sum of all characters in the input string.
    """
    if not input_string:
        return ""

    # Calculate the sum of the ASCII values of the characters
    total = sum(ord(char) for char in input_string)

    # Find the character corresponding to the total modulo 256
    result_char = chr(total % 256)

    return result_char
