
def find_similar_elements(tuple_list1, tuple_list2):
    """
    Find similar elements from two lists of tuples.

    Args:
        tuple_list1 (list of tuple): The first list of tuples.
        tuple_list2 (list of tuple): The second list of tuples.

    Returns:
        list of tuple: A list containing tuples that are present in both input lists.
    """
    # Convert the second list of tuples to a set for faster lookup
    set_tuple_list2 = set(tuple_list2)
    
    # Use a list comprehension to find common tuples
    similar_elements = [tup for tup in tuple_list1 if tup in set_tuple_list2]
    
    return similar_elements

# Example usage:
if __name__ == "__main__":
    list1 = [(1, 2), (3, 4), (5, 6)]
    list2 = [(3, 4), (7, 8), (1, 2)]
    result = find_similar_elements(list1, list2)
    print(result)  # Output: [(1, 2), (3, 4)]
