def remove_digits(strings):
    """
    Remove all digits from a list of strings.

    Args:
        strings (list of str): A list of strings from which digits will be removed.

    Returns:
        list of str: A new list of strings with all digits removed.
    """
    return [''.join(char for char in string if not char.isdigit()) for string in strings]

# Example usage:
if __name__ == "__main__":
    input_strings = ["hello123", "world456", "test789"]
    output_strings = remove_digits(input_strings)
    print(output_strings)  # Output: ['hello', 'world', 'test']
