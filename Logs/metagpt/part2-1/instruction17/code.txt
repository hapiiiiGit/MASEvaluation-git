def remove_characters(str1, str2):
    """
    Remove characters from the first string that are present in the second string.

    Args:
        str1 (str): The original string from which characters will be removed.
        str2 (str): The string containing characters to be removed from str1.

    Returns:
        str: A new string with characters from str1 removed that are present in str2.
    """
    # Create a set of characters to remove for efficient lookup
    chars_to_remove = set(str2)
    
    # Use a list comprehension to build the new string
    result = ''.join([char for char in str1 if char not in chars_to_remove])
    
    return result
