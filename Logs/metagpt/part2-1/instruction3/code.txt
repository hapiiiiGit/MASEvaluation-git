
import unittest
from largest_integers import find_largest_integers

class TestFindLargestIntegers(unittest.TestCase):
    def test_basic_functionality(self):
        self.assertEqual(find_largest_integers([1, 3, 5, 7, 9], 3), [9, 7, 5])
        self.assertEqual(find_largest_integers([10, 20, 30, 40, 50], 2), [50, 40])
        self.assertEqual(find_largest_integers([-1, -2, -3, -4, -5], 2), [-1, -2])

    def test_edge_cases(self):
        self.assertEqual(find_largest_integers([], 3), [])
        self.assertEqual(find_largest_integers([1, 2, 3], 0), [])
        self.assertEqual(find_largest_integers([1], 1), [1])
        self.assertEqual(find_largest_integers([5, 5, 5, 5], 2), [5, 5])

    def test_large_numbers(self):
        self.assertEqual(find_largest_integers([1000000, 999999, 888888, 777777], 2), [1000000, 999999])
        self.assertEqual(find_largest_integers([1, 2, 3, 4, 5, 10000000], 1), [10000000])

if __name__ == '__main__':
    unittest.main()

import heapq
from typing import List

def find_largest_integers(numbers: List[int], n: int) -> List[int]:
    """
    Find the largest integers from a given list of numbers using the heap queue algorithm.

    Args:
        numbers (List[int]): A list of integers from which to find the largest.
        n (int): The number of largest integers to return.

    Returns:
        List[int]: A list containing the n largest integers from the input list.
    """
    if n <= 0:
        return []
    
    # Use heapq to find the n largest integers
    largest_integers = heapq.nlargest(n, numbers)
    
    return largest_integers
