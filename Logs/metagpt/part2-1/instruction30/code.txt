import heapq
from collections import Counter

def top_k_frequent_elements(nums, k):
    count = Counter(nums)
    heap = [(-freq, num) for num, freq in count.items()]
    heapq.heapify(heap)
    return [heapq.heappop(heap)[1] for _ in range(k)]

# Example usage
nums = [1, 1, 1, 2, 2, 3]
k = 2
result = top_k_frequent_elements(nums, k)
print(result)
