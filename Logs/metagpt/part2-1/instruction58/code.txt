def find_nth_octagonal_number(n):
    """Calculate the nth octagonal number using the formula O(n) = n(3n - 2).

    Args:
        n (int): The position of the octagonal number to calculate. Must be a positive integer.

    Returns:
        int: The nth octagonal number, or a message indicating invalid input if n is not positive.
    """
    if n <= 0:
        return "Invalid input. Please provide a positive integer."

    return n * (3 * n - 2)
