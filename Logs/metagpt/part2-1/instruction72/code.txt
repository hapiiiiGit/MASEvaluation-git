import re

def split_string(input_string, delimiters):
    """
    Splits the input string using multiple delimiters defined in the delimiters string.

    Args:
        input_string (str): The string to be split.
        delimiters (str): A string containing all the delimiters to be used for splitting.

    Returns:
        list: A list of substrings obtained by splitting the input string.
    """
    # Create a regex pattern from the delimiters
    regex_pattern = '|'.join(map(re.escape, delimiters))
    
    # Use re.split to split the input string by the regex pattern
    return re.split(regex_pattern, input_string)

# Example usage
if __name__ == "__main__":
    test_string = "Hello, world! This is a test-string; with multiple: delimiters."
    delimiters = ",!;:- "
    result = split_string(test_string, delimiters)
    print(result)
