def sum_repeated_elements(arr):
    """
    Calculate the sum of repeated elements in the given array.

    Args:
        arr (list): A list of integers.

    Returns:
        int: The sum of repeated elements.
    """
    element_count = {}
    repeated_sum = 0

    # Count occurrences of each element
    for num in arr:
        if num in element_count:
            element_count[num] += 1
        else:
            element_count[num] = 1

    # Calculate the sum of elements that are repeated
    for num, count in element_count.items():
        if count > 1:
            repeated_sum += num * count

    return repeated_sum

# Example usage
if __name__ == "__main__":
    example_array = [1, 2, 3, 2, 4, 3, 5]
    result = sum_repeated_elements(example_array)
    print(f"The sum of repeated elements is: {result}")
