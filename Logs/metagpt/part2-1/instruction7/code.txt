def square_elements(input_list):
    """
    This function takes a list of numbers and returns a new list containing the squares of each number.
    
    Args:
        input_list (list): A list of numerical elements.
        
    Returns:
        list: A list containing the squares of the input elements.
    """
    return list(map(lambda x: x ** 2, input_list))

# Example usage:
if __name__ == "__main__":
    numbers = [1, 2, 3, 4, 5]
    squared_numbers = square_elements(numbers)
    print(squared_numbers)  # Output: [1, 4, 9, 16, 25]
