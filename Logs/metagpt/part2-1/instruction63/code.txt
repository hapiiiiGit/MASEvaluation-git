def sort_tuples(tuples_list):
    """
    Sorts a list of tuples based on the first element of each tuple.
    
    Args:
        tuples_list (list of tuple): A list containing tuples to be sorted.
        
    Returns:
        list of tuple: A new list containing the sorted tuples.
    """
    return sorted(tuples_list, key=lambda x: x[0])

# Example usage:
if __name__ == "__main__":
    sample_tuples = [(3, 'three'), (1, 'one'), (2, 'two')]
    sorted_tuples = sort_tuples(sample_tuples)
    print(sorted_tuples)  # Output: [(1, 'one'), (2, 'two'), (3, 'three')]
