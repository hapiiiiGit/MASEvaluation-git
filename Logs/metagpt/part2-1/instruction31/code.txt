def largest_prime_factor(n):
    """Return the largest prime factor of a given number n."""
    # Initialize the largest prime factor
    largest_factor = None
    
    # Check for number of 2s that divide n
    while n % 2 == 0:
        largest_factor = 2
        n //= 2
    
    # n must be odd at this point, so we can skip even numbers
    for i in range(3, int(n**0.5) + 1, 2):
        while n % i == 0:
            largest_factor = i
            n //= i
    
    # This condition is to check if n is a prime number greater than 2
    if n > 2:
        largest_factor = n
    
    return largest_factor

import unittest
from largest_prime_factor import largest_prime_factor

class TestLargestPrimeFactor(unittest.TestCase):
    def test_prime_number(self):
        self.assertEqual(largest_prime_factor(13), 13)

    def test_composite_number(self):
        self.assertEqual(largest_prime_factor(28), 7)
        self.assertEqual(largest_prime_factor(100), 5)

    def test_large_number(self):
        self.assertEqual(largest_prime_factor(13195), 29)
        self.assertEqual(largest_prime_factor(600851475143), 6857)

    def test_edge_cases(self):
        self.assertEqual(largest_prime_factor(1), None)
        self.assertEqual(largest_prime_factor(2), 2)
        self.assertEqual(largest_prime_factor(3), 3)
        self.assertEqual(largest_prime_factor(4), 2)

if __name__ == '__main__':
    unittest.main()
