from typing import List

def sort_matrix_by_row_sum(matrix: List[List[int]]) -> List[List[int]]:
    """
    Sorts a given matrix in ascending order according to the sum of its rows.

    Args:
        matrix (List[List[int]]): A 2D list representing the matrix to be sorted.

    Returns:
        List[List[int]]: A new matrix sorted by the sum of its rows.
    """
    # Calculate the sum of each row and pair it with the row
    row_sums = [(sum(row), row) for row in matrix]
    # Sort the rows based on the calculated sums
    sorted_rows = [row for _, row in sorted(row_sums, key=lambda x: x[0])]
    return sorted_rows

# Example Usage
if __name__ == "__main__":
    matrix = [
        [1, 3, 5],
        [2, 4, 6],
        [0, 8, 7]
    ]
    sorted_matrix = sort_matrix_by_row_sum(matrix)
    print(sorted_matrix)
