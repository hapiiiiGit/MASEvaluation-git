def sort_mixed_list(mixed_list):
    """
    Sorts a given mixed list of integers and strings.

    Args:
        mixed_list (list): A list containing integers and strings.

    Returns:
        list: A new list containing the sorted integers and strings.
    """
    # Separate integers and strings into two lists
    integers = [item for item in mixed_list if isinstance(item, int)]
    strings = [item for item in mixed_list if isinstance(item, str)]

    # Sort both lists
    sorted_integers = sorted(integers)
    sorted_strings = sorted(strings)

    # Combine the sorted lists
    sorted_list = sorted_integers + sorted_strings

    return sorted_list

# Example usage
if __name__ == "__main__":
    mixed_list = [3, "apple", 1, "banana", 2, "cherry"]
    sorted_result = sort_mixed_list(mixed_list)
    print(sorted_result)  # Output: [1, 2, 3, 'apple', 'banana', 'cherry']
