import math

def calculate_sphere_surface_area(radius):
    """
    Calculate the surface area of a sphere given its radius.

    Args:
        radius (float): The radius of the sphere.

    Returns:
        float: The surface area of the sphere.
    
    Raises:
        ValueError: If the radius is negative.
    """
    if radius < 0:
        raise ValueError("Radius cannot be negative.")
    
    surface_area = 4 * math.pi * (radius ** 2)
    return surface_area

import unittest
from geometry import calculate_sphere_surface_area

class TestGeometry(unittest.TestCase):

    def test_surface_area_positive_radius(self):
        # Test for positive radius
        radius = 3
        expected_area = 4 * 3.14159 * (3 ** 2)
        self.assertAlmostEqual(calculate_sphere_surface_area(radius), expected_area, places=5)

    def test_surface_area_zero_radius(self):
        # Test for zero radius
        radius = 0
        expected_area = 0
        self.assertAlmostEqual(calculate_sphere_surface_area(radius), expected_area, places=5)

    def test_surface_area_negative_radius(self):
        # Test for negative radius
        with self.assertRaises(ValueError):
            calculate_sphere_surface_area(-2)

if __name__ == '__main__':
    unittest.main()
