
import unittest
from min_cost_path import min_cost_path

class TestMinCostPath(unittest.TestCase):
    def test_min_cost_path(self):
        # Test case 1: Simple 2x2 matrix
        cost1 = [[1, 2], [3, 4]]
        self.assertEqual(min_cost_path(cost1, 1, 1), 7)

        # Test case 2: 3x3 matrix with varying costs
        cost2 = [[1, 3, 5], [2, 1, 2], [4, 3, 1]]
        self.assertEqual(min_cost_path(cost2, 2, 2), 7)

        # Test case 3: 3x3 matrix with higher costs
        cost3 = [[10, 10, 10], [10, 1, 10], [10, 10, 10]]
        self.assertEqual(min_cost_path(cost3, 2, 2), 31)

        # Test case 4: 1x1 matrix
        cost4 = [[5]]
        self.assertEqual(min_cost_path(cost4, 0, 0), 5)

        # Test case 5: Invalid position
        cost5 = [[1, 2], [3, 4]]
        self.assertEqual(min_cost_path(cost5, -1, -1), float('inf'))
        self.assertEqual(min_cost_path(cost5, 2, 2), float('inf'))

        # Test case 6: Larger matrix
        cost6 = [[1, 2, 3], [4, 8, 2], [1, 5, 3]]
        self.assertEqual(min_cost_path(cost6, 2, 2), 8)

if __name__ == '__main__':
    unittest.main()

def min_cost_path(cost, m, n):
    """
    Function to find the minimum cost path to reach (m, n) from (0, 0) in a given cost matrix.

    Args:
    cost (list of list of int): A 2D list representing the cost matrix.
    m (int): The row index of the destination cell.
    n (int): The column index of the destination cell.

    Returns:
    int: The minimum cost to reach (m, n) from (0, 0).
    """
    if not cost or m < 0 or n < 0 or m >= len(cost) or n >= len(cost[0]):
        return float('inf')

    # Create a 2D array to store the minimum cost to reach each cell
    dp = [[0 for _ in range(n + 1)] for _ in range(m + 1)]

    # Initialize the starting point
    dp[0][0] = cost[0][0]

    # Fill the first row
    for j in range(1, n + 1):
        dp[0][j] = dp[0][j - 1] + cost[0][j]

    # Fill the first column
    for i in range(1, m + 1):
        dp[i][0] = dp[i - 1][0] + cost[i][0]

    # Fill the rest of the dp array
    for i in range(1, m + 1):
        for j in range(1, n + 1):
            dp[i][j] = cost[i][j] + min(dp[i - 1][j], dp[i][j - 1])

    return dp[m][n]
