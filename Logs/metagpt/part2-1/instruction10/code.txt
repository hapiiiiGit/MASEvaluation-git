def remove_first_and_last_occurrence(input_string: str, char_to_remove: str) -> str:
    """
    Remove the first and last occurrence of a given character from the string.

    Args:
        input_string (str): The string from which to remove characters.
        char_to_remove (str): The character to remove from the first and last position.

    Returns:
        str: The modified string with the first and last occurrence of the character removed.
    """
    if not input_string or not char_to_remove:
        return input_string

    first_occurrence_index = input_string.find(char_to_remove)
    last_occurrence_index = input_string.rfind(char_to_remove)

    # If the character is not found or both occurrences are the same
    if first_occurrence_index == -1 or first_occurrence_index == last_occurrence_index:
        return input_string.replace(char_to_remove, '', 1) if first_occurrence_index != -1 else input_string

    # Remove the first occurrence
    modified_string = input_string[:first_occurrence_index] + input_string[first_occurrence_index + 1:]

    # Adjust the last occurrence index since the string has been modified
    last_occurrence_index -= 1  # Adjust for the removed character
    modified_string = modified_string[:last_occurrence_index] + modified_string[last_occurrence_index + 1:]

    return modified_string
