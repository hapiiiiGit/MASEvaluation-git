def zip_tuples(tuple1, tuple2):
    """
    Zips two given tuples into a single tuple of pairs.

    Args:
        tuple1 (tuple): The first tuple to zip.
        tuple2 (tuple): The second tuple to zip.

    Returns:
        tuple: A tuple containing pairs of elements from the input tuples.
               If the input tuples are of different lengths, the resulting
               tuple will be as long as the shorter input tuple.
    """
    return tuple(zip(tuple1, tuple2))

# Example usage:
if __name__ == "__main__":
    tuple_a = (1, 2, 3)
    tuple_b = ('a', 'b', 'c')
    zipped = zip_tuples(tuple_a, tuple_b)
    print(zipped)  # Output: ((1, 'a'), (2, 'b'), (3, 'c'))
