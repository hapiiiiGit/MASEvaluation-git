def find_duplicates(arr):
    """
    Function to determine if there are any duplicate elements in the given array of integers.

    Args:
        arr (list of int): The array of integers to check for duplicates.

    Returns:
        bool: True if duplicates are found, False otherwise.
    """
    seen = set()
    for num in arr:
        if num in seen:
            return True
        seen.add(num)
    return False
