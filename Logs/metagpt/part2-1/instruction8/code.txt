def min_rotations_to_same_string(s: str) -> int:
    """
    Find the minimum number of rotations required to get the same string.

    A rotation of a string is defined as moving the first character to the end of the string.
    For example, rotating "abc" gives "bca" and "cab".

    Args:
        s (str): The input string to analyze.

    Returns:
        int: The minimum number of rotations required to get the same string.
    """
    if not s:
        return 0  # If the string is empty, no rotations are needed.

    n = len(s)
    doubled_s = s + s  # Concatenate the string with itself to facilitate rotation checks.

    # Check for the first occurrence of the original string in the doubled string.
    for i in range(1, n):
        if doubled_s[i:i+n] == s:
            return i  # Return the index as the number of rotations.

    return n  # If no match found, it means the string is unique and requires n rotations.
