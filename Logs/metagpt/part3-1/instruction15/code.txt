# fruit_freshness_classifier.py

import os
import numpy as np
import pandas as pd
import tensorflow as tf
from tensorflow.keras import layers, models
from sklearn.model_selection import train_test_split
from sklearn.preprocessing import LabelEncoder
from tensorflow.keras.preprocessing.image import ImageDataGenerator
import streamlit as st
from PIL import Image

# Constants
IMAGE_SIZE = (150, 150)
BATCH_SIZE = 32
EPOCHS = 20
DATASET_PATH = 'path_to_your_dataset'  # Update this path to your dataset

# Load and preprocess dataset
def load_data(dataset_path):
    # Assuming the dataset is structured in subdirectories for each class
    data_gen = ImageDataGenerator(rescale=1./255, validation_split=0.2)
    
    train_data = data_gen.flow_from_directory(
        dataset_path,
        target_size=IMAGE_SIZE,
        batch_size=BATCH_SIZE,
        class_mode='categorical',
        subset='training'
    )
    
    validation_data = data_gen.flow_from_directory(
        dataset_path,
        target_size=IMAGE_SIZE,
        batch_size=BATCH_SIZE,
        class_mode='categorical',
        subset='validation'
    )
    
    return train_data, validation_data

# Build CNN model
def build_model(num_classes):
    model = models.Sequential([
        layers.Conv2D(32, (3, 3), activation='relu', input_shape=(IMAGE_SIZE[0], IMAGE_SIZE[1], 3)),
        layers.MaxPooling2D(pool_size=(2, 2)),
        layers.Conv2D(64, (3, 3), activation='relu'),
        layers.MaxPooling2D(pool_size=(2, 2)),
        layers.Conv2D(128, (3, 3), activation='relu'),
        layers.MaxPooling2D(pool_size=(2, 2)),
        layers.Flatten(),
        layers.Dense(128, activation='relu'),
        layers.Dense(num_classes, activation='softmax')
    ])
    
    model.compile(optimizer='adam', loss='categorical_crossentropy', metrics=['accuracy'])
    return model

# Train the model
def train_model(model, train_data, validation_data):
    model.fit(train_data, validation_data=validation_data, epochs=EPOCHS)
    return model

# Predict freshness and remaining days
def predict_freshness(model, image):
    image = image.resize(IMAGE_SIZE)
    image = np.array(image) / 255.0
    image = np.expand_dims(image, axis=0)
    predictions = model.predict(image)
    return predictions

# Streamlit GUI
def main():
    st.title("Fruit Freshness Classifier")
    st.write("Upload a fruit image to classify its freshness and predict remaining days before expiry.")
    
    uploaded_file = st.file_uploader("Choose an image...", type=["jpg", "jpeg", "png"])
    
    if uploaded_file is not None:
        image = Image.open(uploaded_file)
        st.image(image, caption='Uploaded Image', use_column_width=True)
        
        # Load data and model
        train_data, validation_data = load_data(DATASET_PATH)
        model = build_model(num_classes=len(train_data.class_indices))
        model = train_model(model, train_data, validation_data)
        
        # Make prediction
        predictions = predict_freshness(model, image)
        predicted_class = np.argmax(predictions, axis=1)
        st.write(f"Predicted Class: {predicted_class}")

if __name__ == "__main__":
    main()
