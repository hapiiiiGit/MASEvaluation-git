# Python script to extract text from screenshots

# Import the necessary libraries
from PIL import Image
import pytesseract

# Function to extract text from a screenshot image
def extract_text_from_screenshot(image_path):
    # Open the image file
    img = Image.open(image_path)
    
    # Use pytesseract to extract text from the image
    text = pytesseract.image_to_string(img)
    
    return text

# Main function to demonstrate text extraction from a screenshot
def main():
    # Path to the screenshot image
    screenshot_path = "/path/to/screenshot.png"
    
    # Extract text from the screenshot
    extracted_text = extract_text_from_screenshot(screenshot_path)
    
    # Print the extracted text
    print("Extracted Text:")
    print(extracted_text)

if __name__ == "__main__":
    main()
