
def fib4(n: int) -> int:
    """The Fib4 number sequence is a sequence similar to the Fibonacci sequence that's defined as follows:
    fib4(0) -> 0
    fib4(1) -> 0
    fib4(2) -> 2
    fib4(3) -> 0
    fib4(n) -> fib4(n-1) + fib4(n-2) + fib4(n-3) + fib4(n-4).
    Please write a function to efficiently compute the n-th element of the fib4 number sequence.  Do not use recursion.
    >>> fib4(5)
    4
    >>> fib4(6)
    8
    >>> fib4(7)
    14
    """
    if n == 0:
        return 0
    elif n == 1:
        return 0
    elif n == 2:
        return 2
    elif n == 3:
        return 0

    # Initialize an array to store the computed values
    fib_values = [0] * (n + 1)
    fib_values[0] = 0
    fib_values[1] = 0
    fib_values[2] = 2
    fib_values[3] = 0

    # Compute the values iteratively
    for i in range(4, n + 1):
        fib_values[i] = fib_values[i - 1] + fib_values[i - 2] + fib_values[i - 3] + fib_values[i - 4]

    return fib_values[n]

import unittest

class TestFib4(unittest.TestCase):
    def test_fib4(self):
        # Test base cases
        self.assertEqual(fib4(0), 0)
        self.assertEqual(fib4(1), 0)
        self.assertEqual(fib4(2), 2)
        self.assertEqual(fib4(3), 0)

        # Test additional cases
        self.assertEqual(fib4(4), 2)  # fib4(4) = fib4(3) + fib4(2) + fib4(1) + fib4(0) = 0 + 2 + 0 + 0 = 2
        self.assertEqual(fib4(5), 4)  # fib4(5) = fib4(4) + fib4(3) + fib4(2) + fib4(1) = 2 + 0 + 2 + 0 = 4
        self.assertEqual(fib4(6), 8)  # fib4(6) = fib4(5) + fib4(4) + fib4(3) + fib4(2) = 4 + 2 + 0 + 2 = 8
        self.assertEqual(fib4(7), 14) # fib4(7) = fib4(6) + fib4(5) + fib4(4) + fib4(3) = 8 + 4 + 2 + 0 = 14

        # Test larger cases
        self.assertEqual(fib4(8), 28) # fib4(8) = fib4(7) + fib4(6) + fib4(5) + fib4(4) = 14 + 8 + 4 + 2 = 28
        self.assertEqual(fib4(9), 54) # fib4(9) = fib4(8) + fib4(7) + fib4(6) + fib4(5) = 28 + 14 + 8 + 4 = 54

if __name__ == '__main__':
    unittest.main()
