
def is_simple_power(x, n):
    """Check if x is a simple power of n.

    A number x is a simple power of n if there exists an integer k such that n**k = x.
    For example:
    is_simple_power(1, 4) => True
    is_simple_power(2, 2) => True
    is_simple_power(8, 2) => True
    is_simple_power(3, 2) => False
    is_simple_power(3, 1) => False
    is_simple_power(5, 3) => False

    Args:
        x (int): The number to check.
        n (int): The base number.

    Returns:
        bool: True if x is a simple power of n, False otherwise.
    """
    if n <= 1 or x <= 0:
        return x == 1  # Only 1 is a power of any number <= 1
    if x == 1:
        return True  # 1 is a power of any number

    power = 1
    while True:
        result = n ** power
        if result == x:
            return True
        elif result > x:
            return False
        power += 1

import unittest
from simple_power import is_simple_power

class TestIsSimplePower(unittest.TestCase):
    def test_simple_power(self):
        # Test cases where x is a simple power of n
        self.assertTrue(is_simple_power(1, 4))  # 4^0 = 1
        self.assertTrue(is_simple_power(2, 2))  # 2^1 = 2
        self.assertTrue(is_simple_power(8, 2))  # 2^3 = 8
        
        # Test cases where x is not a simple power of n
        self.assertFalse(is_simple_power(3, 2))  # No integer k such that 2^k = 3
        self.assertFalse(is_simple_power(3, 1))  # 1^k = 1 for any k, cannot equal 3
        self.assertFalse(is_simple_power(5, 3))  # No integer k such that 3^k = 5

        # Edge cases
        self.assertFalse(is_simple_power(0, 2))  # 2^k cannot be 0
        self.assertFalse(is_simple_power(-1, 2))  # No integer k such that 2^k = -1
        self.assertTrue(is_simple_power(1, 1))  # 1^k = 1 for any k

if __name__ == '__main__':
    unittest.main()
