
import unittest
from grade_converter import numerical_letter_grade

class TestNumericalLetterGrade(unittest.TestCase):
    def test_perfect_grade(self):
        self.assertEqual(numerical_letter_grade([4.0]), ['A+'])

    def test_high_grades(self):
        self.assertEqual(numerical_letter_grade([3.8, 3.6, 3.4]), ['A', 'A-', 'A-'])

    def test_mid_grades(self):
        self.assertEqual(numerical_letter_grade([3.2, 2.8, 2.5]), ['B+', 'B', 'B-'])

    def test_low_grades(self):
        self.assertEqual(numerical_letter_grade([1.9, 1.5, 0.0]), ['C+', 'C-', 'E'])

    def test_mixed_grades(self):
        self.assertEqual(numerical_letter_grade([4.0, 3.5, 2.0, 1.0, 0.5]), ['A+', 'A-', 'C+', 'D+', 'D-'])

    def test_empty_list(self):
        self.assertEqual(numerical_letter_grade([]), [])

if __name__ == '__main__':
    unittest.main()
