
import unittest
from digit_sum import digitSum

class TestDigitSum(unittest.TestCase):
    def test_empty_string(self):
        self.assertEqual(digitSum(""), 0)

    def test_mixed_case_string(self):
        self.assertEqual(digitSum("abAB"), 131)

    def test_uppercase_in_middle(self):
        self.assertEqual(digitSum("abcCd"), 67)

    def test_single_uppercase(self):
        self.assertEqual(digitSum("helloE"), 69)

    def test_multiple_uppercase(self):
        self.assertEqual(digitSum("woArBld"), 131)

    def test_uppercase_with_lowercase(self):
        self.assertEqual(digitSum("aAaaaXa"), 153)

    def test_no_uppercase(self):
        self.assertEqual(digitSum("abcdefg"), 0)

    def test_all_uppercase(self):
        self.assertEqual(digitSum("ABCDEFG"), 65 + 66 + 67 + 68 + 69 + 70 + 71)

if __name__ == '__main__':
    unittest.main()

def digitSum(s):
    """Task
    Write a function that takes a string as input and returns the sum of the upper characters only
    ASCII codes.

    Examples:
        digitSum("") => 0
        digitSum("abAB") => 131
        digitSum("abcCd") => 67
        digitSum("helloE") => 69
        digitSum("woArBld") => 131
        digitSum("aAaaaXa") => 153
    """
    return sum(ord(char) for char in s if char.isupper())
