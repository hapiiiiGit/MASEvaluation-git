
def fibfib(n: int) -> int:
    """The FibFib number sequence is a sequence similar to the Fibonacci sequence that's defined as follows:
    fibfib(0) == 0
    fibfib(1) == 0
    fibfib(2) == 1
    fibfib(n) == fibfib(n-1) + fibfib(n-2) + fibfib(n-3).
    
    Args:
        n (int): The index of the FibFib number to compute.
        
    Returns:
        int: The n-th FibFib number.
    
    Examples:
        >>> fibfib(1)
        0
        >>> fibfib(5)
        4
        >>> fibfib(8)
        24
    """
    if n == 0:
        return 0
    elif n == 1:
        return 0
    elif n == 2:
        return 1
    
    # Initialize a list to store the FibFib numbers up to n
    fibfib_numbers = [0] * (n + 1)
    fibfib_numbers[0] = 0
    fibfib_numbers[1] = 0
    fibfib_numbers[2] = 1
    
    # Compute the FibFib numbers iteratively
    for i in range(3, n + 1):
        fibfib_numbers[i] = fibfib_numbers[i - 1] + fibfib_numbers[i - 2] + fibfib_numbers[i - 3]
    
    return fibfib_numbers[n]

import unittest
from fibfib import fibfib

class TestFibFib(unittest.TestCase):
    def test_fibfib(self):
        # Test base cases
        self.assertEqual(fibfib(0), 0)
        self.assertEqual(fibfib(1), 0)
        self.assertEqual(fibfib(2), 1)
        
        # Test additional cases
        self.assertEqual(fibfib(3), 1)  # fibfib(3) = fibfib(2) + fibfib(1) + fibfib(0) = 1 + 0 + 0
        self.assertEqual(fibfib(4), 2)  # fibfib(4) = fibfib(3) + fibfib(2) + fibfib(1) = 1 + 1 + 0
        self.assertEqual(fibfib(5), 4)  # fibfib(5) = fibfib(4) + fibfib(3) + fibfib(2) = 2 + 1 + 1
        self.assertEqual(fibfib(6), 7)  # fibfib(6) = fibfib(5) + fibfib(4) + fibfib(3) = 4 + 2 + 1
        self.assertEqual(fibfib(7), 13) # fibfib(7) = fibfib(6) + fibfib(5) + fibfib(4) = 7 + 4 + 2
        self.assertEqual(fibfib(8), 24) # fibfib(8) = fibfib(7) + fibfib(6) + fibfib(5) = 13 + 7 + 4

if __name__ == '__main__':
    unittest.main()
