
import unittest
from any_int import any_int

class TestAnyIntFunction(unittest.TestCase):
    def test_valid_cases(self):
        self.assertTrue(any_int(5, 2, 7))  # 5 == 2 + 3
        self.assertTrue(any_int(3, -2, 1))  # 3 == -2 + 1
        self.assertTrue(any_int(0, 0, 0))   # 0 == 0 + 0

    def test_invalid_cases(self):
        self.assertFalse(any_int(3, 2, 2))  # No number equals the sum of the other two
        self.assertFalse(any_int(3.6, -2.2, 2))  # Not all integers
        self.assertFalse(any_int(1, 2, 4))  # No number equals the sum of the other two
        self.assertFalse(any_int(1, 2, '3'))  # One input is a string

    def test_edge_cases(self):
        self.assertFalse(any_int(1, 2, None))  # None is not an integer
        self.assertFalse(any_int(-1, -1, -2))  # -1 == -1 + -2 is True, but all must be integers
        self.assertTrue(any_int(10, 5, 5))  # 10 == 5 + 5

if __name__ == '__main__':
    unittest.main()

def any_int(x, y, z):
    '''
    Create a function that takes 3 numbers.
    Returns true if one of the numbers is equal to the sum of the other two, and all numbers are integers.
    Returns false in any other cases.
    
    Examples
    any_int(5, 2, 7) ➞ True
    any_int(3, 2, 2) ➞ False
    any_int(3, -2, 1) ➞ True
    any_int(3.6, -2.2, 2) ➞ False
    '''
    # Check if all inputs are integers
    if isinstance(x, int) and isinstance(y, int) and isinstance(z, int):
        # Check if one number is equal to the sum of the other two
        return (x == y + z) or (y == x + z) or (z == x + y)
    return False
