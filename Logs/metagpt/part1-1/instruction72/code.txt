
import unittest
from will_it_fly import will_it_fly

class TestWillItFly(unittest.TestCase):
    def test_unbalanced_below_weight(self):
        self.assertFalse(will_it_fly([1, 2], 5), "Should be False: unbalanced but below weight")

    def test_balanced_above_weight(self):
        self.assertFalse(will_it_fly([3, 2, 3], 1), "Should be False: balanced but above weight")

    def test_balanced_below_weight(self):
        self.assertTrue(will_it_fly([3, 2, 3], 9), "Should be True: balanced and below weight")

    def test_single_element_below_weight(self):
        self.assertTrue(will_it_fly([3], 5), "Should be True: single element below weight")

    def test_empty_list(self):
        self.assertTrue(will_it_fly([], 0), "Should be True: empty list is balanced and weight is 0")

    def test_balanced_with_zero_weight(self):
        self.assertTrue(will_it_fly([0, 0, 0], 0), "Should be True: balanced and weight is 0")

    def test_unbalanced_with_large_weight(self):
        self.assertFalse(will_it_fly([1, 2, 3], 10), "Should be False: unbalanced even with large weight")

if __name__ == '__main__':
    unittest.main()

def will_it_fly(q, w):
    '''
    Write a function that returns True if the object q will fly, and False otherwise.
    The object q will fly if it's balanced (it is a palindromic list) and the sum of its elements is less than or equal the maximum possible weight w.

    Example:
    will_it_fly([1, 2], 5) ➞ False 
    # 1+2 is less than the maximum possible weight, but it's unbalanced.

    will_it_fly([3, 2, 3], 1) ➞ False
    # it's balanced, but 3+2+3 is more than the maximum possible weight.

    will_it_fly([3, 2, 3], 9) ➞ True
    # 3+2+3 is less than the maximum possible weight, and it's balanced.

    will_it_fly([3], 5) ➞ True
    # 3 is less than the maximum possible weight, and it's balanced.
    '''
    # Check if the list is a palindrome
    is_balanced = (q == q[::-1])
    
    # Calculate the sum of the elements in the list
    total_weight = sum(q)
    
    # Determine if the object will fly
    return is_balanced and total_weight <= w
