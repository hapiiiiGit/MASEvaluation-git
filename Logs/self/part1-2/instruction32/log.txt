 def is_nested(string):
    '''
    Create a function that takes a string as input which contains only square brackets.
    The function should return True if and only if there is a valid subsequence of brackets 
    where at least one bracket in the subsequence is nested.

    is_nested('[[]]') ➞ True
    is_nested('[]]]]]]][[[[[]') ➞ False
    is_nested('[][]') ➞ False
    is_nested('[]') ➞ False
    is_nested('[[][]]') ➞ True
    is_nested('[[]][[') ➞ True
    '''
    stack = []
    pairs = []

    # Step 1: Find all valid bracket pairs and their positions
    for idx, ch in enumerate(string):
        if ch == '[':
            stack.append(idx)
        elif ch == ']':
            if stack:
                open_idx = stack.pop()
                pairs.append((open_idx, idx))

    # Step 2: For each pair, check if there is another pair strictly inside it
    for i, (start1, end1) in enumerate(pairs):
        for j, (start2, end2) in enumerate(pairs):
            if i != j and start1 < start2 < end2 < end1:
                return True

    return False
