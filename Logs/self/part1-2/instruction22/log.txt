 def add_elements(arr, k):
    """
    Given a non-empty array of integers arr and an integer k, return
    the sum of the elements with at most two digits from the first k elements of arr.

    Example:

        Input: arr = [111,21,3,4000,5,6,7,8,9], k = 4
        Output: 24 # sum of 21 + 3

    Constraints:
        1. 1 <= len(arr) <= 100
        2. 1 <= k <= len(arr)
    """
    # Validate input constraints
    if not isinstance(arr, list) or len(arr) < 1 or len(arr) > 100:
        raise ValueError("arr must be a non-empty list with length between 1 and 100.")
    if not isinstance(k, int) or k < 1 or k > len(arr):
        raise ValueError("k must be an integer between 1 and len(arr).")

    # Extract first k elements
    first_k = arr[:k]

    # Filter elements with at most two digits (absolute value between 0 and 99 inclusive)
    filtered = [x for x in first_k if 0 <= abs(x) <= 99]

    # Sum and return
    return sum(filtered)
