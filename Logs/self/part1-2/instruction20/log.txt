 import heapq

def maximum(arr, k):
    """
    Given an array arr of integers and a positive integer k, return a sorted list 
    of length k with the maximum k numbers in arr.

    Example 1:
        Input: arr = [-3, -4, 5], k = 3
        Output: [-4, -3, 5]

    Example 2:
        Input: arr = [4, -4, 4], k = 2
        Output: [4, 4]

    Example 3:
        Input: arr = [-3, 2, 1, 2, -1, -2, 1], k = 1
        Output: [2]

    Note:
        1. The length of the array will be in the range of [1, 1000].
        2. The elements in the array will be in the range of [-1000, 1000].
        3. 0 <= k <= len(arr)
    """
    # Step 1: Validate input
    if not isinstance(arr, list) or not all(isinstance(x, int) for x in arr):
        raise ValueError("arr must be a list of integers.")
    if not isinstance(k, int) or k < 0 or k > len(arr):
        raise ValueError("k must be a non-negative integer not greater than the length of arr.")

    # Step 2: Handle k == 0
    if k == 0:
        return []

    # Step 3: Find k largest numbers
    largest_k = heapq.nlargest(k, arr)

    # Step 4: Sort the k largest numbers in ascending order
    result = sorted(largest_k)

    # Step 5: Return the sorted list
    return result
