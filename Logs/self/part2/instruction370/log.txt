 def add_spaces_between_capital_words(s):
    if not s:
        return s
    result = [s[0]]
    for i in range(1, len(s)):
        if s[i].isupper() and (s[i-1].islower() or (s[i-1].isupper() and (i+1 < len(s) and s[i+1].islower()))):
            result.append(' ')
        result.append(s[i])
    return ''.join(result)
