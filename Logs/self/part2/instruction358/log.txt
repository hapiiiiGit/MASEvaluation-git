 def locate_left_insertion_point(sorted_list, value):
    """
    Locate the left insertion point for a specified value in sorted order.

    Args:
        sorted_list (list): A list of elements sorted in ascending order.
        value: The value to locate the left insertion point for.

    Returns:
        int: The index where the value can be inserted to maintain sorted order.
    
    Example:
        >>> locate_left_insertion_point([1, 2, 4, 4, 5], 4)
        2
        >>> locate_left_insertion_point([1, 2, 4, 4, 5], 3)
        2
        >>> locate_left_insertion_point([], 10)
        0
        >>> locate_left_insertion_point([1, 2, 3], 0)
        0
        >>> locate_left_insertion_point([1, 2, 3], 4)
        3
    """
    left, right = 0, len(sorted_list)
    while left < right:
        mid = (left + right) // 2
        if sorted_list[mid] < value:
            left = mid + 1
        else:
            right = mid
    return left
