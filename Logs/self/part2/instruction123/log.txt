 def longest_palindromic_subsequence(s: str) -> int:
    # Validate input
    if not isinstance(s, str):
        raise ValueError("Input must be a string.")
    if any(not isinstance(c, str) or len(c) != 1 for c in s):
        raise ValueError("Input must be a valid sequence of characters.")

    n = len(s)
    if n == 0:
        return 0

    # dp[i][j] will hold the length of the longest palindromic subsequence in s[i:j+1]
    dp = [[0] * n for _ in range(n)]

    # Subsequences of length 1 are palindromes of length 1
    for i in range(n):
        dp[i][i] = 1

    # Build the table
    for length in range(2, n + 1):  # length of substring
        for i in range(n - length + 1):
            j = i + length - 1
            if s[i] == s[j]:
                if length == 2:
                    dp[i][j] = 2
                else:
                    dp[i][j] = dp[i + 1][j - 1] + 2
            else:
                dp[i][j] = max(dp[i + 1][j], dp[i][j - 1])

    return dp[0][n - 1]
