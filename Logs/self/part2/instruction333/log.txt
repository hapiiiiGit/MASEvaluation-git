 def replace_whitespace_and_underscore(s: str) -> str:
    if any(c.isspace() for c in s):
        # Replace all whitespace characters with underscores
        return ''.join('_' if c.isspace() else c for c in s)
    elif '_' in s:
        # Replace all underscores with spaces
        return s.replace('_', ' ')
    else:
        # No whitespace or underscore present, return as is
        return s
