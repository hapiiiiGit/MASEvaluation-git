 import math

def calculate_wind_chill(velocity_kmh, temperature_c):
    # Validate input types
    if not (isinstance(velocity_kmh, (int, float)) and isinstance(temperature_c, (int, float))):
        raise ValueError("Both wind velocity and temperature must be numeric values.")
    
    # Validate physical ranges
    if velocity_kmh < 0:
        raise ValueError("Wind velocity cannot be negative.")
    if temperature_c > 10:
        raise ValueError("Wind chill index is only defined for temperatures at or below 10°C.")
    if velocity_kmh < 4.8:
        # Wind chill formula is not valid for velocities below 4.8 km/h
        # For such cases, wind chill is equal to the air temperature
        wind_chill = temperature_c
    else:
        # Standard wind chill index formula (Environment Canada, Celsius, km/h)
        wind_chill = (13.12 + 0.6215 * temperature_c
                      - 11.37 * math.pow(velocity_kmh, 0.16)
                      + 0.3965 * temperature_c * math.pow(velocity_kmh, 0.16))
    
    # Round up to the next integer
    wind_chill_rounded = math.ceil(wind_chill)
    return wind_chill_rounded
