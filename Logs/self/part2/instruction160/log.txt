 def find_max_length_lists(list_of_lists):
    # Validate that the input is a list of lists
    if not isinstance(list_of_lists, list):
        raise ValueError("Input must be a list of lists.")
    if not all(isinstance(inner, list) for inner in list_of_lists):
        raise ValueError("All elements of the input must be lists.")

    # If the input list is empty, return an empty list
    if not list_of_lists:
        return []

    # Determine the maximum length among all the inner lists
    max_len = max(len(inner) for inner in list_of_lists)

    # Identify all inner lists that have the maximum length
    result = [inner for inner in list_of_lists if len(inner) == max_len]

    # Return the list of maximum length inner lists
    return result
