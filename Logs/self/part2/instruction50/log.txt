 def count_divisors(n):
    if not isinstance(n, int):
        raise ValueError("Input must be an integer.")
    if n == 0:
        return float('inf')  # Zero has infinitely many divisors
    count = 0
    abs_n = abs(n)
    for i in range(1, abs_n + 1):
        if n % i == 0:
            count += 1
    return count
