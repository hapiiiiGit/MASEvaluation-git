 import math

def surface_area_cylinder(radius, height):
    """
    Calculate the surface area of a cylinder.

    Parameters:
    radius (float): The radius of the cylinder (must be positive).
    height (float): The height of the cylinder (must be positive).

    Returns:
    float: The surface area of the cylinder.

    Raises:
    ValueError: If radius or height is not a positive number.
    """
    if not (isinstance(radius, (int, float)) and isinstance(height, (int, float))):
        raise TypeError("Radius and height must be numbers.")
    if radius <= 0 or height <= 0:
        raise ValueError("Radius and height must be positive numbers.")
    return 2 * math.pi * radius * (radius + height)
