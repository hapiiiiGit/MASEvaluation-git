 def find_element_after_rotations(arr, rotations, index):
    """
    Finds the element at a given index after performing the specified number of right rotations on the array.

    Parameters:
    arr (list): The input array.
    rotations (int): Number of right rotations to perform.
    index (int): The target index to retrieve after rotations.

    Returns:
    element: The element at the specified index after rotations.
    """
    n = len(arr)
    if n == 0:
        raise ValueError("Array must not be empty.")
    # Normalize rotations in case it's greater than array length
    rotations = rotations % n
    # Perform right rotation
    rotated_arr = arr[-rotations:] + arr[:-rotations]
    # Return the element at the given index
    if index < 0 or index >= n:
        raise IndexError("Index out of bounds.")
    return rotated_arr[index]
