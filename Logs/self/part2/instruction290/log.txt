def find_first_adverb_ly(s):
    # Use regex to find words ending with 'ly'
    pattern = re.compile(r'\b\w+ly\b')
    match = pattern.search(s)
    if match:
        adverb = match.group()
        start, end = match.start(), match.end()
        return {'adverb': adverb, 'start': start, 'end': end}
    else:
        return None
