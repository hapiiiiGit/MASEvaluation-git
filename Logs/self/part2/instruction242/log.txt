 def has_31_days(month_number):
    """
    Check whether the given month number contains 31 days.

    Args:
        month_number (int): The month number (1-12).

    Returns:
        bool: True if the month has 31 days, False otherwise.

    Raises:
        ValueError: If month_number is not an integer between 1 and 12.
    """
    if not isinstance(month_number, int) or not (1 <= month_number <= 12):
        raise ValueError("Month number must be an integer between 1 and 12.")

    months_with_31_days = {1, 3, 5, 7, 8, 10, 12}
    return month_number in months_with_31_days
