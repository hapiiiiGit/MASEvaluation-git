 def ascii_modulo_char(s: str) -> str:
    """
    Returns a character by adding the ASCII values of all characters in the input string,
    taking modulo 26, and mapping the result to an uppercase letter ('A'-'Z').
    """
    total = sum(ord(c) for c in s)
    mod = total % 26
    result_char = chr(ord('A') + mod)
    return result_char
