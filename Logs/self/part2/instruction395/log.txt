 def count_chars_with_vowel_neighbors(s: str) -> int:
    vowels = set('aeiouAEIOU')
    count = 0
    n = len(s)
    for i in range(n):
        has_vowel_neighbor = False
        if i > 0 and s[i-1] in vowels:
            has_vowel_neighbor = True
        if i < n-1 and s[i+1] in vowels:
            has_vowel_neighbor = True
        if has_vowel_neighbor:
            count += 1
    return count
