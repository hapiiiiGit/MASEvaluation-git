 def largest_divisible_subset_size(nums):
    if not nums:
        return 0

    nums.sort()
    n = len(nums)
    dp = [1] * n  # dp[i] = size of largest subset ending with nums[i]

    for i in range(n):
        for j in range(i):
            if nums[i] % nums[j] == 0:
                dp[i] = max(dp[i], dp[j] + 1)

    return max(dp)
