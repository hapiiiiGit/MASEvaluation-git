 def contains_sublist(main_list, sublist):
    if not sublist:
        return True  # Assuming empty sublist is always found

    n, m = len(main_list), len(sublist)
    if m > n:
        return False

    for i in range(n - m + 1):
        if main_list[i:i + m] == sublist:
            return True
    return False
