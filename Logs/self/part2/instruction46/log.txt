 def is_undulating_number(num):
    # Validate input: must be an integer
    if not isinstance(num, int):
        return False

    num_str = str(abs(num))  # Handle negative numbers by taking absolute value

    # Must have at least 3 digits
    if len(num_str) < 3:
        return False

    # Identify the two distinct digits
    distinct_digits = list(dict.fromkeys(num_str))  # Preserves order, gets unique digits

    if len(distinct_digits) != 2:
        return False

    # Check alternation
    expected = [distinct_digits[i % 2] for i in range(len(num_str))]
    if list(num_str) == expected:
        return True
    else:
        return False
