 import math

def regular_polygon_area(number_of_sides, side_length):
    # Input validation
    if not isinstance(number_of_sides, int) or number_of_sides <= 2:
        raise ValueError("number_of_sides must be an integer greater than 2.")
    if not (isinstance(side_length, int) or isinstance(side_length, float)) or side_length <= 0:
        raise ValueError("side_length must be a positive number.")
    
    # Area calculation
    area = (number_of_sides * side_length ** 2) / (4 * math.tan(math.pi / number_of_sides))
    return area
