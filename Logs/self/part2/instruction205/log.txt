 def find_pair_with_highest_product(arr):

        raise ValueError("Input array must contain at least two integers.")

    max1 = max2 = float('-inf')
    min1 = min2 = float('inf')

    for num in arr:
        if num > max1:
            max2 = max1
            max1 = num
        elif num > max2:
            max2 = num

        if num < min1:
            min2 = min1
            min1 = num
        elif num < min2:
            min2 = num

    product1 = max1 * max2
    product2 = min1 * min2

    if product1 >= product2:
        return (max1, max2)
    else:
        return (min1, min2)
