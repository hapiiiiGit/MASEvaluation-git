 def reverse_vowels(s: str) -> str:
    vowels = set('aeiouAEIOU')
    vowel_indices = []
    vowel_chars = []

    # Collect indices and characters of vowels
    for idx, char in enumerate(s):
        if char in vowels:
            vowel_indices.append(idx)
            vowel_chars.append(char)

    # Reverse the list of vowel characters
    reversed_vowels = vowel_chars[::-1]

    # Convert string to list for modification
    s_list = list(s)

    # Replace vowels in original string with reversed vowels
    for idx, new_vowel in zip(vowel_indices, reversed_vowels):
        s_list[idx] = new_vowel

    # Join and return the result
    return ''.join(s_list)
