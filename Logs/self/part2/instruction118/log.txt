 def replace_last_with_list(list1, list2):
    """
    Replaces the last element of list1 with all elements of list2.

    Args:
        list1 (list): The list whose last element will be replaced.
        list2 (list): The list whose elements will replace the last element of list1.

    Returns:
        list: The modified list1.
    """
    if not list1:
        # If list1 is empty, just return list2
        return list2.copy()
    # Remove the last element
    list1 = list1[:-1]
    # Add all elements from list2
    list1.extend(list2)
    return list1
