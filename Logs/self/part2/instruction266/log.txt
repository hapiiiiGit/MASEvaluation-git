 def count_odd_xor_pairs(nums):
    if not isinstance(nums, list) or not all(isinstance(x, int) for x in nums):
        raise ValueError("Input must be a list of integers.")
    count = 0
    n = len(nums)
    for i in range(n):
        for j in range(i + 1, n):
            xor_value = nums[i] ^ nums[j]
            if xor_value % 2 == 1:
                count += 1
    return count
