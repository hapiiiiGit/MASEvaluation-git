 def is_sorted(lst):
    return all(lst[i] <= lst[i+1] for i in range(len(lst)-1))

def find_median_of_two_sorted_lists(list1, list2):
    # Validate input types
    if not isinstance(list1, list) or not isinstance(list2, list):
        raise TypeError("Both inputs must be lists.")
    
    # Validate same size
    if len(list1) != len(list2):
        raise ValueError("Both lists must have the same size.")
    
    # Validate sorted
    if not is_sorted(list1):
        raise ValueError("First list is not sorted.")
    if not is_sorted(list2):
        raise ValueError("Second list is not sorted.")
    
    # Merge two sorted lists
    merged = []
    i, j = 0, 0
    n = len(list1)
    while i < n and j < n:
        if list1[i] <= list2[j]:
            merged.append(list1[i])
            i += 1
        else:
            merged.append(list2[j])
            j += 1
    # Append remaining elements
    while i < n:
        merged.append(list1[i])
        i += 1
    while j < n:
        merged.append(list2[j])
        j += 1
    
    # Find median
    total_len = 2 * n
    mid1 = total_len // 2 - 1
    mid2 = total_len // 2
    median = (merged[mid1] + merged[mid2]) / 2
    
    return median
