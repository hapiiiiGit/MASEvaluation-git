 def find_parabola_directrix(form, **params):
    """
    Finds the directrix of a parabola.

    Parameters:
        form (str): The form of the parabola equation. Supported:
            - 'vertex': (x - h)^2 = 4p(y - k)
            - 'standard': y = ax^2 + bx + c
        params: Parameters required for the chosen form.
            For 'vertex': h, k, p
            For 'standard': a, b, c

    Returns:
        str: The equation of the directrix.
    """
    if form == 'vertex':
        # (x - h)^2 = 4p(y - k)
        h = params.get('h')
        k = params.get('k')
        p = params.get('p')
        if None in (h, k, p):
            raise ValueError("For 'vertex' form, provide h, k, and p.")
        # Directrix: y = k - p
        directrix = k - p
        return f"y = {directrix}"
    elif form == 'standard':
        # y = ax^2 + bx + c
        a = params.get('a')
        b = params.get('b')
        c = params.get('c')
        if None in (a, b, c):
            raise ValueError("For 'standard' form, provide a, b, and c.")
        if a == 0:
            raise ValueError("Coefficient 'a' cannot be zero for a parabola.")
        # Convert to vertex form to find k and p
        h = -b / (2 * a)
        k = a * h ** 2 + b * h + c
        p = 1 / (4 * a)
        directrix = k - p
        return f"y = {directrix}"
    else:
        raise ValueError("Unsupported form. Use 'vertex' or 'standard'.")
