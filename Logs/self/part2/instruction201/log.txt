 def snake_to_camel(snake_str):
    if not isinstance(snake_str, str):
        raise TypeError("Input must be a string.")
    if not snake_str or snake_str.startswith('_') or snake_str.endswith('_') or '__' in snake_str:
        raise ValueError("Input must be a valid snake_case string.")
    if not all(c.islower() or c == '_' or c.isdigit() for c in snake_str):
        raise ValueError("Input must be in snake_case format (lowercase letters, digits, and underscores only).")
    
    words = snake_str.split('_')
    camel_case = words[0] + ''.join(word.capitalize() for word in words[1:])
    return camel_case
