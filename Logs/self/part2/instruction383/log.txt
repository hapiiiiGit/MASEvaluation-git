 def has_30_days(month_number):
    """
    Check whether the given month number contains 30 days.
    Months are numbered from 1 (January) to 12 (December).
    
    Args:
        month_number (int): The month number (1-12).
        
    Returns:
        bool: True if the month has 30 days, False otherwise.
    """
    if not isinstance(month_number, int) or not (1 <= month_number <= 12):
        return False

    months_with_30_days = {4, 6, 9, 11}
    return month_number in months_with_30_days
