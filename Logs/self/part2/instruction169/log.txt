 def count_equal_sum_binary_sequences(n):
    from math import comb

    # For each possible sum k (from 0 to n), count the number of ways
    # to have k ones in the first half and k ones in the second half.
    # For each k, there are comb(n, k) ways for the first half and comb(n, k) for the second half.
    # Total count is sum over k of comb(n, k) ** 2

    total = 0
    for k in range(n + 1):
        total += comb(n, k) ** 2
    return total
