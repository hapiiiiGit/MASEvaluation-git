 def divide_lists_elementwise(list1, list2):
    # Validate that both inputs are lists
    if not isinstance(list1, list) or not isinstance(list2, list):
        raise TypeError("Both inputs must be lists.")
    
    # Check that both lists have the same length
    if len(list1) != len(list2):
        raise ValueError("Both lists must have the same length.")
    
    # Ensure all elements in the second list are not zero
    if any(divisor == 0 for divisor in list2):
        raise ZeroDivisionError("Division by zero detected in the second list.")
    
    # Perform element-wise division
    result = [a / b for a, b in zip(list1, list2)]
    
    return result
