    # Remove surrounding parentheses and whitespace
    cleaned = tuple_str.strip().strip('()')
    if not cleaned:
        return ()
    # Split by comma and convert each element to integer
    elements = [int(item.strip()) for item in cleaned.split(',')]
    return tuple(elements)
