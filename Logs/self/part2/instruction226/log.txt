 def count_bidirectional_tuple_pairs(tuple_list):
    """
    Counts the number of unique bidirectional tuple pairs in the input list.
    A bidirectional pair is defined as (a, b) and (b, a).
    Each pair is counted only once.

    Args:
        tuple_list (list of tuple): List of tuple pairs.

    Returns:
        int: Number of unique bidirectional tuple pairs.
    """
    seen = set()
    counted = set()
    count = 0

    for t in tuple_list:
        # Normalize the tuple so (a, b) and (b, a) are treated the same
        normalized = tuple(sorted(t))
        if normalized in seen and normalized not in counted:
            count += 1
            counted.add(normalized)
        else:
            seen.add(normalized)
    return count
