 def dict_depth(d):
    if not isinstance(d, dict):
        raise TypeError("Input must be a dictionary.")
    if not d:
        return 1
    max_depth = 0
    for value in d.values():
        if isinstance(value, dict):
            depth = dict_depth(value)
            if depth > max_depth:
                max_depth = depth
    return max_depth + 1
