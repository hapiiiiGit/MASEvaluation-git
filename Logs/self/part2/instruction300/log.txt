 def difference_first_even_odd(numbers):
    if not isinstance(numbers, list) or not all(isinstance(x, (int, float)) for x in numbers):
        raise ValueError("Input must be a list of numbers.")

    first_even = None
    first_odd = None

    for num in numbers:
        if first_even is None and isinstance(num, int) and num % 2 == 0:
            first_even = num
        if first_odd is None and isinstance(num, int) and num % 2 != 0:
            first_odd = num
        if first_even is not None and first_odd is not None:
            break

    if first_even is None or first_odd is None:
        raise ValueError("List must contain at least one even and one odd integer.")

    return first_even - first_odd
