 def rearrange_first_n(array, n):
    # Validate input
    if not isinstance(array, list):
        raise TypeError("Input array must be a list.")
    if not isinstance(n, int):
        raise TypeError("n must be an integer.")
    if n < 0 or n > len(array):
        raise ValueError("n must be in the range 0 <= n <= len(array).")

    # Extract the first n elements
    first_n = array[:n]

    # Separate negatives and non-negatives, preserving order
    negatives = [x for x in first_n if x < 0]
    non_negatives = [x for x in first_n if x >= 0]

    # Concatenate negatives and non-negatives
    rearranged = negatives + non_negatives

    # Replace the first n elements in the original array
    result = rearranged + array[n:]

    return result
