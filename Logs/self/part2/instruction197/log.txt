 def next_bigger_number(n):
    digits = list(str(n))
    length = len(digits)

    # Step 1: Find the rightmost digit which is smaller than its next digit
    i = length - 2
    while i >= 0 and digits[i] >= digits[i + 1]:
        i -= 1

    if i == -1:
        return -1  # No bigger permutation exists

    # Step 2: Find the smallest digit on right side of (i) which is greater than digits[i]
    j = length - 1
    while digits[j] <= digits[i]:
        j -= 1

    # Step 3: Swap digits[i] and digits[j]
    digits[i], digits[j] = digits[j], digits[i]

    # Step 4: Reverse the sequence after position i
    digits[i + 1:] = reversed(digits[i + 1:])

    return int(''.join(digits))
