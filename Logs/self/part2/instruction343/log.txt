 def reverse_upto_position(arr, position):
    """
    Reverses the elements of the array from the start up to the given position (inclusive).
    
    Parameters:
        arr (list): The input array.
        position (int): The position up to which to reverse the array (inclusive).
        
    Returns:
        list: The resulting array after reversal.
        
    Raises:
        ValueError: If the input is invalid.
    """
    if not isinstance(arr, list):
        raise ValueError("Input array must be a list.")
    if not isinstance(position, int):
        raise ValueError("Position must be an integer.")
    if position < 0 or position >= len(arr):
        raise ValueError("Position must be within the range of array indices.")
    
    reversed_part = arr[:position+1][::-1]
    remaining_part = arr[position+1:]
    return reversed_part + remaining_part
