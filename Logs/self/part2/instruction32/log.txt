 def find_divisible_tuples(tuple_list, k):
    """
    Returns a list of tuples where all elements are divisible by k.

    Parameters:
    tuple_list (list of tuples): The input list of tuples.
    k (int): The divisor.

    Returns:
    list of tuples: Tuples where all elements are divisible by k.
    """
    result = []
    for tup in tuple_list:
        if all(element % k == 0 for element in tup):
            result.append(tup)
    return result
