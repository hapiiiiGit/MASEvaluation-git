def find_integer_solution(a, b, n):
    # Validate input types
    if not all(isinstance(param, int) for param in (a, b, n)):
        raise ValueError("Parameters a, b, and n must be integers.")

    # Compute gcd
    d = gcd(a, b)
    if n % d != 0:
        return None

    # Extended Euclidean Algorithm to find x0, y0 such that a*x0 + b*y0 = d
    def extended_gcd(a, b):
        if b == 0:
            return (a, 1, 0)
        else:
            g, x1, y1 = extended_gcd(b, a % b)
            x, y = y1, x1 - (a // b) * y1
            return (g, x, y)

    _, x0, y0 = extended_gcd(a, b)
    # Scale solution to n
    x = x0 * (n // d)
    y = y0 * (n // d)
    return (x, y)
