 def count_lists(collection):
    """
    Counts the number of elements in the input collection that are lists.

    Args:
        collection (iterable): A collection containing multiple elements.

    Returns:
        int: The number of elements in the collection that are of type 'list'.
    """
    if not isinstance(collection, (list, tuple, set)):
        raise TypeError("Input must be a collection such as a list, tuple, or set.")

    return sum(1 for item in collection if isinstance(item, list))
