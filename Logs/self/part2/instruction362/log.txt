 def tuple_to_adjacent_dict(tpl):
    if not isinstance(tpl, tuple):
        raise TypeError("Input must be a tuple.")
    if len(tpl) < 2:
        raise ValueError("Tuple must have at least two elements to form adjacent pairs.")
    return {tpl[i]: tpl[i+1] for i in range(len(tpl)-1)}
