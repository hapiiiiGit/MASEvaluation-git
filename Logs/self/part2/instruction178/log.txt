 def set_left_most_unset_bit(n: int) -> int:
    # Get the binary representation without '0b' prefix
    bin_str = bin(n)[2:]
    # Convert to list for mutability
    bin_list = list(bin_str)
    # Find the left most unset bit (from most significant to least significant)
    for i in range(len(bin_list)):
        if bin_list[i] == '0':
            bin_list[i] = '1'
            # Join and convert back to integer
            return int(''.join(bin_list), 2)
    # If all bits are set, add a new leftmost bit
    return int('1' + bin_str, 2)
