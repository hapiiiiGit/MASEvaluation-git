 def find_max_length_list(lists):
    """
    Returns the list with the maximum length from a collection of lists.
    If multiple lists have the same maximum length, returns the first one encountered.
    """
    if not lists:
        return None
    max_list = lists[0]
    max_length = len(max_list)
    for lst in lists[1:]:
        if len(lst) > max_length:
            max_list = lst
            max_length = len(lst)
    return max_list
