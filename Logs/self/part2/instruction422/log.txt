 from typing import List, Any

def sublist_with_highest_sum(lists: List[List[Any]]) -> List[Any]:
    if not isinstance(lists, list) or not lists:
        raise ValueError("Input must be a non-empty list of lists.")

    max_sum = None
    max_sublist = None

    for sublist in lists:
        if not isinstance(sublist, list):
            raise ValueError("Each element of the input must be a list.")
        if not sublist:
            current_sum = 0
        else:
            if not all(isinstance(x, (int, float)) for x in sublist):
                raise ValueError("All elements in sublists must be numeric (int or float).")
            current_sum = sum(sublist)
        if (max_sum is None) or (current_sum > max_sum):
            max_sum = current_sum
            max_sublist = sublist

    return max_sublist
