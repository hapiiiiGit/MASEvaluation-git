 def find_dissimilar_elements(tuple1, tuple2):
    """
    Finds the dissimilar elements between two tuples.

    Args:
        tuple1 (tuple): The first tuple.
        tuple2 (tuple): The second tuple.

    Returns:
        set: A set containing elements that are present in one tuple but not in the other.
    """
    set1 = set(tuple1)
    set2 = set(tuple2)
    dissimilar = set1.symmetric_difference(set2)
    return dissimilar
