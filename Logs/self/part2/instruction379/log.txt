 def count_reverse_pairs(string_list):
    # Validate input
    if not isinstance(string_list, list) or not all(isinstance(s, str) for s in string_list):
        raise ValueError("Input must be a list of strings.")

    count = 0
    n = len(string_list)
    # Use a set to avoid counting duplicate pairs
    seen_pairs = set()

    for i in range(n):
        for j in range(i + 1, n):
            if string_list[i] == string_list[j][::-1]:
                # Create a sorted tuple to avoid counting (a, b) and (b, a) separately
                pair = tuple(sorted([string_list[i], string_list[j]]))
                if pair not in seen_pairs:
                    count += 1
                    seen_pairs.add(pair)
    return count
