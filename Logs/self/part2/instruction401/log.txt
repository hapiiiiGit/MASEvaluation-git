 def sum_odd_length_subarrays(arr):
    total_sum = 0
    n = len(arr)
    # Iterate over all possible odd lengths
    for length in range(1, n + 1, 2):
        # For each odd length, iterate over all possible starting indices
        for start in range(n - length + 1):
            subarray = arr[start:start + length]
            total_sum += sum(subarray)
    return total_sum
