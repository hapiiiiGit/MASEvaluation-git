 def find_substring_occurrences(main_string, substrings):
    """
    Finds all occurrences and positions of each substring in the main string.
    Returns a dictionary mapping each substring to a list of starting indices, or None if not found.
    """
    result = {}
    for sub in substrings:
        positions = []
        start = 0
        while True:
            idx = main_string.find(sub, start)
            if idx == -1:
                break
            positions.append(idx)
            start = idx + 1  # Move past the last found index to allow overlapping matches
        result[sub] = positions if positions else None
    return result
