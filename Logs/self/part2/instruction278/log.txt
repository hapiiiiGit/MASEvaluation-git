 def sort_sublists_of_strings(list_of_lists):
    if not isinstance(list_of_lists, list):
        raise ValueError("Input must be a list of lists.")
    sorted_lists = []
    for sublist in list_of_lists:
        if not isinstance(sublist, list):
            raise ValueError("Each element of the main list must be a list.")
        if not all(isinstance(item, str) for item in sublist):
            raise ValueError("All elements in each sublist must be strings.")
        sorted_lists.append(sorted(sublist))
    return sorted_lists
