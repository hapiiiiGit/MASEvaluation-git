 def max_uppercase_run(s):
    if not isinstance(s, str):
        raise ValueError("Input must be a string.")

    max_run = 0
    current_run = 0

    for char in s:
        if char.isupper():
            current_run += 1
            if current_run > max_run:
                max_run = current_run
        else:
            current_run = 0

    return max_run
