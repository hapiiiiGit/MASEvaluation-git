 def pairwise_add_neighbors(input_tuple):
    # Validate input is a tuple
    if not isinstance(input_tuple, tuple):
        raise TypeError("Input must be a tuple.")
    # Validate tuple has at least two elements
    if len(input_tuple) < 2:
        raise ValueError("Tuple must contain at least two elements.")
    # Validate all elements are numbers
    if not all(isinstance(x, (int, float, complex)) for x in input_tuple):
        raise TypeError("All elements in the tuple must be numbers.")
    
    result = []
    for i in range(len(input_tuple) - 1):
        pair_sum = input_tuple[i] + input_tuple[i + 1]
        result.append(pair_sum)
    return result
