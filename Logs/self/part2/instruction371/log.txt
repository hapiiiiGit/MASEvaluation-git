 def sort_numeric_strings(str_list):
    """
    Sorts a list of strings representing numbers numerically.

    Args:
        str_list (list of str): List of strings, each representing a number.

    Returns:
        list of str: Sorted list of strings, in ascending numeric order.
    """
    # Convert strings to integers, sort, then convert back to strings
    sorted_ints = sorted(int(s) for s in str_list)
    return [str(num) for num in sorted_ints]
