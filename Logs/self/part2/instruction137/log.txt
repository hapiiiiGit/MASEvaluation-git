 def calculate_dog_years(human_years):
    """
    Calculate a dog's age in dog years based on human years.
    Common rule: 1 human year = 7 dog years.
    Args:
        human_years (int or float): The dog's age in human years.
    Returns:
        float: The dog's age in dog years.
    Raises:
        ValueError: If human_years is negative.
        TypeError: If human_years is not a number.
    """
    if not isinstance(human_years, (int, float)):
        raise TypeError("Dog's age must be a number.")
    if human_years < 0:
        raise ValueError("Dog's age cannot be negative.")
    dog_years = human_years * 7
    return dog_years
