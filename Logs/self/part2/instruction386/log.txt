 def nth_polite_number(n: int) -> int:
    """
    Returns the nth polite number.
    Polite numbers are all natural numbers except powers of 2.
    """
    count = 0
    num = 1
    while True:
        # Check if num is not a power of 2
        if num & (num - 1) != 0:
            count += 1
            if count == n:
                return num
        num += 1
