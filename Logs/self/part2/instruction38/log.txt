 def newman_conway(n):
    if n <= 0:
        raise ValueError("Input must be a positive integer.")
    if n == 1 or n == 2:
        return 1
    seq = [0, 1, 1]  # seq[0] is dummy to make indexing easy
    for i in range(3, n + 1):
        seq.append(seq[seq[i - 1]] + seq[i - seq[i - 1]])
    return seq[n]
