 def jacobsthal(n):
    """
    Calculate the nth Jacobsthal number.
    Jacobsthal numbers are defined by:
        J(0) = 0
        J(1) = 1
        J(n) = J(n-1) + 2*J(n-2) for n > 1
    Args:
        n (int): The index of the Jacobsthal number to compute (must be non-negative).
    Returns:
        int: The nth Jacobsthal number.
    Raises:
        ValueError: If n is not a non-negative integer.
    """
    if not isinstance(n, int) or n < 0:
        raise ValueError("Input n must be a non-negative integer.")
    if n == 0:
        return 0
    if n == 1:
        return 1
    j0, j1 = 0, 1
    for _ in range(2, n + 1):
        jn = j1 + 2 * j0
        j0, j1 = j1, jn
    return j1
