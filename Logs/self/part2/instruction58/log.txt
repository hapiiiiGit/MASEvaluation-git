 def sort_sublists(list_of_lists):
    if not isinstance(list_of_lists, list):
        raise TypeError("Input must be a list of lists.")
    sorted_lists = []
    for sublist in list_of_lists:
        if not isinstance(sublist, list):
            raise TypeError("Each element in the main list must be a list.")
        if not all(isinstance(item, str) for item in sublist):
            raise TypeError("All elements in each sublist must be strings.")
        sorted_sublist = sorted(sublist)
        sorted_lists.append(sorted_sublist)
    return sorted_lists
