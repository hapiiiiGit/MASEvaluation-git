 def first_of_smallest_second(lst):
    if not isinstance(lst, list):
        raise TypeError("Input must be a list.")
    if not all(isinstance(t, tuple) and len(t) >= 2 for t in lst):
        raise ValueError("All elements must be tuples with at least two elements.")
    if not lst:
        raise ValueError("Input list cannot be empty.")
    smallest_tuple = min(lst, key=lambda x: x[1])
    return smallest_tuple[0]
