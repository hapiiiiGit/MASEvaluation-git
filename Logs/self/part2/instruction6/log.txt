 def min_rotations_to_same(s: str) -> int:
    """
    Returns the minimum number of rotations (greater than 0) required
    to get the same string. If no such rotation exists, returns -1.
    """
    n = len(s)
    if n == 0:
        return -1  # No rotation possible for empty string
    for i in range(1, n):
        rotated = s[i:] + s[:i]
        if rotated == s:
            return i
    return -1  # No rotation returns the original string
