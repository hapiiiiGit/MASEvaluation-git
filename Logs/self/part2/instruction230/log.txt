 def ratio_of_positive_numbers(arr):
    # Validate input: must be a list of integers
    if not isinstance(arr, list):
        raise TypeError("Input must be a list.")
    if not all(isinstance(x, int) for x in arr):
        raise ValueError("All elements in the array must be integers.")

    total_count = len(arr)
    if total_count == 0:
        return 0.0  # Avoid division by zero, ratio is 0 for empty array

    positive_count = sum(1 for x in arr if x > 0)
    ratio = positive_count / total_count
    return float(ratio)
