 def contains_consecutive_numbers(lst):
    # Validate input: check if lst is a list
    if not isinstance(lst, list):
        raise ValueError("Input must be a list.")
    # Validate input: check if all elements are numbers (int or float)
    if not all(isinstance(x, (int, float)) for x in lst):
        raise ValueError("All elements in the list must be numbers.")
    if len(lst) < 2:
        # A list with 0 or 1 element is trivially consecutive
        return True
    # Sort the list
    sorted_lst = sorted(lst)
    # Check if each adjacent pair differs by exactly 1
    for i in range(1, len(sorted_lst)):
        if sorted_lst[i] - sorted_lst[i - 1] != 1:
            return False
    return True
