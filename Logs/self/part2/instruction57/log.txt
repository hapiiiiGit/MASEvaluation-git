 def eulerian_number(n, m):
    """
    Compute the Eulerian number a(n, m), which counts the number of permutations
    of the numbers 1 to n in which exactly m elements are greater than the previous element.

    Parameters:
        n (int): The number of elements (n >= 1).
        m (int): The number of ascents (0 <= m < n).

    Returns:
        int: The Eulerian number a(n, m).

    Raises:
        ValueError: If n < 1, m < 0, or m >= n.
    """
    if not (isinstance(n, int) and isinstance(m, int)):
        raise ValueError("n and m must be integers.")
    if n < 1:
        raise ValueError("n must be at least 1.")
    if m < 0 or m >= n:
        raise ValueError("m must satisfy 0 <= m < n.")

    # Use dynamic programming to compute Eulerian numbers
    dp = [[0 for _ in range(n)] for _ in range(n + 1)]
    dp[1][0] = 1  # Base case: a(1, 0) = 1

    for i in range(2, n + 1):
        for j in range(i):
            val = 0
            if j < i - 1:
                val += (i - j - 1) * dp[i - 1][j]
            if j > 0:
                val += (j + 1) * dp[i - 1][j - 1]
            dp[i][j] = val

    return dp[n][m]
