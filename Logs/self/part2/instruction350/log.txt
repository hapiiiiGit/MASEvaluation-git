 def consecutive_products(input_tuple):
    # Validate that the input is a tuple
    if not isinstance(input_tuple, tuple):
        raise TypeError("Input must be a tuple.")
    # Validate that the tuple contains at least two elements
    if len(input_tuple) < 2:
        raise ValueError("Input tuple must contain at least two elements.")
    # Validate that all elements are numbers (int or float)
    if not all(isinstance(x, (int, float)) for x in input_tuple):
        raise TypeError("All elements in the tuple must be numbers (int or float).")
    # Compute products of consecutive elements
    products = tuple(input_tuple[i] * input_tuple[i+1] for i in range(len(input_tuple) - 1))
    return products
