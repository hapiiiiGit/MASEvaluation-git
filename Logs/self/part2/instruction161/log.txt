 def num_ways_paint_fence(n, k):
    if n == 0 or k == 0:
        return 0
    if n == 1:
        return k
    if n == 2:
        return k * k

    # same[i]: ways to paint i posts where last two posts have same color
    # diff[i]: ways to paint i posts where last two posts have different colors
    same = k  # For n=2, last two same: k choices (both same color)
    diff = k * (k - 1)  # For n=2, last two different: k choices for first, (k-1) for second

    for i in range(3, n + 1):
        new_same = diff  # Only way to have last two same is if previous two were different
        new_diff = (same + diff) * (k - 1)  # Last two different: previous can be anything, new color is different from last
        same, diff = new_same, new_diff

    return same + diff
