 import math

def binomial_coefficient(n, k):
    if k < 0 or k > n:
        return 0
    return math.comb(n, k)

def sum_of_consecutive_binomial_products(n):
    total = 0
    for k in range(0, n):
        prod = binomial_coefficient(n, k) * binomial_coefficient(n, k + 1)
        total += prod
    return total
