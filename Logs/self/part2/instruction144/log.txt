 def sum_even_natural_numbers_fifth_power(n):
    if not isinstance(n, int) or n <= 0:
        raise ValueError("Input n must be a positive integer.")
    even_numbers = [2 * i for i in range(1, n + 1)]
    total = sum(num ** 5 for num in even_numbers)
    return total
