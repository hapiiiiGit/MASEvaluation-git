    """
    Calculate the geometric sum: 1/2 + 1/4 + 1/8 + ... + 1/(2^n)
    Args:
        n (int): The number of terms in the sum.
    Returns:
        float: The geometric sum as a floating-point number.
    """

    return (1 / (2 ** n)) + geometric_sum(n - 1)
