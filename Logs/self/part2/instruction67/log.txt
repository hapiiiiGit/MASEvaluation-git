 def convert_elements_to_float(list_of_lists):
    result = []
    for sublist in list_of_lists:
        new_sublist = []
        for elem in sublist:
            try:
                new_elem = float(elem)
            except (ValueError, TypeError):
                new_elem = elem
            new_sublist.append(new_elem)
        result.append(new_sublist)
    return result
