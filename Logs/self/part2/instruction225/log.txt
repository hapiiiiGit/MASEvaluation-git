 def remove_odd_characters(s):
    """
    Removes characters at odd indices from the input string.

    Args:
        s (str): The input string.

    Returns:
        str: The string with characters at odd indices removed.
    """
    return ''.join([char for idx, char in enumerate(s) if idx % 2 == 0])
