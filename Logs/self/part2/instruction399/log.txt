 from itertools import combinations

def find_combinations_of_sum(tuple_list, target_sum):
    """
    Finds all combinations of tuples from tuple_list whose elements sum to target_sum.

    Args:
        tuple_list (list of tuple of int): List of tuples to consider.
        target_sum (int): The target sum to match.

    Returns:
        list of list of tuple: List of combinations (each combination is a list of tuples) whose sum matches target_sum.
    """
    result = []
    n = len(tuple_list)
    # Try all possible combination lengths (from 1 to n)
    for r in range(1, n + 1):
        for combo in combinations(tuple_list, r):
            # Flatten the combination to sum all elements
            total = sum(sum(t) for t in combo)
            if total == target_sum:
                result.append(list(combo))
    return result
