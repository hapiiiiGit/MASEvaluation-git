        raise ValueError("Number must be within 0 to 2^32 - 1 (32-bit unsigned integer).")
    
    # Validate the input for the number of bits to rotate
    if not isinstance(d, int):
        raise TypeError("Number of bits to rotate (d) must be an integer.")
    if not (0 <= d <= 31):
        raise ValueError("Number of bits to rotate (d) must be between 0 and 31.")
