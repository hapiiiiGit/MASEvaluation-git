 def min_swaps_to_convert(s1: str, s2: str) -> int:
    # Validate input lengths
    if len(s1) != len(s2):
        raise ValueError("Input strings must be of equal length.")
    # Validate characters
    if not (set(s1) <= {'0', '1'} and set(s2) <= {'0', '1'}):
        raise ValueError("Input strings must contain only '0' and '1'.")
    # Check if conversion is possible
    if s1.count('0') != s2.count('0') or s1.count('1') != s2.count('1'):
        return -1  # Impossible to convert

    # Identify mismatches
    mismatches_01 = 0  # s1 has '0', s2 has '1'
    mismatches_10 = 0  # s1 has '1', s2 has '0'
    for a, b in zip(s1, s2):
        if a != b:
            if a == '0' and b == '1':
                mismatches_01 += 1
            elif a == '1' and b == '0':
                mismatches_10 += 1

    # Each swap can fix one '01' and one '10' mismatch
    swaps = min(mismatches_01, mismatches_10)
    # Remaining mismatches (if any) must be resolved by swapping within the same type
    # Each two mismatches of the same type require two swaps
    remaining = abs(mismatches_01 - mismatches_10)
    swaps += remaining

    return swaps
