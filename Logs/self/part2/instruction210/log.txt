 def concatenate_tuple_elements(input_tuple, delimiter):
    """
    Concatenate each element of the input_tuple by the specified delimiter.

    Args:
        input_tuple (tuple): Tuple containing elements to concatenate.
        delimiter (str): Delimiter to use for concatenation.

    Returns:
        str: Concatenated string.
    """
    # Convert all elements to string
    str_elements = [str(element) for element in input_tuple]
    # Join using the delimiter
    result = delimiter.join(str_elements)
    return result
