 def pack_consecutive_duplicates(lst):
    if not lst:
        return []
    
    result = []
    temp = [lst[0]]
    
    for elem in lst[1:]:
        if elem == temp[-1]:
            temp.append(elem)
        else:
            result.append(temp)
            temp = [elem]
    result.append(temp)
    
    return result
