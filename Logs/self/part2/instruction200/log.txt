 def find_min_in_heterogeneous_list(lst):
    if not isinstance(lst, list):
        raise TypeError("Input must be a list.")

    min_value = None
    for elem in lst:
        if min_value is None:
            min_value = elem
            continue
        try:
            if elem < min_value:
                min_value = elem
        except Exception:
            continue

    # Check if at least one comparable element was found
    if min_value is None:
        return None
    return min_value
