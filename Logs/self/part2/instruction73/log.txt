 def max_zero_one_diff(binary_str):
    # Step 1: Validate input
    if not isinstance(binary_str, str) or any(c not in '01' for c in binary_str):
        raise ValueError("Input must be a binary string containing only '0' and '1'.")

    n = len(binary_str)
    max_diff = float('-inf')

    # Step 2: Iterate through all possible substrings
    for i in range(n):
        zero_count = 0
        one_count = 0
        for j in range(i, n):
            # Step 3: Count 0s and 1s in the current substring
            if binary_str[j] == '0':
                zero_count += 1
            else:
                one_count += 1
            # Step 4: Compute difference and update max_diff
            diff = zero_count - one_count
            if diff > max_diff:
                max_diff = diff

    # Step 5: Return the result
    return max_diff
