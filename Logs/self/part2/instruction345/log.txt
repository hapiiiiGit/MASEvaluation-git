 def max_average_path(matrix):
    # Validate input
    if not matrix or not isinstance(matrix, list):
        raise ValueError("Input must be a non-empty list of lists.")
    N = len(matrix)
    for row in matrix:
        if not isinstance(row, list) or len(row) != N:
            raise ValueError("Matrix must be square (N x N).")
        for val in row:
            if not isinstance(val, (int, float)):
                raise ValueError("Matrix values must be numeric.")

    # Helper function to enumerate all paths using DFS
    def dfs(x, y, path_sum, path_len):
        nonlocal max_avg
        # Add current cell's cost
        path_sum += matrix[x][y]
        path_len += 1
        # If reached bottom-right, update max_avg
        if x == N - 1 and y == N - 1:
            avg = path_sum / path_len
            if avg > max_avg:
                max_avg = avg
            return
        # Move right
        if y + 1 < N:
            dfs(x, y + 1, path_sum, path_len)
        # Move down
        if x + 1 < N:
            dfs(x + 1, y, path_sum, path_len)

    max_avg = float('-inf')
    dfs(0, 0, 0, 0)
    return max_avg
