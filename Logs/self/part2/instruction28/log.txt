 def all_tuples_equal_length(tuples_collection):
    if not tuples_collection:
        return True  # Assuming empty collection means all tuples are "equal" in length
    reference_length = len(tuples_collection[0])
    for tup in tuples_collection[1:]:
        if len(tup) != reference_length:
            return False
    return True
