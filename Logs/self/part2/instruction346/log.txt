 def filter_students_by_height_weight(students: dict, min_height: float, min_weight: float) -> list:
    """
    Filters students whose height and weight are both above the given minimum values.

    Args:
        students (dict): Dictionary with student names as keys and (height, weight) tuples as values.
        min_height (float): Minimal height threshold.
        min_weight (float): Minimal weight threshold.

    Returns:
        list: List of student names who satisfy both conditions.
    """
    filtered_students = []
    for name, (height, weight) in students.items():
        if height > min_height and weight > min_weight:
            filtered_students.append(name)
    return filtered_students
