 def any_value_exists(seq1, seq2):
    """
    Check whether any value in seq1 exists in seq2.

    Args:
        seq1: First sequence (iterable).
        seq2: Second sequence (iterable).

    Returns:
        bool: True if any value from seq1 exists in seq2, False otherwise.
    """
    set_seq2 = set(seq2)
    for value in seq1:
        if value in set_seq2:
            return True
    return False
