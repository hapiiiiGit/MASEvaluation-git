 def max_sum_bitonic_subsequence(arr):
    n = len(arr)
    if n == 0:
        return 0

    # Maximum sum increasing subsequence ending at each index
    inc = arr.copy()
    for i in range(n):
        for j in range(i):
            if arr[j] < arr[i] and inc[i] < inc[j] + arr[i]:
                inc[i] = inc[j] + arr[i]

    # Maximum sum decreasing subsequence starting at each index
    dec = arr.copy()
    for i in range(n-1, -1, -1):
        for j in range(n-1, i, -1):
            if arr[j] < arr[i] and dec[i] < dec[j] + arr[i]:
                dec[i] = dec[j] + arr[i]

    # Combine increasing and decreasing sums for each index
    max_sum = 0
    for i in range(n):
        bitonic_sum = inc[i] + dec[i] - arr[i]
        if bitonic_sum > max_sum:
            max_sum = bitonic_sum

    return max_sum
