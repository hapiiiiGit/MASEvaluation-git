 import re

def extract_quoted_values(s):
    pattern = r"""
        (                               # Start of group 1
            (?<!\\)                     # Not preceded by a backslash (not escaped)
            (['"])                      # Match single or double quote, captured in group 2
            (                           # Start of group 3: the quoted content
                (?:                     # Non-capturing group for quoted content
                    \\.|                # Escaped character (e.g., \" or \')
                    [^\\\2]             # Any character except backslash or the quote type
                )*                      # Zero or more times
            )
            \2                          # Match the same quote as opened
        )
    """
    matches = re.finditer(pattern, s, re.VERBOSE)
    result = []
    for match in matches:
        quoted_content = match.group(3)
        # Unescape escaped quotes and backslashes
        quote_char = match.group(2)
        quoted_content = quoted_content.replace(f'\\{quote_char}', quote_char)
        quoted_content = quoted_content.replace('\\\\', '\\')
        result.append(quoted_content)
    return result
