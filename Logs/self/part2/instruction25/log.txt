 def bell_number(n):
    """
    Calculate the nth Bell number, which is the number of ways to partition a set of n elements.

    Parameters:
    n (int): The number of elements in the set. Must be a non-negative integer.

    Returns:
    int: The nth Bell number.

    Raises:
    ValueError: If n is not a non-negative integer.
    """
    if not isinstance(n, int) or n < 0:
        raise ValueError("n must be a non-negative integer.")

    # Bell triangle method
    bell = [0] * (n + 1)
    bell[0] = 1

    for i in range(1, n + 1):
        bell[i] = bell[i - 1]
        for j in range(i - 1, 0, -1):
            bell[j] = bell[j - 1] + bell[j]
    return bell[n]
