 def count_primes(n):
    if not isinstance(n, int) or n < 0:
        raise ValueError("Input must be a non-negative integer.")

    if n < 2:
        return 0

    is_prime = [True] * n
    is_prime[0:2] = [False, False]  # 0 and 1 are not primes

    for i in range(2, int(n ** 0.5) + 1):
        if is_prime[i]:
            for j in range(i * i, n, i):
                is_prime[j] = False

    return sum(is_prime)
