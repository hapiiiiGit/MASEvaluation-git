 def max_product_increasing_subsequence(arr):
    if not arr:
        return 0

    n = len(arr)
    # dp[i] will store the maximum product of an increasing subsequence ending at index i
    dp = arr.copy()

    for i in range(1, n):
        for j in range(i):
            if arr[i] > arr[j]:
                dp[i] = max(dp[i], dp[j] * arr[i])

    return max(dp)
