 def replace_characters(input_string, targets, replacements):
    """
    Replace specified characters in the input string with given replacements.

    Parameters:
        input_string (str): The string to process.
        targets (str or list): Character(s) to be replaced.
        replacements (str or list): Replacement character(s).

    Returns:
        str: The modified string with replacements applied.
    """
    # Validate input types
    if not isinstance(input_string, str) or not input_string:
        raise ValueError("input_string must be a non-empty string.")
    
    if isinstance(targets, str):
        targets = [targets]
    if isinstance(replacements, str):
        replacements = [replacements]
    
    if not isinstance(targets, list) or not targets:
        raise ValueError("targets must be a non-empty list or string.")
    if not isinstance(replacements, list) or not replacements:
        raise ValueError("replacements must be a non-empty list or string.")
    if len(targets) != len(replacements):
        raise ValueError("targets and replacements must have the same length.")
    
    result = input_string
    for target, replacement in zip(targets, replacements):
        result = result.replace(target, replacement)
    return result
