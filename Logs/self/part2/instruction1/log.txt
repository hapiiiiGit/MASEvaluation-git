 def is_non_prime(n):
    """
    Returns True if n is a non-prime number, False otherwise.
    """
    if not isinstance(n, int):
        raise ValueError("Input must be an integer.")
    if n < 2:
        return True  # Numbers less than 2 are non-prime by definition
    for i in range(2, int(n ** 0.5) + 1):
        if n % i == 0:
            return True  # Has a divisor other than 1 and itself
    return False  # Is prime
