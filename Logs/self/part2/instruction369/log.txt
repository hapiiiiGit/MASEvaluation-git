 def longest_common_subsequence_three(str1, str2, str3):
    # Validate inputs
    if not all(isinstance(s, str) for s in [str1, str2, str3]):
        raise ValueError("All inputs must be strings.")

    m, n, o = len(str1), len(str2), len(str3)
    # Initialize DP table
    dp = [[[0] * (o + 1) for _ in range(n + 1)] for __ in range(m + 1)]

    # Fill DP table
    for i in range(1, m + 1):
        for j in range(1, n + 1):
            for k in range(1, o + 1):
                if str1[i - 1] == str2[j - 1] == str3[k - 1]:
                    dp[i][j][k] = dp[i - 1][j - 1][k - 1] + 1
                else:
                    dp[i][j][k] = max(
                        dp[i - 1][j][k],
                        dp[i][j - 1][k],
                        dp[i][j][k - 1]
                    )

    # Reconstruct LCS from DP table
    lcs = []
    i, j, k = m, n, o
    while i > 0 and j > 0 and k > 0:
        if str1[i - 1] == str2[j - 1] == str3[k - 1]:
            lcs.append(str1[i - 1])
            i -= 1
            j -= 1
            k -= 1
        else:
            if dp[i - 1][j][k] >= dp[i][j - 1][k] and dp[i - 1][j][k] >= dp[i][j][k - 1]:
                i -= 1
            elif dp[i][j - 1][k] >= dp[i][j][k - 1]:
                j -= 1
            else:
                k -= 1

    return ''.join(reversed(lcs))
