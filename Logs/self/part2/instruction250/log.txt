 from typing import List, Union

def max_product_subarray(nums: List[Union[int, float]]) -> Union[int, float]:
    # Validate input
    if not isinstance(nums, list) or len(nums) == 0:
        raise ValueError("Input must be a non-empty list.")
    for num in nums:
        if not isinstance(num, (int, float)):
            raise ValueError("All elements in the array must be numbers.")

    # Initialize variables
    current_max = current_min = overall_max = nums[0]

    # Iterate through the array
    for num in nums[1:]:
        if num == 0:
            current_max, current_min = 0, 0
            overall_max = max(overall_max, 0)
            continue

        temp_max = max(num, current_max * num, current_min * num)
        temp_min = min(num, current_max * num, current_min * num)

        current_max, current_min = temp_max, temp_min
        overall_max = max(overall_max, current_max)

    return overall_max
