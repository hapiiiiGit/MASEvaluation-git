 def calculate_loss(sale_amount, cost_amount):
    """
    Returns the loss amount if there is a loss on the sale,
    otherwise returns 0.

    Parameters:
    sale_amount (float): The amount for which the item was sold.
    cost_amount (float): The original cost of the item.

    Returns:
    float: The loss amount if sale_amount < cost_amount, else 0.
    """
    if sale_amount < cost_amount:
        return cost_amount - sale_amount
    else:
        return 0
