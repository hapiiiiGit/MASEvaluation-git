 def move_numbers_to_end(s: str) -> str:
    non_numbers = []
    numbers = []
    for char in s:
        if char.isdigit():
            numbers.append(char)
        else:
            non_numbers.append(char)
    return ''.join(non_numbers) + ''.join(numbers)
