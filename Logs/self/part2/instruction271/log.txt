 def longest_sublists_length(lst):
    """
    Returns the length of the longest sublist in the given list.

    Parameters:
    lst (list): A list containing sublists.

    Returns:
    int: The length of the longest sublist. Returns 0 if there are no sublists.
    """
    max_length = 0
    for sublist in lst:
        if isinstance(sublist, list):
            max_length = max(max_length, len(sublist))
    return max_length
