 def sum_of_even_factors(n):
    if not isinstance(n, int) or n <= 0:
        raise ValueError("Input must be a positive integer.")
    even_factors_sum = 0
    for i in range(2, n + 1, 2):  # Only check even numbers
        if n % i == 0:
            even_factors_sum += i
    return even_factors_sum
