 import math

def sum_even_index_binomial_coefficients(n):
    if not isinstance(n, int) or n <= 0:
        raise ValueError("Input n must be a positive integer.")
    
    binomial_coeffs = [math.comb(n, k) for k in range(n + 1)]
    even_index_coeffs = [binomial_coeffs[k] for k in range(0, n + 1, 2)]
    return sum(even_index_coeffs)
