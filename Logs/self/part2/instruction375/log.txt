 def min_k_records(tuple_list, k):
    """
    Returns the minimum k records from a list of tuples,
    sorted by the first element of each tuple.
    """
    # Sort the list based on the first element of each tuple
    sorted_list = sorted(tuple_list, key=lambda x: x[0])
    # Return the first k records
    return sorted_list[:k]
