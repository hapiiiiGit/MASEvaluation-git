 def count_list_elements_in_tuple(lst, tpl):
    """
    Counts the occurrence of each element in lst within tpl.

    Args:
        lst (list): List of elements to count.
        tpl (tuple): Tuple in which to count occurrences.

    Returns:
        dict: Dictionary with elements from lst as keys and their counts in tpl as values.
    """
    counts = {}
    for element in lst:
        counts[element] = tpl.count(element)
    return counts
