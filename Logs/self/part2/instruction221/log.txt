 def is_number_greater_than_array():
    try:
        number = float(input("Enter a number: "))
    except ValueError:
        print("Invalid input. Please enter a valid number.")
        return

    array_input = input("Enter array elements separated by spaces: ")
    try:
        array = [float(x) for x in array_input.strip().split()]
    except ValueError:
        print("Invalid array input. Please enter valid numbers separated by spaces.")
        return

    if all(number > elem for elem in array):
        print(f"{number} is greater than all elements in the array.")
        return True
    else:
        print(f"{number} is NOT greater than all elements in the array.")
        return False
