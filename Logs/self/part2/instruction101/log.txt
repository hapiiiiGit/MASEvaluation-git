 def pentagon_perimeter(side_length):
    """
    Calculate the perimeter of a regular pentagon given the length of its sides.

    Parameters:
    side_length (float or int): The length of one side of the pentagon.

    Returns:
    float: The perimeter of the pentagon.

    Raises:
    ValueError: If side_length is not a positive number.
    """
    if not isinstance(side_length, (int, float)):
        raise ValueError("Side length must be a number.")
    if side_length <= 0:
        raise ValueError("Side length must be a positive number.")
    perimeter = 5 * side_length
    return perimeter
