 def find_common_elements(l1, l2, l3):
    if not (isinstance(l1, list) and isinstance(l2, list) and isinstance(l3, list)):
        raise TypeError("All inputs must be lists.")
    
    min_length = min(len(l1), len(l2), len(l3))
    result = []
    
    for i in range(min_length):
        if l1[i] == l2[i] == l3[i]:
            result.append(l1[i])
    
    return result
