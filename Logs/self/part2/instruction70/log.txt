 from typing import List, Tuple, Any
import numbers
import itertools

def max_absolute_product_in_tuples(lst: List[Tuple[Any, ...]]) -> float:
    if not isinstance(lst, list):
        raise ValueError("Input must be a list of tuples.")
    max_abs_product = None
    for tup in lst:
        if not isinstance(tup, tuple):
            raise ValueError("All elements of the list must be tuples.")
        # Filter only numeric elements
        nums = [x for x in tup if isinstance(x, numbers.Number)]
        if len(nums) < 2:
            continue  # Need at least two numbers to form a pair
        for a, b in itertools.combinations(nums, 2):
            abs_product = abs(a * b)
            if (max_abs_product is None) or (abs_product > max_abs_product):
                max_abs_product = abs_product
    if max_abs_product is None:
        raise ValueError("No valid pairs found in input.")
    return max_abs_product
