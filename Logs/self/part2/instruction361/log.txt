 def index_of_smallest_triangular_with_n_digits(n):
    """
    Returns the index of the smallest triangular number with exactly n digits.
    """
    index = 1
    while True:
        triangular = index * (index + 1) // 2
        if len(str(triangular)) == n:
            return index
        index += 1
