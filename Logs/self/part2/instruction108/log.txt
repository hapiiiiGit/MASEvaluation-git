 def are_all_bits_unset(number: int, start: int, end: int) -> bool:
    """
    Check whether all bits are unset (0) in the given range [start, end] (inclusive) of the number.
    Bit positions are 0-indexed from the least significant bit.

    Args:
        number (int): The number to check.
        start (int): The starting bit position (inclusive).
        end (int): The ending bit position (inclusive).

    Returns:
        bool: True if all bits in the range are unset, False otherwise.
    """
    if not isinstance(number, int) or not isinstance(start, int) or not isinstance(end, int):
        raise TypeError("All arguments must be integers.")
    if number < 0 or start < 0 or end < 0:
        raise ValueError("Number and bit positions must be non-negative.")
    if start > end:
        raise ValueError("Start position must be less than or equal to end position.")

    bit_length = number.bit_length()
    # Allow checking bits beyond the current bit length (they are considered unset)
    for pos in range(start, end + 1):
        if pos < bit_length:
            if (number >> pos) & 1:
                return False
    return True
