 def remove_first_last_occurrence(s: str, ch: str) -> str:
    # Validate input
    if not isinstance(s, str):
        raise ValueError("Input string must be of type str.")
    if not isinstance(ch, str) or len(ch) != 1:
        raise ValueError("Character to remove must be a single character string.")

    first_idx = s.find(ch)
    last_idx = s.rfind(ch)

    # If the character does not occur at least twice, return the string unchanged
    if first_idx == -1 or first_idx == last_idx:
        return s

    # Remove first occurrence
    s = s[:first_idx] + s[first_idx+1:]
    # Adjust last_idx since the string is now shorter by one character
    last_idx = last_idx - 1
    # Remove last occurrence
    s = s[:last_idx] + s[last_idx+1:]

    return s
