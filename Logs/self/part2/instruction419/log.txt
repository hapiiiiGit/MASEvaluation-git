 def count_rotations(arr):
    """
    Returns the number of rotations required to generate a sorted array.
    The number of rotations is the index of the minimum element.
    Handles empty arrays and arrays with duplicates.
    """
    n = len(arr)
    if n == 0:
        return 0
    if n == 1 or arr[0] < arr[-1]:
        return 0

    left, right = 0, n - 1
    while left <= right:
        if arr[left] <= arr[right]:
            return left
        mid = left + (right - left) // 2
        next_idx = (mid + 1) % n
        prev_idx = (mid - 1 + n) % n

        if arr[mid] <= arr[next_idx] and arr[mid] <= arr[prev_idx]:
            return mid
        elif arr[mid] >= arr[left]:
            left = mid + 1
        else:
            right = mid - 1
    return 0
