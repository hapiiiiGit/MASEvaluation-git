    # Validate that array is a sequence (list, tuple, etc.)
    if not isinstance(array, (list, tuple)):

    # Validate that k is an integer
    if not isinstance(k, int):
        raise TypeError("k must be an integer.")
    
    # Validate that k is within the valid range (1-based indexing)

        raise IndexError("k is out of bounds for the array.")
    
    # Access the kth element (1-based indexing)
