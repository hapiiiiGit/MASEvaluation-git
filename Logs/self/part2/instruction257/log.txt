 def tuple_intersection(tuple_list):
    # Validate input: must be a list of tuples
    if not isinstance(tuple_list, list) or not all(isinstance(t, tuple) for t in tuple_list):
        raise ValueError("Input must be a list of tuples.")

    if not tuple_list:
        return []

    # Normalize each tuple by sorting its elements
    normalized_tuples = [tuple(sorted(t)) for t in tuple_list]

    # Count occurrences of each normalized tuple
    from collections import Counter
    tuple_counts = Counter(normalized_tuples)

    # The intersection is the set of tuples that appear in all input tuples
    # Since the requirement is to find intersection, we need to find tuples that are present in all
    # For a list of tuples, intersection means the tuples that are common to all lists
    # But since we have a flat list, we assume intersection means duplicates (if any)
    # However, more likely, the requirement is to find the intersection of elements across all tuples

    # Find the intersection of elements in all tuples, irrespective of order
    sets = [set(t) for t in tuple_list]
    if not sets:
        return []
    intersection = set.intersection(*sets)

    # Return as a tuple (since intersection of elements)
    return tuple(intersection)
