 def snake_to_camel(snake_str):
    # Validate input: must be a string and match snake_case pattern
    if not isinstance(snake_str, str):
        raise TypeError("Input must be a string.")
    if not snake_str or not all(c.islower() or c == '_' or c.isdigit() for c in snake_str):
        raise ValueError("Input must be a non-empty snake_case string containing only lowercase letters, digits, and underscores.")
    if snake_str.startswith('_') or snake_str.endswith('_') or '__' in snake_str:
        raise ValueError("Input must not start/end with underscore or contain consecutive underscores.")

    words = snake_str.split('_')
    # Capitalize first letter of each word except the first
    camel_case = words[0] + ''.join(word.capitalize() for word in words[1:])
    return camel_case
