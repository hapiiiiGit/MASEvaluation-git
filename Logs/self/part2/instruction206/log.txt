 from collections import Counter
from typing import List, Tuple, Any, Union

def find_common_first_elements(tuples_list: List[Tuple[Any, ...]]) -> Union[List[Any], str]:
    # Validate input
    if not isinstance(tuples_list, list) or not all(isinstance(t, tuple) and len(t) > 0 for t in tuples_list):
        raise ValueError("Input must be a list of non-empty tuples.")
    
    # Extract first elements
    first_elements = [t[0] for t in tuples_list]
    
    # Count occurrences
    counts = Counter(first_elements)
    
    # Find elements that appear more than once
    common_elements = [elem for elem, count in counts.items() if count > 1]
    
    # Return result
    if common_elements:
        return common_elements
    else:
        return "No common first element found."
