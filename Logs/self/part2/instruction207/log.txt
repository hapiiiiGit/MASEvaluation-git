 def find_max_length_element(lst):
    if not isinstance(lst, list):
        raise TypeError("Input must be a list.")
    if not lst:
        raise ValueError("Input list is empty.")

    max_length = -1
    max_element = None
    found = False

    for elem in lst:
        try:
            elem_length = len(elem)
        except TypeError:
            continue  # Skip elements that do not have a length
        if elem_length > max_length:
            max_length = elem_length
            max_element = elem
            found = True

    if not found:
        raise ValueError("No element with length found in the list.")

    return max_element
