 from functools import reduce
from operator import ixor

def xor_across_tuples(*tuples):
    # Validate that all inputs are tuples containing only integers
    if not all(isinstance(t, tuple) for t in tuples):
        raise TypeError("All inputs must be tuples.")
    for t in tuples:
        if not all(isinstance(x, int) for x in t):
            raise TypeError("All elements in each tuple must be integers.")

    # Flatten the tuples into a single sequence of integers
    elements = [x for t in tuples for x in t]

    # If no elements, raise an error
    if not elements:
        raise ValueError("No integers provided for xor operation.")

    # Apply bitwise xor across all elements
    result = reduce(ixor, elements)

    return result
