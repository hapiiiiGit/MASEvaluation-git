 def largest_number_from_digits(digits):
    # Validate input: check if digits is a list of integers between 0 and 9
    if not isinstance(digits, list):
        raise ValueError("Input must be a list of digits.")
    for d in digits:
        if not isinstance(d, int) or d < 0 or d > 9:
            raise ValueError("All elements must be integers between 0 and 9.")

    # Sort digits in descending order
    sorted_digits = sorted(digits, reverse=True)

    # Combine digits into a single number (as string)
    largest_number_str = ''.join(str(d) for d in sorted_digits)

    # Return the result as an integer
    return int(largest_number_str) if largest_number_str else 0
