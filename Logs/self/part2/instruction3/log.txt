 def differ_at_one_bit_position(a: int, b: int) -> bool:
    """
    Check if two integers differ at exactly one bit position.

    Args:
        a (int): First integer.
        b (int): Second integer.

    Returns:
        bool: True if they differ at exactly one bit position, False otherwise.
    """
    xor_result = a ^ b
    return xor_result != 0 and (xor_result & (xor_result - 1)) == 0
