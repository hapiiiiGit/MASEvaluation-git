 def count_odd_rotations(binary_str: str, rotations: int) -> int:
    count = 0
    n = len(binary_str)
    current = binary_str
    for i in range(rotations):
        # Rotate left by 1
        current = current[1:] + current[0]
        value = int(current, 2)
        if value % 2 == 1:
            count += 1
    return count
