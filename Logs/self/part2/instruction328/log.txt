 import math

def largest_inscribed_triangle_area(radius):
    """
    Returns the area of the largest triangle that can be inscribed in a semicircle of given radius.
    """
    # The largest triangle is an isosceles triangle with its base as the diameter of the semicircle
    # and its vertex at the highest point of the semicircle.
    # Base = 2 * radius, Height = radius
    base = 2 * radius
    height = radius
    area = 0.5 * base * height
    return area
