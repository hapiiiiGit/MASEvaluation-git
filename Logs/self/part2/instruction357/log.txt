 def toggle_bits_except_first_last(n: int) -> int:
    # Convert number to binary string without '0b' prefix
    bin_str = bin(n)[2:]
    length = len(bin_str)
    
    # If the number has less than 3 bits, nothing to toggle
    if length < 3:
        return n

    # Convert to list for mutability
    bits = list(bin_str)
    
    # Toggle bits except first and last
    for i in range(1, length - 1):
        bits[i] = '1' if bits[i] == '0' else '0'
    
    # Reconstruct binary string and convert to integer
    toggled_bin_str = ''.join(bits)
    return int(toggled_bin_str, 2)
