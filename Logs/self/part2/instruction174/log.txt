 def get_tuple_slice(tpl, start=None, end=None, step=None):
    """
    Returns a slice of the tuple using colon notation.
    
    Parameters:
        tpl (tuple): The input tuple.
        start (int, optional): The starting index of the slice.
        end (int, optional): The ending index of the slice (exclusive).
        step (int, optional): The step of the slice.
        
    Returns:
        tuple: The sliced tuple.
    """
    if not isinstance(tpl, tuple):
        raise TypeError("Input must be a tuple.")
    return tpl[slice(start, end, step)]
