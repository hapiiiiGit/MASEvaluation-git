 def extract_nth_element(tuple_list, n):
    if not isinstance(tuple_list, list):
        raise TypeError("Input must be a list of tuples.")
    if not all(isinstance(t, tuple) for t in tuple_list):
        raise TypeError("All elements in the list must be tuples.")
    if not isinstance(n, int) or n < 0:
        raise ValueError("n must be a non-negative integer.")

    result = []
    for idx, t in enumerate(tuple_list):
        if n >= len(t):
            raise IndexError(f"Tuple at index {idx} does not have an element at position {n}.")
        result.append(t[n])
    return result
