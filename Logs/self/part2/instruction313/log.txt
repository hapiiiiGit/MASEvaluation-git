 def bell_number(n):
    """
    Compute the nth Bell number.

    The Bell number counts the number of ways to partition a set of n elements.
    This implementation uses the Bell triangle method for efficient computation.

    Parameters:
        n (int): The index of the Bell number to compute (n >= 0).

    Returns:
        int: The nth Bell number.

    Raises:
        ValueError: If n is negative.

    Example:
        >>> bell_number(3)
        5
    """
    if not isinstance(n, int):
        raise TypeError("Input n must be an integer.")
    if n < 0:
        raise ValueError("Input n must be non-negative.")

    bell = [0] * (n + 1)
    bell[0] = 1

    for i in range(1, n + 1):
        bell[i] = 0
        prev = bell[i - 1]
        for j in range(i):
            bell[i] += prev
            prev = bell[i] - prev

    return bell[n]
