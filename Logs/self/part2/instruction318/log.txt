 def average_per_tuple(tuples):
    if not isinstance(tuples, tuple) or not all(isinstance(inner, tuple) for inner in tuples):
        raise TypeError("Input must be a tuple of tuples.")
    averages = []
    for inner in tuples:
        if len(inner) == 0:
            raise ValueError("Inner tuples must not be empty.")
        avg = sum(inner) / len(inner)
        averages.append(avg)
    return averages
