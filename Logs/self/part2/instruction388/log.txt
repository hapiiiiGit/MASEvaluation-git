 def count_pairs_with_sum(numbers, target_sum):
    # Input validation
    if not isinstance(numbers, list):
        raise ValueError("Input 'numbers' must be a list.")
    if not all(isinstance(num, (int, float)) for num in numbers):
        raise ValueError("All elements in 'numbers' must be integers or floats.")
    if not isinstance(target_sum, (int, float)):
        raise ValueError("Input 'target_sum' must be an integer or float.")

    count = 0
    n = len(numbers)
    # Use a set to avoid counting duplicate pairs
    seen_pairs = set()
    for i in range(n):
        for j in range(i + 1, n):
            if numbers[i] + numbers[j] == target_sum:
                pair = tuple(sorted((numbers[i], numbers[j])))
                if pair not in seen_pairs:
                    seen_pairs.add(pair)
                    count += 1
    return count
