 def max_pair_difference(tuple_list):
    if not isinstance(tuple_list, list):
        raise ValueError("Input must be a list of tuples.")
    max_diff = None
    for item in tuple_list:
        if (not isinstance(item, tuple)) or (len(item) != 2):
            raise ValueError("Each item must be a tuple of two elements.")
        a, b = item
        if not (isinstance(a, (int, float)) and isinstance(b, (int, float))):
            raise ValueError("Tuple elements must be numeric (int or float).")
        diff = abs(a - b)
        if (max_diff is None) or (diff > max_diff):
            max_diff = diff
    return max_diff
