 def extract_strings_by_size(strings, size):
    """
    Extract strings of a specified size from a given list of string values.

    Parameters:
        strings (list): List of string values.
        size (int): The specified size to filter strings.

    Returns:
        list: List of strings whose length matches the specified size.

    Raises:
        ValueError: If inputs are invalid.
    """
    if not isinstance(strings, list):
        raise ValueError("Input must be a list of strings.")
    if not all(isinstance(s, str) for s in strings):
        raise ValueError("All elements in the list must be strings.")
    if not isinstance(size, int) or size <= 0:
        raise ValueError("Size must be a positive integer.")

    return [s for s in strings if len(s) == size]
