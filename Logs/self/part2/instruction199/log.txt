 def min_product_from_tuples(tuple_list):
    # Validate input: must be a list of tuples, each tuple with exactly two numeric elements
    if not isinstance(tuple_list, list):
        raise ValueError("Input must be a list.")
    products = []
    for item in tuple_list:
        if not (isinstance(item, tuple) and len(item) == 2):
            raise ValueError("Each element must be a tuple with exactly two elements.")
        a, b = item
        if not (isinstance(a, (int, float)) and isinstance(b, (int, float))):
            raise ValueError("Tuple elements must be numeric (int or float).")
        products.append(a * b)
    if not products:
        raise ValueError("Input list must contain at least one valid tuple.")
    return min(products)
