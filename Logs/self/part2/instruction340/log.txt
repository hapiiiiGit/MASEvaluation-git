 def remove_parentheses(s: str) -> str:
    result = []
    depth = 0
    for char in s:
        if char == '(':
            depth += 1
        elif char == ')':
            if depth > 0:
                depth -= 1
        elif depth == 0:
            result.append(char)
    return ''.join(result)
