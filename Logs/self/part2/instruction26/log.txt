 def is_monotonic(arr):
    if len(arr) <= 1:
        return True

    non_increasing = True
    non_decreasing = True

    for i in range(1, len(arr)):
        if arr[i] > arr[i - 1]:
            non_increasing = False
        if arr[i] < arr[i - 1]:
            non_decreasing = False

    return non_increasing or non_decreasing
