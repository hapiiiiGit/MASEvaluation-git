 def lists_to_nested_dict(*lists):
    if len(lists) < 2:
        raise ValueError("At least two lists are required.")

    # Check all inputs are lists and not empty
    for idx, lst in enumerate(lists):
        if not isinstance(lst, list):
            raise TypeError(f"Argument {idx+1} is not a list.")
        if not lst:
            raise ValueError(f"List {idx+1} is empty.")

    # Find the minimum length to avoid IndexError
    min_len = min(len(lst) for lst in lists)

    def build_nested(idx, pos):
        if idx == len(lists) - 1:
            return lists[idx][pos]
        return {lists[idx][pos]: build_nested(idx + 1, pos)}

    result = {}
    for i in range(min_len):
        current = build_nested(0, i)
        # Merge current into result
        def merge(d1, d2):
            for k, v in d2.items():
                if k in d1 and isinstance(d1[k], dict) and isinstance(v, dict):
                    merge(d1[k], v)
                else:
                    d1[k] = v
        merge(result, current)

    return result
