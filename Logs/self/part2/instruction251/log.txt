 def all_values_same(d):
    """
    Check if all values in the dictionary are the same.

    Args:
        d (dict): The input dictionary.

    Returns:
        bool: True if all values are the same, False otherwise.
    """
    values = list(d.values())
    if not values:
        return True  # Empty dictionary, considered as all values same
    first_value = values[0]
    return all(value == first_value for value in values)
