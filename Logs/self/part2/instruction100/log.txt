 def sum_in_range(numbers, start_idx, end_idx):
    """
    Returns the sum of numbers in the list 'numbers' from start_idx to end_idx (inclusive).
    
    Args:
        numbers (list): List of numbers.
        start_idx (int): Start index (inclusive).
        end_idx (int): End index (inclusive).
        
    Returns:
        int or float: Sum of the numbers in the specified range.
        
    Raises:
        ValueError: If the input list is empty or indices are out of bounds or invalid.
    """
    if not isinstance(numbers, list) or len(numbers) == 0:
        raise ValueError("Input list must be a non-empty list.")
    if not (isinstance(start_idx, int) and isinstance(end_idx, int)):
        raise ValueError("Indices must be integers.")
    if start_idx < 0 or end_idx < 0 or start_idx >= len(numbers) or end_idx >= len(numbers):
        raise ValueError("Indices are out of bounds.")
    if start_idx > end_idx:
        raise ValueError("Start index cannot be greater than end index.")
    
    sublist = numbers[start_idx:end_idx+1]
    return sum(sublist)
