 def trim_tuples_by_k(tuple_list, k):
    """
    Trims k elements from both the start and end of each tuple in the list.
    If a tuple has fewer than 2*k elements, returns an empty tuple for that entry.

    Args:
        tuple_list (list of tuples): The list of tuples to trim.
        k (int): Number of elements to trim from both ends.

    Returns:
        list of tuples: The list of trimmed tuples.
    """
    trimmed = []
    for t in tuple_list:
        if len(t) < 2 * k:
            trimmed.append(())
        else:
            trimmed.append(t[k:len(t)-k])
    return trimmed
