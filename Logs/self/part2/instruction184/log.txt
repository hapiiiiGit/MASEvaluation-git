 def max_sum(n, memo=None):
    if memo is None:
        memo = {}
    if n == 0:
        return 0
    if n in memo:
        return memo[n]
    sum_parts = (
        max_sum(n // 2, memo) +
        max_sum(n // 3, memo) +
        max_sum(n // 4, memo) +
        max_sum(n // 5, memo)
    )
    result = max(sum_parts, n)
    memo[n] = result
    return result
