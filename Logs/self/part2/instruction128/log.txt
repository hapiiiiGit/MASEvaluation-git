 import math

def complex_to_polar(z):
    if not isinstance(z, complex):
        raise TypeError("Input must be a complex number.")
    real = z.real
    imag = z.imag
    magnitude = math.hypot(real, imag)
    angle = math.atan2(imag, real)
    return (magnitude, angle)
