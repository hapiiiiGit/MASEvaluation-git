 def sum_odd_natural_numbers(l, r):
    # Validate input: l and r must be integers, l <= r, and both >= 1 (natural numbers)
    if not (isinstance(l, int) and isinstance(r, int)):
        raise ValueError("Both l and r must be integers.")
    if l > r:
        raise ValueError("l must be less than or equal to r.")
    if l < 1 or r < 1:
        raise ValueError("Both l and r must be natural numbers (>= 1).")
    
    total = 0
    for num in range(l, r + 1):
        if num % 2 == 1:
            total += num
    return total
