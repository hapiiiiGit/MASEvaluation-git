 def follows_pattern(sequence, patterns):
    # Validate inputs
    if not isinstance(sequence, (list, tuple)) or not isinstance(patterns, (list, tuple)):
        raise ValueError("Both sequence and patterns must be lists or tuples.")
    if len(sequence) != len(patterns):
        return False

    # Compare each element
    for seq_elem, pat_elem in zip(sequence, patterns):
        if seq_elem != pat_elem:
            return False
    return True
