 def count_substrings_with_sum_equal_length(s: str) -> int:
    count = 0
    n = len(s)
    for i in range(n):
        sum_digits = 0
        for j in range(i, n):
            sum_digits += int(s[j])
            length = j - i + 1
            if sum_digits == length:
                count += 1
    return count
