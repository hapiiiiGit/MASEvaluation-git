 from typing import List, Tuple

def k_smallest_pairs(nums1: List[int], nums2: List[int], k: int) -> List[Tuple[int, int]]:
    # Validate input
    if not isinstance(nums1, list) or not isinstance(nums2, list):
        raise ValueError("Both nums1 and nums2 must be lists.")
    if not nums1 or not nums2:
        raise ValueError("Input arrays must be non-empty.")
    if not isinstance(k, int) or k <= 0:
        raise ValueError("k must be a positive integer.")

    pairs = []
    for a in nums1:
        for b in nums2:
            pairs.append((a, b, a + b))

    # Sort pairs by their sum
    pairs.sort(key=lambda x: x[2])

    # Select the first k pairs
    result = [(a, b) for a, b, _ in pairs[:k]]

    return result
