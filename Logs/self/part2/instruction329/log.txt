 def find_smallest_missing_number(nums):
    # Validate input: must be a list of sorted natural numbers (positive integers)
    if not isinstance(nums, list):
        raise ValueError("Input must be a list.")
    if any(not isinstance(x, int) or x <= 0 for x in nums):
        raise ValueError("All elements must be positive integers.")
    if any(nums[i] > nums[i+1] for i in range(len(nums)-1)):
        raise ValueError("List must be sorted in ascending order.")

    expected = 1
    for num in nums:
        if num != expected:
            return expected
        expected += 1
    return expected
