 def can_be_sum_of_nonzero_powers_of_2(n):
    # Validate the input
    if not isinstance(n, int) or n <= 0:
        return False

    # Generate non-zero powers of 2 less than or equal to n
    powers = []
    power = 1
    while power <= n:
        powers.append(power)
        power *= 2

    # Check if n can be represented as the sum of one or more of these powers
    # This is equivalent to checking if n can be written as a sum of distinct powers of 2
    # i.e., binary representation of n contains only 0s and 1s (which is always true for positive integers)
    # But since the requirement is "non-zero powers", we need to ensure at least one power is used

    # Since 0 is not a non-zero power, and all positive integers can be represented as sum of non-zero powers of 2,
    # we just need to check if n > 0 and n can be represented using the available powers

    # For completeness, let's use subset sum approach
    from itertools import combinations

    for r in range(1, len(powers) + 1):
        for combo in combinations(powers, r):
            if sum(combo) == n:
                return True
    return False
