 def cube_sum_even_naturals(n):
    """
    Returns the sum of cubes of the first n even natural numbers.

    Parameters:
    n (int): The number of even natural numbers to consider.

    Returns:
    int: The sum of cubes of the first n even natural numbers.
    """
    even_numbers = [2 * i for i in range(1, n + 1)]
    cube_sum = sum(num ** 3 for num in even_numbers)
