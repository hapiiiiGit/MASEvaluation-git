 def remove_tuples(input_tuple):
    """
    Removes all tuple elements from the given tuple.

    Args:
        input_tuple (tuple): The input tuple possibly containing tuple elements.

    Returns:
        tuple: A new tuple with all tuple elements removed.
    """
    return tuple(item for item in input_tuple if not isinstance(item, tuple))
