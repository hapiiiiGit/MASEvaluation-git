    # Flatten the list of lists
    flattened = [item for sublist in list_of_lists for item in sublist]
    # Initialize frequency dictionary
    freq_dict = {}
    # Count frequencies
    for item in flattened:
        freq_dict[item] = freq_dict.get(item, 0) + 1
    return freq_dict
