        if isinstance(element, tuple):

        elif isinstance(element, int):
            return element if element % 2 == 0 else None
        else:
            return element

    def clean_tuple(t):
        result = []
        for item in t:
            if isinstance(item, tuple):
                cleaned = clean_tuple(item)

                    result.append(cleaned)

                result.append(item)
