 def sum_per_digit_difference(a, b):
    # Validate that the inputs are integers
    if not isinstance(a, int) or not isinstance(b, int):
        raise ValueError("Both inputs must be integers.")

    # Convert both integers to strings to access individual digits
    str_a = str(abs(a))
    str_b = str(abs(b))

    # Pad the shorter string with leading zeros so both have the same length
    max_len = max(len(str_a), len(str_b))
    str_a = str_a.zfill(max_len)
    str_b = str_b.zfill(max_len)

    # Iterate through each digit position and calculate the absolute difference
    total = 0
    for digit_a, digit_b in zip(str_a, str_b):
        total += abs(int(digit_a) - int(digit_b))

    # Return the final sum
    return total
