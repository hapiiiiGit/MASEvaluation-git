 def index_wise_addition(tuple1, tuple2):
    # Validate that both inputs are tuples
    if not (isinstance(tuple1, tuple) and isinstance(tuple2, tuple)):
        raise TypeError("Both inputs must be tuples.")
    # Validate that both tuples have the same length
    if len(tuple1) != len(tuple2):
        raise ValueError("Both tuples must have the same length.")
    # Validate that all elements are tuples and have the same length
    for t1, t2 in zip(tuple1, tuple2):
        if not (isinstance(t1, tuple) and isinstance(t2, tuple)):
            raise TypeError("All elements of the input tuples must be tuples.")
        if len(t1) != len(t2):
            raise ValueError("Inner tuples must have the same length.")

    # Perform index-wise addition
    result = tuple(
        tuple(a + b for a, b in zip(inner1, inner2))
        for inner1, inner2 in zip(tuple1, tuple2)
    )
    return result
