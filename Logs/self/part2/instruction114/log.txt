 def set_all_even_bits(n: int) -> int:
    # Determine the number of bits needed to represent n
    num_bits = n.bit_length()
    # To handle cases where n is 0, ensure at least 1 bit is considered
    num_bits = max(num_bits, 1)
    # Create a mask with all even bits set to 1
    mask = 0
    for i in range(0, num_bits, 2):  # Even positions: 0, 2, 4, ...
        mask |= (1 << i)
    # Set even bits in n
    return n | mask
