 def find_kth_element(arr1, arr2, k):
    """
    Find the kth element from the merged sorted array formed by arr1 and arr2.
    k is 1-based index.
    """
    # Validate input
    if not isinstance(arr1, list) or not isinstance(arr2, list):
        raise ValueError("Input arrays must be lists.")
    if not all(arr1[i] <= arr1[i+1] for i in range(len(arr1)-1)):
        raise ValueError("First array is not sorted.")
    if not all(arr2[i] <= arr2[i+1] for i in range(len(arr2)-1)):
        raise ValueError("Second array is not sorted.")
    total_len = len(arr1) + len(arr2)
    if k < 1 or k > total_len:
        raise ValueError("k is out of valid range.")

    # Helper function for binary search
    def kth(arr1, arr2, k):
        if not arr1:
            return arr2[k-1]
        if not arr2:
            return arr1[k-1]
        if k == 1:
            return min(arr1[0], arr2[0])

        i = min(len(arr1), k // 2)
        j = min(len(arr2), k // 2)

        if arr1[i-1] <= arr2[j-1]:
            return kth(arr1[i:], arr2, k - i)
        else:
            return kth(arr1, arr2[j:], k - j)

    return kth(arr1, arr2, k)
