 def count_elements_before_target(input_tuple, target_element):
    """
    Returns the number of elements that occur before the first occurrence of the target_element in input_tuple.
    If input_tuple is not a tuple or is empty, raises ValueError.
    If target_element is not found, returns -1.
    """
    if not isinstance(input_tuple, tuple) or len(input_tuple) == 0:
        raise ValueError("Input must be a non-empty tuple.")

    try:
        index = input_tuple.index(target_element)
        return index
    except ValueError:
        return -1
