 import math

def surface_area_square_pyramid(base_edge: float, height: float) -> float:
    """
    Calculate the surface area of a square pyramid.

    Parameters:
    base_edge (float): The length of the base edge of the pyramid.
    height (float): The vertical height from the base to the apex.

    Returns:
    float: The surface area of the square pyramid.
    """
    # Calculate the slant height using the Pythagorean theorem
    slant_height = math.sqrt((base_edge / 2) ** 2 + height ** 2)
    
    # Area of the square base
    base_area = base_edge ** 2
    
    # Area of the four triangular faces
    lateral_area = 2 * base_edge * slant_height
    
    # Total surface area
    surface_area = base_area + lateral_area
    
    return surface_area
