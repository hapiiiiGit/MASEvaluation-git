    """
    Generate the first n Perrin numbers.
    Perrin sequence: P(0) = 3, P(1) = 0, P(2) = 2
    P(n) = P(n-2) + P(n-3) for n >= 3
    """
    if n <= 0:
        return []
    perrin = [3, 0, 2]
    if n <= 3:
