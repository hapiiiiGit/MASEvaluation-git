 def remove_duplicates_from_lists(*lists):
    from collections import Counter

    # Flatten all lists to count occurrences of each number
    all_numbers = [num for lst in lists for num in lst]
    num_counts = Counter(all_numbers)

    # Only keep numbers that appear once across all lists
    result = []
    for lst in lists:
        filtered = [num for num in lst if num_counts[num] == 1]
        result.append(filtered)
    return result
