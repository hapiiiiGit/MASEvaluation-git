 def apply_format_to_list(input_list, format_string):
    # Validate input_list is iterable
    if not hasattr(input_list, '__iter__'):
        raise TypeError("input_list must be iterable")
    
    # Validate format_string is a string
    if not isinstance(format_string, str):
        raise TypeError("format_string must be a string")
    
    formatted_list = []
    for elem in input_list:
        try:
            formatted_elem = format_string.format(elem)
        except Exception as e:
            raise ValueError(f"Element '{elem}' cannot be formatted with the given format string: {e}")
        formatted_list.append(formatted_elem)
    
    return formatted_list
