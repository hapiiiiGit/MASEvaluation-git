 from typing import List

def n_largest_integers(numbers: List[int], n: int) -> List[int]:
    # Validate input list
    if not isinstance(numbers, list):
        raise TypeError("Input must be a list.")
    if not all(isinstance(x, int) for x in numbers):
        raise ValueError("All elements in the list must be integers.")
    
    # Validate n
    if not isinstance(n, int):
        raise TypeError("n must be an integer.")
    if n <= 0:
        raise ValueError("n must be a positive integer.")
    if n > len(numbers):
        raise ValueError("n cannot be greater than the length of the list.")
    
    # Find n largest integers and sort in descending order
    largest = sorted(numbers, reverse=True)[:n]
    return largest
