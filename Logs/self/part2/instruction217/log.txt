 def extract_elementwise_and_tuples(tuple1, tuple2):
    # Validate inputs
    if not isinstance(tuple1, tuple) or not isinstance(tuple2, tuple):
        raise TypeError("Both inputs must be tuples.")
    if len(tuple1) != len(tuple2):
        raise ValueError("Both tuples must have the same length.")

    # Perform elementwise 'and' operation
    and_result = tuple(a and b for a, b in zip(tuple1, tuple2))

    # Return the result and the original tuples
    return (and_result, tuple1, tuple2)
