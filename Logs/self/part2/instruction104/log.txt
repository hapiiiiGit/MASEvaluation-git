 def is_majority_element(arr, n, element):
    # Validate input
    if not isinstance(arr, list) or not isinstance(n, int) or n != len(arr):
        raise ValueError("Invalid input: array and length do not match or are not valid types.")
    if n == 0:
        return False
    if element is None:
        raise ValueError("Element must be provided.")

    # Check if array is sorted
    if any(arr[i] > arr[i+1] for i in range(n-1)):
        raise ValueError("Array must be sorted.")

    # Efficiently count occurrences using binary search
    def find_first(arr, n, x):
        left, right = 0, n-1
        result = -1
        while left <= right:
            mid = (left + right) // 2
            if arr[mid] == x:
                result = mid
                right = mid - 1
            elif arr[mid] < x:
                left = mid + 1
            else:
                right = mid - 1
        return result

    def find_last(arr, n, x):
        left, right = 0, n-1
        result = -1
        while left <= right:
            mid = (left + right) // 2
            if arr[mid] == x:
                result = mid
                left = mid + 1
            elif arr[mid] < x:
                left = mid + 1
            else:
                right = mid - 1
        return result

    first = find_first(arr, n, element)
    if first == -1:
        return False
    last = find_last(arr, n, element)
    count = last - first + 1

    return count > n // 2
