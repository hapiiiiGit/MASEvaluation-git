 import math

def find_third_side(side1, side2, missing_side='hypotenuse'):
    """
    Calculate the third side of a right-angled triangle using the Pythagorean theorem.

    Parameters:
    - side1 (float): Length of the first known side (must be positive).
    - side2 (float): Length of the second known side (must be positive).
    - missing_side (str): Which side to calculate: 'hypotenuse', 'leg1', or 'leg2'.
                          Default is 'hypotenuse'.

    Returns:
    - float: Length of the missing side.

    Raises:
    - ValueError: If inputs are invalid or do not form a valid right-angled triangle.
    """
    # Validate input values
    if not (isinstance(side1, (int, float)) and isinstance(side2, (int, float))):
        raise ValueError("Side lengths must be numbers.")
    if side1 <= 0 or side2 <= 0:
        raise ValueError("Side lengths must be positive numbers.")

    missing_side = missing_side.lower()
    if missing_side == 'hypotenuse':
        # Calculate hypotenuse
        hypotenuse = math.hypot(side1, side2)
        return hypotenuse
    elif missing_side in ('leg1', 'leg2'):
        # Calculate a leg, side1 and side2 must be hypotenuse and leg
        hypotenuse = max(side1, side2)
        leg = min(side1, side2)
        if hypotenuse <= leg:
            raise ValueError("Hypotenuse must be the longest side.")
        diff = hypotenuse ** 2 - leg ** 2
        if diff < 0:
            raise ValueError("Invalid triangle sides: cannot compute square root of negative number.")
        return math.sqrt(diff)
    else:
        raise ValueError("missing_side must be 'hypotenuse', 'leg1', or 'leg2'.")
