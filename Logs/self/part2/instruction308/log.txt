 def is_lucid_number(n):
    """
    A lucid number is a positive integer that can be represented as the sum of distinct powers of 2,
    where the exponents are consecutive integers starting from 0.
    That is, numbers of the form 2^0 + 2^1 + ... + 2^k for some k >= 0.
    These numbers are 1, 3, 7, 15, 31, 63, 127, ...
    """
    # A lucid number is of the form 2^(k+1) - 1
    # So, n+1 should be a power of 2
    return n > 0 and ((n + 1) & n) == 0

def get_lucid_numbers(upto):
    """
    Returns a list of all lucid numbers less than or equal to 'upto'.
    """
    result = []
    for i in range(1, upto + 1):
        if is_lucid_number(i):
            result.append(i)
    return result
