def is_prime(num):
    if num < 2:
        return False
    if num == 2:
        return True
    if num % 2 == 0:
        return False
    sqrt_n = int(math.isqrt(num)) + 1
    for i in range(3, sqrt_n, 2):
        if num % i == 0:
            return False
    return True

def nth_nsw_prime(n):
    # NSW numbers: S(0) = 1, S(1) = 1, S(n) = 2*S(n-1) + S(n-2)
    S_prev2 = 1
    S_prev1 = 1
    count = 0
    idx = 1  # S(1)
    while True:
        if is_prime(S_prev1):
            count += 1
            if count == n:
                return S_prev1
        # Generate next NSW number
        S_next = 2 * S_prev1 + S_prev2
        S_prev2, S_prev1 = S_prev1, S_next
        idx += 1
