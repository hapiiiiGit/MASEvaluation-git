 import math

def min_jumps_to_reach(d, jump_length):
    """
    Calculate the minimum number of jumps required to reach or pass the point (d, 0)
    from the origin (0, 0) on the x-axis, given each jump is of fixed length.

    Parameters:
    d (int): Target distance on the x-axis.
    jump_length (int): Length of each jump.

    Returns:
    int: Minimum number of jumps required.

    Raises:
    ValueError: If jump_length is not a positive integer.
    """
    if not isinstance(d, int) or not isinstance(jump_length, int):
        raise ValueError("Both d and jump_length must be integers.")
    if jump_length <= 0:
        raise ValueError("Jump length must be a positive integer.")

    distance = abs(d)
    if distance == 0:
        return 0

    jumps = math.ceil(distance / jump_length)
    return jumps
