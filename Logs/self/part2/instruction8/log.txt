 from typing import List, Any

def sort_matrix_by_row_sum(matrix: List[List[Any]]) -> List[List[Any]]:
    # Validate input: must be a list of lists with numeric values
    if not isinstance(matrix, list) or not all(isinstance(row, list) for row in matrix):
        raise ValueError("Input must be a list of lists (matrix).")
    for row in matrix:
        if not all(isinstance(item, (int, float)) for item in row):
            raise ValueError("All elements in the matrix must be numeric (int or float).")
    
    # Sort rows by the sum of their elements
    sorted_matrix = sorted(matrix, key=lambda row: sum(row))
    return sorted_matrix
