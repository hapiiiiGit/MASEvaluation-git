 def max_sum_incr_subseq_with_k(arr, i, k):
    """
    Finds the maximum sum of increasing subsequence from prefix until ith index,
    and also including a given kth element (where k > i).
    
    :param arr: List[int] - input array
    :param i: int - end index of prefix (inclusive)
    :param k: int - index of the element to be included (k > i)
    :return: int - maximum sum
    """
    n = len(arr)
    if not (0 <= i < n and 0 <= k < n and k > i):
        raise ValueError("Invalid indices: ensure 0 <= i < n, 0 <= k < n, and k > i")
    
    # dp[j]: max sum of increasing subsequence ending at j (0 <= j <= i)
    dp = [arr[j] for j in range(i+1)]
    
    for j in range(i+1):
        for l in range(j):
            if arr[l] < arr[j]:
                dp[j] = max(dp[j], dp[l] + arr[j])
    
    max_sum = 0
    for j in range(i+1):
        if arr[j] < arr[k]:
            max_sum = max(max_sum, dp[j] + arr[k])
    
    return max_sum
