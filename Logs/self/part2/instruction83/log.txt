 def ratio_zeroes_to_nonzeroes(arr):
    if not isinstance(arr, list):
        raise TypeError("Input must be a list of integers.")
    if not all(isinstance(x, int) for x in arr):
        raise ValueError("All elements in the array must be integers.")

    zeroes_count = 0
    non_zeroes_count = 0

    for num in arr:
        if num == 0:
            zeroes_count += 1
        else:
            non_zeroes_count += 1

    if non_zeroes_count == 0:
        raise ZeroDivisionError("No non-zero elements in the array; ratio is undefined.")

    return zeroes_count / non_zeroes_count
