 def max_subarray_sum_repeated(arr, k):
    if not arr or k <= 0:
        return 0

    def kadane(nums):
        max_sum = curr_sum = nums[0]
        for num in nums[1:]:
            curr_sum = max(num, curr_sum + num)
            max_sum = max(max_sum, curr_sum)
        return max_sum

    total_sum = sum(arr)
    max_kadane = kadane(arr)

    if k == 1:
        return max_kadane

    # Kadane's on arr repeated twice to handle subarrays crossing the boundary
    max_kadane_twice = kadane(arr * 2)

    if total_sum > 0:
        return max(max_kadane_twice, max_kadane + (k - 2) * total_sum)
    else:
        return max_kadane_twice
