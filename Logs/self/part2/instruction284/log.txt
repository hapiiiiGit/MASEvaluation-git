 def interleave_three_lists(list1, list2, list3):
    # Validate that all inputs are lists
    if not all(isinstance(lst, list) for lst in (list1, list2, list3)):
        raise TypeError("All inputs must be lists.")
    
    # Check that all lists have the same length
    length = len(list1)
    if len(list2) != length or len(list3) != length:
        raise ValueError("All lists must have the same length.")
    
    # Interleave the elements
    result = []
    for i in range(length):
        result.append(list1[i])
        result.append(list2[i])
        result.append(list3[i])
    
    return result
