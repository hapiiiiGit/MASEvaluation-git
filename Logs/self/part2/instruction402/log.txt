 def rgb_to_hsv(r, g, b):
    """
    Convert RGB color to HSV color.
    Input:
        r, g, b: Integers in the range 0-255
    Output:
        h: Hue in range 0-360
        s: Saturation in percentage (0-100)
        v: Value in percentage (0-100)
    Returns:
        (h, s, v) as a tuple
    """
    # Validate input
    if not (0 <= r <= 255 and 0 <= g <= 255 and 0 <= b <= 255):
        raise ValueError("RGB values must be in the range 0-255.")

    # Normalize RGB values to [0, 1]
    r_norm = r / 255.0
    g_norm = g / 255.0
    b_norm = b / 255.0

    max_c = max(r_norm, g_norm, b_norm)
    min_c = min(r_norm, g_norm, b_norm)
    delta = max_c - min_c

    # Compute Hue
    if delta == 0:
        h = 0
    elif max_c == r_norm:
        h = 60 * (((g_norm - b_norm) / delta) % 6)
    elif max_c == g_norm:
        h = 60 * (((b_norm - r_norm) / delta) + 2)
    elif max_c == b_norm:
        h = 60 * (((r_norm - g_norm) / delta) + 4)

    # Compute Saturation
    if max_c == 0:
        s = 0
    else:
        s = (delta / max_c) * 100

    # Compute Value
    v = max_c * 100

    # Ensure Hue is in [0, 360]
    h = round(h)
    s = round(s)
    v = round(v)

    return (h, s, v)
