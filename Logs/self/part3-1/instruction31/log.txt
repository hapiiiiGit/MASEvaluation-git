 import os
import subprocess
import zipfile

# Step 2: Create a Python virtual environment
def create_virtualenv(env_name):
    subprocess.run(['python3', '-m', 'venv', env_name])
    print(f"Virtual environment '{env_name}' created.")

# Step 3: Install KeyBERT and keyphrase-vectorizers
def install_dependencies(env_name):
    subprocess.run([os.path.join(env_name, 'bin', 'pip'), 'install', 'keybert', 'keyphrase-vectorizers'])
    print("Dependencies installed.")

# Step 4: Package the installed libraries into a .zip file
def package_lambda_layer(env_name, zip_file_name):
    site_packages = os.path.join(env_name, 'lib', 'python3.*/site-packages')
    with zipfile.ZipFile(zip_file_name, 'w') as zipf:
        for root, dirs, files in os.walk(site_packages):
            for file in files:
                file_path = os.path.join(root, file)
                zipf.write(file_path, os.path.relpath(file_path, site_packages))
    print(f"Packaged libraries into '{zip_file_name}'.")

# Step 5: Write a sample AWS Lambda function
def create_lambda_function():
    lambda_function_code = """
import json
from keybert import KeyBERT

def lambda_handler(event, context):
    text = event.get('text', '')
    kw_model = KeyBERT()
    keywords = kw_model.extract_keywords(text)
    return {
        'statusCode': 200,
        'body': json.dumps(keywords)
    }
"""
    with open('lambda_function.py', 'w') as f:
        f.write(lambda_function_code)
    print("Sample Lambda function created.")

# Main execution
if __name__ == "__main__":
    env_name = 'lambda_layer_env'
    zip_file_name = 'keybert_layer.zip'
    
    create_virtualenv(env_name)
    install_dependencies(env_name)
    package_lambda_layer(env_name, zip_file_name)
    create_lambda_function()
