 import requests
import csv
import pandas as pd
import logging
from flask import Flask, request, jsonify

# Set up logging
logging.basicConfig(level=logging.INFO)

# Flask app for the Chrome Extension
app = Flask(__name__)

# WooCommerce API credentials
WC_API_URL = 'https://your-woocommerce-site.com/wp-json/wc/v3/products'
WC_CONSUMER_KEY = 'your_consumer_key'
WC_CONSUMER_SECRET = 'your_consumer_secret'

def fetch_product_data():
    try:
        response = requests.get(WC_API_URL, auth=(WC_CONSUMER_KEY, WC_CONSUMER_SECRET))
        response.raise_for_status()
        return response.json()
    except requests.exceptions.RequestException as e:
        logging.error(f"Error fetching product data: {e}")
        return []

def process_data(products):
    processed_data = []
    for product in products:
        processed_data.append({
            'ID': product['id'],
            'Name': product['name'],
            'Price': product['price'],
            'SKU': product['sku'],
            'Stock': product['stock_quantity'],
            'Description': product['description']
        })
    return processed_data

def generate_csv(data, filename='products.csv'):
    keys = data[0].keys()
    with open(filename, 'w', newline='', encoding='utf-8') as output_file:
        dict_writer = csv.DictWriter(output_file, fieldnames=keys)
        dict_writer.writeheader()
        dict_writer.writerows(data)

def generate_excel(data, filename='products.xlsx'):
    df = pd.DataFrame(data)
    df.to_excel(filename, index=False)

@app.route('/scrape', methods=['POST'])
def scrape():
    products = fetch_product_data()
    if not products:
        return jsonify({"error": "No products found"}), 404

    processed_data = process_data(products)
    generate_csv(processed_data)
    generate_excel(processed_data)

    return jsonify({"message": "Data scraped and files generated successfully!"}), 200

if __name__ == '__main__':
    app.run(port=5000)
