 import sqlite3
import requests
import logging

# Configure logging
logging.basicConfig(level=logging.INFO, format='%(asctime)s - %(levelname)s - %(message)s')

def connect_to_database(db_path):
    """Establish a connection to the database."""
    try:
        conn = sqlite3.connect(db_path)
        logging.info("Database connection established.")
        return conn
    except Exception as e:
        logging.error(f"Error connecting to database: {e}")
        return None

def retrieve_user_data(conn):
    """Extract relevant user data from the database."""
    cursor = conn.cursor()
    cursor.execute("SELECT name, email, password FROM users")  # Example query
    user_data = cursor.fetchall()
    logging.info("User data retrieved from database.")
    return user_data

def fill_registration_form(user_data):
    """Automate the filling of the web registration form."""
    url = "https://example.com/register"  # Replace with actual registration URL
    for user in user_data:
        name, email, password = user
        payload = {
            'name': name,
            'email': email,
            'password': password
        }
        try:
            response = requests.post(url, data=payload)
            response.raise_for_status()
            logging.info(f"Successfully submitted registration for {name}.")
        except requests.exceptions.RequestException as e:
            logging.error(f"Failed to submit registration for {name}: {e}")

def main():
    db_path = 'path_to_your_database.db'  # Replace with actual database path
    conn = connect_to_database(db_path)
    if conn:
        user_data = retrieve_user_data(conn)
        fill_registration_form(user_data)
        conn.close()

if __name__ == "__main__":
    main()
