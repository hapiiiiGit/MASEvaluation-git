class CandidateDatabase:
    def __init__(self):
        self.candidates = []

    def add_candidate(self, candidate):
        self.candidates.append(candidate)

    def screen_candidates(self, required_skills):
        screened_candidates = []
        for candidate in self.candidates:
            if all(skill in candidate.skills for skill in required_skills):
                screened_candidates.append(candidate)
        return screened_candidates

    def present_candidates(self, candidates):
        for candidate in candidates:
            print(f"Name: {candidate.name}")
            print(f"Skills: {', '.join(candidate.skills)}")
            print(f"CV: {candidate.cv}")
            print(f"Interview Notes: {candidate.interview_notes}\n")
