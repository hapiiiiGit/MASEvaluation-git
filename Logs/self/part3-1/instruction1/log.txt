def fetch_price_data(index, start_date, end_date):
    nse = Nse()
    price_data = nse.get_historical_data(index, start_date, end_date)
    return price_data

# Function to calculate daily returns
def calculate_daily_returns(price_data):
    price_data['daily_returns'] = price_data['Close'].pct_change()
    return price_data

# Function to calculate average return
def calculate_average_return(price_data):
    average_return = price_data['daily_returns'].mean()
    return average_return

# Function to calculate standard deviation of daily returns
def calculate_standard_deviation(price_data):
    standard_deviation = price_data['daily_returns'].std()
    return standard_deviation

# Function to calculate momentum
def calculate_momentum(price_data):
    price_data['momentum'] = price_data['Close'] - price_data['Close'].shift(30)
    return price_data

# Function to rank stocks based on metrics
def rank_stocks(price_data):
    price_data['rank'] = price_data['daily_returns'] + price_data['average_return'] + price_data['standard_deviation'] + price_data['momentum']
    price_data = price_data.sort_values('rank', ascending=False)
    return price_data

# Function to export data to Excel workbook
def export_to_excel(price_data):
    filename = 'stock_data.xlsx'

    rank_data = price_data[['Stock', 'rank']]
    rank_data.to_excel(writer, index=False, sheet_name='Ranking Table')
    writer.save()
