 from flask import Flask, jsonify
import requests
import schedule
import time
from datetime import datetime, timedelta
import pytz

app = Flask(__name__)

SHOPIFY_API_KEY = 'your_api_key'
SHOPIFY_PASSWORD = 'your_api_password'
SHOPIFY_STORE_NAME = 'your_store_name'
SHOPIFY_API_VERSION = '2023-01'  # Use the appropriate version
TIMEZONE = 'America/New_York'  # Set your store's timezone

order_count = 0

def fetch_daily_order_count():
    global order_count
    url = f"https://{SHOPIFY_API_KEY}:{SHOPIFY_PASSWORD}@{SHOPIFY_STORE_NAME}.myshopify.com/admin/api/{SHOPIFY_API_VERSION}/orders.json"
    today = datetime.now(pytz.timezone(TIMEZONE)).date()
    params = {
        'created_at_min': f"{today}T00:00:00-00:00",
        'created_at_max': f"{today}T23:59:59-00:00",
        'status': 'any'
    }
    response = requests.get(url, params=params)
    if response.status_code == 200:
        orders = response.json().get('orders', [])
        order_count = len(orders)
    else:
        order_count = 0

@app.route('/order_count', methods=['GET'])
def get_order_count():
    return jsonify({'order_count': order_count})

def reset_order_count():
    global order_count
    order_count = 0

# Schedule the reset at midnight
schedule.every().day.at("00:00").do(reset_order_count)

if __name__ == '__main__':
    fetch_daily_order_count()
    schedule.run_pending()
    app.run(host='0.0.0.0', port=5000)
