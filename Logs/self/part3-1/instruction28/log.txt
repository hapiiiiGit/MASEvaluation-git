 from flask import Flask, jsonify, request
import requests
import threading
import time
import websocket

app = Flask(__name__)

live_prices = {}
order_status = "Pending"

def fetch_live_prices():
    global live_prices
    while True:
        # Simulate fetching live prices from an API
        live_prices = {
            "symbol": "BTC/USD",
            "bid": round(30000 + (time.time() % 1000), 2),
            "ask": round(30000 + (time.time() % 1000) + 10, 2)
        }
        time.sleep(0.2)  # Update every 200ms

# WebSocket for live price streaming
def on_message(ws, message):
    print("Received message:", message)

def on_error(ws, error):
    print("Error:", error)

def on_close(ws):
    print("WebSocket closed")

def on_open(ws):
    print("WebSocket opened")

# Start WebSocket in a separate thread
def start_websocket():
    ws = websocket.WebSocketApp("ws://example.com/live-prices",
                                  on_message=on_message,
                                  on_error=on_error,
                                  on_close=on_close)
    ws.run_forever()

# Route to get live prices
@app.route('/live-prices', methods=['GET'])
def get_live_prices():
    return jsonify(live_prices)

# Route to get public IP address
@app.route('/public-ip', methods=['GET'])
def get_public_ip():
    ip_info = requests.get('https://api.ipify.org?format=json').json()
    return jsonify(ip_info)

# Route for one-click trading
@app.route('/trade', methods=['POST'])
def trade():
    global order_status
    data = request.json
    # Simulate order processing
    if 'symbol' in data and 'amount' in data:
        order_status = "Executed"
        return jsonify({"status": order_status, "message": "Order executed successfully."}), 200
    return jsonify({"status": "Failed", "message": "Invalid order data."}), 400

# Route to get order status
@app.route('/order-status', methods=['GET'])
def get_order_status():
    return jsonify({"status": order_status})

if __name__ == '__main__':
    threading.Thread(target=fetch_live_prices).start()
    start_websocket()
    app.run(debug=True)
