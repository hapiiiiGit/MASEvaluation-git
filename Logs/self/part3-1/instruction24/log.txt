 import os
import json
import datetime
from googleapiclient.discovery import build

# Set up the YouTube Data API client
def setup_youtube_client(api_key):
    return build('youtube', 'v3', developerKey=api_key)

# Function to get user input for search term or channel
def get_user_input():
    return input("Enter a search term or channel ID: ")

# Function to fetch the last 10 days of videos based on the search term or channel
def fetch_videos(youtube, search_term, days=10):
    today = datetime.datetime.now()
    published_after = (today - datetime.timedelta(days=days)).isoformat("T") + "Z"
    
    request = youtube.search().list(
        q=search_term,
        part='id,snippet',
        publishedAfter=published_after,
        maxResults=50,
        type='video'
    )
    response = request.execute()
    return response.get('items', [])

# Function to analyze the fetched videos based on specified criteria
def analyze_videos(youtube, videos):
    analyzed_data = []
    for video in videos:
        video_id = video['id']['videoId']
        video_details = youtube.videos().list(
            part='statistics',
            id=video_id
        ).execute()
        
        stats = video_details['items'][0]['statistics']
        analyzed_data.append({
            'title': video['snippet']['title'],
            'publishedAt': video['snippet']['publishedAt'],
            'viewCount': int(stats.get('viewCount', 0)),
            'likeCount': int(stats.get('likeCount', 0)),
            'commentCount': int(stats.get('commentCount', 0))
        })
    return analyzed_data

# Function to write the JSON output to a file
def write_to_json_file(data, filename='output.json'):
    with open(filename, 'w') as json_file:
        json.dump(data, json_file, indent=4)

def main():
    api_key = os.getenv('YOUTUBE_API_KEY')  # Ensure to set your API key in the environment variable
    youtube = setup_youtube_client(api_key)
    
    search_term = get_user_input()
    videos = fetch_videos(youtube, search_term)
    analyzed_data = analyze_videos(youtube, videos)
    write_to_json_file(analyzed_data)

if __name__ == "__main__":
    main()
