 def median(l: list):
    """Return median of elements in the list l.
    >>> median([3, 1, 2, 4, 5])
    3
    >>> median([-10, 4, 6, 1000, 10, 20])
    15.0
    """
    # Input Validation
    if not isinstance(l, list) or not all(isinstance(x, (int, float)) for x in l):
        raise ValueError("Input must be a list of numeric elements.")
    
    # Sorting the List
    l.sort()
    
    # Finding the Median
    n = len(l)
    if n % 2 == 1:
        return l[n // 2]
    else:
        return (l[n // 2 - 1] + l[n // 2]) / 2
