 def sort_even(l: list):
    even_indices = [l[i] for i in range(len(l)) if i % 2 == 0]
    sorted_even_indices = sorted(even_indices)
    result = []
    even_index = 0
    for i in range(len(l)):
        if i % 2 == 0:
            result.append(sorted_even_indices[even_index])
            even_index += 1
        else:
            result.append(l[i])
    return result
