 def shift_character(ch: str, shift: int) -> str:
    if 'a' <= ch <= 'z':
        return chr(((ord(ch) - ord('a') + shift) % 26) + ord('a'))
    raise ValueError("Input must be a lowercase alphabetic character.")

def encode_shift(s: str) -> str:
    if not s.isalpha() or not s.islower():
        raise ValueError("Input must be a string of lowercase alphabetic characters.")
    return "".join(shift_character(ch, 5) for ch in s)

def decode_shift(s: str) -> str:
    if not s.isalpha() or not s.islower():
        raise ValueError("Input must be a string of lowercase alphabetic characters.")
    return "".join(shift_character(ch, -5) for ch in s)
