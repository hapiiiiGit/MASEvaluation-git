 def is_prime(num: int) -> bool:
    if num <= 1:
        return False
    for i in range(2, int(num**0.5) + 1):
        if num % i == 0:
            return False
    return True

def fibonacci_up_to(limit: int):
    fibs = [0, 1]
    while True:
        next_fib = fibs[-1] + fibs[-2]
        if next_fib > limit:
            break
        fibs.append(next_fib)
    return fibs

def prime_fib(n: int):
    """ 
    prime_fib returns n-th number that is a Fibonacci number and it's also prime.
    >>> prime_fib(1)
    2
    >>> prime_fib(2)
    3
    >>> prime_fib(3)
    5
    >>> prime_fib(4)
    13
    >>> prime_fib(5)
    89
    """
    fib_primes = []
    limit = 100000  # Arbitrary limit to generate Fibonacci numbers
    fibs = fibonacci_up_to(limit)
    
    for fib in fibs:
        if is_prime(fib):
            fib_primes.append(fib)
        if len(fib_primes) == n:
            return fib_primes[-1]
