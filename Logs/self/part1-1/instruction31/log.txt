def is_prime(n):

    >>> is_prime(6)
    False
    >>> is_prime(101)
    True
    >>> is_prime(11)
    True
    >>> is_prime(13441)
    True
    >>> is_prime(61)
    True
    >>> is_prime(4)
    False
    >>> is_prime(1)
    False
    """
    # Input Validation
    if not isinstance(n, int):
        raise ValueError("Input must be an integer.")
    
    # Handle Special Cases
    if n < 2:
        return False
    
    # Prime Check Logic

        if n % i == 0:
            return False
