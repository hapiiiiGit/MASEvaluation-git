 def starts_one_ends(n):
    """
    Given a positive integer n, return the count of the numbers of n-digit
    positive integers that start or end with 1.
    """
    if n < 1:
        return 0

    if n == 1:
        return 1  # Only the number 1 itself

    # Count of n-digit integers starting with 1
    start_with_1 = 10 ** (n - 1)

    # Count of n-digit integers ending with 1
    end_with_1 = 10 ** (n - 1)

    # Count of n-digit integers that start and end with 1
    start_and_end_with_1 = 10 ** (n - 2) if n > 1 else 0

    # Combine the results, avoiding double counting
    total_count = start_with_1 + end_with_1 - start_and_end_with_1

    return total_count
