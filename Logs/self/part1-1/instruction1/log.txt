 from typing import List

def separate_paren_groups(paren_string: str) -> List[str]:
    """ Input to this function is a string containing multiple groups of nested parentheses. Your goal is to
    separate those group into separate strings and return the list of those.
    Separate groups are balanced (each open brace is properly closed) and not nested within each other
    Ignore any spaces in the input string.
    >>> separate_paren_groups('( ) (( )) (( )( ))')
    ['()', '(())', '(()())']
    """
    # Step 2: Remove all spaces from the input string
    paren_string = paren_string.replace(" ", "")
    
    # Step 3: Initialize an empty list to store the separated groups
    groups = []
    stack = []
    current_group = ""
    
    # Step 4: Iterate through the characters of the string
    for char in paren_string:
        if char == '(':
            if not stack:  # New group starts
                current_group = char
            else:  # Nested group
                current_group += char
            stack.append(char)
        elif char == ')':
            if stack:
                current_group += char
                stack.pop()
                if not stack:  # Group is complete
                    groups.append(current_group)
                    current_group = ""
    
    # Step 7: Return the list of separated groups
    return groups
