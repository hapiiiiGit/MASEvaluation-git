 def is_palindrome(string: str) -> bool:
    """ Test if given string is a palindrome """
    return string == string[::-1]

def find_longest_palindromic_suffix(string: str) -> str:
    """ Find the longest palindromic suffix of a given string """
    for i in range(len(string)):
        if is_palindrome(string[i:]):
            return string[i:]
    return ""

def reverse_string(string: str) -> str:
    """ Reverse a given string """
    return string[::-1]

def make_palindrome(string: str) -> str:
    """ Find the shortest palindrome that begins with a supplied string.
    Algorithm idea is simple:
    - Find the longest postfix of supplied string that is a palindrome.
    - Append to the end of the string reverse of a string prefix that comes before the palindromic suffix.
    >>> make_palindrome('')
    ''
    >>> make_palindrome('cat')
    'catac'
    >>> make_palindrome('cata')
    'catac'
    """
    longest_suffix = find_longest_palindromic_suffix(string)
    prefix = string[:-len(longest_suffix)]  # Get the part of the string before the palindromic suffix
    return string + reverse_string(prefix)
