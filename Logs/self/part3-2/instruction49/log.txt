 def ingest_data(obd_path='obd_data.csv', sensor_path='sensor_data.csv'):
    obd_df = pd.read_csv(obd_path)
    sensor_df = pd.read_csv(sensor_path)
    # Merge on timestamp or vehicle_id
    data = pd.merge(obd_df, sensor_df, on=['timestamp', 'vehicle_id'], how='inner')
    return data

# ---------------------------
# 2. Data Cleaning & Preprocessing
# ---------------------------
def clean_preprocess_data(df):
    # Drop duplicates
    df = df.drop_duplicates()
    # Fill missing values with median
    for col in df.select_dtypes(include=[np.number]).columns:
        df[col] = df[col].fillna(df[col].median())
    # Drop rows with missing labels if any
    if 'label' in df.columns:
        df = df.dropna(subset=['label'])
    # Feature scaling
    scaler = StandardScaler()
    feature_cols = [col for col in df.columns if col not in ['timestamp', 'vehicle_id', 'label']]
    df[feature_cols] = scaler.fit_transform(df[feature_cols])
    return df, scaler

# ---------------------------
# 3. Data Labeling
# ---------------------------
def label_data(df):
    # Example: If 'fault_code' exists, label as 1 (fault) or 0 (normal)
    if 'fault_code' in df.columns:
        df['label'] = df['fault_code'].apply(lambda x: 1 if pd.notnull(x) and x != 0 else 0)
    else:
        # If no fault_code, use anomaly detection (simple rule-based)
        df['label'] = 0
        for col in df.select_dtypes(include=[np.number]).columns:
            mean = df[col].mean()
            std = df[col].std()
            df.loc[(df[col] > mean + 3*std) | (df[col] < mean - 3*std), 'label'] = 1
    return df
