def create_subscription_plan(plan_name, price, features):
    plan = frappe.get_doc({
        "doctype": "SubscriptionPlan",
        "plan_name": plan_name,
        "price": price,
        "features": features
    })
    plan.insert()
    return plan

def get_available_plans():
    return frappe.get_all("SubscriptionPlan", fields=["name", "plan_name", "price", "features"])

def initiate_telr_payment(user_email, plan_id, amount, return_url):
    payload = {
        "ivp_method": "create",
        "ivp_store": TELR_STORE_ID,
        "ivp_authkey": TELR_API_KEY,
        "ivp_cart": plan_id,
        "ivp_amount": str(amount),
        "ivp_currency": "USD",
        "ivp_desc": f"Subscription for plan {plan_id}",
        "ivp_test": "1",
        "ivp_customer": {
            "name": user_email,
            "email": user_email
        },
        "return_url": return_url
    }
    response = requests.post(TELR_API_URL, json=payload)
    response.raise_for_status()
    return response.json()

@frappe.whitelist(allow_guest=True)
def telr_webhook_handler():
    data = frappe.local.form_dict
    # Validate payment status
    if data.get("status") == "paid":
        user_email = data.get("customer_email")
        plan_id = data.get("cart_id")
        provision_new_tenant_site(user_email, plan_id)
        return "OK"
    return "FAILED"
