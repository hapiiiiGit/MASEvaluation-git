 def gather_files(source_dir):
    files = []
    for root, _, filenames in os.walk(source_dir):
        for filename in filenames:
            files.append(os.path.join(root, filename))
    return files

def categorize_file(file_path):
    ext = os.path.splitext(file_path)[1].lower()
    for dir_name, exts in DIR_STRUCTURE.items():
        if ext in exts or (dir_name == 'framework/laravel' and 'laravel' in file_path.lower()):
            return dir_name
    # Default to backend/php for unknown .php files
    if ext == '.php':
        return 'backend/php'
    return None

def organize_files(files, target_root):
    for file_path in files:
        category = categorize_file(file_path)
        if category:
            target_dir = os.path.join(target_root, category)
            os.makedirs(target_dir, exist_ok=True)
            shutil.copy2(file_path, target_dir)

def verify_and_fix_code(target_root):
    # Syntax check for Python files
    python_dir = os.path.join(target_root, 'backend/python')
    if os.path.exists(python_dir):
        for filename in os.listdir(python_dir):
            if filename.endswith('.py'):
                file_path = os.path.join(python_dir, filename)

                    subprocess.check_output(['python', '-m', 'py_compile', file_path])
