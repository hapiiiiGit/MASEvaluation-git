def preprocess_data(df_postgres, df_csv):
    # Merge and preprocess data as needed
    if df_postgres.empty and df_csv.empty:
        return pd.DataFrame()
    elif df_postgres.empty:
        return df_csv
    elif df_csv.empty:
        return df_postgres
    else:
        # Assuming both have similar columns, merge on common columns
        common_cols = list(set(df_postgres.columns) & set(df_csv.columns))
        if common_cols:
            merged_df = pd.concat([df_postgres, df_csv], ignore_index=True)
        else:
            # If no common columns, join side by side
            merged_df = pd.concat([df_postgres, df_csv], axis=1)
        return merged_df
