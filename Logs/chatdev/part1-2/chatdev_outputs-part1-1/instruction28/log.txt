'''

This file contains the main application logic for the Interval Intersection Checker.
It uses Tkinter for the GUI and provides functionality to check if the length of the intersection
of two intervals is a prime number.

'''

import tkinter as tk

def is_prime(n):
    '''
    Determines if a given number n is a prime number.
    '''
    if n <= 1:
        return False
    if n <= 3:
        return True
    if n % 2 == 0 or n % 3 == 0:
        return False
    i = 5
    while i * i <= n:
        if n % i == 0 or n % (i + 2) == 0:
            return False
        i += 6
    return True

def intersection_length(interval1, interval2):
    '''
    Calculates the length of the intersection of two intervals.
    '''
    start1, end1 = interval1
    start2, end2 = interval2
    start_intersection = max(start1, start2)
    end_intersection = min(end1, end2)
    if start_intersection <= end_intersection:
        return end_intersection - start_intersection + 1
    return 0

class IntervalApp:
    '''
    This class manages the GUI application for checking interval intersections.
    '''
    def __init__(self, master):
        self.master = master
        master.title("Interval Intersection Checker")

        self.label = tk.Label(master, text="Enter two intervals:")
        self.label.pack()

        self.interval1_entry = tk.Entry(master)
        self.interval1_entry.pack()
        self.interval1_entry.insert(0, "e.g., 1,3")

        self.interval2_entry = tk.Entry(master)
        self.interval2_entry.pack()
        self.interval2_entry.insert(0, "e.g., 2,4")

        self.check_button = tk.Button(master, text="Check Intersection", command=self.check_intersection)
        self.check_button.pack()

        self.result_label = tk.Label(master, text="")
        self.result_label.pack()

    def check_intersection(self):
        interval1_str = self.interval1_entry.get()
        interval2_str = self.interval2_entry.get()

        try:
            interval1 = tuple(map(int, interval1_str.split(',')))
            interval2 = tuple(map(int, interval2_str.split(',')))

            length = intersection_length(interval1, interval2)
            if is_prime(length):
                result = "YES"
            else:
                result = "NO"
        except ValueError:
            result = "Invalid input. Please enter intervals as two integers separated by a comma."

        self.result_label.config(text=result)

if __name__ == "__main__":
    root = tk.Tk()
    app = IntervalApp(root)
    root.mainloop()

'''
This file contains the main application logic for the Interval Intersection Checker.
It uses Tkinter for the GUI and provides functionality to check if the length of the intersection
of two intervals is a prime number.

import tkinter as tk
def is_prime(n):
    '''
    Determines if a given number n is a prime number.
    '''
    if n <= 1:
        return False
    if n <= 3:
        return True
    if n % 2 == 0 or n % 3 == 0:
        return False
    i = 5
    while i * i <= n:
        if n % i == 0 or n % (i + 2) == 0:
            return False
        i += 6
    return True
def intersection_length(interval1, interval2):
    '''
    Calculates the length of the intersection of two intervals.
    '''
    start1, end1 = interval1
    start2, end2 = interval2
    start_intersection = max(start1, start2)
    end_intersection = min(end1, end2)
    if start_intersection <= end_intersection:
        return end_intersection - start_intersection + 1
    return 0
class IntervalApp:
    '''
    This class manages the GUI application for checking interval intersections.
    '''
    def __init__(self, master):
        self.master = master
        master.title("Interval Intersection Checker")
        self.label = tk.Label(master, text="Enter two intervals:")
        self.label.pack()
        self.interval1_entry = tk.Entry(master)
        self.interval1_entry.pack()
        self.interval1_entry.insert(0, "e.g., 1,3")
        self.interval2_entry = tk.Entry(master)
        self.interval2_entry.pack()
        self.interval2_entry.insert(0, "e.g., 2,4")
        self.check_button = tk.Button(master, text="Check Intersection", command=self.check_intersection)
        self.check_button.pack()
        self.result_label = tk.Label(master, text="")
        self.result_label.pack()
    def check_intersection(self):
        interval1_str = self.interval1_entry.get()
        interval2_str = self.interval2_entry.get()
        try:
            interval1 = tuple(map(int, interval1_str.split(',')))
            interval2 = tuple(map(int, interval2_str.split(',')))
            length = intersection_length(interval1, interval2)
            if is_prime(length):
                result = "YES"
            else:
                result = "NO"
        except ValueError:
            result = "Invalid input. Please enter intervals as two integers separated by a comma."
        self.result_label.config(text=result)
if __name__ == "__main__":
    root = tk.Tk()
    app = IntervalApp(root)
    root.mainloop()

'''

It uses Tkinter for the GUI and provides functionality to check if the length of the intersection

import tkinter as tk  
def is_prime(n):  
    '''  
    Determines if a given number n is a prime number.  
    '''  
    if n <= 1:  
        return False  
    if n <= 3:  
        return True  
    if n % 2 == 0 or n % 3 == 0:  
        return False  
    i = 5  
    while i * i <= n:  
        if n % i == 0 or n % (i + 2) == 0:  
            return False  
        i += 6  
    return True  
def intersection_length(interval1, interval2):  
    '''  
    Calculates the length of the intersection of two intervals.  
    '''  
    start1, end1 = interval1  
    start2, end2 = interval2  
    start_intersection = max(start1, start2)  
    end_intersection = min(end1, end2)  
    if start_intersection <= end_intersection:  
        return end_intersection - start_intersection + 1  
    return 0  
class IntervalApp:  
    '''  
    This class manages the GUI application for checking interval intersections.  
    '''  
    def __init__(self, master):  
        self.master = master  
        master.title("Interval Intersection Checker")  
        self.label = tk.Label(master, text="Enter two intervals:")  
        self.label.pack()  
        self.interval1_entry = tk.Entry(master)  
        self.interval1_entry.pack()  
        self.interval1_entry.insert(0, "e.g., 1,3")  
        self.interval2_entry = tk.Entry(master)  
        self.interval2_entry.pack()  
        self.interval2_entry.insert(0, "e.g., 2,4")  
        self.check_button = tk.Button(master, text="Check Intersection", command=self.check_intersection)  
        self.check_button.pack()  
        self.result_label = tk.Label(master, text="")  
        self.result_label.pack()  
    def check_intersection(self):  
        interval1_str = self.interval1_entry.get()  
        interval2_str = self.interval2_entry.get()  
        try:  
            interval1 = tuple(map(int, interval1_str.split(',')))  
            interval2 = tuple(map(int, interval2_str.split(',')))  
            if len(interval1) != 2 or len(interval2) != 2:  
                raise ValueError("Invalid interval format.")  
            length = intersection_length(interval1, interval2)  
            if is_prime(length):  
                result = "YES"  
            else:  
                result = "NO"  
        except ValueError:  
            result = "Invalid input. Please enter intervals as two integers separated by a comma."  
        self.result_label.config(text=result)

'''
This file contains the main application logic for the Interval Intersection Checker.
It uses Tkinter for the GUI and provides functionality to check if the length of the intersection
of two intervals is a prime number.
'''
import tkinter as tk
def is_prime(n):
    '''
    Determines if a given number n is a prime number.
    '''
    if n <= 1:
        return False
    if n <= 3:
        return True
    if n % 2 == 0 or n % 3 == 0:
        return False
    i = 5
    while i * i <= n:
        if n % i == 0 or n % (i + 2) == 0:
            return False
        i += 6
    return True
def intersection_length(interval1, interval2):
    '''
    Calculates the length of the intersection of two intervals.
    '''
    start1, end1 = interval1
    start2, end2 = interval2
    start_intersection = max(start1, start2)
    end_intersection = min(end1, end2)
    if start_intersection <= end_intersection:
        return end_intersection - start_intersection + 1
    return 0
class IntervalApp:
    '''
    This class manages the GUI application for checking interval intersections.
    '''
    def __init__(self, master):
        self.master = master
        master.title("Interval Intersection Checker")
        self.label = tk.Label(master, text="Enter two intervals:")
        self.label.pack()
        self.interval1_entry = tk.Entry(master)
        self.interval1_entry.pack()
        self.interval1_entry.insert(0, "e.g., 1,3")
        self.interval2_entry = tk.Entry(master)
        self.interval2_entry.pack()
        self.interval2_entry.insert(0, "e.g., 2,4")
        self.check_button = tk.Button(master, text="Check Intersection", command=self.check_intersection)
        self.check_button.pack()
        self.result_label = tk.Label(master, text="")
        self.result_label.pack()
    def check_intersection(self):
        interval1_str = self.interval1_entry.get()
        interval2_str = self.interval2_entry.get()
        try:
            interval1 = tuple(map(int, interval1_str.split(',')))
            interval2 = tuple(map(int, interval2_str.split(',')))
            if len(interval1) != 2 or len(interval2) != 2:
                raise ValueError("Invalid interval format.")
            length = intersection_length(interval1, interval2)
            if is_prime(length):
                result = "YES"
            else:
                result = "NO"
        except ValueError:
            result = "Invalid input. Please enter intervals as two integers separated by a comma."
        except Exception as e:
            result = f"An error occurred: {str(e)}"
        self.result_label.config(text=result)
if __name__ == "__main__":
    root = tk.Tk()
    app = IntervalApp(root)
    root.mainloop()

'''
This file contains the main application logic for the Interval Intersection Checker.
It uses Tkinter for the GUI and provides functionality to check if the length of the intersection
of two intervals is a prime number.
'''
import tkinter as tk
def is_prime(n):
    '''
    Determines if a given number n is a prime number.
    '''
    if n <= 1:
        return False
    if n <= 3:
        return True
    if n % 2 == 0 or n % 3 == 0:
        return False
    i = 5
    while i * i <= n:
        if n % i == 0 or n % (i + 2) == 0:
            return False
        i += 6
    return True
def intersection_length(interval1, interval2):
    '''
    Calculates the length of the intersection of two intervals.
    '''
    start1, end1 = interval1
    start2, end2 = interval2
    start_intersection = max(start1, start2)
    end_intersection = min(end1, end2)
    if start_intersection <= end_intersection:
        return end_intersection - start_intersection + 1
    return 0
class IntervalApp:
    '''
    This class manages the GUI application for checking interval intersections.
    '''
    def __init__(self, master):
        self.master = master
        master.title("Interval Intersection Checker")
        self.label = tk.Label(master, text="Enter two intervals:")
        self.label.pack()
        self.interval1_entry = tk.Entry(master)
        self.interval1_entry.pack()
        self.interval1_entry.insert(0, "e.g., 1,3")
        self.interval2_entry = tk.Entry(master)
        self.interval2_entry.pack()
        self.interval2_entry.insert(0, "e.g., 2,4")
        self.check_button = tk.Button(master, text="Check Intersection", command=self.check_intersection)
        self.check_button.pack()
        self.result_label = tk.Label(master, text="")
        self.result_label.pack()
    def check_intersection(self):
        interval1_str = self.interval1_entry.get()
        interval2_str = self.interval2_entry.get()
        try:
            interval1 = tuple(map(int, interval1_str.split(',')))
            interval2 = tuple(map(int, interval2_str.split(',')))
            if len(interval1) != 2 or len(interval2) != 2:
                raise ValueError("Invalid interval format.")
            length = intersection_length(interval1, interval2)
            if is_prime(length):
                result = "YES"
            else:
                result = "NO"
        except ValueError:
            result = "Invalid input. Please enter intervals as two integers separated by a comma."
        except Exception as e:
            result = f"An error occurred: {str(e)}"
        self.result_label.config(text=result)
if __name__ == "__main__":
    root = tk.Tk()
    app = IntervalApp(root)
    root.mainloop()

'''

It uses Tkinter for the GUI and provides functionality to check if the length of the intersection
of two intervals is a prime number.

import tkinter as tk
def is_prime(n):
    '''
    Determines if a given number n is a prime number.
    '''
    if n <= 1:
        return False
    if n <= 3:
        return True
    if n % 2 == 0 or n % 3 == 0:
        return False
    i = 5
    while i * i <= n:
        if n % i == 0 or n % (i + 2) == 0:
            return False
        i += 6
    return True
def intersection_length(interval1, interval2):
    '''
    Calculates the length of the intersection of two intervals.
    '''
    start1, end1 = interval1
    start2, end2 = interval2
    start_intersection = max(start1, start2)
    end_intersection = min(end1, end2)
    if start_intersection <= end_intersection:
        return end_intersection - start_intersection + 1
    return 0
class IntervalApp:
    '''
    This class manages the GUI application for checking interval intersections.
    '''
    def __init__(self, master):
        self.master = master
        master.title("Interval Intersection Checker")
        self.label = tk.Label(master, text="Enter two intervals:")
        self.label.pack()
        self.interval1_entry = tk.Entry(master)
        self.interval1_entry.pack()
        self.interval1_entry.insert(0, "e.g., 1,3")
        self.interval2_entry = tk.Entry(master)
        self.interval2_entry.pack()
        self.interval2_entry.insert(0, "e.g., 2,4")
        self.check_button = tk.Button(master, text="Check Intersection", command=self.check_intersection)
        self.check_button.pack()
        self.result_label = tk.Label(master, text="")
        self.result_label.pack()
    def check_intersection(self):
        interval1_str = self.interval1_entry.get()
        interval2_str = self.interval2_entry.get()
        try:
            interval1 = tuple(map(int, interval1_str.split(',')))
            interval2 = tuple(map(int, interval2_str.split(',')))
            if len(interval1) != 2 or len(interval2) != 2:
                raise ValueError("Invalid interval format.")
            length = intersection_length(interval1, interval2)
            if is_prime(length):
                result = "YES"
            else:
                result = "NO"
        except ValueError:
            result = "Invalid input. Please enter intervals as two integers separated by a comma."
        except Exception as e:
            result = f"An error occurred: {str(e)}"
        self.result_label.config(text=result)
if __name__ == "__main__":
    root = tk.Tk()
