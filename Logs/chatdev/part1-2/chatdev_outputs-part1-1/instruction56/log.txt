def even_odd_count(num):
    """Given an integer, return a tuple that has the number of even and odd digits respectively."""
    num_str = str(abs(num))  # Convert number to string and handle negative numbers
    even_count = sum(1 for digit in num_str if int(digit) % 2 == 0)
    odd_count = sum(1 for digit in num_str if int(digit) % 2 != 0)
    return even_count, odd_count

def run_application():
    """Initialize and run the GUI application."""
    from gui import EvenOddCounterApp
    app = EvenOddCounterApp()
    app.mainloop()

    run_application()

import tkinter as tk
from tkinter import messagebox
from main import even_odd_count

class EvenOddCounterApp(tk.Tk):
    """Class to create the GUI application using tkinter."""
    
    def __init__(self):
        super().__init__()
        self.title("Even and Odd Digit Counter")
        self.geometry("300x200")
        self.create_widgets()

    def create_widgets(self):
        """Create and place widgets in the application window."""
        self.label = tk.Label(self, text="Enter an integer:")
        self.label.pack(pady=10)

        self.entry = tk.Entry(self)
        self.entry.pack(pady=10)

        self.button = tk.Button(self, text="Calculate", command=self.calculate_and_display)
        self.button.pack(pady=10)

    def calculate_and_display(self):
        """Handle user input, call even_odd_count, and display the result."""
        try:
            num = int(self.entry.get())
            even_count, odd_count = even_odd_count(num)
            messagebox.showinfo("Result", f"Even digits: {even_count}, Odd digits: {odd_count}")
        except ValueError:
            messagebox.showerror("Error", "Please enter a valid integer.")

def even_odd_count(num):
    """Given an integer, return a tuple that has the number of even and odd digits respectively."""
    num_str = str(abs(num))  # Convert number to string and handle negative numbers
    even_count = sum(1 for digit in num_str if int(digit) % 2 == 0)
    odd_count = sum(1 for digit in num_str if int(digit) % 2 != 0)
    return even_count, odd_count

def even_odd_count(num):
    """Given an integer, return a tuple that has the number of even and odd digits respectively."""
    num_str = str(abs(num))  # Convert number to string and handle negative numbers
    even_count = sum(1 for digit in num_str if int(digit) % 2 == 0)
    odd_count = sum(1 for digit in num_str if int(digit) % 2 != 0)
    return even_count, odd_count
