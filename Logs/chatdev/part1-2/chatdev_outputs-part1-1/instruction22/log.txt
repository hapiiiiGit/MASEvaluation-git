class OddSumApp:
    def __init__(self, master):
        self.master = master

        self.label = tk.Label(master, text="Enter a list of integers (comma-separated):")
        self.label.pack()

        self.entry.pack()
        self.calculate_button = tk.Button(master, text="Calculate", command=self.calculate_and_display_sum)
        self.calculate_button.pack()

    def calculate_and_display_sum(self):
        input_text = self.entry.get()

            lst = list(map(int, input_text.split(',')))

            self.result_label.config(text=f"Sum of odd elements at even positions: {result}")

import tkinter as tk
from calculator import OddSumCalculator
class OddSumApp:
    def __init__(self, master):
        self.master = master
        master.title("Odd Sum Calculator")
        self.label = tk.Label(master, text="Enter a list of integers (comma-separated):")
        self.label.pack()
        self.entry = tk.Entry(master)
        self.entry.pack()
        self.calculate_button = tk.Button(master, text="Calculate", command=self.calculate_and_display_sum)
        self.calculate_button.pack()
        self.result_label = tk.Label(master, text="")

    def calculate_and_display_sum(self):
        input_text = self.entry.get()

            self.result_label.config(text=f"Sum of odd elements at even positions: {result}")
        except ValueError:
