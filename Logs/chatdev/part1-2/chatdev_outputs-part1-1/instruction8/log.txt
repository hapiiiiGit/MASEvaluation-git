'''

This file contains the main application logic for the "Even and Odd Palindrome Counter".
It includes the GUI implementation using tkinter and the logic to calculate even and odd palindromes.

'''

import tkinter as tk
from tkinter import messagebox


def is_palindrome(num):
    '''
    Check if a given number is a palindrome.
    
    Args:
        num (int): The number to check.
    
    Returns:
        bool: True if the number is a palindrome, False otherwise.
    '''
    str_num = str(num)
    return str_num == str_num[::-1]


def even_odd_palindrome(n):
    '''
    Calculate the number of even and odd integer palindromes within the range [1, n].
    
    Args:
        n (int): The upper limit of the range.
    
    Returns:
        tuple: A tuple containing the count of even and odd palindromes respectively.
    '''
    even_count = 0
    odd_count = 0
    
    for num in range(1, n + 1):
        if is_palindrome(num):
            if num % 2 == 0:
                even_count += 1
            else:
                odd_count += 1
    
    return even_count, odd_count


def calculate_palindromes():
    '''
    Event handler for the "Calculate" button. Reads input, calculates results, and displays them.
    '''
    try:
        # Get the input value from the entry field
        n = int(entry.get())
        
        if n < 1 or n > 1000:

        # Display the result
        result_label.config(text=f"Even Palindromes: {even_count}, Odd Palindromes: {odd_count}")
    except ValueError as e:
        # Show an error message if the input is invalid
        messagebox.showerror("Invalid Input", str(e))


# Create the main application window
root = tk.Tk()
root.title("Even and Odd Palindrome Counter")

calculate_button.pack(pady=10)

result_label.pack(pady=5)

# Start the main event loop
root.mainloop()

This file contains the main application logic for the "Even and Odd Palindrome Counter".
It includes the GUI implementation using tkinter and the logic to calculate even and odd palindromes.
'''
import tkinter as tk
from tkinter import messagebox

def is_palindrome(num):
    '''
    Check if a given number is a palindrome.
    Args:
        num (int): The number to check.
    Returns:
        bool: True if the number is a palindrome, False otherwise.
    '''
    str_num = str(num)
    return str_num == str_num[::-1]

def even_odd_palindrome(n):
    '''
    Calculate the number of even and odd integer palindromes within the range [1, n].
    Args:
        n (int): The upper limit of the range.
    Returns:
        tuple: A tuple containing the count of even and odd palindromes respectively.
    '''
    even_count = 0
    odd_count = 0
    for num in range(1, n + 1):
        if is_palindrome(num):
            if num % 2 == 0:
                even_count += 1
            else:
                odd_count += 1
    return even_count, odd_count

def calculate_palindromes():
    '''
    Event handler for the "Calculate" button. Reads input, calculates results, and displays them.
    '''
    try:
        # Get the input value from the entry field
        n = int(entry.get())
        if n < 1 or n > 1000:
            raise ValueError("Input must be between 1 and 1000.")
        # Calculate the even and odd palindromes
        even_count, odd_count = even_odd_palindrome(n)
        # Display the result
        result_label.config(text=f"Even Palindromes: {even_count}, Odd Palindromes: {odd_count}")
    except ValueError as e:
        # Show an error message if the input is invalid
        messagebox.showerror("Invalid Input", str(e))

# Create the main application window
root = tk.Tk()
root.title("Even and Odd Palindrome Counter")

# Create and place widgets
tk.Label(root, text="Enter a positive integer (1-1000):").pack(pady=5)
entry = tk.Entry(root)
entry.pack(pady=5)
calculate_button = tk.Button(root, text="Calculate", command=calculate_palindromes)
calculate_button.pack(pady=10)
result_label = tk.Label(root, text="")
result_label.pack(pady=5)

# Start the main event loop
root.mainloop()

'''
This file contains the main application logic for the "Even and Odd Palindrome Counter".
It includes the GUI implementation using tkinter and the logic to calculate even and odd palindromes.

import tkinter as tk
from tkinter import messagebox
def is_palindrome(num):
    '''
    Check if a given number is a palindrome.
    Args:
        num (int): The number to check.
    Returns:
        bool: True if the number is a palindrome, False otherwise.
    '''
    str_num = str(num)
    return str_num == str_num[::-1]
def even_odd_palindrome(n):
    '''
    Calculate the number of even and odd integer palindromes within the range [1, n].
    Args:
        n (int): The upper limit of the range.
    Returns:
        tuple: A tuple containing the count of even and odd palindromes respectively.
    '''
    even_count = 0
    odd_count = 0
    for num in range(1, n + 1):
        if is_palindrome(num):
            if num % 2 == 0:
                even_count += 1
            else:
                odd_count += 1
    return even_count, odd_count
def calculate_palindromes():

    except ValueError as e:
        # Show an error message if the input is invalid

# Create the main application window

import tkinter as tk
from tkinter import messagebox
def is_palindrome(num):
    '''
    Check if a given number is a palindrome.
    Args:
        num (int): The number to check.
    Returns:
        bool: True if the number is a palindrome, False otherwise.
    '''
    str_num = str(num)
    return str_num == str_num[::-1]
def even_odd_palindrome(n):
    '''
    Calculate the number of even and odd integer palindromes within the range [1, n].
    Args:
        n (int): The upper limit of the range.
    Returns:
        tuple: A tuple containing the count of even and odd palindromes respectively.
    '''
    even_count = 0
    odd_count = 0
    for num in range(1, n + 1):
        if is_palindrome(num):
            if num % 2 == 0:
                even_count += 1
            else:
                odd_count += 1
    return even_count, odd_count
