def order_by_points(nums):
    """
    Sorts the given list of integers in ascending order according to the sum of their digits.
    If there are several items with similar sum of their digits, order them based on their index in the original list.

    Args:
        nums (list): A list of integers.

        list: A sorted list of integers.

'''
This module provides the implementation of the order_by_points function and a graphical user interface for the application using tkinter.

def order_by_points(nums):
    """
    Sorts the given list of integers in ascending order according to the sum of their digits.
    If there are several items with similar sum of their digits, order them based on their index in the original list.
    """
    def digit_sum(n):
        return sum(int(digit) for digit in str(abs(n)))
    return sorted(nums, key=lambda x: (digit_sum(x), nums.index(x)))
import tkinter as tk
from tkinter import messagebox
class OrderByPointsGUI:
    """
    A simple GUI application to input a list of integers and display them sorted by the sum of their digits.
    """
    def __init__(self, root):
        self.root = root
        self.root.title("Order By Points")
        self.label = tk.Label(root, text="Enter a list of integers (comma-separated):")
        self.label.pack()
        self.entry = tk.Entry(root, width=50)
        self.entry.pack()
        self.button = tk.Button(root, text="Sort", command=self.sort_numbers)
        self.button.pack()
        self.result_label = tk.Label(root, text="")
        self.result_label.pack()
    def sort_numbers(self):
        """
        Retrieves the list of integers from the entry widget, sorts them using order_by_points, and displays the result.
        """
        try:
            nums = list(map(int, self.entry.get().split(',')))
            sorted_nums = order_by_points(nums)
            self.result_label.config(text=f"Sorted list: {sorted_nums}")
        except ValueError:
            messagebox.showerror("Invalid input", "Please enter a valid list of integers.")
if __name__ == "__main__":
    root = tk.Tk()
    app = OrderByPointsGUI(root)
    root.mainloop()
