'''

Main application file for the score comparison GUI.

'''

import tkinter as tk
from logic import compare

class MainApp:
    def __init__(self, root):
        self.root = root
        self.root.title("Score Comparison")

        # Labels and Entry fields for input
        self.label_game = tk.Label(root, text="Enter Game Scores (comma-separated):")
        self.label_game.pack()
        self.entry_game = tk.Entry(root)
        self.entry_game.pack()

        self.label_guess = tk.Label(root, text="Enter Guesses (comma-separated):")
        self.label_guess.pack()
        self.entry_guess = tk.Entry(root)
        self.entry_guess.pack()

        # Button to trigger comparison
        self.compare_button = tk.Button(root, text="Compare", command=self.compare_scores)
        self.compare_button.pack()

        # Label to display result
        self.result_label = tk.Label(root, text="")
        self.result_label.pack()

    def compare_scores(self):
        # Retrieve and process input
        game_input = self.entry_game.get()
        guess_input = self.entry_guess.get()

        try:
            game_scores = list(map(int, game_input.split(',')))
            guess_scores = list(map(int, guess_input.split(',')))

            # Call the compare function
            result = compare(game_scores, guess_scores)

            # Display result
            self.result_label.config(text=f"Result: {result}")

        except ValueError:
            self.result_label.config(text="Invalid input. Please enter numbers separated by commas.")

if __name__ == "__main__":
    root = tk.Tk()
    app = MainApp(root)
    root.mainloop()
```

```python
'''

Logic module for comparing game scores and guesses.

def compare(game, guess):
    """
    Compare the game scores with the guesses and return the differences.

    :param game: List of actual game scores.
    :param guess: List of guessed scores.
    :return: List of differences between actual scores and guesses.
    """
    return [abs(g - s) for g, s in zip(game, guess)]
```

def compare(game, guess):
    """
    Compare the game scores with the guesses and return the differences.
    :param game: List of actual game scores.
    :param guess: List of guessed scores.
    :return: List of differences between actual scores and guesses.
    """
    return [abs(g - s) for g, s in zip(game, guess)]

'''
Main application file for the score comparison GUI.
'''
import tkinter as tk
from logic import compare

class MainApp:
    def __init__(self, root):
        self.root = root
        self.root.title("Score Comparison")
        # Labels and Entry fields for input
        self.label_game = tk.Label(root, text="Enter Game Scores (comma-separated):")
        self.label_game.pack()
        self.entry_game = tk.Entry(root)
        self.entry_game.pack()
        self.label_guess = tk.Label(root, text="Enter Guesses (comma-separated):")
        self.label_guess.pack()
        self.entry_guess = tk.Entry(root)
        self.entry_guess.pack()
        # Button to trigger comparison
        self.compare_button = tk.Button(root, text="Compare", command=self.compare_scores)
        self.compare_button.pack()
        # Label to display result
        self.result_label = tk.Label(root, text="")
        self.result_label.pack()

    def compare_scores(self):
        # Retrieve and process input
        game_input = self.entry_game.get()
        guess_input = self.entry_guess.get()
        try:
            game_scores = list(map(int, game_input.split(',')))
            guess_scores = list(map(int, guess_input.split(',')))
            # Call the compare function
            result = compare(game_scores, guess_scores)
            # Display result
            self.result_label.config(text=f"Result: {result}")
        except ValueError:
            self.result_label.config(text="Invalid input. Please enter numbers separated by commas.")

if __name__ == "__main__":
    root = tk.Tk()
    app = MainApp(root)
    root.mainloop()

def compare(game, guess):
    """
    Compare the game scores with the guesses and return the differences.
    :param game: List of actual game scores.
    :param guess: List of guessed scores.
    :return: List of differences between actual scores and guesses.
    """
    return [abs(g - s) for g, s in zip(game, guess)]

'''

def compare(game, guess):
    """
    Compare the game scores with the guesses and return the differences.
    :param game: List of actual game scores.
    :param guess: List of guessed scores.
    :return: List of differences between actual scores and guesses.
    """
    return [abs(g - s) for g, s in zip(game, guess)]

import tkinter as tk
from logic import compare
class MainApp:
    def __init__(self, root):
        self.root = root
        self.root.title("Score Comparison")
        # Labels and Entry fields for input
        self.label_game = tk.Label(root, text="Enter Game Scores (comma-separated):")
        self.label_game.pack()
        self.entry_game = tk.Entry(root)
        self.entry_game.pack()
        self.label_guess = tk.Label(root, text="Enter Guesses (comma-separated):")
        self.label_guess.pack()
        self.entry_guess = tk.Entry(root)
        self.entry_guess.pack()
        # Button to trigger comparison
        self.compare_button = tk.Button(root, text="Compare", command=self.compare_scores)
        self.compare_button.pack()
        # Label to display result
        self.result_label = tk.Label(root, text="")
        self.result_label.pack()
    def compare_scores(self):
        # Retrieve and process input
        game_input = self.entry_game.get()
        guess_input = self.entry_guess.get()
        try:
            game_scores = list(map(int, game_input.split(',')))
            guess_scores = list(map(int, guess_input.split(',')))
            # Call the compare function
            result = compare(game_scores, guess_scores)
            # Display result
            self.result_label.config(text=f"Result: {result}")
        except ValueError:
            self.result_label.config(text="Invalid input. Please enter numbers separated by commas.")
if __name__ == "__main__":
    root = tk.Tk()
    app = MainApp(root)
    root.mainloop()

'''
Logic module for comparing game scores and guesses.

def compare(game, guess):
    """
    Compare the game scores with the guesses and return the differences.
    :param game: List of actual game scores.
    :param guess: List of guessed scores.
    :return: List of differences between actual scores and guesses.
    """
    return [abs(g - s) for g, s in zip(game, guess)]

import tkinter as tk
from logic import compare
class MainApp:
    def __init__(self, root):

        self.entry_game.pack()
        self.label_guess = tk.Label(root, text="Enter Guesses (comma-separated):")

        # Button to trigger comparison

def compare(game, guess):
    """
    Compare the game scores with the guesses and return the differences.
    :param game: List of actual game scores.
    :param guess: List of guessed scores.
    :return: List of differences between actual scores and guesses.
    """
    return [abs(g - s) for g, s in zip(game, guess)]

# Score Comparison Application

Welcome to the Score Comparison Application! This software allows users to compare actual game scores with their guesses and determine how accurate their predictions were. The application provides a simple graphical user interface (GUI) for inputting scores and viewing results.

## Main Functions

The Score Comparison Application offers the following main functions:

- **Input Game Scores and Guesses**: Users can enter the actual game scores and their guesses through a user-friendly interface.
- **Compare Scores**: The application calculates the absolute differences between the actual scores and the guesses, indicating how far off each guess was.
- **Display Results**: The results are displayed on the interface, showing users the accuracy of their guesses.

## Installation

### Prerequisites

- **Python**: Ensure that Python is installed on your system. You can download it from [python.org](https://www.python.org/downloads/).

    ```

4. **Run the Application**: Execute the following command to start the application:

    ```bash
    python main.py
    ```

## How to Use

1. **Launch the Application**: After running `main.py`, a window titled "Score Comparison" will appear.

3. **Enter Guesses**: In the "Enter Guesses" field, input your guessed scores, separated by commas. For example: `1,2,3,4,2,-2`.

4. **Compare Scores**: Click the "Compare" button to calculate the differences between the actual scores and your guesses.

5. **View Results**: The results will be displayed below the button, showing the absolute differences for each match. A value of `0` indicates a correct guess, while any other number indicates how far off the guess was.

6. **Handle Invalid Input**: If you enter non-numeric values or mismatched lengths of scores and guesses, an error message will prompt you to correct your input.
