'''
Module to compute the histogram of letters in a given string.
'''

def histogram(test):
    """
    Given a string representing space-separated lowercase letters, return a dictionary
    of the letter with the most repetition and containing the corresponding count.
    If several letters have the same occurrence, return all of them.
    """
    if not test:
        return {}

    # Split the input string into individual letters
    letters = test.split()

    # Count occurrences of each letter
    count_dict = {}
    for letter in letters:
        if letter in count_dict:
            count_dict[letter] += 1
        else:
            count_dict[letter] = 1

    # Find the maximum occurrence count
    max_count = max(count_dict.values())

    # Collect all letters with the maximum occurrence count
    result = {letter: count for letter, count in count_dict.items() if count == max_count}

    return result

'''
Module to compute the histogram of letters in a given string.
'''
def histogram(test):
    """
    Given a string representing space-separated lowercase letters, return a dictionary
    of the letter with the most repetition and containing the corresponding count.
    If several letters have the same occurrence, return all of them.
    """
    if not test:
        return {}
    # Split the input string into individual letters
    letters = test.split()
    # Count occurrences of each letter
    count_dict = {}
    for letter in letters:
        if letter in count_dict:
            count_dict[letter] += 1
        else:
            count_dict[letter] = 1
    # Find the maximum occurrence count
    max_count = max(count_dict.values())
    # Collect all letters with the maximum occurrence count
    result = {letter: count for letter, count in count_dict.items() if count == max_count}
    return result

'''
Module to compute the histogram of letters in a given string.
'''
def histogram(test):
    """
    Given a string representing space-separated lowercase letters, return a dictionary
    of the letter with the most repetition and containing the corresponding count.
    If several letters have the same occurrence, return all of them.
    """
    if not test:
        return {}
    # Split the input string into individual letters
    letters = test.split()
    # Count occurrences of each letter
    count_dict = {}
    for letter in letters:
        if letter in count_dict:
            count_dict[letter] += 1
        else:
            count_dict[letter] = 1
    # Find the maximum occurrence count
    max_count = max(count_dict.values())
    # Collect all letters with the maximum occurrence count
    result = {letter: count for letter, count in count_dict.items() if count == max_count}
    return result

'''
Module to compute the histogram of letters in a given string.
'''
def histogram(test):
    """
    Given a string representing space-separated lowercase letters, return a dictionary
    of the letter with the most repetition and containing the corresponding count.
    If several letters have the same occurrence, return all of them.
    """
    if not test:
        return {}
    # Split the input string into individual letters
    letters = test.split()
    # Count occurrences of each letter
    count_dict = {}
    for letter in letters:
        if letter in count_dict:
            count_dict[letter] += 1
        else:
            count_dict[letter] = 1
    # Find the maximum occurrence count
    max_count = max(count_dict.values())
    # Collect all letters with the maximum occurrence count
    result = {letter: count for letter, count in count_dict.items() if count == max_count}
    return result

'''
Main application file for the histogram GUI application.
'''
import tkinter as tk
from histogram import histogram
class MainApp:
    def __init__(self, root):
        self.root = root
        self.root.title("Histogram Application")
        self.create_widgets()
    def create_widgets(self):
        # Input label and entry
        self.label = tk.Label(self.root, text="Enter space-separated letters:")
        self.label.pack()
        self.entry = tk.Entry(self.root, width=50)
        self.entry.pack()
        # Compute button
        self.button = tk.Button(self.root, text="Compute Histogram", command=self.compute_histogram)
        self.button.pack()
        # Output label
        self.output_label = tk.Label(self.root, text="Histogram Output:")
        self.output_label.pack()
        self.output = tk.Text(self.root, height=10, width=50)
        self.output.pack()
    def compute_histogram(self):
        input_text = self.entry.get()
        result = histogram(input_text)
        self.output.delete(1.0, tk.END)
        self.output.insert(tk.END, str(result))
if __name__ == "__main__":
    root = tk.Tk()
    app = MainApp(root)
    root.mainloop()

'''
Module to compute the histogram of letters in a given string.
'''
def histogram(test):
    """
    Given a string representing space-separated lowercase letters, return a dictionary
    of the letter with the most repetition and containing the corresponding count.
    If several letters have the same occurrence, return all of them.
    """
    if not test:
        return {}
    # Split the input string into individual letters
    letters = test.split()
    # Count occurrences of each letter
    count_dict = {}
    for letter in letters:
        if letter in count_dict:
            count_dict[letter] += 1
        else:
            count_dict[letter] = 1
    # Find the maximum occurrence count
    max_count = max(count_dict.values())
    # Collect all letters with the maximum occurrence count
    result = {letter: count for letter, count in count_dict.items() if count == max_count}
    return result
