import tkinter as tk
from tkinter import messagebox

def factorial(num):
    """Calculate the factorial of a number."""
    if num == 0 or num == 1:
        return 1
    else:
        result = 1
        for i in range(2, num + 1):
            result *= i
        return result

def f(n):
    """Implement the function f that takes n as a parameter,
    and returns a list of size n, such that the value of the element at index i is the factorial of i if i is even
    or the sum of numbers from 1 to i otherwise.
    i starts from 1.
    """
    result = []
    for i in range(1, n + 1):
        if i % 2 == 0:
            result.append(factorial(i))
        else:
            result.append(sum(range(1, i + 1)))
    return result

class App:
    def __init__(self, root):
        self.root = root
        self.root.title("Factorial and Sum List Generator")

        self.label = tk.Label(root, text="Enter a number:")
        self.label.pack()

        self.entry = tk.Entry(root)
        self.entry.pack()

        self.button = tk.Button(root, text="Generate List", command=self.generate_list)
        self.button.pack()

        self.result_label = tk.Label(root, text="")
        self.result_label.pack()

    def generate_list(self):
        try:
            n = int(self.entry.get())
            if n < 1:
                raise ValueError("Number must be greater than 0.")
            result = f(n)
            self.result_label.config(text=f"Result: {result}")
        except ValueError as e:
            messagebox.showerror("Invalid Input", str(e))

if __name__ == "__main__":
    root = tk.Tk()
    app = App(root)
    root.mainloop()

import tkinter as tk
from tkinter import messagebox

def factorial(num):
    """Calculate the factorial of a number."""
    if num == 0 or num == 1:
        return 1
    else:
        result = 1
        for i in range(2, num + 1):
            result *= i
        return result

def f(n):
    """Implement the function f that takes n as a parameter,
    and returns a list of size n, such that the value of the element at index i is the factorial of i if i is even
    or the sum of numbers from 1 to i otherwise.
    i starts from 1.
    """
    result = []
    for i in range(1, n + 1):
        if i % 2 == 0:
            result.append(factorial(i))
        else:
            result.append(sum(range(1, i + 1)))
    return result

class App:
    """GUI Application for generating a list based on the function f(n)."""
    def __init__(self, root):
        """Initialize the GUI components."""
        self.root = root
        self.root.title("Factorial and Sum List Generator")
        self.label = tk.Label(root, text="Enter a number:")
        self.label.pack()
        self.entry = tk.Entry(root)
        self.entry.pack()
        self.button = tk.Button(root, text="Generate List", command=self.generate_list)
        self.button.pack()
        self.result_label = tk.Label(root, text="")
        self.result_label.pack()

    def generate_list(self):
        """Generate the list based on user input and display the result."""
        try:
            n = int(self.entry.get())
            if n < 1:
                raise ValueError("Number must be greater than 0.")
            result = f(n)
            self.result_label.config(text=f"Result: {result}")
        except ValueError as e:
            messagebox.showerror("Invalid Input", str(e))

if __name__ == "__main__":
    root = tk.Tk()
    app = App(root)
    root.mainloop()

import tkinter as tk
from tkinter import messagebox
def factorial(num):
    """Calculate the factorial of a number."""
    if num == 0 or num == 1:
        return 1
    else:
        result = 1
        for i in range(2, num + 1):
            result *= i
        return result
def f(n):
    """Implement the function f that takes n as a parameter,
    and returns a list of size n, such that the value of the element at index i is the factorial of i if i is even
    or the sum of numbers from 1 to i otherwise.
    i starts from 1.
    """
    result = []
    for i in range(1, n + 1):
        if i % 2 == 0:
            result.append(factorial(i))
        else:
            result.append(sum(range(1, i + 1)))
    return result
class App:
    """GUI Application for generating a list based on the function f(n)."""
    def __init__(self, root):
        """Initialize the GUI components."""
        self.root = root

        self.label = tk.Label(root, text="Enter a number:")
        self.label.pack()

        self.entry.pack()

        self.button.pack()
        self.result_label = tk.Label(root, text="")
        self.result_label.pack()

        """Generate the list based on user input and display the result."""
        try:
            n = int(self.entry.get())
            if n < 1:
                raise ValueError("Number must be greater than 0.")
            result = f(n)
            self.result_label.config(text=f"Result: {result}")
        except ValueError as e:
