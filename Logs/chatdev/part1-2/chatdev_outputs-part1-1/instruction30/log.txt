        self.grid = grid
        self.k = k

            if self.min_path is None or path < self.min_path:

class GridPathFinder:
    def __init__(self, grid, k):
        self.grid = grid
        self.k = k
        self.n = len(grid)
        self.min_path = None
    def find_min_path(self):
        self.min_path = None
        for i in range(self.n):
            for j in range(self.n):
                self._dfs([self.grid[i][j]], i, j)
        return self.min_path
    def _dfs(self, path, x, y):
        if len(path) == self.k:
            if self.min_path is None or path < self.min_path:
                self.min_path = path[:]
            return
        for dx, dy in [(-1, 0), (1, 0), (0, -1), (0, 1)]:
            nx, ny = x + dx, y + dy
            if 0 <= nx < self.n and 0 <= ny < self.n:
                path.append(self.grid[nx][ny])
                self._dfs(path, nx, ny)
                path.pop()
        if len(path) < self.k:
            path.append(self.grid[x][y])
            self._dfs(path, x, y)
            path.pop()

class GridPathFinder:
    def __init__(self, grid, k):
        self.grid = grid
        self.k = k
        self.n = len(grid)
        self.min_path = None
    def find_min_path(self):
        self.min_path = None
        for i in range(self.n):
            for j in range(self.n):
                self._dfs([self.grid[i][j]], i, j)
        return self.min_path
    def _dfs(self, path, x, y):
        if len(path) == self.k:
            if self.min_path is None or path < self.min_path:
                self.min_path = path[:]
            return
        for dx, dy in [(-1, 0), (1, 0), (0, -1), (0, 1)]:
            nx, ny = x + dx, y + dy
            if 0 <= nx < self.n and 0 <= ny < self.n:
                path.append(self.grid[nx][ny])
                self._dfs(path, nx, ny)
                path.pop()
        if len(path) < self.k:
            path.append(self.grid[x][y])
            self._dfs(path, x, y)
            path.pop()

class GridPathFinder:
    def __init__(self, grid, k):
        self.grid = grid
        self.k = k
        self.n = len(grid)
        self.min_path = None
    def find_min_path(self):
        self.min_path = None
        for i in range(self.n):
            for j in range(self.n):
                self._dfs([self.grid[i][j]], i, j)
        return self.min_path
    def _dfs(self, path, x, y):
        if len(path) == self.k:

                path.pop()
        # Ensure the path is extended to exactly k steps
        if len(path) < self.k:

class GridPathFinder:
    def __init__(self, grid, k):
        self.grid = grid
        self.k = k
        self.n = len(grid)
        self.min_path = None
    def find_min_path(self):
        self.min_path = None
        for i in range(self.n):
            for j in range(self.n):
                self._dfs([self.grid[i][j]], i, j)
        return self.min_path
    def _dfs(self, path, x, y):
        if len(path) == self.k:
            if self.min_path is None or path < self.min_path:
                self.min_path = path[:]
            return
        for dx, dy in [(-1, 0), (1, 0), (0, -1), (0, 1)]:
            nx, ny = x + dx, y + dy
            if 0 <= nx < self.n and 0 <= ny < self.n:
                path.append(self.grid[nx][ny])
                self._dfs(path, nx, ny)
                path.pop()
        # Ensure the path is extended to exactly k steps
        if len(path) < self.k:
            path.append(self.grid[x][y])
            self._dfs(path, x, y)
            path.pop()

'''
Grid Path Finder logic to find the minimum path of length k.
'''
class GridPathFinder:
    def __init__(self, grid, k):
        self.grid = grid
        self.k = k
        self.n = len(grid)
        self.min_path = None
    def find_min_path(self):
        self.min_path = None
        for i in range(self.n):
            for j in range(self.n):
                self._dfs([self.grid[i][j]], i, j)
        return self.min_path
    def _dfs(self, path, x, y):
        if len(path) == self.k:
            if self.min_path is None or path < self.min_path:
                self.min_path = path[:]
            return
        for dx, dy in [(-1, 0), (1, 0), (0, -1), (0, 1)]:
            nx, ny = x + dx, y + dy
            if 0 <= nx < self.n and 0 <= ny < self.n:
                path.append(self.grid[nx][ny])
                self._dfs(path, nx, ny)
                path.pop()
        # Ensure the path is extended to exactly k steps
        while len(path) < self.k:
            path.append(self.grid[x][y])
        if len(path) == self.k:
            if self.min_path is None or path < self.min_path:
                self.min_path = path[:]
        while len(path) > self.k:
            path.pop()

'''
Grid Path Finder logic to find the minimum path of length k.
'''
class GridPathFinder:
    def __init__(self, grid, k):
        self.grid = grid
        self.k = k
        self.n = len(grid)
        self.min_path = None
    def find_min_path(self):
        self.min_path = None
        for i in range(self.n):
            for j in range(self.n):
                self._dfs([self.grid[i][j]], i, j)
        return self.min_path
    def _dfs(self, path, x, y):
        if len(path) == self.k:
            if self.min_path is None or path < self.min_path:
                self.min_path = path[:]
            return
        for dx, dy in [(-1, 0), (1, 0), (0, -1), (0, 1)]:
            nx, ny = x + dx, y + dy
            if 0 <= nx < self.n and 0 <= ny < self.n:
                path.append(self.grid[nx][ny])
                self._dfs(path, nx, ny)
                path.pop()
        # Ensure the path is extended to exactly k steps
        while len(path) < self.k:
            path.append(self.grid[x][y])
        if len(path) == self.k:
            if self.min_path is None or path < self.min_path:
                self.min_path = path[:]
        while len(path) > self.k:
            path.pop()

## Introduction

Grid Path Finder is a GUI application designed to help users find the minimum path of a specified length in a given grid. The grid is composed of unique integers, and the path is determined based on lexicographical order. This application is useful for educational purposes, algorithm exploration, and grid-based pathfinding tasks.

## Main Functions

- **Grid Input**: Users can input a grid using comma-separated values for each row, separated by semicolons.
- **Path Length Input**: Users specify the desired path length (k) for the pathfinding operation.
