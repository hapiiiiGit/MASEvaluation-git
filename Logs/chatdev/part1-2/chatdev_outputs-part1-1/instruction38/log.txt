'''

This module contains the CompareApp class for the GUI and the compare_one function to compare two values.

'''

import tkinter as tk
from tkinter import messagebox

def compare_one(a, b):
    '''
    Compares two values (integers, floats, or strings representing real numbers)
    and returns the larger one. Returns None if they are equal.
    '''
    def convert_to_float(value):
        if isinstance(value, str):
            value = value.replace(',', '.')
        return float(value)

    try:
        a_float = convert_to_float(a)
        b_float = convert_to_float(b)
    except ValueError:
        return None

    if a_float > b_float:
        return a
    elif b_float > a_float:
        return b
    else:
        return None

class CompareApp:
    def __init__(self, master):
        self.master = master
        master.title("Compare Values")

        self.label_a = tk.Label(master, text="Value A:")
        self.label_a.pack()

        self.entry_a = tk.Entry(master)
        self.entry_a.pack()

        self.label_b = tk.Label(master, text="Value B:")
        self.label_b.pack()

        self.entry_b = tk.Entry(master)
        self.entry_b.pack()

        self.compare_button = tk.Button(master, text="Compare", command=self.compare_values)
        self.compare_button.pack()

        self.result_label = tk.Label(master, text="", fg="blue")
        self.result_label.pack()

    def compare_values(self):
        a = self.entry_a.get()
        b = self.entry_b.get()
        result = compare_one(a, b)
        if result is None:
            message = "The values are equal."
        else:
            message = f"The larger value is: {result}"
        self.result_label.config(text=message)
        messagebox.showinfo("Result", message)

if __name__ == "__main__":
    root = tk.Tk()
    app = CompareApp(root)
    root.mainloop()

'''
This module contains the CompareApp class for the GUI and the compare_one function to compare two values.
'''
import tkinter as tk
from tkinter import messagebox
def compare_one(a, b):
    '''
    Compares two values (integers, floats, or strings representing real numbers)
    and returns the larger one in its original type. Returns None if they are equal.
    '''
    def convert_to_float(value):
        if isinstance(value, str):
            try:
                value = value.replace(',', '.')
                return float(value)
            except ValueError:
                raise ValueError(f"Invalid input: {value} cannot be converted to a float.")
        return float(value)
    a_float = convert_to_float(a)
    b_float = convert_to_float(b)
    if a_float > b_float:
        return a
    elif b_float > a_float:
        return b
    else:
        return None
class CompareApp:
    '''
    A GUI application for comparing two values using the compare_one function.
    '''
    def __init__(self, master):
        self.master = master
        master.title("Compare Values")
        self.label_a = tk.Label(master, text="Value A:")
        self.label_a.pack()
        self.entry_a = tk.Entry(master)
        self.entry_a.pack()
        self.label_b = tk.Label(master, text="Value B:")
        self.label_b.pack()
        self.entry_b = tk.Entry(master)
        self.entry_b.pack()
        self.compare_button = tk.Button(master, text="Compare", command=self.compare_values)
        self.compare_button.pack()
        self.result_label = tk.Label(master, text="", fg="blue")
        self.result_label.pack()
    def compare_values(self):
        '''
        Retrieves input values from the GUI, compares them, and displays the result.
        '''
        a = self.entry_a.get()
        b = self.entry_b.get()
        try:
            result = compare_one(a, b)
            if result is None:
                message = "The values are equal."
            else:
                message = f"The larger value is: {result}"
        except ValueError as e:
            message = str(e)
        self.result_label.config(text=message)
        messagebox.showinfo("Result", message)
if __name__ == "__main__":
    root = tk.Tk()
    app = CompareApp(root)
    root.mainloop()

'''
This module contains the CompareApp class for the GUI and the compare_one function to compare two values.
'''
import tkinter as tk
from tkinter import messagebox
def compare_one(a, b):
    '''
    Compares two values (integers, floats, or strings representing real numbers)
    and returns the larger one in its original type. Returns None if they are equal.
    '''
    def convert_to_float(value):
        if isinstance(value, str):
            try:
                value = value.replace(',', '.')
                return float(value)
            except ValueError:
                raise ValueError(f"Invalid input: {value} cannot be converted to a float.")
        return float(value)
    a_float = convert_to_float(a)
    b_float = convert_to_float(b)
    if a_float > b_float:
        return a
    elif b_float > a_float:
        return b
    else:
        return None
class CompareApp:
    '''
    A GUI application for comparing two values using the compare_one function.
    '''
    def __init__(self, master):
        self.master = master
        master.title("Compare Values")
        self.label_a = tk.Label(master, text="Value A:")
        self.label_a.pack()
        self.entry_a = tk.Entry(master)
        self.entry_a.pack()
        self.label_b = tk.Label(master, text="Value B:")
        self.label_b.pack()
        self.entry_b = tk.Entry(master)
        self.entry_b.pack()
        self.compare_button = tk.Button(master, text="Compare", command=self.compare_values)
        self.compare_button.pack()
        self.result_label = tk.Label(master, text="", fg="blue")
        self.result_label.pack()
    def compare_values(self):
        '''
        Retrieves input values from the GUI, compares them, and displays the result.
        '''
        a = self.entry_a.get()
        b = self.entry_b.get()
        try:
            result = compare_one(a, b)
            if result is None:
                message = "The values are equal."
            else:
                message = f"The larger value is: {result}"
        except ValueError as e:
            message = str(e)
        self.result_label.config(text=message)
        messagebox.showinfo("Result", message)
if __name__ == "__main__":
    root = tk.Tk()
    app = CompareApp(root)
    root.mainloop()
