'''

This module contains the main application code for finding the word with the maximum number of unique characters from a list of words provided by the user.

'''

import tkinter as tk
from tkinter import messagebox

def find_max(words):
    '''
    Write a function that accepts a list of strings.
    The list contains different words. Return the word with maximum number
    of unique characters. If multiple strings have maximum number of unique
    characters, return the one which comes first in lexicographical order.

    :param words: List of strings
    :return: String with maximum unique characters
    '''
    max_unique_count = 0
    result_word = ""
    
    for word in words:
        unique_chars = set(word)
        unique_count = len(unique_chars)
        
        if unique_count > max_unique_count or (unique_count == max_unique_count and word < result_word):
            max_unique_count = unique_count
            result_word = word
    
    return result_word

class App:
    def __init__(self, master):
        '''
        Initialize the application with GUI components.
        
        :param master: The root window
        '''
        self.master = master
        master.title("Find Max Unique Characters")
        
        self.label = tk.Label(master, text="Enter words separated by commas:")
        self.label.pack()
        
        self.entry = tk.Entry(master, width=50)
        self.entry.pack()
        
        self.find_button = tk.Button(master, text="Find", command=self.process_input)
        self.find_button.pack()
        
        self.result_label = tk.Label(master, text="", fg="blue")
        self.result_label.pack()
    
    def process_input(self):
        '''
        Process the input from the user and display the result.
        '''
        input_text = self.entry.get()
        words = [word.strip() for word in input_text.split(',')]
        
        if not words or words == ['']:
            messagebox.showerror("Input Error", "Please enter at least one word.")
            return
        
        result = find_max(words)
        self.result_label.config(text=f"Word with max unique characters: {result}")

if __name__ == "__main__":
    root = tk.Tk()
    app = App(root)
    root.mainloop()
```

'''

import tkinter as tk
from tkinter import messagebox
def find_max(words):
    '''
    Write a function that accepts a list of strings.
    The list contains different words. Return the word with maximum number
    of unique characters. If multiple strings have maximum number of unique
    characters, return the one which comes first in lexicographical order.
    :param words: List of strings
    :return: String with maximum unique characters
    '''
    max_unique_count = 0
    result_word = ""
    for word in words:
        unique_chars = set(word)
        unique_count = len(unique_chars)
        if unique_count > max_unique_count or (unique_count == max_unique_count and word < result_word):
            max_unique_count = unique_count
            result_word = word
    return result_word

def find_max(words):
    '''
    Write a function that accepts a list of strings.
    The list contains different words. Return the word with maximum number
    of unique characters. If multiple strings have maximum number of unique
    characters, return the one which comes first in lexicographical order.
    :param words: List of strings
    :return: String with maximum unique characters
    '''
    max_unique_count = 0
    result_word = ""
    for word in words:
        unique_chars = set(word)
        unique_count = len(unique_chars)
        if unique_count > max_unique_count or (unique_count == max_unique_count and word < result_word):
            max_unique_count = unique_count
            result_word = word
    return result_word

'''
This module contains the main application code for finding the word with the maximum number of unique characters from a list of words provided by the user.
'''
import tkinter as tk
from tkinter import messagebox
def find_max(words):
    '''
    Write a function that accepts a list of strings.
    The list contains different words. Return the word with maximum number
    of unique characters. If multiple strings have maximum number of unique
    characters, return the one which comes first in lexicographical order.
    :param words: List of strings
    :return: String with maximum unique characters
    '''
    max_unique_count = 0
    result_word = ""
    for word in words:
        unique_chars = set(word)
        unique_count = len(unique_chars)
        if unique_count > max_unique_count or (unique_count == max_unique_count and word < result_word):
            max_unique_count = unique_count
            result_word = word
    return result_word
class App:
    def __init__(self, master):
        '''
        Initialize the application with GUI components.
        :param master: The root window
        '''
        self.master = master
        master.title("Find Max Unique Characters")
        self.label = tk.Label(master, text="Enter words separated by commas:")
        self.label.pack()
        self.entry = tk.Entry(master, width=50)
        self.entry.pack()
        self.find_button = tk.Button(master, text="Find", command=self.process_input)
        self.find_button.pack()
        self.result_label = tk.Label(master, text="", fg="blue")
        self.result_label.pack()
    def process_input(self):
        '''
        Process the input from the user and display the result.
        '''
        input_text = self.entry.get()
        words = [word.strip() for word in input_text.split(',')]
        if not words or words == ['']:
            messagebox.showerror("Input Error", "Please enter at least one word.")
            return
        result = find_max(words)
        self.result_label.config(text=f"Word with max unique characters: {result}")
if __name__ == "__main__":
    root = tk.Tk()
    app = App(root)
    root.mainloop()
