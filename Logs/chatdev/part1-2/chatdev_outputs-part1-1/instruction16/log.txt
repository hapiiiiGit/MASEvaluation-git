'''

main.py

This file contains the main GUI application for the "max_fill" task. It uses tkinter for the graphical user interface and imports the logic from logic.py.

import tkinter as tk
from tkinter import messagebox
from logic import max_fill, parse_input


class MainApp:
    def __init__(self, root):
        '''
        Initializes the GUI components.
        '''
        self.root = root
        self.root.title("Max Fill Calculator")

        # Grid input label and text box
        self.grid_label = tk.Label(root, text="Enter Grid (e.g., [[0,0,1],[1,1,0]]):")

        self.calculate_button = tk.Button(root, text="Calculate", command=self.calculate)

'''

logic.py

This file contains the core logic for the "max_fill" task, including the max_fill function and helper functions.

'''

import math


def max_fill(grid, capacity):
    '''
    Calculates the number of times the buckets need to be lowered to empty the wells.

    Parameters:
        grid (list of list of int): The grid representing the wells.
        capacity (int): The capacity of the bucket.

    Returns:
        int: The number of times the buckets need to be lowered.
    '''
    total_water = sum(sum(row) for row in grid)
    return math.ceil(total_water / capacity)


def parse_input(grid_input):
    '''
    Parses the grid input string into a 2D list of integers.

    Parameters:
        grid_input (str): The grid input as a string.

    Returns:
        list of list of int: The parsed grid.
    '''
    try:
        grid = eval(grid_input)
        if not isinstance(grid, list) or not all(isinstance(row, list) for row in grid):
            raise ValueError("Grid must be a list of lists.")
        if not all(all(cell in (0, 1) for cell in row) for row in grid):
            raise ValueError("Grid cells must be 0 or 1.")
        return grid
    except Exception as e:
        raise ValueError(f"Invalid grid format: {e}")

import math
def max_fill(grid, capacity):
    '''
    Calculates the number of times the buckets need to be lowered to empty the wells.

        capacity (int): The capacity of the bucket.

    '''
    total_water = sum(sum(row) for row in grid)

import math
def max_fill(grid, capacity):
    '''
    Calculates the number of times the buckets need to be lowered to empty the wells.
    Parameters:
        grid (list of list of int): The grid representing the wells.
        capacity (int): The capacity of the bucket.
    Returns:
        int: The number of times the buckets need to be lowered.
    '''
    total_water = sum(sum(row) for row in grid)
    return math.ceil(total_water / capacity)
def parse_input(grid_input):
    '''
    Parses the grid input string into a 2D list of integers.
    Parameters:
        grid_input (str): The grid input as a string.
    Returns:
        list of list of int: The parsed grid.
    '''
    try:
        grid = eval(grid_input)
        if not isinstance(grid, list) or not all(isinstance(row, list) for row in grid):
            raise ValueError("Grid must be a list of lists.")
        if not all(all(cell in (0, 1) for cell in row) for row in grid):
            raise ValueError("Grid cells must be 0 or 1.")
        return grid
    except Exception as e:
        raise ValueError(f"Invalid grid format: {e}")

import math
def max_fill(grid, capacity):
    '''
    Calculates the number of times the buckets need to be lowered to empty the wells.
    Parameters:
        grid (list of list of int): The grid representing the wells.
        capacity (int): The capacity of the bucket.
    Returns:
        int: The number of times the buckets need to be lowered.
    '''
    total_water = sum(sum(row) for row in grid)
    return math.ceil(total_water / capacity)
def parse_input(grid_input):
    '''

            raise ValueError("Grid cells must be 0 or 1.")
        return grid

import math
import ast

def max_fill(grid, capacity):
    '''
    Calculates the number of times the buckets need to be lowered to empty the wells.
    Parameters:
        grid (list of list of int): The grid representing the wells.
        capacity (int): The capacity of the bucket.
    Returns:
        int: The number of times the buckets need to be lowered.
    '''
    total_water = sum(sum(row) for row in grid)
    return math.ceil(total_water / capacity)

def parse_input(grid_input):
    '''
    Parses the grid input string into a 2D list of integers.
    Parameters:
        grid_input (str): The grid input as a string.
    Returns:
        list of list of int: The parsed grid.
    '''
    try:
        grid = ast.literal_eval(grid_input)
        if not isinstance(grid, list) or not all(isinstance(row, list) for row in grid):
            raise ValueError("Grid must be a list of lists.")
        if not all(all(cell in (0, 1) for cell in row) for row in grid):
            raise ValueError("Grid cells must be 0 or 1.")
        return grid
    except Exception as e:
        raise ValueError(f"Invalid grid format: {e}")

import math

def max_fill(grid, capacity):
    '''
    Calculates the number of times the buckets need to be lowered to empty the wells.
    Parameters:
        grid (list of list of int): The grid representing the wells.
        capacity (int): The capacity of the bucket.
    Returns:
        int: The number of times the buckets need to be lowered.
    '''
    total_water = sum(sum(row) for row in grid)
    return math.ceil(total_water / capacity)
def parse_input(grid_input):
    '''

    '''
    try:

            raise ValueError("Grid must be a list of lists.")
        if not all(all(cell in (0, 1) for cell in row) for row in grid):
            raise ValueError("Grid cells must be 0 or 1.")
        return grid
    except Exception as e:

import math
import ast

def max_fill(grid, capacity):
    '''
    Calculates the number of times the buckets need to be lowered to empty the wells.
    Parameters:
        grid (list of list of int): The grid representing the wells.
        capacity (int): The capacity of the bucket.
    Returns:
        int: The number of times the buckets need to be lowered.
    '''
    total_water = sum(sum(row) for row in grid)
    return math.ceil(total_water / capacity)

def parse_input(grid_input):
    '''
    Parses the grid input string into a 2D list of integers.
    Parameters:
        grid_input (str): The grid input as a string.
    Returns:
        list of list of int: The parsed grid.
    '''
    try:
        grid = ast.literal_eval(grid_input)
        if not isinstance(grid, list) or not all(isinstance(row, list) for row in grid):
            raise ValueError("Grid must be a list of lists.")
        if not all(all(cell in (0, 1) for cell in row) for row in grid):
            raise ValueError("Grid cells must be 0 or 1.")
        return grid
    except Exception as e:
        raise ValueError(f"Invalid grid format: {e}")

'''
logic.py
This file contains the core logic for the "max_fill" task, including the max_fill function and helper functions.
'''
import math
import ast
def max_fill(grid, capacity):
    '''
    Calculates the number of times the buckets need to be lowered to empty the wells.
    Parameters:
        grid (list of list of int): The grid representing the wells.
        capacity (int): The capacity of the bucket.
    Returns:
        int: The number of times the buckets need to be lowered.
    '''
    total_water = sum(sum(row) for row in grid)
    return math.ceil(total_water / capacity)
def parse_input(grid_input):
    '''
    Parses the grid input string into a 2D list of integers.
    Parameters:
        grid_input (str): The grid input as a string.
    Returns:
        list of list of int: The parsed grid.
    '''
    try:
        grid = ast.literal_eval(grid_input)
        if not isinstance(grid, list) or not all(isinstance(row, list) for row in grid):
            raise ValueError("Grid must be a list of lists.")
        if not all(all(cell in (0, 1) for cell in row) for row in grid):
            raise ValueError("Grid cells must be 0 or 1.")
        return grid
    except Exception as e:
        raise ValueError(f"Invalid grid format: {e}")

'''
logic.py
This file contains the core logic for the "max_fill" task, including the max_fill function and helper functions.
'''
import math
import ast
def max_fill(grid, capacity):
    '''
    Calculates the number of times the buckets need to be lowered to empty the wells.
    Parameters:
        grid (list of list of int): The grid representing the wells.
        capacity (int): The capacity of the bucket.
    Returns:
        int: The number of times the buckets need to be lowered.
    '''
    total_water = sum(sum(row) for row in grid)
    return math.ceil(total_water / capacity)
def parse_input(grid_input):
    '''
    Parses the grid input string into a 2D list of integers.
    Parameters:
        grid_input (str): The grid input as a string.
    Returns:
        list of list of int: The parsed grid.
    '''
    try:
        grid = ast.literal_eval(grid_input)
        if not isinstance(grid, list) or not all(isinstance(row, list) for row in grid):
            raise ValueError("Grid must be a list of lists.")
        if not all(all(cell in (0, 1) for cell in row) for row in grid):
            raise ValueError("Grid cells must be 0 or 1.")
        return grid
    except Exception as e:
        raise ValueError(f"Invalid grid format: {e}")
