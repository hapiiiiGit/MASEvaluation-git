'''

This module contains the function to find the closest vowel between two consonants
from the right side of the word, excluding vowels at the beginning and end.

'''

def get_closest_vowel(word):
    """
    You are given a word. Your task is to find the closest vowel that stands between 
    two consonants from the right side of the word (case sensitive).
    
    Vowels in the beginning and ending doesn't count. Return empty string if you didn't
    find any vowel met the above condition. 

    You may assume that the given string contains English letter only.

    Example:
    get_closest_vowel("yogurt") ==> "u"
    get_closest_vowel("FULL") ==> "U"
    get_closest_vowel("quick") ==> ""
    get_closest_vowel("ab") ==> ""
    """
    vowels = "AEIOUaeiou"
    length = len(word)
    
    # Exclude vowels at the beginning and end
    if length < 3 or word[0] in vowels or word[-1] in vowels:
        return ""
    
    for i in range(length - 2, 0, -1):
        if word[i] in vowels and word[i-1] not in vowels and word[i+1] not in vowels:
            return word[i]
    
    return ""

'''
This module contains the function to find the closest vowel between two consonants
from the right side of the word, excluding vowels at the beginning and end.
'''
def get_closest_vowel(word):
    """
    You are given a word. Your task is to find the closest vowel that stands between 
    two consonants from the right side of the word (case sensitive).
    Vowels in the beginning and ending doesn't count. Return empty string if you didn't
    find any vowel met the above condition. 
    You may assume that the given string contains English letter only.
    Example:
    get_closest_vowel("yogurt") ==> "u"
    get_closest_vowel("FULL") ==> "U"
    get_closest_vowel("quick") ==> ""
    get_closest_vowel("ab") ==> ""
    """
    vowels = "AEIOUaeiou"
    length = len(word)
    # Exclude vowels at the beginning and end
    if length < 3 or word[0] in vowels or word[-1] in vowels:
        return ""
    for i in range(length - 2, 0, -1):
        if word[i] in vowels and word[i-1] not in vowels and word[i+1] not in vowels:
            return word[i]
    return ""

'''
This module contains the function to find the closest vowel between two consonants
from the right side of the word, excluding vowels at the beginning and end.
'''
def get_closest_vowel(word):
    """
    You are given a word. Your task is to find the closest vowel that stands between 
    two consonants from the right side of the word (case sensitive).
    Vowels in the beginning and ending doesn't count. Return empty string if you didn't
    find any vowel met the above condition. 
    You may assume that the given string contains English letter only.
    Example:
    get_closest_vowel("yogurt") ==> "u"
    get_closest_vowel("FULL") ==> "U"
    get_closest_vowel("quick") ==> ""
    get_closest_vowel("ab") ==> ""
    """
    vowels = "AEIOUaeiou"
    length = len(word)
    # Exclude vowels at the beginning and end
    if length < 3 or word[0] in vowels or word[-1] in vowels:
        return ""
    for i in range(length - 2, 0, -1):
        if word[i] in vowels and word[i-1] not in vowels and word[i+1] not in vowels:
            return word[i]
    return ""

'''
This module contains the function to find the closest vowel between two consonants
from the right side of the word, excluding vowels at the beginning and end.
'''
def get_closest_vowel(word):
    """
    You are given a word. Your task is to find the closest vowel that stands between 
    two consonants from the right side of the word (case sensitive).
    Vowels in the beginning and ending doesn't count. Return empty string if you didn't
    find any vowel met the above condition. 
    You may assume that the given string contains English letter only.
    Example:
    get_closest_vowel("yogurt") ==> "u"
    get_closest_vowel("FULL") ==> "U"
    get_closest_vowel("quick") ==> ""
    get_closest_vowel("ab") ==> ""
    """
    vowels = "AEIOUaeiou"
    length = len(word)
    # Exclude vowels at the beginning and end
    if length < 3 or word[0] in vowels or word[-1] in vowels:
        return ""
    for i in range(length - 2, 0, -1):
        if word[i] in vowels and word[i-1] not in vowels and word[i+1] not in vowels:
            return word[i]
    return ""

'''
This module contains the function to find the closest vowel between two consonants
from the right side of the word, excluding vowels at the beginning and end.
'''
def get_closest_vowel(word):
    """
    You are given a word. Your task is to find the closest vowel that stands between 
    two consonants from the right side of the word (case sensitive).
    Vowels in the beginning and ending doesn't count. Return empty string if you didn't
    find any vowel met the above condition. 
    You may assume that the given string contains English letter only.
    Example:
    get_closest_vowel("yogurt") ==> "u"
    get_closest_vowel("FULL") ==> "U"
    get_closest_vowel("quick") ==> ""
    get_closest_vowel("ab") ==> ""
    """
    vowels = "AEIOUaeiou"
    length = len(word)
    # Exclude vowels at the beginning and end
    if length < 3 or word[0] in vowels or word[-1] in vowels:
        return ""
    for i in range(length - 2, 0, -1):
        if word[i] in vowels and word[i-1] not in vowels and word[i+1] not in vowels:
            return word[i]
    return ""

    def __init__(self, root):

    def run(self):

'''
This module contains the function to find the closest vowel between two consonants
from the right side of the word, excluding vowels at the beginning and end.
'''
def get_closest_vowel(word):
    """
    You are given a word. Your task is to find the closest vowel that stands between 
    two consonants from the right side of the word (case sensitive).
    Vowels in the beginning and ending doesn't count. Return empty string if you didn't
    find any vowel met the above condition. 
    You may assume that the given string contains English letter only.
    Example:
    get_closest_vowel("yogurt") ==> "u"
    get_closest_vowel("FULL") ==> "U"
    get_closest_vowel("quick") ==> ""
    get_closest_vowel("ab") ==> ""
    """
    vowels = "AEIOUaeiou"
    length = len(word)
    # Exclude vowels at the beginning and end
    if length < 3 or word[0] in vowels or word[-1] in vowels:
        return ""
    for i in range(length - 2, 0, -1):
        if word[i] in vowels and word[i-1] not in vowels and word[i+1] not in vowels:
            return word[i]
    return ""

```python
'''
This module contains the VowelFinderApp class which creates a GUI for finding the closest vowel
between two consonants from the right side of the word.
'''
import tkinter as tk
from vowel_finder import get_closest_vowel
class VowelFinderApp:
    def __init__(self, root):
        self.root = root
        self.root.title("Closest Vowel Finder")
        self.label = tk.Label(root, text="Enter a word:")
        self.label.pack()
        self.entry = tk.Entry(root)
        self.entry.pack()
        self.button = tk.Button(root, text="Find Closest Vowel", command=self.find_vowel)
        self.button.pack()
        self.result_label = tk.Label(root, text="")
        self.result_label.pack()
    def find_vowel(self):
        word = self.entry.get()
        closest_vowel = get_closest_vowel(word)
        if closest_vowel:
            self.result_label.config(text=f"Closest vowel: {closest_vowel}")
        else:
            self.result_label.config(text="No vowel found between consonants.")
    def run(self):
        self.root.mainloop()
if __name__ == "__main__":
    root = tk.Tk()
    app = VowelFinderApp(root)
    app.run()

'''
This module contains the function to find the closest vowel between two consonants
from the right side of the word, excluding vowels at the beginning and end.
'''
def get_closest_vowel(word):
    """
    You are given a word. Your task is to find the closest vowel that stands between 
    two consonants from the right side of the word (case sensitive).
    Vowels in the beginning and ending doesn't count. Return empty string if you didn't
    find any vowel met the above condition. 
    You may assume that the given string contains English letter only.
    Example:
    get_closest_vowel("yogurt") ==> "u"
    get_closest_vowel("FULL") ==> "U"
    get_closest_vowel("quick") ==> ""
    get_closest_vowel("ab") ==> ""
    """
    vowels = "AEIOUaeiou"
    length = len(word)
    # Exclude vowels at the beginning and end
    if length < 3 or word[0] in vowels or word[-1] in vowels:
        return ""
    for i in range(length - 2, 0, -1):
        if word[i] in vowels and word[i-1] not in vowels and word[i+1] not in vowels:
            return word[i]
    return ""

import tkinter as tk
from vowel_finder import get_closest_vowel
class VowelFinderApp:
    def __init__(self, root):
        self.root = root

        self.label.pack()

        self.button = tk.Button(root, text="Find Closest Vowel", command=self.find_vowel)

        self.result_label.pack()
    def find_vowel(self):
        word = self.entry.get()

            self.result_label.config(text=f"Closest vowel: {closest_vowel}")
        else:

        self.root.mainloop()
