import tkinter as tk
from tkinter import messagebox
from odd_count import odd_count

class OddCountApp:
    def __init__(self, root):
        self.root = root
        self.root.title("Odd Count Application")

        self.label = tk.Label(root, text="Enter a list of digit strings (comma-separated):")
        self.label.pack()

        self.entry = tk.Entry(root, width=50)
        self.entry.pack()

        self.button = tk.Button(root, text="Calculate Odd Counts", command=self.calculate_odd_counts)
        self.button.pack()

        self.result_label = tk.Label(root, text="")
        self.result_label.pack()

    def calculate_odd_counts(self):
        input_data = self.entry.get()
        try:
            input_list = [s.strip() for s in input_data.split(',')]
            result = odd_count(input_list)
            self.result_label.config(text="\n".join(result))
        except Exception as e:
            messagebox.showerror("Error", str(e))

if __name__ == "__main__":
    root = tk.Tk()
    app = OddCountApp(root)
    root.mainloop()

def count_odds(s):
    """Count the number of odd digits in a string."""
    odd_digits = {'1', '3', '5', '7', '9'}
    return sum(1 for char in s if char in odd_digits)

def odd_count(lst):
    """Given a list of strings, where each string consists of only digits, return a list.
    Each element i of the output should be "the number of odd elements in the
    string i of the input." where all the i's should be replaced by the number
    of odd digits in the i'th string of the input.
    """
    result = []
    for s in lst:
        odd_count = count_odds(s)
        result.append(f"the number of odd elements {odd_count}n the str{odd_count}ng {odd_count} of the {odd_count}nput.")
    return result

def count_odds(s):
    """Count the number of odd digits in a string."""
    odd_digits = {'1', '3', '5', '7', '9'}
    return sum(1 for char in s if char in odd_digits)

    """Given a list of strings, where each string consists of only digits, return a list.

class OddCountApp:

        self.label.pack()

    def calculate_odd_counts(self):

            self.result_label.config(text="\n".join(result))
        except Exception as e:

def count_odds(s):
    """Count the number of odd digits in a string."""
    odd_digits = {'1', '3', '5', '7', '9'}
    return sum(1 for char in s if char in odd_digits)
def odd_count(lst):
    """Given a list of strings, where each string consists of only digits, return a list.
    Each element i of the output should be "the number of odd elements in the
    string i of the input." where all the i's should be replaced by the number
    of odd digits in the i'th string of the input.

     "the number of odd elements 8 in the string 8 of the input."]
    """
    result = []
    for index, s in enumerate(lst):
        odd_count_value = count_odds(s)
        result.append(f"the number of odd elements {odd_count_value} in the string {odd_count_value} of the input.")
    return result
