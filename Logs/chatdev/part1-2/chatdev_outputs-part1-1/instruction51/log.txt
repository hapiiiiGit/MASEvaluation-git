def x_or_y(n, x, y):
    """
    Determines if n is a prime number and returns x if true, otherwise returns y.

        return True

import tkinter as tk
def x_or_y(n, x, y):
    """
    Determines if n is a prime number and returns x if true, otherwise returns y.

        if num <= 1:

        return True
    return x if is_prime(n) else y
