'''

This file contains the implementation of the MoveOneBallApp class, which provides a GUI
for users to input an array and check if it can be sorted in non-decreasing order by performing
right shift operations.

'''

import tkinter as tk
from tkinter import messagebox

class MoveOneBallApp:
    def __init__(self, master):
        '''
        Initializes the GUI components.
        '''
        self.master = master
        master.title("Move One Ball Checker")

        self.label = tk.Label(master, text="Enter array elements separated by commas:")
        self.label.pack()

        self.entry = tk.Entry(master)
        self.entry.pack()

        self.check_button = tk.Button(master, text="Check", command=self.check_array)
        self.check_button.pack()

        self.result_label = tk.Label(master, text="", fg="blue")
        self.result_label.pack()

    def check_array(self):
        '''
        Retrieves user input, processes it, and displays the result.
        '''
        input_text = self.entry.get()
        try:
            arr = list(map(int, input_text.split(',')))
            result = self.move_one_ball(arr)
            if result:
                self.result_label.config(text="The array can be sorted by right shifts.")
            else:
                self.result_label.config(text="The array cannot be sorted by right shifts.")
        except ValueError:
            messagebox.showerror("Invalid input", "Please enter a valid list of integers separated by commas.")

    def move_one_ball(self, arr):
        '''
        Determines if the array can be sorted by right shifts.
        '''
        if not arr:
            return True

        n = len(arr)
        count = 0

        for i in range(1, n):
            if arr[i] < arr[i - 1]:
                count += 1
                if count > 1:
                    return False

        if count == 0:
            return True

        return arr[0] >= arr[-1]

if __name__ == "__main__":
    root = tk.Tk()
    app = MoveOneBallApp(root)
    root.mainloop()
```

'''
This file contains the implementation of the MoveOneBallApp class, which provides a GUI
for users to input an array and check if it can be sorted in non-decreasing order by performing

import tkinter as tk
from tkinter import messagebox
class MoveOneBallApp:
    def __init__(self, master):
        '''
        Initializes the GUI components.
        '''
        self.master = master
        master.title("Move One Ball Checker")
        self.label = tk.Label(master, text="Enter array elements separated by commas:")
        self.label.pack()
        self.entry = tk.Entry(master)
        self.entry.pack()
        self.check_button = tk.Button(master, text="Check", command=self.check_array)
        self.check_button.pack()
        self.result_label = tk.Label(master, text="", fg="blue")
        self.result_label.pack()
    def check_array(self):
        '''
        Retrieves user input, processes it, and displays the result.
        '''
        input_text = self.entry.get()
        try:
            arr = list(map(int, input_text.split(',')))
            result = self.move_one_ball(arr)
            if result:
                self.result_label.config(text="The array can be sorted by right shifts.")
            else:
                self.result_label.config(text="The array cannot be sorted by right shifts.")
        except ValueError:
            messagebox.showerror("Invalid input", "Please enter a valid list of integers separated by commas.")
    def move_one_ball(self, arr):
        '''
        Determines if the array can be sorted by right shifts.
        '''
        if not arr:
            return True
        n = len(arr)
        count = 0
        split_point = -1
        for i in range(1, n):
            if arr[i] < arr[i - 1]:
                count += 1
                split_point = i
                if count > 1:
                    return False
        # Check if the array can be split into two sorted parts
        if count == 1:
            # Check if the array can be rearranged by right shifts
            if arr[-1] <= arr[0]:
                return True
            # Check if the split point creates two sorted parts
            if arr[split_point:] + arr[:split_point] == sorted(arr):
                return True
            return False
        return True
if __name__ == "__main__":
    root = tk.Tk()
    app = MoveOneBallApp(root)
    root.mainloop()
