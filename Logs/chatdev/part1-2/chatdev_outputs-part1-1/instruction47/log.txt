import tkinter as tk
from tkinter import messagebox

def specialFilter(nums):
    """
    Write a function that takes an array of numbers as input and returns 
    the number of elements in the array that are greater than 10 and both 
    first and last digits of a number are odd (1, 3, 5, 7, 9).
    """
    def is_odd_digit(digit):
        return digit in {'1', '3', '5', '7', '9'}
    
    count = 0
    for num in nums:
        str_num = str(abs(num))
        if num > 10 and is_odd_digit(str_num[0]) and is_odd_digit(str_num[-1]):
            count += 1
    return count

class MainApp:
    def __init__(self, root):
        self.root = root
        self.root.title("Special Filter Application")
        
        self.label = tk.Label(root, text="Enter numbers separated by commas:")
        self.label.pack()
        
        self.entry = tk.Entry(root, width=50)
        self.entry.pack()
        
        self.button = tk.Button(root, text="Filter", command=self.filter_numbers)
        self.button.pack()
        
        self.result_label = tk.Label(root, text="")
        self.result_label.pack()
    
    def filter_numbers(self):
        try:
            nums = list(map(int, self.entry.get().split(',')))
            result = specialFilter(nums)
            self.result_label.config(text=f"Result: {result}")
        except ValueError:
            messagebox.showerror("Invalid input", "Please enter valid integers separated by commas.")

if __name__ == "__main__":
    root = tk.Tk()
    app = MainApp(root)
    root.mainloop()

import tkinter as tk
from tkinter import messagebox
def specialFilter(nums):
    """
    Write a function that takes an array of numbers as input and returns 
    the number of elements in the array that are greater than 10 and both

    def is_odd_digit(digit):

import tkinter as tk  
from tkinter import messagebox  
def specialFilter(nums):  
    """  
    Write a function that takes an array of numbers as input and returns  
    the number of elements in the array that are greater than 10 and both  
    first and last digits of a number are odd (1, 3, 5, 7, 9).

    def is_odd_digit(digit):

    for num in nums:

import tkinter as tk
from tkinter import messagebox
def specialFilter(nums):
    """
    Write a function that takes an array of numbers as input and returns
    the number of elements in the array that are greater than 10 and both
    first and last digits of a number are odd (1, 3, 5, 7, 9).
    """
    def is_odd_digit(digit):
        return digit in {'1', '3', '5', '7', '9'}
    count = 0
    for num in nums:
        str_num = str(abs(num))
        if num > 10 and is_odd_digit(str_num[0]) and is_odd_digit(str_num[-1]):
            count += 1
    return count
class MainApp:
    def __init__(self, root):
        """
        Initialize the GUI application with input field, button, and result display.
        """
        self.root = root
        self.root.title("Special Filter Application")
        self.label = tk.Label(root, text="Enter numbers separated by commas:")
        self.label.pack()
        self.entry = tk.Entry(root, width=50)
        self.entry.pack()
        self.button = tk.Button(root, text="Filter", command=self.filter_numbers)
        self.button.pack()
        self.result_label = tk.Label(root, text="")
        self.result_label.pack()
    def filter_numbers(self):
        """
        Process the input numbers, apply the specialFilter function, and display the result.
        """
        try:
            nums = list(map(int, self.entry.get().split(',')))
            result = specialFilter(nums)
            self.result_label.config(text=f"Result: {result}")
        except ValueError:
            messagebox.showerror("Invalid input", "Please enter valid integers separated by commas.")
if __name__ == "__main__":
    """
    Run the GUI application.
    """
    root = tk.Tk()
    app = MainApp(root)
    root.mainloop()
