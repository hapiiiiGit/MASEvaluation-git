```python
'''

Main application file for the parentheses matching GUI.

'''

import tkinter as tk
from tkinter import messagebox

def match_parens(lst):
    '''
    Check if it is possible to concatenate the two strings in some order
    such that the resulting string will be balanced.
    
    Parameters:
    lst (list): A list of two strings consisting of '(' and ')'.
    
    Returns:
    str: 'Yes' if a balanced string can be formed, 'No' otherwise.
    '''
    # Count the number of open and close parentheses in both strings
    open_count_1 = lst[0].count('(')
    close_count_1 = lst[0].count(')')
    open_count_2 = lst[1].count('(')
    close_count_2 = lst[1].count(')')

    # Check if the total number of open and close parentheses are equal
    if open_count_1 + open_count_2 == close_count_1 + close_count_2:
        return 'Yes'
    return 'No'

class MainApp:
    def __init__(self, root):
        '''
        Initialize the main application window and widgets.
        
        Parameters:
        root (Tk): The root window object.
        '''
        self.root = root
        self.root.title("Parentheses Matcher")

        # Create and place widgets
        self.label1 = tk.Label(root, text="Enter first string:")
        self.label1.pack()

        self.entry1 = tk.Entry(root)
        self.entry1.pack()

        self.label2 = tk.Label(root, text="Enter second string:")
        self.label2.pack()

        self.entry2 = tk.Entry(root)
        self.entry2.pack()

        self.submit_button = tk.Button(root, text="Check", command=self.on_submit)
        self.submit_button.pack()

    def on_submit(self):
        '''
        Handle the submit button click event.
        '''
        str1 = self.entry1.get()
        str2 = self.entry2.get()
        result = match_parens([str1, str2])
        messagebox.showinfo("Result", f"Can form a balanced string: {result}")

if __name__ == "__main__":
    root = tk.Tk()
    app = MainApp(root)
    root.mainloop()
```

'''
Main application file for the parentheses matching GUI.
'''
import tkinter as tk
from tkinter import messagebox

def match_parens(lst):
    '''
    Check if it is possible to concatenate the two strings in some order
    such that the resulting string will be balanced.
    Parameters:
    lst (list): A list of two strings consisting of '(' and ')'.
    Returns:
    str: 'Yes' if a balanced string can be formed, 'No' otherwise.
    '''
    def is_balanced(s):
        balance = 0
        for char in s:
            if char == '(':
                balance += 1
            else:
                balance -= 1
            if balance < 0:
                return False
        return balance == 0

    # Check both possible concatenations
    if is_balanced(lst[0] + lst[1]) or is_balanced(lst[1] + lst[0]):
        return 'Yes'
    return 'No'

class MainApp:
    def __init__(self, root):
        '''
        Initialize the main application window and widgets.
        Parameters:
        root (Tk): The root window object.
        '''
        self.root = root
        self.root.title("Parentheses Matcher")
        # Create and place widgets
        self.label1 = tk.Label(root, text="Enter first string:")
        self.label1.pack()
        self.entry1 = tk.Entry(root)
        self.entry1.pack()
        self.label2 = tk.Label(root, text="Enter second string:")
        self.label2.pack()
        self.entry2 = tk.Entry(root)
        self.entry2.pack()
        self.submit_button = tk.Button(root, text="Check", command=self.on_submit)
        self.submit_button.pack()

    def on_submit(self):
        '''
        Handle the submit button click event.
        '''
        str1 = self.entry1.get()
        str2 = self.entry2.get()
        result = match_parens([str1, str2])
        messagebox.showinfo("Result", f"Can form a balanced string: {result}")

if __name__ == "__main__":
    root = tk.Tk()
    app = MainApp(root)
    root.mainloop()

'''
Main application file for the parentheses matching GUI.
'''
import tkinter as tk
from tkinter import messagebox
def match_parens(lst):
    '''
    Check if it is possible to concatenate the two strings in some order
    such that the resulting string will be balanced.
    Parameters:
    lst (list): A list of two strings consisting of '(' and ')'.
    Returns:
    str: 'Yes' if a balanced string can be formed, 'No' otherwise.
    '''
    def is_balanced(s):
        balance = 0
        for char in s:
            if char == '(':
                balance += 1
            else:
                balance -= 1
            if balance < 0:
                return False
        return balance == 0
    # Check both possible concatenations
    if is_balanced(lst[0] + lst[1]) or is_balanced(lst[1] + lst[0]):
        return 'Yes'
    return 'No'
class MainApp:
    def __init__(self, root):
        '''
        Initialize the main application window and widgets.
        Parameters:
        root (Tk): The root window object.
        '''
        self.root = root
        self.root.title("Parentheses Matcher")
        # Create and place widgets
        self.label1 = tk.Label(root, text="Enter first string:")
        self.label1.pack()
        self.entry1 = tk.Entry(root)
        self.entry1.pack()
        self.label2 = tk.Label(root, text="Enter second string:")
        self.label2.pack()
        self.entry2 = tk.Entry(root)
        self.entry2.pack()
        self.submit_button = tk.Button(root, text="Check", command=self.on_submit)
        self.submit_button.pack()
    def on_submit(self):
        '''
        Handle the submit button click event.
        '''
        str1 = self.entry1.get()
        str2 = self.entry2.get()
        # Validate input
        if not all(c in '()' for c in str1) or not all(c in '()' for c in str2):
            messagebox.showerror("Invalid Input", "Please enter only '(' and ')' characters.")
            return
        result = match_parens([str1, str2])
        messagebox.showinfo("Result", f"Can form a balanced string: {result}")
if __name__ == "__main__":
    root = tk.Tk()
    app = MainApp(root)
    root.mainloop()

'''
Main application file for the parentheses matching GUI.
'''
import tkinter as tk
from tkinter import messagebox
def match_parens(lst):
    '''
    Check if it is possible to concatenate the two strings in some order
    such that the resulting string will be balanced.
    Parameters:
    lst (list): A list of two strings consisting of '(' and ')'.
    Returns:
    str: 'Yes' if a balanced string can be formed, 'No' otherwise.
    '''
    def is_balanced(s):
        balance = 0
        for char in s:
            if char == '(':
                balance += 1
            else:
                balance -= 1
            if balance < 0:
                return False
        return balance == 0
    # Check both possible concatenations
    if is_balanced(lst[0] + lst[1]) or is_balanced(lst[1] + lst[0]):
        return 'Yes'
    return 'No'
class MainApp:
    def __init__(self, root):
        '''
        Initialize the main application window and widgets.
        Parameters:
        root (Tk): The root window object.
        '''
        self.root = root
        self.root.title("Parentheses Matcher")
        # Create and place widgets
        self.label1 = tk.Label(root, text="Enter first string:")
        self.label1.pack()
        self.entry1 = tk.Entry(root)
        self.entry1.pack()
        self.label2 = tk.Label(root, text="Enter second string:")
        self.label2.pack()
        self.entry2 = tk.Entry(root)
        self.entry2.pack()
        self.submit_button = tk.Button(root, text="Check", command=self.on_submit)
        self.submit_button.pack()
    def on_submit(self):
        '''
        Handle the submit button click event.
        '''
        str1 = self.entry1.get()
        str2 = self.entry2.get()
        # Validate input
        if not all(c in '()' for c in str1) or not all(c in '()' for c in str2):
            messagebox.showerror("Invalid Input", "Please enter only '(' and ')' characters.")
            return
        result = match_parens([str1, str2])
        messagebox.showinfo("Result", f"Can form a balanced string: {result}")
if __name__ == "__main__":
    root = tk.Tk()
    app = MainApp(root)
    root.mainloop()

'''
Main application file for the parentheses matching GUI.
'''
import tkinter as tk
from tkinter import messagebox
def match_parens(lst):
    '''
    Check if it is possible to concatenate the two strings in some order
    such that the resulting string will be balanced.
    Parameters:
    lst (list): A list of two strings consisting of '(' and ')'.
    Returns:
    str: 'Yes' if a balanced string can be formed, 'No' otherwise.
    '''
    def is_balanced(s):
        balance = 0
        for char in s:
            if char == '(':
                balance += 1
            else:
                balance -= 1
            if balance < 0:
                return False
        return balance == 0
    # Check both possible concatenations
    if is_balanced(lst[0] + lst[1]) or is_balanced(lst[1] + lst[0]):
        return 'Yes'
    return 'No'
class MainApp:
    def __init__(self, root):
        '''
        Initialize the main application window and widgets.
        Parameters:
        root (Tk): The root window object.
        '''
        self.root = root
        self.root.title("Parentheses Matcher")
        # Create and place widgets
        self.label1 = tk.Label(root, text="Enter first string:")
        self.label1.pack()
        self.entry1 = tk.Entry(root)
        self.entry1.pack()
        self.label2 = tk.Label(root, text="Enter second string:")
        self.label2.pack()
        self.entry2 = tk.Entry(root)
        self.entry2.pack()
        self.submit_button = tk.Button(root, text="Check", command=self.on_submit)
        self.submit_button.pack()
    def on_submit(self):
        '''
        Handle the submit button click event.
        '''
        str1 = self.entry1.get()
        str2 = self.entry2.get()
        # Validate input
        if not all(c in '()' for c in str1) or not all(c in '()' for c in str2):
            messagebox.showerror("Invalid Input", "Please enter only '(' and ')' characters.")
            return
        result = match_parens([str1, str2])
        messagebox.showinfo("Result", f"Can form a balanced string: {result}")
if __name__ == "__main__":
    root = tk.Tk()
    app = MainApp(root)
    root.mainloop()
