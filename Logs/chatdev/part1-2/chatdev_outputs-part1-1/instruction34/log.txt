'''

Main application file for the Sum Squares GUI application.

'''

import tkinter as tk
from tkinter import messagebox
from utils import sum_squares

class SumSquaresApp:
    def __init__(self, root):
        self.root = root
        self.root.title("Sum of Squares Calculator")

        self.label = tk.Label(root, text="Enter numbers separated by commas:")
        self.label.pack()

        self.entry = tk.Entry(root, width=50)
        self.entry.pack()

        self.calculate_button = tk.Button(root, text="Calculate", command=self.calculate_sum_squares)
        self.calculate_button.pack()

        self.result_label = tk.Label(root, text="")
        self.result_label.pack()

    def calculate_sum_squares(self):
        try:
            input_text = self.entry.get()
            numbers = list(map(float, input_text.split(',')))
            result = sum_squares(numbers)
            self.display_result(result)
        except ValueError:
            messagebox.showerror("Invalid input", "Please enter a valid list of numbers.")

    def display_result(self, result):
        self.result_label.config(text=f"Sum of squares: {result}")

if __name__ == "__main__":
    root = tk.Tk()
    app = SumSquaresApp(root)
    root.mainloop()

    '''
    return sum(math.ceil(x) ** 2 for x in lst)

'''
Main application file for the Sum Squares GUI application.
'''
import tkinter as tk
from tkinter import messagebox
from utils import sum_squares
class SumSquaresApp:
    def __init__(self, root):
        self.root = root
        self.root.title("Sum of Squares Calculator")
        self.label = tk.Label(root, text="Enter numbers separated by commas:")
        self.label.pack()
        self.entry = tk.Entry(root, width=50)
        self.entry.pack()
        self.calculate_button = tk.Button(root, text="Calculate", command=self.calculate_sum_squares)
        self.calculate_button.pack()
        self.result_label = tk.Label(root, text="")
        self.result_label.pack()
    def calculate_sum_squares(self):
        try:
            input_text = self.entry.get()
            numbers = list(map(float, input_text.split(',')))
            result = sum_squares(numbers)
            self.display_result(result)
        except ValueError:
            messagebox.showerror("Invalid input", "Please enter a valid list of numbers.")
    def display_result(self, result):
        self.result_label.config(text=f"Sum of squares: {result}")
if __name__ == "__main__":
    root = tk.Tk()
    app = SumSquaresApp(root)
    root.mainloop()

        self.label.pack()
