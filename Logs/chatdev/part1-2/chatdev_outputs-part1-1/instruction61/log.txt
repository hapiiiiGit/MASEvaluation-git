import tkinter as tk
from algebra_evaluator import AlgebraEvaluator

class AlgebraApp:
    def __init__(self, root):
        self.root = root
        self.root.title("Algebra Evaluator")

        self.operator_label = tk.Label(root, text="Operators (e.g., +, *, -):")
        self.operator_label.pack()

        self.operator_entry = tk.Entry(root)
        self.operator_entry.pack()

        self.operand_label = tk.Label(root, text="Operands (e.g., 2, 3, 4, 5):")
        self.operand_label.pack()

        self.operand_entry = tk.Entry(root)
        self.operand_entry.pack()

        self.calculate_button = tk.Button(root, text="Calculate", command=self.calculate_result)
        self.calculate_button.pack()

        self.result_label = tk.Label(root, text="Result: ")
        self.result_label.pack()

    def calculate_result(self):
        operators = self.operator_entry.get().split(',')
        operands = list(map(int, self.operand_entry.get().split(',')))

        evaluator = AlgebraEvaluator()
        result = evaluator.evaluate_expression(operators, operands)

        self.result_label.config(text=f"Result: {result}")

if __name__ == "__main__":
    root = tk.Tk()
    app = AlgebraApp(root)
    root.mainloop()

'''

This module contains the AlgebraEvaluator class which evaluates algebraic expressions.

'''

class AlgebraEvaluator:
    def evaluate_expression(self, operators, operands):
        if len(operators) != len(operands) - 1:
            raise ValueError("The number of operators must be one less than the number of operands.")

        result = operands[0]
        for i, operator in enumerate(operators):
            if operator == '+':
                result += operands[i + 1]
            elif operator == '-':
                result -= operands[i + 1]
            elif operator == '*':
                result *= operands[i + 1]
            elif operator == '//':
                result //= operands[i + 1]
            elif operator == '**':
                result **= operands[i + 1]
            else:
                raise ValueError(f"Unsupported operator: {operator}")

        return result

class AlgebraEvaluator:
    def evaluate_expression(self, operators, operands):
        if len(operators) != len(operands) - 1:
            raise ValueError("The number of operators must be one less than the number of operands.")
        result = operands[0]
        for i, operator in enumerate(operators):
            if operator == '+':
                result += operands[i + 1]
            elif operator == '-':
                result -= operands[i + 1]
            elif operator == '*':
                result *= operands[i + 1]
            elif operator == '//':
                result //= operands[i + 1]
            elif operator == '**':
                result **= operands[i + 1]
            else:
                raise ValueError(f"Unsupported operator: {operator}")
        return result

'''
This module contains the AlgebraEvaluator class which evaluates algebraic expressions.

class AlgebraEvaluator:
    def evaluate_expression(self, operators, operands):
        if len(operators) != len(operands) - 1:
            raise ValueError("The number of operators must be one less than the number of operands.")
        result = operands[0]
        for i, operator in enumerate(operators):
            if operator == '+':
                result += operands[i + 1]
            elif operator == '-':
                result -= operands[i + 1]
            elif operator == '*':
                result *= operands[i + 1]
            elif operator == '//':
                if operands[i + 1] == 0:
                    raise ZeroDivisionError("Cannot divide by zero.")
                result //= operands[i + 1]
            elif operator == '**':
                result **= operands[i + 1]
            else:
                raise ValueError(f"Unsupported operator: {operator}")
        return result

class AlgebraEvaluator:
    def evaluate_expression(self, operators, operands):
        if len(operators) != len(operands) - 1:
            raise ValueError("The number of operators must be one less than the number of operands.")
        result = operands[0]
        for i, operator in enumerate(operators):
            if operator == '+':
                result += operands[i + 1]
            elif operator == '-':
                result -= operands[i + 1]
            elif operator == '*':
                result *= operands[i + 1]
            elif operator == '//':
                if operands[i + 1] == 0:
                    raise ZeroDivisionError("Cannot divide by zero.")
                result //= operands[i + 1]
            elif operator == '**':
                result **= operands[i + 1]
            else:
                raise ValueError(f"Unsupported operator: {operator}")
        return result

'''
This module contains the AlgebraEvaluator class which evaluates algebraic expressions.

class AlgebraEvaluator:
    def evaluate_expression(self, operators, operands):
        if len(operators) != len(operands) - 1:
            raise ValueError("The number of operators must be one less than the number of operands.")
        result = operands[0]
        for i, operator in enumerate(operators):
            if operator == '+':
                result += operands[i + 1]
            elif operator == '-':
                result -= operands[i + 1]
            elif operator == '*':
                result *= operands[i + 1]
            elif operator == '//':
                if operands[i + 1] == 0:
                    raise ZeroDivisionError("Cannot divide by zero.")
                result //= operands[i + 1]
            elif operator == '**':
                result **= operands[i + 1]
            else:
                raise ValueError(f"Unsupported operator: {operator}")
        return result

'''
This module contains the AlgebraEvaluator class which evaluates algebraic expressions.
'''
class AlgebraEvaluator:
    def evaluate_expression(self, operators, operands):
        if len(operators) != len(operands) - 1:
            raise ValueError("The number of operators must be one less than the number of operands.")
        result = operands[0]
        for i, operator in enumerate(operators):
            if operator == '+':
                result += operands[i + 1]
            elif operator == '-':
                result -= operands[i + 1]
            elif operator == '*':
                result *= operands[i + 1]
            elif operator == '//':
                if operands[i + 1] == 0:
                    raise ZeroDivisionError("Cannot divide by zero.")
                result //= operands[i + 1]
            elif operator == '**':
                result **= operands[i + 1]
            else:
                raise ValueError(f"Unsupported operator: {operator}")
        return result

# Algebra Evaluator User Manual

Welcome to the Algebra Evaluator application! This software allows users to evaluate algebraic expressions using basic operations through a graphical user interface (GUI). Below, you will find detailed instructions on how to install, use, and understand the main functions of the application.

## Introduction

The Algebra Evaluator is a Python-based application that provides users with a simple interface to input algebraic operations and operands, and then calculates the result of the expression. The application supports basic algebraic operations such as addition, subtraction, multiplication, floor division, and exponentiation.

## Installation

### Prerequisites

- Ensure you have Python installed on your system. You can download it from the [official Python website](https://www.python.org/downloads/).

### Quick Install

1. Clone the repository or download the source code to your local machine.

   ```bash
   # No external dependencies are required for this project.
   ```

4. Run the application:

   ```bash
   python main.py
   ```

## Usage

1. Launch the application by running the `main.py` script.

2. A GUI window will appear with fields to input operators and operands.

3. Enter the operators in the "Operators" field, separated by commas (e.g., `+, *, -`).

4. Enter the operands in the "Operands" field, separated by commas (e.g., `2, 3, 4, 5`).

5. Click the "Calculate" button to evaluate the expression.

6. The result will be displayed below the "Result" label.

## Main Functions

- **Operators Input**: Accepts a list of algebraic operators (`+`, `-`, `*`, `//`, `**`) separated by commas.

- **Operands Input**: Accepts a list of non-negative integers separated by commas.

- **Calculate Button**: Triggers the evaluation of the expression using the provided operators and operands.

- **Result Display**: Shows the result of the evaluated expression or an error message if the input is invalid.

## Troubleshooting

- Ensure that the number of operators is one less than the number of operands.

- Only use supported operators: `+`, `-`, `*`, `//`, `**`.

- Ensure operands are non-negative integers.

- If you encounter a "Cannot divide by zero" error, check that you are not performing floor division by zero.

- For any unsupported operator errors, verify that all operators are valid and correctly entered.

Thank you for using the Algebra Evaluator! If you have any questions or need further assistance, please feel free to reach out.
