'''

This file contains the main application logic for the fraction simplification checker.
It includes the `simplify` function to determine if the product of two fractions simplifies to a whole number,
and a `GUI` class to provide a graphical user interface for user interaction.

'''

from tkinter import Tk, Label, Entry, Button, StringVar
from fractions import Fraction


def simplify(x, n):
    '''
    Determines if the product of two fractions simplifies to a whole number.

    Args:
        x (str): A string representing the first fraction in the format "numerator/denominator".
        n (str): A string representing the second fraction in the format "numerator/denominator".

    Returns:
        bool: True if the product simplifies to a whole number, False otherwise.
    '''
    # Convert the string fractions to Fraction objects
    fraction_x = Fraction(x)
    fraction_n = Fraction(n)

    # Calculate the product
    product = fraction_x * fraction_n

    # Check if the product is a whole number
    return product.denominator == 1


class GUI:
    '''
    A class to create and manage the graphical user interface for the fraction simplification checker.
    '''
    def __init__(self, root):
        '''
        Initializes the GUI components.

        Args:
            root (Tk): The root window of the tkinter application.
        '''
        self.root = root
        self.root.title("Fraction Simplification Checker")

        # Input labels and entry fields
        Label(root, text="Enter first fraction (x):").grid(row=0, column=0, padx=10, pady=10)
        self.x_input = Entry(root)
        self.x_input.grid(row=0, column=1, padx=10, pady=10)

        Label(root, text="Enter second fraction (n):").grid(row=1, column=0, padx=10, pady=10)
        self.n_input = Entry(root)
        self.n_input.grid(row=1, column=1, padx=10, pady=10)

        # Button to calculate the result
        self.calculate_button = Button(root, text="Simplify", command=self.calculate)
        self.calculate_button.grid(row=2, column=0, columnspan=2, pady=10)

        # Label to display the result
        self.result_var = StringVar()
        self.result_label = Label(root, textvariable=self.result_var, fg="blue")
        self.result_label.grid(row=3, column=0, columnspan=2, pady=10)

    def calculate(self):
        '''
        Retrieves user input, calls the `simplify` function, and displays the result.
        '''
        # Get the input fractions
        x = self.x_input.get()
        n = self.n_input.get()

        try:
            # Call the simplify function
            result = simplify(x, n)

            # Display the result
            if result:
                self.result_var.set("The product simplifies to a whole number: True")
            else:
                self.result_var.set("The product does not simplify to a whole number: False")
        except Exception as e:
            # Handle invalid input
            self.result_var.set(f"Error: {str(e)}")


if __name__ == "__main__":
    # Create the main application window
    root = Tk()
    app = GUI(root)
    root.mainloop()
```

'''

from tkinter import Tk, Label, Entry, Button, StringVar
from fractions import Fraction
def simplify(x, n):
    '''
    Determines if the product of two fractions simplifies to a whole number.
    Args:
        x (str): A string representing the first fraction in the format "numerator/denominator".
        n (str): A string representing the second fraction in the format "numerator/denominator".
    Returns:
        bool: True if the product simplifies to a whole number, False otherwise.
    '''
    # Convert the string fractions to Fraction objects
    fraction_x = Fraction(x)
    fraction_n = Fraction(n)
    # Calculate the product
    product = fraction_x * fraction_n
    # Check if the product is a whole number
    return product.denominator == 1
class GUI:
    '''
    A class to create and manage the graphical user interface for the fraction simplification checker.
    '''

        Label(root, text="Enter second fraction (n):").grid(row=1, column=0, padx=10, pady=10)

        self.n_input.grid(row=1, column=1, padx=10, pady=10)

        self.result_label = Label(root, textvariable=self.result_var, fg="blue")

        '''

def simplify(x, n):
    '''
    Determines if the product of two fractions simplifies to a whole number.
    Args:

        n (str): A string representing the second fraction in the format "numerator/denominator".

    '''

    fraction_x = Fraction(x)
    fraction_n = Fraction(n)

    product = fraction_x * fraction_n

    return product.denominator == 1

'''

from tkinter import Tk, Label, Entry, Button, StringVar

    '''
    Determines if the product of two fractions simplifies to a whole number.
    Args:

'''
This file contains the main application logic for the fraction simplification checker.
It includes the `simplify` function to determine if the product of two fractions simplifies to a whole number,
and a `GUI` class to provide a graphical user interface for user interaction.
'''
from tkinter import Tk, Label, Entry, Button, StringVar
from fractions import Fraction
def simplify(x, n):
    '''
    Determines if the product of two fractions simplifies to a whole number.
    Args:
        x (str): A string representing the first fraction in the format "numerator/denominator".
        n (str): A string representing the second fraction in the format "numerator/denominator".
    Returns:
        bool: True if the product simplifies to a whole number, False otherwise.
    '''
    # Convert the string fractions to Fraction objects
    fraction_x = Fraction(x)
    fraction_n = Fraction(n)
    # Calculate the product
    product = fraction_x * fraction_n
    # Check if the product is a whole number
    return product.denominator == 1
class GUI:
    '''
    A class to create and manage the graphical user interface for the fraction simplification checker.
    '''
    def __init__(self, root):
        '''
        Initializes the GUI components.
        Args:
            root (Tk): The root window of the tkinter application.
        '''
        self.root = root
        self.root.title("Fraction Simplification Checker")
        # Input labels and entry fields
        Label(root, text="Enter first fraction (x):").grid(row=0, column=0, padx=10, pady=10)
        self.x_input = Entry(root)
        self.x_input.grid(row=0, column=1, padx=10, pady=10)
        Label(root, text="Enter second fraction (n):").grid(row=1, column=0, padx=10, pady=10)
        self.n_input = Entry(root)
        self.n_input.grid(row=1, column=1, padx=10, pady=10)
        # Button to calculate the result
        self.calculate_button = Button(root, text="Simplify", command=self.calculate)
        self.calculate_button.grid(row=2, column=0, columnspan=2, pady=10)
        # Label to display the result
        self.result_var = StringVar()
        self.result_label = Label(root, textvariable=self.result_var, fg="blue")
        self.result_label.grid(row=3, column=0, columnspan=2, pady=10)
    def calculate(self):
        '''
        Retrieves user input, calls the `simplify` function, and displays the result.
        '''
        # Get the input fractions
        x = self.x_input.get()
        n = self.n_input.get()
        try:
            # Call the simplify function
            result = simplify(x, n)
            # Display the result
            if result:
                self.result_var.set("The product simplifies to a whole number: True")
            else:
                self.result_var.set("The product does not simplify to a whole number: False")
        except Exception as e:
            # Handle invalid input
            self.result_var.set(f"Error: {str(e)}")
if __name__ == "__main__":
    # Create the main application window
    root = Tk()
    app = GUI(root)
    root.mainloop()

'''
This file contains the main application logic for the fraction simplification checker.
It includes the `simplify` function to determine if the product of two fractions simplifies to a whole number,
and a `GUI` class to provide a graphical user interface for user interaction.
'''
from tkinter import Tk, Label, Entry, Button, StringVar
from fractions import Fraction
def simplify(x, n):
    '''
    Determines if the product of two fractions simplifies to a whole number.
    Args:
        x (str): A string representing the first fraction in the format "numerator/denominator".
        n (str): A string representing the second fraction in the format "numerator/denominator".
    Returns:
        bool: True if the product simplifies to a whole number, False otherwise.
    '''
    # Convert the string fractions to Fraction objects
    fraction_x = Fraction(x)
    fraction_n = Fraction(n)
    # Calculate the product
    product = fraction_x * fraction_n
    # Check if the product is a whole number
    return product.denominator == 1
class GUI:
    '''
    A class to create and manage the graphical user interface for the fraction simplification checker.
    '''
    def __init__(self, root):
        '''
        Initializes the GUI components.
        Args:
            root (Tk): The root window of the tkinter application.
        '''
        self.root = root
        self.root.title("Fraction Simplification Checker")
        # Input labels and entry fields
        Label(root, text="Enter first fraction (x):").grid(row=0, column=0, padx=10, pady=10)
        self.x_input = Entry(root)
        self.x_input.grid(row=0, column=1, padx=10, pady=10)
        Label(root, text="Enter second fraction (n):").grid(row=1, column=0, padx=10, pady=10)
        self.n_input = Entry(root)
        self.n_input.grid(row=1, column=1, padx=10, pady=10)
        # Button to calculate the result
        self.calculate_button = Button(root, text="Simplify", command=self.calculate)
        self.calculate_button.grid(row=2, column=0, columnspan=2, pady=10)
        # Label to display the result
        self.result_var = StringVar()
        self.result_label = Label(root, textvariable=self.result_var, fg="blue")
        self.result_label.grid(row=3, column=0, columnspan=2, pady=10)
    def calculate(self):
        '''
        Retrieves user input, calls the `simplify` function, and displays the result.

            # Call the simplify function
            result = simplify(x, n)
            # Display the result
            if result:
                self.result_var.set("The product simplifies to a whole number: True")
            else:
                self.result_var.set("The product does not simplify to a whole number: False")
        except Exception as e:
            # Handle invalid input
            self.result_var.set(f"Error: {str(e)}")
if __name__ == "__main__":
    # Create the main application window
    root = Tk()
    app = GUI(root)
    root.mainloop()

from tkinter import Tk, Label, Entry, Button, StringVar
from fractions import Fraction
def simplify(x, n):
    '''
    Determines if the product of two fractions simplifies to a whole number.
    Args:
        x (str): A string representing the first fraction in the format "numerator/denominator".
        n (str): A string representing the second fraction in the format "numerator/denominator".
    Returns:
        bool: True if the product simplifies to a whole number, False otherwise.
    '''
    # Convert the string fractions to Fraction objects
    fraction_x = Fraction(x)
    fraction_n = Fraction(n)
    # Calculate the product
    product = fraction_x * fraction_n
    # Check if the product is a whole number
    return product.denominator == 1

        Initializes the GUI components.

        Label(root, text="Enter first fraction (x):").grid(row=0, column=0, padx=10, pady=10)

        Label(root, text="Enter second fraction (n):").grid(row=1, column=0, padx=10, pady=10)

'''
This file contains the main application logic for the fraction simplification checker.
It includes the `simplify` function to determine if the product of two fractions simplifies to a whole number,
and a `GUI` class to provide a graphical user interface for user interaction.
'''
from tkinter import Tk, Label, Entry, Button, StringVar
from fractions import Fraction
def simplify(x, n):
    '''
    Determines if the product of two fractions simplifies to a whole number.
    Args:
        x (str): A string representing the first fraction in the format "numerator/denominator".
        n (str): A string representing the second fraction in the format "numerator/denominator".
    Returns:
        bool: True if the product simplifies to a whole number, False otherwise.
    '''
    # Convert the string fractions to Fraction objects
    fraction_x = Fraction(x)
    fraction_n = Fraction(n)
    # Calculate the product
    product = fraction_x * fraction_n
    # Check if the product is a whole number
    return product.denominator == 1
class GUI:
    '''
    A class to create and manage the graphical user interface for the fraction simplification checker.
    '''
    def __init__(self, root):
        '''
        Initializes the GUI components.
        Args:
            root (Tk): The root window of the tkinter application.
        '''
        self.root = root
        self.root.title("Fraction Simplification Checker")
        # Input labels and entry fields
        Label(root, text="Enter first fraction (x):").grid(row=0, column=0, padx=10, pady=10)
        self.x_input = Entry(root)
        self.x_input.grid(row=0, column=1, padx=10, pady=10)
        Label(root, text="Enter second fraction (n):").grid(row=1, column=0, padx=10, pady=10)
        self.n_input = Entry(root)
        self.n_input.grid(row=1, column=1, padx=10, pady=10)
        # Button to calculate the result
        self.calculate_button = Button(root, text="Simplify", command=self.calculate)
        self.calculate_button.grid(row=2, column=0, columnspan=2, pady=10)
        # Label to display the result
        self.result_var = StringVar()
        self.result_label = Label(root, textvariable=self.result_var, fg="blue")
        self.result_label.grid(row=3, column=0, columnspan=2, pady=10)
    def calculate(self):
        '''
        Retrieves user input, calls the `simplify` function, and displays the result.
        '''
        # Get the input fractions
        x = self.x_input.get()
        n = self.n_input.get()
        try:
            # Call the simplify function
            result = simplify(x, n)
            # Display the result
            if result:
                self.result_var.set("The product simplifies to a whole number: True")
            else:
                self.result_var.set("The product does not simplify to a whole number: False")
        except Exception as e:
            # Handle invalid input
            self.result_var.set(f"Error: {str(e)}")
if __name__ == "__main__":
    # Create the main application window
    root = Tk()
    app = GUI(root)
    root.mainloop()

- Ensure you have Python 3.6 or higher installed on your system. You can download it from the [official Python website](https://www.python.org/downloads/).

### Quick Install

1. **Clone the Repository**: Clone the project repository to your local machine using the following command:
   ```bash
   git clone <repository-url>
   ```

2. **Navigate to the Project Directory**: Change your directory to the project folder:
   ```bash
   cd <project-directory>
   ```

3. **Install Dependencies**: Install the required dependencies using pip:
   ```bash
   pip install -r requirements.txt
   ```

   Ensure that your Python version is 3.6 or higher as specified in the `requirements.txt`.

## How to Use

1. **Run the Application**: Start the application by executing the `main.py` file:
   ```bash
   python main.py
   ```

2. **Input Fractions**: Enter the first fraction in the "Enter first fraction (x):" field and the second fraction in the "Enter second fraction (n):" field. The fractions should be in the format "numerator/denominator".

3. **Check Simplification**: Click the "Simplify" button to determine if the product of the two fractions simplifies to a whole number.

4. **View Results**: The result will be displayed below the button. It will indicate whether the product simplifies to a whole number (True) or not (False).

## Troubleshooting

- **Invalid Input**: If you enter an invalid fraction format, an error message will be displayed. Ensure that the fractions are in the correct format "numerator/denominator" with positive whole numbers.

- **Dependencies**: If you encounter issues with dependencies, ensure that your Python environment is set up correctly and that all required packages are installed.
