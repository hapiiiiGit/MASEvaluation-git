'''

Main application file for evaluating if a number can be expressed as the sum of four positive even numbers.

'''

import tkinter as tk
from tkinter import messagebox

def is_equal_to_sum_even(n):
    '''
    Evaluate whether the given number n can be written as the sum of exactly 4 positive even numbers.
    
    Parameters:
    n (int): The number to evaluate.
    
    Returns:
    bool: True if n can be expressed as the sum of four positive even numbers, otherwise False.
    '''
    # A number can be expressed as the sum of four positive even numbers if it is at least 8
    # and even. The smallest sum of four positive even numbers is 2 + 2 + 2 + 2 = 8.
    return n >= 8 and n % 2 == 0

class SumEvenApp:
    '''
    GUI Application class for evaluating numbers.
    '''
    def __init__(self, master):
        '''
        Initialize the GUI components.
        
        Parameters:
        master (tk.Tk): The root window.
        '''
        self.master = master
        master.title("Sum of Four Even Numbers Checker")

        self.label = tk.Label(master, text="Enter a number:")
        self.label.pack()

        self.entry = tk.Entry(master)
        self.entry.pack()

        self.check_button = tk.Button(master, text="Check", command=self.evaluate_number)
        self.check_button.pack()

        self.result_label = tk.Label(master, text="")
        self.result_label.pack()

    def evaluate_number(self):
        '''
        Evaluate the input number and display the result.
        '''
        try:
            n = int(self.entry.get())
            if is_equal_to_sum_even(n):
                result_text = f"{n} can be expressed as the sum of four positive even numbers."
            else:
                result_text = f"{n} cannot be expressed as the sum of four positive even numbers."
            self.result_label.config(text=result_text)
        except ValueError:
            messagebox.showerror("Invalid input", "Please enter a valid integer.")

if __name__ == "__main__":
    root = tk.Tk()
    app = SumEvenApp(root)
    root.mainloop()

    '''

'''
Main application file for evaluating if a number can be expressed as the sum of four positive even numbers.
'''
import tkinter as tk
from tkinter import messagebox

def is_equal_to_sum_even(n):
    '''
    Evaluate whether the given number n can be written as the sum of exactly 4 positive even numbers.
    Parameters:
    n (int): The number to evaluate.
    Returns:
    bool: True if n can be expressed as the sum of four positive even numbers, otherwise False.
    '''
    # A number can be expressed as the sum of four positive even numbers if it is at least 8
    # and even. The smallest sum of four positive even numbers is 2 + 2 + 2 + 2 = 8.
    return n >= 8 and n % 2 == 0

class SumEvenApp:
    '''
    GUI Application class for evaluating numbers.
    '''
    def __init__(self, master):
        '''
        Initialize the GUI components.
        Parameters:
        master (tk.Tk): The root window.
        '''
        self.master = master
        master.title("Sum of Four Even Numbers Checker")
        self.label = tk.Label(master, text="Enter a number:")
        self.label.pack()
        self.entry = tk.Entry(master)
        self.entry.pack()
        self.check_button = tk.Button(master, text="Check", command=self.evaluate_number)
        self.check_button.pack()
        self.result_label = tk.Label(master, text="")
        self.result_label.pack()

    def evaluate_number(self):
        '''
        Evaluate the input number and display the result.
        '''
        try:
            n = int(self.entry.get())
            if is_equal_to_sum_even(n):
                result_text = f"{n} can be expressed as the sum of four positive even numbers."
            else:
                result_text = f"{n} cannot be expressed as the sum of four positive even numbers."
            self.result_label.config(text=result_text)
        except ValueError:
            messagebox.showerror("Invalid input", "Please enter a valid integer.")

if __name__ == "__main__":
    root = tk.Tk()
    app = SumEvenApp(root)
    root.mainloop()

'''
Main application file for evaluating if a number can be expressed as the sum of four positive even numbers.
'''
import tkinter as tk
from tkinter import messagebox
def is_equal_to_sum_even(n):
    '''
    Evaluate whether the given number n can be written as the sum of exactly 4 positive even numbers.
    Parameters:
    n (int): The number to evaluate.
    Returns:
    bool: True if n can be expressed as the sum of four positive even numbers, otherwise False.
    '''
    # A number can be expressed as the sum of four positive even numbers if it is at least 8
    # and even. The smallest sum of four positive even numbers is 2 + 2 + 2 + 2 = 8.
    return n >= 8 and n % 2 == 0
class SumEvenApp:
    '''
    GUI Application class for evaluating numbers.
    '''
    def __init__(self, master):
        '''
        Initialize the GUI components.
        Parameters:
        master (tk.Tk): The root window.
        '''
        self.master = master
        master.title("Sum of Four Even Numbers Checker")
        self.label = tk.Label(master, text="Enter a number:")
        self.label.pack()
        self.entry = tk.Entry(master)
        self.entry.pack()
        self.check_button = tk.Button(master, text="Check", command=self.evaluate_number)
        self.check_button.pack()
        self.result_label = tk.Label(master, text="")
        self.result_label.pack()
    def evaluate_number(self):
        '''
        Evaluate the input number and display the result.
        '''
        try:
            n = int(self.entry.get())
            if is_equal_to_sum_even(n):
                result_text = f"{n} can be expressed as the sum of four positive even numbers."
            else:
                result_text = f"{n} cannot be expressed as the sum of four positive even numbers."
            self.result_label.config(text=result_text)
        except ValueError:
            messagebox.showerror("Invalid input", "Please enter a valid integer.")
if __name__ == "__main__":
    root = tk.Tk()
    app = SumEvenApp(root)
    root.mainloop()
