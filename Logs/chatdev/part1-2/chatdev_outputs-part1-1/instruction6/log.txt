import tkinter as tk
from tkinter import messagebox
from processing import by_length

class MainApp:
    def __init__(self, root):
        self.root = root
        self.root.title("Number Processor")

        self.label = tk.Label(root, text="Enter numbers separated by commas:")
        self.label.pack()

        self.entry = tk.Entry(root, width=50)

        self.process_button = tk.Button(root, text="Process", command=self.process_input)

'''

Processing functions for the number array.

'''

def by_length(arr):
    '''
    Given an array of integers, sort the integers that are between 1 and 9 inclusive,
    reverse the resulting array, and then replace each digit by its corresponding name from
    "One", "Two", "Three", "Four", "Five", "Six", "Seven", "Eight", "Nine".
    '''
    def filter_and_sort(arr):
        # Filter numbers between 1 and 9
        filtered = [num for num in arr if 1 <= num <= 9]
        # Sort and reverse the list
        return sorted(filtered, reverse=True)

    def convert_to_names(arr):
        # Mapping of numbers to names
        num_to_name = {
            1: "One", 2: "Two", 3: "Three", 4: "Four",
            5: "Five", 6: "Six", 7: "Seven", 8: "Eight", 9: "Nine"
        }
        return [num_to_name[num] for num in arr]

    sorted_filtered = filter_and_sort(arr)
    return convert_to_names(sorted_filtered)
```

'''
Processing functions for the number array.
'''
def by_length(arr):
    '''
    Given an array of integers, sort the integers that are between 1 and 9 inclusive,
    reverse the resulting array, and then replace each digit by its corresponding name from
    "One", "Two", "Three", "Four", "Five", "Six", "Seven", "Eight", "Nine".
    '''
    def filter_and_sort(arr):
        # Filter numbers between 1 and 9
        filtered = [num for num in arr if 1 <= num <= 9]
        # Sort and reverse the list
        return sorted(filtered, reverse=True)
    def convert_to_names(arr):
        # Mapping of numbers to names
        num_to_name = {
            1: "One", 2: "Two", 3: "Three", 4: "Four",
            5: "Five", 6: "Six", 7: "Seven", 8: "Eight", 9: "Nine"
        }
        return [num_to_name[num] for num in arr]
    sorted_filtered = filter_and_sort(arr)
    return convert_to_names(sorted_filtered)

def by_length(arr):

'''
Processing functions for the number array.
'''
def by_length(arr):
    '''
    Given an array of integers, sort the integers that are between 1 and 9 inclusive,
    reverse the resulting array, and then replace each digit by its corresponding name from
    "One", "Two", "Three", "Four", "Five", "Six", "Seven", "Eight", "Nine".
    '''
    def filter_and_sort(arr):
        # Sort the entire array first
        sorted_arr = sorted(arr)
        # Filter numbers between 1 and 9
        filtered = [num for num in sorted_arr if 1 <= num <= 9]
        # Reverse the filtered list
        return list(reversed(filtered))
    def convert_to_names(arr):
        # Mapping of numbers to names
        num_to_name = {
            1: "One", 2: "Two", 3: "Three", 4: "Four",
            5: "Five", 6: "Six", 7: "Seven", 8: "Eight", 9: "Nine"
        }
        return [num_to_name[num] for num in arr]
    sorted_filtered = filter_and_sort(arr)
    return convert_to_names(sorted_filtered)

'''
Processing functions for the number array.
'''
def by_length(arr):
    '''
    Given an array of integers, sort the integers that are between 1 and 9 inclusive,
    reverse the resulting array, and then replace each digit by its corresponding name from
    "One", "Two", "Three", "Four", "Five", "Six", "Seven", "Eight", "Nine".
    '''
    def filter_and_sort(arr):
        # Sort the entire array first
        sorted_arr = sorted(arr)
        # Filter numbers between 1 and 9
        filtered = [num for num in sorted_arr if 1 <= num <= 9]
        # Reverse the filtered list
        return list(reversed(filtered))
    def convert_to_names(arr):
        # Mapping of numbers to names
        num_to_name = {
            1: "One", 2: "Two", 3: "Three", 4: "Four",
            5: "Five", 6: "Six", 7: "Seven", 8: "Eight", 9: "Nine"
        }
        return [num_to_name[num] for num in arr]
    sorted_filtered = filter_and_sort(arr)
    return convert_to_names(sorted_filtered)

'''
Processing functions for the number array.
'''
def by_length(arr):
    '''
    Given an array of integers, sort the integers that are between 1 and 9 inclusive,
    reverse the resulting array, and then replace each digit by its corresponding name from
    "One", "Two", "Three", "Four", "Five", "Six", "Seven", "Eight", "Nine".
    '''
    def filter_and_sort(arr):
        # Sort the entire array first
        sorted_arr = sorted(arr)
        # Filter numbers between 1 and 9
        filtered = [num for num in sorted_arr if 1 <= num <= 9]
        # Reverse the filtered list
        return list(reversed(filtered))
    def convert_to_names(arr):
        # Mapping of numbers to names
        num_to_name = {
            1: "One", 2: "Two", 3: "Three", 4: "Four",
            5: "Five", 6: "Six", 7: "Seven", 8: "Eight", 9: "Nine"
        }
        return [num_to_name[num] for num in arr]
    sorted_filtered = filter_and_sort(arr)
    return convert_to_names(sorted_filtered)
