import tkinter as tk
from tribonacci_calculator import TribonacciCalculator

class TribonacciApp:

'''

Module for calculating the Tribonacci sequence.

class TribonacciCalculator:
    def calculate_sequence(self, n):
        sequence = [1] * (n + 1)
        
        for i in range(1, n + 1):
            if i == 1:
                sequence[i] = 3
            elif i % 2 == 0:
                sequence[i] = 1 + i // 2
            else:
                sequence[i] = sequence[i - 1] + sequence[i - 2] + sequence[i + 1] if i + 1 <= n else sequence[i - 1] + sequence[i - 2]
        
        return sequence

class TribonacciCalculator:
    def calculate_sequence(self, n):
        sequence = [1] * (n + 1)
        for i in range(1, n + 1):
            if i == 1:
                sequence[i] = 3
            elif i % 2 == 0:
                sequence[i] = 1 + i // 2
            else:
                sequence[i] = sequence[i - 1] + sequence[i - 2] + sequence[i + 1] if i + 1 <= n else sequence[i - 1] + sequence[i - 2]
        return sequence

class TribonacciCalculator:
    def calculate_sequence(self, n):
        sequence = [1] * (n + 1)
        for i in range(1, n + 1):
            if i == 1:
                sequence[i] = 3
            elif i % 2 == 0:
                sequence[i] = 1 + i // 2
            else:
                if i + 1 <= n:
                    sequence[i] = sequence[i - 1] + sequence[i - 2] + sequence[i + 1]
                else:
                    sequence[i] = sequence[i - 1] + sequence[i - 2]
        return sequence

class TribonacciCalculator:
    def calculate_sequence(self, n):
        if n < 0:
            raise ValueError("The number must be non-negative.")
        sequence = [0] * (n + 1)
        if n >= 1:

                    sequence[i] = sequence[i - 1] + sequence[i - 2] + next_value
        return sequence

'''
Module for calculating the Tribonacci sequence.
'''
class TribonacciCalculator:
    def calculate_sequence(self, n):
        if n < 0:
            raise ValueError("The number must be non-negative.")
        sequence = [0] * (n + 1)
        if n >= 1:
            sequence[1] = 3
        for i in range(2, n + 1):
            if i % 2 == 0:
                sequence[i] = 1 + i // 2
            else:
                if i + 1 <= n:
                    sequence[i] = sequence[i - 1] + sequence[i - 2] + sequence[i + 1]
                else:
                    # Calculate tri(n+1) dynamically if it is out of bounds
                    next_value = 1 + (i + 1) // 2 if (i + 1) % 2 == 0 else sequence[i - 1] + sequence[i - 2]
                    sequence[i] = sequence[i - 1] + sequence[i - 2] + next_value
        return sequence

'''
Module for calculating the Tribonacci sequence.
'''
class TribonacciCalculator:
    def calculate_sequence(self, n):
        if n < 0:
            raise ValueError("The number must be non-negative.")
        sequence = [0] * (n + 1)
        if n >= 1:
            sequence[1] = 3
        for i in range(2, n + 1):
            if i % 2 == 0:
                sequence[i] = 1 + i // 2
            else:
                if i + 1 <= n:
                    sequence[i] = sequence[i - 1] + sequence[i - 2] + sequence[i + 1]
                else:
                    # Calculate tri(n+1) dynamically if it is out of bounds
                    next_value = 1 + (i + 1) // 2 if (i + 1) % 2 == 0 else sequence[i - 1] + sequence[i - 2]
                    sequence[i] = sequence[i - 1] + sequence[i - 2] + next_value
        return sequence
