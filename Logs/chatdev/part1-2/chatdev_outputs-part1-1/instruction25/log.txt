'''

This is the main file for the Date Validator application. It sets up the GUI using tkinter and includes the logic for validating date strings.

'''

import tkinter as tk
from tkinter import messagebox

def valid_date(date):
    '''
    Validates a given date string and returns True if the date is valid otherwise False.
    The date is valid if all of the following rules are satisfied:
    1. The date string is not empty.
    2. The number of days is not less than 1 or higher than 31 days for months 1,3,5,7,8,10,12.
       And the number of days is not less than 1 or higher than 30 days for months 4,6,9,11.
       And, the number of days is not less than 1 or higher than 29 for the month 2.
    3. The months should not be less than 1 or higher than 12.
    4. The date should be in the format: mm-dd-yyyy

    :param date: The date string to validate.
    :return: True if the date is valid, False otherwise.
    '''
    if not date:
        return False

    parts = date.split('-')
    if len(parts) != 3:
        return False

    try:
        month, day, year = map(int, parts)
    except ValueError:
        return False

    if month < 1 or month > 12:
        return False

    if month in {1, 3, 5, 7, 8, 10, 12}:
        if day < 1 or day > 31:
            return False
    elif month in {4, 6, 9, 11}:
        if day < 1 or day > 30:
            return False
    elif month == 2:
        if day < 1 or day > 29:
            return False

    return True

class DateValidatorApp:
    '''
    The DateValidatorApp class sets up the tkinter GUI for the date validation application.
    '''
    def __init__(self, root):
        self.root = root
        self.root.title("Date Validator")

        self.label = tk.Label(root, text="Enter date (mm-dd-yyyy):")
        self.label.pack()

        self.entry = tk.Entry(root)
        self.entry.pack()

        self.validate_button = tk.Button(root, text="Validate", command=self.validate_date)
        self.validate_button.pack()

    def validate_date(self):
        date = self.entry.get()
        if valid_date(date):
            messagebox.showinfo("Result", "The date is valid.")
        else:
            messagebox.showerror("Result", "The date is invalid.")

if __name__ == "__main__":
    root = tk.Tk()
    app = DateValidatorApp(root)
    root.mainloop()

    '''
    Validates a given date string and returns True if the date is valid otherwise False.
    The date is valid if all of the following rules are satisfied:
    1. The date string is not empty.
    2. The number of days is not less than 1 or higher than 31 days for months 1,3,5,7,8,10,12.
       And the number of days is not less than 1 or higher than 30 days for months 4,6,9,11.
       And, the number of days is not less than 1 or higher than 29 for the month 2.
    3. The months should not be less than 1 or higher than 12.
    4. The date should be in the format: mm-dd-yyyy
    :param date: The date string to validate.
    :return: True if the date is valid, False otherwise.
    '''
    if not date:
        return False
    parts = date.split('-')
    if len(parts) != 3:
        return False
    try:
        month, day, year = map(int, parts)
    except ValueError:
        return False
    if month < 1 or month > 12:
        return False
    if month in {1, 3, 5, 7, 8, 10, 12}:
        if day < 1 or day > 31:
            return False
    elif month in {4, 6, 9, 11}:
        if day < 1 or day > 30:
            return False
    elif month == 2:
        if day < 1 or day > 29:
            return False
    return True

    '''

    The date is valid if all of the following rules are satisfied:
    1. The date string is not empty.
    2. The number of days is not less than 1 or higher than 31 days for months 1,3,5,7,8,10,12.
       And the number of days is not less than 1 or higher than 30 days for months 4,6,9,11.
       And, the number of days is not less than 1 or higher than 29 for the month 2.
    3. The months should not be less than 1 or higher than 12.
    4. The date should be in the format: mm-dd-yyyy
    :param date: The date string to validate.
    :return: True if the date is valid, False otherwise.
    '''
    if not date:
        return False
    parts = date.split('-')
    if len(parts) != 3:
        return False
    try:
        month, day, year = map(int, parts)
    except ValueError:
        return False
    if month < 1 or month > 12:
        return False
    if month in {1, 3, 5, 7, 8, 10, 12}:
        if day < 1 or day > 31:
            return False
    elif month in {4, 6, 9, 11}:
        if day < 1 or day > 30:
            return False
    elif month == 2:
        if day < 1 or day > 29:
            return False
    return True

def valid_date(date):
    '''
    Validates a given date string and returns True if the date is valid otherwise False.
    The date is valid if all of the following rules are satisfied:
    1. The date string is not empty and does not contain spaces or invalid separators like '/'.
    2. The number of days is not less than 1 or higher than 31 days for months 1,3,5,7,8,10,12.
       And the number of days is not less than 1 or higher than 30 days for months 4,6,9,11.
       And, the number of days is not less than 1 or higher than 29 for the month 2 in a leap year,
       or 28 in a non-leap year.
    3. The months should not be less than 1 or higher than 12.
    4. The date should be in the format: mm-dd-yyyy
    :param date: The date string to validate.
    :return: True if the date is valid, False otherwise.
    '''
    if not date or ' ' in date or '/' in date:
        return False
    parts = date.split('-')
    if len(parts) != 3:
        return False
    try:
        month, day, year = map(int, parts)
    except ValueError:
        return False
    if month < 1 or month > 12:
        return False
    if day < 1:
        return False
    if month in {1, 3, 5, 7, 8, 10, 12}:
        if day > 31:
            return False
    elif month in {4, 6, 9, 11}:
        if day > 30:
            return False
    elif month == 2:
        # Check for leap year
        if (year % 4 == 0 and year % 100 != 0) or (year % 400 == 0):
            if day > 29:
                return False
        else:
            if day > 28:
                return False
    return True
class DateValidatorApp:
    '''
    The DateValidatorApp class sets up the tkinter GUI for the date validation application.
    '''

        self.root = root

        date = self.entry.get()
        if valid_date(date):
            messagebox.showinfo("Result", "The date is valid.")
        else:
            messagebox.showerror("Result", "The date is invalid.")

'''
This is the main file for the Date Validator application. It sets up the GUI using tkinter and includes the logic for validating date strings.
'''
import tkinter as tk
from tkinter import messagebox
def valid_date(date):
    '''
    Validates a given date string and returns True if the date is valid otherwise False.
    The date is valid if all of the following rules are satisfied:
    1. The date string is not empty and does not contain spaces or invalid separators like '/'.
    2. The number of days is not less than 1 or higher than 31 days for months 1,3,5,7,8,10,12.
       And the number of days is not less than 1 or higher than 30 days for months 4,6,9,11.
       And, the number of days is not less than 1 or higher than 29 for the month 2 in a leap year,
       or 28 in a non-leap year.
    3. The months should not be less than 1 or higher than 12.
    4. The date should be in the format: mm-dd-yyyy
    :param date: The date string to validate.
    :return: True if the date is valid, False otherwise.
    '''
    if not date or ' ' in date or '/' in date:
        return False
    parts = date.split('-')
    if len(parts) != 3:
        return False
    try:
        month, day, year = map(int, parts)
    except ValueError:
        return False
    if year < 0:
        return False
    if month < 1 or month > 12:
        return False
    if day < 1:
        return False
    if month in {1, 3, 5, 7, 8, 10, 12}:
        if day > 31:
            return False
    elif month in {4, 6, 9, 11}:
        if day > 30:
            return False
    elif month == 2:
        # Check for leap year
        if (year % 4 == 0 and year % 100 != 0) or (year % 400 == 0):
            if day > 29:
                return False
        else:
            if day > 28:
                return False
    return True
class DateValidatorApp:
    '''
    The DateValidatorApp class sets up the tkinter GUI for the date validation application.
    '''
    def __init__(self, root):
        self.root = root
        self.root.title("Date Validator")
        self.label = tk.Label(root, text="Enter date (mm-dd-yyyy):")
        self.label.pack()
        self.entry = tk.Entry(root)
        self.entry.pack()
        self.validate_button = tk.Button(root, text="Validate", command=self.validate_date)
        self.validate_button.pack()
    def validate_date(self):
        date = self.entry.get()
        if valid_date(date):
            messagebox.showinfo("Result", "The date is valid.")
        else:
            messagebox.showerror("Result", "The date is invalid.")
if __name__ == "__main__":
    root = tk.Tk()
    app = DateValidatorApp(root)
    root.mainloop()

'''
This is the main file for the Date Validator application. It sets up the GUI using tkinter and includes the logic for validating date strings.
'''
import tkinter as tk
from tkinter import messagebox
def valid_date(date):
    '''
    Validates a given date string and returns True if the date is valid otherwise False.
    The date is valid if all of the following rules are satisfied:
    1. The date string is not empty and does not contain spaces or invalid separators like '/'.
    2. The number of days is not less than 1 or higher than 31 days for months 1,3,5,7,8,10,12.
       And the number of days is not less than 1 or higher than 30 days for months 4,6,9,11.
       And, the number of days is not less than 1 or higher than 29 for the month 2 in a leap year,
       or 28 in a non-leap year.
    3. The months should not be less than 1 or higher than 12.
    4. The date should be in the format: mm-dd-yyyy
    :param date: The date string to validate.
    :return: True if the date is valid, False otherwise.
    '''
    if not date or ' ' in date or '/' in date:
        return False
    parts = date.split('-')
    if len(parts) != 3:
        return False
    try:
        month, day, year = map(int, parts)
    except ValueError:
        return False
    if year < 0:
        return False
    if month < 1 or month > 12:
        return False
    if day < 1:
        return False
    if month in {1, 3, 5, 7, 8, 10, 12}:
        if day > 31:
            return False
    elif month in {4, 6, 9, 11}:
        if day > 30:
            return False
    elif month == 2:
        # Check for leap year
        if (year % 4 == 0 and year % 100 != 0) or (year % 400 == 0):
            if day > 29:
                return False
        else:
            if day > 28:
                return False
    return True
class DateValidatorApp:
    '''
    The DateValidatorApp class sets up the tkinter GUI for the date validation application.
    '''
    def __init__(self, root):
        self.root = root
        self.root.title("Date Validator")
        self.label = tk.Label(root, text="Enter date (mm-dd-yyyy):")
        self.label.pack()
        self.entry = tk.Entry(root)
        self.entry.pack()
        self.validate_button = tk.Button(root, text="Validate", command=self.validate_date)
        self.validate_button.pack()
    def validate_date(self):
        date = self.entry.get()
        if valid_date(date):
            messagebox.showinfo("Result", "The date is valid.")
        else:
            messagebox.showerror("Result", "The date is invalid.")
if __name__ == "__main__":
    root = tk.Tk()
    app = DateValidatorApp(root)
    root.mainloop()

'''
This is the main file for the Date Validator application. It sets up the GUI using tkinter and includes the logic for validating date strings.
'''
import tkinter as tk
from tkinter import messagebox
def valid_date(date):
    '''
    Validates a given date string and returns True if the date is valid otherwise False.
    The date is valid if all of the following rules are satisfied:
    1. The date string is not empty and does not contain spaces or invalid separators like '/'.
    2. The number of days is not less than 1 or higher than 31 days for months 1,3,5,7,8,10,12.
       And the number of days is not less than 1 or higher than 30 days for months 4,6,9,11.
       And, the number of days is not less than 1 or higher than 29 for the month 2 in a leap year,
       or 28 in a non-leap year.
    3. The months should not be less than 1 or higher than 12.
    4. The date should be in the format: mm-dd-yyyy
    :param date: The date string to validate.
    :return: True if the date is valid, False otherwise.
    '''
    if not date or ' ' in date or '/' in date:
        return False
    parts = date.split('-')
    if len(parts) != 3:
        return False
    try:
        month, day, year = map(int, parts)
    except ValueError:
        return False
    if year < 0:
        return False
    if month < 1 or month > 12:
        return False
    if day < 1:
        return False
    if month in {1, 3, 5, 7, 8, 10, 12}:
        if day > 31:
            return False
    elif month in {4, 6, 9, 11}:
        if day > 30:
            return False
    elif month == 2:
        # Check for leap year
        if (year % 4 == 0 and year % 100 != 0) or (year % 400 == 0):
            if day > 29:
                return False
        else:
            if day > 28:
                return False
    return True
class DateValidatorApp:
    '''
    The DateValidatorApp class sets up the tkinter GUI for the date validation application.
    '''
    def __init__(self, root):
        self.root = root
        self.root.title("Date Validator")
        self.label = tk.Label(root, text="Enter date (mm-dd-yyyy):")
        self.label.pack()
        self.entry = tk.Entry(root)
        self.entry.pack()
        self.validate_button = tk.Button(root, text="Validate", command=self.validate_date)
        self.validate_button.pack()
    def validate_date(self):
        date = self.entry.get()
        if valid_date(date):
            messagebox.showinfo("Result", "The date is valid.")
        else:
            messagebox.showerror("Result", "The date is invalid.")
if __name__ == "__main__":
    root = tk.Tk()
    app = DateValidatorApp(root)
    root.mainloop()
