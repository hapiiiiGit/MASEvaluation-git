import tkinter as tk
from tkinter import messagebox
from utils import is_sorted

class SortedCheckerApp:
    def __init__(self, root):
        self.root = root
        self.root.title("Sorted Checker")
        
        self.label = tk.Label(root, text="Enter a list of numbers separated by commas:")
        self.label.pack(pady=10)
        
        self.entry = tk.Entry(root, width=50)
        self.entry.pack(pady=10)
        
        self.check_button = tk.Button(root, text="Check", command=self.check_sorted)
        self.check_button.pack(pady=10)

        user_input = self.entry.get()

            if result:

                messagebox.showinfo("Result", "The list is not sorted correctly.")

if __name__ == "__main__":

'''

def is_sorted(lst):
    '''
    Given a list of numbers, return whether or not they are sorted
    in ascending order. If list has more than 1 duplicate of the same
    number, return False. Assume no negative numbers and only integers.

    Examples
    is_sorted([5]) ➞ True
    is_sorted([1, 2, 3, 4, 5]) ➞ True
    is_sorted([1, 3, 2, 4, 5]) ➞ False
    is_sorted([1, 2, 3, 4, 5, 6]) ➞ True
    is_sorted([1, 2, 3, 4, 5, 6, 7]) ➞ True
    is_sorted([1, 3, 2, 4, 5, 6, 7]) ➞ False
    is_sorted([1, 2, 2, 3, 3, 4]) ➞ True
    is_sorted([1, 2, 2, 2, 3, 4]) ➞ False
    '''
    if len(lst) <= 1:
        return True
    
    for i in range(len(lst) - 1):
        if lst[i] > lst[i + 1]:
            return False
    
    counts = {}
    for num in lst:
        if num in counts:
            counts[num] += 1
            if counts[num] > 2:
                return False
        else:
            counts[num] = 1
    
    return True

import tkinter as tk
from tkinter import messagebox
from utils import is_sorted
class SortedCheckerApp:
    def __init__(self, root):
        self.root = root
        self.root.title("Sorted Checker")
        self.label = tk.Label(root, text="Enter a list of numbers separated by commas:")
        self.label.pack(pady=10)
        self.entry = tk.Entry(root, width=50)
        self.entry.pack(pady=10)
        self.check_button = tk.Button(root, text="Check", command=self.check_sorted)
        self.check_button.pack(pady=10)
    def check_sorted(self):
        user_input = self.entry.get()
        try:
            num_list = list(map(int, user_input.split(',')))
            result = is_sorted(num_list)
            if result:
                messagebox.showinfo("Result", "The list is sorted correctly.")
            else:
                messagebox.showinfo("Result", "The list is not sorted correctly.")
        except ValueError:
            messagebox.showerror("Error", "Please enter a valid list of integers.")
if __name__ == "__main__":
    root = tk.Tk()
    app = SortedCheckerApp(root)
    root.mainloop()

def is_sorted(lst):
    '''
    Given a list of numbers, return whether or not they are sorted
    in ascending order. If list has more than 1 duplicate of the same
    number, return False. Assume no negative numbers and only integers.
    Examples
    is_sorted([5]) ➞ True
    is_sorted([1, 2, 3, 4, 5]) ➞ True
    is_sorted([1, 3, 2, 4, 5]) ➞ False
    is_sorted([1, 2, 3, 4, 5, 6]) ➞ True
    is_sorted([1, 2, 3, 4, 5, 6, 7]) ➞ True
    is_sorted([1, 3, 2, 4, 5, 6, 7]) ➞ False
    is_sorted([1, 2, 2, 3, 3, 4]) ➞ True
    is_sorted([1, 2, 2, 2, 3, 4]) ➞ False
    '''
    if len(lst) <= 1:
        return True
    for i in range(len(lst) - 1):
        if lst[i] > lst[i + 1]:
            return False
    counts = {}
    for num in lst:
        if num in counts:
            counts[num] += 1
            if counts[num] > 2:
                return False
        else:
            counts[num] = 1
    return True

def is_sorted(lst):
    '''
    Given a list of numbers, return whether or not they are sorted
    in ascending order. If list has more than 1 duplicate of the same
    number, return False. Assume no negative numbers and only integers.
    Examples
    is_sorted([5]) ➞ True
    is_sorted([1, 2, 3, 4, 5]) ➞ True
    is_sorted([1, 3, 2, 4, 5]) ➞ False
    is_sorted([1, 2, 3, 4, 5, 6]) ➞ True
    is_sorted([1, 2, 3, 4, 5, 6, 7]) ➞ True
    is_sorted([1, 3, 2, 4, 5, 6, 7]) ➞ False
    is_sorted([1, 2, 2, 3, 3, 4]) ➞ True
    is_sorted([1, 2, 2, 2, 3, 4]) ➞ False
    '''
    if len(lst) <= 1:
        return True
    for i in range(len(lst) - 1):
        if lst[i] > lst[i + 1]:
            return False
    counts = {}
    for num in lst:
        if num in counts:
            counts[num] += 1
            if counts[num] > 2:
                return False
        else:
            counts[num] = 1
    return True

def is_sorted(lst):
    '''
    Given a list of numbers, return whether or not they are sorted
    in ascending order. If list has more than 1 duplicate of the same

    '''
    if len(lst) <= 1:

import tkinter as tk

'''

def is_sorted(lst):
    '''
    Given a list of numbers, return whether or not they are sorted
    in ascending order. If list has more than 1 duplicate of the same
    number, return False. Assume no negative numbers and only integers.
    '''
    if len(lst) <= 1:
        return True
    for i in range(len(lst) - 1):
        if lst[i] > lst[i + 1]:
            return False
    counts = {}
    for num in lst:
        if num in counts:
            counts[num] += 1
            if counts[num] > 1:  # Corrected to disallow more than one duplicate
                return False
        else:
            counts[num] = 1
    return True

'''
Utility functions for SortedCheckerApp.

def is_sorted(lst):
    '''
    Given a list of numbers, return whether or not they are sorted
    in ascending order. If list has more than 1 duplicate of the same
    number, return False. Assume no negative numbers and only integers.
    '''
    if len(lst) <= 1:
        return True
    for i in range(len(lst) - 1):
        if lst[i] > lst[i + 1]:
            return False
    counts = {}
    for num in lst:
        if num in counts:
            counts[num] += 1
            if counts[num] > 1:  # Corrected to disallow more than one duplicate
                return False
        else:
            counts[num] = 1
    return True
