'''

This is the main application file for the Strongest Extension Finder.
It uses tkinter for the GUI and defines the logic to find the strongest extension.

'''

import tkinter as tk
from tkinter import messagebox

def Strongest_Extension(class_name, extensions):
    """
    Computes the strongest extension based on the number of uppercase and lowercase letters.
    Returns the class name concatenated with the strongest extension.
    """
    max_strength = None
    strongest_extension = None
    
    for extension in extensions:
        cap_count = sum(1 for c in extension if c.isupper())
        sm_count = sum(1 for c in extension if c.islower())
        strength = cap_count - sm_count
        
        if max_strength is None or strength > max_strength:
            max_strength = strength
            strongest_extension = extension
    
    return f"{class_name}.{strongest_extension}"

class App:
    def __init__(self, root):
        self.root = root
        self.root.title("Strongest Extension Finder")
        
        # Class Name Entry
        self.class_label = tk.Label(root, text="Class Name:")
        self.class_label.pack()
        self.class_entry = tk.Entry(root)
        self.class_entry.pack()
        
        # Extensions Entry
        self.extensions_label = tk.Label(root, text="Extensions (comma-separated):")
        self.extensions_label.pack()
        self.extensions_entry = tk.Entry(root)
        self.extensions_entry.pack()
        
        # Compute Button
        self.compute_button = tk.Button(root, text="Find Strongest Extension", command=self.compute_strongest_extension)
        self.compute_button.pack()
        
        # Result Label
        self.result_label = tk.Label(root, text="")
        self.result_label.pack()
    
    def compute_strongest_extension(self):
        class_name = self.class_entry.get().strip()
        extensions_input = self.extensions_entry.get().strip()
        
        if not class_name or not extensions_input:
            messagebox.showerror("Input Error", "Please enter both class name and extensions.")
            return
        
        extensions = [ext.strip() for ext in extensions_input.split(',')]
        
        if not extensions:
            messagebox.showerror("Input Error", "Please enter valid extensions.")
            return
        
        result = Strongest_Extension(class_name, extensions)
        self.result_label.config(text=f"Strongest Extension: {result}")

if __name__ == "__main__":
    root = tk.Tk()
    app = App(root)
    root.mainloop()
```

import tkinter as tk
from tkinter import messagebox
def Strongest_Extension(class_name, extensions):
    """
    Computes the strongest extension based on the number of uppercase and lowercase letters.
    Returns the class name concatenated with the strongest extension.
    If two extensions have the same strength, the first one in the list is chosen.
    """
    max_strength = None
    strongest_extension = None
    for extension in extensions:
        cap_count = sum(1 for c in extension if c.isupper())
        sm_count = sum(1 for c in extension if c.islower())
        strength = cap_count - sm_count
        # Ensure the first extension with the highest strength is chosen
        if max_strength is None or strength > max_strength:
            max_strength = strength
            strongest_extension = extension
    return f"{class_name}.{strongest_extension}"

    def __init__(self, root):

        self.extensions_entry.pack()
        # Compute Button
        self.compute_button = tk.Button(root, text="Find Strongest Extension", command=self.compute_strongest_extension)
        self.compute_button.pack()
        # Result Label
        self.result_label = tk.Label(root, text="")
        self.result_label.pack()
    def compute_strongest_extension(self):
        class_name = self.class_entry.get().strip()
        extensions_input = self.extensions_entry.get().strip()
        if not class_name or not extensions_input:
            messagebox.showerror("Input Error", "Please enter both class name and extensions.")
            return
        extensions = [ext.strip() for ext in extensions_input.split(',')]
        if not extensions:
            messagebox.showerror("Input Error", "Please enter valid extensions.")
            return
        result = Strongest_Extension(class_name, extensions)
        self.result_label.config(text=f"Strongest Extension: {result}")

import tkinter as tk
from tkinter import messagebox
def Strongest_Extension(class_name, extensions):
    """
    Computes the strongest extension based on the number of uppercase and lowercase letters.
    Returns the class name concatenated with the strongest extension.
    If two extensions have the same strength, the first one in the list is chosen.
    """
    max_strength = None
    strongest_extension = None
    for extension in extensions:
        cap_count = sum(1 for c in extension if c.isupper())
        sm_count = sum(1 for c in extension if c.islower())
        strength = cap_count - sm_count
        # Ensure the first extension with the highest strength is chosen
        if max_strength is None or strength > max_strength:
            max_strength = strength
            strongest_extension = extension
    return f"{class_name}.{strongest_extension}"

    def compute_strongest_extension(self):
        class_name = self.class_entry.get().strip()

        if not extensions:

import tkinter as tk
from tkinter import messagebox
def Strongest_Extension(class_name, extensions):
    """
    Computes the strongest extension based on the number of uppercase and lowercase letters.
    Returns the class name concatenated with the strongest extension.
    If two extensions have the same strength, the first one in the list is chosen.
    """
    max_strength = None
    strongest_extension = None
    for extension in extensions:
        cap_count = sum(1 for c in extension if c.isupper())
        sm_count = sum(1 for c in extension if c.islower())
        strength = cap_count - sm_count
        # Ensure the first extension with the highest strength is chosen
        # If two extensions have the same strength, the first one in the list is chosen
        if max_strength is None or strength > max_strength:
            max_strength = strength
            strongest_extension = extension
    return f"{class_name}.{strongest_extension}"

        class_name = self.class_entry.get().strip()

        extensions = [ext.strip() for ext in extensions_input.split(',')]
        if not extensions:
            messagebox.showerror("Input Error", "Please enter valid extensions.")

        self.result_label.config(text=f"Strongest Extension: {result}")
