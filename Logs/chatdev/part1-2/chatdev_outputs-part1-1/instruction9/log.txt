```python
'''

This module contains the main application class and logic for counting numbers with a sum of digits greater than zero.

'''

import tkinter as tk
from tkinter import messagebox

class App:
    def __init__(self, root):
        '''
        Initialize the main application with GUI components.
        
        :param root: The root window of the tkinter application.
        '''
        self.root = root
        self.root.title("Count Numbers with Positive Digit Sum")
        
        self.label = tk.Label(root, text="Enter a list of integers separated by commas:")
        self.label.pack()
        
        self.entry = tk.Entry(root, width=50)
        self.entry.pack()
        
        self.submit_button = tk.Button(root, text="Submit", command=self.on_submit)
        self.submit_button.pack()
        
        self.result_label = tk.Label(root, text="")
        self.result_label.pack()
    
    def count_nums(self, arr):
        '''
        Count the number of elements in the array with a sum of digits > 0.
        
        :param arr: List of integers.
        :return: Integer count of numbers with a positive digit sum.
        '''
        count = 0
        for num in arr:
            # Convert number to string and handle negative sign
            digits = str(num)
            if digits[0] == '-':
                digit_sum = -int(digits[1]) + sum(int(d) for d in digits[2:])
            else:
                digit_sum = sum(int(d) for d in digits)
            
            if digit_sum > 0:
                count += 1
        return count
    
    def on_submit(self):
        '''
        Handle the submit button click event, process input, and display result.
        '''
        try:
            input_text = self.entry.get()
            # Convert input string to a list of integers
            arr = list(map(int, input_text.split(',')))
            result = self.count_nums(arr)
            self.result_label.config(text=f"Count: {result}")
        except ValueError:
            messagebox.showerror("Invalid Input", "Please enter a valid list of integers.")

if __name__ == "__main__":
    root = tk.Tk()
    app = App(root)
    root.mainloop()
```

import tkinter as tk
from tkinter import messagebox
class App:
    def __init__(self, root):
        '''
        Initialize the main application with GUI components.
        :param root: The root window of the tkinter application.
        '''
        self.root = root
        self.root.title("Count Numbers with Positive Digit Sum")
        self.label = tk.Label(root, text="Enter a list of integers separated by commas:")
        self.label.pack()
        self.entry = tk.Entry(root, width=50)
        self.entry.pack()
        self.submit_button = tk.Button(root, text="Submit", command=self.on_submit)
        self.submit_button.pack()
        self.result_label = tk.Label(root, text="")
        self.result_label.pack()
    def count_nums(self, arr):
        '''
        Count the number of elements in the array with a sum of digits > 0.
        :param arr: List of integers.
        :return: Integer count of numbers with a positive digit sum.
        '''
        count = 0
        for num in arr:
            # Convert number to string and handle negative sign
            digits = str(num)
            if digits[0] == '-':
                digit_sum = -int(digits[1]) + sum(int(d) for d in digits[2:])
            else:
                digit_sum = sum(int(d) for d in digits)
            if digit_sum > 0:
                count += 1
        return count

        '''
        Handle the submit button click event, process input, and display result.
        '''
        try:
            input_text = self.entry.get()
            # Convert input string to a list of integers
            arr = list(map(int, input_text.split(',')))
            result = self.count_nums(arr)
            self.result_label.config(text=f"Count: {result}")
        except ValueError:
            messagebox.showerror("Invalid Input", "Please enter a valid list of integers.")

'''
This module contains the main application class and logic for counting numbers with a sum of digits greater than zero.

import tkinter as tk
from tkinter import messagebox
class App:
    def __init__(self, root):
        '''
        Initialize the main application with GUI components.
        :param root: The root window of the tkinter application.
        '''
        self.root = root
        self.root.title("Count Numbers with Positive Digit Sum")
        self.label = tk.Label(root, text="Enter a list of integers separated by commas:")
        self.label.pack()
        self.entry = tk.Entry(root, width=50)
        self.entry.pack()
        self.submit_button = tk.Button(root, text="Submit", command=self.on_submit)
        self.submit_button.pack()
        self.result_label = tk.Label(root, text="")
        self.result_label.pack()
    def count_nums(self, arr):
        '''
        Count the number of elements in the array with a sum of digits > 0.
        :param arr: List of integers.
        :return: Integer count of numbers with a positive digit sum.
        '''
        count = 0
        for num in arr:
            # Convert number to string and handle negative sign
            digits = str(num)
            if digits[0] == '-':
                digit_sum = -int(digits[1]) + sum(int(d) for d in digits[2:])
            else:
                digit_sum = sum(int(d) for d in digits)
            if digit_sum > 0:
                count += 1
        return count
    def on_submit(self):
        '''
        Handle the submit button click event, process input, and display result.
        '''
        try:
            input_text = self.entry.get()
            # Convert input string to a list of integers
            arr = list(map(int, input_text.split(',')))
            result = self.count_nums(arr)
            self.result_label.config(text=f"Count: {result}")
        except ValueError:
            messagebox.showerror("Invalid Input", "Please enter a valid list of integers.")
if __name__ == "__main__":
    root = tk.Tk()
    app = App(root)
    root.mainloop()
