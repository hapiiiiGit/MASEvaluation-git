'''

This file contains the main application code for the Bracket Checker App.
It includes a GUI for user interaction and a function to check nested brackets.

'''

import tkinter as tk
from tkinter import messagebox

def is_nested(string):
    '''
    Create a function that takes a string as input which contains only square brackets.
    The function should return True if and only if there is a valid subsequence of brackets
    where at least one bracket in the subsequence is nested.

    Examples:
    is_nested('[[]]') ➞ True
    is_nested('[]]]]]]][[[[[]') ➞ False
    is_nested('[][]') ➞ False
    is_nested('[]') ➞ False
    is_nested('[[][]]') ➞ True
    is_nested('[[]][[') ➞ True
    '''
    stack = []
    for char in string:
        if char == '[':
            stack.append(char)
        elif char == ']':
            if stack and stack[-1] == '[':
                stack.pop()
                if stack:
                    return True
            else:
                return False
    return False

class BracketCheckerApp:
    def __init__(self, master):
        self.master = master
        master.title("Bracket Checker App")

        self.label = tk.Label(master, text="Enter a string of brackets:")
        self.label.pack()

        self.entry = tk.Entry(master)
        self.entry.pack()

        self.check_button = tk.Button(master, text="Check", command=self.check_brackets)
        self.check_button.pack()

        self.result_label = tk.Label(master, text="")
        self.result_label.pack()

    def check_brackets(self):
        input_string = self.entry.get()
        if is_nested(input_string):
            result = "The string contains a nested subsequence."
        else:
            result = "The string does not contain a nested subsequence."
        self.result_label.config(text=result)
        messagebox.showinfo("Result", result)

if __name__ == "__main__":
    root = tk.Tk()
    app = BracketCheckerApp(root)
    root.mainloop()
```

import tkinter as tk
from tkinter import messagebox

def is_nested(string):
    '''
    Create a function that takes a string as input which contains only square brackets.
    The function should return True if and only if there is a valid subsequence of brackets
    where at least one bracket in the subsequence is nested.
    '''
    stack = []
    for char in string:
        if char == '[':
            stack.append(char)
        elif char == ']':
            if stack and stack[-1] == '[':
                stack.pop()
                if stack:  # Check if there are still open brackets after a valid pair
                    return True
            else:
                return False
    return False

        self.result_label.pack()

'''
This file contains the main application code for the Bracket Checker App.
It includes a GUI for user interaction and a function to check nested brackets.
'''
import tkinter as tk
from tkinter import messagebox

def is_nested(string):
    '''
    Create a function that takes a string as input which contains only square brackets.
    The function should return True if and only if there is a valid subsequence of brackets
    where at least one bracket in the subsequence is nested.
    '''
    stack = []
    for char in string:
        if char == '[':
            stack.append(char)
        elif char == ']':
            if stack:
                stack.pop()
                if stack:  # If there are still open brackets, it means nesting occurred
                    return True
            else:
                return False
    return False

class BracketCheckerApp:
    def __init__(self, master):
        '''
        Initialize the Bracket Checker App with a GUI for user interaction.
        '''
        self.master = master
        master.title("Bracket Checker App")
        self.label = tk.Label(master, text="Enter a string of brackets:")
        self.label.pack()
        self.entry = tk.Entry(master)
        self.entry.pack()
        self.check_button = tk.Button(master, text="Check", command=self.check_brackets)
        self.check_button.pack()
        self.result_label = tk.Label(master, text="")
        self.result_label.pack()

    def check_brackets(self):
        '''
        Check the input string for nested brackets and display the result.
        '''
        input_string = self.entry.get()
        if is_nested(input_string):
            result = "The string contains a nested subsequence."
        else:
            result = "The string does not contain a nested subsequence."
        self.result_label.config(text=result)
        messagebox.showinfo("Result", result)

if __name__ == "__main__":
    root = tk.Tk()
    app = BracketCheckerApp(root)
    root.mainloop()

'''
This file contains the main application code for the Bracket Checker App.
It includes a GUI for user interaction and a function to check nested brackets.

import tkinter as tk
from tkinter import messagebox
def is_nested(string):
    '''
    Create a function that takes a string as input which contains only square brackets.
    The function should return True if and only if there is a valid subsequence of brackets
    where at least one bracket in the subsequence is nested.
    '''
    stack = []
    for char in string:
        if char == '[':
            stack.append(char)
        elif char == ']':
            if stack:
                stack.pop()
                if stack:  # If there are still open brackets, it means nesting occurred
                    return True
            else:
                return False
    return False
class BracketCheckerApp:
    def __init__(self, master):
        '''
        Initialize the Bracket Checker App with a GUI for user interaction.
        '''
        self.master = master
        master.title("Bracket Checker App")
        self.label = tk.Label(master, text="Enter a string of brackets:")
        self.label.pack()
        self.entry = tk.Entry(master)
        self.entry.pack()
        self.check_button = tk.Button(master, text="Check", command=self.check_brackets)
        self.check_button.pack()
        self.result_label = tk.Label(master, text="")
        self.result_label.pack()
    def check_brackets(self):
        '''
        Check the input string for nested brackets and display the result.
        '''
        input_string = self.entry.get()
        if is_nested(input_string):
            result = "The string contains a nested subsequence."
        else:
            result = "The string does not contain a nested subsequence."
        self.result_label.config(text=result)
        messagebox.showinfo("Result", result)
if __name__ == "__main__":
    root = tk.Tk()

# Bracket Checker App

A simple application to check if a string of square brackets contains a valid subsequence where at least one bracket is nested.

## 🤔 What is this?

The Bracket Checker App is a Python-based application with a graphical user interface (GUI) that allows users to input a string of square brackets and determine if it contains a nested subsequence. This app is designed to be user-friendly and does not require any external dependencies beyond Python's standard library.

### Main Features:
- **Nested Bracket Checker**: Input a string of square brackets, and the app will determine if it contains a valid nested subsequence.
- **User-Friendly GUI**: The app provides an intuitive graphical interface for easy interaction.
- **Instant Results**: Get immediate feedback on whether the input string contains nested brackets.

---

## 🚀 Quick Install

### Prerequisites:
- Python 3.6 or higher installed on your system.

### Installation Steps:
1. Clone or download the project files to your local machine.
2. Navigate to the project directory.
3. Ensure you have Python 3.6 or higher installed. You can check your Python version by running:
   ```bash
   python --version
   ```
4. No additional dependencies are required as the app uses Python's standard library. However, you can create a virtual environment for the project (optional but recommended):
   ```bash
   python -m venv venv
   source venv/bin/activate  # On Windows, use `venv\Scripts\activate`
   ```
5. Install the required Python version if not already installed:
   ```bash
   pip install -r requirements.txt
   ```

---

## 🛠 How to Use the Bracket Checker App

### Running the Application:
1. Open a terminal or command prompt.
2. Navigate to the directory where the project files are located.
3. Run the application by executing the following command:
   ```bash
   python main.py
   ```

### Using the Application:
1. Once the application starts, a GUI window titled "Bracket Checker App" will appear.
2. Enter a string of square brackets (e.g., `[[]]`, `[][]`, `[]]]]]]][[[[[]`) into the input field.
3. Click the "Check" button to analyze the input string.
4. The result will be displayed below the input field and in a pop-up message box:
   - **"The string contains a nested subsequence."**: Indicates that the input string contains at least one nested bracket.
   - **"The string does not contain a nested subsequence."**: Indicates that the input string does not contain any nested brackets.

---

## 📖 Example Usage

Here are some examples of input strings and their expected results:

| Input String         | Result                                      |
|----------------------|---------------------------------------------|
| `[[]]`               | The string contains a nested subsequence.  |
| `[]]]]]]][[[[[]`     | The string does not contain a nested subsequence. |
| `[][]`               | The string does not contain a nested subsequence. |
| `[]`                 | The string does not contain a nested subsequence. |
| `[[][]]`             | The string contains a nested subsequence.  |
| `[[]][[`             | The string contains a nested subsequence.  |

---

## 🧩 How It Works

The application uses a stack-based approach to determine if a string contains nested brackets:
1. Iterate through each character in the input string.
2. Push opening brackets (`[`) onto a stack.
3. For closing brackets (`]`), check if the stack is not empty:
   - If the stack is not empty, pop the top element and check if there are still elements in the stack. If so, it indicates nesting.
   - If the stack is empty, it means the brackets are unbalanced, and the function returns `False`.
4. If the entire string is processed without finding a nested subsequence, the function returns `False`.

---

## 🛡 Troubleshooting

- **The app doesn't start**: Ensure you have Python 3.6 or higher installed and that you are running the `main.py` file from the correct directory.
- **Unexpected results**: Double-check your input string to ensure it contains only square brackets (`[` and `]`).
