'''
This file contains the main application logic for a GUI that calculates
the minimum sum of any non-empty sub-array of integers provided by the user.
'''
import tkinter as tk
from tkinter import messagebox
def min_sub_array_sum(nums):
    '''
    Given an array of integers nums, find the minimum sum of any non-empty sub-array of nums.
    Parameters:
    nums (list): A list of integers.
    Returns:
    int: The minimum sum of any non-empty sub-array.
    '''
    min_sum = float('inf')
    current_sum = 0
    for num in nums:
        current_sum += num
        if current_sum < min_sum:
            min_sum = current_sum
        if current_sum > 0:
            current_sum = 0
    return min_sum
class MainApp:
    def __init__(self, root):
        '''
        Initialize the main application with GUI components.
        Parameters:
        root (tk.Tk): The root window of the application.
        '''
        self.root = root
        self.root.title("Minimum Sub-array Sum Calculator")
        self.label = tk.Label(root, text="Enter integers separated by commas:")
        self.label.pack()
        self.entry = tk.Entry(root, width=50)
        self.entry.pack()
        self.calculate_button = tk.Button(root, text="Calculate", command=self.calculate_min_sum)
        self.calculate_button.pack()
        self.result_label = tk.Label(root, text="")
        self.result_label.pack()
    def calculate_min_sum(self):
        '''
        Retrieve user input, calculate the minimum sub-array sum, and display the result.
        '''
        try:
            nums = list(map(int, self.entry.get().split(',')))
            result = min_sub_array_sum(nums)
            self.result_label.config(text=f"Minimum Sub-array Sum: {result}")
        except ValueError:
            messagebox.showerror("Invalid input", "Please enter a valid list of integers separated by commas.")
if __name__ == "__main__":
    root = tk.Tk()
    app = MainApp(root)
    root.mainloop()
