```python
'''

Main application file for the exchange checker with GUI using tkinter.

'''

import tkinter as tk
from tkinter import messagebox

def exchange(lst1, lst2):
    '''
    Determines if it's possible to exchange elements between lst1 and lst2
    to make lst1 consist only of even numbers.
    
    Parameters:
    lst1 (list): First list of numbers.
    lst2 (list): Second list of numbers.
    
    Returns:
    str: "YES" if possible, otherwise "NO".
    '''
    # Check if all elements in lst1 are even
    if all(x % 2 == 0 for x in lst1):
        return "YES"
    
    # Check if there is at least one even number in lst2
    if any(x % 2 == 0 for x in lst2):
        return "YES"
    
    return "NO"

class ExchangeApp:
    '''
    GUI Application for checking the possibility of exchanging elements
    between two lists to make the first list consist only of even numbers.
    '''
    def __init__(self, master):
        self.master = master
        master.title("Exchange Checker")

        self.label1 = tk.Label(master, text="List 1 (comma-separated):")
        self.label1.pack()

        self.entry1 = tk.Entry(master)
        self.entry1.pack()

        self.label2 = tk.Label(master, text="List 2 (comma-separated):")
        self.label2.pack()

        self.entry2 = tk.Entry(master)
        self.entry2.pack()

        self.check_button = tk.Button(master, text="Check Exchange", command=self.check_exchange)
        self.check_button.pack()

        self.result_label = tk.Label(master, text="")
        self.result_label.pack()

    def check_exchange(self):
        try:
            lst1 = list(map(int, self.entry1.get().split(',')))
            lst2 = list(map(int, self.entry2.get().split(',')))
            result = exchange(lst1, lst2)
            self.result_label.config(text=f"Result: {result}")
        except ValueError:
            messagebox.showerror("Invalid input", "Please enter valid integers separated by commas.")

if __name__ == "__main__":
    root = tk.Tk()
    app = ExchangeApp(root)
    root.mainloop()
```

import tkinter as tk
from tkinter import messagebox

def exchange(lst1, lst2):
    '''
    Determines if it's possible to exchange elements between lst1 and lst2
    to make lst1 consist only of even numbers.
    Parameters:
    lst1 (list): First list of numbers.
    lst2 (list): Second list of numbers.
    Returns:
    str: "YES" if possible, otherwise "NO".
    '''
    # Count odd numbers in lst1
    odd_count_lst1 = sum(1 for x in lst1 if x % 2 != 0)
    # Count even numbers in lst2
    even_count_lst2 = sum(1 for x in lst2 if x % 2 == 0)
    # Check if there are enough even numbers in lst2 to replace all odd numbers in lst1
    if even_count_lst2 >= odd_count_lst1:
        return "YES"
    return "NO"

        self.entry1 = tk.Entry(master)

        self.label2 = tk.Label(master, text="List 2 (comma-separated):")

            lst2 = list(map(int, self.entry2.get().split(',')))

'''
Main application file for the exchange checker with GUI using tkinter.
'''
import tkinter as tk
from tkinter import messagebox
def exchange(lst1, lst2):
    '''
    Determines if it's possible to exchange elements between lst1 and lst2
    to make lst1 consist only of even numbers.
    Parameters:
    lst1 (list): First list of numbers.
    lst2 (list): Second list of numbers.
    Returns:
    str: "YES" if possible, otherwise "NO".
    '''
    # Count odd numbers in lst1
    odd_count_lst1 = sum(1 for x in lst1 if x % 2 != 0)
    # Count even numbers in lst2
    even_count_lst2 = sum(1 for x in lst2 if x % 2 == 0)
    # Check if there are enough even numbers in lst2 to replace all odd numbers in lst1
    if even_count_lst2 >= odd_count_lst1:
        return "YES"
    return "NO"
class ExchangeApp:
    '''
    GUI Application for checking the possibility of exchanging elements
    between two lists to make the first list consist only of even numbers.
    '''
    def __init__(self, master):
        self.master = master
        master.title("Exchange Checker")
        self.label1 = tk.Label(master, text="List 1 (comma-separated):")
        self.label1.pack()
        self.entry1 = tk.Entry(master)
        self.entry1.pack()
        self.label2 = tk.Label(master, text="List 2 (comma-separated):")
        self.label2.pack()
        self.entry2 = tk.Entry(master)
        self.entry2.pack()
        self.check_button = tk.Button(master, text="Check Exchange", command=self.check_exchange)
        self.check_button.pack()
        self.result_label = tk.Label(master, text="")
        self.result_label.pack()
    def check_exchange(self):
        try:
            lst1 = list(map(int, self.entry1.get().split(',')))
            lst2 = list(map(int, self.entry2.get().split(',')))
            result = exchange(lst1, lst2)
            self.result_label.config(text=f"Result: {result}")
        except ValueError:
            messagebox.showerror("Invalid input", "Please enter valid integers separated by commas.")
if __name__ == "__main__":
    root = tk.Tk()
    app = ExchangeApp(root)
    root.mainloop()

```python
'''
Main application file for the exchange checker with GUI using tkinter.
'''
import tkinter as tk
from tkinter import messagebox
def exchange(lst1, lst2):
    '''
    Determines if it's possible to exchange elements between lst1 and lst2
    to make lst1 consist only of even numbers.
    Parameters:
    lst1 (list): First list of numbers.
    lst2 (list): Second list of numbers.
    Returns:
    str: "YES" if possible, otherwise "NO".
    '''
    # Count odd numbers in lst1
    odd_count_lst1 = sum(1 for x in lst1 if x % 2 != 0)
    # Count even numbers in lst2
    even_count_lst2 = sum(1 for x in lst2 if x % 2 == 0)
    # Check if there are enough even numbers in lst2 to replace all odd numbers in lst1
    if even_count_lst2 >= odd_count_lst1:
        return "YES"
    return "NO"
class ExchangeApp:
    '''
    GUI Application for checking the possibility of exchanging elements
    between two lists to make the first list consist only of even numbers.
    '''
    def __init__(self, master):
        self.master = master
        master.title("Exchange Checker")
        self.label1 = tk.Label(master, text="List 1 (comma-separated):")
        self.label1.pack()
        self.entry1 = tk.Entry(master)
        self.entry1.pack()
        self.label2 = tk.Label(master, text="List 2 (comma-separated):")
        self.label2.pack()
        self.entry2 = tk.Entry(master)
        self.entry2.pack()
        self.check_button = tk.Button(master, text="Check Exchange", command=self.check_exchange)
        self.check_button.pack()
        self.result_label = tk.Label(master, text="")
        self.result_label.pack()
    def check_exchange(self):
        try:
            lst1 = list(map(int, self.entry1.get().split(',')))
            lst2 = list(map(int, self.entry2.get().split(',')))
            result = exchange(lst1, lst2)
            self.result_label.config(text=f"Result: {result}")
        except ValueError:
            messagebox.showerror("Invalid input", "Please enter valid integers separated by commas.")
if __name__ == "__main__":
    root = tk.Tk()
    app = ExchangeApp(root)
    root.mainloop()

'''
Main application file for the exchange checker with GUI using tkinter.
'''
import tkinter as tk
from tkinter import messagebox
def exchange(lst1, lst2):
    '''
    Determines if it's possible to exchange elements between lst1 and lst2
    to make lst1 consist only of even numbers.
    Parameters:
    lst1 (list): First list of numbers.
    lst2 (list): Second list of numbers.
    Returns:
    str: "YES" if possible, otherwise "NO".
    '''
    # Count odd numbers in lst1
    odd_count_lst1 = sum(1 for x in lst1 if x % 2 != 0)
    # Count even numbers in lst2
    even_count_lst2 = sum(1 for x in lst2 if x % 2 == 0)
    # Check if there are enough even numbers in lst2 to replace all odd numbers in lst1
    if even_count_lst2 >= odd_count_lst1:
        return "YES"
    return "NO"
class ExchangeApp:
    '''
    GUI Application for checking the possibility of exchanging elements
    between two lists to make the first list consist only of even numbers.
    '''
    def __init__(self, master):
        self.master = master
        master.title("Exchange Checker")
        self.label1 = tk.Label(master, text="List 1 (comma-separated):")
        self.label1.pack()
        self.entry1 = tk.Entry(master)
        self.entry1.pack()
        self.label2 = tk.Label(master, text="List 2 (comma-separated):")
        self.label2.pack()
        self.entry2 = tk.Entry(master)
        self.entry2.pack()
        self.check_button = tk.Button(master, text="Check Exchange", command=self.check_exchange)
        self.check_button.pack()
        self.result_label = tk.Label(master, text="")
        self.result_label.pack()
    def check_exchange(self):
        try:
            lst1 = list(map(int, self.entry1.get().split(',')))
            lst2 = list(map(int, self.entry2.get().split(',')))
            result = exchange(lst1, lst2)
            self.result_label.config(text=f"Result: {result}")
        except ValueError:
            messagebox.showerror("Invalid input", "Please enter valid integers separated by commas.")
if __name__ == "__main__":
    root = tk.Tk()
    app = ExchangeApp(root)
    root.mainloop()
