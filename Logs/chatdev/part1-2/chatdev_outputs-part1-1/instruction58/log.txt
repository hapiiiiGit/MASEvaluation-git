'''

This module contains the main application code for checking if three sides form a right-angled triangle using a GUI.

'''

import tkinter as tk
from tkinter import messagebox

class RightAngleTriangleChecker:
    '''
    This class provides a method to check if three sides can form a right-angled triangle.
    '''
    
    @staticmethod
    def is_right_angle_triangle(a, b, c):
        '''
        Determine if the sides form a right-angled triangle using the Pythagorean theorem.
        
        :param a: Length of the first side
        :param b: Length of the second side
        :param c: Length of the third side
        :return: True if the sides form a right-angled triangle, False otherwise
        '''
        sides = sorted([a, b, c])
        return sides[0]**2 + sides[1]**2 == sides[2]**2

class TriangleApp:
    '''
    This class creates the GUI for the application.
    '''
    
    def __init__(self, master):
        '''
        Initialize the GUI components.
        
        :param master: The root window
        '''
        self.master = master
        master.title("Right-Angle Triangle Checker")
        
        self.label = tk.Label(master, text="Enter the lengths of the three sides:")
        self.label.pack()
        
        self.side1_entry = tk.Entry(master)
        self.side1_entry.pack()
        
        self.side2_entry = tk.Entry(master)
        self.side2_entry.pack()
        
        self.side3_entry = tk.Entry(master)
        self.side3_entry.pack()
        
        self.check_button = tk.Button(master, text="Check", command=self.check_triangle)
        self.check_button.pack()
        
    def check_triangle(self):
        '''
        Retrieve input values, use RightAngleTriangleChecker to check the triangle, and display the result.
        '''
        try:
            a = float(self.side1_entry.get())
            b = float(self.side2_entry.get())
            c = float(self.side3_entry.get())
            
            if RightAngleTriangleChecker.is_right_angle_triangle(a, b, c):
                messagebox.showinfo("Result", "The sides form a right-angled triangle.")
            else:
                messagebox.showinfo("Result", "The sides do not form a right-angled triangle.")
        except ValueError:
            messagebox.showerror("Invalid input", "Please enter valid numbers for the sides.")

if __name__ == "__main__":
    root = tk.Tk()
    app = TriangleApp(root)
    root.mainloop()

import tkinter as tk
from tkinter import messagebox
class RightAngleTriangleChecker:
    '''
    This class provides a method to check if three sides can form a right-angled triangle.
    '''
    @staticmethod
    def is_right_angle_triangle(a, b, c):
        '''
        Determine if the sides form a right-angled triangle using the Pythagorean theorem.
        :param a: Length of the first side
        :param b: Length of the second side
        :param c: Length of the third side
        :return: True if the sides form a right-angled triangle, False otherwise
        '''
        sides = sorted([a, b, c])
        return sides[0]**2 + sides[1]**2 == sides[2]**2

import tkinter as tk

class RightAngleTriangleChecker:
    '''
    This class provides a method to check if three sides can form a right-angled triangle.
    '''
    @staticmethod
    def is_right_angle_triangle(a, b, c):
        '''
        Determine if the sides form a right-angled triangle using the Pythagorean theorem.

'''
This module contains the main application code for checking if three sides form a right-angled triangle using a GUI.
'''
import tkinter as tk
from tkinter import messagebox
class RightAngleTriangleChecker:
    '''
    This class provides a method to check if three sides can form a right-angled triangle.
    '''
    @staticmethod
    def is_right_angle_triangle(a, b, c):
        '''
        Determine if the sides form a right-angled triangle using the Pythagorean theorem.
        :param a: Length of the first side
        :param b: Length of the second side
        :param c: Length of the third side
        :return: True if the sides form a right-angled triangle, False otherwise
        '''
        sides = sorted([a, b, c])
        return sides[0]**2 + sides[1]**2 == sides[2]**2
class TriangleApp:
    '''
    This class creates the GUI for the application.
    '''
    def __init__(self, master):
        '''
        Initialize the GUI components.
        :param master: The root window
        '''
        self.master = master
        master.title("Right-Angle Triangle Checker")
        self.label = tk.Label(master, text="Enter the lengths of the three sides:")
        self.label.pack()
        self.side1_entry = tk.Entry(master)
        self.side1_entry.pack()
        self.side2_entry = tk.Entry(master)
        self.side2_entry.pack()
        self.side3_entry = tk.Entry(master)
        self.side3_entry.pack()
        self.check_button = tk.Button(master, text="Check", command=self.check_triangle)
        self.check_button.pack()
    def check_triangle(self):
        '''
        Retrieve input values, use RightAngleTriangleChecker to check the triangle, and display the result.
        '''
        try:
            a = float(self.side1_entry.get())
            b = float(self.side2_entry.get())
            c = float(self.side3_entry.get())
            if RightAngleTriangleChecker.is_right_angle_triangle(a, b, c):
                messagebox.showinfo("Result", "The sides form a right-angled triangle.")
            else:
                messagebox.showinfo("Result", "The sides do not form a right-angled triangle.")
        except ValueError:
            messagebox.showerror("Invalid input", "Please enter valid numbers for the sides.")
if __name__ == "__main__":
    root = tk.Tk()
    app = TriangleApp(root)
    root.mainloop()
