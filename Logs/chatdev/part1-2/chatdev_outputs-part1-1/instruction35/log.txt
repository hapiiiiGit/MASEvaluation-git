'''

This file contains the main application logic and GUI setup using tkinter.
It includes a function to check if the last character of a string is an alphabetical character
and a class to manage the GUI.

'''

import tkinter as tk
from tkinter import messagebox

def check_if_last_char_is_a_letter(txt):
    '''
    Create a function that returns True if the last character
    of a given string is an alphabetical character and is not
    a part of a word, and False otherwise.
    Note: "word" is a group of characters separated by space.

    Examples:
    check_if_last_char_is_a_letter("apple pie") ➞ False
    check_if_last_char_is_a_letter("apple pi e") ➞ True
    check_if_last_char_is_a_letter("apple pi e ") ➞ False
    check_if_last_char_is_a_letter("") ➞ False
    '''
    txt = txt.strip()
    if not txt:
        return False
    last_char = txt[-1]
    if last_char.isalpha():
        return len(txt.split()[-1]) == 1
    return False

class App:
    def __init__(self, root):
        self.root = root
        self.root.title("Last Character Checker")

        self.label = tk.Label(root, text="Enter a string:")
        self.label.pack()

        self.entry = tk.Entry(root, width=50)
        self.entry.pack()

        self.check_button = tk.Button(root, text="Check", command=self.check_last_char)
        self.check_button.pack()

        self.result_label = tk.Label(root, text="")
        self.result_label.pack()

    def check_last_char(self):
        user_input = self.entry.get()
        result = check_if_last_char_is_a_letter(user_input)
        if result:
            self.result_label.config(text="The last character is a standalone letter.")
        else:
            self.result_label.config(text="The last character is NOT a standalone letter.")

if __name__ == "__main__":
    root = tk.Tk()
    app = App(root)
    root.mainloop()
```

    Create a function that returns True if the last character

import tkinter as tk
from tkinter import messagebox
def check_if_last_char_is_a_letter(txt):
    '''
    Create a function that returns True if the last character
    of a given string is an alphabetical character and is not
    a part of a word, and False otherwise.
    Note: "word" is a group of characters separated by space.
    Examples:
    check_if_last_char_is_a_letter("apple pie") ➞ False
    check_if_last_char_is_a_letter("apple pi e") ➞ True
    check_if_last_char_is_a_letter("apple pi e ") ➞ False
    check_if_last_char_is_a_letter("") ➞ False
    '''
    txt = txt.strip()
    if not txt:
        return False
    last_char = txt[-1]
    if last_char.isalpha():
        return len(txt.split()[-1]) == 1
    return False
class App:
    def __init__(self, root):
        self.root = root
        self.root.title("Last Character Checker")
        self.label = tk.Label(root, text="Enter a string:")
        self.label.pack()
        self.entry = tk.Entry(root, width=50)
        self.entry.pack()
        self.check_button = tk.Button(root, text="Check", command=self.check_last_char)
        self.check_button.pack()
        self.result_label = tk.Label(root, text="")
        self.result_label.pack()
    def check_last_char(self):
        user_input = self.entry.get()
        result = check_if_last_char_is_a_letter(user_input)
        if result:
            self.result_label.config(text="The last character is a standalone letter.")
        else:
            self.result_label.config(text="The last character is NOT a standalone letter.")
if __name__ == "__main__":
    root = tk.Tk()
    app = App(root)

'''

and a class to manage the GUI.
'''
import tkinter as tk
from tkinter import messagebox
def check_if_last_char_is_a_letter(txt):
    '''
    Create a function that returns True if the last character
    of a given string is an alphabetical character and is not
    a part of a word, and False otherwise.
    Note: "word" is a group of characters separated by space.
    Examples:
    check_if_last_char_is_a_letter("apple pie") ➞ False
    check_if_last_char_is_a_letter("apple pi e") ➞ True
    check_if_last_char_is_a_letter("apple pi e ") ➞ False
    check_if_last_char_is_a_letter("") ➞ False
    '''
    txt = txt.strip()
    if not txt:
        return False
    last_char = txt[-1]
    if last_char.isalpha():
        return len(txt.split()[-1]) == 1
    return False
class App:
    '''
    This class manages the GUI for the Last Character Checker application.
    It includes a label, an entry widget for user input, a button to trigger
    the check, and a label to display the result.
    '''
    def __init__(self, root):
        self.root = root
        self.root.title("Last Character Checker")
        self.label = tk.Label(root, text="Enter a string:")
        self.label.pack()
        self.entry = tk.Entry(root, width=50)
        self.entry.pack()
        self.check_button = tk.Button(root, text="Check", command=self.check_last_char)
        self.check_button.pack()
        self.result_label = tk.Label(root, text="")
        self.result_label.pack()
    def check_last_char(self):
        '''
        This method retrieves the user input from the entry widget,
        checks if the last character is a standalone letter using the
        check_if_last_char_is_a_letter function, and updates the result
        label with the appropriate message.
        '''
        user_input = self.entry.get()
        result = check_if_last_char_is_a_letter(user_input)
        if result:
            self.result_label.config(text="The last character is a standalone letter.")
        else:
            self.result_label.config(text="The last character is NOT a standalone letter.")
if __name__ == "__main__":
    root = tk.Tk()
    app = App(root)
    root.mainloop()

    a part of a word, and False otherwise.
    Note: "word" is a group of characters separated by space.
