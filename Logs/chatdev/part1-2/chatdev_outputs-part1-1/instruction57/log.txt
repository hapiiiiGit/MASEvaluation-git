```python
'''

Main application file for converting integers to Roman numerals using a GUI.

'''

import tkinter as tk
from tkinter import messagebox

def int_to_mini_roman(number):
    '''
    Given a positive integer, obtain its roman numeral equivalent as a string,
    and return it in lowercase.
    Restrictions: 1 <= num <= 1000

    Examples:
    >>> int_to_mini_roman(19) == 'xix'
    >>> int_to_mini_roman(152) == 'clii'
    >>> int_to_mini_roman(426) == 'cdxxvi'
    '''
    roman_numerals = [
        (1000, 'M'), (900, 'CM'), (500, 'D'), (400, 'CD'),
        (100, 'C'), (90, 'XC'), (50, 'L'), (40, 'XL'),
        (10, 'X'), (9, 'IX'), (5, 'V'), (4, 'IV'), (1, 'I')
    ]
    result = []
    for value, numeral in roman_numerals:
        while number >= value:
            result.append(numeral)
            number -= value
    return ''.join(result).lower()

class RomanConverterApp:
    '''
    A class to create a GUI application for converting integers to Roman numerals.
    '''
    def __init__(self, master):
        self.master = master
        master.title("Integer to Roman Numeral Converter")

        self.label = tk.Label(master, text="Enter an integer (1-1000):")
        self.label.pack()

        self.entry = tk.Entry(master)
        self.entry.pack()

        self.convert_button = tk.Button(master, text="Convert", command=self.convert)
        self.convert_button.pack()

        self.result_label = tk.Label(master, text="", fg="blue")
        self.result_label.pack()

    def convert(self):
        try:
            number = int(self.entry.get())
            if 1 <= number <= 1000:
                roman_numeral = int_to_mini_roman(number)
                self.result_label.config(text=f"Roman Numeral: {roman_numeral}")
            else:
                messagebox.showerror("Error", "Please enter a number between 1 and 1000.")
        except ValueError:
            messagebox.showerror("Error", "Invalid input. Please enter a valid integer.")

if __name__ == "__main__":
    root = tk.Tk()
    app = RomanConverterApp(root)
    root.mainloop()
```

'''

import tkinter as tk
from tkinter import messagebox
def int_to_mini_roman(number):
    '''
    Given a positive integer, obtain its roman numeral equivalent as a string,
    and return it in lowercase.
    Restrictions: 1 <= num <= 1000
    Examples:
    >>> int_to_mini_roman(19) == 'xix'
    >>> int_to_mini_roman(152) == 'clii'
    >>> int_to_mini_roman(426) == 'cdxxvi'
    '''
    roman_numerals = [
        (1000, 'M'), (900, 'CM'), (500, 'D'), (400, 'CD'),
        (100, 'C'), (90, 'XC'), (50, 'L'), (40, 'XL'),
        (10, 'X'), (9, 'IX'), (5, 'V'), (4, 'IV'), (1, 'I')
    ]
    result = []
    for value, numeral in roman_numerals:
        while number >= value:
            result.append(numeral)
            number -= value
    return ''.join(result).lower()
class RomanConverterApp:
    '''
    A class to create a GUI application for converting integers to Roman numerals.
    '''
    def __init__(self, master):
        self.master = master
        master.title("Integer to Roman Numeral Converter")
        self.label = tk.Label(master, text="Enter an integer (1-1000):")
        self.label.pack()
        self.entry = tk.Entry(master)
        self.entry.pack()
        self.convert_button = tk.Button(master, text="Convert", command=self.convert)
        self.convert_button.pack()
        self.result_label = tk.Label(master, text="", fg="blue")
        self.result_label.pack()
    def convert(self):
        try:
            number = int(self.entry.get())
            if 1 <= number <= 1000:
                roman_numeral = int_to_mini_roman(number)
                self.result_label.config(text=f"Roman Numeral: {roman_numeral}")
            else:
                messagebox.showerror("Error", "Please enter a number between 1 and 1000.")
        except ValueError:
            messagebox.showerror("Error", "Invalid input. Please enter a valid integer.")
if __name__ == "__main__":
    root = tk.Tk()

    root.mainloop()

'''
Main application file for converting integers to Roman numerals using a GUI.
'''
import tkinter as tk
from tkinter import messagebox
def int_to_mini_roman(number):
    '''
    Given a positive integer, obtain its roman numeral equivalent as a string,
    and return it in lowercase.
    Restrictions: 1 <= num <= 1000
    Examples:
    >>> int_to_mini_roman(19) == 'xix'
    >>> int_to_mini_roman(152) == 'clii'
    >>> int_to_mini_roman(426) == 'cdxxvi'
    '''
    roman_numerals = [
        (1000, 'M'), (900, 'CM'), (500, 'D'), (400, 'CD'),
        (100, 'C'), (90, 'XC'), (50, 'L'), (40, 'XL'),
        (10, 'X'), (9, 'IX'), (5, 'V'), (4, 'IV'), (1, 'I')
    ]
    result = []
    for value, numeral in roman_numerals:
        while number >= value:
            result.append(numeral)
            number -= value
    return ''.join(result).lower()
class RomanConverterApp:
    '''
    A class to create a GUI application for converting integers to Roman numerals.
    '''
    def __init__(self, master):
        self.master = master
        master.title("Integer to Roman Numeral Converter")
        self.label = tk.Label(master, text="Enter an integer (1-1000):")
        self.label.pack()
        self.entry = tk.Entry(master)
        self.entry.pack()
        self.convert_button = tk.Button(master, text="Convert", command=self.convert)
        self.convert_button.pack()
        self.result_label = tk.Label(master, text="", fg="blue")
        self.result_label.pack()
    def convert(self):
        try:
            number = int(self.entry.get())
            if 1 <= number <= 1000:
                roman_numeral = int_to_mini_roman(number)
                self.result_label.config(text=f"Roman Numeral: {roman_numeral}")
            else:
                messagebox.showerror("Error", "Please enter a number between 1 and 1000.")
        except ValueError:
            messagebox.showerror("Error", "Invalid input. Please enter a valid integer.")
if __name__ == "__main__":
    root = tk.Tk()
    app = RomanConverterApp(root)
    root.mainloop()

'''
Main application file for converting integers to Roman numerals using a GUI.
'''
import tkinter as tk
from tkinter import messagebox
def int_to_mini_roman(number):
    '''
    Given a positive integer, obtain its roman numeral equivalent as a string,
    and return it in lowercase.
    Restrictions: 1 <= num <= 1000
    Examples:
    >>> int_to_mini_roman(19) == 'xix'
    >>> int_to_mini_roman(152) == 'clii'
    >>> int_to_mini_roman(426) == 'cdxxvi'
    '''
    roman_numerals = [
        (1000, 'M'), (900, 'CM'), (500, 'D'), (400, 'CD'),
        (100, 'C'), (90, 'XC'), (50, 'L'), (40, 'XL'),
        (10, 'X'), (9, 'IX'), (5, 'V'), (4, 'IV'), (1, 'I')
    ]
    result = []
    for value, numeral in roman_numerals:
        while number >= value:
            result.append(numeral)
            number -= value
    return ''.join(result).lower()
class RomanConverterApp:
    '''
    A class to create a GUI application for converting integers to Roman numerals.
    '''
    def __init__(self, master):
        self.master = master
        master.title("Integer to Roman Numeral Converter")
        self.label = tk.Label(master, text="Enter an integer (1-1000):")
        self.label.pack()
        self.entry = tk.Entry(master)
        self.entry.pack()
        self.convert_button = tk.Button(master, text="Convert", command=self.convert)
        self.convert_button.pack()
        self.result_label = tk.Label(master, text="", fg="blue")
        self.result_label.pack()
    def convert(self):
        try:
            number = int(self.entry.get())
            if 1 <= number <= 1000:
                roman_numeral = int_to_mini_roman(number)
                self.result_label.config(text=f"Roman Numeral: {roman_numeral}")
            else:
                messagebox.showerror("Error", "Please enter a number between 1 and 1000.")
        except ValueError:
            messagebox.showerror("Error", "Invalid input. Please enter a valid integer.")
if __name__ == "__main__":
    root = tk.Tk()
    app = RomanConverterApp(root)
    root.mainloop()
