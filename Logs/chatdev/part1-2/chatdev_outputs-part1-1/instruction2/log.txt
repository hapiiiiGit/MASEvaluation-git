'''

Main application file for the WordsApp GUI.

'''

import tkinter as tk
from utils import words_string

class WordsApp:
    def __init__(self, root):
        '''
        Initialize the WordsApp GUI with input and output fields.
        '''
        self.root = root
        self.root.title("Words String Splitter")

        # Input label and entry
        self.input_label = tk.Label(root, text="Enter a string:")
        self.input_label.pack()

        self.input_entry = tk.Entry(root, width=50)
        self.input_entry.pack()

        # Split button
        self.split_button = tk.Button(root, text="Split Words", command=self.split_words)
        self.split_button.pack()

        # Output label
        self.output_label = tk.Label(root, text="Words:")
        self.output_label.pack()

        self.output_text = tk.Text(root, height=10, width=50)
        self.output_text.pack()

    def split_words(self):
        '''
        Split the input string into words and display the result.
        '''
        input_string = self.input_entry.get()
        words = words_string(input_string)
        self.output_text.delete(1.0, tk.END)
        self.output_text.insert(tk.END, str(words))

if __name__ == "__main__":
    root = tk.Tk()
    app = WordsApp(root)
    root.mainloop()
```

        self.output_text = tk.Text(root, height=10, width=50)
        self.output_text.pack()
    def split_words(self):
        '''
        Split the input string into words and display the result.
        '''
        input_string = self.input_entry.get()

'''
Main application file for the WordsApp GUI.

import tkinter as tk
from utils import words_string
class WordsApp:
    def __init__(self, root):
        '''
        Initialize the WordsApp GUI with input and output fields.
        '''
        self.root = root
        self.root.title("Words String Splitter")
        # Input label and entry
        self.input_label = tk.Label(root, text="Enter a string:")
        self.input_label.pack()
        self.input_entry = tk.Entry(root, width=50)
        self.input_entry.pack()
        # Split button
        self.split_button = tk.Button(root, text="Split Words", command=self.split_words)
        self.split_button.pack()
        # Output label
        self.output_label = tk.Label(root, text="Words:")
        self.output_label.pack()
        self.output_text = tk.Text(root, height=10, width=50)
        self.output_text.pack()
    def split_words(self):
        '''
        Split the input string into words and display the result.
        '''
        input_string = self.input_entry.get()
        words = words_string(input_string)
        self.output_text.delete(1.0, tk.END)
        self.output_text.insert(tk.END, str(words))
if __name__ == "__main__":
    root = tk.Tk()
    app = WordsApp(root)
    root.mainloop()
