import tkinter as tk
from utils import sort_array_logic

class SortArrayApp:
    def __init__(self, root):
        '''
        Initialize the GUI application.
        '''
        self.root = root
        self.root.title("Sort Array Application")

        self.label = tk.Label(root, text="Enter array of non-negative integers (comma separated):")
        self.label.pack()

        self.entry = tk.Entry(root, width=50)
        self.entry.pack()

        self.sort_button = tk.Button(root, text="Sort Array", command=self.sort_array)
        self.sort_button.pack()

        self.result_label = tk.Label(root, text="")
        self.result_label.pack()

    def sort_array(self):
        '''
        Sort the array based on binary representation and update the result.
        '''
        input_data = self.entry.get()
        try:
            arr = list(map(int, input_data.split(',')))
            sorted_arr = sort_array_logic(arr)
            self.result_label.config(text=f"Sorted Array: {sorted_arr}")
        except ValueError:
            self.result_label.config(text="Invalid input. Please enter integers separated by commas.")

if __name__ == "__main__":
    root = tk.Tk()
    app = SortArrayApp(root)

import tkinter as tk
from utils import sort_array_logic
class SortArrayApp:
    def __init__(self, root):
        '''
        Initialize the GUI application.
        '''
        self.root = root
        self.root.title("Sort Array Application")
        self.label = tk.Label(root, text="Enter array of non-negative integers (comma separated):")
        self.label.pack()
        self.entry = tk.Entry(root, width=50)
        self.entry.pack()
        self.sort_button = tk.Button(root, text="Sort Array", command=self.sort_array)
        self.sort_button.pack()
        self.result_label = tk.Label(root, text="")
        self.result_label.pack()
    def sort_array(self):
        '''
        Sort the array based on binary representation and update the result.
        '''
        input_data = self.entry.get()
        try:
            arr = list(map(int, input_data.split(',')))
            if any(x < 0 for x in arr):
                raise ValueError("Array contains negative integers, which are not supported.")
            sorted_arr = sort_array_logic(arr)
            self.result_label.config(text=f"Sorted Array: {sorted_arr}")
        except ValueError as e:
            self.result_label.config(text=str(e))
if __name__ == "__main__":
    root = tk.Tk()
    app = SortArrayApp(root)
    root.mainloop()

'''
Utility functions for sorting array based on binary representation.
'''
def sort_array_logic(arr):
    '''
    Sort an array of non-negative integers according to the number of ones in their binary representation.
    For similar number of ones, sort based on decimal value.
    Note: This function assumes all integers are non-negative.
    '''
    if any(x < 0 for x in arr):
        raise ValueError("Array contains negative integers, which are not supported.")
    return sorted(arr, key=lambda x: (bin(x).count('1'), x))

The Sort Array Application is designed to help users sort arrays of non-negative integers according to the number of ones in their binary representation.
