'''

This module contains the MainApp class for the GUI application and the unique_digits function to process the list of integers.

'''

import tkinter as tk
from tkinter import messagebox

def unique_digits(x):
    '''
    Given a list of positive integers x, return a sorted list of all elements that don't have any even digit.

    :param x: List of positive integers
    :return: Sorted list of integers without even digits
    '''
    def has_even_digit(num):
        return any(int(digit) % 2 == 0 for digit in str(num))

    result = [num for num in x if not has_even_digit(num)]
    return sorted(result)

class MainApp:
    '''
    Main application class for the GUI.
    '''
    def __init__(self, root):
        self.root = root
        self.root.title("Unique Digits Finder")

        self.label = tk.Label(root, text="Enter a list of positive integers separated by commas:")
        self.label.pack()

        self.entry = tk.Entry(root, width=50)
        self.entry.pack()

        self.button = tk.Button(root, text="Find Unique Digits", command=self.process_input)
        self.button.pack()

        self.result_label = tk.Label(root, text="")
        self.result_label.pack()

    def process_input(self):
        '''
        Process the input from the entry widget, find numbers without even digits, and display the result.
        '''
        try:
            input_text = self.entry.get()
            numbers = list(map(int, input_text.split(',')))
            result = unique_digits(numbers)
            self.result_label.config(text=f"Result: {result}")
        except ValueError:
            messagebox.showerror("Invalid Input", "Please enter a valid list of integers separated by commas.")

if __name__ == "__main__":
    root = tk.Tk()
    app = MainApp(root)
    root.mainloop()

import tkinter as tk
from tkinter import messagebox
def unique_digits(x):
    '''
    Given a list of positive integers x, return a sorted list of all elements that don't have any even digit.
    :param x: List of positive integers
    :return: Sorted list of integers without even digits
    '''
    def has_even_digit(num):
        return any(int(digit) % 2 == 0 for digit in str(num))
    result = [num for num in x if not has_even_digit(num)]
    return sorted(result)

    '''
    Main application class for the GUI.
    '''

        self.entry.pack()

            input_text = self.entry.get()

import tkinter as tk
from tkinter import messagebox
def unique_digits(x):
    '''
    Given a list of positive integers x, return a sorted list of all elements that don't have any even digit.
    :param x: List of positive integers
    :return: Sorted list of integers without even digits
    '''
    def has_even_digit(num):
