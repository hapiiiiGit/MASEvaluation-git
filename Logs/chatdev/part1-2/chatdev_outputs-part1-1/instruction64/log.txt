def generate_integers(a, b):
    """Given two positive integers a and b, return the even digits between a and b, in ascending order."""
    if a > b:
        a, b = b, a
    even_digits = [i for i in range(a, b + 1) if i % 2 == 0]
    return even_digits

def generate_integers(a, b):
    """Given two positive integers a and b, return the even digits between a and b, in ascending order."""
    if a > b:
        a, b = b, a
    even_digits = [i for i in range(a, b + 1) if i % 2 == 0]
    return even_digits

def generate_integers(a, b):
    """Given two positive integers a and b, return the even digits between a and b, in ascending order."""
    if a > b:
        a, b = b, a
    even_digits = [i for i in range(a, b + 1) if i % 2 == 0]
    return even_digits
