'''

Main file for the application. Contains the MainApp class for GUI and the add_elements function for calculation.

import tkinter as tk
from tkinter import messagebox

def add_elements(arr, k):
    '''
    Given a non-empty array of integers arr and an integer k, return
    the sum of the elements with at most two digits from the first k elements of arr.

    :param arr: List[int] - List of integers
    :param k: int - Number of elements to consider from the start of the list
    :return: int - Sum of elements with at most two digits
    '''
    return sum(x for x in arr[:k] if -99 <= x <= 99)

class MainApp:
    '''
    Main application class to create the GUI and handle user interactions.
    '''
    def __init__(self, root):
        self.root = root
        self.root.title("Sum of Elements with At Most Two Digits")

        # Input for array
        self.label_arr = tk.Label(root, text="Enter array (comma-separated):")
        self.label_arr.pack()
        self.entry_arr = tk.Entry(root, width=50)
        self.entry_arr.pack()

        # Input for k
        self.label_k = tk.Label(root, text="Enter k:")
        self.label_k.pack()
        self.entry_k = tk.Entry(root, width=10)
        self.entry_k.pack()

        # Calculate button
        self.button_calculate = tk.Button(root, text="Calculate", command=self.calculate_sum)
        self.button_calculate.pack()

        # Result display
        self.label_result = tk.Label(root, text="Result: ")
        self.label_result.pack()

    def calculate_sum(self):
        '''
        Retrieve user inputs, perform the calculation, and display the result.
        '''
        try:
            # Parse array input
            arr_input = self.entry_arr.get()
            arr = list(map(int, arr_input.split(',')))

            # Parse k input
            k = int(self.entry_k.get())

            # Validate inputs
            if not (1 <= len(arr) <= 100):
                raise ValueError("Array length must be between 1 and 100.")
            if not (1 <= k <= len(arr)):
                raise ValueError("k must be between 1 and the length of the array.")

            # Calculate result
            result = add_elements(arr, k)
            self.label_result.config(text=f"Result: {result}")

        except ValueError as e:
            messagebox.showerror("Input Error", str(e))
        except Exception as e:
            messagebox.showerror("Error", "An unexpected error occurred.")

if __name__ == "__main__":
    root = tk.Tk()
    app = MainApp(root)
    root.mainloop()

import tkinter as tk
from tkinter import messagebox
def add_elements(arr, k):
    '''
    Given a non-empty array of integers arr and an integer k, return
    the sum of the elements with at most two digits from the first k elements of arr.
    :param arr: List[int] - List of integers
    :param k: int - Number of elements to consider from the start of the list
    :return: int - Sum of elements with at most two digits
    '''
    return sum(x for x in arr[:k] if -99 <= x <= 99)

        self.button_calculate.pack()

        '''

        '''

            arr = list(map(int, arr_input.split(',')))
            # Parse k input
            k = int(self.entry_k.get())
            # Validate inputs
            if not (1 <= len(arr) <= 100):
                raise ValueError("Array length must be between 1 and 100.")
            if not (1 <= k <= len(arr)):

            # Calculate result
            result = add_elements(arr, k)
            self.label_result.config(text=f"Result: {result}")

import tkinter as tk
from tkinter import messagebox
def add_elements(arr, k):
    '''
    Given a non-empty array of integers arr and an integer k, return
    the sum of the elements with at most two digits from the first k elements of arr.
    :param arr: List[int] - List of integers
    :param k: int - Number of elements to consider from the start of the list
    :return: int - Sum of elements with at most two digits
    '''
    return sum(x for x in arr[:k] if -99 <= x <= 99)
