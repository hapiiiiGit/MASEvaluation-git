'''

Contains the logic for splitting words according to specified rules.

def split_words(txt):
    if ' ' in txt:
        return txt.split()
    elif ',' in txt:
        return txt.split(',')
    else:
        odd_count = sum(1 for char in txt if char.islower() and (ord(char) - ord('a')) % 2 == 1)
        return odd_count
```

def split_words(txt):
    if ' ' in txt:
        return txt.split()
    elif ',' in txt:
        return txt.split(',')
    else:
        odd_count = sum(1 for char in txt if char.islower() and (ord(char) - ord('a')) % 2 == 1)
        return odd_count

'''

def split_words(txt):
    if ' ' in txt:
        return txt.split()
    elif ',' in txt:
        return txt.split(',')
    else:
        odd_count = sum(1 for char in txt if char.islower() and (ord(char) - ord('a')) % 2 == 1)
        return odd_count
