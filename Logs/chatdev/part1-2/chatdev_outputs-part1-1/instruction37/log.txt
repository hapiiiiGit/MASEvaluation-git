'''

This module contains the main application logic for the LargestSmallestApp,
which provides a GUI to find the largest negative and smallest positive integers
from a user-provided list.

'''

import tkinter as tk
from tkinter import messagebox

class LargestSmallestApp:
    def __init__(self, master):
        self.master = master
        master.title("Largest Smallest Integers Finder")

        self.label = tk.Label(master, text="Enter a list of integers separated by commas:")
        self.label.pack()

        self.entry = tk.Entry(master, width=50)
        self.entry.pack()

        self.calculate_button = tk.Button(master, text="Calculate", command=self.calculate)
        self.calculate_button.pack()

        self.result_label = tk.Label(master, text="")
        self.result_label.pack()

    def calculate(self):
        input_text = self.entry.get()
        try:
            lst = list(map(int, input_text.split(',')))
            largest_neg, smallest_pos = self.largest_smallest_integers(lst)
            result_text = f"Largest Negative: {largest_neg}, Smallest Positive: {smallest_pos}"
            self.result_label.config(text=result_text)
        except ValueError:
            messagebox.showerror("Invalid input", "Please enter a valid list of integers separated by commas.")

    def largest_smallest_integers(self, lst):
        largest_neg = None
        smallest_pos = None

        for num in lst:
            if num < 0:
                if largest_neg is None or num > largest_neg:
                    largest_neg = num
            elif num > 0:
                if smallest_pos is None or num < smallest_pos:
                    smallest_pos = num

        return (largest_neg, smallest_pos)

def main():
    root = tk.Tk()
    app = LargestSmallestApp(root)
    root.mainloop()

if __name__ == "__main__":
    main()

which provides a GUI to find the largest negative and smallest positive integers
from a user-provided list.

import tkinter as tk
from tkinter import messagebox

class LargestSmallestApp:
    def __init__(self, master):
        '''
        Initializes the LargestSmallestApp with GUI components.
        '''
        self.master = master
        master.title("Largest Smallest Integers Finder")
        
        self.label = tk.Label(master, text="Enter a list of integers separated by commas:")

        self.entry.pack()
        
        self.calculate_button = tk.Button(master, text="Calculate", command=self.calculate)

        self.result_label = tk.Label(master, text="")
        self.result_label.pack()

    def calculate(self):

        Handles the calculation of largest negative and smallest positive integers
        from the user input and updates the result label.
        '''

        '''

        '''

        self.result_label = tk.Label(master, text="")
        self.result_label.pack()

        '''
        Handles the calculation of largest negative and smallest positive integers

        '''
        Returns a tuple containing the largest negative integer and the smallest positive integer

        largest_neg = None
        smallest_pos = None
        for num in lst:

                    largest_neg = num
            elif num > 0:
                if smallest_pos is None or num < smallest_pos:
                    smallest_pos = num
        return (largest_neg, smallest_pos)

import tkinter as tk
from tkinter import messagebox
class LargestSmallestApp:
    def __init__(self, master):
        '''
        Initializes the LargestSmallestApp with GUI components.
        '''
        self.master = master
        master.title("Largest Smallest Integers Finder")
        self.label = tk.Label(master, text="Enter a list of integers separated by commas:")
        self.label.pack()
        self.entry = tk.Entry(master, width=50)
        self.entry.pack()
        self.calculate_button = tk.Button(master, text="Calculate", command=self.calculate)
        self.calculate_button.pack()
        self.result_label = tk.Label(master, text="")
        self.result_label.pack()
    def calculate(self):
        '''
        Handles the calculation of largest negative and smallest positive integers
        from the user input and updates the result label.
        '''
        input_text = self.entry.get()
        if not input_text.strip():
            messagebox.showerror("Invalid input", "Input cannot be empty. Please enter a list of integers separated by commas.")
            return
        try:
            lst = list(map(int, input_text.split(',')))
            largest_neg, smallest_pos = self.largest_smallest_integers(lst)
            result_text = f"Largest Negative: {largest_neg}, Smallest Positive: {smallest_pos}"
            self.result_label.config(text=result_text)
        except ValueError:
            invalid_values = [val for val in input_text.split(',') if not val.strip().isdigit()]
            messagebox.showerror("Invalid input", f"Invalid integer values: {', '.join(invalid_values)}. Please enter a valid list of integers separated by commas.")
    def largest_smallest_integers(self, lst):
        '''
        Returns a tuple containing the largest negative integer and the smallest positive integer
        from the provided list. If no such integers exist, returns None for each.
        '''
        largest_neg = None
        smallest_pos = None
        for num in lst:
            if num < 0:
                if largest_neg is None or num > largest_neg:
                    largest_neg = num
            elif num > 0:
                if smallest_pos is None or num < smallest_pos:
                    smallest_pos = num
        return (largest_neg, smallest_pos)
def main():
    '''
    Main function to run the LargestSmallestApp.
    '''
    root = tk.Tk()
    app = LargestSmallestApp(root)
    root.mainloop()
if __name__ == "__main__":

    def __init__(self, master):
        '''
        Initializes the LargestSmallestApp with GUI components.
        '''
        self.master = master
        master.title("Largest Smallest Integers Finder")
        self.label = tk.Label(master, text="Enter a list of integers separated by commas:")
        self.label.pack()
        self.entry = tk.Entry(master, width=50)
        self.entry.pack()
        self.calculate_button = tk.Button(master, text="Calculate", command=self.calculate)
        self.calculate_button.pack()
        self.result_label = tk.Label(master, text="")
        self.result_label.pack()

        '''
        Handles the calculation of largest negative and smallest positive integers
        from the user input and updates the result label.
        '''

        '''
