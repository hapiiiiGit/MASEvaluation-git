def special_factorial(n):
    """
    Calculate the Brazilian factorial of a given integer n.
    
    :param n: Integer for which the special factorial is calculated
    :return: The special factorial of n
    """
    if n <= 0:
        raise ValueError("n must be greater than 0")
    
    def factorial(x):
        if x == 0 or x == 1:
            return 1
        else:
            return x * factorial(x - 1)
    
    result = 1
    for i in range(1, n + 1):
        result *= factorial(i)
    
    return result

def special_factorial(n):
    """
    Calculate the Brazilian factorial of a given integer n.
    :param n: Integer for which the special factorial is calculated
    :return: The special factorial of n
    """
    if n <= 0:
        raise ValueError("n must be greater than 0")
    def factorial(x):
        if x == 0 or x == 1:
            return 1
        else:
            return x * factorial(x - 1)
    result = 1
    for i in range(1, n + 1):
        result *= factorial(i)
    return result

def special_factorial(n):
    """
    Calculate the Brazilian factorial of a given integer n.
    :param n: Integer for which the special factorial is calculated
    :return: The special factorial of n
    """
    if n <= 0:
        raise ValueError("n must be greater than 0")
    def factorial(x):
        if x == 0 or x == 1:
            return 1
        else:
            return x * factorial(x - 1)
    result = 1
    for i in range(1, n + 1):
        result *= factorial(i)
    return result

'''
Contains the logic for calculating the special factorial (Brazilian factorial).
'''
def special_factorial(n):
    """
    Calculate the Brazilian factorial of a given integer n.
    :param n: Integer for which the special factorial is calculated
    :return: The special factorial of n
    """
    if n <= 0:
        raise ValueError("n must be greater than 0")
    def factorial(x):
        if x == 0 or x == 1:
            return 1
        else:
            return x * factorial(x - 1)
    result = 1
    for i in range(1, n + 1):
        result *= factorial(i)
    return result

'''
Contains the logic for calculating the special factorial (Brazilian factorial).
'''
def special_factorial(n):
    """
    Calculate the Brazilian factorial of a given integer n.
    :param n: Integer for which the special factorial is calculated
    :return: The special factorial of n
    """
    if n <= 0:
        raise ValueError("n must be greater than 0")
    def factorial(x):
        if x == 0 or x == 1:
            return 1
        else:
            return x * factorial(x - 1)
    result = 1
    for i in range(1, n + 1):
        result *= factorial(i)
    return result
