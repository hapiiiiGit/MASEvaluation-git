import wx
from shapes import Rectangle, Circle
class CanvasPanel(wx.Panel):
    def __init__(self, parent):
        super(CanvasPanel, self).__init__(parent)
        self.shapes = []
        self.selected_shape = None
        self.drag_start_pos = None
        self.Bind(wx.EVT_PAINT, self.on_paint)
        self.Bind(wx.EVT_LEFT_DOWN, self.on_left_down)
        self.Bind(wx.EVT_LEFT_UP, self.on_left_up)
        self.Bind(wx.EVT_MOTION, self.on_motion)
    def on_paint(self, event):
        dc = wx.PaintDC(self)
        for shape in self.shapes:
            shape.draw(dc)
    def on_left_down(self, event):
        x, y = event.GetPosition()
        self.selected_shape = None
        for shape in reversed(self.shapes):
            if shape.contains_point(x, y):
                self.selected_shape = shape
                shape.selected = True
                self.drag_start_pos = (x, y)
                break
        self.Refresh()
    def on_left_up(self, event):
        if self.selected_shape:
            self.selected_shape.selected = False
            self.selected_shape = None
        self.drag_start_pos = None
        self.Refresh()
    def on_motion(self, event):
        if event.Dragging() and self.selected_shape:
            x, y = event.GetPosition()
            dx = x - self.drag_start_pos[0]
            dy = y - self.drag_start_pos[1]
            self.selected_shape.move(dx, dy)
            self.drag_start_pos = (x, y)
            self.Refresh()
    def add_shape(self, shape):
        self.shapes.append(shape)
        self.Refresh()
class ShapeManager:
    def __init__(self, canvas):
        self.canvas = canvas
    def create_rectangle(self, x, y, width, height):

    def move(self, dx, dy):

        self.y += dy

    def __init__(self, x, y, width, height):
        super().__init__(x, y)
        self.width = width
        self.height = height

        if self.selected:

        dc.SetPen(pen)

'''
Defines shape classes for drawing and manipulating shapes.
'''
import wx
class Shape:
    def __init__(self, x, y):
        self.x = x
        self.y = y
        self.selected = False
    def draw(self, dc):
        raise NotImplementedError("Draw method must be implemented by subclasses.")
    def contains_point(self, x, y):
        raise NotImplementedError("Contains_point method must be implemented by subclasses.")
    def move(self, dx, dy):
        self.x += dx
        self.y += dy
class Rectangle(Shape):
    def __init__(self, x, y, width, height):
        super().__init__(x, y)
        self.width = width
        self.height = height
    def draw(self, dc):
        pen = wx.Pen(wx.BLACK, 2)
        if self.selected:
            pen.SetStyle(wx.PENSTYLE_DOT)
        dc.SetPen(pen)
        dc.SetBrush(wx.Brush(wx.BLUE if self.selected else wx.RED))
        dc.DrawRectangle(self.x, self.y, self.width, self.height)
    def contains_point(self, x, y):
        return self.x <= x <= self.x + self.width and self.y <= y <= self.y + self.height
class Circle(Shape):
    def __init__(self, x, y, radius):
        super().__init__(x, y)
        self.radius = radius
    def draw(self, dc):
        pen = wx.Pen(wx.BLACK, 2)
        if self.selected:
            pen.SetStyle(wx.PENSTYLE_DOT)
        dc.SetPen(pen)
        dc.SetBrush(wx.Brush(wx.GREEN if self.selected else wx.YELLOW))
        dc.DrawCircle(self.x, self.y, self.radius)
    def contains_point(self, x, y):
        return (x - self.x) ** 2 + (y - self.y) ** 2 <= self.radius ** 2

'''
Canvas panel for drawing and managing shapes.
'''
import wx
from shapes import Rectangle, Circle
class CanvasPanel(wx.Panel):
    def __init__(self, parent):
        super(CanvasPanel, self).__init__(parent)
        self.shapes = []
        self.selected_shape = None
        self.drag_start_pos = None
        self.Bind(wx.EVT_PAINT, self.on_paint)
        self.Bind(wx.EVT_LEFT_DOWN, self.on_left_down)
        self.Bind(wx.EVT_LEFT_UP, self.on_left_up)
        self.Bind(wx.EVT_MOTION, self.on_motion)
    def on_paint(self, event):
        dc = wx.PaintDC(self)
        for shape in self.shapes:
            shape.draw(dc)
    def on_left_down(self, event):
        x, y = event.GetPosition()
        self.selected_shape = None
        for shape in reversed(self.shapes):
            if shape.contains_point(x, y):
                self.selected_shape = shape
                shape.selected = True
                self.drag_start_pos = (x, y)
                break
        self.Refresh()
    def on_left_up(self, event):
        if self.selected_shape:
            self.selected_shape.selected = False
            self.selected_shape = None
        self.drag_start_pos = None
        self.Refresh()
    def on_motion(self, event):
        if event.Dragging() and self.selected_shape:
            x, y = event.GetPosition()
            dx = x - self.drag_start_pos[0]
            dy = y - self.drag_start_pos[1]
            self.selected_shape.move(dx, dy)
            self.drag_start_pos = (x, y)
            self.Refresh()
    def add_shape(self, shape):
        self.shapes.append(shape)
        self.Refresh()
class ShapeManager:
    def __init__(self, canvas):
        self.canvas = canvas
    def create_rectangle(self, x, y, width, height):
        rect = Rectangle(x, y, width, height)
        self.canvas.add_shape(rect)
    def create_circle(self, x, y, radius):
        circle = Circle(x, y, radius)
        self.canvas.add_shape(circle)

'''

import wx

class Shape:
    def __init__(self, x, y):
        self.x = x
        self.y = y
        self.selected = False

    def draw(self, dc):
        raise NotImplementedError("Draw method must be implemented by subclasses.")

    def contains_point(self, x, y):
        raise NotImplementedError("Contains_point method must be implemented by subclasses.")

    def move(self, dx, dy):
        self.x += dx
        self.y += dy

class Rectangle(Shape):
    def __init__(self, x, y, width, height):
        super().__init__(x, y)
        self.width = width
        self.height = height

    def draw(self, dc):
        pen = wx.Pen(wx.BLACK, 2)
        if self.selected:
            pen.SetStyle(wx.PENSTYLE_DOT)
        dc.SetPen(pen)
        dc.SetBrush(wx.Brush(wx.BLUE if self.selected else wx.RED))
        dc.DrawRectangle(self.x, self.y, self.width, self.height)

    def contains_point(self, x, y):
        return self.x <= x <= self.x + self.width and self.y <= y <= self.y + self.height

class Circle(Shape):
    def __init__(self, x, y, radius):
        super().__init__(x, y)
        self.radius = radius

    def draw(self, dc):
        pen = wx.Pen(wx.BLACK, 2)
        if self.selected:
            pen.SetStyle(wx.PENSTYLE_DOT)
        dc.SetPen(pen)
        dc.SetBrush(wx.Brush(wx.GREEN if self.selected else wx.YELLOW))
        dc.DrawCircle(self.x, self.y, self.radius)

    def contains_point(self, x, y):
        return (x - self.x) ** 2 + (y - self.y) ** 2 <= self.radius ** 2

'''
Canvas panel for drawing and managing shapes.
'''
import wx
from shapes import Rectangle, Circle

class CanvasPanel(wx.Panel):
    def __init__(self, parent):
        super(CanvasPanel, self).__init__(parent)
        self.shapes = []
        self.selected_shape = None
        self.drag_start_pos = None
        self.Bind(wx.EVT_PAINT, self.on_paint)
        self.Bind(wx.EVT_LEFT_DOWN, self.on_left_down)
        self.Bind(wx.EVT_LEFT_UP, self.on_left_up)
        self.Bind(wx.EVT_MOTION, self.on_motion)

    def on_paint(self, event):
        dc = wx.PaintDC(self)
        for shape in self.shapes:
            shape.draw(dc)

    def on_left_down(self, event):
        x, y = event.GetPosition()
        self.selected_shape = None
        for shape in reversed(self.shapes):
            if shape.contains_point(x, y):
                self.selected_shape = shape
                shape.selected = True
                self.drag_start_pos = (x, y)
                break
        self.Refresh()

    def on_left_up(self, event):
        if self.selected_shape:
            self.selected_shape.selected = False
            self.selected_shape = None
        self.drag_start_pos = None
        self.Refresh()

    def on_motion(self, event):
        if event.Dragging() and self.selected_shape:
            x, y = event.GetPosition()
            dx = x - self.drag_start_pos[0]
            dy = y - self.drag_start_pos[1]
            self.selected_shape.move(dx, dy)
            self.drag_start_pos = (x, y)
            self.Refresh()

    def add_shape(self, shape):
        self.shapes.append(shape)
        self.Refresh()

class ShapeManager:
    def __init__(self, canvas):
        self.canvas = canvas

    def create_rectangle(self, x, y, width, height):
        rect = Rectangle(x, y, width, height)
        self.canvas.add_shape(rect)

    def create_circle(self, x, y, radius):
        circle = Circle(x, y, radius)
        self.canvas.add_shape(circle)

'''
Defines shape classes for drawing and manipulating shapes.
'''
import wx
class Shape:
    def __init__(self, x, y):
        self.x = x
        self.y = y
        self.selected = False
    def draw(self, dc):
        raise NotImplementedError("Draw method must be implemented by subclasses.")
    def contains_point(self, x, y):
        raise NotImplementedError("Contains_point method must be implemented by subclasses.")
    def move(self, dx, dy):
        self.x += dx
        self.y += dy
class Rectangle(Shape):
    def __init__(self, x, y, width, height):
        super().__init__(x, y)
        self.width = width
        self.height = height
    def draw(self, dc):
        pen = wx.Pen(wx.BLACK, 2)
        if self.selected:
            pen.SetStyle(wx.PENSTYLE_DOT)
        dc.SetPen(pen)
        dc.SetBrush(wx.Brush(wx.BLUE if self.selected else wx.RED))
        dc.DrawRectangle(self.x, self.y, self.width, self.height)
    def contains_point(self, x, y):
        return self.x <= x <= self.x + self.width and self.y <= y <= self.y + self.height
class Circle(Shape):
    def __init__(self, x, y, radius):
        super().__init__(x, y)
        self.radius = radius
    def draw(self, dc):
        pen = wx.Pen(wx.BLACK, 2)
        if self.selected:
            pen.SetStyle(wx.PENSTYLE_DOT)
        dc.SetPen(pen)
        dc.SetBrush(wx.Brush(wx.GREEN if self.selected else wx.YELLOW))
        dc.DrawCircle(self.x, self.y, self.radius)
    def contains_point(self, x, y):
        return (x - self.x) ** 2 + (y - self.y) ** 2 <= self.radius ** 2

'''
Canvas panel for drawing and managing shapes.
'''
import wx
from shapes import Rectangle, Circle
class CanvasPanel(wx.Panel):
    def __init__(self, parent):
        super(CanvasPanel, self).__init__(parent)
        self.shapes = []
        self.selected_shape = None
        self.drag_start_pos = None
        self.Bind(wx.EVT_PAINT, self.on_paint)
        self.Bind(wx.EVT_LEFT_DOWN, self.on_left_down)
        self.Bind(wx.EVT_LEFT_UP, self.on_left_up)
        self.Bind(wx.EVT_MOTION, self.on_motion)
    def on_paint(self, event):
        dc = wx.PaintDC(self)
        for shape in self.shapes:
            shape.draw(dc)
    def on_left_down(self, event):
        x, y = event.GetPosition()
        self.selected_shape = None
        for shape in reversed(self.shapes):
            if shape.contains_point(x, y):
                self.selected_shape = shape
                shape.selected = True
                self.drag_start_pos = (x, y)
                break
        self.Refresh()
    def on_left_up(self, event):
        if self.selected_shape:
            self.selected_shape.selected = False
            self.selected_shape = None
        self.drag_start_pos = None
        self.Refresh()
    def on_motion(self, event):
        if event.Dragging() and self.selected_shape:
            x, y = event.GetPosition()
            dx = x - self.drag_start_pos[0]
            dy = y - self.drag_start_pos[1]
            self.selected_shape.move(dx, dy)
            self.drag_start_pos = (x, y)
            self.Refresh()
    def add_shape(self, shape):
        self.shapes.append(shape)
        self.Refresh()
class ShapeManager:
    def __init__(self, canvas):
        self.canvas = canvas
    def create_rectangle(self, x, y, width, height):
        rect = Rectangle(x, y, width, height)
        self.canvas.add_shape(rect)
    def create_circle(self, x, y, radius):
        circle = Circle(x, y, radius)
        self.canvas.add_shape(circle)
