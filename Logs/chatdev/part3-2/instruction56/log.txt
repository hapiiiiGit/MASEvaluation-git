'''

This file contains the JobListingApp class which sets up the GUI for the job listing.

import tkinter as tk
from job_details import JobDetails

class JobListingApp(tk.Tk):
    '''
    A class to create the main application window for the job listing.
    '''
    def __init__(self):
        super().__init__()
        self.title("Job Listing - Senior AI/ML Engineer")
        self.geometry("500x400")
        self.create_widgets()

    def create_widgets(self):
        '''
        Create and place the widgets in the application window.
        '''
        job_details = JobDetails()
        details_text = job_details.get_details()

        label = tk.Label(self, text="Job Listing", font=("Helvetica", 16))
        label.pack(pady=10)

        details_label = tk.Label(self, text=details_text, justify=tk.LEFT, anchor="w")
        details_label.pack(padx=10, pady=10, fill=tk.BOTH, expand=True)

        quit_button = tk.Button(self, text="Quit", command=self.quit)
        quit_button.pack(pady=10)

'''
This file contains the JobListingApp class which sets up the GUI for the job listing.
'''
import tkinter as tk
from job_details import JobDetails
class JobListingApp(tk.Tk):
    '''
    A class to create the main application window for the job listing.
    '''
    def __init__(self):
        super().__init__()
        self.title("Job Listing - Senior AI/ML Engineer")
        self.geometry("500x400")
        self.create_widgets()
    def create_widgets(self):
        '''
        Create and place the widgets in the application window.
        '''
        job_details = JobDetails()
        details_text = job_details.get_details()
        label = tk.Label(self, text="Job Listing", font=("Helvetica", 16))
        label.pack(pady=10)
        details_label = tk.Label(self, text=details_text, justify=tk.LEFT, anchor="w")
        details_label.pack(padx=10, pady=10, fill=tk.BOTH, expand=True)
        quit_button = tk.Button(self, text="Quit", command=self.quit)
        quit_button.pack(pady=10)

The Job Listing Application is a GUI-based program that displays the details of a job listing for a Senior AI/ML Engineer. The main functions include:

   python main.py
