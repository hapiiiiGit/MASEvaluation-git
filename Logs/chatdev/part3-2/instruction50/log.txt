'''
Analyzes and repairs syntax errors in Python files.
'''
import ast

class SyntaxAnalyzer:
    def analyze_file(self, file_path):
        errors = []
        with open(file_path, 'r') as file:
            try:
                tree = ast.parse(file.read())
            except SyntaxError as e:
                errors.append(e)
        return errors

    def repair_syntax(self, file_path, errors):
        changes = []
        with open(file_path, 'r') as file:
            lines = file.readlines()

        for error in errors:
            lineno = error.lineno - 1
            # Example logic to repair syntax errors
            # This is a placeholder for actual repair logic
            if "unexpected EOF" in error.msg:
                lines[lineno] = lines[lineno].rstrip() + " # Fixed unexpected EOF\n"
                changes.append(f"Fixed error at line {error.lineno}: {error.msg}")

        with open(file_path, 'w') as file:
            file.writelines(lines)

        return changes

'''
Analyzes and repairs syntax errors in Python files.
'''
import ast
class SyntaxAnalyzer:
    def analyze_file(self, file_path):
        errors = []
        with open(file_path, 'r') as file:
            try:
                tree = ast.parse(file.read())
            except SyntaxError as e:
                errors.append(e)
        return errors
    def repair_syntax(self, file_path, errors):
        changes = []
        with open(file_path, 'r') as file:
            lines = file.readlines()
        for error in errors:
            lineno = error.lineno - 1
            # Example logic to repair syntax errors
            # This is a placeholder for actual repair logic
            if "unexpected EOF" in error.msg:
                lines[lineno] = lines[lineno].rstrip() + " # Fixed unexpected EOF\n"
                changes.append(f"Fixed error at line {error.lineno}: {error.msg}")
        with open(file_path, 'w') as file:
            file.writelines(lines)
        return changes

'''
import tkinter as tk
from tkinter import filedialog, messagebox
class FileSelector:
    def __init__(self, root):
        self.root = root
    def create_main_window(self, syntax_analyzer, logger):
        # Create a button to select files
        select_button = tk.Button(self.root, text="Select File", command=lambda: self.select_file(syntax_analyzer, logger))
        select_button.pack(pady=20)
    def select_file(self, syntax_analyzer, logger):
        # Open file dialog to select a Python file
        file_path = filedialog.askopenfilename(filetypes=[("Python Files", "*.py")])
        if file_path:
            errors = syntax_analyzer.analyze_file(file_path)
            if errors:
                changes = syntax_analyzer.repair_syntax(file_path, errors)
                logger.log_changes(changes)
                messagebox.showinfo("Repair Status", "Syntax errors repaired. Check changes_log.txt for details.")
            else:
                messagebox.showinfo("Repair Status", "No syntax errors found.")

'''
Analyzes and repairs syntax errors in Python files.
'''
import ast
class SyntaxAnalyzer:
    def analyze_file(self, file_path):
        errors = []
        with open(file_path, 'r') as file:
            try:
                tree = ast.parse(file.read())
            except SyntaxError as e:
                errors.append(e)
        return errors
    def repair_syntax(self, file_path, errors):
        changes = []
        with open(file_path, 'r') as file:
            lines = file.readlines()
        for error in errors:
            lineno = error.lineno - 1
            # Example logic to repair syntax errors
            # This is a placeholder for actual repair logic
            if "unexpected EOF" in error.msg:
                lines[lineno] = lines[lineno].rstrip() + " # Fixed unexpected EOF\n"
                changes.append(f"Fixed error at line {error.lineno}: {error.msg}")
        with open(file_path, 'w') as file:
            file.writelines(lines)
        return changes

import ast
class SyntaxAnalyzer:
    def analyze_file(self, file_path):
        errors = []
        with open(file_path, 'r') as file:
            try:
                tree = ast.parse(file.read())
            except SyntaxError as e:
                errors.append(e)
        return errors
    def repair_syntax(self, file_path, errors):
        changes = []
        with open(file_path, 'r') as file:
            lines = file.readlines()
        for error in errors:
            lineno = error.lineno - 1
            # Example logic to repair syntax errors
            # This is a placeholder for actual repair logic
            if "unexpected EOF" in error.msg:
                lines[lineno] = lines[lineno].rstrip() + " # Fixed unexpected EOF\n"
                changes.append(f"Fixed error at line {error.lineno}: {error.msg}")
        with open(file_path, 'w') as file:
            file.writelines(lines)
        return changes
