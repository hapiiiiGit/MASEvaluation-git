'''
Main application file to initialize the GUI and manage user interactions.
'''
import tkinter as tk
from ingestion import DataIngestion
from transformation import DataTransformation
from loading import DataLoader
from semantic import SemanticLayer
from quality import SchemaManager, DataQualityChecker

class DataPipelineApp:
    def __init__(self, root):
        self.root = root
        self.root.title("Data Pipeline Application")
        self.ingestion = DataIngestion()
        self.transformation = DataTransformation()
        self.loader = DataLoader()
        self.semantic_layer = SemanticLayer()
        self.schema_manager = SchemaManager()
        self.quality_checker = DataQualityChecker()
        self.create_widgets()

    def create_widgets(self):
        tk.Button(self.root, text="Ingest Data", command=self.ingest_data).pack()
        tk.Button(self.root, text="Transform Data", command=self.transform_data).pack()
        tk.Button(self.root, text="Load Data", command=self.load_data).pack()
        tk.Button(self.root, text="Create Semantic Layer", command=self.create_semantic_layer).pack()
        tk.Button(self.root, text="Check Schema", command=self.check_schema).pack()
        tk.Button(self.root, text="Check Data Quality", command=self.check_data_quality).pack()

    def ingest_data(self):
        data = self.ingestion.ingest_data()
        self.transformed_data = self.transformation.transform_data(data)

    def transform_data(self):
        self.transformed_data = self.transformation.transform_data(self.transformed_data)

    def load_data(self):
        self.loader.load_data(self.transformed_data)

    def create_semantic_layer(self):
        self.semantic_layer.create_semantic_layer()

    def check_schema(self):
        self.schema_manager.check_schema(self.transformed_data)

    def check_data_quality(self):
        self.quality_checker.check_data_quality(self.transformed_data)

if __name__ == "__main__":
    root = tk.Tk()
    app = DataPipelineApp(root)
    root.mainloop()

import tkinter as tk
from ingestion import DataIngestion
from transformation import DataTransformation
from loading import DataLoader
from semantic import SemanticLayer
from quality import SchemaManager, DataQualityChecker
class DataPipelineApp:
    def __init__(self, root):
        self.root = root
        self.root.title("Data Pipeline Application")
        self.ingestion = DataIngestion()
        self.transformation = DataTransformation()
        self.loader = DataLoader()
        self.semantic_layer = SemanticLayer()
        self.schema_manager = SchemaManager()
        self.quality_checker = DataQualityChecker()
        self.create_widgets()
    def create_widgets(self):
        tk.Button(self.root, text="Ingest Data", command=self.ingest_data).pack()
        tk.Button(self.root, text="Transform Data", command=self.transform_data).pack()
        tk.Button(self.root, text="Load Data", command=self.load_data).pack()
        tk.Button(self.root, text="Create Semantic Layer", command=self.create_semantic_layer).pack()
        tk.Button(self.root, text="Check Schema", command=self.check_schema).pack()
        tk.Button(self.root, text="Check Data Quality", command=self.check_data_quality).pack()
    def ingest_data(self):
        data = self.ingestion.ingest_data()
        self.transformed_data = self.transformation.transform_data(data)
    def transform_data(self):
        self.transformed_data = self.transformation.transform_data(self.transformed_data)
    def load_data(self):
        self.loader.load_data(self.transformed_data)
    def create_semantic_layer(self):
        self.semantic_layer.create_semantic_layer()
    def check_schema(self):
        self.schema_manager.check_schema(self.transformed_data)
    def check_data_quality(self):
        self.quality_checker.check_data_quality(self.transformed_data)
if __name__ == "__main__":
    root = tk.Tk()
    app = DataPipelineApp(root)
    root.mainloop()
