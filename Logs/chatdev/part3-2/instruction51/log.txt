from sympy import sympify  # Example library for formula evaluation
from PyPDF2 import PdfReader

def generate_summary(file_path):
    # Determine file type and extract text
    _, file_extension = os.path.splitext(file_path)
    content = ""
    if file_extension.lower() == '.pdf':
        with open(file_path, 'rb') as file:
            reader = PdfReader(file)
            for page in reader.pages:
                content += page.extract_text()

        doc = DocxDocument(file_path)
        for para in doc.paragraphs:

from django.shortcuts import render, redirect
from django.http import HttpResponse

def upload_documents(request):
    if request.method == 'POST':
        form = DocumentForm(request.POST, request.FILES)
        if form.is_valid():
            document = form.save()
            summary_text = generate_summary(document.file.path)
            Summary.objects.create(document=document, text=summary_text)
            return redirect('document_list')
    else:
        form = DocumentForm()

'''
Utility functions for generating summaries and evaluating formulas.
'''
import os
from gensim.summarization import summarize  # Example library for summarization
from sympy import sympify  # Example library for formula evaluation
from PyPDF2 import PdfReader
from docx import Document as DocxDocument
def generate_summary(file_path):
    try:
        # Determine file type and extract text
        _, file_extension = os.path.splitext(file_path)
        content = ""
        if file_extension.lower() == '.pdf':
            with open(file_path, 'rb') as file:
                reader = PdfReader(file)
                for page in reader.pages:
                    content += page.extract_text()
        elif file_extension.lower() == '.docx':
            doc = DocxDocument(file_path)
            for para in doc.paragraphs:
                content += para.text + '\n'
        elif file_extension.lower() in ['.txt', '.text']:
            with open(file_path, 'r') as file:
                content = file.read()
        else:
            raise ValueError("Unsupported file type")
        # Check if content is large enough for summarization
        if len(content.split()) < 50:  # Example threshold
            return "Content too short for summarization."
        # Generate summary
        return summarize(content)
    except Exception as e:
        return f"An error occurred during summarization: {str(e)}"
def evaluate_formula(expression):
    # Implement logic to evaluate formula
    return sympify(expression).evalf()

'''
Utility functions for generating summaries and evaluating formulas.
'''
import os
from gensim.summarization import summarize  # Example library for summarization
from sympy import sympify  # Example library for formula evaluation
from PyPDF2 import PdfReader
from docx import Document as DocxDocument
def generate_summary(file_path):
    try:
        # Determine file type and extract text
        _, file_extension = os.path.splitext(file_path)
        content = ""
        if file_extension.lower() == '.pdf':
            with open(file_path, 'rb') as file:
                reader = PdfReader(file)
                for page in reader.pages:
                    content += page.extract_text()
        elif file_extension.lower() == '.docx':
            doc = DocxDocument(file_path)
            for para in doc.paragraphs:
                content += para.text + '\n'
        elif file_extension.lower() in ['.txt', '.text']:
            with open(file_path, 'r') as file:
                content = file.read()
        else:
            raise ValueError("Unsupported file type")
        # Check if content is large enough for summarization
        if len(content.split()) < 50:  # Example threshold
            return "Content too short for summarization."
        # Generate summary
        return summarize(content)
    except Exception as e:
        return f"An error occurred during summarization: {str(e)}"
def evaluate_formula(expression):
    # Implement logic to evaluate formula
    return sympify(expression).evalf()

from django.shortcuts import render, redirect
from django.http import HttpResponse
from .forms import DocumentForm, FormulaForm
from .models import Document, Summary, Formula
from .utils import generate_summary, evaluate_formula
def upload_documents(request):
    if request.method == 'POST':
        form = DocumentForm(request.POST, request.FILES)
        if form.is_valid():
            document = form.save()
            summary_text = generate_summary(document.file.path)
            Summary.objects.create(document=document, text=summary_text)
            return redirect('document_list')
    else:
        form = DocumentForm()
    return render(request, 'upload.html', {'form': form})
def summarize_document(request, doc_id):
    document = Document.objects.get(id=doc_id)
    summary = Summary.objects.get(document=document)
    return render(request, 'summary.html', {'summary': summary})
def formula_builder(request):
    if request.method == 'POST':
        form = FormulaForm(request.POST)
        if form.is_valid():
            formula = form.save()
            result = evaluate_formula(formula.expression)
            return render(request, 'formula_result.html', {'result': result})
    else:
        form = FormulaForm()
    return render(request, 'formula_builder.html', {'form': form})

'''
Utility functions for generating summaries and evaluating formulas.
'''
import os
from gensim.summarization import summarize  # Example library for summarization
from sympy import sympify  # Example library for formula evaluation
from PyPDF2 import PdfReader
from docx import Document as DocxDocument
def generate_summary(file_path):
    try:
        # Determine file type and extract text
        _, file_extension = os.path.splitext(file_path)
        content = ""
        if file_extension.lower() == '.pdf':
            with open(file_path, 'rb') as file:
                reader = PdfReader(file)
                for page in reader.pages:
                    content += page.extract_text()
        elif file_extension.lower() == '.docx':
            doc = DocxDocument(file_path)
            for para in doc.paragraphs:
                content += para.text + '\n'
        elif file_extension.lower() in ['.txt', '.text']:
            with open(file_path, 'r') as file:
                content = file.read()
        else:
            raise ValueError("Unsupported file type")
        # Check if content is large enough for summarization
        if len(content.split()) < 50:  # Example threshold
            return "Content too short for summarization."
        # Generate summary
        return summarize(content)
    except Exception as e:
        return f"An error occurred during summarization: {str(e)}"
def evaluate_formula(expression):
    # Implement logic to evaluate formula
    return sympify(expression).evalf()

# Document Summarization and Formula Builder Application

This application is a browser-based Python/Django application that allows users to upload multiple documents (PDF, Word, plain text), generates concise summaries for each document, and features a formula builder for custom financial calculations with the ability to save and reuse formulas.

## Main Functions

1. **Document Upload and Summarization**:
   - Users can upload documents in PDF, Word, or plain text formats.
   - The application extracts text from the uploaded documents and generates concise summaries.

2. **Formula Builder**:
   - Users can create custom financial formulas.
   - The application evaluates these formulas and displays the results.
   - Users can save and reuse formulas for future calculations.
