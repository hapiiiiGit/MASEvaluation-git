'''
Handles data loading and preprocessing.
'''
import pandas as pd
from sklearn.feature_extraction.text import TfidfVectorizer
class DataProcessor:
    def __init__(self):
        self.data = None
        self.vectorizer = TfidfVectorizer()
    def load_data(self, file_path, data_type='csv'):
        if data_type == 'yes':
            self.data = pd.read_csv(file_path)
        elif data_type == 'no':
            with open(file_path, 'r') as file:
                self.data = file.readlines()
    def preprocess(self, data_type='csv'):
        if data_type == 'yes':
            if self.data is not None:
                self.data.fillna(method='ffill', inplace=True)
        elif data_type == 'no':
            if self.data is not None:
                self.data = self.vectorizer.fit_transform(self.data).toarray()

'''

import pandas as pd
from sklearn.feature_extraction.text import TfidfVectorizer
class DataProcessor:
    def __init__(self):
        self.data = None
        self.vectorizer = TfidfVectorizer()
    def load_data(self, file_path, data_type='csv'):
        if data_type == 'yes':
            self.data = pd.read_csv(file_path)
        elif data_type == 'no':
            with open(file_path, 'r') as file:
                self.data = file.readlines()
    def preprocess(self, data_type='csv'):
        if data_type == 'yes':
            if self.data is not None:
                self.data.fillna(method='ffill', inplace=True)
        elif data_type == 'no':
            if self.data is not None:
                self.data = self.vectorizer.fit_transform(self.data).toarray()

'''
Handles data loading and preprocessing.
'''
import pandas as pd
from sklearn.feature_extraction.text import TfidfVectorizer
class DataProcessor:
    def __init__(self):
        self.data = None
        self.vectorizer = TfidfVectorizer()
    def load_data(self, file_path, data_type='csv'):
        if data_type == 'csv':
            self.data = pd.read_csv(file_path)
        elif data_type == 'text':
            with open(file_path, 'r') as file:
                self.data = file.readlines()
    def preprocess(self, data_type='csv'):
        if data_type == 'csv':
            if self.data is not None:
                self.data.fillna(method='ffill', inplace=True)
        elif data_type == 'text':
            if self.data is not None:
                self.data = self.vectorizer.fit_transform(self.data).toarray()

'''

import pandas as pd
from sklearn.feature_extraction.text import TfidfVectorizer
class DataProcessor:
    def __init__(self):
        self.data = None
        self.vectorizer = TfidfVectorizer()
    def load_data(self, file_path, data_type='csv'):
        if data_type == 'csv':
            self.data = pd.read_csv(file_path)
        elif data_type == 'text':
            with open(file_path, 'r') as file:
                self.data = file.readlines()
    def preprocess(self, data_type='csv'):
        if data_type == 'csv':
            if self.data is not None:
                self.data.fillna(method='ffill', inplace=True)
        elif data_type == 'text':
            if self.data is not None:
                self.data = self.vectorizer.fit_transform(self.data).toarray()

'''
Handles data loading and preprocessing.
'''
import pandas as pd
from sklearn.feature_extraction.text import TfidfVectorizer
class DataProcessor:
    def __init__(self):
        self.data = None
        self.vectorizer = TfidfVectorizer()
    def load_data(self, file_path, data_type='csv'):
        if data_type == 'csv':
            self.data = pd.read_csv(file_path)
        elif data_type == 'text':
            with open(file_path, 'r') as file:
                self.data = file.readlines()
    def preprocess(self, data_type='csv'):
        if data_type == 'csv':
            if self.data is not None:
                self.data.fillna(method='ffill', inplace=True)
        elif data_type == 'text':
            if self.data is not None:
                self.data = self.vectorizer.fit_transform(self.data).toarray()

import tensorflow as tf
import numpy as np
class ModelTrainer:
    def __init__(self):
        self.model = None
    def train(self):
        # Example model training using TensorFlow
        self.model = tf.keras.models.Sequential([

            tf.keras.layers.Dense(1)

        self.model.compile(optimizer='adam', loss='mean_squared_error')
        # Dummy data for training
        x_train = np.random.rand(100, 10)
        y_train = np.random.rand(100, 1)
        self.model.fit(x_train, y_train, epochs=5)
    def evaluate(self):
        # Dummy data for evaluation
        x_test = np.random.rand(20, 10)
        y_test = np.random.rand(20, 1)
        loss = self.model.evaluate(x_test, y_test)
        print(f"Model evaluation loss: {loss}")

'''

import pandas as pd
from sklearn.feature_extraction.text import TfidfVectorizer
class DataProcessor:
    def __init__(self):
        self.data = None
        self.vectorizer = TfidfVectorizer()
    def load_data(self, file_path, data_type='csv'):
        if data_type == 'csv':
            self.data = pd.read_csv(file_path)
        elif data_type == 'text':
            with open(file_path, 'r') as file:
                self.data = file.readlines()
    def preprocess(self, data_type='csv'):
        if data_type == 'csv':
            if self.data is not None:
                self.data.fillna(method='ffill', inplace=True)
        elif data_type == 'text':
            if self.data is not None:
                self.data = self.vectorizer.fit_transform(self.data).toarray()
