'''

import pandas as pd
import requests

class DataHandler:
    def __init__(self):
        self.data = None

    def fetch_data(self, file_path=None, api_url=None):
        if file_path:
            self.data = pd.read_csv(file_path)
        elif api_url:
            response = requests.get(api_url)
            if response.status_code == 200:
                self.data = pd.DataFrame(response.json())
            else:
                raise Exception("Failed to fetch data from API.")

    def clean_data(self):
        # Example cleaning process
        self.data.dropna(inplace=True)
        self.data.drop_duplicates(inplace=True)

    def analyze_data(self):
        # Example analysis
        self.summary = self.data.describe()

    def generate_insights(self):
        # Example insights generation
        insights = "Data contains {} rows and {} columns.".format(self.data.shape[0], self.data.shape[1])
        return insights

import pandas as pd
import requests
class DataHandler:
    def __init__(self):
        self.data = None
    def fetch_data(self, file_path=None, api_url=None):
        if file_path:
            self.data = pd.read_csv(file_path)
        elif api_url:
            response = requests.get(api_url)
            if response.status_code == 200:
                self.data = pd.DataFrame(response.json())
            else:
                raise Exception("Failed to fetch data from API.")
    def clean_data(self):
        # Example cleaning process
        self.data.dropna(inplace=True)
        self.data.drop_duplicates(inplace=True)
    def analyze_data(self):
        # Example analysis
        self.summary = self.data.describe()
    def generate_insights(self):
        # Example insights generation
        insights = "Data contains {} rows and {} columns.".format(self.data.shape[0], self.data.shape[1])
        return insights

'''
DataHandler class to handle data fetching, cleaning, and analysis.
'''
import pandas as pd
import requests
from scipy import stats

class DataHandler:
    def __init__(self):
        self.data = None
        self.summary = None
        self.correlation_matrix = None

    def fetch_data(self, file_path=None, api_url=None):
        if file_path:
            self.data = pd.read_csv(file_path)
        elif api_url:
            response = requests.get(api_url)
            if response.status_code == 200:
                self.data = pd.DataFrame(response.json())
            else:
                raise Exception("Failed to fetch data from API.")

    def clean_data(self):
        # Example cleaning process
        self.data.dropna(inplace=True)
        self.data.drop_duplicates(inplace=True)

    def analyze_data(self):
        # Basic statistical summary
        self.summary = self.data.describe()
        
        # Additional analysis: Correlation matrix
        self.correlation_matrix = self.data.corr()
        
        # Example: Hypothesis testing or trend analysis can be added here
        # For instance, using scipy.stats for t-tests or similar analyses
        # Example: t-test between two columns if applicable
        if 'column1' in self.data.columns and 'column2' in self.data.columns:
            t_stat, p_value = stats.ttest_ind(self.data['column1'], self.data['column2'])
            print(f"T-test between column1 and column2: t-statistic = {t_stat}, p-value = {p_value}")

    def generate_insights(self):
        # Example insights generation
        insights = "Data contains {} rows and {} columns.".format(self.data.shape[0], self.data.shape[1])
        if self.correlation_matrix is not None:
            insights += "\nCorrelation matrix:\n" + str(self.correlation_matrix)
        return insights

        # Basic statistical summary

from scipy import stats

'''
DataHandler class to handle data fetching, cleaning, and analysis.
'''
import pandas as pd
import requests
from scipy import stats
class DataHandler:
    def __init__(self):
        self.data = None
        self.summary = None
        self.correlation_matrix = None
    def fetch_data(self, file_path=None, api_url=None):
        if file_path:
            self.data = pd.read_csv(file_path)
        elif api_url:
            response = requests.get(api_url)
            if response.status_code == 200:
                self.data = pd.DataFrame(response.json())
            else:
                raise Exception("Failed to fetch data from API.")
    def clean_data(self):
        # Example cleaning process
        self.data.dropna(inplace=True)
        self.data.drop_duplicates(inplace=True)
    def analyze_data(self):
        # Basic statistical summary
        self.summary = self.data.describe()
        # Additional analysis: Correlation matrix
        self.correlation_matrix = self.data.corr()
        # Example: Hypothesis testing or trend analysis can be added here
        # For instance, using scipy.stats for t-tests or similar analyses
        # Example: t-test between two columns if applicable
        if 'column1' in self.data.columns and 'column2' in self.data.columns:
            t_stat, p_value = stats.ttest_ind(self.data['column1'], self.data['column2'])
            print(f"T-test between column1 and column2: t-statistic = {t_stat}, p-value = {p_value}")
    def generate_insights(self):
        # Example insights generation
        insights = "Data contains {} rows and {} columns.".format(self.data.shape[0], self.data.shape[1])
        if self.correlation_matrix is not None:
            insights += "\nCorrelation matrix:\n" + str(self.correlation_matrix)
        return insights

class Visualization:
    def plot_data(self, data):

        numerical_columns = data.select_dtypes(include=['number']).columns
        for i, column in enumerate(numerical_columns):
            plt.subplot(2, 2, i + 1)
            sns.histplot(data[column], kde=True)
            plt.title(f"Distribution of {column}")
        plt.tight_layout()
        plt.show()

'''
DataHandler class to handle data fetching, cleaning, and analysis.
'''
import pandas as pd
import requests
from scipy import stats
class DataHandler:
    def __init__(self):
        self.data = None
        self.summary = None
        self.correlation_matrix = None
    def fetch_data(self, file_path=None, api_url=None):
        if file_path:
            self.data = pd.read_csv(file_path)
        elif api_url:
            response = requests.get(api_url)
            if response.status_code == 200:
                self.data = pd.DataFrame(response.json())
            else:
                raise Exception("Failed to fetch data from API.")
    def clean_data(self):
        # Example cleaning process
        self.data.dropna(inplace=True)
        self.data.drop_duplicates(inplace=True)
    def analyze_data(self):
        # Basic statistical summary
        self.summary = self.data.describe()
        # Additional analysis: Correlation matrix
        self.correlation_matrix = self.data.corr()
        # Example: Hypothesis testing or trend analysis can be added here
        # For instance, using scipy.stats for t-tests or similar analyses
        # Example: t-test between two columns if applicable
        if 'column1' in self.data.columns and 'column2' in self.data.columns:
            t_stat, p_value = stats.ttest_ind(self.data['column1'], self.data['column2'])
            print(f"T-test between column1 and column2: t-statistic = {t_stat}, p-value = {p_value}")
    def generate_insights(self):
        # Example insights generation
        insights = "Data contains {} rows and {} columns.".format(self.data.shape[0], self.data.shape[1])
        if self.correlation_matrix is not None:
            insights += "\nCorrelation matrix:\n" + str(self.correlation_matrix)
        return insights

'''
DataHandler class to handle data fetching, cleaning, and analysis.
'''
import pandas as pd
import requests
from scipy import stats
class DataHandler:
    def __init__(self):
        self.data = None
        self.summary = None
        self.correlation_matrix = None
    def fetch_data(self, file_path=None, api_url=None):
        if file_path:
            self.data = pd.read_csv(file_path)
        elif api_url:
            response = requests.get(api_url)
            if response.status_code == 200:
                self.data = pd.DataFrame(response.json())
            else:
                raise Exception("Failed to fetch data from API.")
    def clean_data(self):
        # Example cleaning process
        self.data.dropna(inplace=True)
        self.data.drop_duplicates(inplace=True)
    def analyze_data(self):
        # Basic statistical summary
        self.summary = self.data.describe()
        # Additional analysis: Correlation matrix
        self.correlation_matrix = self.data.corr()
        # Example: Hypothesis testing or trend analysis can be added here
        # For instance, using scipy.stats for t-tests or similar analyses
        # Example: t-test between two columns if applicable
        if 'column1' in self.data.columns and 'column2' in self.data.columns:
            t_stat, p_value = stats.ttest_ind(self.data['column1'], self.data['column2'])
            print(f"T-test between column1 and column2: t-statistic = {t_stat}, p-value = {p_value}")
    def generate_insights(self):
        # Example insights generation
        insights = "Data contains {} rows and {} columns.".format(self.data.shape[0], self.data.shape[1])
        if self.correlation_matrix is not None:
            insights += "\nCorrelation matrix:\n" + str(self.correlation_matrix)
        return insights

import matplotlib.pyplot as plt
import seaborn as sns
class Visualization:
    def plot_data(self, data):
        plt.figure(figsize=(12, 8))
        numerical_columns = data.select_dtypes(include=['number']).columns
        for i, column in enumerate(numerical_columns):
            plt.subplot(2, 2, i + 1)
            sns.histplot(data[column], kde=True)
            plt.title(f"Distribution of {column}")
        plt.tight_layout()
        plt.show()
