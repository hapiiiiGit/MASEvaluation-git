'''

Handles interaction with mStock Type A User APIs, including API key generation, login, session token generation, logout, and error handling.

'''

import requests

class APIManager:
    BASE_URL = "https://api.mstock.com/type_a_user"

    def generate_api_key(self):
        response = requests.post(f"{self.BASE_URL}/generate_api_key")
        if response.status_code == 200:
            return response.json()["api_key"]
        else:
            self.handle_error(response.status_code)

    def login(self, username, password):
        response = requests.post(f"{self.BASE_URL}/login", json={"username": username, "password": password})
        if response.status_code == 200:
            return response.json()["session_token"]
        else:
            self.handle_error(response.status_code)

    def logout(self, session_token):
        response = requests.post(f"{self.BASE_URL}/logout", headers={"Authorization": f"Bearer {session_token}"})
        if response.status_code == 200:
            return True
        else:
            self.handle_error(response.status_code)

    def get_fund_details(self, session_token):
        response = requests.get(f"{self.BASE_URL}/fund_details", headers={"Authorization": f"Bearer {session_token}"})
        if response.status_code == 200:
            return response.json()
        else:
            self.handle_error(response.status_code)

    def handle_error(self, error_code):
        error_messages = {
            400: "Bad Request",
            401: "Unauthorized",
            403: "Forbidden",
            404: "Not Found",
            500: "Internal Server Error"
        }
        raise Exception(error_messages.get(error_code, "Unknown Error"))

'''
Handles interaction with mStock Type A User APIs, including API key generation, login, session token generation, logout, and error handling.
'''
import requests
class APIManager:
    BASE_URL = "https://api.mstock.com/type_a_user"
    def generate_api_key(self):
        response = requests.post(f"{self.BASE_URL}/generate_api_key")
        if response.status_code == 200:
            return response.json()["api_key"]
        else:
            self.handle_error(response.status_code)
    def login(self, username, password):
        response = requests.post(f"{self.BASE_URL}/login", json={"username": username, "password": password})
        if response.status_code == 200:
            return response.json()["session_token"]
        else:
            self.handle_error(response.status_code)
    def logout(self, session_token):
        response = requests.post(f"{self.BASE_URL}/logout", headers={"Authorization": f"Bearer {session_token}"})
        if response.status_code == 200:
            return True
        else:
            self.handle_error(response.status_code)
    def get_fund_details(self, session_token):
        response = requests.get(f"{self.BASE_URL}/fund_details", headers={"Authorization": f"Bearer {session_token}"})
        if response.status_code == 200:
            return response.json()
        else:
            self.handle_error(response.status_code)
    def handle_error(self, error_code):
        error_messages = {
            400: "Bad Request",
            401: "Unauthorized",
            403: "Forbidden",
            404: "Not Found",
            500: "Internal Server Error"
        }
        raise Exception(error_messages.get(error_code, "Unknown Error"))

    def __init__(self):
        self.sessions = {}
    def create_session(self, user_id, session_token):
        self.sessions[session_token] = user_id
    def validate_session(self, session_token):
        return session_token in self.sessions
    def end_session(self, session_token):
        if session_token in self.sessions:

'''
Handles interaction with mStock Type A User APIs, including API key generation, login, session token generation, logout, and error handling.
'''
import requests
# Define custom exceptions
class BadRequestError(Exception):
    pass
class UnauthorizedError(Exception):
    pass
class ForbiddenError(Exception):
    pass
class NotFoundError(Exception):
    pass
class InternalServerError(Exception):
    pass
class APIManager:
    BASE_URL = "https://api.mstock.com/type_a_user"
    def generate_api_key(self):
        response = requests.post(f"{self.BASE_URL}/generate_api_key")
        if response.status_code == 200:
            return response.json()["api_key"]
        else:
            self.handle_error(response.status_code)
    def login(self, username, password):
        response = requests.post(f"{self.BASE_URL}/login", json={"username": username, "password": password})
        if response.status_code == 200:
            return response.json()["session_token"]
        else:
            self.handle_error(response.status_code)
    def logout(self, session_token):
        response = requests.post(f"{self.BASE_URL}/logout", headers={"Authorization": f"Bearer {session_token}"})
        if response.status_code == 200:
            return True
        else:
            self.handle_error(response.status_code)
    def get_fund_details(self, session_token):
        response = requests.get(f"{self.BASE_URL}/fund_details", headers={"Authorization": f"Bearer {session_token}"})
        if response.status_code == 200:
            return response.json()
        else:
            self.handle_error(response.status_code)
    def handle_error(self, error_code):
        error_messages = {
            400: "Bad Request",
            401: "Unauthorized",
            403: "Forbidden",
            404: "Not Found",
            500: "Internal Server Error"
        }
        error_classes = {
            400: BadRequestError,
            401: UnauthorizedError,
            403: ForbiddenError,
            404: NotFoundError,
            500: InternalServerError
        }
        error_class = error_classes.get(error_code, Exception)
        raise error_class(f"Error {error_code}: {error_messages.get(error_code, 'Unknown Error')}")

'''
Manages user sessions, including creation, validation, and termination.

class SessionManager:
    def __init__(self):
        self.sessions = {}
    def create_session(self, user_id, session_token):
        self.sessions[session_token] = user_id
    def validate_session(self, session_token):
        return session_token in self.sessions
    def end_session(self, session_token):
        if session_token in self.sessions:
            del self.sessions[session_token]

import requests
# Define custom exceptions
class BadRequestError(Exception):

    pass

    pass

    pass

    BASE_URL = "https://api.mstock.com/type_a_user"
    def generate_api_key(self):
        response = requests.post(f"{self.BASE_URL}/generate_api_key")
        if response.status_code == 200:
            return response.json()["api_key"]
        else:
            self.handle_error(response.status_code)
    def login(self, username, password):
        response = requests.post(f"{self.BASE_URL}/login", json={"username": username, "password": password})
        if response.status_code == 200:
            return response.json()["session_token"]
        else:
            self.handle_error(response.status_code)
    def logout(self, session_token):
        response = requests.post(f"{self.BASE_URL}/logout", headers={"Authorization": f"Bearer {session_token}"})
        if response.status_code == 200:
            return True
        else:
            self.handle_error(response.status_code)
    def get_fund_details(self, session_token):
        response = requests.get(f"{self.BASE_URL}/fund_details", headers={"Authorization": f"Bearer {session_token}"})
        if response.status_code == 200:
            return response.json()
        else:
            self.handle_error(response.status_code)
    def handle_error(self, error_code):
        error_messages = {
            400: "Bad Request",
            401: "Unauthorized",
            403: "Forbidden",
            404: "Not Found",
            500: "Internal Server Error"

'''
Handles interaction with mStock Type A User APIs, including API key generation, login, session token generation, logout, and error handling.
'''
import requests
# Define custom exceptions
class BadRequestError(Exception):
    pass
class UnauthorizedError(Exception):
    pass
class ForbiddenError(Exception):
    pass
class NotFoundError(Exception):
    pass
class InternalServerError(Exception):
    pass
class APIManager:
    BASE_URL = "https://api.mstock.com/type_a_user"
    def generate_api_key(self):
        try:
            response = requests.post(f"{self.BASE_URL}/generate_api_key")
            if response.status_code == 200:
                return response.json()["api_key"]
            else:
                self.handle_error(response.status_code)
        except Exception as e:
            print(f"API Key generation failed: {str(e)}")
            return None
    def login(self, username, password):
        try:
            response = requests.post(f"{self.BASE_URL}/login", json={"username": username, "password": password})
            if response.status_code == 200:
                return response.json()["session_token"]
            else:
                self.handle_error(response.status_code)
        except Exception as e:
            print(f"Login failed: {str(e)}")
            return None
    def logout(self, session_token):
        try:
            response = requests.post(f"{self.BASE_URL}/logout", headers={"Authorization": f"Bearer {session_token}"})
            if response.status_code == 200:
                return True
            else:
                self.handle_error(response.status_code)
        except Exception as e:
            print(f"Logout failed: {str(e)}")
            return False
    def get_fund_details(self, session_token):
        try:
            response = requests.get(f"{self.BASE_URL}/fund_details", headers={"Authorization": f"Bearer {session_token}"})
            if response.status_code == 200:
                return response.json()
            else:
                self.handle_error(response.status_code)
        except Exception as e:
            print(f"Fetching fund details failed: {str(e)}")
            return None
    def handle_error(self, error_code):
        error_messages = {
            400: "Bad Request",
            401: "Unauthorized",
            403: "Forbidden",
            404: "Not Found",
            500: "Internal Server Error"
        }
        error_classes = {
            400: BadRequestError,
            401: UnauthorizedError,
            403: ForbiddenError,
            404: NotFoundError,
            500: InternalServerError
        }
        error_class = error_classes.get(error_code, Exception)
        raise error_class(f"Error {error_code}: {error_messages.get(error_code, 'Unknown Error')}")

'''
Manages user sessions, including creation, validation, and termination.
'''
class SessionManager:
    def __init__(self):
        self.sessions = {}
    def create_session(self, user_id, session_token):
        self.sessions[session_token] = user_id
    def validate_session(self, session_token):
        return session_token in self.sessions
    def end_session(self, session_token):
        if session_token in self.sessions:
            del self.sessions[session_token]

'''

import tkinter as tk
from tkinter import messagebox
class GUIManager:
    def __init__(self, api_manager, session_manager):
        self.api_manager = api_manager
        self.session_manager = session_manager
    def create_login_window(self):
        def login():
            username = username_entry.get()
            password = password_entry.get()
            session_token = self.api_manager.login(username, password)
            if session_token:
                self.session_manager.create_session(username, session_token)
                self.create_dashboard_window(session_token)
                login_window.destroy()
            else:
                self.show_error_message("Login failed. Please check your credentials.")
        login_window = tk.Tk()
        login_window.title("Login")
        tk.Label(login_window, text="Username").pack()
        username_entry = tk.Entry(login_window)
        username_entry.pack()
        tk.Label(login_window, text="Password").pack()
        password_entry = tk.Entry(login_window, show="*")
        password_entry.pack()
        tk.Button(login_window, text="Login", command=login).pack()
        login_window.mainloop()
    def create_dashboard_window(self, session_token):
        def logout():
            if self.api_manager.logout(session_token):
                self.session_manager.end_session(session_token)
                dashboard_window.destroy()
                self.create_login_window()
            else:
                self.show_error_message("Logout failed.")
        def fetch_fund_details():
            fund_details = self.api_manager.get_fund_details(session_token)
            if fund_details:
                fund_label.config(text=str(fund_details))
            else:
                self.show_error_message("Failed to fetch fund details.")
        dashboard_window = tk.Tk()
        dashboard_window.title("Dashboard")
        tk.Button(dashboard_window, text="Fetch Fund Details", command=fetch_fund_details).pack()
        fund_label = tk.Label(dashboard_window, text="")
        fund_label.pack()
        tk.Button(dashboard_window, text="Logout", command=logout).pack()
        dashboard_window.mainloop()
    def show_error_message(self, message):
        messagebox.showerror("Error", message)

'''
Handles interaction with mStock Type A User APIs, including API key generation, login, session token generation, logout, and error handling.
'''
import requests
# Define custom exceptions
class BadRequestError(Exception):
    pass
class UnauthorizedError(Exception):
    pass
class ForbiddenError(Exception):
    pass
class NotFoundError(Exception):
    pass
class InternalServerError(Exception):
    pass
class APIManager:
    BASE_URL = "https://api.mstock.com/type_a_user"
    def generate_api_key(self):
        try:
            response = requests.post(f"{self.BASE_URL}/generate_api_key")
            if response.status_code == 200:
                return response.json()["api_key"]
            else:
                self.handle_error(response.status_code)
        except Exception as e:
            print(f"API Key generation failed: {str(e)}")
            return None
    def login(self, username, password):
        try:
            response = requests.post(f"{self.BASE_URL}/login", json={"username": username, "password": password})
            if response.status_code == 200:
                return response.json()["session_token"]
            else:
                self.handle_error(response.status_code)
        except Exception as e:
            print(f"Login failed: {str(e)}")
            return None
    def logout(self, session_token):
        try:
            response = requests.post(f"{self.BASE_URL}/logout", headers={"Authorization": f"Bearer {session_token}"})
            if response.status_code == 200:
                return True
            else:
                self.handle_error(response.status_code)
        except Exception as e:
            print(f"Logout failed: {str(e)}")
            return False
    def get_fund_details(self, session_token):
        try:
            response = requests.get(f"{self.BASE_URL}/fund_details", headers={"Authorization": f"Bearer {session_token}"})
            if response.status_code == 200:
                return response.json()
            else:
                self.handle_error(response.status_code)
        except Exception as e:
            print(f"Fetching fund details failed: {str(e)}")
            return None
    def handle_error(self, error_code):
        error_messages = {
            400: "Bad Request",
            401: "Unauthorized",
            403: "Forbidden",
            404: "Not Found",
            500: "Internal Server Error"
        }
        error_classes = {
            400: BadRequestError,
            401: UnauthorizedError,
            403: ForbiddenError,
            404: NotFoundError,
            500: InternalServerError
        }
        error_class = error_classes.get(error_code, Exception)
        raise error_class(f"Error {error_code}: {error_messages.get(error_code, 'Unknown Error')}")

class SessionManager:
    def __init__(self):
        self.sessions = {}
    def create_session(self, user_id, session_token):
        self.sessions[session_token] = user_id
    def validate_session(self, session_token):
        return session_token in self.sessions
    def end_session(self, session_token):
        if session_token in self.sessions:

# Define custom exceptions
class BadRequestError(Exception):
    pass

    pass

    pass

    pass

    pass
class APIManager:
    BASE_URL = "https://api.mstock.com/type_a_user"
    def generate_api_key(self):
        try:
            response = requests.post(f"{self.BASE_URL}/generate_api_key")
            if response.status_code == 200:
                return response.json()["api_key"]
            else:
                self.handle_error(response.status_code)
        except Exception as e:
            logging.error(f"API Key generation failed: {str(e)}")
            return None
    def login(self, username, password):
        try:
            response = requests.post(f"{self.BASE_URL}/login", json={"username": username, "password": password})
            if response.status_code == 200:

            else:

'''

'''
import requests
import logging
# Configure logging
logging.basicConfig(level=logging.INFO, format='%(asctime)s - %(levelname)s - %(message)s')
# Define custom exceptions
class BadRequestError(Exception):
    pass
class UnauthorizedError(Exception):
    pass
class ForbiddenError(Exception):
    pass
class NotFoundError(Exception):
    pass
class InternalServerError(Exception):
    pass
class APIManager:
    BASE_URL = "https://api.mstock.com/type_a_user"
    def generate_api_key(self):
        try:
            response = requests.post(f"{self.BASE_URL}/generate_api_key")
            if response.status_code == 200:
                return response.json()["api_key"]
            else:
                self.handle_error(response.status_code)
        except Exception as e:
            logging.error(f"API Key generation failed: {str(e)}")
            return None
    def login(self, username, password):
        try:
            response = requests.post(f"{self.BASE_URL}/login", json={"username": username, "password": password})
            if response.status_code == 200:
                return response.json()["session_token"]
            else:
                self.handle_error(response.status_code)
        except Exception as e:
            logging.error(f"Login failed: {str(e)}")
            return None
    def logout(self, session_token):
        try:
            response = requests.post(f"{self.BASE_URL}/logout", headers={"Authorization": f"Bearer {session_token}"})
            if response.status_code == 200:
                return True
            else:
                self.handle_error(response.status_code)
        except Exception as e:
            logging.error(f"Logout failed: {str(e)}")
            return False
    def get_fund_details(self, session_token):
        try:
            response = requests.get(f"{self.BASE_URL}/fund_details", headers={"Authorization": f"Bearer {session_token}"})
            if response.status_code == 200:
                return response.json()
            else:
                self.handle_error(response.status_code)
        except Exception as e:
            logging.error(f"Fetching fund details failed: {str(e)}")
            return None
    def handle_error(self, error_code):
        error_messages = {
            400: "Bad Request",
            401: "Unauthorized",
            403: "Forbidden",
            404: "Not Found",
            500: "Internal Server Error"
        }
        error_classes = {
            400: BadRequestError,
            401: UnauthorizedError,
            403: ForbiddenError,
            404: NotFoundError,
            500: InternalServerError
        }
        error_class = error_classes.get(error_code, Exception)
        logging.error(f"Error {error_code}: {error_messages.get(error_code, 'Unknown Error')}")
        raise error_class(f"Error {error_code}: {error_messages.get(error_code, 'Unknown Error')}")

'''
Manages user sessions, including creation, validation, and termination.
'''
class SessionManager:
    def __init__(self):
        self.sessions = {}
    def create_session(self, user_id, session_token):
        self.sessions[session_token] = user_id
    def validate_session(self, session_token):
        return session_token in self.sessions
    def end_session(self, session_token):
        if session_token in self.sessions:
            del self.sessions[session_token]

'''
Handles interaction with mStock Type A User APIs, including API key generation, login, session token generation, logout, and error handling.
'''
import requests
import logging
# Configure logging
logging.basicConfig(level=logging.INFO, format='%(asctime)s - %(levelname)s - %(message)s')
# Define custom exceptions
class BadRequestError(Exception):
    pass
class UnauthorizedError(Exception):
    pass
class ForbiddenError(Exception):
    pass
class NotFoundError(Exception):
    pass
class InternalServerError(Exception):
    pass
class APIManager:
    BASE_URL = "https://api.mstock.com/type_a_user"
    def generate_api_key(self):
        try:
            response = requests.post(f"{self.BASE_URL}/generate_api_key")
            if response.status_code == 200:
                return response.json()["api_key"]
            else:
                self.handle_error(response.status_code)
        except Exception as e:
            logging.error(f"API Key generation failed: {str(e)}")
            return None
    def login(self, username, password):
        try:
            response = requests.post(f"{self.BASE_URL}/login", json={"username": username, "password": password})
            if response.status_code == 200:
                return response.json()["session_token"]
            else:
                self.handle_error(response.status_code)
        except Exception as e:
            logging.error(f"Login failed: {str(e)}")
            return None
    def logout(self, session_token):
        try:
            response = requests.post(f"{self.BASE_URL}/logout", headers={"Authorization": f"Bearer {session_token}"})
            if response.status_code == 200:
                return True
            else:
                self.handle_error(response.status_code)
        except Exception as e:
            logging.error(f"Logout failed: {str(e)}")
            return False
    def get_fund_details(self, session_token):
        try:
            response = requests.get(f"{self.BASE_URL}/fund_details", headers={"Authorization": f"Bearer {session_token}"})
            if response.status_code == 200:
                return response.json()
            else:
                self.handle_error(response.status_code)
        except Exception as e:
            logging.error(f"Fetching fund details failed: {str(e)}")
            return None
    def handle_error(self, error_code):
        error_messages = {
            400: "Bad Request",
            401: "Unauthorized",
            403: "Forbidden",
            404: "Not Found",
            500: "Internal Server Error"
        }
        error_classes = {
            400: BadRequestError,
            401: UnauthorizedError,
            403: ForbiddenError,
            404: NotFoundError,
            500: InternalServerError
        }
        error_class = error_classes.get(error_code, Exception)
        logging.error(f"Error {error_code}: {error_messages.get(error_code, 'Unknown Error')}")
        raise error_class(f"Error {error_code}: {error_messages.get(error_code, 'Unknown Error')}")

'''

class SessionManager:
    def __init__(self):
        self.sessions = {}
    def create_session(self, user_id, session_token):
        self.sessions[session_token] = user_id
    def validate_session(self, session_token):
        return session_token in self.sessions
    def end_session(self, session_token):
        if session_token in self.sessions:
            del self.sessions[session_token]

class GUIManager:

            username = username_entry.get()
