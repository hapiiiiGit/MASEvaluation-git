class DataAnalyzer:
    def analyze_data(self, data):
        result = []
        for token in data:
            prices = token['prices']
            exchanges = list(prices.keys())
            if len(exchanges) < 2:
                continue
            min_price = min(prices.values())
            max_price = max(prices.values())
            price_diff = ((max_price - min_price) / min_price) * 100
            if price_diff > 1.8:
                result.append({
                    'name': token['name'],
                    'contract': token['contract'],
                    'lowest_exchange': exchanges[prices.values().index(min_price)],
                    'highest_exchange': exchanges[prices.values().index(max_price)],
                    'price_diff': price_diff
                })
        return result

'''

class DataAnalyzer:
    def analyze_data(self, data):
        result = []
        for token in data:
            prices = token['prices']
            exchanges = list(prices.keys())
            if len(exchanges) < 2:
                continue
            min_price = min(prices.values())
            max_price = max(prices.values())
            price_diff = ((max_price - min_price) / min_price) * 100
            if price_diff > 1.8:
                result.append({
                    'name': token['name'],
                    'contract': token['contract'],
                    'lowest_exchange': exchanges[prices.values().index(min_price)],
                    'highest_exchange': exchanges[prices.values().index(max_price)],
                    'price_diff': price_diff
                })
        return result

'''
Module to analyze token data and find price differences.
'''
class DataAnalyzer:
    def analyze_data(self, data):
        result = []
        for token in data:
            prices = token['prices']
            exchanges = list(prices.keys())
            if len(exchanges) < 2:
                continue
            min_price = min(prices.values())
            max_price = max(prices.values())
            price_diff = ((max_price - min_price) / min_price) * 100
            if price_diff > 1.8:
                result.append({
                    'name': token['name'],
                    'contract': token['contract'],
                    'lowest_exchange': exchanges[prices.values().index(min_price)],
                    'highest_exchange': exchanges[prices.values().index(max_price)],
                    'price_diff': price_diff
                })
        return result

                f"Contract: {token['contract']}\n"

                f"Highest Exchange: {token['highest_exchange']}\n"
                f"Price Difference: {token['price_diff']:.2f}%\n\n"
            )

class DataAnalyzer:
    def analyze_data(self, data):
        result = []
        for token in data:
            prices = token['prices']
            exchanges = list(prices.keys())
            if len(exchanges) < 2:
                continue
            min_price = min(prices.values())
            max_price = max(prices.values())
            price_diff = ((max_price - min_price) / min_price) * 100
            if price_diff > 1.8:
                result.append({
                    'name': token['name'],
                    'contract': token['contract'],
                    'lowest_exchange': exchanges[prices.values().index(min_price)],
                    'highest_exchange': exchanges[prices.values().index(max_price)],
                    'price_diff': price_diff
                })
        return result

class DataAnalyzer:
    def analyze_data(self, data):
        result = []
        for token in data:
            prices = token['prices']
            exchanges = list(prices.keys())
            if len(exchanges) < 2:
                continue
            min_price = min(prices.values())
            max_price = max(prices.values())
            price_diff = ((max_price - min_price) / min_price) * 100
            if price_diff > 1.8:
                result.append({
                    'name': token['name'],
                    'contract': token['contract'],
                    'lowest_exchange': exchanges[prices.values().index(min_price)],
                    'highest_exchange': exchanges[prices.values().index(max_price)],
                    'price_diff': price_diff
                })
        return result

'''
Module to analyze token data and find price differences.
'''
class DataAnalyzer:
    def analyze_data(self, data):
        result = []
        for token in data:
            prices = token['prices']
            exchanges = list(prices.keys())
            if len(exchanges) < 2:
                continue
            min_price = min(prices.values())
            max_price = max(prices.values())
            price_diff = ((max_price - min_price) / min_price) * 100
            if price_diff > 1.8:
                result.append({
                    'name': token['name'],
                    'contract': token['contract'],
                    'lowest_exchange': exchanges[prices.values().index(min_price)],
                    'highest_exchange': exchanges[prices.values().index(max_price)],
                    'price_diff': price_diff
                })
        return result

'''
Module to fetch token data from Uniswap, PancakeSwap, and Sushiswap.

import requests
class DataFetcher:
    def fetch_data(self):
        # Fetch data from Uniswap
        uniswap_query = """
        {
          tokens(first: 5) {
            id
            symbol
            derivedETH
          }
        }
        """
        uniswap_response = requests.post(
            'https://api.thegraph.com/subgraphs/name/uniswap/uniswap-v2',
            json={'query': uniswap_query}
        )
        uniswap_data = uniswap_response.json().get('data', {}).get('tokens', [])
        # Fetch data from PancakeSwap
        pancakeswap_query = """
        {
          tokens(first: 5) {
            id
            symbol
            derivedBNB
          }
        }
        """
        pancakeswap_response = requests.post(
            'https://bsc.streamingfast.io/subgraphs/name/pancakeswap/exchange-v2',
            json={'query': pancakeswap_query}
        )
        pancakeswap_data = pancakeswap_response.json().get('data', {}).get('tokens', [])
        # Fetch data from Sushiswap
        sushiswap_query = """
        {
          tokens(first: 5) {
            id
            symbol
            derivedETH
          }
        }
        """
        sushiswap_response = requests.post(
            'https://api.thegraph.com/subgraphs/name/sushiswap/exchange',
            json={'query': sushiswap_query}
        )
        sushiswap_data = sushiswap_response.json().get('data', {}).get('tokens', [])
        # Combine and format data
        combined_data = self.format_data(uniswap_data, pancakeswap_data, sushiswap_data)
        return combined_data
    def format_data(self, uniswap_data, pancakeswap_data, sushiswap_data):
        combined_data = []
        # Create a dictionary to store tokens by their contract address
        token_dict = {}
        # Helper function to add token data to the dictionary
        def add_token_data(exchange_name, token_data, price_key):
            for token in token_data:
                contract = token['id']
                if contract not in token_dict:
                    token_dict[contract] = {
                        'name': token['symbol'],
                        'contract': contract,
                        'prices': {}
                    }
                token_dict[contract]['prices'][exchange_name] = token[price_key]
        # Add data from each exchange
        add_token_data('Uniswap', uniswap_data, 'derivedETH')
        add_token_data('PancakeSwap', pancakeswap_data, 'derivedBNB')
        add_token_data('Sushiswap', sushiswap_data, 'derivedETH')
        # Convert the dictionary to a list
        for token in token_dict.values():
            combined_data.append(token)
        return combined_data

'''
Module to analyze token data and find price differences.
'''
class DataAnalyzer:
    def analyze_data(self, data):
        result = []
        for token in data:
            prices = token['prices']
            exchanges = list(prices.keys())
            if len(exchanges) < 2:
                continue
            min_price = min(prices.values())
            max_price = max(prices.values())
            price_diff = ((max_price - min_price) / min_price) * 100
            if price_diff > 1.8:
                result.append({
                    'name': token['name'],
                    'contract': token['contract'],
                    'lowest_exchange': exchanges[prices.values().index(min_price)],
                    'highest_exchange': exchanges[prices.values().index(max_price)],
                    'price_diff': price_diff
                })
        return result

'''
Module to fetch token data from Uniswap, PancakeSwap, and Sushiswap.
'''
import requests
class DataFetcher:
    def fetch_data(self):
        # Fetch data from Uniswap
        uniswap_query = """
        {
          tokens(first: 5) {
            id
            symbol
            derivedETH
          }
        }
        """
        uniswap_response = requests.post(
            'https://api.thegraph.com/subgraphs/name/uniswap/uniswap-v2',
            json={'query': uniswap_query}
        )
        uniswap_data = uniswap_response.json().get('data', {}).get('tokens', [])
        # Fetch data from PancakeSwap
        pancakeswap_query = """
        {
          tokens(first: 5) {
            id
            symbol
            derivedBNB
          }
        }
        """
        pancakeswap_response = requests.post(
            'https://bsc.streamingfast.io/subgraphs/name/pancakeswap/exchange-v2',
            json={'query': pancakeswap_query}
        )
        pancakeswap_data = pancakeswap_response.json().get('data', {}).get('tokens', [])
        # Fetch data from Sushiswap
        sushiswap_query = """
        {
          tokens(first: 5) {
            id
            symbol
            derivedETH
          }
        }
        """
        sushiswap_response = requests.post(
            'https://api.thegraph.com/subgraphs/name/sushiswap/exchange',
            json={'query': sushiswap_query}
        )
        sushiswap_data = sushiswap_response.json().get('data', {}).get('tokens', [])
        # Combine and format data
        combined_data = self.format_data(uniswap_data, pancakeswap_data, sushiswap_data)
        return combined_data
    def format_data(self, uniswap_data, pancakeswap_data, sushiswap_data):
        combined_data = []
        # Create a dictionary to store tokens by their contract address
        token_dict = {}
        # Helper function to add token data to the dictionary
        def add_token_data(exchange_name, token_data, price_key):
            for token in token_data:
                contract = token['id']
                if contract not in token_dict:
                    token_dict[contract] = {
                        'name': token['symbol'],
                        'contract': contract,
                        'prices': {}
                    }
                token_dict[contract]['prices'][exchange_name] = token[price_key]
        # Add data from each exchange
        add_token_data('Uniswap', uniswap_data, 'derivedETH')
        add_token_data('PancakeSwap', pancakeswap_data, 'derivedBNB')
        add_token_data('Sushiswap', sushiswap_data, 'derivedETH')
        # Convert the dictionary to a list
        for token in token_dict.values():
            combined_data.append(token)
        return combined_data

'''
Module to analyze token data and find price differences.
'''
class DataAnalyzer:
    def analyze_data(self, data):
        result = []
        for token in data:
            prices = token['prices']
            exchanges = list(prices.keys())
            if len(exchanges) < 2:
                continue
            min_price = min(prices.values())
            max_price = max(prices.values())
            price_diff = ((max_price - min_price) / min_price) * 100
            if price_diff > 1.8:
                result.append({
                    'name': token['name'],
                    'contract': token['contract'],
                    'lowest_exchange': exchanges[prices.values().index(min_price)],
                    'highest_exchange': exchanges[prices.values().index(max_price)],
                    'price_diff': price_diff
                })
        return result

'''
Module to fetch token data from Uniswap, PancakeSwap, and Sushiswap.
'''
import requests
class DataFetcher:
    def fetch_data(self):
        # Fetch data from Uniswap
        uniswap_query = """
        {
          tokens(first: 5) {
            id
            symbol
            derivedETH
          }
        }
        """
        uniswap_response = requests.post(
            'https://api.thegraph.com/subgraphs/name/uniswap/uniswap-v2',
            json={'query': uniswap_query}
        )
        uniswap_data = uniswap_response.json().get('data', {}).get('tokens', [])
        # Fetch data from PancakeSwap
        pancakeswap_query = """
        {
          tokens(first: 5) {
            id
            symbol
            derivedBNB
          }
        }
        """
        pancakeswap_response = requests.post(
            'https://bsc.streamingfast.io/subgraphs/name/pancakeswap/exchange-v2',
            json={'query': pancakeswap_query}
        )
        pancakeswap_data = pancakeswap_response.json().get('data', {}).get('tokens', [])
        # Fetch data from Sushiswap
        sushiswap_query = """
        {
          tokens(first: 5) {
            id
            symbol
            derivedETH
          }
        }
        """
        sushiswap_response = requests.post(
            'https://api.thegraph.com/subgraphs/name/sushiswap/exchange',
            json={'query': sushiswap_query}
        )
        sushiswap_data = sushiswap_response.json().get('data', {}).get('tokens', [])
        # Combine and format data
        combined_data = self.format_data(uniswap_data, pancakeswap_data, sushiswap_data)
        return combined_data
    def format_data(self, uniswap_data, pancakeswap_data, sushiswap_data):
        combined_data = []
        # Create a dictionary to store tokens by their contract address
        token_dict = {}
        # Helper function to add token data to the dictionary
        def add_token_data(exchange_name, token_data, price_key):
            for token in token_data:
                contract = token['id']
                if contract not in token_dict:
                    token_dict[contract] = {
                        'name': token['symbol'],
                        'contract': contract,
                        'prices': {}
                    }
                token_dict[contract]['prices'][exchange_name] = token[price_key]
        # Add data from each exchange
        add_token_data('Uniswap', uniswap_data, 'derivedETH')
        add_token_data('PancakeSwap', pancakeswap_data, 'derivedBNB')
        add_token_data('Sushiswap', sushiswap_data, 'derivedETH')
        # Convert the dictionary to a list
        for token in token_dict.values():
            combined_data.append(token)
        return combined_data

'''
Module to analyze token data and find price differences.
'''
class DataAnalyzer:
    def analyze_data(self, data):
        result = []
        for token in data:
            prices = token['prices']
            exchanges = list(prices.keys())
            if len(exchanges) < 2:
                continue
            min_price = min(prices.values())
            max_price = max(prices.values())
            price_diff = ((max_price - min_price) / min_price) * 100
            if price_diff > 1.8:
                result.append({
                    'name': token['name'],
                    'contract': token['contract'],
                    'lowest_exchange': exchanges[list(prices.values()).index(min_price)],
                    'highest_exchange': exchanges[list(prices.values()).index(max_price)],
                    'price_diff': price_diff
                })
        return result

import tkinter as tk
class Dashboard:
    def __init__(self, root):
        self.root = root
        self.frame = tk.Frame(root)
        self.frame.pack()

        self.data_display.pack()
    def update_dashboard(self, data):
        self.data_display.delete(1.0, tk.END)
        for token in data:
            display_text = (
                f"Name: {token['name']}\n"
                f"Contract: {token['contract']}\n"
                f"Lowest Exchange: {token['lowest_exchange']}\n"
                f"Highest Exchange: {token['highest_exchange']}\n"
                f"Price Difference: {token['price_diff']:.2f}%\n\n"
            )
            self.data_display.insert(tk.END, display_text)

'''
Module to fetch token data from Uniswap, PancakeSwap, and Sushiswap.

import requests
class DataFetcher:
    def fetch_data(self):
        # Fetch data from Uniswap
        uniswap_query = """
        {
          tokens(first: 5) {
            id
            symbol
            derivedETH
          }
        }
        """
        uniswap_response = requests.post(
            'https://api.thegraph.com/subgraphs/name/uniswap/uniswap-v2',
            json={'query': uniswap_query}
        )
        uniswap_data = uniswap_response.json().get('data', {}).get('tokens', [])
        # Fetch data from PancakeSwap
        pancakeswap_query = """
        {
          tokens(first: 5) {
            id
            symbol
            derivedBNB
          }
        }
        """
        pancakeswap_response = requests.post(
            'https://bsc.streamingfast.io/subgraphs/name/pancakeswap/exchange-v2',
            json={'query': pancakeswap_query}
        )
        pancakeswap_data = pancakeswap_response.json().get('data', {}).get('tokens', [])
        # Fetch data from Sushiswap
        sushiswap_query = """
        {
          tokens(first: 5) {
            id
            symbol
            derivedETH
          }
        }
        """
        sushiswap_response = requests.post(
            'https://api.thegraph.com/subgraphs/name/sushiswap/exchange',
            json={'query': sushiswap_query}
        )
        sushiswap_data = sushiswap_response.json().get('data', {}).get('tokens', [])
        # Combine and format data
        combined_data = self.format_data(uniswap_data, pancakeswap_data, sushiswap_data)
        return combined_data
    def format_data(self, uniswap_data, pancakeswap_data, sushiswap_data):
        combined_data = []
        # Create a dictionary to store tokens by their contract address
        token_dict = {}
        # Helper function to add token data to the dictionary
        def add_token_data(exchange_name, token_data, price_key):
            for token in token_data:
                contract = token['id']
                if contract not in token_dict:
                    token_dict[contract] = {
                        'name': token['symbol'],
                        'contract': contract,
                        'prices': {}
                    }
                token_dict[contract]['prices'][exchange_name] = token[price_key]
        # Add data from each exchange
        add_token_data('Uniswap', uniswap_data, 'derivedETH')
        add_token_data('PancakeSwap', pancakeswap_data, 'derivedBNB')
        add_token_data('Sushiswap', sushiswap_data, 'derivedETH')
        # Convert the dictionary to a list
        for token in token_dict.values():
            combined_data.append(token)
        return combined_data

'''
Module to analyze token data and find price differences.
'''
class DataAnalyzer:
    def analyze_data(self, data):
        result = []
        for token in data:
            prices = token['prices']
            exchanges = list(prices.keys())
            if len(exchanges) < 2:
                continue
            min_price = min(prices.values())
            max_price = max(prices.values())
            price_diff = ((max_price - min_price) / min_price) * 100
            if price_diff > 1.8:
                result.append({
                    'name': token['name'],
                    'contract': token['contract'],
                    'lowest_exchange': exchanges[list(prices.values()).index(min_price)],
                    'highest_exchange': exchanges[list(prices.values()).index(max_price)],
                    'price_diff': price_diff
                })
        return result

'''
Module to fetch token data from Uniswap, PancakeSwap, and Sushiswap.

import requests
class DataFetcher:
    def fetch_data(self):
        # Fetch data from Uniswap
        uniswap_query = """
        {
          tokens(first: 5) {
            id
            symbol
            derivedETH
          }
        }
        """
        uniswap_response = requests.post(
            'https://api.thegraph.com/subgraphs/name/uniswap/uniswap-v2',
            json={'query': uniswap_query}
        )
        uniswap_data = uniswap_response.json().get('data', {}).get('tokens', [])
        # Fetch data from PancakeSwap
        pancakeswap_query = """
        {
          tokens(first: 5) {
            id
            symbol
            derivedBNB
          }
        }
        """
        pancakeswap_response = requests.post(
            'https://bsc.streamingfast.io/subgraphs/name/pancakeswap/exchange-v2',
            json={'query': pancakeswap_query}
        )
        pancakeswap_data = pancakeswap_response.json().get('data', {}).get('tokens', [])
        # Fetch data from Sushiswap
        sushiswap_query = """
        {
          tokens(first: 5) {
            id
            symbol
            derivedETH
          }
        }
        """
        sushiswap_response = requests.post(
            'https://api.thegraph.com/subgraphs/name/sushiswap/exchange',
            json={'query': sushiswap_query}
        )
        sushiswap_data = sushiswap_response.json().get('data', {}).get('tokens', [])
        # Combine and format data
        combined_data = self.format_data(uniswap_data, pancakeswap_data, sushiswap_data)
        return combined_data
    def format_data(self, uniswap_data, pancakeswap_data, sushiswap_data):
        combined_data = []
        # Create a dictionary to store tokens by their contract address
        token_dict = {}
        # Helper function to add token data to the dictionary
        def add_token_data(exchange_name, token_data, price_key):
            for token in token_data:
                contract = token['id']
                if contract not in token_dict:
                    token_dict[contract] = {
                        'name': token['symbol'],
                        'contract': contract,
                        'prices': {}
                    }
                token_dict[contract]['prices'][exchange_name] = token[price_key]
        # Add data from each exchange
        add_token_data('Uniswap', uniswap_data, 'derivedETH')
        add_token_data('PancakeSwap', pancakeswap_data, 'derivedBNB')
        add_token_data('Sushiswap', sushiswap_data, 'derivedETH')
        # Convert the dictionary to a list
        for token in token_dict.values():
            combined_data.append(token)
        return combined_data

'''
Module to analyze token data and find price differences.
'''
class DataAnalyzer:
    def analyze_data(self, data):
        result = []
        for token in data:
            prices = token['prices']
            exchanges = list(prices.keys())
            if len(exchanges) < 2:
                continue
            min_price = min(prices.values())
            max_price = max(prices.values())
            price_diff = ((max_price - min_price) / min_price) * 100
            if price_diff > 1.8:
                result.append({
                    'name': token['name'],
                    'contract': token['contract'],
                    'lowest_exchange': exchanges[list(prices.values()).index(min_price)],
                    'highest_exchange': exchanges[list(prices.values()).index(max_price)],
                    'price_diff': price_diff
                })
        return result
