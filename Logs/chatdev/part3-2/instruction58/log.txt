import requests
import csv
from random import choice
class JobScraper:
    def __init__(self, proxies):
        self.proxies = proxies
    def scrape_jobs(self, zip_code, proxy):
        url = f"https://www.indeed.com/jobs?q=&l={zip_code}"
        headers = {
            "User-Agent": "Mozilla/5.0 (Windows NT 10.0; Win64; x64) AppleWebKit/537.36 (KHTML, like Gecko) Chrome/85.0.4183.121 Safari/537.36"
        }
        try:
            response = requests.get(url, headers=headers, proxies={"http": proxy, "https": proxy})
            response.raise_for_status()
            # Parse the response (placeholder for actual parsing logic)
            jobs = [{"title": "Sample Job", "company": "Sample Company", "location": zip_code}]
            return jobs
        except requests.RequestException as e:
            print(f"Error scraping {zip_code} with proxy {proxy}: {e}")
            return []
    def rotate_proxy(self):
        return choice(self.proxies)
    def save_to_csv(self, data, filename):
        with open(filename, mode="w", newline="", encoding="utf-8") as file:
            writer = csv.DictWriter(file, fieldnames=["title", "company", "location"])

'''
This module contains the JobScraper class, which handles the scraping logic for Indeed jobs, including proxy rotation and saving results to a CSV file.
'''
import requests
import csv
from random import choice
class JobScraper:
    def __init__(self, proxies):
        self.proxies = proxies
    def scrape_jobs(self, zip_code, proxy):
        url = f"https://www.indeed.com/jobs?q=&l={zip_code}"
        headers = {
            "User-Agent": "Mozilla/5.0 (Windows NT 10.0; Win64; x64) AppleWebKit/537.36 (KHTML, like Gecko) Chrome/85.0.4183.121 Safari/537.36"
        }
        try:
            response = requests.get(url, headers=headers, proxies={"http": proxy, "https": proxy})
            response.raise_for_status()
            # Parse the response (placeholder for actual parsing logic)
            jobs = [{"title": "Sample Job", "company": "Sample Company", "location": zip_code}]
            return jobs
        except requests.RequestException as e:
            print(f"Error scraping {zip_code} with proxy {proxy}: {e}")
            return []
    def rotate_proxy(self):
        return choice(self.proxies)
    def save_to_csv(self, data, filename):
        with open(filename, mode="w", newline="", encoding="utf-8") as file:
            writer = csv.DictWriter(file, fieldnames=["title", "company", "location"])
            writer.writeheader()
            writer.writerows(data)

        with ThreadPoolExecutor(max_workers=10) as executor:  # Adjust max_workers as needed

import requests
import csv
from random import choice
class JobScraper:
    def __init__(self, proxies):
        self.proxies = proxies
    def scrape_jobs(self, zip_code, proxy):
        url = f"https://www.indeed.com/jobs?q=&l={zip_code}"
        headers = {
            "User-Agent": "Mozilla/5.0 (Windows NT 10.0; Win64; x64) AppleWebKit/537.36 (KHTML, like Gecko) Chrome/85.0.4183.121 Safari/537.36"
        }
        try:
            response = requests.get(url, headers=headers, proxies={"http": proxy, "https": proxy})
            response.raise_for_status()
            # Parse the response (placeholder for actual parsing logic)
            jobs = [{"title": "Sample Job", "company": "Sample Company", "location": zip_code}]
            return jobs
        except requests.RequestException as e:
            print(f"Error scraping {zip_code} with proxy {proxy}: {e}")
            return []
    def rotate_proxy(self):
        return choice(self.proxies)
    def save_to_csv(self, data, filename):
        with open(filename, mode="w", newline="", encoding="utf-8") as file:
            writer = csv.DictWriter(file, fieldnames=["title", "company", "location"])
            writer.writeheader()
            writer.writerows(data)

'''  
This module contains the JobScraper class, which handles the scraping logic for Indeed jobs, including proxy rotation, retry mechanism, and saving results to a CSV file.  
'''  
import requests  
import csv  
from random import choice  
class JobScraper:  
    def __init__(self, proxies):  
        self.proxies = proxies  
    def scrape_jobs(self, zip_code):  
        url = f"https://www.indeed.com/jobs?q=&l={zip_code}"  
        headers = {  
            "User-Agent": "Mozilla/5.0 (Windows NT 10.0; Win64; x64) AppleWebKit/537.36 (KHTML, like Gecko) Chrome/85.0.4183.121 Safari/537.36"  
        }  
        for _ in range(len(self.proxies)):  
            proxy = self.rotate_proxy()  
            try:  
                response = requests.get(url, headers=headers, proxies={"http": proxy, "https": proxy})  
                response.raise_for_status()  
                # Parse the response (placeholder for actual parsing logic)  
                jobs = [{"title": "Sample Job", "company": "Sample Company", "location": zip_code}]  
                return jobs  
            except requests.RequestException as e:  
                print(f"Error scraping {zip_code} with proxy {proxy}: {e}")  
        return []  
    def rotate_proxy(self):  
        return choice(self.proxies)  
    def save_to_csv(self, data, filename):  
        with open(filename, mode="w", newline="", encoding="utf-8") as file:  
            writer = csv.DictWriter(file, fieldnames=["title", "company", "location"])  
            writer.writeheader()  
            writer.writerows(data)

'''

import tkinter as tk  
from tkinter import filedialog, messagebox  
from scraper import JobScraper  
from excel_processor import ExcelProcessor

class GUI:  
    def __init__(self):  
        self.root = tk.Tk()

        self.root.geometry("400x300")  
        self.proxies = ["http://proxy1.com", "http://proxy2.com"]  # Example proxies  
        self.scraper = JobScraper(self.proxies)  
        self.processor = ExcelProcessor()  
        self.file_path = tk.StringVar()  
        self.output_file = tk.StringVar(value="output.csv")  
        self.create_widgets()  
    def create_widgets(self):  
        tk.Label(self.root, text="Excel File:").pack(pady=5)  
        tk.Entry(self.root, textvariable=self.file_path, width=40).pack(pady=5)  
        tk.Button(self.root, text="Browse", command=self.browse_file).pack(pady=5)  
        tk.Label(self.root, text="Output CSV File:").pack(pady=5)  
        tk.Entry(self.root, textvariable=self.output_file, width=40).pack(pady=5)  
        tk.Button(self.root, text="Start Scraping", command=self.start_scraping).pack(pady=20)  
    def browse_file(self):  
        file_path = filedialog.askopenfilename(filetypes=[("Excel files", "*.xlsx")])  
        self.file_path.set(file_path)  
    def start_scraping(self):  
        file_path = self.file_path.get()  
        output_file = self.output_file.get()  
        if not file_path:  
            messagebox.showerror("Error", "Please select an Excel file.")  
            return  
        zip_codes = self.processor.read_zip_codes(file_path)  
        def scrape(zip_code):  
            return self.scraper.scrape_jobs(zip_code)  
        all_jobs = []  
        with ThreadPoolExecutor(max_workers=10) as executor:  # Adjust max_workers as needed  
            futures = [executor.submit(scrape, zip_code) for zip_code in zip_codes]  
            for future in futures:  
                all_jobs.extend(future.result())  
        self.scraper.save_to_csv(all_jobs, output_file)  
        messagebox.showinfo("Success", f"Scraping completed. Results saved to {output_file}.")  
    def run(self):  
        self.root.mainloop()

'''  
This module contains the JobScraper class, which handles the scraping logic for Indeed jobs, including proxy rotation, retry mechanism, and saving results to a CSV file.  
'''  
import requests  
import csv  
from random import choice  
class JobScraper:  
    def __init__(self, proxies):  
        self.proxies = proxies  
    def scrape_jobs(self, zip_code):  
        url = f"https://www.indeed.com/jobs?q=&l={zip_code}"  
        headers = {  
            "User-Agent": "Mozilla/5.0 (Windows NT 10.0; Win64; x64) AppleWebKit/537.36 (KHTML, like Gecko) Chrome/85.0.4183.121 Safari/537.36"  
        }  
        for _ in range(len(self.proxies)):  
            proxy = self.rotate_proxy()  
            try:  
                response = requests.get(url, headers=headers, proxies={"http": proxy, "https": proxy})  
                response.raise_for_status()  
                # Parse the response (placeholder for actual parsing logic)  
                jobs = [{"title": "Sample Job", "company": "Sample Company", "location": zip_code}]  
                return jobs  
            except requests.RequestException as e:  
                print(f"Error scraping {zip_code} with proxy {proxy}: {e}")  
        return []  
    def rotate_proxy(self):  
        return choice(self.proxies)  
    def save_to_csv(self, data, filename):  
        with open(filename, mode="w", newline="", encoding="utf-8") as file:  
            writer = csv.DictWriter(file, fieldnames=["title", "company", "location"])  
            writer.writeheader()  
            writer.writerows(data)

'''
This module contains the JobScraper class, which handles the scraping logic for Indeed jobs, including proxy rotation, retry mechanism, and saving results to a CSV file.
'''
import requests
import csv
from random import choice
class JobScraper:
    def __init__(self, proxies):
        self.proxies = proxies
    def scrape_jobs(self, zip_code):
        url = f"https://www.indeed.com/jobs?q=&l={zip_code}"
        headers = {
            "User-Agent": "Mozilla/5.0 (Windows NT 10.0; Win64; x64) AppleWebKit/537.36 (KHTML, like Gecko) Chrome/85.0.4183.121 Safari/537.36"
        }
        for attempt in range(len(self.proxies)):
            proxy = self.rotate_proxy()
            try:
                response = requests.get(url, headers=headers, proxies={"http": proxy, "https": proxy})
                response.raise_for_status()
                # Parse the response (placeholder for actual parsing logic)
                jobs = [{"title": "Sample Job", "company": "Sample Company", "location": zip_code}]
                return jobs
            except requests.RequestException as e:
                print(f"Error scraping {zip_code} with proxy {proxy}: {e}")
        print(f"Failed to scrape jobs for ZIP code {zip_code} after {len(self.proxies)} attempts.")
        return []
    def rotate_proxy(self):
        return choice(self.proxies)
    def save_to_csv(self, data, filename):
        with open(filename, mode="w", newline="", encoding="utf-8") as file:
            writer = csv.DictWriter(file, fieldnames=["title", "company", "location"])
            writer.writeheader()
            writer.writerows(data)

'''

'''
import requests
import csv
from random import choice
class JobScraper:
    def __init__(self, proxies):
        self.proxies = proxies
    def scrape_jobs(self, zip_code):
        url = f"https://www.indeed.com/jobs?q=&l={zip_code}"
        headers = {
            "User-Agent": "Mozilla/5.0 (Windows NT 10.0; Win64; x64) AppleWebKit/537.36 (KHTML, like Gecko) Chrome/85.0.4183.121 Safari/537.36"
        }
        for attempt in range(len(self.proxies)):
            proxy = self.rotate_proxy()
            try:
                response = requests.get(url, headers=headers, proxies={"http": proxy, "https": proxy})
                response.raise_for_status()
                # Parse the response (placeholder for actual parsing logic)
                jobs = [{"title": "Sample Job", "company": "Sample Company", "location": zip_code}]
                return jobs
            except requests.RequestException as e:
                print(f"Error scraping {zip_code} with proxy {proxy}: {e}")
        print(f"Failed to scrape jobs for ZIP code {zip_code} after {len(self.proxies)} attempts.")
        return []
    def rotate_proxy(self):
        return choice(self.proxies)
    def save_to_csv(self, data, filename):
        with open(filename, mode="w", newline="", encoding="utf-8") as file:
            writer = csv.DictWriter(file, fieldnames=["title", "company", "location"])
            writer.writeheader()
            writer.writerows(data)

'''
This module contains the JobScraper class, which handles the scraping logic for Indeed jobs, including proxy rotation, retry mechanism, and saving results to a CSV file.
'''
import requests
import csv
from random import choice
class JobScraper:
    def __init__(self, proxies):
        self.proxies = proxies
    def scrape_jobs(self, zip_code):
        url = f"https://www.indeed.com/jobs?q=&l={zip_code}"
        headers = {
            "User-Agent": "Mozilla/5.0 (Windows NT 10.0; Win64; x64) AppleWebKit/537.36 (KHTML, like Gecko) Chrome/85.0.4183.121 Safari/537.36"
        }
        for attempt in range(len(self.proxies)):
            proxy = self.rotate_proxy()
            try:
                response = requests.get(url, headers=headers, proxies={"http": proxy, "https": proxy})
                response.raise_for_status()
                # Parse the response (placeholder for actual parsing logic)
                jobs = [{"title": "Sample Job", "company": "Sample Company", "location": zip_code}]
                return jobs
            except requests.RequestException as e:
                print(f"Error scraping {zip_code} with proxy {proxy}: {e}")
        print(f"Failed to scrape jobs for ZIP code {zip_code} after {len(self.proxies)} attempts.")
        return []
    def rotate_proxy(self):
        return choice(self.proxies)
    def save_to_csv(self, data, filename):
        with open(filename, mode="w", newline="", encoding="utf-8") as file:
            writer = csv.DictWriter(file, fieldnames=["title", "company", "location"])
            writer.writeheader()
            writer.writerows(data)

'''
This module contains the GUI class, which provides a graphical user interface for the application.
'''
import tkinter as tk
from tkinter import filedialog, messagebox
from scraper import JobScraper
from excel_processor import ExcelProcessor
from threading import Thread
from concurrent.futures import ThreadPoolExecutor
class GUI:
    def __init__(self):
        self.root = tk.Tk()
        self.root.title("Indeed Job Scraper")
        self.root.geometry("400x300")
        self.proxies = ["http://proxy1.com", "http://proxy2.com"]  # Example proxies
        self.scraper = JobScraper(self.proxies)
        self.processor = ExcelProcessor()
        self.file_path = tk.StringVar()
        self.output_file = tk.StringVar(value="output.csv")
        self.create_widgets()
    def create_widgets(self):
        tk.Label(self.root, text="Excel File:").pack(pady=5)
        tk.Entry(self.root, textvariable=self.file_path, width=40).pack(pady=5)
        tk.Button(self.root, text="Browse", command=self.browse_file).pack(pady=5)
        tk.Label(self.root, text="Output CSV File:").pack(pady=5)
        tk.Entry(self.root, textvariable=self.output_file, width=40).pack(pady=5)
        tk.Button(self.root, text="Start Scraping", command=self.start_scraping).pack(pady=20)
    def browse_file(self):
        file_path = filedialog.askopenfilename(filetypes=[("Excel files", "*.xlsx")])
        self.file_path.set(file_path)
    def start_scraping(self):
        file_path = self.file_path.get()
        output_file = self.output_file.get()
        if not file_path:
            messagebox.showerror("Error", "Please select an Excel file.")
            return
        zip_codes = self.processor.read_zip_codes(file_path)
        def scrape(zip_code):
            return self.scraper.scrape_jobs(zip_code)
        all_jobs = []
        with ThreadPoolExecutor(max_workers=10) as executor:  # Adjust max_workers as needed
            futures = [executor.submit(scrape, zip_code) for zip_code in zip_codes]
            for future in futures:
                all_jobs.extend(future.result())
        self.scraper.save_to_csv(all_jobs, output_file)
        messagebox.showinfo("Success", f"Scraping completed. Results saved to {output_file}.")
    def run(self):
        self.root.mainloop()
