'''

Main application file to initialize the GUI and handle user interactions.

'''

import tkinter as tk
from tkinter import filedialog, messagebox
import pandas as pd
from data_processing import DataProcessor
from visualization import Visualizer
from report_generation import ReportGenerator

class App:
    def __init__(self, root):
        self.root = root
        self.root.title("Descriptive Analysis Tool")
        
        self.data_processor = DataProcessor()
        self.visualizer = Visualizer()
        self.report_generator = ReportGenerator()
        
        self.create_widgets()
        
    def create_widgets(self):
        self.upload_button = tk.Button(self.root, text="Upload Dataset", command=self.upload_file)
        self.upload_button.pack(pady=10)
        
        self.analyze_button = tk.Button(self.root, text="Analyze", command=self.analyze_data)
        self.analyze_button.pack(pady=10)
        
        self.report_button = tk.Button(self.root, text="Generate Report", command=self.generate_report)
        self.report_button.pack(pady=10)
        
    def upload_file(self):
        file_path = filedialog.askopenfilename()
        if file_path:
            self.data = pd.read_csv(file_path)
            messagebox.showinfo("Success", "Dataset uploaded successfully!")
        
    def analyze_data(self):
        if hasattr(self, 'data'):
            self.cleaned_data = self.data_processor.clean_data(self.data)
            self.metrics = self.data_processor.compute_metrics(self.cleaned_data)
            self.visualizer.plot_data(self.cleaned_data)
            messagebox.showinfo("Success", "Data analysis complete!")
        else:
            messagebox.showwarning("Warning", "Please upload a dataset first.")
        
    def generate_report(self):
        if hasattr(self, 'metrics'):
            self.report_generator.generate_pdf(self.metrics)
            messagebox.showinfo("Success", "Report generated successfully!")
        else:
            messagebox.showwarning("Warning", "Please analyze data first.")

if __name__ == "__main__":
    root = tk.Tk()
    app = App(root)
    root.mainloop()
```

import pandas as pd

class DataProcessor:
    def clean_data(self, data):
        # Example data cleaning steps
        data = data.dropna()  # Drop missing values
        data = data.drop_duplicates()  # Drop duplicate rows
        return data
    
    def compute_metrics(self, data):
        metrics = {
            'count': data.count(),
            'mean': data.mean(),
            'median': data.median(),
            'std': data.std(),
            'frequency': data.apply(lambda x: x.value_counts())
        }
        return metrics

Main application file to initialize the GUI and handle user interactions.
'''
import tkinter as tk
from tkinter import filedialog, messagebox
import pandas as pd
from data_processing import DataProcessor
from visualization import Visualizer
from report_generation import ReportGenerator

class App:
    def __init__(self, root):
        self.root = root
        self.root.title("Descriptive Analysis Tool")
        self.data_processor = DataProcessor()
        self.visualizer = Visualizer()
        self.report_generator = ReportGenerator()
        self.create_widgets()

    def create_widgets(self):
        self.upload_button = tk.Button(self.root, text="Upload Dataset", command=self.upload_file)
        self.upload_button.pack(pady=10)
        self.analyze_button = tk.Button(self.root, text="Analyze", command=self.analyze_data)
        self.analyze_button.pack(pady=10)
        self.report_button = tk.Button(self.root, text="Generate Report", command=self.generate_report)
        self.report_button.pack(pady=10)

    def upload_file(self):
        file_path = filedialog.askopenfilename()
        if file_path:
            self.data = pd.read_csv(file_path)
            messagebox.showinfo("Success", "Dataset uploaded successfully!")

    def analyze_data(self):
        if hasattr(self, 'data'):
            self.cleaned_data = self.data_processor.clean_data(self.data)
            self.metrics = self.data_processor.compute_metrics(self.cleaned_data)
            self.visualizer.plot_data(self.cleaned_data)
            messagebox.showinfo("Success", "Data analysis complete!")
        else:
            messagebox.showwarning("Warning", "Please upload a dataset first.")

    def generate_report(self):
        if hasattr(self, 'metrics'):
            self.report_generator.generate_pdf(self.metrics)
            messagebox.showinfo("Success", "Report generated successfully!")
        else:
            messagebox.showwarning("Warning", "Please analyze data first.")

if __name__ == "__main__":
    root = tk.Tk()
    app = App(root)
    root.mainloop()
```

'''
Module for data cleaning and computation of metrics.
'''
import pandas as pd

class DataProcessor:
    def clean_data(self, data):
        # Example data cleaning steps
        data = data.dropna()  # Drop missing values
        data = data.drop_duplicates()  # Drop duplicate rows
        return data

    def compute_metrics(self, data):
        metrics = {
            'count': data.count(),
            'mean': data.mean(),
            'median': data.median(),
            'std': data.std(),
            'frequency': data.apply(lambda x: x.value_counts())
        }
        return metrics
```

from fpdf import FPDF

class ReportGenerator:
    def generate_pdf(self, metrics):
        pdf = FPDF()
        pdf.add_page()
        pdf.set_font("Arial", size=12)
        pdf.cell(200, 10, txt="Descriptive Analysis Report", ln=True, align='C')
        for key, value in metrics.items():
            pdf.cell(200, 10, txt=f"{key}: {value}", ln=True)

import pandas as pd
class DataProcessor:
    def clean_data(self, data):
        # Example data cleaning steps
        data = data.dropna()  # Drop missing values
        data = data.drop_duplicates()  # Drop duplicate rows

import pandas as pd

    def clean_data(self, data):
        # Example data cleaning steps

'''

import tkinter as tk
from tkinter import filedialog, messagebox
import pandas as pd
from data_processing import DataProcessor
from visualization import Visualizer
from report_generation import ReportGenerator

class App:
    def __init__(self, root):
        self.root = root
        self.root.title("Descriptive Analysis Tool")
        self.data_processor = DataProcessor()
        self.visualizer = Visualizer()
        self.report_generator = ReportGenerator()
        self.create_widgets()

    def create_widgets(self):
        self.upload_button = tk.Button(self.root, text="Upload Dataset", command=self.upload_file)
        self.upload_button.pack(pady=10)
        self.analyze_button = tk.Button(self.root, text="Analyze", command=self.analyze_data)
        self.analyze_button.pack(pady=10)
        self.report_button = tk.Button(self.root, text="Generate Report", command=self.generate_report)
        self.report_button.pack(pady=10)

    def upload_file(self):
        file_path = filedialog.askopenfilename()
        if file_path:
            self.data = pd.read_csv(file_path)
            messagebox.showinfo("Success", "Dataset uploaded successfully!")

    def analyze_data(self):
        if hasattr(self, 'data'):
            self.cleaned_data = self.data_processor.clean_data(self.data)
            self.metrics = self.data_processor.compute_metrics(self.cleaned_data)
            self.visualizer.plot_data(self.cleaned_data)
            messagebox.showinfo("Success", "Data analysis complete!")

    def generate_report(self):
        if hasattr(self, 'metrics'):
            self.report_generator.generate_pdf(self.metrics)
            messagebox.showinfo("Success", "Report generated successfully!")
        else:
            messagebox.showwarning("Warning", "Please analyze data first.")

if __name__ == "__main__":
    root = tk.Tk()

'''
Module for data cleaning and computation of metrics.
'''
import pandas as pd

class DataProcessor:
    def clean_data(self, data):
        # Example data cleaning steps
        data = data.dropna()  # Drop missing values
        data = data.drop_duplicates()  # Drop duplicate rows
        return data

    def compute_metrics(self, data):
        metrics = {
            'count': data.count(),
            'mean': data.mean(),
            'median': data.median(),
            'std': data.std(),
            'frequency': data.apply(lambda x: x.value_counts())
        }
        return metrics

'''
Module for data cleaning and computation of metrics.
'''
import pandas as pd
class DataProcessor:
    def clean_data(self, data):
        # Example data cleaning steps
        data = data.dropna()  # Drop missing values
        data = data.drop_duplicates()  # Drop duplicate rows
        return data
    def compute_metrics(self, data):
        metrics = {
            'count': data.count(),
            'mean': data.mean(),
            'median': data.median(),
            'std': data.std(),
            'frequency': data.apply(lambda x: x.value_counts())
        }
        return metrics
