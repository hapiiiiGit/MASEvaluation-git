'''
Handles data ingestion from PostgreSQL and CSV files.
'''
import psycopg2
import csv
class DataIngestion:
    def ingest_postgresql_data(self):
        try:
            # Connect to PostgreSQL and fetch data
            with psycopg2.connect("dbname=test user=postgres password=secret") as conn:
                with conn.cursor() as cur:
                    cur.execute("SELECT * FROM operational_data")
                    data = cur.fetchall()
            return data
        except psycopg2.Error as e:
            print(f"Error connecting to PostgreSQL: {e}")
            return []
    def ingest_csv_data(self):
        # Read data from CSV file
        data = []
        try:
            with open('data.csv', newline='') as csvfile:
                reader = csv.reader(csvfile)
                for row in reader:
                    data.append(row)
        except FileNotFoundError as e:
            print(f"CSV file not found: {e}")
        return data

import psycopg2
import csv
class DataIngestion:
    def ingest_postgresql_data(self):
        try:
            # Connect to PostgreSQL and fetch data
            with psycopg2.connect("dbname=test user=postgres password=secret") as conn:
                with conn.cursor() as cur:
                    cur.execute("SELECT * FROM operational_data")
                    data = cur.fetchall()
            return data
        except psycopg2.Error as e:
            print(f"Error connecting to PostgreSQL: {e}")
            return []
    def ingest_csv_data(self):
        # Read data from CSV file
        data = []
        try:
            with open('data.csv', newline='') as csvfile:
                reader = csv.reader(csvfile)
                for row in reader:
                    data.append(row)
        except FileNotFoundError as e:
            print(f"CSV file not found: {e}")
        return data

'''

import psycopg2
import csv
class DataIngestion:
    def ingest_postgresql_data(self):
        try:
            # Connect to PostgreSQL and fetch data
            with psycopg2.connect("dbname=test user=postgres password=secret") as conn:
                with conn.cursor() as cur:
                    cur.execute("SELECT * FROM operational_data")
                    data = cur.fetchall()
            return data
        except psycopg2.Error as e:
            raise Exception(f"Error connecting to PostgreSQL: {e}")
    def ingest_csv_data(self):
        # Read data from CSV file
        data = []
        try:
            with open('data.csv', newline='') as csvfile:
                reader = csv.reader(csvfile)
                for row in reader:
                    data.append(row)
        except FileNotFoundError as e:
            raise Exception(f"CSV file not found: {e}")
        return data

import psycopg2
import csv
class DataIngestion:
    def ingest_postgresql_data(self):
        try:
            # Connect to PostgreSQL and fetch data
            with psycopg2.connect("dbname=test user=postgres password=secret") as conn:
                with conn.cursor() as cur:
                    cur.execute("SELECT * FROM operational_data")
                    data = cur.fetchall()
            return data
        except psycopg2.Error as e:
            raise Exception(f"Error connecting to PostgreSQL: {e}")
    def ingest_csv_data(self):
        # Read data from CSV file
        data = []
        try:
            with open('data.csv', newline='') as csvfile:
                reader = csv.reader(csvfile)
                for row in reader:
                    data.append(row)
        except FileNotFoundError as e:
            raise Exception(f"CSV file not found: {e}")
        return data

'''
Handles data ingestion from PostgreSQL and CSV files.
'''
import psycopg2
import csv
class DataIngestion:
    def ingest_postgresql_data(self):
        try:
            # Connect to PostgreSQL and fetch data
            with psycopg2.connect("dbname=test user=postgres password=secret") as conn:
                with conn.cursor() as cur:
                    cur.execute("SELECT * FROM operational_data")
                    data = cur.fetchall()
            return data
        except psycopg2.Error as e:
            raise Exception(f"Error connecting to PostgreSQL: {e}")
    def ingest_csv_data(self):
        # Read data from CSV file
        data = []
        try:
            with open('data.csv', newline='') as csvfile:
                reader = csv.reader(csvfile)
                for row in reader:
                    data.append(row)
        except FileNotFoundError as e:
            raise Exception(f"CSV file not found: {e}")
        return data
