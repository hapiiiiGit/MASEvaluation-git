    description = models.TextField()
    date = models.DateTimeField()
    location = models.CharField(max_length=200)
    organizer = models.ForeignKey(UserProfile, on_delete=models.CASCADE)
class Ticket(models.Model):
    event = models.ForeignKey(Event, on_delete=models.CASCADE)
    holder = models.ForeignKey(UserProfile, on_delete=models.CASCADE)
    purchase_date = models.DateTimeField(auto_now_add=True)
class Payment(models.Model):

    def get(self, request, event_id):
        event = get_object_or_404(Event, id=event_id)
        return render(request, 'events/event_detail.html', {'event': event})
class UserProfileView(View):
    def get(self, request):
        form = UserProfileForm()
        return render(request, 'profiles/user_profile.html', {'form': form})
    def post(self, request):
        form = UserProfileForm(request.POST)
        if form.is_valid():
            user_profile = form.save(commit=False)
            user_profile.user = request.user  # Associate with the current user
            user_profile.save()
            return redirect('event_list')
        return render(request, 'profiles/user_profile.html', {'form': form})
class TicketPurchaseView(View):
    def get(self, request, event_id):
        form = TicketForm()

        form = TicketForm(request.POST)
        if form.is_valid():
            ticket = form.save(commit=False)
            ticket.event = get_object_or_404(Event, id=event_id)  # Associate with the specific event
            ticket.holder = request.user.userprofile  # Associate with the current user's profile
            ticket.save()

from django.db import models
from django.contrib.auth.models import User
class UserProfile(models.Model):
    user = models.OneToOneField(User, on_delete=models.CASCADE)
    role = models.CharField(max_length=50, choices=[('delegate', 'Delegate'), ('organizer', 'Organizer'), ('speaker', 'Speaker')])
class Event(models.Model):
    name = models.CharField(max_length=200)
    description = models.TextField()
    date = models.DateTimeField()
    location = models.CharField(max_length=200)
    organizer = models.ForeignKey(UserProfile, on_delete=models.CASCADE)
class Ticket(models.Model):
    event = models.ForeignKey(Event, on_delete=models.CASCADE)
    holder = models.ForeignKey(UserProfile, on_delete=models.CASCADE)
    purchase_date = models.DateTimeField(auto_now_add=True)
class Payment(models.Model):
    ticket = models.OneToOneField(Ticket, on_delete=models.CASCADE)
    amount = models.DecimalField(max_digits=10, decimal_places=2)
    payment_date = models.DateTimeField(auto_now_add=True)
    status = models.CharField(max_length=50)

'''

from django.shortcuts import render, get_object_or_404, redirect
from django.views import View
from .models import Event, UserProfile, Ticket
from .forms import UserProfileForm, EventForm, TicketForm
class EventListView(View):
    def get(self, request):
        events = Event.objects.all()
        return render(request, 'events/event_list.html', {'events': events})
class EventDetailView(View):
    def get(self, request, event_id):
        event = get_object_or_404(Event, id=event_id)
        return render(request, 'events/event_detail.html', {'event': event})
class UserProfileView(View):
    def get(self, request):
        form = UserProfileForm()
        return render(request, 'profiles/user_profile.html', {'form': form})
    def post(self, request):
        form = UserProfileForm(request.POST)
        if form.is_valid():
            user_profile = form.save(commit=False)
            user_profile.user = request.user  # Associate with the current user
            user_profile.save()
            return redirect('event_list')
        return render(request, 'profiles/user_profile.html', {'form': form})
class TicketPurchaseView(View):
    def get(self, request, event_id):
        form = TicketForm()
        return render(request, 'tickets/ticket_purchase.html', {'form': form})
    def post(self, request, event_id):
        form = TicketForm(request.POST)
        if form.is_valid():
            ticket = form.save(commit=False)
            ticket.event = get_object_or_404(Event, id=event_id)  # Associate with the specific event
            ticket.holder = request.user.userprofile  # Associate with the current user's profile
            ticket.save()
            return redirect('event_detail', event_id=event_id)
        return render(request, 'tickets/ticket_purchase.html', {'form': form})
class AdminDashboardView(View):
    def get(self, request):
        return render(request, 'admin/dashboard.html')

from django.shortcuts import render, get_object_or_404, redirect
from django.views import View
from django.contrib.auth.decorators import login_required, user_passes_test
from django.utils.decorators import method_decorator
from .models import Event, UserProfile, Ticket
from .forms import UserProfileForm, EventForm, TicketForm
def is_admin(user):
    return user.is_staff  # Assuming admin users are marked as staff
@method_decorator(login_required, name='dispatch')
class EventListView(View):
    def get(self, request):
        events = Event.objects.all()
        return render(request, 'events/event_list.html', {'events': events})
@method_decorator(login_required, name='dispatch')
class EventDetailView(View):
    def get(self, request, event_id):
        event = get_object_or_404(Event, id=event_id)
        return render(request, 'events/event_detail.html', {'event': event})
@method_decorator(login_required, name='dispatch')
class UserProfileView(View):
    def get(self, request):
        form = UserProfileForm()
        return render(request, 'profiles/user_profile.html', {'form': form})
    def post(self, request):
        form = UserProfileForm(request.POST)
        if form.is_valid():
            user_profile = form.save(commit=False)
            user_profile.user = request.user  # Associate with the current user
            user_profile.save()
            return redirect('event_list')
        return render(request, 'profiles/user_profile.html', {'form': form})
@method_decorator(login_required, name='dispatch')
class TicketPurchaseView(View):
    def get(self, request, event_id):
        form = TicketForm()
        return render(request, 'tickets/ticket_purchase.html', {'form': form})
    def post(self, request, event_id):
        form = TicketForm(request.POST)
        if form.is_valid():
            ticket = form.save(commit=False)
            ticket.event = get_object_or_404(Event, id=event_id)  # Associate with the specific event
            ticket.holder = request.user.userprofile  # Associate with the current user's profile
            ticket.save()
            return redirect('event_detail', event_id=event_id)
        return render(request, 'tickets/ticket_purchase.html', {'form': form})
@method_decorator([login_required, user_passes_test(is_admin)], name='dispatch')
class AdminDashboardView(View):
    def get(self, request):
        return render(request, 'admin/dashboard.html')

def is_admin(user):
    return user.is_staff  # Assuming admin users are marked as staff
@method_decorator(login_required, name='dispatch')
class EventListView(View):
    def get(self, request):
        events = Event.objects.all()
        return render(request, 'events/event_list.html', {'events': events})
@method_decorator(login_required, name='dispatch')
class EventDetailView(View):
    def get(self, request, event_id):
        event = get_object_or_404(Event, id=event_id)
        return render(request, 'events/event_detail.html', {'event': event})
@method_decorator(login_required, name='dispatch')
class UserProfileView(View):
    def get(self, request):
        form = UserProfileForm()
        return render(request, 'profiles/user_profile.html', {'form': form})
    def post(self, request):
        form = UserProfileForm(request.POST)
        if form.is_valid():
            user_profile = form.save(commit=False)
            user_profile.user = request.user  # Associate with the current user
            user_profile.save()
            return redirect('event_list')
        return render(request, 'profiles/user_profile.html', {'form': form})
@method_decorator(login_required, name='dispatch')
class TicketPurchaseView(View):
    def get(self, request, event_id):
        form = TicketForm()
        return render(request, 'tickets/ticket_purchase.html', {'form': form})
    def post(self, request, event_id):
        form = TicketForm(request.POST)
        if form.is_valid():
            ticket = form.save(commit=False)
            ticket.event = get_object_or_404(Event, id=event_id)  # Associate with the specific event
            ticket.holder = request.user.userprofile  # Associate with the current user's profile
            ticket.save()
            return redirect('event_detail', event_id=event_id)
        return render(request, 'tickets/ticket_purchase.html', {'form': form})
@method_decorator([login_required, user_passes_test(is_admin)], name='dispatch')
class AdminDashboardView(View):
    def get(self, request):
        return render(request, 'admin/dashboard.html')

    return user.is_staff  # Assuming admin users are marked as staff
@method_decorator(login_required, name='dispatch')
class EventListView(View):
    def get(self, request):
        events = Event.objects.all()
        return render(request, 'events/event_list.html', {'events': events})
@method_decorator(login_required, name='dispatch')
class EventDetailView(View):
    def get(self, request, event_id):
        event = get_object_or_404(Event, id=event_id)
        return render(request, 'events/event_detail.html', {'event': event})
@method_decorator(login_required, name='dispatch')
class UserProfileView(View):
    def get(self, request):
        form = UserProfileForm()
        return render(request, 'profiles/user_profile.html', {'form': form})
    def post(self, request):
        form = UserProfileForm(request.POST)
        if form.is_valid():
            user_profile = form.save(commit=False)
            user_profile.user = request.user  # Associate with the current user
            user_profile.save()
            return redirect('event_list')
        return render(request, 'profiles/user_profile.html', {'form': form})
@method_decorator(login_required, name='dispatch')
class TicketPurchaseView(View):
    def get(self, request, event_id):
        form = TicketForm()
        return render(request, 'tickets/ticket_purchase.html', {'form': form})
    def post(self, request, event_id):
        form = TicketForm(request.POST)
        if form.is_valid():
            ticket = form.save(commit=False)
            ticket.event = get_object_or_404(Event, id=event_id)  # Associate with the specific event
            ticket.holder = request.user.userprofile  # Associate with the current user's profile
            ticket.save()
            return redirect('event_detail', event_id=event_id)
        return render(request, 'tickets/ticket_purchase.html', {'form': form})
@method_decorator([login_required, user_passes_test(is_admin)], name='dispatch')
class AdminDashboardView(View):

# Summit Aggregator Web Application

Welcome to the Summit Aggregator Web Application! This application is designed to serve as an aggregator for global summits and conferences, providing a platform for delegates, organizers, and speakers to interact, manage events, and facilitate ticketing and payments. Additionally, it includes an AI chatbot for enhanced user interaction and administrative controls for event management and monetization.

## Main Features

1. **User Profiles**: 
   - Create and manage profiles for delegates, organizers, and speakers.
   - Assign roles to users to tailor their experience and access within the application.

2. **Event Management**:
   - Organizers can create and manage events, including details such as name, description, date, and location.
   - Delegates and speakers can view event details and participate accordingly.

3. **Ticketing and Payment Integration**:
   - Purchase tickets for events with integrated payment processing using Stripe.
   - Manage purchased tickets and view payment history.

4. **AI Chatbot Functionality**:
   - Interact with an AI chatbot for assistance and information about events and the platform.
   - The chatbot is powered by ChatterBot and trained on a corpus of English language data.

5. **Administrative Controls**:
   - Admin users have access to a dashboard for managing events and monetization strategies.
   - Administrative controls ensure smooth operation and management of the platform.

- ChatterBot for AI chatbot functionality
