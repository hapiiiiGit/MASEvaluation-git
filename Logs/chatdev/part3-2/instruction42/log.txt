import psycopg2
from psycopg2 import sql, OperationalError
class QuestDBConnector:
    def __init__(self):
        self.connection = None
    def connect(self):
        try:
            self.connection = psycopg2.connect(
                dbname='your_db_name', user='your_user', password='your_password', host='your_host', port='your_port'
            )
            print("Connected to QuestDB")
        except OperationalError as e:
            print(f"ERROR: Could not connect to QuestDB: {e}")
    def insert_data(self, data):
        if self.connection is None:
            print("ERROR: No connection to QuestDB")
            return
        try:
            with self.connection.cursor() as cursor:
                insert_query = sql.SQL("INSERT INTO your_table (column_name) VALUES (%s)")
                cursor.execute(insert_query, (data,))
                self.connection.commit()
                print(f"Inserted data into QuestDB: {data}")
        except Exception as e:
            print(f"ERROR: Failed to insert data: {e}")
    def close_connection(self):
        if self.connection:
            self.connection.close()
            print("Connection to QuestDB closed")

'''

import os
import psycopg2
from psycopg2 import sql, OperationalError
class QuestDBConnector:
    def __init__(self):
        self.connection = None
    def connect(self):
        try:
            self.connection = psycopg2.connect(
                dbname=os.getenv('DB_NAME'),
                user=os.getenv('DB_USER'),
                password=os.getenv('DB_PASSWORD'),
                host=os.getenv('DB_HOST'),
                port=os.getenv('DB_PORT')
            )
            print("Connected to QuestDB")
        except OperationalError as e:
            print(f"ERROR: Could not connect to QuestDB: {e}")
    def insert_data(self, data):
        if self.connection is None:
            print("ERROR: No connection to QuestDB")
            return
        try:
            with self.connection.cursor() as cursor:
                insert_query = sql.SQL("INSERT INTO your_table (column_name) VALUES (%s)")
                cursor.execute(insert_query, (data,))
                self.connection.commit()
                print(f"Inserted data into QuestDB: {data}")
        except Exception as e:
            print(f"ERROR: Failed to insert data: {e}")
    def close_connection(self):
        if self.connection:
            self.connection.close()
            print("Connection to QuestDB closed")

```python
'''
Handles connections and operations with QuestDB.
'''
import os
import psycopg2
from psycopg2 import sql, OperationalError
class QuestDBConnector:
    def __init__(self):
        self.connection = None
    def connect(self):
        try:
            self.connection = psycopg2.connect(
                dbname=os.getenv('DB_NAME'),
                user=os.getenv('DB_USER'),
                password=os.getenv('DB_PASSWORD'),
                host=os.getenv('DB_HOST'),
                port=os.getenv('DB_PORT')
            )
            print("Connected to QuestDB")
        except OperationalError as e:
            print(f"ERROR: Could not connect to QuestDB: {e}")
    def insert_data(self, data):
        if self.connection is None:
            print("ERROR: No connection to QuestDB")
            return
        try:
            with self.connection.cursor() as cursor:
                insert_query = sql.SQL("INSERT INTO your_table (column_name) VALUES (%s)")
                cursor.execute(insert_query, (data,))
                self.connection.commit()
                print(f"Inserted data into QuestDB: {data}")
        except Exception as e:
            print(f"ERROR: Failed to insert data: {e}")
    def close_connection(self):
        if self.connection:
            self.connection.close()
            print("Connection to QuestDB closed")

'''
Handles connections and operations with QuestDB.
'''
import os
import psycopg2
from psycopg2 import sql, OperationalError
class QuestDBConnector:
    def __init__(self):
        self.connection = None
    def connect(self):
        try:
            self.connection = psycopg2.connect(
                dbname=os.getenv('DB_NAME'),
                user=os.getenv('DB_USER'),
                password=os.getenv('DB_PASSWORD'),
                host=os.getenv('DB_HOST'),
                port=os.getenv('DB_PORT')
            )
            print("Connected to QuestDB")
        except OperationalError as e:
            print(f"ERROR: Could not connect to QuestDB: {e}")
    def insert_data(self, data):
        if self.connection is None:
            print("ERROR: No connection to QuestDB")
            return
        try:
            with self.connection.cursor() as cursor:
                insert_query = sql.SQL("INSERT INTO your_table (column_name) VALUES (%s)")
                cursor.execute(insert_query, (data,))
                self.connection.commit()
                print(f"Inserted data into QuestDB: {data}")
        except Exception as e:
            print(f"ERROR: Failed to insert data: {e}")
    def close_connection(self):
        if self.connection:
            self.connection.close()
            print("Connection to QuestDB closed")

        self.logger.log_info("Extracting data...")

        self.logger.log_info("Loading data into QuestDB...")

        transformed_data = self.transform_data(data)
        self.load_data(transformed_data)

'''
Handles connections and operations with QuestDB.
'''
import os
import psycopg2
from psycopg2 import sql, OperationalError
class QuestDBConnector:
    def __init__(self):
        self.connection = None
    def connect(self):
        try:
            self.connection = psycopg2.connect(
                dbname=os.getenv('DB_NAME'),
                user=os.getenv('DB_USER'),
                password=os.getenv('DB_PASSWORD'),
                host=os.getenv('DB_HOST'),
                port=os.getenv('DB_PORT')
            )
            print("Connected to QuestDB")
        except OperationalError as e:
            print(f"ERROR: Could not connect to QuestDB: {e}")
    def insert_data(self, data):
        if self.connection is None:
            print("ERROR: No connection to QuestDB")
            return
        try:
            with self.connection.cursor() as cursor:
                insert_query = sql.SQL("INSERT INTO your_table (column_name) VALUES (%s)")
                cursor.execute(insert_query, (data,))
                self.connection.commit()
                print(f"Inserted data into QuestDB: {data}")
        except Exception as e:
            print(f"ERROR: Failed to insert data: {e}")
    def close_connection(self):
        if self.connection:
            self.connection.close()
            print("Connection to QuestDB closed")

'''
Provides a user interface for interacting with the ETL pipeline.
'''
import tkinter as tk
from tkinter import messagebox
class GUI:
    def __init__(self, etl_pipeline):
        self.etl_pipeline = etl_pipeline
    def run(self):
        root = tk.Tk()
        root.title("ETL Pipeline GUI")
        def start_pipeline():
            try:
                self.etl_pipeline.run_pipeline()
                messagebox.showinfo("Success", "ETL Pipeline executed successfully!")
            except Exception as e:
                messagebox.showerror("Error", f"An error occurred: {e}")
        start_button = tk.Button(root, text="Start ETL Pipeline", command=start_pipeline)
        start_button.pack(pady=20)
        root.mainloop()

'''
Handles connections and operations with QuestDB.
'''
import os
import psycopg2
from psycopg2 import sql, OperationalError
class QuestDBConnector:
    def __init__(self):
        self.connection = None
    def connect(self):
        try:
            self.connection = psycopg2.connect(
                dbname=os.getenv('DB_NAME'),
                user=os.getenv('DB_USER'),
                password=os.getenv('DB_PASSWORD'),
                host=os.getenv('DB_HOST'),
                port=os.getenv('DB_PORT')
            )
            print("Connected to QuestDB")
        except OperationalError as e:
            print(f"ERROR: Could not connect to QuestDB: {e}")
    def insert_data(self, data):
        if self.connection is None:
            print("ERROR: No connection to QuestDB")
            return
        try:
            with self.connection.cursor() as cursor:
                insert_query = sql.SQL("INSERT INTO your_table (column_name) VALUES (%s)")
                cursor.execute(insert_query, (data,))
                self.connection.commit()
                print(f"Inserted data into QuestDB: {data}")
        except Exception as e:
            print(f"ERROR: Failed to insert data: {e}")
    def close_connection(self):
        if self.connection:
            self.connection.close()
            print("Connection to QuestDB closed")

        start_button.pack(pady=20)
