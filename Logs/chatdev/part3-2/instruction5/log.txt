class DesignUtils:

    @staticmethod
    def create_square_logo(text, output_path):

        image = Image.new("RGB", size, "white")

'''
Utility functions for creating images for the logo and cover image.
'''
from PIL import Image, ImageDraw, ImageFont
class DesignUtils:
    @staticmethod
    def create_square_logo(text, output_path):
        # Create a square logo with the specified text
        size = (300, 300)
        image = Image.new("RGB", size, "white")
        draw = ImageDraw.Draw(image)
        font = DesignUtils.load_font(40)
        text_width, text_height = draw.textsize(text, font=font)
        text_position = ((size[0] - text_width) / 2, (size[1] - text_height) / 2)
        draw.text(text_position, text, fill="black", font=font)
        image.save(output_path)
    @staticmethod
    def create_cover_image(text, output_path):
        # Create a cover image with the specified text
        size = (820, 312)
        image = Image.new("RGB", size, "white")
        draw = ImageDraw.Draw(image)
        font = DesignUtils.load_font(40)
        text_width, text_height = draw.textsize(text, font=font)
        text_position = ((size[0] - text_width) / 2, (size[1] - text_height) / 2)
        draw.text(text_position, text, fill="black", font=font)
        image.save(output_path)
    @staticmethod
    def load_font(size):
        # Load a font, with a fallback mechanism
        try:
            font = ImageFont.truetype("arial.ttf", size)
        except OSError:
            try:
                # Attempt to load a bundled font file
                font = ImageFont.truetype("path/to/bundled/font.ttf", size)
            except OSError:
                # Fallback to default font and notify the user
                font = ImageFont.load_default()
                print("Warning: Custom font not found. Using default font.")
        return font
