from smartcard.System import readers

class ACSReader:
    def __init__(self):
        self.reader = None
    def connect_reader(self):
        # Logic to connect to the ACS ACR1552U reader
        print("Connecting to ACS ACR1552U reader...")
        available_readers = readers()
        if available_readers:
            self.reader = available_readers[0]
            print(f"Connected to reader: {self.reader}")
        else:
            print("No readers available.")
    def disconnect_reader(self):
        # Logic to disconnect the reader
        print("Disconnecting from ACS ACR1552U reader...")
        self.reader = None
    def send_command(self, command):
        # Logic to send a command to the reader
        print(f"Sending command to reader: {command}")
        if self.reader:
            connection = self.reader.createConnection()
            connection.connect()
            response, sw1, sw2 = connection.transmit(command)
            print(f"Response: {toHexString(response)}, Status: {sw1:X} {sw2:X}")
        else:
            print("No reader connected.")
