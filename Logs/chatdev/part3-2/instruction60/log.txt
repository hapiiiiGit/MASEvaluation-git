    db_handler.save_candidate(candidate)
    flash('Candidate information submitted successfully!', 'success')

'''

import json

class DatabaseHandler:
    def __init__(self, filename='candidates.json'):
        self.filename = filename

    def save_candidate(self, candidate):
        try:
            candidates = self.load_candidates()
        except FileNotFoundError:
            candidates = []
        candidates.append(candidate.__dict__)
        with open(self.filename, 'w') as file:
            json.dump(candidates, file, indent=4)

    def load_candidates(self):
        with open(self.filename, 'r') as file:
            return json.load(file)

'''
Handles data storage and retrieval for candidates.
'''
import json
class DatabaseHandler:
    def __init__(self, filename='candidates.json'):
        self.filename = filename
    def save_candidate(self, candidate):
        try:
            candidates = self.load_candidates()
        except FileNotFoundError:
            candidates = []
        candidates.append(candidate.__dict__)
        with open(self.filename, 'w') as file:
            json.dump(candidates, file, indent=4)
    def load_candidates(self):
        with open(self.filename, 'r') as file:
            return json.load(file)

'''
Handles data storage and retrieval for candidates.
'''
import json
class DatabaseHandler:
    def __init__(self, filename='candidates.json'):
        self.filename = filename
    def save_candidate(self, candidate):
        try:
            candidates = self.load_candidates()
        except FileNotFoundError:
            candidates = []
        candidates.append(candidate.__dict__)
        with open(self.filename, 'w') as file:
            json.dump(candidates, file, indent=4)
    def load_candidates(self):
        with open(self.filename, 'r') as file:
            return json.load(file)

import tkinter as tk
import tkinter.messagebox as messagebox
from candidate import Candidate
from database_handler import DatabaseHandler
class CandidateForm:
    def __init__(self, master):
        self.master = master
        self.db_handler = DatabaseHandler()
    def create_form(self):
        self.name_label = tk.Label(self.master, text="Name:")
        self.name_label.grid(row=0, column=0)
        self.name_entry = tk.Entry(self.master)
        self.name_entry.grid(row=0, column=1)
        self.email_label = tk.Label(self.master, text="Email:")
        self.email_label.grid(row=1, column=0)
        self.email_entry = tk.Entry(self.master)
        self.email_entry.grid(row=1, column=1)
        self.experience_label = tk.Label(self.master, text="Experience:")
        self.experience_label.grid(row=2, column=0)
        self.experience_entry = tk.Entry(self.master)
        self.experience_entry.grid(row=2, column=1)
        self.development_approach_label = tk.Label(self.master, text="Development Approach:")
        self.development_approach_label.grid(row=3, column=0)
        self.development_approach_entry = tk.Entry(self.master)
        self.development_approach_entry.grid(row=3, column=1)
        self.submit_button = tk.Button(self.master, text="Submit", command=self.submit_form)
        self.submit_button.grid(row=4, column=1)
    def submit_form(self):
        name = self.name_entry.get()
        email = self.email_entry.get()
        experience = self.experience_entry.get()
        development_approach = self.development_approach_entry.get()
        candidate = Candidate(name, email, experience, development_approach)
        self.db_handler.save_candidate(candidate)
        self.name_entry.delete(0, tk.END)
        self.email_entry.delete(0, tk.END)
        self.experience_entry.delete(0, tk.END)
        self.development_approach_entry.delete(0, tk.END)
        messagebox.showinfo("Success", "Candidate information submitted successfully!")

'''
Handles data storage and retrieval for candidates.
'''
import json
class DatabaseHandler:
    def __init__(self, filename='candidates.json'):
        self.filename = filename
    def save_candidate(self, candidate):
        try:
            candidates = self.load_candidates()
        except FileNotFoundError:
            candidates = []
        candidates.append(candidate.__dict__)
        with open(self.filename, 'w') as file:
            json.dump(candidates, file, indent=4)
    def load_candidates(self):
        with open(self.filename, 'r') as file:
            return json.load(file)

from database_handler import DatabaseHandler
class CandidateForm:
    def __init__(self, master):
        self.master = master
        self.db_handler = DatabaseHandler()
    def create_form(self):
        self.name_label = tk.Label(self.master, text="Name:")
        self.name_label.grid(row=0, column=0)
        self.name_entry = tk.Entry(self.master)
        self.name_entry.grid(row=0, column=1)
        self.email_label = tk.Label(self.master, text="Email:")
        self.email_label.grid(row=1, column=0)
        self.email_entry = tk.Entry(self.master)
        self.email_entry.grid(row=1, column=1)
        self.experience_label = tk.Label(self.master, text="Experience:")
        self.experience_label.grid(row=2, column=0)
        self.experience_entry = tk.Entry(self.master)
        self.experience_entry.grid(row=2, column=1)
        self.development_approach_label = tk.Label(self.master, text="Development Approach:")
        self.development_approach_label.grid(row=3, column=0)
        self.development_approach_entry = tk.Entry(self.master)
        self.development_approach_entry.grid(row=3, column=1)
        self.submit_button = tk.Button(self.master, text="Submit", command=self.submit_form)
        self.submit_button.grid(row=4, column=1)
    def submit_form(self):
        name = self.name_entry.get()
        email = self.email_entry.get()
        experience = self.experience_entry.get()
        development_approach = self.development_approach_entry.get()
        candidate = Candidate(name, email, experience, development_approach)
        self.db_handler.save_candidate(candidate)
        self.name_entry.delete(0, tk.END)
        self.email_entry.delete(0, tk.END)
        self.experience_entry.delete(0, tk.END)
        self.development_approach_entry.delete(0, tk.END)
        messagebox.showinfo("Success", "Candidate information submitted successfully!")
