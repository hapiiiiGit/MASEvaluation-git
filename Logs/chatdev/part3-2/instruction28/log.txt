def store_consent(customer_id):
    # Connect to the SQLite database
    conn = sqlite3.connect('consent_data.db')
    cursor = conn.cursor()

        CREATE TABLE IF NOT EXISTS consent (

            customer_id TEXT NOT NULL,
            consent_given BOOLEAN NOT NULL
        )
    ''')

        INSERT INTO consent (customer_id, consent_given)

'''

import sqlite3
def store_consent(customer_id):
    # Connect to the SQLite database
    conn = sqlite3.connect('consent_data.db')
    cursor = conn.cursor()
    # Create table if it doesn't exist
    cursor.execute('''
        CREATE TABLE IF NOT EXISTS consent (
            id INTEGER PRIMARY KEY AUTOINCREMENT,
            customer_id TEXT NOT NULL,
            consent_given BOOLEAN NOT NULL
        )
    ''')
    # Insert consent data into the table
    cursor.execute('''
        INSERT INTO consent (customer_id, consent_given)
        VALUES (?, ?)
    ''', (customer_id, True))
    # Commit changes and close the connection
    conn.commit()
    conn.close()

'''
Backend logic for handling customer consent storage using a database.

import sqlite3
def store_consent(customer_id):
    # Connect to the SQLite database
    conn = sqlite3.connect('consent_data.db')
    cursor = conn.cursor()
    # Create table if it doesn't exist
    cursor.execute('''
        CREATE TABLE IF NOT EXISTS consent (
            id INTEGER PRIMARY KEY AUTOINCREMENT,
            customer_id TEXT NOT NULL,
            consent_given BOOLEAN NOT NULL
        )
    ''')
    # Insert consent data into the table
    cursor.execute('''
        INSERT INTO consent (customer_id, consent_given)
        VALUES (?, ?)
    ''', (customer_id, True))
    # Commit changes and close the connection
    conn.commit()
    conn.close()

class CustomerPortal(http.Controller):

'''
Backend logic for handling customer consent storage using a database.
'''
import sqlite3
def store_consent(customer_id):
    # Connect to the SQLite database
    conn = sqlite3.connect('consent_data.db')
    cursor = conn.cursor()
    # Create table if it doesn't exist
    cursor.execute('''
        CREATE TABLE IF NOT EXISTS consent (
            id INTEGER PRIMARY KEY AUTOINCREMENT,
            customer_id TEXT NOT NULL,
            consent_given BOOLEAN NOT NULL
        )
    ''')
    # Insert consent data into the table
    cursor.execute('''
        INSERT INTO consent (customer_id, consent_given)
        VALUES (?, ?)
    ''', (customer_id, True))
    # Commit changes and close the connection
    conn.commit()
    conn.close()

'''

from odoo import http
from odoo.http import request
class CustomerPortal(http.Controller):
    @http.route('/customer_portal', type='http', auth='public', website=True)
    def index(self, **kw):
        return request.render('customer_portal_redesign.index_template')
    @http.route('/submit_consent', type='http', auth='public', methods=['POST'], website=True)
    def submit_consent(self, **post):
        customer_id = int(post.get('customer_id'))
        consent_given = post.get('terms_checkbox') == 'on'
        if consent_given:
            request.env['customer.consent'].create({
                'customer_id': customer_id,
                'consent_given': True
            })
            return request.render('customer_portal_redesign.thank_you_template')
        return request.redirect('/customer_portal')

'''
Backend logic for handling customer consent storage using a database.
'''
import sqlite3
def store_consent(customer_id):
    # Connect to the SQLite database
    conn = sqlite3.connect('consent_data.db')
    cursor = conn.cursor()
    # Create table if it doesn't exist
    cursor.execute('''
        CREATE TABLE IF NOT EXISTS consent (
            id INTEGER PRIMARY KEY AUTOINCREMENT,
            customer_id TEXT NOT NULL,
            consent_given BOOLEAN NOT NULL
        )
    ''')
    # Insert consent data into the table
    cursor.execute('''
        INSERT INTO consent (customer_id, consent_given)
        VALUES (?, ?)
    ''', (customer_id, True))
    # Commit changes and close the connection
    conn.commit()
    conn.close()
