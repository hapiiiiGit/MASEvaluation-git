import librosa
import numpy as np

class AudioAnalyzer:
    def __init__(self, model_path):

'''
AudioAnalyzer class for processing audio files and predicting speaker gender.
'''
import librosa
import numpy as np
from sklearn.externals import joblib
class AudioAnalyzer:
    def __init__(self, model_path):
        self.model = joblib.load(model_path)
    def load_audio(self, file_path):
        # Load audio file
        audio_data, _ = librosa.load(file_path, sr=16000)
        return audio_data
    def extract_features(self, audio_data):
        # Extract MFCC features
        mfccs = librosa.feature.mfcc(y=audio_data, sr=16000, n_mfcc=13)
        return np.mean(mfccs.T, axis=0)
    def predict_gender(self, features):
        # Predict gender
        prediction = self.model.predict([features])
        confidence = max(self.model.predict_proba([features])[0])
        gender = 'Male' if prediction[0] == 1 else 'Female'
        return gender, confidence

'''

import librosa
import numpy as np
import joblib

class AudioAnalyzer:
    def __init__(self, model_path):
        self.model = joblib.load(model_path)

    def load_audio(self, file_path):
        # Load audio file
        audio_data, _ = librosa.load(file_path, sr=16000)
        return audio_data

    def extract_features(self, audio_data):
        # Extract MFCC features
        mfccs = librosa.feature.mfcc(y=audio_data, sr=16000, n_mfcc=13)
        return np.mean(mfccs.T, axis=0)

    def predict_gender(self, features):
        # Predict gender
        prediction = self.model.predict([features])
        confidence = max(self.model.predict_proba([features])[0])
        gender = 'Male' if prediction[0] == 1 else 'Female'
        return gender, confidence

```python
'''
AudioAnalyzer class for processing audio files and predicting speaker gender.
'''
import librosa
import numpy as np
import joblib
class AudioAnalyzer:
    def __init__(self, model_path):
        self.model = joblib.load(model_path)
    def load_audio(self, file_path):
        # Load audio file
        audio_data, _ = librosa.load(file_path, sr=16000)
        return audio_data
    def extract_features(self, audio_data):
        # Extract MFCC features
        mfccs = librosa.feature.mfcc(y=audio_data, sr=16000, n_mfcc=13)
        return np.mean(mfccs.T, axis=0)
    def predict_gender(self, features):
        # Predict gender
        prediction = self.model.predict([features])
        confidence = max(self.model.predict_proba([features])[0])
        gender = 'Male' if prediction[0] == 1 else 'Female'
        return gender, confidence

'''
AudioAnalyzer class for processing audio files and predicting speaker gender.
'''
import librosa
import numpy as np
import joblib
class AudioAnalyzer:
    def __init__(self, model_path):
        self.model = joblib.load(model_path)
    def load_audio(self, file_path):
        # Load audio file
        audio_data, _ = librosa.load(file_path, sr=16000)
        return audio_data
    def extract_features(self, audio_data):
        # Extract MFCC features
        mfccs = librosa.feature.mfcc(y=audio_data, sr=16000, n_mfcc=13)
        return np.mean(mfccs.T, axis=0)
    def predict_gender(self, features):
        # Predict gender
        prediction = self.model.predict([features])
        confidence = max(self.model.predict_proba([features])[0])
        gender = 'Male' if prediction[0] == 1 else 'Female'
        return gender, confidence

'''
AudioAnalyzer class for processing audio files and predicting speaker gender.
'''
import librosa
import numpy as np
import joblib

class AudioAnalyzer:
    def __init__(self, model_path):
        '''
        Initialize the AudioAnalyzer with a pre-trained model.
        
        :param model_path: Path to the pre-trained model file.
        '''
        self.model = joblib.load(model_path)

    def load_audio(self, file_path):
        '''
        Load an audio file and return the audio data.
        
        :param file_path: Path to the audio file.
        :return: Audio data as a numpy array.
        '''
        audio_data, _ = librosa.load(file_path, sr=16000)
        return audio_data

    def extract_features(self, audio_data):
        '''
        Extract MFCC features from the audio data.
        
        :param audio_data: Audio data as a numpy array.
        :return: Extracted MFCC features.
        '''
        mfccs = librosa.feature.mfcc(y=audio_data, sr=16000, n_mfcc=13)
        return np.mean(mfccs.T, axis=0)

    def predict_gender(self, features):
        '''
        Predict the gender of the speaker from the extracted features.
        
        :param features: Extracted MFCC features.
        :return: Predicted gender and confidence score.
        '''
        prediction = self.model.predict([features])
        confidence = max(self.model.predict_proba([features])[0])
        gender = 'Male' if prediction[0] == 1 else 'Female'
        return gender, confidence

'''
AudioAnalyzer class for processing audio files and predicting speaker gender.
'''
import librosa
import numpy as np
import joblib
class AudioAnalyzer:
    def __init__(self, model_path):
        '''
        Initialize the AudioAnalyzer with a pre-trained model.
        :param model_path: Path to the pre-trained model file.
        '''
        self.model = joblib.load(model_path)
    def load_audio(self, file_path):
        '''
        Load an audio file and return the audio data.
        :param file_path: Path to the audio file.
        :return: Audio data as a numpy array.
        '''
        audio_data, _ = librosa.load(file_path, sr=16000)
        return audio_data
    def extract_features(self, audio_data):
        '''
        Extract MFCC features from the audio data.
        :param audio_data: Audio data as a numpy array.
        :return: Extracted MFCC features.
        '''
        mfccs = librosa.feature.mfcc(y=audio_data, sr=16000, n_mfcc=13)
        return np.mean(mfccs.T, axis=0)
    def predict_gender(self, features):
        '''
        Predict the gender of the speaker from the extracted features.
        :param features: Extracted MFCC features.
        :return: Predicted gender and confidence score.
        '''
        prediction = self.model.predict([features])
        confidence = max(self.model.predict_proba([features])[0])
        gender = 'Male' if prediction[0] == 1 else 'Female'
        return gender, confidence

import tkinter as tk
from tkinter import filedialog, messagebox
from audio_analyzer import AudioAnalyzer
class GenderDetectorGUI:
    def __init__(self):
        '''
        Initialize the GUI for the gender detection application.
        '''
        self.root = tk.Tk()
        self.root.title("Gender Detector")
        self.analyzer = AudioAnalyzer("gender_model.pkl")
        self.label = tk.Label(self.root, text="Select an audio file (MP3 or WAV):")
        self.label.pack()
        self.select_button = tk.Button(self.root, text="Select File", command=self.load_file)
        self.select_button.pack()
        self.result_label = tk.Label(self.root, text="")
        self.result_label.pack()
    def load_file(self):
        '''
        Open a file dialog to select an audio file and analyze it.
        '''
        file_path = filedialog.askopenfilename(filetypes=[("Audio Files", "*.mp3 *.wav")])
        if file_path:
            self.analyze_audio(file_path)
    def analyze_audio(self, file_path):
        '''
        Analyze the selected audio file to predict the speaker's gender.
        :param file_path: Path to the audio file.
        '''
        try:
            audio_data = self.analyzer.load_audio(file_path)
            features = self.analyzer.extract_features(audio_data)
            gender, confidence = self.analyzer.predict_gender(features)
            self.result_label.config(text=f"Gender: {gender}, Confidence: {confidence:.2f}")

            messagebox.showerror("Error", f"Failed to analyze audio: {e}")

audio_analyzer.py
```python
'''
AudioAnalyzer class for processing audio files and predicting speaker gender.
'''
import librosa
import numpy as np
import joblib
class AudioAnalyzer:
    def __init__(self, model_path):
        '''
        Initialize the AudioAnalyzer with a pre-trained model.
        :param model_path: Path to the pre-trained model file.
        '''
        self.model = joblib.load(model_path)
    def load_audio(self, file_path):
        '''
        Load an audio file and return the audio data.
        :param file_path: Path to the audio file.
        :return: Audio data as a numpy array.
        '''
        audio_data, _ = librosa.load(file_path, sr=16000)
        return audio_data
    def extract_features(self, audio_data):
        '''
        Extract MFCC features from the audio data.
        :param audio_data: Audio data as a numpy array.
        :return: Extracted MFCC features.
        '''
        mfccs = librosa.feature.mfcc(y=audio_data, sr=16000, n_mfcc=13)
        return np.mean(mfccs.T, axis=0)
    def predict_gender(self, features):
        '''
        Predict the gender of the speaker from the extracted features.
        :param features: Extracted MFCC features.
        :return: Predicted gender and confidence score.
        '''
        prediction = self.model.predict([features])
        confidence = max(self.model.predict_proba([features])[0])
        gender = 'Male' if prediction[0] == 1 else 'Female'
        return gender, confidence

'''
GenderDetectorGUI class for the application's graphical user interface.
'''
import tkinter as tk
from tkinter import filedialog, messagebox
from audio_analyzer import AudioAnalyzer
class GenderDetectorGUI:
    def __init__(self):
        '''
        Initialize the GUI for the gender detection application.
        '''
        self.root = tk.Tk()
        self.root.title("Gender Detector")
        self.analyzer = AudioAnalyzer("gender_model.pkl")
        self.label = tk.Label(self.root, text="Select an audio file (MP3 or WAV):")
        self.label.pack()
        self.select_button = tk.Button(self.root, text="Select File", command=self.load_file)
        self.select_button.pack()
        self.result_label = tk.Label(self.root, text="")
        self.result_label.pack()
    def load_file(self):
        '''
        Open a file dialog to select an audio file and analyze it.
        '''
        file_path = filedialog.askopenfilename(filetypes=[("Audio Files", "*.mp3 *.wav")])
        if file_path:
            self.analyze_audio(file_path)
    def analyze_audio(self, file_path):
        '''
        Analyze the selected audio file to predict the speaker's gender.
        :param file_path: Path to the audio file.
        '''
        try:
            audio_data = self.analyzer.load_audio(file_path)
            features = self.analyzer.extract_features(audio_data)
            gender, confidence = self.analyzer.predict_gender(features)
            self.result_label.config(text=f"Gender: {gender}, Confidence: {confidence:.2f}")
        except Exception as e:
            messagebox.showerror("Error", f"Failed to analyze audio: {e}")
    def run(self):
        '''
        Run the main loop of the GUI application.
        '''
        self.root.mainloop()

'''
AudioAnalyzer class for processing audio files and predicting speaker gender.
'''
import librosa
import numpy as np
import joblib
class AudioAnalyzer:
    def __init__(self, model_path):
        '''
        Initialize the AudioAnalyzer with a pre-trained model.
        :param model_path: Path to the pre-trained model file.
        '''
        self.model = joblib.load(model_path)
    def load_audio(self, file_path):
        '''
        Load an audio file and return the audio data.
        :param file_path: Path to the audio file.
        :return: Audio data as a numpy array.
        '''
        audio_data, _ = librosa.load(file_path, sr=16000)
        return audio_data
    def extract_features(self, audio_data):
        '''
        Extract MFCC features from the audio data.
        :param audio_data: Audio data as a numpy array.
        :return: Extracted MFCC features.
        '''
        mfccs = librosa.feature.mfcc(y=audio_data, sr=16000, n_mfcc=13)
        return np.mean(mfccs.T, axis=0)
    def predict_gender(self, features):
        '''
        Predict the gender of the speaker from the extracted features.
        :param features: Extracted MFCC features.
        :return: Predicted gender and confidence score.
        '''
        prediction = self.model.predict([features])
        confidence = max(self.model.predict_proba([features])[0])
        gender = 'Male' if prediction[0] == 1 else 'Female'
        return gender, confidence

'''
AudioAnalyzer class for processing audio files and predicting speaker gender.
'''
import librosa
import numpy as np
import joblib
class AudioAnalyzer:
    def __init__(self, model_path):
        '''
        Initialize the AudioAnalyzer with a pre-trained model.
        :param model_path: Path to the pre-trained model file.
        '''
        self.model = joblib.load(model_path)
    def load_audio(self, file_path):
        '''
        Load an audio file and return the audio data.
        :param file_path: Path to the audio file.
        :return: Audio data as a numpy array.
        '''
        audio_data, _ = librosa.load(file_path, sr=16000)
        return audio_data
    def extract_features(self, audio_data):
        '''
        Extract MFCC features from the audio data.
        :param audio_data: Audio data as a numpy array.
        :return: Extracted MFCC features.
        '''
        mfccs = librosa.feature.mfcc(y=audio_data, sr=16000, n_mfcc=13)
        return np.mean(mfccs.T, axis=0)
    def predict_gender(self, features):
        '''
        Predict the gender of the speaker from the extracted features.
        :param features: Extracted MFCC features.
        :return: Predicted gender and confidence score.
        '''
        prediction = self.model.predict([features])
        confidence = max(self.model.predict_proba([features])[0])
        gender = 'Male' if prediction[0] == 1 else 'Female'
        return gender, confidence
