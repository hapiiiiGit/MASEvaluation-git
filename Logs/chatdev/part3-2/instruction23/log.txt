    def read_excel_file(self):
        wb = load_workbook("output_data.xlsx")

'''
Client for interacting with the GPT-4 API to generate input data.
'''
import os
import requests
class GPT4Client:
    def __init__(self):
        self.api_url = "https://api.openai.com/v1/engines/gpt-4/completions"
        self.api_key = os.getenv("GPT4_API_KEY")
    def generate_input_data(self):
        headers = {
            "Authorization": f"Bearer {self.api_key}",
            "Content-Type": "application/json"
        }
        data = {
            "prompt": "Generate input data for automated test scenarios.",
            "max_tokens": 100
        }
        try:
            response = requests.post(self.api_url, headers=headers, json=data)
            response.raise_for_status()  # Raises an HTTPError for bad responses
            return response.json()["choices"][0]["text"]
        except requests.exceptions.RequestException as e:
            print(f"An error occurred: {e}")
            return None

'''
Manages creation and manipulation of Excel files.

from openpyxl import Workbook, load_workbook
class ExcelManager:
    def create_excel_file(self, data):
        wb = Workbook()
        ws = wb.active
        ws.append(data.split("\n"))
        wb.save("input_data.xlsx")
    def read_excel_file(self):
        wb = load_workbook("output_data.xlsx")
        ws = wb.active
        return [row for row in ws.iter_rows(values_only=True)]

'''

import sqlite3
class DatabaseManager:
    def __init__(self):
        self.conn = sqlite3.connect('data.db')
        self.initialize_db()
    def initialize_db(self):
        cursor = self.conn.cursor()
        cursor.execute('''
            CREATE TABLE IF NOT EXISTS logs (
                id INTEGER PRIMARY KEY AUTOINCREMENT,
                message TEXT
            )
        ''')
        self.conn.commit()
    def log_to_db(self, message):
        cursor = self.conn.cursor()
        cursor.execute('INSERT INTO logs (message) VALUES (?)', (message,))
        self.conn.commit()
    def retrieve_data(self):
        cursor = self.conn.cursor()
        cursor.execute('SELECT * FROM logs')
        return cursor.fetchall()

'''
Client for interacting with the GPT-4 API to generate input data.
'''
import os
import requests
class GPT4Client:
    def __init__(self):
        self.api_url = "https://api.openai.com/v1/engines/gpt-4/completions"
        self.api_key = os.getenv("GPT4_API_KEY")
    def generate_input_data(self):
        headers = {
            "Authorization": f"Bearer {self.api_key}",
            "Content-Type": "application/json"
        }
        data = {
            "prompt": "Generate input data for automated test scenarios.",
            "max_tokens": 100
        }
        try:
            response = requests.post(self.api_url, headers=headers, json=data)
            response.raise_for_status()  # Raises an HTTPError for bad responses
            return response.json()["choices"][0]["text"]
        except requests.exceptions.RequestException as e:
            print(f"An error occurred: {e}")
            return None

import os

class GPT4Client:
    def __init__(self):
        self.api_url = "https://api.openai.com/v1/engines/gpt-4/completions"

import os
import requests
import tkinter as tk
from tkinter import simpledialog

class GPT4Client:
    def __init__(self):
        self.api_url = "https://api.openai.com/v1/engines/gpt-4/completions"
        self.api_key = os.getenv("GPT4_API_KEY") or self.prompt_for_api_key()
        self.db_manager = DatabaseManager()
    def prompt_for_api_key(self):
        root = tk.Tk()
        root.withdraw()  # Hide the main window
        api_key = simpledialog.askstring("API Key", "Enter your GPT-4 API Key:")
        root.destroy()
        if not api_key:
            raise ValueError("API Key is required to proceed.")
        return api_key

        headers = {
            "Authorization": f"Bearer {self.api_key}",
            "Content-Type": "application/json"

            "max_tokens": 100
        }

            response = requests.post(self.api_url, headers=headers, json=data)
            response.raise_for_status()  # Raises an HTTPError for bad responses

import sqlite3
class DatabaseManager:
    def __init__(self):
        self.conn = sqlite3.connect('data.db')
        self.initialize_db()
    def initialize_db(self):
        cursor = self.conn.cursor()
        cursor.execute('''
            CREATE TABLE IF NOT EXISTS logs (
                id INTEGER PRIMARY KEY AUTOINCREMENT,
                message TEXT
            )
        ''')
        self.conn.commit()
    def log_to_db(self, message):
        cursor = self.conn.cursor()
        cursor.execute('INSERT INTO logs (message) VALUES (?)', (message,))
        self.conn.commit()
    def retrieve_data(self):
        cursor = self.conn.cursor()
        cursor.execute('SELECT * FROM logs')
        return cursor.fetchall()

'''

import os
import requests
import tkinter as tk
from tkinter import simpledialog
from database_manager import DatabaseManager
class GPT4Client:
    def __init__(self):
        self.api_url = "https://api.openai.com/v1/engines/gpt-4/completions"
        self.api_key = os.getenv("GPT4_API_KEY") or self.prompt_for_api_key()
        self.db_manager = DatabaseManager()
    def prompt_for_api_key(self):
        root = tk.Tk()
        root.withdraw()  # Hide the main window
        api_key = simpledialog.askstring("API Key", "Enter your GPT-4 API Key:")
        root.destroy()
        if not api_key:
            raise ValueError("API Key is required to proceed.")
        return api_key
    def generate_input_data(self):
        headers = {
            "Authorization": f"Bearer {self.api_key}",
            "Content-Type": "application/json"
        }
        data = {
            "prompt": "Generate input data for automated test scenarios.",
            "max_tokens": 100
        }
        try:
            response = requests.post(self.api_url, headers=headers, json=data)
            response.raise_for_status()  # Raises an HTTPError for bad responses
            return response.json()["choices"][0]["text"]
        except requests.exceptions.RequestException as e:
            error_message = f"An error occurred: {e}"
            print(error_message)
            # Log error to database
            self.db_manager.log_to_db(error_message)
            # Return an empty string or raise a custom exception
            return ""

'''
Manages creation and manipulation of Excel files.

from openpyxl import Workbook, load_workbook
class ExcelManager:
    def create_excel_file(self, data):
        wb = Workbook()
        ws = wb.active
        ws.append(data.split("\n"))
        wb.save("input_data.xlsx")
    def read_excel_file(self):
        wb = load_workbook("output_data.xlsx")
        ws = wb.active
        return [row for row in ws.iter_rows(values_only=True)]

import sqlite3
class DatabaseManager:
    def __init__(self):
        self.conn = sqlite3.connect('data.db')
        self.initialize_db()
    def initialize_db(self):
        cursor = self.conn.cursor()
        cursor.execute('''
            CREATE TABLE IF NOT EXISTS logs (
                id INTEGER PRIMARY KEY AUTOINCREMENT,
                message TEXT
            )
        ''')
        self.conn.commit()
    def log_to_db(self, message):
        cursor = self.conn.cursor()
        cursor.execute('INSERT INTO logs (message) VALUES (?)', (message,))
        self.conn.commit()
    def retrieve_data(self):
        cursor = self.conn.cursor()
        cursor.execute('SELECT * FROM logs')
        return cursor.fetchall()
