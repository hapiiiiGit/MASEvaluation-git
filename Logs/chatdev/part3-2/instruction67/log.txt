'''

Main application file to launch the GUI for automation tasks.

'''

import tkinter as tk
from task_manager import TaskManager

class MainApp:
    def __init__(self):
        self.root = tk.Tk()
        self.root.title("Automation Script Application")
        self.task_manager = TaskManager()
        self.create_widgets()

    def create_widgets(self):
        self.label = tk.Label(self.root, text="Select a task to execute:")
        self.label.pack(pady=10)

        self.task_listbox = tk.Listbox(self.root)
        for task in self.task_manager.tasks:
            self.task_listbox.insert(tk.END, task)
        self.task_listbox.pack(pady=10)

        self.execute_button = tk.Button(self.root, text="Execute Task", command=self.execute_task)
        self.execute_button.pack(pady=10)

    def execute_task(self):
        selected_task = self.task_listbox.get(tk.ACTIVE)
        if selected_task:
            self.task_manager.execute_task(selected_task)

    def run(self):
        self.root.mainloop()

if __name__ == "__main__":
    app = MainApp()
    app.run()

'''

import os
import shutil
class TaskManager:
    def __init__(self):
        self.tasks = {
            "Task One": self.task_one,
            "Task Two": self.task_two
        }
    def execute_task(self, task_name):
        task_function = self.tasks.get(task_name)
        if task_function:
            try:
                task_function()
                return f"Task '{task_name}' executed successfully."
            except Exception as e:
                return f"An error occurred during '{task_name}': {e}"
        else:
            return f"Task '{task_name}' not found."
    def task_one(self):
        print("Executing Task One...")
        # Example: Automate file operations
        source_dir = "source_directory"
        destination_dir = "destination_directory"
        # Ensure directories exist
        if not os.path.exists(source_dir):
            os.makedirs(source_dir)
        if not os.path.exists(destination_dir):
            os.makedirs(destination_dir)
        try:
            files = os.listdir(source_dir)
            for file in files:
                full_file_name = os.path.join(source_dir, file)
                if os.path.isfile(full_file_name):
                    shutil.copy(full_file_name, destination_dir)
            print(f"Files copied from {source_dir} to {destination_dir}.")
        except Exception as e:
            print(f"An error occurred during file operations: {e}")
    def task_two(self):
        print("Executing Task Two...")
        # Example: Perform data processing tasks

        processed_data = [x**2 for x in data]
        print(f"Processed data: {processed_data}")

'''
Task manager module to handle and execute automation tasks.
'''
import os
import shutil
class TaskManager:
    def __init__(self):
        self.tasks = {
            "Task One": self.task_one,
            "Task Two": self.task_two
        }
    def execute_task(self, task_name):
        task_function = self.tasks.get(task_name)
        if task_function:
            try:
                task_function()
                return f"Task '{task_name}' executed successfully."
            except Exception as e:
                return f"An error occurred during '{task_name}': {e}"
        else:
            return f"Task '{task_name}' not found."
    def task_one(self):
        print("Executing Task One...")
        # Example: Automate file operations
        source_dir = "source_directory"
        destination_dir = "destination_directory"
        # Ensure directories exist
        if not os.path.exists(source_dir):
            os.makedirs(source_dir)
        if not os.path.exists(destination_dir):
            os.makedirs(destination_dir)
        try:
            files = os.listdir(source_dir)
            for file in files:
                full_file_name = os.path.join(source_dir, file)
                if os.path.isfile(full_file_name):
                    shutil.copy(full_file_name, destination_dir)
            print(f"Files copied from {source_dir} to {destination_dir}.")
        except Exception as e:
            print(f"An error occurred during file operations: {e}")
    def task_two(self):
        print("Executing Task Two...")
        # Example: Perform data processing tasks
        data = [1, 2, 3, 4, 5]
        processed_data = [x**2 for x in data]
        print(f"Processed data: {processed_data}")
