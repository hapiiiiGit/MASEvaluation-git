'''
Main entry point for the web application.
'''

from flask import Flask, jsonify
from gui import GUI
from utils import fetch_data, secure_key_handling
import threading

class App:
    def __init__(self):
        self.app = Flask(__name__)
        self.prices = {}
        self.api_key = secure_key_handling.load_key()

        @self.app.route('/prices')
        def get_prices():
            return jsonify(self.prices)

        @self.app.route('/ip')
        def get_ip():
            return jsonify({"ip": fetch_data.get_public_ip()})

        @self.app.route('/trade', methods=['POST'])
        def place_order():
            # Logic for placing an order
            return jsonify({"status": "Order placed"})

    def update_prices(self):
        while True:
            self.prices = fetch_data.get_live_prices()
            threading.Event().wait(0.2)  # 200ms delay

    def run(self):
        threading.Thread(target=self.update_prices).start()
        self.app.run(debug=True)

if __name__ == '__main__':
    app = App()
    app.run()

        self.prices = {}
        self.api_key = secure_key_handling.load_key()

        @self.app.route('/prices')
        def get_prices():
            return jsonify(self.prices)

        @self.app.route('/ip')
        def get_ip():
            return jsonify({"ip": fetch_data.get_public_ip()})

        @self.app.route('/trade', methods=['POST'])
        def place_order():
            # Logic for placing an order
            return jsonify({"status": "Order placed"})

    def update_prices(self):
        while True:
            self.prices = fetch_data.get_live_prices()
            threading.Event().wait(0.2)  # 200ms delay

    def run(self):
        threading.Thread(target=self.update_prices).start()
        self.app.run(debug=True)

if __name__ == '__main__':

from fastapi import FastAPI, HTTPException
from pydantic import BaseModel
import uvicorn
from utils import get_live_prices, get_public_ip, secure_key_handling

app = FastAPI()

class TradeRequest(BaseModel):
    api_key: str

@app.get("/live-prices")
async def live_prices():
    try:
        prices = get_live_prices()
        return prices
    except Exception as e:
        raise HTTPException(status_code=500, detail="Error fetching live prices")

@app.get("/public-ip")
async def public_ip():
    return {"ip": get_public_ip()}

@app.post("/trade")
async def execute_trade(trade_request: TradeRequest):
    # Logic to execute trade
    if trade_request.api_key != secure_key_handling():
        raise HTTPException(status_code=403, detail="Invalid API Key")
    # Simulate trade execution
    return {"status": "success", "message": "Trade executed successfully"}

if __name__ == "__main__":
    uvicorn.run(app, host="0.0.0.0", port=5000)

import tkinter as tk
import requests
from utils import get_public_ip, secure_key_handling
class GUI:
    def __init__(self, master):
        self.master = master
        self.master.title("Live Futures Trading")
        self.ip_label = tk.Label(master, text="Public IP: ")
        self.ip_label.pack()
        self.price_label = tk.Label(master, text="Live Prices: ")
        self.price_label.pack()
        self.trade_button = tk.Button(master, text="Place Order", command=self.execute_trade)
        self.trade_button.pack()
        self.refresh_ip()
        self.update_prices_display()
        self.master.after(200, self.update_prices_display)  # Update prices every 200ms
    def refresh_ip(self):
        ip = get_public_ip()
        self.ip_label.config(text=f"Public IP: {ip}")
    def update_prices_display(self):
        try:
            prices_data = requests.get("http://localhost:5000/live-prices").json()
            self.price_label.config(text=f"Live Prices: {prices_data}")
        except Exception as e:
            self.price_label.config(text="Error fetching prices")
            print("Error:", e)
    def execute_trade(self):
        # Logic to execute trade
        response = requests.post('http://localhost:5000/trade', json={"api_key": secure_key_handling()})
        if response.status_code == 200:
            trade_info = response.json()
            print(f"Trade executed: {trade_info}")
        else:
            print("Error executing trade:", response.json())
if __name__ == '__main__':
    root = tk.Tk()
    gui = GUI(root)
    root.mainloop()

'''
Graphical User Interface for the trading application using Tkinter.

import tkinter as tk
import requests
from utils import get_public_ip, secure_key_handling
class GUI:
    def __init__(self, master):
        self.master = master
        self.master.title("Live Futures Trading")
        self.ip_label = tk.Label(master, text="Public IP: ")
        self.ip_label.pack()
        self.price_label = tk.Label(master, text="Live Prices: ")
        self.price_label.pack()
        self.trade_button = tk.Button(master, text="Place Order", command=self.execute_trade)
        self.trade_button.pack()
        self.refresh_ip()
        self.update_prices_display()
        self.master.after(200, self.update_prices_display)  # Update prices every 200ms
    def refresh_ip(self):
        ip = get_public_ip()
        self.ip_label.config(text=f"Public IP: {ip}")
    def update_prices_display(self):
        try:
            prices_data = requests.get("http://localhost:5000/live-prices").json()
            self.price_label.config(text=f"Live Prices: {prices_data}")
        except Exception as e:
            self.price_label.config(text="Error fetching prices")
            print("Error:", e)
    def execute_trade(self):
        # Logic to execute trade
        response = requests.post('http://localhost:5000/trade', json={"api_key": secure_key_handling()})
        if response.status_code == 200:
            trade_info = response.json()
            print(f"Trade executed: {trade_info}")
        else:
            print("Error executing trade:", response.json())
if __name__ == '__main__':
    root = tk.Tk()
    gui = GUI(root)
    root.mainloop()
