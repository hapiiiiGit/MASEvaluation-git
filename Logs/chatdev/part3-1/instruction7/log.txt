        Calculate Kundli based on user input and display the results.
        """
        birth_date = self.entry_date.get()

        location = self.entry_location.get().split(',')
        if len(location) != 2:
            messagebox.showerror("Input Error", "Please enter valid latitude and longitude.")
            return

            # Validate date and time format
            datetime.strptime(birth_date, '%Y-%m-%d')
            datetime.strptime(birth_time, '%H:%M')
            lat = float(location[0])
            lon = float(location[1])
            planetary_positions = self.calculate_planetary_positions(birth_date, birth_time, (lat, lon))
