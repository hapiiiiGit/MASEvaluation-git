'''
Scraper file containing functions to scrape data from https://www.loenoverblik.dk/
'''
import requests
from bs4 import BeautifulSoup
import time

def scrape_data(base_url):
    all_data = []
    # Iterate through all available periods and sector levels down to level 3
    for year in range(2010, 2024):  # Example range of years
        for sector_level in range(1, 4):  # Assuming sector levels 1 to 3
            url = f"{base_url}/data?year={year}&sector_level={sector_level}"
            response = requests.get(url)
            if response.status_code == 200:
                html_content = response.text
                data = extract_data(html_content)
                all_data.extend(data)
            time.sleep(1)  # Respectful scraping by adding delay
    return all_data

def extract_data(html_content):
    soup = BeautifulSoup(html_content, 'html.parser')
    data = []
    # Extract 'Årsværk' data for each 'Personalekategori'
    for category in soup.find_all(class_='personalekategori'):
        year_work = category.find(class_='årsværk').text.strip()
        category_name = category.find(class_='kategori_navn').text.strip()
        data.append([category_name, year_work])
    return data

import requests
from bs4 import BeautifulSoup
import time
def scrape_data(base_url):
    all_data = []
    # Iterate through all available periods and sector levels down to level 3
    for year in range(2010, 2024):  # Example range of years
        for sector_level in range(1, 4):  # Assuming sector levels 1 to 3
            url = f"{base_url}/data?year={year}&sector_level={sector_level}"
            try:
                response = requests.get(url)
                response.raise_for_status()  # Raise an error for bad responses
                html_content = response.text
                data = extract_data(html_content)
                all_data.extend(data)
            except requests.RequestException as e:
                print(f"Request failed for {url}: {e}")
            except Exception as e:
                print(f"Error extracting data for {url}: {e}")
            time.sleep(1)  # Respectful scraping by adding delay
    return all_data
def extract_data(html_content):
    soup = BeautifulSoup(html_content, 'html.parser')

        category_name = category.find(class_='actual_class_name_for_kategori_navn').text.strip()  # Update class name
        data.append([category_name, year_work])
    return data

import requests
from bs4 import BeautifulSoup
import time

def scrape_data(base_url):
    all_data = []
    # Iterate through all available periods and sector levels down to level 3
    for year in range(2010, 2024):  # Example range of years
        for sector_level in range(1, 4):  # Assuming sector levels 1 to 3
            url = f"{base_url}/data?year={year}&sector_level={sector_level}"
            try:
                response = requests.get(url)
                response.raise_for_status()  # Raise an error for bad responses
                html_content = response.text
                data = extract_data(html_content)
                all_data.extend(data)
            except requests.RequestException as e:
                print(f"Request failed for {url}: {e}")
            except Exception as e:
                print(f"Error extracting data for {url}: {e}")
            time.sleep(1)  # Respectful scraping by adding delay
    return all_data

def extract_data(html_content):
    soup = BeautifulSoup(html_content, 'html.parser')
    data = []
    # Update the class names based on the actual HTML structure
    for category in soup.find_all(class_='personalekategori_class_name'):  # Replace with actual class name
        year_work = category.find(class_='årsværk_class_name').text.strip()  # Replace with actual class name
        category_name = category.find(class_='kategori_navn_class_name').text.strip()  # Replace with actual class name
        data.append([category_name, year_work])
    return data

import time
def scrape_data(base_url):
    all_data = []
    # Iterate through all available periods and sector levels down to level 3
    for year in range(2010, 2024):  # Example range of years
        for sector_level in range(1, 4):  # Assuming sector levels 1 to 3
            url = f"{base_url}/data?year={year}&sector_level={sector_level}"
            try:
                response = requests.get(url)
                response.raise_for_status()  # Raise an error for bad responses
                html_content = response.text
                data = extract_data(html_content)
                all_data.extend(data)
            except requests.RequestException as e:
                print(f"Request failed for {url}: {e}")
            except Exception as e:
                print(f"Error extracting data for {url}: {e}")
            time.sleep(1)  # Respectful scraping by adding delay
    return all_data

'''
Scraper file containing functions to scrape data from https://www.loenoverblik.dk/
'''
import requests
from bs4 import BeautifulSoup
import time
def scrape_data(base_url):
    all_data = []
    # Iterate through all available periods and sector levels down to level 3
    for year in range(2010, 2024):  # Example range of years
        for sector_level in range(1, 4):  # Assuming sector levels 1 to 3
            url = f"{base_url}/data?year={year}&sector_level={sector_level}"
            try:
                response = requests.get(url)
                response.raise_for_status()  # Raise an error for bad responses
                html_content = response.text
                data = extract_data(html_content)
                all_data.extend(data)
            except requests.RequestException as e:
                print(f"Request failed for {url}: {e}")
            except Exception as e:
                print(f"Error extracting data for {url}: {e}")
            time.sleep(1)  # Respectful scraping by adding delay
    return all_data
def extract_data(html_content):
    soup = BeautifulSoup(html_content, 'html.parser')
    data = []
    try:
        # Update the class names based on the actual HTML structure
        for category in soup.find_all(class_='actual_personalekategori_class_name'):  # Update with actual class name
            year_work = category.find(class_='actual_årsværk_class_name').text.strip()  # Update with actual class name
            category_name = category.find(class_='actual_kategori_navn_class_name').text.strip()  # Update with actual class name
            data.append([category_name, year_work])
    except AttributeError as e:
        print(f"Error extracting data: {e}")
    return data

'''
Scraper file containing functions to scrape data from https://www.loenoverblik.dk/
'''
import requests
from bs4 import BeautifulSoup
import time
def scrape_data(base_url):
    all_data = []
    # Iterate through all available periods and sector levels down to level 3
    for year in range(2010, 2024):  # Example range of years
        for sector_level in range(1, 4):  # Assuming sector levels 1 to 3
            url = f"{base_url}/data?year={year}&sector_level={sector_level}"
            try:
                response = requests.get(url)
                response.raise_for_status()  # Raise an error for bad responses
                html_content = response.text
                data = extract_data(html_content)
                all_data.extend(data)
            except requests.RequestException as e:
                print(f"Request failed for {url}: {e}")
            except Exception as e:
                print(f"Error extracting data for {url}: {e}")
            time.sleep(1)  # Respectful scraping by adding delay
    return all_data
def extract_data(html_content):
    soup = BeautifulSoup(html_content, 'html.parser')
    data = []
    try:
        # Update the class names based on the actual HTML structure
        for category in soup.find_all(class_='actual_personalekategori_class_name'):  # Update with actual class name
            year_work = category.find(class_='actual_årsværk_class_name').text.strip()  # Update with actual class name
            category_name = category.find(class_='actual_kategori_navn_class_name').text.strip()  # Update with actual class name
            data.append([category_name, year_work])
    except AttributeError as e:
        print(f"Error extracting data: {e}")
    return data
