import sqlite3

def init_db():
    '''
    Initializes the database.
    '''
    conn = sqlite3.connect('annotations.db')
    c = conn.cursor()
    c.execute('''
        CREATE TABLE IF NOT EXISTS annotations (
            id INTEGER PRIMARY KEY,
            text TEXT,
            annotation TEXT
        )
    ''')
    conn.commit()
    conn.close()

def save_annotation(text, annotation):
    '''
    Saves the annotation to the database.
    '''
    conn = sqlite3.connect('annotations.db')
    c = conn.cursor()
    c.execute('INSERT INTO annotations (text, annotation) VALUES (?, ?)', (text, annotation))
    conn.commit()
    conn.close()

def fetch_annotations():
    '''
    Fetches all stored annotations.
    '''
    conn = sqlite3.connect('annotations.db')
    c = conn.cursor()
    c.execute('SELECT * FROM annotations')
    annotations = c.fetchall()
    conn.close()
    return [{'id': id, 'text': text, 'annotation': annotation} for id, text, annotation in annotations]

import sqlite3
def init_db():
    '''
    Initializes the database.
    '''
    conn = sqlite3.connect('annotations.db')
    c = conn.cursor()
    c.execute('''
        CREATE TABLE IF NOT EXISTS annotations (

            text TEXT,
            annotation TEXT
        )
    ''')
    conn.commit()
    conn.close()

    '''
    Saves the annotation to the database.
    '''
    conn = sqlite3.connect('annotations.db')
    c = conn.cursor()
    c.execute('INSERT INTO annotations (text, annotation) VALUES (?, ?)', (text, annotation))
    conn.commit()
    conn.close()

    '''
    Fetches all stored annotations.

    conn.close()
    return [{'id': id, 'text': text, 'annotation': annotation} for id, text, annotation in annotations]

'''

import sqlite3
def init_db():
    '''
    Initializes the database.
    '''
    conn = sqlite3.connect('annotations.db')
    c = conn.cursor()
    c.execute('''
        CREATE TABLE IF NOT EXISTS annotations (
            id INTEGER PRIMARY KEY,
            text TEXT,
            annotation TEXT
        )
    ''')
    conn.commit()
    conn.close()
def save_annotation(text, annotation):
    '''
    Saves the annotation to the database.
    '''
    conn = sqlite3.connect('annotations.db')
    c = conn.cursor()
    c.execute('INSERT INTO annotations (text, annotation) VALUES (?, ?)', (text, annotation))
    conn.commit()
    conn.close()
def fetch_annotations():
    '''
    Fetches all stored annotations.
    '''
    conn = sqlite3.connect('annotations.db')
    c = conn.cursor()
    c.execute('SELECT * FROM annotations')
    annotations = c.fetchall()
    conn.close()
    return [{'id': id, 'text': text, 'annotation': annotation} for id, text, annotation in annotations]

import sqlite3
def init_db():
    '''
    Initializes the database.
    '''
    conn = sqlite3.connect('annotations.db')

            text TEXT,
            annotation TEXT
        )
    ''')
    conn.commit()

    '''
    Saves the annotation to the database.
    '''

    '''

    annotations = c.fetchall()
    conn.close()
    return [{'id': id, 'text': text, 'annotation': annotation} for id, text, annotation in annotations]

'''
Creates a simple web interface for user interaction using Flask.
'''
from flask import Flask, request, jsonify
from model import load_model, annotate_text
from database import init_db, save_annotation, fetch_annotations
app = Flask(__name__)
@app.route('/upload', methods=['POST'])
def upload_file():
    '''
    Handles file uploads and returns annotations.
    '''
    if 'file' not in request.files:
        return jsonify({'error': 'No file uploaded'}), 400
    file = request.files['file']
    text = file.read().decode('utf-8')  # Assuming the file is in UTF-8
    annotation = annotate_text(text)
    save_annotation(text, annotation)
    return jsonify({'annotation': annotation})
@app.route('/annotations', methods=['GET'])
def get_annotations():
    '''
    Fetches all stored annotations.
    '''
    annotations = fetch_annotations()
    return jsonify(annotations)
if __name__ == '__main__':
    init_db()  # Initialize the database
    load_model()  # Load the Hugging Face model
    app.run(debug=True)

'''
Manages the persistent storage of annotations using SQLite.
'''
import sqlite3
def init_db():
    '''
    Initializes the database.
    '''
    conn = sqlite3.connect('annotations.db')
    c = conn.cursor()
    c.execute('''
        CREATE TABLE IF NOT EXISTS annotations (
            id INTEGER PRIMARY KEY,
            text TEXT,
            annotation TEXT
        )
    ''')
    conn.commit()
    conn.close()
def save_annotation(text, annotation):
    '''
    Saves the annotation to the database.
    '''
    conn = sqlite3.connect('annotations.db')
    c = conn.cursor()
    c.execute('INSERT INTO annotations (text, annotation) VALUES (?, ?)', (text, annotation))
    conn.commit()
    conn.close()
def fetch_annotations():
    '''
    Fetches all stored annotations.
    '''
    conn = sqlite3.connect('annotations.db')
    c = conn.cursor()
    c.execute('SELECT * FROM annotations')
    annotations = c.fetchall()
    conn.close()
    return [{'id': id, 'text': text, 'annotation': annotation} for id, text, annotation in annotations]

'''
Manages the persistent storage of annotations using SQLite.
'''
import sqlite3
def init_db():
    '''
    Initializes the database.
    '''
    conn = sqlite3.connect('annotations.db')
    c = conn.cursor()
    c.execute('''
        CREATE TABLE IF NOT EXISTS annotations (
            id INTEGER PRIMARY KEY,
            text TEXT,
            annotation TEXT
        )
    ''')
    conn.commit()
    conn.close()
def save_annotation(text, annotation):
    '''
    Saves the annotation to the database.
    '''
    conn = sqlite3.connect('annotations.db')
    c = conn.cursor()
    c.execute('INSERT INTO annotations (text, annotation) VALUES (?, ?)', (text, annotation))
    conn.commit()
    conn.close()
def fetch_annotations():
    '''
    Fetches all stored annotations.
    '''
    conn = sqlite3.connect('annotations.db')
    c = conn.cursor()
    c.execute('SELECT * FROM annotations')
    annotations = c.fetchall()
    conn.close()
    return [{'id': id, 'text': text, 'annotation': annotation} for id, text, annotation in annotations]
