class Order:

        self.items = []
        self.total_amount = 0.0
        self.status = "Pending"

    def add_item(self, item, price):

        self.calculate_total()

    def remove_item(self, item):
        self.items = [i for i in self.items if i[0] != item]
        self.calculate_total()

    def calculate_total(self):
        self.total_amount = sum(price for _, price in self.items)

    def update_status(self, status):

class Inventory:

    def remove_item(self, item_name):

            del self.items[item_name]

    def update_item_quantity(self, item_name, quantity):

class Order:
    def __init__(self, order_id):
        self.order_id = order_id
        self.items = []
        self.total_amount = 0.0
        self.status = "Pending"
    def add_item(self, item, price):

        self.calculate_total()
    def remove_item(self, item):
        self.items = [i for i in self.items if i[0] != item]
        self.calculate_total()
    def calculate_total(self):
        self.total_amount = sum(price for _, price in self.items)
    def update_status(self, status):
        self.status = status

class Inventory:

'''
Module for the main POS system logic.

from order import Order
from inventory import Inventory
class POSSystem:
    def __init__(self):
        self.orders = []
        self.inventory = Inventory()
    def create_order(self):
        order_id = len(self.orders) + 1
        new_order = Order(order_id)
        self.orders.append(new_order)
        return new_order
    def process_order(self, order):
        order.update_status("Completed")
    def track_inventory(self):
        return self.inventory.items

class Order:
    def __init__(self, order_id):
        self.order_id = order_id
        self.items = []
        self.total_amount = 0.0
        self.status = "Pending"
    def add_item(self, item, price):
        self.items.append((item, price))
        self.calculate_total()
    def remove_item(self, item):
        self.items = [i for i in self.items if i[0] != item]
        self.calculate_total()
    def calculate_total(self):
        self.total_amount = sum(price for _, price in self.items)
    def update_status(self, status):
        self.status = status

        self.inventory.add_item(item_name, quantity, price)

        self.orders = []
        self.inventory = Inventory()
    def create_order(self):
        order_id = len(self.orders) + 1
        new_order = Order(order_id)
        self.orders.append(new_order)
        return new_order
    def process_order(self, order):
        order.update_status("Completed")
    def track_inventory(self):
        return self.inventory.items
    def add_inventory_item(self, item_name, quantity, price):
        self.inventory.add_item(item_name, quantity, price)
    def update_inventory_item(self, item_name, quantity):
        self.inventory.update_item_quantity(item_name, quantity)
