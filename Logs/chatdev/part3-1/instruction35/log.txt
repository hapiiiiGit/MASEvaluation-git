import requests
class TelegramNotifier:
    def __init__(self, token, chat_id):
        self.token = token
        self.chat_id = chat_id
    def send_notification(self, message):
        url = f"https://api.telegram.org/bot{self.token}/sendMessage"
        payload = {

import requests
import json
from delta_adjuster import DeltaAdjuster
from telegram_notifier import TelegramNotifier
class TradingBot:
    def __init__(self, api_key, api_secret, target_delta, telegram_token, chat_id):
        self.api_key = api_key
        self.api_secret = api_secret
        self.target_delta = target_delta
        self.delta_adjuster = DeltaAdjuster()
        self.telegram_notifier = TelegramNotifier(telegram_token, chat_id)
        self.base_url = "https://www.deribit.com/api/v2/"
        self.session = requests.Session()
        self.session.headers.update({'Content-Type': 'application/json'})
    def connect_to_deribit(self):
        # Connect to Deribit API
        # Here you can implement authentication if needed
        print("Connected to Deribit API")
    def adjust_delta(self):
        current_delta = self.get_current_delta()
        adjustment = self.delta_adjuster.calculate_adjustment(current_delta, self.target_delta)
        return adjustment  # Return adjustment for use in trading
    def get_current_delta(self):
        # Logic to get current delta from Deribit
        response = self.session.get(self.base_url + "public/get_account_summary", params={"currency": "BTC"})
        if response.status_code == 200:
            data = response.json()
            # Assuming the delta is part of the account summary response
            current_delta = data['result']['delta']  # Placeholder for actual delta retrieval logic
            return current_delta
        else:
            print("Error retrieving current delta:", response.text)
            return 0  # Fallback if there's an error
    def execute_trade(self, adjustment, instrument_name, order_type, price):
        # Logic to execute trade on Deribit
        order_params = {
            "instrument_name": instrument_name,
            "amount": adjustment,
            "type": order_type,
            "price": price
        }
        response = self.session.post(self.base_url + "private/buy", json=order_params)
        if response.status_code == 200:
            self.telegram_notifier.send_notification(f"Executed trade with adjustment: {adjustment} for {instrument_name}")
        else:
            self.telegram_notifier.send_notification(f"Failed to execute trade: {response.text}")

import requests
class TelegramNotifier:
    def __init__(self, token, chat_id):
        self.token = token
        self.chat_id = chat_id
    def send_notification(self, message):
        url = f"https://api.telegram.org/bot{self.token}/sendMessage"
        payload = {
            'chat_id': self.chat_id,
            'text': message
        }
        requests.post(url, data=payload)

class TelegramNotifier:

        self.chat_id = chat_id
    def send_notification(self, message):

            'chat_id': self.chat_id,
            'text': message
        }

'''
Trading bot class for managing trades on Deribit.
'''
import requests
import json
from delta_adjuster import DeltaAdjuster
from telegram_notifier import TelegramNotifier
class TradingBot:
    def __init__(self, api_key, api_secret, target_delta, telegram_token, chat_id):
        self.api_key = api_key
        self.api_secret = api_secret
        self.target_delta = target_delta
        self.delta_adjuster = DeltaAdjuster()
        self.telegram_notifier = TelegramNotifier(telegram_token, chat_id)
        self.base_url = "https://www.deribit.com/api/v2/"
        self.session = requests.Session()
        self.session.headers.update({'Content-Type': 'application/json'})
    def connect_to_deribit(self):
        # Connect to Deribit API
        print("Connected to Deribit API")
    def adjust_delta(self):
        current_delta = self.get_current_delta()
        adjustment = self.delta_adjuster.calculate_adjustment(current_delta, self.target_delta)
        return adjustment  # Return adjustment for use in trading
    def get_current_delta(self):
        # Logic to get current delta from Deribit
        response = self.session.get(self.base_url + "public/get_account_summary", params={"currency": "BTC"})
        if response.status_code == 200:
            data = response.json()
            # Assuming the delta is part of the account summary response
            current_delta = data['result']['delta']  # Placeholder for actual delta retrieval logic
            return current_delta
        else:
            print("Error retrieving current delta:", response.text)
            return 0  # Fallback if there's an error
    def execute_trade(self, adjustment, instrument_name, order_type, price):
        if adjustment <= 0:
            self.telegram_notifier.send_notification(f"No trade executed. Adjustment is zero or negative: {adjustment}")
            return  # Exit the method if no valid trade adjustment
        order_params = {
            "instrument_name": instrument_name,
            "amount": adjustment,
            "type": order_type,
            "price": price
        }
        response = self.session.post(self.base_url + "private/buy", json=order_params)
        if response.status_code == 200:
            self.telegram_notifier.send_notification(f"Executed trade with adjustment: {adjustment} for {instrument_name}")
        else:
            self.telegram_notifier.send_notification(f"Failed to execute trade: {response.text}")

import requests
import json
from delta_adjuster import DeltaAdjuster

class TradingBot:
    def __init__(self, api_key, api_secret, target_delta, telegram_token, chat_id):
        self.api_key = api_key
        self.api_secret = api_secret
        self.target_delta = target_delta
        self.delta_adjuster = DeltaAdjuster()
        self.telegram_notifier = TelegramNotifier(telegram_token, chat_id)
        self.base_url = "https://www.deribit.com/api/v2/"
        self.session = requests.Session()
        self.session.headers.update({'Content-Type': 'application/json'})
    def connect_to_deribit(self):
        # Connect to Deribit API
        print("Connected to Deribit API")
    def adjust_delta(self):
        current_delta = self.get_current_delta()
        adjustment = self.delta_adjuster.calculate_adjustment(current_delta, self.target_delta)
        return adjustment  # Return adjustment for use in trading
    def get_current_delta(self):
        # Logic to get current delta from Deribit
        response = self.session.get(self.base_url + "public/get_account_summary", params={"currency": "BTC"})
        if response.status_code == 200:
            data = response.json()
            # Assuming the delta is part of the account summary response
            current_delta = data['result']['delta']  # Placeholder for actual delta retrieval logic
            return current_delta
        else:
            print("Error retrieving current delta:", response.text)
            return 0  # Fallback if there's an error

        if adjustment <= 0:

            "instrument_name": instrument_name,
            "amount": adjustment,
            "type": order_type,

        if response.status_code == 200:
            self.telegram_notifier.send_notification(f"Executed trade with adjustment: {adjustment} for {instrument_name}")

            self.telegram_notifier.send_notification(f"Failed to execute trade: {response.text}")
