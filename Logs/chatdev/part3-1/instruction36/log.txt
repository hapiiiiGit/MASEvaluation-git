'''
Contains the WooCommerceScraper class to scrape product data from WooCommerce websites.
'''

import requests
from bs4 import BeautifulSoup
import pandas as pd

class WooCommerceScraper:
    def __init__(self, url):
        self.url = url
        self.product_data = []

    def fetch_product_data(self):
        response = requests.get(self.url)
        soup = BeautifulSoup(response.content, 'html.parser')
        products = soup.find_all(class_='product')

        for product in products:
            title = product.find(class_='woocommerce-loop-product__title').text
            price = product.find(class_='price').text
            self.product_data.append({'Title': title, 'Price': price})

    def generate_csv(self, data):
        df = pd.DataFrame(data)
        df.to_csv('products.csv', index=False)

    def generate_excel(self, data):
        df = pd.DataFrame(data)
        df.to_excel('products.xlsx', index=False)
```

import requests
from bs4 import BeautifulSoup
import pandas as pd
class WooCommerceScraper:
    def __init__(self, url):
        self.url = url
        self.product_data = []
    def fetch_product_data(self):
        response = requests.get(self.url)
        soup = BeautifulSoup(response.content, 'html.parser')
        products = soup.find_all(class_='product')
        for product in products:
            title = product.find(class_='woocommerce-loop-product__title').text
            price = product.find(class_='price').text
            self.product_data.append({'Title': title, 'Price': price})
    def generate_csv(self, data):
        df = pd.DataFrame(data)
        df.to_csv('products.csv', index=False)
    def generate_excel(self, data):
        df = pd.DataFrame(data)
        df.to_excel('products.xlsx', index=False)

        df.to_csv('products.csv', index=False)

    def generate_excel(self, data):

        df.to_excel('products.xlsx', index=False)

'''
Contains the WooCommerceScraper class to scrape product data from WooCommerce websites.
'''
import requests
from bs4 import BeautifulSoup
import pandas as pd
class WooCommerceScraper:
    def __init__(self, url):
        self.url = url
        self.product_data = []
    def fetch_product_data(self):
        response = requests.get(self.url)
        soup = BeautifulSoup(response.content, 'html.parser')
        products = soup.find_all(class_='product')
        for product in products:
            title = product.find(class_='woocommerce-loop-product__title').text
            price = product.find(class_='price').text
            self.product_data.append({'Title': title, 'Price': price})
    def generate_csv(self, data):
        df = pd.DataFrame(data)
        df.to_csv('products.csv', index=False)
    def generate_excel(self, data):
        df = pd.DataFrame(data)
        df.to_excel('products.xlsx', index=False)

'''

import requests
from bs4 import BeautifulSoup
import pandas as pd

class WooCommerceScraper:
    def __init__(self, url):
        self.url = url
        self.product_data = []

    def fetch_product_data(self):
        response = requests.get(self.url)
        if response.status_code == 200:
            soup = BeautifulSoup(response.content, 'html.parser')
            products = soup.find_all(class_='product')
            for product in products:
                title = product.find(class_='woocommerce-loop-product__title').text.strip()
                price = product.find(class_='price').text.strip()
                self.product_data.append({'Title': title, 'Price': price})
        else:
            raise Exception("Failed to fetch data from the URL provided.")

    def generate_csv(self, data):
        df = pd.DataFrame(data)
        df.to_csv('products.csv', index=False)

    def generate_excel(self, data):
        df = pd.DataFrame(data)
        df.to_excel('products.xlsx', index=False)

'''
Contains the WooCommerceScraper class to scrape product data from WooCommerce websites.
'''
import requests
from bs4 import BeautifulSoup
import pandas as pd
class WooCommerceScraper:
    def __init__(self, url):
        self.url = url
        self.product_data = []
    def fetch_product_data(self):
        response = requests.get(self.url)
        if response.status_code == 200:
            soup = BeautifulSoup(response.content, 'html.parser')
            products = soup.find_all(class_='product')
            for product in products:
                title = product.find(class_='woocommerce-loop-product__title').text.strip()
                price = product.find(class_='price').text.strip()
                self.product_data.append({'Title': title, 'Price': price})
        else:
            raise Exception("Failed to fetch data from the URL provided.")
    def generate_csv(self, data):
        df = pd.DataFrame(data)
        df.to_csv('products.csv', index=False)
    def generate_excel(self, data):
        df = pd.DataFrame(data)
        df.to_excel('products.xlsx', index=False)

'''
Contains the WooCommerceScraper class to scrape product data from WooCommerce websites.
'''
import requests
from bs4 import BeautifulSoup
import pandas as pd
class WooCommerceScraper:
    def __init__(self, url):
        self.url = url
        self.product_data = []
    def fetch_product_data(self):
        response = requests.get(self.url)
        if response.status_code == 200:
            soup = BeautifulSoup(response.content, 'html.parser')
            products = soup.find_all(class_='product')
            for product in products:
                title = product.find(class_='woocommerce-loop-product__title').text.strip()
                price = product.find(class_='price').text.strip()
                self.product_data.append({'Title': title, 'Price': price})
        else:
            raise Exception("Failed to fetch data from the URL provided.")
    def generate_csv(self, data):
        df = pd.DataFrame(data)
        df.to_csv('products.csv', index=False)
    def generate_excel(self, data):
        df = pd.DataFrame(data)
        df.to_excel('products.xlsx', index=False)

import tkinter as tk
from scraper import WooCommerceScraper
from chrome_extension import ChromeExtension
class GUI:
    def __init__(self, master):
        self.master = master
        master.title("WooCommerce Scraper")
        self.label = tk.Label(master, text="Enter WooCommerce URL:")
        self.label.pack()
        self.url_entry = tk.Entry(master)
        self.url_entry.pack()
        self.scrape_button = tk.Button(master, text="Scrape Products", command=self.start_scraping)
        self.scrape_button.pack()
        self.setup_extension_button = tk.Button(master, text="Setup Chrome Extension", command=self.setup_chrome_extension)
        self.setup_extension_button.pack()
        self.trigger_scraping_button = tk.Button(master, text="Trigger Scraping from Extension", command=self.trigger_scraping)
        self.trigger_scraping_button.pack()
        self.status_label = tk.Label(master, text="")
        self.status_label.pack()
        self.chrome_extension = ChromeExtension("http://localhost:5000/scrape")  # Example URL for triggering scraping
    def start_scraping(self):
        url = self.url_entry.get()
        scraper = WooCommerceScraper(url)
        try:
            scraper.fetch_product_data()
            scraper.generate_csv(scraper.product_data)
            scraper.generate_excel(scraper.product_data)
            self.update_status("Scraping completed! Files generated.")
        except Exception as e:
            self.update_status(f"Error: {str(e)}")
    def setup_chrome_extension(self):
        self.chrome_extension.setup_extension()
        self.update_status("Chrome Extension setup complete.")
    def trigger_scraping(self):
        result = self.chrome_extension.trigger_scraping()
        if "error" in result:
            self.update_status(result["error"])
        else:
            self.update_status("Scraping triggered from Chrome Extension.")
