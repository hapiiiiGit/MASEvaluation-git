'''
Main application file for the HotFinder program.
'''

import tkinter as tk
from tkinter import messagebox
from youtube_api import YouTubeAPI
from utils import write_to_json

class HotFinderApp:
    def __init__(self, master):
        self.master = master
        master.title("HotFinder - YouTube Video Analyzer")

        self.label = tk.Label(master, text="Enter Search Term or Channel:")
        self.label.pack()

        self.entry = tk.Entry(master)
        self.entry.pack()

        self.search_button = tk.Button(master, text="Search", command=self.search_videos)
        self.search_button.pack()

        self.results_text = tk.Text(master)
        self.results_text.pack()

    def search_videos(self):
        query = self.entry.get()
        if not query:
            messagebox.showwarning("Input Error", "Please enter a search term or channel.")
            return

        api_key = "YOUR_YOUTUBE_API_KEY"  # Replace with your actual API key
        youtube_api = YouTubeAPI(api_key)
        videos = youtube_api.search(query)

        if videos:
            filtered_videos = youtube_api.filter_videos(videos)
            write_to_json("results.json", filtered_videos)
            self.results_text.delete(1.0, tk.END)
            self.results_text.insert(tk.END, f"Results saved to results.json\n{filtered_videos}")
        else:
            messagebox.showinfo("No Results", "No videos found for the given query.")

if __name__ == "__main__":
    root = tk.Tk()
    app = HotFinderApp(root)
    root.mainloop()
```

```python
'''
YouTube API handler for searching and filtering videos.
'''

import requests
from datetime import datetime, timedelta

class YouTubeAPI:
    def __init__(self, api_key):
        self.api_key = api_key
        self.base_url = "https://www.googleapis.com/youtube/v3/search"

    def search(self, query):
        params = {
            'part': 'snippet',
            'q': query,
            'type': 'video',
            'publishedAfter': (datetime.now() - timedelta(days=10)).isoformat() + 'Z',
            'key': self.api_key
        }
        response = requests.get(self.base_url, params=params)
        if response.status_code == 200:
            return response.json().get('items', [])
        return []

    def filter_videos(self, videos):
        # Example filter: only return videos with more than 100 views (this would require additional API calls)
        filtered_videos = []
        for video in videos:
            # Here you would normally check the view count, but for simplicity, we just return the video data
            filtered_videos.append(video)
        return filtered_videos
```

'''
YouTube API handler for searching and filtering videos.
'''
import requests
from datetime import datetime, timedelta
class YouTubeAPI:
    def __init__(self, api_key):
        self.api_key = api_key
        self.base_url = "https://www.googleapis.com/youtube/v3/search"
    def search(self, query):
        params = {
            'part': 'snippet',
            'q': query,
            'type': 'video',
            'publishedAfter': (datetime.now() - timedelta(days=10)).isoformat() + 'Z',
            'key': self.api_key
        }
        response = requests.get(self.base_url, params=params)
        if response.status_code == 200:
            return response.json().get('items', [])
        return []
    def filter_videos(self, videos):
        '''
        Filters videos based on specified criteria.
        Currently, it returns all videos without additional filtering.
        '''
        filtered_videos = []
        for video in videos:
            # Here you would normally check the view count, but for simplicity, we just return the video data
            filtered_videos.append(video)
        return filtered_videos

'''
YouTube API handler for searching and filtering videos.

import requests
from datetime import datetime, timedelta
class YouTubeAPI:
    def __init__(self, api_key):
        self.api_key = api_key
        self.base_url = "https://www.googleapis.com/youtube/v3/search"
    def search(self, query):
        params = {
            'part': 'snippet',
            'q': query,
            'type': 'video',
            'publishedAfter': (datetime.now() - timedelta(days=10)).isoformat() + 'Z',
            'key': self.api_key
        }
        response = requests.get(self.base_url, params=params)
        if response.status_code == 200:
            return response.json().get('items', [])
        return []
    def filter_videos(self, videos):
        '''
        Filters videos based on specified criteria.
        Currently, it returns all videos without additional filtering.
        '''
        filtered_videos = []
        for video in videos:
            # Here you would normally check the view count, but for simplicity, we just return the video data
            filtered_videos.append(video)
        return filtered_videos

import requests
from datetime import datetime, timedelta
class YouTubeAPI:
    def __init__(self, api_key):
        self.api_key = api_key
        self.base_url = "https://www.googleapis.com/youtube/v3/search"
    def search(self, query):
        params = {
            'part': 'snippet',
            'q': query,
            'type': 'video',
            'publishedAfter': (datetime.now() - timedelta(days=10)).isoformat() + 'Z',
            'key': self.api_key
        }
        response = requests.get(self.base_url, params=params)
        if response.status_code == 200:
            return response.json().get('items', [])
        return []
    def filter_videos(self, videos):
        '''
        Filters videos based on specified criteria.
        Currently, it returns all videos without additional filtering.
        '''
        filtered_videos = []
        for video in videos:
            # Here you would normally check the view count, but for simplicity, we just return the video data
            filtered_videos.append(video)
        return filtered_videos

import tkinter as tk

from youtube_api import YouTubeAPI

'''
YouTube API handler for searching and filtering videos.
'''
import requests
from datetime import datetime, timedelta
class YouTubeAPI:
    def __init__(self, api_key):
        self.api_key = api_key
        self.base_url = "https://www.googleapis.com/youtube/v3/search"
    def search(self, query):
        params = {
            'part': 'snippet',
            'q': query,
            'type': 'video',
            'publishedAfter': (datetime.now() - timedelta(days=10)).isoformat() + 'Z',
            'key': self.api_key
        }
        response = requests.get(self.base_url, params=params)
        if response.status_code == 200:
            return response.json().get('items', [])
        return []
    def filter_videos(self, videos):
        '''
        Filters videos based on specified criteria.
        Currently, it returns all videos without additional filtering.
        '''
        filtered_videos = []
        for video in videos:
            # Here you would normally check the view count, but for simplicity, we just return the video data
            filtered_videos.append(video)
        return filtered_videos

'''

import tkinter as tk
from tkinter import messagebox
from youtube_api import YouTubeAPI
from utils import write_to_json
from config import YOUTUBE_API_KEY  # Importing the API key from config
class HotFinderApp:
    def __init__(self, master):
        self.master = master
        master.title("HotFinder - YouTube Video Analyzer")
        self.label = tk.Label(master, text="Enter Search Term or Channel:")
        self.label.pack()

        self.entry.pack()
        self.search_button = tk.Button(master, text="Search", command=self.search_videos)
        self.search_button.pack()
        self.results_text = tk.Text(master)
        self.results_text.pack()
    def search_videos(self):

'''
YouTube API handler for searching and filtering videos.
'''
import requests
from datetime import datetime, timedelta
class YouTubeAPI:
    def __init__(self, api_key):
        self.api_key = api_key
        self.base_url = "https://www.googleapis.com/youtube/v3/search"
    def search(self, query):
        params = {
            'part': 'snippet',
            'q': query,
            'type': 'video',
            'publishedAfter': (datetime.now() - timedelta(days=10)).isoformat() + 'Z',
            'key': self.api_key
        }
        response = requests.get(self.base_url, params=params)
        if response.status_code == 200:
            return response.json().get('items', [])
        return []
    def filter_videos(self, videos):
        '''
        Filters videos based on specified criteria.
        Currently, it returns all videos without additional filtering.
        '''
        filtered_videos = []
        for video in videos:
            # Here you would normally check the view count, but for simplicity, we just return the video data
            filtered_videos.append(video)
        return filtered_videos
