'''
Main application file for the Stock Analyzer utility.
'''

import tkinter as tk
from tkinter import messagebox
from data_fetcher import fetch_data
from data_processor import calculate_daily_returns, calculate_average_return, calculate_standard_deviation, calculate_momentum, rank_stocks
from exporter import export_to_excel

class StockAnalyzerApp:
    def __init__(self, master):
        self.master = master
        master.title("Stock Analyzer")

        self.label = tk.Label(master, text="Enter Index Basket and Time Window:")
        self.label.pack()

        self.index_entry = tk.Entry(master)
        self.index_entry.pack()
        self.index_entry.insert(0, "NSE Index Basket")

        self.time_window_entry = tk.Entry(master)
        self.time_window_entry.pack()
        self.time_window_entry.insert(0, "Time Window (e.g., 30 days)")

        self.fetch_button = tk.Button(master, text="Fetch Data", command=self.fetch_and_process_data)
        self.fetch_button.pack()

    def fetch_and_process_data(self):
        index_basket = self.index_entry.get()
        time_window = int(self.time_window_entry.get())

        try:
            data = fetch_data(index_basket, time_window)
            daily_returns = calculate_daily_returns(data)
            avg_return = calculate_average_return(daily_returns)
            std_dev = calculate_standard_deviation(daily_returns)
            momentum = calculate_momentum(data)
            ranked_stocks = rank_stocks(data, daily_returns, avg_return, std_dev, momentum)

            export_to_excel(data, ranked_stocks)
            messagebox.showinfo("Success", "Data fetched and exported successfully!")
        except Exception as e:
            messagebox.showerror("Error", str(e))

if __name__ == "__main__":
    root = tk.Tk()
    app = StockAnalyzerApp(root)
    root.mainloop()
```

```python
'''
Module for processing stock data.
'''

import pandas as pd
import numpy as np

def calculate_daily_returns(data):
    return data.pct_change().dropna()

def calculate_average_return(returns):
    return returns.mean()

def calculate_standard_deviation(returns):
    return returns.std()

def calculate_momentum(data):
    return data.iloc[-1] - data.iloc[0]

def rank_stocks(data, daily_returns, avg_return, std_dev, momentum):
    metrics = pd.DataFrame({
        'Average Return': avg_return,
        'Standard Deviation': std_dev,
        'Momentum': momentum
    })
    return metrics.rank(ascending=False)
```

        ranked_stocks.to_excel(writer, sheet_name='Ranked Stocks')

'''

import pandas as pd
import numpy as np

def calculate_daily_returns(data):
    return data.pct_change().dropna()

def calculate_average_return(returns):
    return returns.mean()

def calculate_standard_deviation(returns):
    return returns.std()

def calculate_momentum(data):
    return data.iloc[-1] - data.iloc[0]

    metrics = pd.DataFrame({

        'Standard Deviation': std_dev,
        'Momentum': momentum
    })
    return metrics.rank(ascending=False)

'''
Module for processing stock data.
'''
import pandas as pd
import numpy as np
def calculate_daily_returns(data):
    return data.pct_change().dropna()
def calculate_average_return(returns):
    return returns.mean()
def calculate_standard_deviation(returns):
    return returns.std()
def calculate_momentum(data):
    return data.iloc[-1] - data.iloc[0]
def rank_stocks(data, daily_returns):
    # Calculate metrics for each stock
    avg_return = daily_returns.mean()
    std_dev = daily_returns.std()
    momentum = data.iloc[-1] - data.iloc[0]
    # Create a DataFrame to hold the metrics
    metrics = pd.DataFrame({
        'Average Return': avg_return,
        'Standard Deviation': std_dev,
        'Momentum': momentum
    })
    # Rank the stocks based on the metrics
    ranked_stocks = metrics.rank(ascending=False)
    return ranked_stocks

'''
Module for processing stock data.
'''
import pandas as pd
import numpy as np
def calculate_daily_returns(data):
    return data.pct_change().dropna()
def calculate_average_return(returns):
    return returns.mean()
def calculate_standard_deviation(returns):
    return returns.std()
def calculate_momentum(data):
    return data.iloc[-1] - data.iloc[0]
def rank_stocks(data, daily_returns):
    # Calculate metrics for each stock
    avg_return = daily_returns.mean()
    std_dev = daily_returns.std()
    momentum = data.iloc[-1] - data.iloc[0]
    # Create a DataFrame to hold the metrics
    metrics = pd.DataFrame({
        'Average Return': avg_return,
        'Standard Deviation': std_dev,
        'Momentum': momentum
    })
    # Rank the stocks based on the metrics
    ranked_stocks = metrics.rank(ascending=False)
    return ranked_stocks

'''
Module for processing stock data.
'''
import pandas as pd
import numpy as np
def calculate_daily_returns(data):
    return data.pct_change().dropna()
def calculate_average_return(returns):
    return returns.mean()
def calculate_standard_deviation(returns):
    return returns.std()
def calculate_momentum(data):
    return data.iloc[-1] - data.iloc[0]
def rank_stocks(data, daily_returns):
    # Calculate metrics for each stock
    avg_return = daily_returns.mean()
    std_dev = daily_returns.std()
    momentum = data.iloc[-1] - data.iloc[0]
    # Create a DataFrame to hold the metrics
    metrics = pd.DataFrame({
        'Average Return': avg_return,
        'Standard Deviation': std_dev,
        'Momentum': momentum
    })
    # Rank the stocks based on the metrics
    ranked_stocks = metrics.rank(ascending=False)
    return ranked_stocks
