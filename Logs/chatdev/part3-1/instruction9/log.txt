'''
Main entry point for the Heart Attack Risk Prediction application.
'''

import tkinter as tk
from tkinter import filedialog, messagebox
from image_processor import ImageProcessor
from model import HeartAttackModel
from heatmap import generate_heatmap

class HeartAttackRiskApp:
    def __init__(self, master):
        self.master = master
        self.master.title("Heart Attack Risk Prediction")
        
        self.image_processor = ImageProcessor()
        self.model = HeartAttackModel()
        
        self.label = tk.Label(master, text="Upload Retina-Fundus Image:")
        self.label.pack()
        
        self.upload_button = tk.Button(master, text="Upload Image", command=self.load_image)
        self.upload_button.pack()
        
        self.predict_button = tk.Button(master, text="Predict Risk", command=self.predict_risk)
        self.predict_button.pack()
        
        self.result_label = tk.Label(master, text="")
        self.result_label.pack()
        
    def load_image(self):
        file_path = filedialog.askopenfilename()
        if file_path:
            self.image_processor.load_image(file_path)
            messagebox.showinfo("Image Loaded", "Image loaded successfully!")
    
    def predict_risk(self):
        if self.image_processor.image is None:
            messagebox.showerror("Error", "No image loaded!")
            return
        
        risk_score, confidence, heatmap = self.model.predict(self.image_processor.image)
        self.result_label.config(text=f"Risk Score: {risk_score}, Confidence: {confidence}")
        generate_heatmap(self.image_processor.image, heatmap)

if __name__ == "__main__":
    root = tk.Tk()
    app = HeartAttackRiskApp(root)
    root.mainloop()
```
