import tkinter as tk
from tkinter import messagebox
from user import User
from product import Product
from order import Order
from notification import NotificationManager
from nlp import NLPProcessor

class OrderingSystemGUI:
    def __init__(self):
        self.root = tk.Tk()
        self.root.title("AI-Powered Ordering System")
        self.notification_manager = NotificationManager()
        self.nlp_processor = NLPProcessor()
        self.users = {}
        self.products = [Product(1, "Pizza", 10.99), Product(2, "Burger", 5.99)]
        self.orders = []

        self.create_widgets()

    def create_widgets(self):
        self.label = tk.Label(self.root, text="Enter your order:")
        self.label.pack()

        self.entry = tk.Entry(self.root)
        self.entry.pack()

        self.order_button = tk.Button(self.root, text="Place Order", command=self.place_order)
        self.order_button.pack()

    def place_order(self):
        input_text = self.entry.get()
        product_name = self.nlp_processor.interpret_order(input_text)[0]
        product = next((p for p in self.products if p.name.lower() == product_name.lower()), None)

        if product:
            order = Order(len(self.orders) + 1, User(1, "Customer", "customer"), product)
            self.orders.append(order)
            self.notification_manager.notify(order.user, f"Your order for {product.get_details()} has been placed.")
            messagebox.showinfo("Success", f"Order placed for {product.get_details()}.")
        else:
            messagebox.showerror("Error", "Product not found.")

    def run(self):
        self.root.mainloop()
```

        self.root.title("AI-Powered Ordering System")
        self.notification_manager = NotificationManager()
        self.nlp_processor = NLPProcessor()
        self.users = {}
        self.products = [Product(1, "Pizza", 10.99), Product(2, "Burger", 5.99)]
        self.orders = []
        self.create_widgets()
    def create_widgets(self):
        self.label = tk.Label(self.root, text="Enter your order:")
        self.label.pack()
        self.entry = tk.Entry(self.root)

        self.order_button = tk.Button(self.root, text="Place Order", command=self.place_order)
        self.order_button.pack()
    def place_order(self):
        input_text = self.entry.get()
        product_name = self.nlp_processor.interpret_order(input_text)[0]
        product = next((p for p in self.products if p.name.lower() == product_name.lower()), None)
        if product:
            order = Order(len(self.orders) + 1, User(1, "Customer", "customer"), product)
            self.orders.append(order)
            self.notification_manager.notify(order.user, f"Your order for {product.get_details()} has been placed.")
            messagebox.showinfo("Success", f"Order placed for {product.get_details()}.")
        else:
            messagebox.showerror("Error", "Product not found.")
    def run(self):
        self.root.mainloop()

                messagebox.showerror("Error", "Product not found.")
        else:

'''
Module to create the graphical user interface for the application.
'''
import tkinter as tk
from tkinter import messagebox
from user import User
from product import Product
from order import Order
from notification import NotificationManager
from nlp import NLPProcessor
class OrderingSystemGUI:
    def __init__(self):
        self.root = tk.Tk()
        self.root.title("AI-Powered Ordering System")
        self.notification_manager = NotificationManager()
        self.nlp_processor = NLPProcessor()
        self.users = {1: User(1, "Customer", "customer")}  # Example user
        self.products = [Product(1, "Pizza", 10.99), Product(2, "Burger", 5.99)]
        self.orders = []
        self.create_widgets()
    def create_widgets(self):
        self.label = tk.Label(self.root, text="Enter your order:")
        self.label.pack()
        self.entry = tk.Entry(self.root)
        self.entry.pack()
        self.order_button = tk.Button(self.root, text="Place Order", command=self.place_order)
        self.order_button.pack()
    def place_order(self):
        input_text = self.entry.get()
        tokens = self.nlp_processor.interpret_order(input_text)
        if tokens:
            product_name = tokens[0]  # Assuming the first token is the product name
            product = next((p for p in self.products if p.name.lower() == product_name.lower()), None)
            if product:
                user = self.users[1]  # Using the first user for simplicity
                order = Order(len(self.orders) + 1, user, product)
                self.orders.append(order)
                self.notification_manager.notify(user, f"Your order for {product.get_details()} has been placed.")
                messagebox.showinfo("Success", f"Order placed for {product.get_details()}.")
            else:
                messagebox.showerror("Error", "Product not found.")
        else:
            messagebox.showerror("Error", "Invalid input. Please try again.")
    def run(self):
        self.root.mainloop()

'''
Module to create the graphical user interface for the application.
'''
import tkinter as tk
from tkinter import messagebox
from user import User
from product import Product
from order import Order
from notification import NotificationManager
from nlp import NLPProcessor
class OrderingSystemGUI:
    def __init__(self):
        self.root = tk.Tk()
        self.root.title("AI-Powered Ordering System")
        self.notification_manager = NotificationManager()
        self.nlp_processor = NLPProcessor()
        self.users = {1: User(1, "Customer", "customer")}  # Example user
        self.products = [Product(1, "Pizza", 10.99), Product(2, "Burger", 5.99)]
        self.orders = []
        self.create_widgets()
    def create_widgets(self):
        self.label = tk.Label(self.root, text="Enter your order:")
        self.label.pack()
        self.entry = tk.Entry(self.root)
        self.entry.pack()
        self.order_button = tk.Button(self.root, text="Place Order", command=self.place_order)
        self.order_button.pack()
    def place_order(self):
        input_text = self.entry.get()
        tokens = self.nlp_processor.interpret_order(input_text)
        if tokens:
            product_name = tokens[0]  # Assuming the first token is the product name
            product = next((p for p in self.products if p.name.lower() == product_name.lower()), None)
            if product:
                user = self.users[1]  # Using the first user for simplicity
                order = Order(len(self.orders) + 1, user, product)
                self.orders.append(order)
                self.notification_manager.notify(user, f"Your order for {product.get_details()} has been placed.")
                messagebox.showinfo("Success", f"Order placed for {product.get_details()}.")
            else:
                messagebox.showerror("Error", "Product not found.")
        else:
            messagebox.showerror("Error", "Invalid input. Please try again.")
    def run(self):
        self.root.mainloop()

'''
import nltk  # Ensure you have nltk installed
from nltk.tokenize import word_tokenize
class NLPProcessor:
    def __init__(self):
        nltk.download('punkt')  # Download necessary resources
    def interpret_order(self, input_text):
        # Tokenize the input text to extract keywords
        tokens = word_tokenize(input_text.lower())
        return tokens

'''
Module to create the graphical user interface for the application.
'''
import tkinter as tk
from tkinter import messagebox
from user import User
from product import Product
from order import Order
from notification import NotificationManager
from nlp import NLPProcessor
class OrderingSystemGUI:
    def __init__(self):
        self.root = tk.Tk()
        self.root.title("AI-Powered Ordering System")
        self.notification_manager = NotificationManager()
        self.nlp_processor = NLPProcessor()
        self.users = {1: User(1, "Customer", "customer")}  # Example user
        self.products = [Product(1, "Pizza", 10.99), Product(2, "Burger", 5.99)]
        self.orders = []
        self.create_widgets()
    def create_widgets(self):
        self.label = tk.Label(self.root, text="Enter your order:")
        self.label.pack()
        self.entry = tk.Entry(self.root)
        self.entry.pack()
        self.order_button = tk.Button(self.root, text="Place Order", command=self.place_order)
        self.order_button.pack()
    def place_order(self):
        input_text = self.entry.get()
        tokens = self.nlp_processor.interpret_order(input_text)
        if tokens:
            product_name = tokens[0]  # Assuming the first token is the product name
            product = next((p for p in self.products if p.name.lower() == product_name.lower()), None)
            if product:
                user = self.users[1]  # Using the first user for simplicity
                order = Order(len(self.orders) + 1, user, product)
                self.orders.append(order)
                self.notification_manager.notify(user, f"Your order for {product.get_details()} has been placed.")
                messagebox.showinfo("Success", f"Order placed for {product.get_details()}.")
            else:
                messagebox.showerror("Error", "Product not found.")
        else:
            messagebox.showerror("Error", "Invalid input. Please try again.")
    def run(self):
        self.root.mainloop()
