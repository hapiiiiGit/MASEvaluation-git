Main application file for the web scraping tool.

import tkinter as tk
from tkinter import scrolledtext
from scraper import WebScraper
from data_processor import DataProcessor

class WebScraperApp:
    def __init__(self, root):
        self.root = root
        self.root.title("Web Scraper Application")
        
        self.url_label = tk.Label(root, text="Enter URL:")
        self.url_label.pack()
        
        self.url_entry = tk.Entry(root, width=50)
        self.url_entry.pack()
        
        self.scrape_button = tk.Button(root, text="Scrape Data", command=self.run_scraper)
        self.scrape_button.pack()
        
        self.results_area = scrolledtext.ScrolledText(root, width=60, height=20)
        self.results_area.pack()
        
    def run_scraper(self):
        url = self.url_entry.get()
        scraper = WebScraper(url)
        data = scraper.scrape()
        processor = DataProcessor(data)
        processed_data = processor.process_data()
        self.display_results(processed_data)

'''
Module for web scraping using Scrapy and BeautifulSoup.
'''

import requests
from bs4 import BeautifulSoup

class WebScraper:
    def __init__(self, url):
        self.url = url

    def scrape(self):
        response = requests.get(self.url)
        if response.status_code == 200:
            soup = BeautifulSoup(response.text, 'html.parser')
            data = self.extract_data(soup)
            return data
        else:
            raise Exception("Failed to retrieve data from the URL.")

    def extract_data(self, soup):
        # Example: Extracting all paragraph texts
        paragraphs = soup.find_all('p')
        return [para.get_text() for para in paragraphs]
```

        df.dropna(inplace=True)

import requests
from bs4 import BeautifulSoup
class WebScraper:
    def __init__(self, url):
        self.url = url
    def scrape(self):
        response = requests.get(self.url)
        if response.status_code == 200:
            soup = BeautifulSoup(response.text, 'html.parser')
            data = self.extract_data(soup)
            return data
        else:
            raise Exception("Failed to retrieve data from the URL.")
    def extract_data(self, soup):
        # Example: Extracting all paragraph texts
        paragraphs = soup.find_all('p')
        return [para.get_text() for para in paragraphs if para.get_text().strip() != '']

'''
Module for web scraping using Scrapy and BeautifulSoup.
'''
import requests
from bs4 import BeautifulSoup
class WebScraper:
    def __init__(self, url):
        self.url = url
    def scrape(self):
        response = requests.get(self.url)
        if response.status_code == 200:
            soup = BeautifulSoup(response.text, 'html.parser')
            data = self.extract_data(soup)
            return data
        else:
            raise Exception("Failed to retrieve data from the URL.")
    def extract_data(self, soup):
        # Example: Extracting all paragraph texts
        paragraphs = soup.find_all('p')
        return [para.get_text() for para in paragraphs if para.get_text().strip() != '']

'''

'''
import requests
from bs4 import BeautifulSoup

class WebScraper:
    def __init__(self, url, mode):
        self.url = url
        self.mode = mode

    def scrape(self):
        try:
            response = requests.get(self.url)
            response.raise_for_status()  # Raises an HTTPError for bad responses
            
            if self.mode == "api" and response.headers.get('Content-Type') == 'application/json':
                return response.json()  # Return JSON data directly for API
            else:
                soup = BeautifulSoup(response.text, 'html.parser')
                data = self.extract_data(soup)
                return data
        except requests.exceptions.RequestException as e:
            raise Exception(f"Network error: {str(e)}")

    def extract_data(self, soup):
        # Example: Extracting all paragraph texts
        paragraphs = soup.find_all('p')
        return [para.get_text() for para in paragraphs if para.get_text().strip() != '']

'''
Module for web scraping using Scrapy and BeautifulSoup.
'''
import requests
from bs4 import BeautifulSoup
class WebScraper:
    def __init__(self, url, mode):
        self.url = url
        self.mode = mode
    def scrape(self):
        try:
            response = requests.get(self.url)
            response.raise_for_status()  # Raises an HTTPError for bad responses
            if self.mode == "api" and response.headers.get('Content-Type') == 'application/json':
                return response.json()  # Return JSON data directly for API
            else:
                soup = BeautifulSoup(response.text, 'html.parser')
                data = self.extract_data(soup)
                return data
        except requests.exceptions.RequestException as e:
            raise Exception(f"Network error: {str(e)}")
    def extract_data(self, soup):
        # Example: Extracting all paragraph texts
        paragraphs = soup.find_all('p')
        return [para.get_text() for para in paragraphs if para.get_text().strip() != '']

'''
Module for web scraping using Scrapy and BeautifulSoup.
'''
import requests
from bs4 import BeautifulSoup
class WebScraper:
    def __init__(self, url, mode):
        self.url = url
        self.mode = mode
    def scrape(self):
        try:
            response = requests.get(self.url)
            response.raise_for_status()  # Raises an HTTPError for bad responses
            if self.mode == "api" and response.headers.get('Content-Type') == 'application/json':
                return response.json()  # Return JSON data directly for API
            else:
                soup = BeautifulSoup(response.text, 'html.parser')
                data = self.extract_data(soup)
                return data
        except requests.exceptions.RequestException as e:
            raise Exception(f"Network error: {str(e)}")
    def extract_data(self, soup):
        # Example: Extracting all paragraph texts
        paragraphs = soup.find_all('p')
        return [para.get_text() for para in paragraphs if para.get_text().strip() != '']
