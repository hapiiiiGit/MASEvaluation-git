from candidate import Candidate
class SourcingPlan:
    def source_candidates(self):
        # Logic to source candidates based on specified criteria
        candidates = [
            Candidate("Alice", ["JavaScript", "Python", "Java"], 5, "alice_cv.pdf"),
            Candidate("Bob", ["JavaScript", "Java"], 3, "bob_cv.pdf"),
            Candidate("Charlie", ["Python", "Java"], 4, "charlie_cv.pdf"),
        ]
        print("Candidates sourced:")
        for candidate in candidates:
            print(candidate.name)
        return candidates
    def screen_candidates(self, candidates):
        # Logic to screen candidates based on skills and experience
        screened_candidates = [c for c in candidates if any(skill in c.skills for skill in ["JavaScript", "Python", "Java"]) and c.experience >= 3]
        print("Screened candidates:")
        for candidate in screened_candidates:
            print(candidate.name)
        return screened_candidates

        # Logic to present candidate profiles with CVs and interview notes
