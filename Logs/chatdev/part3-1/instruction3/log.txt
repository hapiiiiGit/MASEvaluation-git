'''
Image processing module for detecting cars and license plates.
'''

import cv2
import numpy as np
import pytesseract
from PIL import Image

class ImageProcessor:
    def __init__(self):
        self.car_cascade = cv2.CascadeClassifier(cv2.data.haarcascades + 'haarcascade_car.xml')
        self.plate_cascade = cv2.CascadeClassifier(cv2.data.haarcascades + 'haarcascade_license_plate.xml')

    def process_image(self, file_path):
        image = cv2.imread(file_path)
        gray = cv2.cvtColor(image, cv2.COLOR_BGR2GRAY)
        
        cars = self.detect_cars(gray)
        plates = self.detect_license_plates(gray)

        self.draw_boxes(image, cars, (0, 255, 0))  # Green for cars
        self.draw_boxes(image, plates, (255, 0, 0))  # Blue for plates

        plate_texts = self.extract_license_plate_text(image, plates)
        print("Detected License Plate Texts:", plate_texts)

        return Image.fromarray(cv2.cvtColor(image, cv2.COLOR_BGR2RGB))

    def detect_cars(self, gray):
        return self.car_cascade.detectMultiScale(gray, 1.1, 1)

    def detect_license_plates(self, gray):
        return self.plate_cascade.detectMultiScale(gray, 1.1, 1)

    def draw_boxes(self, image, detections, color):
        for (x, y, w, h) in detections:
            cv2.rectangle(image, (x, y), (x + w, y + h), color, 2)

    def extract_license_plate_text(self, image, plates):
        texts = []
        for (x, y, w, h) in plates:
            roi = image[y:y + h, x:x + w]
            text = pytesseract.image_to_string(roi)
            texts.append(text.strip())
        return texts
```

'''

import cv2
import numpy as np
import pytesseract
from PIL import Image
class ImageProcessor:
    def __init__(self):
        self.car_cascade = cv2.CascadeClassifier(cv2.data.haarcascades + 'haarcascade_car.xml')
        self.plate_cascade = cv2.CascadeClassifier(cv2.data.haarcascades + 'haarcascade_license_plate.xml')
    def process_image(self, file_path):
        image = cv2.imread(file_path)
        gray = cv2.cvtColor(image, cv2.COLOR_BGR2GRAY)
        cars = self.detect_cars(gray)
        plates = self.detect_license_plates(gray)
        self.draw_boxes(image, cars, (0, 255, 0))  # Green for cars
        self.draw_boxes(image, plates, (255, 0, 0))  # Blue for plates
        plate_texts = self.extract_license_plate_text(image, plates)
        print("Detected License Plate Texts:", plate_texts)
        return Image.fromarray(cv2.cvtColor(image, cv2.COLOR_BGR2RGB))
    def detect_cars(self, gray):
        return self.car_cascade.detectMultiScale(gray, 1.1, 1)
    def detect_license_plates(self, gray):
        return self.plate_cascade.detectMultiScale(gray, 1.1, 1)
    def draw_boxes(self, image, detections, color):
        for (x, y, w, h) in detections:
            cv2.rectangle(image, (x, y), (x + w, y + h), color, 2)

        texts = []
        for (x, y, w, h) in plates:
            roi = image[y:y + h, x:x + w]
            text = pytesseract.image_to_string(roi)
            texts.append(text.strip())
        return texts

import cv2
import numpy as np
from PIL import Image
import pytesseract
class ImageProcessor:
    def __init__(self):
        self.car_cascade = cv2.CascadeClassifier(cv2.data.haarcascades + 'haarcascade_car.xml')
        self.plate_cascade = cv2.CascadeClassifier(cv2.data.haarcascades + 'haarcascade_license_plate.xml')
    def process_image(self, file_path):
        image = cv2.imread(file_path)
        gray = cv2.cvtColor(image, cv2.COLOR_BGR2GRAY)
        cars = self.detect_cars(gray)
        plates = self.detect_license_plates(gray)
        self.draw_boxes(image, cars, (0, 255, 0))  # Green for cars
        self.draw_boxes(image, plates, (255, 0, 0))  # Blue for plates
        plate_texts = self.extract_license_plate_text(image, plates)
        print("Detected License Plate Texts:", plate_texts)
        return Image.fromarray(cv2.cvtColor(image, cv2.COLOR_BGR2RGB)
    def detect_cars(self, gray):
        return self.car_cascade.detectMultiScale(gray, 1.1, 1)

        print("Detected License Plate Texts:", plate_texts)
        return Image.fromarray(cv2.cvtColor(image, cv2.COLOR_BGR2RGB))

'''
Image processing module for detecting cars and license plates.
'''
import cv2
import numpy as np
from PIL import Image
import pytesseract
class ImageProcessor:
    def __init__(self):
        self.car_cascade = cv2.CascadeClassifier(cv2.data.haarcascades + 'haarcascade_car.xml')
        self.plate_cascade = cv2.CascadeClassifier(cv2.data.haarcascades + 'haarcascade_license_plate.xml')
    def process_image(self, file_path):
        image = cv2.imread(file_path)
        gray = cv2.cvtColor(image, cv2.COLOR_BGR2GRAY)
        cars = self.detect_cars(gray)
        plates = self.detect_license_plates(gray)
        self.draw_boxes(image, cars, (0, 255, 0))  # Green for cars
        self.draw_boxes(image, plates, (255, 0, 0))  # Blue for plates
        plate_texts = self.extract_license_plate_text(image, plates)
        print("Detected License Plate Texts:", plate_texts)
        return Image.fromarray(cv2.cvtColor(image, cv2.COLOR_BGR2RGB))
    def detect_cars(self, gray):
        return self.car_cascade.detectMultiScale(gray, 1.1, 1)
    def detect_license_plates(self, gray):
        return self.plate_cascade.detectMultiScale(gray, 1.1, 1)
    def draw_boxes(self, image, detections, color):
        for (x, y, w, h) in detections:
            cv2.rectangle(image, (x, y), (x + w, y + h), color, 2)
    def extract_license_plate_text(self, image, plates):
        texts = []
        for (x, y, w, h) in plates:
            roi = image[y:y + h, x:x + w]
            text = pytesseract.image_to_string(roi)
            texts.append(text.strip())
        return texts

'''
Image processing module for detecting cars and license plates.
'''
import cv2
import numpy as np
from PIL import Image
import pytesseract
class ImageProcessor:
    def __init__(self):
        self.car_cascade = cv2.CascadeClassifier(cv2.data.haarcascades + 'haarcascade_car.xml')
        self.plate_cascade = cv2.CascadeClassifier(cv2.data.haarcascades + 'haarcascade_license_plate.xml')
    def process_image(self, file_path):
        image = cv2.imread(file_path)
        gray = cv2.cvtColor(image, cv2.COLOR_BGR2GRAY)
        cars = self.detect_cars(gray)
        plates = self.detect_license_plates(gray)
        self.draw_boxes(image, cars, (0, 255, 0))  # Green for cars
        self.draw_boxes(image, plates, (255, 0, 0))  # Blue for plates
        plate_texts = self.extract_license_plate_text(image, plates)
        print("Detected License Plate Texts:", plate_texts)
        return Image.fromarray(cv2.cvtColor(image, cv2.COLOR_BGR2RGB))
    def detect_cars(self, gray):
        return self.car_cascade.detectMultiScale(gray, 1.1, 1)
    def detect_license_plates(self, gray):
        return self.plate_cascade.detectMultiScale(gray, 1.1, 1)
    def draw_boxes(self, image, detections, color):
        for (x, y, w, h) in detections:
            cv2.rectangle(image, (x, y), (x + w, y + h), color, 2)
    def extract_license_plate_text(self, image, plates):
        texts = []
        for (x, y, w, h) in plates:
            roi = image[y:y + h, x:x + w]
            text = pytesseract.image_to_string(roi)
            texts.append(text.strip())
        return texts

# Car Plate Detector User Manual

## Introduction

Welcome to the Car Plate Detector application! This software is designed to process images by detecting cars and their license plates, drawing squares around both, and extracting the license plate numbers as text. This user manual will guide you through the main functions of the software, how to install the required environment dependencies, and how to use the application effectively.

## Main Functions

1. **Load Image**: Allows you to select an image file from your device.
2. **Process Image**: Processes the selected image by detecting cars and license plates, drawing squares around them, and extracting the license plate numbers as text.
3. **Display Result**: Shows the modified image with squares around the detected cars and license plates.

## Installation

To use the Car Plate Detector application, you need to install the required dependencies listed in the `requirements.txt` file. You can do this using `pip` by running the following command in your terminal:

```bash
pip install -r requirements.txt
```

Make sure you have Python installed on your system before proceeding with the installation.

## How to Use

1. Run the `main.py` file using Python to start the application.
2. Click on the "Load Image" button to select an image file from your device.
3. The application will process the image and display the modified version with squares around the detected cars and license plates.
4. The license plate numbers extracted from the image will be printed in the console.
