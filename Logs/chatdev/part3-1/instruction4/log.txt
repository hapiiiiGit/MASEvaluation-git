import tkinter as tk

from ocr_processor import OCRProcessor

class OCRApp:
    def __init__(self, master):
        self.master = master

        self.label = tk.Label(master, text="Load an image for OCR:")

        self.result_text = tk.Text(master, height=10, width=50)
        self.result_text.pack()

        self.ocr_processor = OCRProcessor()

    def load_image(self):
        file_path = filedialog.askopenfilename()
        if file_path:
            self.ocr_processor.set_image(file_path)
            messagebox.showinfo("Image Loaded", "Image loaded successfully!")

    def run_ocr(self):
        if self.ocr_processor.image_path:
            result = self.ocr_processor.extract_text()
            self.display_result(result)
        else:
            messagebox.showwarning("No Image", "Please load an image first.")

        self.result_text.delete(1.0, tk.END)

'''
OCR processing module for Arabic handwriting using AHDoc and AHR-OCR2024.
'''
import ahdoc  # Import for AHDoc
import ahr_ocr  # Import for AHR-OCR2024
class OCRProcessor:
    def __init__(self):
        self.image_path = None
    def set_image(self, image_path):
        self.image_path = image_path
    def extract_text(self):
        if self.image_path:
            # Utilizing AHDoc and AHR-OCR2024 for OCR operations
            printed_text = ahdoc.process_image(self.image_path)  # Process image for printed text
            cursive_text = ahr_ocr.recognize_handwriting(self.image_path)  # Recognize handwriting for cursive text
            return f"Printed Text: {printed_text}\nCursive Text: {cursive_text}"
        return "No image set for OCR."

    def __init__(self):

'''
Main application file for Arabic handwriting OCR using AHDoc and AHR-OCR2024.
'''
import tkinter as tk
from tkinter import filedialog, messagebox
from ocr_processor import OCRProcessor
class OCRApp:
    def __init__(self, master):
        self.master = master
        master.title("Arabic Handwriting OCR")
        self.label = tk.Label(master, text="Load an image for OCR:")
        self.label.pack()
        self.load_button = tk.Button(master, text="Load Image", command=self.load_image)
        self.load_button.pack()
        self.run_button = tk.Button(master, text="Run OCR", command=self.run_ocr)
        self.run_button.pack()
        self.result_text = tk.Text(master, height=10, width=50)
        self.result_text.pack()
        self.ocr_processor = OCRProcessor()
    def load_image(self):
        file_path = filedialog.askopenfilename()
        if file_path:
            self.ocr_processor.set_image(file_path)
            messagebox.showinfo("Image Loaded", "Image loaded successfully!")
    def run_ocr(self):
        if self.ocr_processor.image_path:
            result = self.ocr_processor.extract_text()
            self.display_result(result)
        else:
            messagebox.showwarning("No Image", "Please load an image first.")
    def display_result(self, result):
        self.result_text.delete(1.0, tk.END)
        self.result_text.insert(tk.END, result)
if __name__ == "__main__":
    root = tk.Tk()
    app = OCRApp(root)
    root.mainloop()

```python
'''
OCR processing module for Arabic handwriting using AHDoc and AHR-OCR2024.
'''
import ahdoc  # Import for AHDoc
import ahr_ocr  # Import for AHR-OCR2024
class OCRProcessor:
    def __init__(self):
        self.image_path = None
    def set_image(self, image_path):
        self.image_path = image_path
    def extract_text(self):
        if self.image_path:
            try:
                printed_text = ahdoc.process_image(self.image_path)  # Process image for printed text
                cursive_text = ahr_ocr.recognize_handwriting(self.image_path)  # Recognize handwriting for cursive text
                return f"Printed Text: {printed_text}\nCursive Text: {cursive_text}"
            except Exception as e:
                return f"Error during OCR processing: {str(e)}"
        return "No image set for OCR."

Main application file for Arabic handwriting OCR using AHDoc and AHR-OCR2024.
'''
import tkinter as tk
from tkinter import filedialog, messagebox
from ocr_processor import OCRProcessor
class OCRApp:
    def __init__(self, master):
        self.master = master
        master.title("Arabic Handwriting OCR")
        self.label = tk.Label(master, text="Load an image for OCR:")
        self.label.pack()
        self.load_button = tk.Button(master, text="Load Image", command=self.load_image)
        self.load_button.pack()
        self.run_button = tk.Button(master, text="Run OCR", command=self.run_ocr)
        self.run_button.pack()
        self.result_text = tk.Text(master, height=10, width=50)
        self.result_text.pack()
        self.ocr_processor = OCRProcessor()
    def load_image(self):
        file_path = filedialog.askopenfilename()
        if file_path:
            self.ocr_processor.set_image(file_path)
            messagebox.showinfo("Image Loaded", "Image loaded successfully!")
    def run_ocr(self):
        if self.ocr_processor.image_path:
            result = self.ocr_processor.extract_text()
            self.display_result(result)
        else:
            messagebox.showwarning("No Image", "Please load an image first.")
    def display_result(self, result):
        self.result_text.delete(1.0, tk.END)
        self.result_text.insert(tk.END, result)
if __name__ == "__main__":
    root = tk.Tk()
    app = OCRApp(root)
    root.mainloop()

'''
OCR processing module for Arabic handwriting using AHDoc and AHR-OCR2024.
'''
import ahdoc  # Import for AHDoc
import ahr_ocr  # Import for AHR-OCR2024
class OCRProcessor:
    def __init__(self):
        self.image_path = None
    def set_image(self, image_path):
        self.image_path = image_path
    def extract_text(self):
        if self.image_path:
            try:
                printed_text = ahdoc.process_image(self.image_path)  # Process image for printed text
                cursive_text = ahr_ocr.recognize_handwriting(self.image_path)  # Recognize handwriting for cursive text
                return f"Printed Text: {printed_text}\nCursive Text: {cursive_text}"
            except Exception as e:
                return f"Error during OCR processing: {str(e)}"
        return "No image set for OCR."

# Arabic Handwriting OCR Application

This application allows users to perform Optical Character Recognition (OCR) on Arabic handwriting, supporting both printed and cursive styles. It utilizes the AHDoc and AHR-OCR2024 libraries to extract and recognize text from images.

## Main Functions

- **Load Image**: Users can load an image file containing Arabic handwriting or printed text.
- **Run OCR**: After loading an image, users can run the OCR process to extract printed text and recognize cursive handwriting.
- **Display Results**: The extracted text is displayed in a text area for easy viewing.

## Installation Instructions

To set up the environment and run the application, follow these steps:

1. **Install Python**: Ensure you have Python 3.6 or higher installed on your machine. You can download it from [python.org](https://www.python.org/downloads/).

2. **Install Required Dependencies**: Open your terminal or command prompt and run the following commands to install the necessary libraries:

   ```bash
   pip install ahdoc
   pip install ahr-ocr
   pip install tkinter
   ```

   Note: If you encounter any issues with the installation, please refer to the documentation of the respective libraries for troubleshooting.

3. **Download the Application Files**: Clone or download the application files (`main.py`, `ocr_processor.py`, and `utils.py`) from the repository.

4. **Run the Application**: Navigate to the directory where the application files are located and run the following command:

   ```bash
   python main.py
   ```
