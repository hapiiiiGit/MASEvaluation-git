```python
'''
Class to create a lightweight AWS Lambda Layer for KeyBERT and keyphrase-vectorizers.
'''

import os
import subprocess
import zipfile

class LambdaLayerCreator:
    def __init__(self, layer_name):
        self.layer_name = layer_name
        self.dependencies = ["keybert", "keyphrase-vectorizers"]

    def package_dependencies(self):
        try:
            os.makedirs(f"{self.layer_name}/python", exist_ok=True)
            for dependency in self.dependencies:
                subprocess.check_call([f"pip", "install", dependency, "-t", f"{self.layer_name}/python"])
            return True
        except Exception as e:
            print(f"Error packaging dependencies: {e}")
            return False

    def zip_layer(self):
        with zipfile.ZipFile(f"{self.layer_name}.zip", 'w') as zipf:
            for root, dirs, files in os.walk(self.layer_name):
                for file in files:
                    zipf.write(os.path.join(root, file),
                                os.path.relpath(os.path.join(root, file),
                                                os.path.join(self.layer_name, '..')))
        print(f"Layer zipped as {self.layer_name}.zip")

```

'''
Class to create a lightweight AWS Lambda Layer for KeyBERT and keyphrase-vectorizers.
'''
import os
import subprocess
import zipfile
class LambdaLayerCreator:
    def __init__(self, layer_name):
        self.layer_name = layer_name
        self.dependencies = ["keybert", "keyphrase-vectorizers"]
    def package_dependencies(self):
        try:
            os.makedirs(f"{self.layer_name}/python", exist_ok=True)
            for dependency in self.dependencies:
                subprocess.check_call([f"pip", "install", dependency, "-t", f"{self.layer_name}/python"])
            return True
        except Exception as e:
            print(f"Error packaging dependencies: {e}")
            return False
    def zip_layer(self):
        with zipfile.ZipFile(f"{self.layer_name}.zip", 'w') as zipf:
            for root, dirs, files in os.walk(self.layer_name):
                for file in files:
                    zipf.write(os.path.join(root, file),
                                os.path.relpath(os.path.join(root, file),
                                                os.path.join(self.layer_name, '..')))
        print(f"Layer zipped as {self.layer_name}.zip")

import os
import subprocess
import zipfile

class LambdaLayerCreator:
    def __init__(self, layer_name):
        self.layer_name = layer_name
        self.dependencies = ["keybert", "keyphrase-vectorizers"]

    def package_dependencies(self):
        try:

'''
Class to create a lightweight AWS Lambda Layer for KeyBERT and keyphrase-vectorizers.
'''
import os
import subprocess
import zipfile
class LambdaLayerCreator:
    def __init__(self, layer_name):
        self.layer_name = layer_name
        self.dependencies = ["keybert", "keyphrase-vectorizers"]
    def package_dependencies(self):
        try:
            os.makedirs(f"{self.layer_name}/python", exist_ok=True)
            for dependency in self.dependencies:
                subprocess.check_call([f"pip", "install", dependency, "-t", f"{self.layer_name}/python"])
            return True
        except Exception as e:
            print(f"Error packaging dependencies: {e}")
            return False
    def zip_layer(self):
        with zipfile.ZipFile(f"{self.layer_name}.zip", 'w') as zipf:
            for root, dirs, files in os.walk(self.layer_name):
                for file in files:
                    zipf.write(os.path.join(root, file),
                                os.path.relpath(os.path.join(root, file),
                                                os.path.join(self.layer_name, '..')))
        if os.path.getsize(f"{self.layer_name}.zip") > 250 * 1024 * 1024:  # Check if size exceeds 250MB
            print("Warning: The zipped layer exceeds 250MB.")
        print(f"Layer zipped as {self.layer_name}.zip")

'''
Class to create a lightweight AWS Lambda Layer for KeyBERT and keyphrase-vectorizers.
'''
import os
import subprocess
import zipfile
class LambdaLayerCreator:
    def __init__(self, layer_name):
        self.layer_name = layer_name
        self.dependencies = ["keybert", "keyphrase-vectorizers"]
    def package_dependencies(self):
        try:
            os.makedirs(f"{self.layer_name}/python", exist_ok=True)
            for dependency in self.dependencies:
                subprocess.check_call([f"pip", "install", dependency, "-t", f"{self.layer_name}/python"])
            return True
        except Exception as e:
            print(f"Error packaging dependencies: {e}")
            return False
    def zip_layer(self):
        with zipfile.ZipFile(f"{self.layer_name}.zip", 'w') as zipf:
            for root, dirs, files in os.walk(self.layer_name):
                for file in files:
                    zipf.write(os.path.join(root, file),
                                os.path.relpath(os.path.join(root, file),
                                                os.path.join(self.layer_name, '..')))
        if os.path.getsize(f"{self.layer_name}.zip") > 250 * 1024 * 1024:  # Check if size exceeds 250MB
            print("Warning: The zipped layer exceeds 250MB.")
        print(f"Layer zipped as {self.layer_name}.zip")

'''
Class to create a lightweight AWS Lambda Layer for KeyBERT and keyphrase-vectorizers.
'''
import os
import subprocess
import zipfile
class LambdaLayerCreator:
    def __init__(self, layer_name):
        self.layer_name = layer_name
        self.dependencies = ["keybert", "keyphrase-vectorizers"]
    def package_dependencies(self):
        try:
            os.makedirs(f"{self.layer_name}/python", exist_ok=True)
            for dependency in self.dependencies:
                subprocess.check_call([f"pip", "install", dependency, "-t", f"{self.layer_name}/python"])
            return True
        except Exception as e:
            print(f"Error packaging dependencies: {e}")
            return False
    def zip_layer(self):
        with zipfile.ZipFile(f"{self.layer_name}.zip", 'w') as zipf:
            for root, dirs, files in os.walk(self.layer_name):
                for file in files:
                    zipf.write(os.path.join(root, file),
                                os.path.relpath(os.path.join(root, file),
                                                os.path.join(self.layer_name, '..')))
        if os.path.getsize(f"{self.layer_name}.zip") > 250 * 1024 * 1024:  # Check if size exceeds 250MB
            print("Warning: The zipped layer exceeds 250MB.")
        print(f"Layer zipped as {self.layer_name}.zip")

'''
Class to create a lightweight AWS Lambda Layer for KeyBERT and keyphrase-vectorizers.
'''
import os
import subprocess
import zipfile
class LambdaLayerCreator:
    def __init__(self, layer_name):
        self.layer_name = layer_name
        self.dependencies = ["keybert", "keyphrase-vectorizers"]
    def package_dependencies(self):
        try:
            os.makedirs(f"{self.layer_name}/python", exist_ok=True)
            for dependency in self.dependencies:
                subprocess.check_call([f"pip", "install", dependency, "-t", f"{self.layer_name}/python"])
            return True
        except Exception as e:
            print(f"Error packaging dependencies: {e}")
            return False
    def zip_layer(self):
        with zipfile.ZipFile(f"{self.layer_name}.zip", 'w') as zipf:
            for root, dirs, files in os.walk(self.layer_name):
                for file in files:
                    zipf.write(os.path.join(root, file),
                                os.path.relpath(os.path.join(root, file),
                                                os.path.join(self.layer_name, '..')))
        if os.path.getsize(f"{self.layer_name}.zip") > 250 * 1024 * 1024:  # Check if size exceeds 250MB
            print("Warning: The zipped layer exceeds 250MB.")
        print(f"Layer zipped as {self.layer_name}.zip")

'''
Class to create a lightweight AWS Lambda Layer for KeyBERT and keyphrase-vectorizers.
'''
import os
import subprocess
import zipfile
class LambdaLayerCreator:
    def __init__(self, layer_name):
        self.layer_name = layer_name
        self.dependencies = ["keybert", "keyphrase-vectorizers"]
    def package_dependencies(self):
        try:
            os.makedirs(f"{self.layer_name}/python", exist_ok=True)
            for dependency in self.dependencies:
                subprocess.check_call([f"pip", "install", dependency, "-t", f"{self.layer_name}/python"])
            return True
        except Exception as e:
            print(f"Error packaging dependencies: {e}")
            return False
    def zip_layer(self):
        with zipfile.ZipFile(f"{self.layer_name}.zip", 'w') as zipf:
            for root, dirs, files in os.walk(self.layer_name):
                for file in files:
                    zipf.write(os.path.join(root, file),
                                os.path.relpath(os.path.join(root, file),
                                                os.path.join(self.layer_name, '..')))
        if os.path.getsize(f"{self.layer_name}.zip") > 250 * 1024 * 1024:  # Check if size exceeds 250MB
            print("Warning: The zipped layer exceeds 250MB.")
        print(f"Layer zipped as {self.layer_name}.zip")

'''
Class to create a lightweight AWS Lambda Layer for KeyBERT and keyphrase-vectorizers.
'''
import os
import subprocess
import zipfile
class LambdaLayerCreator:
    def __init__(self, layer_name):
        self.layer_name = layer_name
        self.dependencies = ["keybert", "keyphrase-vectorizers"]
    def package_dependencies(self):
        try:
            os.makedirs(f"{self.layer_name}/python", exist_ok=True)
            for dependency in self.dependencies:
                subprocess.check_call([f"pip", "install", dependency, "-t", f"{self.layer_name}/python"])
            return True
        except Exception as e:
            print(f"Error packaging dependencies: {e}")
            return False
    def zip_layer(self):
        with zipfile.ZipFile(f"{self.layer_name}.zip", 'w') as zipf:
            for root, dirs, files in os.walk(self.layer_name):
                for file in files:
                    zipf.write(os.path.join(root, file),
                                os.path.relpath(os.path.join(root, file),
                                                os.path.join(self.layer_name, '..')))
        if os.path.getsize(f"{self.layer_name}.zip") > 250 * 1024 * 1024:  # Check if size exceeds 250MB
            print("Warning: The zipped layer exceeds 250MB.")
        print(f"Layer zipped as {self.layer_name}.zip")

'''
Class to create a lightweight AWS Lambda Layer for KeyBERT and keyphrase-vectorizers.
'''
import os
import subprocess
import zipfile
class LambdaLayerCreator:
    def __init__(self, layer_name):
        self.layer_name = layer_name
        self.dependencies = ["keybert", "keyphrase-vectorizers"]
    def package_dependencies(self):
        try:
            os.makedirs(f"{self.layer_name}/python", exist_ok=True)
            for dependency in self.dependencies:
                subprocess.check_call([f"pip", "install", dependency, "-t", f"{self.layer_name}/python"])
            return True
        except Exception as e:
            print(f"Error packaging dependencies: {e}")
            return False
    def zip_layer(self):
        with zipfile.ZipFile(f"{self.layer_name}.zip", 'w') as zipf:
            for root, dirs, files in os.walk(self.layer_name):
                for file in files:
                    zipf.write(os.path.join(root, file),
                                os.path.relpath(os.path.join(root, file),
                                                os.path.join(self.layer_name, '..')))
        if os.path.getsize(f"{self.layer_name}.zip") > 250 * 1024 * 1024:  # Check if size exceeds 250MB
            print("Warning: The zipped layer exceeds 250MB.")
        print(f"Layer zipped as {self.layer_name}.zip")

'''
Class to create a lightweight AWS Lambda Layer for KeyBERT and keyphrase-vectorizers.
'''
import os
import subprocess
import zipfile
class LambdaLayerCreator:
    def __init__(self, layer_name):
        self.layer_name = layer_name
        self.dependencies = ["keybert", "keyphrase-vectorizers"]
    def package_dependencies(self):
        try:
            os.makedirs(f"{self.layer_name}/python", exist_ok=True)
            for dependency in self.dependencies:
                subprocess.check_call([f"pip", "install", dependency, "-t", f"{self.layer_name}/python"])
            return True
        except Exception as e:
            print(f"Error packaging dependencies: {e}")
            return False
    def zip_layer(self):
        with zipfile.ZipFile(f"{self.layer_name}.zip", 'w') as zipf:
            for root, dirs, files in os.walk(self.layer_name):
                for file in files:
                    zipf.write(os.path.join(root, file),
                                os.path.relpath(os.path.join(root, file),
                                                os.path.join(self.layer_name, '..')))
        if os.path.getsize(f"{self.layer_name}.zip") > 250 * 1024 * 1024:  # Check if size exceeds 250MB
            print("Warning: The zipped layer exceeds 250MB.")
        print(f"Layer zipped as {self.layer_name}.zip")
