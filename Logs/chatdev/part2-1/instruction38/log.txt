'''
Utility functions for checking if a string can be rearranged
so that no two adjacent characters are the same.
'''

from collections import Counter

def can_rearrange(s):
    '''
    Check if the letters of the given string can be rearranged
    such that no two adjacent characters are the same.
    
    :param s: Input string
    :return: True if rearrangement is possible, False otherwise
    '''
    if not s:
        return True

    char_count = Counter(s)
    max_count = max(char_count.values())
    n = len(s)

    # If the most frequent character is more than half of the string length (rounded up), it's impossible
    return max_count <= (n + 1) // 2
```

'''
Utility functions for checking if a string can be rearranged
so that no two adjacent characters are the same.

from collections import Counter
def can_rearrange(s):
    '''
    Check if the letters of the given string can be rearranged
    such that no two adjacent characters are the same.
    :param s: Input string
    :return: True if rearrangement is possible, False otherwise
    '''
    if not s:
        return True
    char_count = Counter(s)
    max_count = max(char_count.values())
    n = len(s)
    # If the most frequent character is more than half of the string length (rounded up), it's impossible
    return max_count <= (n + 1) // 2

from collections import Counter
def can_rearrange(s):

from collections import Counter
def can_rearrange(s):
    '''

'''
Main application file for checking if letters of a string can be rearranged
so that no two adjacent characters are the same.
'''
import tkinter as tk
from tkinter import messagebox
from utils import can_rearrange
class App:
    def __init__(self, master):
        self.master = master
        master.title("String Rearrangement Checker")
        self.label = tk.Label(master, text="Enter a string:")
        self.label.pack()
        self.entry = tk.Entry(master)
        self.entry.pack()
        self.check_button = tk.Button(master, text="Check", command=self.check_rearrangement)
        self.check_button.pack()
        self.result_label = tk.Label(master, text="")
        self.result_label.pack()
    def check_rearrangement(self):
        input_string = self.entry.get()
        # Validate input to ensure it contains only alphabetic characters
        if not input_string.isalpha():
            messagebox.showerror("Invalid Input", "Please enter only alphabetic characters.")
            return
        if can_rearrange(input_string):
            self.result_label.config(text="Yes, it can be rearranged!")
        else:
            self.result_label.config(text="No, it cannot be rearranged.")
if __name__ == "__main__":
    root = tk.Tk()
    app = App(root)
    root.mainloop()

'''
Utility functions for checking if a string can be rearranged
so that no two adjacent characters are the same.
'''
from collections import Counter
def can_rearrange(s):
    '''
    Check if the letters of the given string can be rearranged
    such that no two adjacent characters are the same.
    :param s: Input string
    :return: True if rearrangement is possible, False otherwise
    '''
    if not s:
        return True
    char_count = Counter(s)
    max_count = max(char_count.values())
    n = len(s)
    # If the most frequent character is more than half of the string length (rounded up), it's impossible
    return max_count <= (n + 1) // 2
