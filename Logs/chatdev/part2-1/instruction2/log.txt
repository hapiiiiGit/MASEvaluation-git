'''
This module contains the NonPrimeIdentifier class which provides methods
to identify non-prime numbers.
'''

class NonPrimeIdentifier:
    @staticmethod
    def is_non_prime(num):
        """Check if a number is non-prime."""
        if num < 2:
            return True  # 0 and 1 are non-prime
        for i in range(2, int(num**0.5) + 1):
            if num % i == 0:
                return True  # Found a divisor, hence non-prime
        return False  # The number is prime

    @staticmethod
    def get_non_primes_in_range(start, end):
        """Get a list of non-prime numbers in the given range."""
        return [num for num in range(start, end + 1) if NonPrimeIdentifier.is_non_prime(num)]

'''
This module contains the NonPrimeIdentifier class which provides methods
to identify non-prime numbers.
'''
class NonPrimeIdentifier:
    @staticmethod
    def is_non_prime(num):
        """Check if a number is non-prime."""
        if num < 2:
            return True  # 0 and 1 are non-prime
        for i in range(2, int(num**0.5) + 1):
            if num % i == 0:
                return True  # Found a divisor, hence non-prime
        return False  # The number is prime
    @staticmethod
    def get_non_primes_in_range(start, end):
        """Get a list of non-prime numbers in the given range."""
        return [num for num in range(start, end + 1) if NonPrimeIdentifier.is_non_prime(num)]

class NonPrimeIdentifier:
    @staticmethod
    def is_non_prime(num):
        """Check if a number is non-prime."""
        if num < 2:
            return True  # 0 and 1 are non-prime
        for i in range(2, int(num**0.5) + 1):

    @staticmethod
    def get_non_primes_in_range(start, end):
        """Get a list of non-prime numbers in the given range."""
        return [num for num in range(start, end + 1) if NonPrimeIdentifier.is_non_prime(num)]

'''
This module contains the NonPrimeIdentifier class which provides methods
to identify non-prime numbers.
'''
class NonPrimeIdentifier:
    @staticmethod
    def is_non_prime(num):
        """Check if a number is non-prime."""
        if num < 2:
            return True  # 0 and 1 are non-prime
        if num == 2:
            return False  # 2 is a prime number
        for i in range(2, int(num**0.5) + 1):
            if num % i == 0:
                return True  # Found a divisor, hence non-prime
        return False  # The number is prime
    @staticmethod
    def get_non_primes_in_range(start, end):
        """Get a list of non-prime numbers in the given range."""
        return [num for num in range(start, end + 1) if NonPrimeIdentifier.is_non_prime(num)]

'''
This module contains the NonPrimeIdentifier class which provides methods
to identify non-prime numbers.
'''
class NonPrimeIdentifier:
    @staticmethod
    def is_non_prime(num):
        """Check if a number is non-prime."""
        if num < 2:
            return True  # 0 and 1 are non-prime
        if num == 2:
            return False  # 2 is a prime number
        for i in range(2, int(num**0.5) + 1):
            if num % i == 0:
                return True  # Found a divisor, hence non-prime
        return False  # The number is prime
    @staticmethod
    def get_non_primes_in_range(start, end):
        """Get a list of non-prime numbers in the given range."""
        return [num for num in range(start, end + 1) if NonPrimeIdentifier.is_non_prime(num)]

import tkinter as tk

class App:
    def __init__(self):
        """Initialize the GUI components."""
        self.root = tk.Tk()
        self.root.title("Non-Prime Number Identifier")
        self.label_start = tk.Label(self.root, text="Start:")

        self.button_check.pack()
        self.result_label = tk.Label(self.root, text="")
        self.result_label.pack()
    def check_non_primes(self):
