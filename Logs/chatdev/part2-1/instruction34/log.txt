Main entry point for the Rectangular Number Calculator application.
'''

from tkinter import Tk, Label, Entry, Button, StringVar
from calculator import RectangularNumberCalculator

class App:
    def __init__(self):
        self.window = Tk()
        self.window.title("Rectangular Number Calculator")

        self.label = Label(self.window, text="Enter n:")
        self.label.pack()

        self.input_value = StringVar()
        self.entry = Entry(self.window, textvariable=self.input_value)
        self.entry.pack()

        self.calculate_button = Button(self.window, text="Calculate", command=self.calculate_rectangular_number)
        self.calculate_button.pack()

        self.result_label = Label(self.window, text="")
        self.result_label.pack()

        self.calculator = RectangularNumberCalculator()

    def calculate_rectangular_number(self):
        try:
            n = int(self.input_value.get())
            result = self.calculator.calculate(n)
            self.result_label.config(text=f"The {n}-th rectangular number is: {result}")
        except ValueError:
            self.result_label.config(text="Please enter a valid integer.")

    def run(self):
        self.window.mainloop()

if __name__ == "__main__":
    app = App()
    app.run()
```

class RectangularNumberCalculator:
    def calculate(self, n):
        '''
        Calculate the n-th rectangular number.
        The n-th rectangular number is given by the formula: n * (n + 1) / 2.
        '''
        if n < 1:
            raise ValueError("n must be a positive integer.")
        return n * (n + 1) // 2

'''

from tkinter import Tk, Label, Entry, Button, StringVar
from calculator import RectangularNumberCalculator
class App:
    def __init__(self):
        self.window = Tk()
        self.window.title("Rectangular Number Calculator")
        self.label = Label(self.window, text="Enter n:")
        self.label.pack()
        self.input_value = StringVar()
        self.entry = Entry(self.window, textvariable=self.input_value)
        self.entry.pack()

        self.calculator = RectangularNumberCalculator()

            n = int(self.input_value.get())
            result = self.calculator.calculate(n)
            self.result_label.config(text=f"The {n}-th rectangular number is: {result}")
        except ValueError:
            self.result_label.config(text="Please enter a valid integer.")

        self.window.mainloop()
