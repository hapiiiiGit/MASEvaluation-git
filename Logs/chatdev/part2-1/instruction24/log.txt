'''
Module to calculate the product of non-repeated elements in an array.

class ProductCalculator:
    def __init__(self, array):
        self.array = array

    def product_of_non_repeated(self):
        element_count = {}
        for num in self.array:
            if num in element_count:
                element_count[num] += 1
            else:
                element_count[num] = 1
        
        product = 1
        has_non_repeated = False
        for num, count in element_count.items():
            if count == 1:
                product *= num
                has_non_repeated = True
        
        return product if has_non_repeated else 0

class ProductCalculator:
    def __init__(self, array):
        self.array = array
    def product_of_non_repeated(self):
        element_count = {}
        for num in self.array:
            if num in element_count:
                element_count[num] += 1

                element_count[num] = 1
        product = 1
        has_non_repeated = False
        for num, count in element_count.items():
            if count == 1:
                product *= num
                has_non_repeated = True
        return product if has_non_repeated else 0

Module to calculate the product of non-repeated elements in an array.

class ProductCalculator:
    def __init__(self, array):
        self.array = array
    def product_of_non_repeated(self):
        element_count = {}
        for num in self.array:
            if num in element_count:
                element_count[num] += 1
            else:
                element_count[num] = 1
        product = 1
        has_non_repeated = False
        for num, count in element_count.items():
            if count == 1:
                product *= num
                has_non_repeated = True
        return product if has_non_repeated else 0

'''
Module to calculate the product of non-repeated elements in an array.
'''
class ProductCalculator:
    def __init__(self, array):
        self.array = array
    def product_of_non_repeated(self):
        element_count = {}
        for num in self.array:
            if num in element_count:
                element_count[num] += 1
            else:
                element_count[num] = 1
        product = 1
        has_non_repeated = False
        for num, count in element_count.items():
            if count == 1:
                product *= num
                has_non_repeated = True
        return product if has_non_repeated else 0

'''
Main application file for the product of non-repeated elements in an array.
'''
from tkinter import Tk, Label, Entry, Button, StringVar
from product_calculator import ProductCalculator
class MainApp:
    def __init__(self):
        self.root = Tk()
        self.root.title("Product of Non-Repeated Elements")
        self.label = Label(self.root, text="Enter numbers separated by commas:")
        self.label.pack()
        self.input_var = StringVar()
        self.entry = Entry(self.root, textvariable=self.input_var)
        self.entry.pack()
        self.result_label = Label(self.root, text="")
        self.result_label.pack()
        self.calculate_button = Button(self.root, text="Calculate Product", command=self.calculate_product)
        self.calculate_button.pack()
        self.root.mainloop()
    def calculate_product(self):
        input_data = self.input_var.get()
        if not input_data.strip():  # Check for empty input
            self.result_label.config(text="Please enter some numbers.")
            return
        try:
            array = list(map(int, input_data.split(',')))
            calculator = ProductCalculator(array)
            product = calculator.product_of_non_repeated()
            if product == 0:
                self.result_label.config(text="No non-repeated elements found.")
            else:
                self.result_label.config(text=f"Product: {product}")
        except ValueError:
            self.result_label.config(text="Invalid input! Please enter integers separated by commas.")
if __name__ == "__main__":
    MainApp()

'''
Module to calculate the product of non-repeated elements in an array.
'''
class ProductCalculator:
    def __init__(self, array):
        self.array = array
    def product_of_non_repeated(self):
        element_count = {}
        for num in self.array:
            if num in element_count:
                element_count[num] += 1
            else:
                element_count[num] = 1
        product = 1
        has_non_repeated = False
        for num, count in element_count.items():
            if count == 1:
                product *= num
                has_non_repeated = True
        return product if has_non_repeated else 0
