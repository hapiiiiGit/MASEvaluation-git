'''
This module contains the GCDCalculator class which provides methods to calculate the GCD of an array of numbers.
'''

import math
from functools import reduce

class GCDCalculator:
    @staticmethod
    def gcd(a, b):
        '''Computes the GCD of two numbers using the Euclidean algorithm.'''
        while b:
            a, b = b, a % b
        return abs(a)

    @staticmethod
    def find_gcd(arr):
        '''Computes the GCD of an array of numbers.'''
        return reduce(GCDCalculator.gcd, arr)
```

import tkinter as tk

        self.label.pack()

        self.entry.pack()

    def calculate_gcd(self):
        '''Retrieves input from the user, calculates the GCD, and displays the result.'''
        try:
            numbers = list(map(int, self.entry.get().split(',')))
            result = GCDCalculator.find_gcd(numbers)
            self.result_label.config(text=f"GCD: {result}")
        except ValueError:
            messagebox.showerror("Input Error", "Please enter valid integers separated by commas.")

import math
from functools import reduce
class GCDCalculator:
    @staticmethod
    def gcd(a, b):
        '''Computes the GCD of two numbers using the Euclidean algorithm.'''
        while b:
            a, b = b, a % b

'''
This module contains the GCDCalculator class which provides methods to calculate the GCD of an array of numbers.
'''
import math
from functools import reduce
class GCDCalculator:
    @staticmethod
    def gcd(a, b):
        '''Computes the GCD of two numbers using the Euclidean algorithm.'''
        while b:
            a, b = b, a % b
        return abs(a)
    @staticmethod
    def find_gcd(arr):
        '''Computes the GCD of an array of numbers.'''
        return reduce(GCDCalculator.gcd, arr)

class GCDCalculator:

import math
from functools import reduce
class GCDCalculator:
    @staticmethod

        '''Computes the GCD of two numbers using the Euclidean algorithm.'''
        while b:
            a, b = b, a % b
        return abs(a)

    def find_gcd(arr):
        '''Computes the GCD of an array of numbers.'''
        return reduce(GCDCalculator.gcd, arr)

```python
'''
This module contains the GCDCalculator class which provides methods to calculate the GCD of an array of numbers.
'''
import math
from functools import reduce
class GCDCalculator:
    @staticmethod
    def gcd(a, b):
        '''Computes the GCD of two numbers using the Euclidean algorithm.'''
        while b:
            a, b = b, a % b
        return abs(a)
    @staticmethod
    def find_gcd(arr):
        '''Computes the GCD of an array of numbers.'''
        return reduce(GCDCalculator.gcd, arr)
```

    def calculate_gcd(self):
        '''Retrieves input from the user, validates input, calculates the GCD, and displays the result.'''

'''
This module contains the GCDCalculator class which provides methods to calculate the GCD of an array of numbers.
'''
import math
from functools import reduce
class GCDCalculator:
    @staticmethod
    def gcd(a, b):
        '''Computes the GCD of two numbers using the Euclidean algorithm.'''
        while b:
            a, b = b, a % b
        return abs(a)
    @staticmethod
    def find_gcd(arr):
        '''Computes the GCD of an array of numbers.'''
        return reduce(GCDCalculator.gcd, arr)

    def calculate_gcd(self):
        '''Retrieves input from the user, validates input, calculates the GCD, and displays the result.'''
        input_str = self.entry.get().strip()  # Remove leading/trailing whitespaces

        try:
            numbers = [int(num) for num in input_str.split(',') if num.strip().isdigit()]

            result = GCDCalculator.find_gcd(numbers)
            self.result_label.config(text=f"GCD: {result}")
        except ValueError:
            messagebox.showerror("Input Error", "Please enter valid integers separated by commas.")
