'''

def count_ways(n):
    if n == 0:
        return 1
    if n == 1:
        return 0
    if n == 2:
        return 3
    
    # Dynamic programming array to store the number of ways
    dp = [0] * (n + 1)
    dp[0] = 1
    dp[1] = 0
    dp[2] = 3
    
    for i in range(3, n + 1):
        dp[i] = 4 * dp[i - 2] + dp[i - 4]
    
    return dp[n]

def count_ways(n):
    if n == 0:
        return 1
    if n == 1:
        return 0
    if n == 2:

    # Dynamic programming array to store the number of ways
    dp = [0] * (n + 1)

    dp[1] = 0
    dp[2] = 3
    for i in range(3, n + 1):
        dp[i] = 4 * dp[i - 2] + dp[i - 4]
    return dp[n]

'''
Module to calculate the number of ways to fill a 3 x n board with 2 x 1 dominoes.
'''
def count_ways(n):
    '''
    Function to calculate the number of ways to fill a 3 x n board with 2 x 1 dominoes.

    Parameters:
    n (int): The width of the board.

    Returns:
    int: The number of ways to fill the board.
    '''
    # If n is odd, it's impossible to fill the board completely with 2 x 1 dominoes
    if n % 2 != 0:
        return 0

    # Dynamic programming array to store the number of ways
    dp = [0] * (n + 1)
    dp[0] = 1  # Base case: 1 way to fill a 3 x 0 board (do nothing)
    dp[2] = 3  # Base case: 3 ways to fill a 3 x 2 board

def count_ways(n):
    '''
    Function to calculate the number of ways to fill a 3 x n board with 2 x 1 dominoes.
    Parameters:
    n (int): The width of the board.
    Returns:
    int: The number of ways to fill the board.
    '''
    # If n is odd, it's impossible to fill the board completely with 2 x 1 dominoes
    if n % 2 != 0:
        return 0
    # Dynamic programming array to store the number of ways
    dp = [0] * (n + 1)
    dp[0] = 1  # Base case: 1 way to fill a 3 x 0 board (do nothing)
    dp[2] = 3  # Base case: 3 ways to fill a 3 x 2 board
    # Calculate the number of ways using dynamic programming
    for i in range(4, n + 1, 2):
        # The recurrence relation to calculate the number of ways
        dp[i] = 4 * dp[i - 2] - dp[i - 4]
    return dp[n]

'''
Module to calculate the number of ways to fill a 3 x n board with 2 x 1 dominoes.
'''
def count_ways(n):
    '''
    Function to calculate the number of ways to fill a 3 x n board with 2 x 1 dominoes.
    Parameters:
    n (int): The width of the board.
    Returns:
    int: The number of ways to fill the board.
    '''
    # If n is odd, it's impossible to fill the board completely with 2 x 1 dominoes
    if n % 2 != 0:
        return 0
    # Dynamic programming array to store the number of ways
    dp = [0] * (n + 1)
    dp[0] = 1  # Base case: 1 way to fill a 3 x 0 board (do nothing)
    if n >= 2:
        dp[2] = 3  # Base case: 3 ways to fill a 3 x 2 board
    # Calculate the number of ways using dynamic programming
    for i in range(4, n + 1, 2):
        # The recurrence relation to calculate the number of ways
        dp[i] = 4 * dp[i - 2] - dp[i - 4]
    return dp[n]

'''
Module to calculate the number of ways to fill a 3 x n board with 2 x 1 dominoes.
'''
def count_ways(n):
    '''
    Function to calculate the number of ways to fill a 3 x n board with 2 x 1 dominoes.
    Parameters:
    n (int): The width of the board.
    Returns:
    int: The number of ways to fill the board.
    '''
    # If n is odd, it's impossible to fill the board completely with 2 x 1 dominoes
    if n % 2 != 0:
        return 0
    # Dynamic programming array to store the number of ways
    dp = [0] * (n + 1)
    dp[0] = 1  # Base case: 1 way to fill a 3 x 0 board (do nothing)
    if n >= 2:
        dp[2] = 3  # Base case: 3 ways to fill a 3 x 2 board
    # Calculate the number of ways using dynamic programming
    for i in range(4, n + 1, 2):
        # The recurrence relation to calculate the number of ways
        dp[i] = 4 * dp[i - 2] - dp[i - 4]
    return dp[n]
