        for tpl in self.sorted_tuples:

            lbl.pack()

if __name__ == "__main__":

def sort_tuples(tuple_list, key=lambda x: x[1]):
    """
    Sorts a list of tuples based on the specified key.
    Args:
        tuple_list (list): List of tuples to be sorted.
        key (function, optional): Key function to determine the sorting order. Defaults to sorting based on the second element of each tuple.
