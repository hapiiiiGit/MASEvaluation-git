import tkinter as tk
from tkinter import messagebox

class WoodballChecker:
    def __init__(self, master):
        self.master = master
        self.master.title("Woodball Number Checker")
        self.create_gui()

    def is_woodball(self, number):
        """
        Check if the given number is a woodball number.

                messagebox.showinfo("Result", f"{number} is a woodball number.")

                messagebox.showinfo("Result", f"{number} is not a woodball number.")
        except ValueError:
            messagebox.showerror("Error", "Please enter a valid integer.")

        """
        Set up the GUI components.
        """
        self.label = tk.Label(self.master, text="Enter a number:")
        self.label.pack(pady=10)

        self.entry = tk.Entry(self.master)
        self.entry.pack(pady=10)

        self.check_button = tk.Button(self.master, text="Check", command=self.check_number)
        self.check_button.pack(pady=10)

'''
import tkinter as tk
from tkinter import messagebox
class WoodballChecker:
    def __init__(self, master):
        self.master = master
        self.master.title("Woodball Number Checker")
        self.create_gui()
    def is_woodball(self, number):
        """
        Check if the given number is a woodball number.
        A woodball number is defined as a number that is a multiple of 3 or 5.
        """
        return number % 3 == 0 or number % 5 == 0
    def check_number(self):
        """
        Retrieve the input from the user, validate it, and display the result.
        """
        input_value = self.entry.get()

            number = int(input_value)

                messagebox.showinfo("Result", f"{number} is a woodball number.")
            else:
                messagebox.showinfo("Result", f"{number} is not a woodball number.")
        except ValueError:
            messagebox.showerror("Error", "Please enter a valid integer.")
    def create_gui(self):
        """
        Set up the GUI components.
        """
        self.label = tk.Label(self.master, text="Enter a number:")
        self.label.pack(pady=10)
        self.entry = tk.Entry(self.master)
        self.entry.pack(pady=10)
        self.check_button = tk.Button(self.master, text="Check", command=self.check_number)
        self.check_button.pack(pady=10)
        self.quit_button = tk.Button(self.master, text="Quit", command=self.master.quit)
        self.quit_button.pack(pady=10)

'''

import tkinter as tk

class WoodballChecker:
    def __init__(self, master):
        self.master = master
        self.master.title("Woodball Number Checker")
        self.create_gui()

        Check if the given number is a woodball number.
        A woodball number is defined as a number that is a multiple of 3 or 5.

## 🤔 What is this?

Large language models (LLMs) are emerging as a transformative technology, enabling developers to build applications that they previously could not. However, using these LLMs in isolation is often insufficient for creating a truly powerful app - the real power comes when you can combine them with other sources of computation or knowledge.

This library aims to assist in the development of those types of applications. Common examples of these applications include:

**❓ Question Answering over specific documents**

- Documentation

- End-to-end Example: Question Answering over Notion Database

**🤖 Agents**

- Documentation

- End-to-end Example: GPT+WolframAlpha
