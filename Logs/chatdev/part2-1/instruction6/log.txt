'''
Main application file for the Word Finder GUI.
'''

import tkinter as tk
from tkinter import scrolledtext, messagebox
from regex_util import find_long_words

class WordFinderApp:
    def __init__(self, root):
        self.root = root
        self.root.title("Word Finder")
        
        self.label = tk.Label(root, text="Enter a string:")
        self.label.pack(pady=10)

        self.text_input = tk.Text(root, height=10, width=50)
        self.text_input.pack(pady=10)

        self.find_button = tk.Button(root, text="Find Words", command=self.find_words)
        self.find_button.pack(pady=10)

        self.results_label = tk.Label(root, text="Found Words:")
        self.results_label.pack(pady=10)

        self.results_text = scrolledtext.ScrolledText(root, height=10, width=50)
        self.results_text.pack(pady=10)

    def find_words(self):
        input_text = self.text_input.get("1.0", tk.END).strip()
        if not input_text:
            messagebox.showwarning("Input Error", "Please enter a string.")
            return
        
        words = find_long_words(input_text)
        self.display_results(words)

    def display_results(self, words):
        self.results_text.delete("1.0", tk.END)  # Clear previous results
        if words:
            self.results_text.insert(tk.END, "\n".join(words))
        else:
            self.results_text.insert(tk.END, "No words found.")

if __name__ == "__main__":
    root = tk.Tk()
    app = WordFinderApp(root)
    root.mainloop()

import re

def find_long_words(input_string):
    '''
    Finds all words in the input string that are at least 4 characters long.
    
    Args:
        input_string (str): The string to search for words.
        
    Returns:
        list: A list of words that are at least 4 characters long.
    '''
    pattern = r'\b\w{4,}\b'

'''
Main application file for the Word Finder GUI.

import tkinter as tk
from tkinter import scrolledtext, messagebox
from regex_util import find_long_words
class WordFinderApp:
    def __init__(self, root):
        self.root = root
        self.root.title("Word Finder")
        self.label = tk.Label(root, text="Enter a string:")
        self.label.pack(pady=10)
        self.text_input = tk.Text(root, height=10, width=50)
        self.text_input.pack(pady=10)
        self.find_button = tk.Button(root, text="Find Words", command=self.find_words)
        self.find_button.pack(pady=10)
        self.results_label = tk.Label(root, text="Found Words:")
        self.results_label.pack(pady=10)
        self.results_text = scrolledtext.ScrolledText(root, height=10, width=50)
        self.results_text.pack(pady=10)
    def find_words(self):
        input_text = self.text_input.get("1.0", tk.END).strip()
        if not input_text:
            messagebox.showwarning("Input Error", "Please enter a string.")
            return
        words = find_long_words(input_text)
        self.display_results(words)
    def display_results(self, words):
        self.results_text.delete("1.0", tk.END)  # Clear previous results
        if words:
            self.results_text.insert(tk.END, "\n".join(words))
        else:
            self.results_text.insert(tk.END, "No words found.")
if __name__ == "__main__":
    root = tk.Tk()
    app = WordFinderApp(root)
    root.mainloop()

'''
Utility module for regex operations to find long words.
'''
import re
def find_long_words(input_string):
    '''
    Finds all words in the input string that are at least 4 characters long.
    Args:
        input_string (str): The string to search for words.
    Returns:
        list: A list of words that are at least 4 characters long.
    '''
    pattern = r'\b\w{4,}\b'
    return re.findall(pattern, input_string)

'''
Main application file for the Word Finder GUI.

import tkinter as tk
from tkinter import scrolledtext, messagebox
from regex_util import find_long_words
class WordFinderApp:
    def __init__(self, root):
        self.root = root
        self.root.title("Word Finder")
        self.label = tk.Label(root, text="Enter a string:")
        self.label.pack(pady=10)
        self.text_input = tk.Text(root, height=10, width=50)
        self.text_input.pack(pady=10)
        self.find_button = tk.Button(root, text="Find Words", command=self.find_words)
        self.find_button.pack(pady=10)
        self.results_label = tk.Label(root, text="Found Words:")
        self.results_label.pack(pady=10)
        self.results_text = scrolledtext.ScrolledText(root, height=10, width=50)
        self.results_text.pack(pady=10)
    def find_words(self):
        input_text = self.text_input.get("1.0", tk.END).strip()
        if not input_text:
            messagebox.showwarning("Input Error", "Please enter a string.")
            return
        words = find_long_words(input_text)
        self.display_results(words)
    def display_results(self, words):
        self.results_text.delete("1.0", tk.END)  # Clear previous results
        if words:
            self.results_text.insert(tk.END, "\n".join(words))
        else:
            self.results_text.insert(tk.END, "No words found.")
if __name__ == "__main__":
    root = tk.Tk()
    app = WordFinderApp(root)

'''
Main application file for the Word Finder GUI.

import tkinter as tk
from tkinter import scrolledtext, messagebox
from regex_util import find_long_words
class WordFinderApp:
    def __init__(self, root):
        self.root = root
        self.root.title("Word Finder")
        self.label = tk.Label(root, text="Enter a string:")
        self.label.pack(pady=10)
        self.text_input = tk.Text(root, height=10, width=50)
        self.text_input.pack(pady=10)
        self.find_button = tk.Button(root, text="Find Words", command=self.find_words)
        self.find_button.pack(pady=10)
        self.results_label = tk.Label(root, text="Found Words:")
        self.results_label.pack(pady=10)
        self.results_text = scrolledtext.ScrolledText(root, height=10, width=50)
        self.results_text.pack(pady=10)
    def find_words(self):
        input_text = self.text_input.get("1.0", tk.END).strip()
        if not input_text:
            messagebox.showwarning("Input Error", "Please enter a string.")
            return
        words = find_long_words(input_text)
        self.display_results(words)
    def display_results(self, words):
        self.results_text.delete("1.0", tk.END)  # Clear previous results
        if words:
            self.results_text.insert(tk.END, "\n".join(words))
        else:
            self.results_text.insert(tk.END, "No words found.")
if __name__ == "__main__":
    root = tk.Tk()
    app = WordFinderApp(root)
    root.mainloop()

'''
Utility module for regex operations to find long words.
'''
import re
def find_long_words(input_string):
    '''
    Finds all words in the input string that are at least 4 characters long.
    Args:
        input_string (str): The string to search for words.
    Returns:
        list: A list of words that are at least 4 characters long.
    '''
    pattern = r'\b\w{4,}\b'
    return re.findall(pattern, input_string)

The Word Finder application is designed to find all words that are at least 4 characters long in a given input string using regular expressions (regex). The application provides a simple graphical user interface (GUI) for users to input a string and view the words that meet the specified criteria.

1. **Enter a String**: Users can input a string in the text box provided.
