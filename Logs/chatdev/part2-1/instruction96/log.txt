'''
Main application file for the Frequency Counter GUI.
'''

import tkinter as tk
from tkinter import messagebox
from frequency_counter import FrequencyCounter

class FrequencyCounterApp:
    def __init__(self):
        self.root = tk.Tk()
        self.root.title("Frequency Counter")
        
        self.label = tk.Label(self.root, text="Enter a list of lists (e.g. [[1, 2], [2, 3], [1]])")
        self.label.pack()
        
        self.input_text = tk.Text(self.root, height=5, width=50)
        self.input_text.pack()
        
        self.calculate_button = tk.Button(self.root, text="Calculate Frequency", command=self.calculate_frequency)
        self.calculate_button.pack()
        
        self.result_label = tk.Label(self.root, text="")
        self.result_label.pack()
        
        self.root.mainloop()

    def calculate_frequency(self):
        try:
            input_data = eval(self.input_text.get("1.0", tk.END).strip())
            if not isinstance(input_data, list) or not all(isinstance(i, list) for i in input_data):
                raise ValueError("Input must be a list of lists.")
            counter = FrequencyCounter(input_data)
            result = counter.count_frequency()
            self.display_result(result)
        except Exception as e:
            messagebox.showerror("Error", str(e))

    def display_result(self, result):
        self.result_label.config(text=str(result))

if __name__ == "__main__":
    FrequencyCounterApp()

import tkinter as tk
from tkinter import messagebox
import json
from frequency_counter import FrequencyCounter

class FrequencyCounterApp:
    def __init__(self):
        self.root = tk.Tk()
        self.root.title("Frequency Counter")
        self.label = tk.Label(self.root, text="Enter a list of lists (e.g. [[1, 2], [2, 3], [1]])")
        self.label.pack()
        self.input_text = tk.Text(self.root, height=5, width=50)
        self.input_text.pack()
        self.calculate_button = tk.Button(self.root, text="Calculate Frequency", command=self.calculate_frequency)
        self.calculate_button.pack()
        self.result_label = tk.Label(self.root, text="")
        self.result_label.pack()
        self.root.mainloop()

    def calculate_frequency(self):
        try:
            input_data = json.loads(self.input_text.get("1.0", tk.END).strip())
            if not isinstance(input_data, list) or not all(isinstance(i, list) for i in input_data):
                raise ValueError("Input must be a list of lists.")
            counter = FrequencyCounter(input_data)
            result = counter.count_frequency()
            self.display_result(result)
        except json.decoder.JSONDecodeError:
            messagebox.showerror("Error", "Invalid JSON format. Please enter a valid list of lists.")
        except ValueError as ve:
            messagebox.showerror("Error", str(ve))
        except Exception as e:
            messagebox.showerror("Error", str(e))

    def display_result(self, result):
        self.result_label.config(text=str(result))

if __name__ == "__main__":
    FrequencyCounterApp()

import tkinter as tk
from tkinter import messagebox
import json
from frequency_counter import FrequencyCounter
class FrequencyCounterApp:
    def __init__(self):
        self.root = tk.Tk()
        self.root.title("Frequency Counter")
        self.label = tk.Label(self.root, text="Enter a list of lists (e.g. [[1, 2], [2, 3], [1]])")
        self.label.pack()
        self.input_text = tk.Text(self.root, height=5, width=50)
        self.input_text.pack()
        self.calculate_button = tk.Button(self.root, text="Calculate Frequency", command=self.calculate_frequency)
        self.calculate_button.pack()
        self.result_label = tk.Label(self.root, text="")
        self.result_label.pack()
        self.root.mainloop()
    def calculate_frequency(self):
        try:
            input_data = json.loads(self.input_text.get("1.0", tk.END).strip())
            if not isinstance(input_data, list) or not all(isinstance(i, list) for i in input_data):
                raise ValueError("Input must be a list of lists.")
            counter = FrequencyCounter(input_data)
            result = counter.count_frequency()
            self.display_result(result)
        except json.decoder.JSONDecodeError:
            messagebox.showerror("Error", "Invalid JSON format. Please enter a valid list of lists.")
        except ValueError as ve:
            messagebox.showerror("Error", str(ve))
        except Exception as e:
            messagebox.showerror("Error", str(e))
    def display_result(self, result):
        self.result_label.config(text=str(result))
if __name__ == "__main__":
    FrequencyCounterApp()

'''
Utility file for counting frequency of elements in a list of lists.
'''
class FrequencyCounter:
    def __init__(self, data):
        self.data = data
    def count_frequency(self):
        frequency = {}
        for sublist in self.data:
            for item in sublist:
                if item in frequency:
                    frequency[item] += 1
                else:
                    frequency[item] = 1
        return frequency
