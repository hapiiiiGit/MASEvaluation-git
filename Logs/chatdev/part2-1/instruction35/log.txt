'''
Module to find the nth digit in the proper fraction of two given numbers.
'''

class FractionFinder:
    '''
    Class to calculate the proper fraction and find the nth digit.
    '''

    def __init__(self, numerator, denominator):
        self.numerator = numerator
        self.denominator = denominator

    def proper_fraction(self):
        '''
        Computes the proper fraction as a string.
        '''
        if self.denominator == 0:
            raise ValueError("Denominator cannot be zero.")
        
        # Get the integer part and the fractional part
        integer_part = self.numerator // self.denominator
        fractional_part = self.numerator % self.denominator

        # Create the proper fraction string
        fraction_str = str(integer_part) + '.'
        fractional_digits = []

        # Perform long division to get the fractional part
        seen_remainders = {}
        index = 0

        while fractional_part != 0:
            if fractional_part in seen_remainders:
                break  # Repeat detected
            seen_remainders[fractional_part] = index

            fractional_part *= 10
            fractional_digits.append(fractional_part // self.denominator)
            fractional_part = fractional_part % self.denominator
            index += 1

        fraction_str += ''.join(map(str, fractional_digits))
        return fraction_str

    def nth_digit(self, n):
        '''
        Finds the nth digit in the proper fraction.
        '''
        fraction_str = self.proper_fraction()
        # Check if n is valid
        if n < 1 or n >= len(fraction_str):
            return None  # N is out of bounds
        return fraction_str[n]
```

class FractionFinder:
    '''
    Class to calculate the proper fraction and find the nth digit.
    '''
    def __init__(self, numerator, denominator):
        self.numerator = numerator
        self.denominator = denominator
    def proper_fraction(self):
        '''
        Computes the proper fraction as a string.
        '''
        if self.denominator == 0:
            raise ValueError("Denominator cannot be zero.")
        # Get the integer part and the fractional part
        integer_part = self.numerator // self.denominator
        fractional_part = self.numerator % self.denominator
        # Create the proper fraction string
        fraction_str = str(integer_part) + '.'
        fractional_digits = []
        # Perform long division to get the fractional part
        seen_remainders = {}
        index = 0
        while fractional_part != 0:
            if fractional_part in seen_remainders:
                break  # Repeat detected
            seen_remainders[fractional_part] = index
            fractional_part *= 10
            fractional_digits.append(fractional_part // self.denominator)
            fractional_part = fractional_part % self.denominator
            index += 1
        fraction_str += ''.join(map(str, fractional_digits))
        return fraction_str
    def nth_digit(self, n):
        '''
        Finds the nth digit in the proper fraction.
        '''
        fraction_str = self.proper_fraction()
        # Check if n is valid
        if n < 1 or n >= len(fraction_str):

class FractionFinder:
    '''
    Class to calculate the proper fraction and find the nth digit.
    '''
    def __init__(self, numerator, denominator):
        self.numerator = numerator
        self.denominator = denominator
    def proper_fraction(self):
        '''
        Computes the proper fraction as a string.
        '''
        if self.denominator == 0:
            raise ValueError("Denominator cannot be zero.")
        # Get the integer part and the fractional part
        integer_part = self.numerator // self.denominator
        fractional_part = self.numerator % self.denominator
        # Create the proper fraction string
        fraction_str = str(integer_part) + '.'
        fractional_digits = []
        # Perform long division to get the fractional part
        seen_remainders = {}
        index = 0
        while fractional_part != 0:
            if fractional_part in seen_remainders:
                break  # Repeat detected
            seen_remainders[fractional_part] = index
            fractional_part *= 10
            fractional_digits.append(fractional_part // self.denominator)
            fractional_part = fractional_part % self.denominator
            index += 1
        fraction_str += ''.join(map(str, fractional_digits))
        return fraction_str
    def nth_digit(self, n):
        '''
        Finds the nth digit in the proper fraction.
        '''
        fraction_str = self.proper_fraction()
        # Check if n is valid
        if n < 1 or n >= len(fraction_str):
            return None  # N is out of bounds
        return fraction_str[n]

class FractionFinder:
    '''
    Class to calculate the proper fraction and find the nth digit.
    '''
    def __init__(self, numerator, denominator):
        self.numerator = numerator
        self.denominator = denominator

    def proper_fraction(self):
        '''
        Computes the proper fraction as a string.
        '''
        if self.denominator == 0:
            raise ValueError("Denominator cannot be zero.")
        # Get the integer part and the fractional part
        integer_part = self.numerator // self.denominator
        fractional_part = self.numerator % self.denominator
        # Create the proper fraction string
        fraction_str = str(integer_part) + '.'
        fractional_digits = []
        # Perform long division to get the fractional part
        seen_remainders = {}
        index = 0
        while fractional_part != 0:
            if fractional_part in seen_remainders:
                break  # Repeat detected
            seen_remainders[fractional_part] = index
            fractional_part *= 10
            fractional_digits.append(fractional_part // self.denominator)
            fractional_part = fractional_part % self.denominator
            index += 1
        fraction_str += ''.join(map(str, fractional_digits))
        return fraction_str

    def nth_digit(self, n):
        '''
        Finds the nth digit in the proper fraction.
        '''
        fraction_str = self.proper_fraction()
        # Check if n is valid
        if n < 1 or n >= len(fraction_str):
            return None  # N is out of bounds
        return fraction_str[n]

class FractionFinder:
    '''
    Class to calculate the proper fraction and find the nth digit.
    '''
    def __init__(self, numerator, denominator):
        self.numerator = numerator
        self.denominator = denominator

    def proper_fraction(self):
        '''
        Computes the proper fraction as a string.
        '''
        if self.denominator == 0:
            raise ValueError("Denominator cannot be zero.")
        # Get the integer part and the fractional part
        integer_part = self.numerator // self.denominator
        fractional_part = self.numerator % self.denominator
        # Create the proper fraction string
        fraction_str = str(integer_part) + '.'
        fractional_digits = []
        # Perform long division to get the fractional part
        seen_remainders = {}
        index = 0
        while fractional_part != 0:
            if fractional_part in seen_remainders:
                break  # Repeat detected
            seen_remainders[fractional_part] = index
            fractional_part *= 10
            fractional_digits.append(fractional_part // self.denominator)
            fractional_part = fractional_part % self.denominator
            index += 1
        fraction_str += ''.join(map(str, fractional_digits))
        return fraction_str

    def nth_digit(self, n):
        '''
        Finds the nth digit in the proper fraction.
        '''
        fraction_str = self.proper_fraction()
        # Split the fraction string to get the fractional part
        fractional_part = fraction_str.split('.')[1] if '.' in fraction_str else ''
        # Check if n is valid
        if n < 1 or n > len(fractional_part):
            return None  # N is out of bounds
        return fractional_part[n - 1]  # Adjust for zero-based index

'''
Module to find the nth digit in the proper fraction of two given numbers.
'''
class FractionFinder:
    '''
    Class to calculate the proper fraction and find the nth digit.
    '''
    def __init__(self, numerator, denominator):
        self.numerator = numerator
        self.denominator = denominator
    def proper_fraction(self):
        '''
        Computes the proper fraction as a string.
        '''
        if self.denominator == 0:
            raise ValueError("Denominator cannot be zero.")
        # Get the integer part and the fractional part
        integer_part = self.numerator // self.denominator
        fractional_part = self.numerator % self.denominator
        # Create the proper fraction string
        fraction_str = str(integer_part) + '.'
        fractional_digits = []
        # Perform long division to get the fractional part
        seen_remainders = {}
        index = 0
        while fractional_part != 0:
            if fractional_part in seen_remainders:
                break  # Repeat detected
            seen_remainders[fractional_part] = index
            fractional_part *= 10
            fractional_digits.append(fractional_part // self.denominator)
            fractional_part = fractional_part % self.denominator
            index += 1
        fraction_str += ''.join(map(str, fractional_digits))
        return fraction_str
    def nth_digit(self, n):
        '''
        Finds the nth digit in the proper fraction.
        '''
        fraction_str = self.proper_fraction()
        # Split the fraction string to get the fractional part
        fractional_part = fraction_str.split('.')[1] if '.' in fraction_str else ''
        # Check if n is valid
        if n < 1 or n > len(fractional_part):
            return None  # N is out of bounds
        return fractional_part[n - 1]  # Adjust for zero-based index

from tkinter import Tk, Label, Entry, Button, StringVar
from fraction_finder import FractionFinder
class App:
    '''
    Class to create the GUI for the Fraction Finder application.
    '''
    def __init__(self, master):
        self.master = master
        master.title("Nth Digit Finder in Proper Fraction")
        self.numerator_label = Label(master, text="Numerator:")
        self.numerator_label.pack()
        self.numerator_entry = Entry(master)
        self.numerator_entry.pack()
        self.denominator_label = Label(master, text="Denominator:")
        self.denominator_label.pack()
        self.denominator_entry = Entry(master)
        self.denominator_entry.pack()
        self.nth_digit_label = Label(master, text="Nth Digit:")
        self.nth_digit_label.pack()
        self.nth_digit_entry = Entry(master)
        self.nth_digit_entry.pack()
        self.result_var = StringVar()
        self.result_label = Label(master, textvariable=self.result_var)
        self.result_label.pack()
        self.calculate_button = Button(master, text="Find Nth Digit", command=self.calculate)
        self.calculate_button.pack()
    def calculate(self):
        '''
        Handles the button click event to perform calculations and display results.
        '''
        try:
            numerator = int(self.numerator_entry.get())
            denominator = int(self.denominator_entry.get())
            n = int(self.nth_digit_entry.get())
            finder = FractionFinder(numerator, denominator)
            digit = finder.nth_digit(n)
            if digit is not None:
                self.result_var.set(f"The {n}th digit is: {digit}")
            else:
                self.result_var.set("Error: N is too large.")
        except ValueError:
            self.result_var.set("Error: Please enter valid integers.")
if __name__ == "__main__":
    root = Tk()
    app = App(root)
    root.mainloop()

'''
Module to find the nth digit in the proper fraction of two given numbers.
'''
class FractionFinder:
    '''
    Class to calculate the proper fraction and find the nth digit.
    '''
    def __init__(self, numerator, denominator):
        self.numerator = numerator
        self.denominator = denominator
    def proper_fraction(self):
        '''
        Computes the proper fraction as a string.
        '''
        if self.denominator == 0:
            raise ValueError("Denominator cannot be zero.")
        # Get the integer part and the fractional part
        integer_part = self.numerator // self.denominator
        fractional_part = self.numerator % self.denominator
        # Create the proper fraction string
        fraction_str = str(integer_part) + '.'
        fractional_digits = []
        # Perform long division to get the fractional part
        seen_remainders = {}
        index = 0
        while fractional_part != 0:
            if fractional_part in seen_remainders:
                break  # Repeat detected
            seen_remainders[fractional_part] = index
            fractional_part *= 10
            fractional_digits.append(fractional_part // self.denominator)
            fractional_part = fractional_part % self.denominator
            index += 1
        fraction_str += ''.join(map(str, fractional_digits))
        return fraction_str
    def nth_digit(self, n):
        '''
        Finds the nth digit in the proper fraction.
        '''
        fraction_str = self.proper_fraction()
        # Split the fraction string to get the fractional part
        fractional_part = fraction_str.split('.')[1] if '.' in fraction_str else ''
        # Check if n is valid
        if n < 1 or n > len(fractional_part):
            return None  # N is out of bounds
        return fractional_part[n - 1]  # Adjust for zero-based index
