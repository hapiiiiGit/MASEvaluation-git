class PatternChecker:

    '''
    def check_pattern(self, sequence, patterns):
        '''
        Checks if the sequence follows the patterns.
        Args:
            sequence (list): A list of elements in the sequence.
            patterns (list): A list of patterns to check against.
        Returns:
            bool: True if the sequence follows the patterns, False otherwise.
        '''
        if not sequence and not patterns:
            return True

        if len(sequence) != len(patterns):
            return False

        for seq_item, pattern in zip(sequence, patterns):
            if seq_item.strip() != pattern.strip():
                return False

        return True

'''

import tkinter as tk
from tkinter import messagebox
from pattern_checker import PatternChecker
class App:
    '''
    GUI application for checking sequences against patterns.
    '''
    def __init__(self, master):
        self.master = master
        master.title("Pattern Checker")
        self.label = tk.Label(master, text="Enter a sequence (comma-separated):")
        self.label.pack()

        self.patterns_label.pack()
        self.patterns_entry = tk.Entry(master)
        self.patterns_entry.pack()
        self.check_button = tk.Button(master, text="Check Pattern", command=self.check_sequence)
        self.check_button.pack()
    def check_sequence(self):
        '''
        Retrieves input from the user and checks the sequence against the patterns.
        '''
        sequence = self.sequence_entry.get().split(',')

            messagebox.showwarning("Result", "The sequence does NOT follow the patterns.")
if __name__ == "__main__":
    root = tk.Tk()
    app = App(root)
    root.mainloop()

'''
Module containing the PatternChecker class to validate sequences against patterns.
'''
class PatternChecker:
    '''
    Class to check if a sequence follows the given patterns.
    '''
    def check_pattern(self, sequence, patterns):
        '''
        Checks if the sequence follows the patterns.
        Args:
            sequence (list): A list of elements in the sequence.
            patterns (list): A list of patterns to check against.
        Returns:
            bool: True if the sequence follows the patterns, False otherwise.
        '''
        if not sequence and not patterns:
            return True
        if len(sequence) < len(patterns):  # Check if patterns list is longer than sequence list
            return False
        for seq_item, pattern in zip(sequence, patterns):
            if seq_item.strip() != pattern.strip():
                return False
        return True

import tkinter as tk
from tkinter import messagebox
from pattern_checker import PatternChecker
class App:
    '''
    GUI application for checking sequences against patterns.
    '''
    def __init__(self, master):
        self.master = master
        master.title("Pattern Checker")
        self.label = tk.Label(master, text="Enter a sequence (comma-separated):")
        self.label.pack()
        self.sequence_entry = tk.Entry(master)
        self.sequence_entry.pack()
        self.patterns_label = tk.Label(master, text="Enter patterns (comma-separated):")
        self.patterns_label.pack()
        self.patterns_entry = tk.Entry(master)
        self.patterns_entry.pack()
        self.check_button = tk.Button(master, text="Check Pattern", command=self.check_sequence)
        self.check_button.pack()
    def check_sequence(self):
        '''
        Retrieves input from the user and checks the sequence against the patterns.
        '''
        sequence = self.sequence_entry.get().split(',')
        patterns = self.patterns_entry.get().split(',')
        checker = PatternChecker()
        result = checker.check_pattern(sequence, patterns)
        if result:
            messagebox.showinfo("Result", "The sequence follows the patterns.")
        else:
            messagebox.showwarning("Result", "The sequence does NOT follow the patterns.")
if __name__ == "__main__":

'''
Module containing the PatternChecker class to validate sequences against patterns.
'''
class PatternChecker:
    '''
    Class to check if a sequence follows the given patterns.
    '''
    def check_pattern(self, sequence, patterns):
        '''
        Checks if the sequence follows the patterns.
        Args:
            sequence (list): A list of elements in the sequence.
            patterns (list): A list of patterns to check against.
        Returns:
            bool: True if the sequence follows the patterns, False otherwise.
        '''
        if not sequence and not patterns:
            return True
        if len(sequence) < len(patterns):  # Check if patterns list is longer than sequence list
            return False
        for seq_item, pattern in zip(sequence, patterns):
            if seq_item.strip() != pattern.strip():
                return False
        return True
