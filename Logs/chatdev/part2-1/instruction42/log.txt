import re

def find_lowercase_sequences(input_string):
    '''
    Finds sequences of lowercase letters joined with underscores in the input string.
    
    Parameters:
        input_string (str): The string to search for patterns.
    
    Returns:
        list: A list of matched sequences.
    '''
    pattern = r'\b[a-z]+(?:_[a-z]+)*\b'
    matches = re.findall(pattern, input_string)
    return matches

'''
Utility file containing regex functions for finding patterns.
'''
import re
def find_lowercase_sequences(input_string):
    '''
    Finds sequences of lowercase letters joined with underscores in the input string.
    Parameters:
        input_string (str): The string to search for patterns.
    Returns:
        list: A list of matched sequences.
    '''
    pattern = r'\b[a-z]+(?:_[a-z]+)*\b'
    matches = re.findall(pattern, input_string)
    return matches

'''

import re

def find_lowercase_sequences(input_string):
    '''
    Finds sequences of lowercase letters joined with underscores in the input string.
    Parameters:
        input_string (str): The string to search for patterns.
    Returns:
        list: A list of matched sequences.
    '''
    if not input_string.strip():  # Check if input string is empty or contains only whitespace
        return []

    pattern = r'[a-z]+(?:_[a-z]+)*'
    matches = re.findall(pattern, input_string)
    return matches

import re

Main application file for the Regex Finder GUI.

import tkinter as tk
from tkinter import scrolledtext, messagebox
from regex_util import find_lowercase_sequences
class RegexFinderApp:
    '''
    Main application class to create the GUI for finding regex patterns.
    '''
    def __init__(self, master):
        self.master = master
        master.title("Regex Finder")
        self.label = tk.Label(master, text="Enter text:")
        self.label.pack()
        self.text_area = scrolledtext.ScrolledText(master, width=40, height=10)
        self.text_area.pack()
        self.find_button = tk.Button(master, text="Find Sequences", command=self.find_sequences)
        self.find_button.pack()
        self.result_label = tk.Label(master, text="Results:")
        self.result_label.pack()
        self.result_area = scrolledtext.ScrolledText(master, width=40, height=10, state='disabled')
        self.result_area.pack()
    def find_sequences(self):
        '''
        Finds sequences of lowercase letters joined with underscores in the input text.
        '''
        input_text = self.text_area.get("1.0", tk.END)
        results = find_lowercase_sequences(input_text)
        self.result_area.config(state='normal')
        self.result_area.delete("1.0", tk.END)  # Clear previous results
        if results:
            self.result_area.insert(tk.END, "\n".join(results))
        else:
            self.result_area.insert(tk.END, "No matches found.")
        self.result_area.config(state='disabled')
if __name__ == "__main__":
    root = tk.Tk()
    app = RegexFinderApp(root)
    root.mainloop()

'''
Utility file containing regex functions for finding patterns.
'''
import re
def find_lowercase_sequences(input_string):
    '''
    Finds sequences of lowercase letters joined with underscores in the input string.
    Parameters:
        input_string (str): The string to search for patterns.
    Returns:
        list: A list of matched sequences.
    '''
    if not input_string.strip():  # Check if input string is empty or contains only whitespace
        return []
    pattern = r'[a-z]+(?:_[a-z]+)*'
    matches = re.findall(pattern, input_string)
    return matches
