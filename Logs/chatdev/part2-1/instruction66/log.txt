'''
Module to calculate Bell numbers and the number of partitions.

class BellNumberCalculator:
    def bell_number(self, n):
        """Compute the nth Bell number using dynamic programming."""
        bell = [[0 for _ in range(n + 1)] for _ in range(n + 1)]
        bell[0][0] = 1

        for i in range(1, n + 1):
            bell[i][0] = bell[i - 1][i - 1]
            for j in range(1, i + 1):
                bell[i][j] = bell[i - 1][j - 1] + bell[i][j - 1]

        return bell[n][0]

    def partition_count(self, n):
        """Compute the number of ways to partition the set of Bell numbers up to n."""
        if n < 0:
            return 0
        return self.bell_number(n)

Module to calculate Bell numbers and the number of partitions.

class BellNumberCalculator:
    def bell_number(self, n):
        """Compute the nth Bell number using dynamic programming."""
        bell = [[0 for _ in range(n + 1)] for _ in range(n + 1)]
        bell[0][0] = 1
        for i in range(1, n + 1):
            bell[i][0] = bell[i - 1][i - 1]
            for j in range(1, i + 1):
                bell[i][j] = bell[i - 1][j - 1] + bell[i][j - 1]
        return bell[n][0]
    def partition_count(self, n):
        """Compute the number of ways to partition the set of Bell numbers up to n."""
        if n < 0:
            return 0
        return self.bell_number(n)

    def bell_number(self, n):
        """Compute the nth Bell number using dynamic programming."""
        bell = [[0 for _ in range(n + 1)] for _ in range(n + 1)]
        bell[0][0] = 1
        for i in range(1, n + 1):
            bell[i][0] = bell[i - 1][i - 1]
            for j in range(1, i + 1):
                bell[i][j] = bell[i - 1][j - 1] + bell[i][j - 1]
        return bell[n][0]
    def partition_count(self, n):
        """Compute the number of ways to partition the set of Bell numbers up to n."""
        if n < 0:
            return 0
        return self.bell_number(n)

'''
Module to calculate Bell numbers and the number of partitions.

import tkinter as tk
class BellNumberCalculator:
    def bell_number(self, n):
        """Compute the nth Bell number using dynamic programming."""
        bell = [[0 for _ in range(n + 1)] for _ in range(n + 1)]
        bell[0][0] = 1
        for i in range(1, n + 1):
            bell[i][0] = bell[i - 1][i - 1]
            for j in range(1, i + 1):
                bell[i][j] = bell[i - 1][j - 1] + bell[i][j - 1]
        return bell[n][0]
    def partition_count(self, n):
        """Compute the number of ways to partition the set of Bell numbers up to n."""
        if n < 0:
            return 0
        partitions = [0 for _ in range(n + 1)]
        partitions[0] = 1
        for i in range(1, n + 1):
            partitions[i] = 0
            for j in range(i):
                partitions[i] += self.bell_number(j) * partitions[i - j - 1]
        return partitions[n]

class BellNumberCalculator:
    def bell_number(self, n):
        """Compute the nth Bell number using dynamic programming."""
        bell = [[0 for _ in range(n + 1)] for _ in range(n + 1)]
        bell[0][0] = 1
        for i in range(1, n + 1):
            bell[i][0] = bell[i - 1][i - 1]
            for j in range(1, i + 1):
                bell[i][j] = bell[i - 1][j - 1] + bell[i][j - 1]
        return bell[n][0]
    def partition_count(self, n):
        """Compute the number of ways to partition the set of Bell numbers up to n."""
        if n < 0:
            return 0
        partitions = [0 for _ in range(n + 1)]
        partitions[0] = 1
        for i in range(1, n + 1):
            partitions[i] = 0
            for j in range(i):
                partitions[i] += self.bell_number(j) * partitions[i - j - 1]
        return partitions[n]

'''
Module to calculate Bell numbers and the number of partitions.
'''
class BellNumberCalculator:
    def bell_number(self, n):
        """Compute the nth Bell number using dynamic programming."""
        bell = [[0 for _ in range(n + 1)] for _ in range(n + 1)]
        bell[0][0] = 1
        for i in range(1, n + 1):
            bell[i][0] = bell[i - 1][i - 1]
            for j in range(1, i + 1):
                bell[i][j] = bell[i - 1][j - 1] + bell[i][j - 1]
        return bell[n][0]
    def partition_count(self, n):
        """Compute the number of ways to partition the set of Bell numbers up to n."""
        if n < 0:
            return 0
        partitions = [0 for _ in range(n + 1)]
        partitions[0] = 1
        for i in range(1, n + 1):
            partitions[i] = 0
            for j in range(i):
                partitions[i] += self.bell_number(j) * partitions[i - j - 1]
        return partitions[n]

'''
Module to calculate Bell numbers and the number of partitions.
'''
class BellNumberCalculator:
    def bell_number(self, n):
        """Compute the nth Bell number using dynamic programming."""
        bell = [[0 for _ in range(n + 1)] for _ in range(n + 1)]
        bell[0][0] = 1
        for i in range(1, n + 1):
            bell[i][0] = bell[i - 1][i - 1]
            for j in range(1, i + 1):
                bell[i][j] = bell[i - 1][j - 1] + bell[i][j - 1]
        return bell[n][0]
    def partition_count(self, n):
        """Compute the number of ways to partition the set of Bell numbers up to n."""
        if n < 0:
            return 0
        partitions = [0 for _ in range(n + 1)]
        partitions[0] = 1
        for i in range(1, n + 1):
            partitions[i] = 0
            for j in range(i):
                partitions[i] += self.bell_number(j) * partitions[i - j - 1]
        return partitions[n]
