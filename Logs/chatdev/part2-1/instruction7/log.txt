import tkinter as tk

class GUI:
    def __init__(self, square_finder):
        self.square_finder = square_finder
        self.root = tk.Tk()
        self.root.title("Square Finder")

        self.input_label = tk.Label(self.root, text="Enter numbers separated by commas:")
        self.input_label.pack()

        self.input_entry = tk.Entry(self.root)
        self.input_entry.pack()

        self.result_label = tk.Label(self.root, text="Squares:")
        self.result_label.pack()

        self.result_text = tk.Text(self.root, height=10, width=50)
        self.result_text.pack()

        self.calculate_button = tk.Button(self.root, text="Calculate", command=self.calculate_squares)
        self.calculate_button.pack()

    def calculate_squares(self):
        input_numbers = list(map(int, self.input_entry.get().split(',')))
        squares = self.square_finder.find_squares(input_numbers)
        self.result_text.delete(1.0, tk.END)
        self.result_text.insert(tk.END, '\n'.join(map(str, squares)))

    def run(self):
        self.root.mainloop()

class GUI:
    def __init__(self, square_finder):

'''
This file contains the GUI class responsible for the graphical user interface.
'''
import tkinter as tk
from tkinter import messagebox
class GUI:
    def __init__(self, square_finder):
        self.square_finder = square_finder
        self.root = tk.Tk()
        self.root.title("Square Finder")
        self.input_label = tk.Label(self.root, text="Enter numbers separated by commas:")
        self.input_label.pack()
        self.input_entry = tk.Entry(self.root)
        self.input_entry.pack()
        self.result_label = tk.Label(self.root, text="Squares:")
        self.result_label.pack()
        self.result_text = tk.Text(self.root, height=10, width=50)
        self.result_text.pack()
        self.calculate_button = tk.Button(self.root, text="Calculate", command=self.calculate_squares)
        self.calculate_button.pack()
    def calculate_squares(self):
        input_str = self.input_entry.get()
        if not input_str:
            # Handle empty input
            messagebox.showerror("Error", "Input cannot be empty. Please enter numbers separated by commas.")
            return
        try:
            input_numbers = list(map(int, input_str.split(',')))
            squares = self.square_finder.find_squares(input_numbers)
            self.result_text.delete(1.0, tk.END)
            self.result_text.insert(tk.END, '\n'.join(map(str, squares)))
        except ValueError:
            # Handle invalid input (non-numeric values)
            messagebox.showerror("Error", "Please enter valid numbers separated by commas.")
    def run(self):
        self.root.mainloop()
