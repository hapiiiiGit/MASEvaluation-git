def remove_first_last_occurrence(input_string, char_to_remove):
    first_index = input_string.find(char_to_remove)
    last_index = input_string.rfind(char_to_remove)

    if first_index != -1 and last_index != -1:
        return input_string[:first_index] + input_string[first_index+1:last_index] + input_string[last_index+1:]
    else:

'''
This file contains the main application logic for removing the first and last occurrence of a given character from a string.
'''
class RemoveCharacterApp:
    def __init__(self):
        pass
    def remove_first_and_last_occurrence(self, input_string, char_to_remove):
        first_index = input_string.find(char_to_remove)
        last_index = input_string.rfind(char_to_remove)
        if first_index != -1 and last_index != -1:
            if first_index != last_index:
                modified_string = input_string[:first_index] + input_string[first_index+1:last_index] + input_string[last_index+1:]
                return modified_string
            else:
                return "Character occurs only once in the string."
        else:
            return "Character not found in the string."

class RemoveCharacterApp:
    def __init__(self):

    def remove_first_and_last_occurrence(self, input_string, char_to_remove):

# Remove Character App User Manual

## Introduction
Welcome to the Remove Character App user manual. This application allows you to remove the first and last occurrence of a given character from a string. This manual will guide you on how to install the application and how to use it effectively.

## Main Functions
The main function of the Remove Character App is to remove the first and last occurrence of a specified character from a string. The application provides a graphical user interface (GUI) for user interaction.

## Installation
To install the Remove Character App, follow these steps:

1. Clone the repository containing the application files.
2. Ensure you have Python installed on your system.
3. Install the required dependencies by running the following command:
   ```
   pip install tkinter
   ```
4. Run the `main.py` file to launch the GUI of the application.

## How to Use
Once you have installed the application, follow these steps to use it:

1. Enter a string in the "Enter a string:" field on the GUI.
2. Enter the character you want to remove in the "Enter a character to remove:" field.
3. Click on the "Remove Character" button.
4. The modified string, after removing the first and last occurrence of the specified character, will be displayed below.

## Example
For example, if you enter the string "hello world" and specify the character "o" to remove, the modified string will be "hell world".

## Troubleshooting
If you encounter any issues while using the Remove Character App, please ensure that you have entered valid input strings and characters. If the character is not found in the string, the application will display a message indicating that the character was not found.

For further assistance or inquiries, please contact our support team at support@chatdev.com.

Thank you for using the Remove Character App!
