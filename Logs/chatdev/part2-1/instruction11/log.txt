'''
This module contains classes to handle matrix sorting and GUI display.

import tkinter as tk
from tkinter import messagebox

class MatrixSorterApp:
    '''
    Main application class for sorting a matrix.
    '''

    def __init__(self):
        self.root = tk.Tk()
        self.root.title("Matrix Sorter")
        
        self.input_frame = MatrixInputFrame(self.root, self.sort_matrix)
        self.input_frame.pack(pady=10)

        self.display_frame = MatrixDisplayFrame(self.root)
        self.display_frame.pack(pady=10)

    def sort_matrix(self):
        '''
        Sorts the matrix based on the sum of its rows and updates the display.
        '''
        matrix = self.input_frame.get_matrix()
        if matrix is None:
            return
        
        sorted_matrix = sorted(matrix, key=sum)
        self.display_frame.update_display(sorted_matrix)

    def run(self):
        '''
        Runs the main loop of the application.
        '''
        self.root.mainloop()


class MatrixInputFrame(tk.Frame):
    '''
    Frame for inputting the matrix.
    '''

    def __init__(self, master, sort_callback):
        super().__init__(master)
        self.sort_callback = sort_callback
        
        self.label = tk.Label(self, text="Enter matrix (comma-separated rows):")
        self.label.pack()

        self.text_area = tk.Text(self, height=10, width=30)
        self.text_area.pack()

        self.sort_button = tk.Button(self, text="Sort Matrix", command=self.sort_matrix)
        self.sort_button.pack()

    def get_matrix(self):
        '''
        Retrieves the matrix input from the user.
        '''
        try:
            input_text = self.text_area.get("1.0", tk.END).strip()
            matrix = [list(map(int, row.split(','))) for row in input_text.splitlines() if row]
            return matrix
        except ValueError:
            messagebox.showerror("Input Error", "Please enter a valid matrix.")
            return None

    def sort_matrix(self):
        '''
        Calls the sort callback function.
        '''
        self.sort_callback()


class MatrixDisplayFrame(tk.Frame):
    '''
    Frame for displaying the sorted matrix.
    '''

    def __init__(self, master):
        super().__init__(master)
        self.label = tk.Label(self, text="Sorted Matrix:")
        self.label.pack()

        self.display_area = tk.Text(self, height=10, width=30)
        self.display_area.pack()

    def update_display(self, sorted_matrix):
        '''
        Updates the display with the sorted matrix.
        '''
        self.display_area.delete("1.0", tk.END)
        for row in sorted_matrix:
            self.display_area.insert(tk.END, ', '.join(map(str, row)) + '\n')

import tkinter as tk
from tkinter import messagebox
class MatrixSorterApp:
    '''
    Main application class for sorting a matrix.
    '''
    def __init__(self):
        self.root = tk.Tk()
        self.root.title("Matrix Sorter")
        self.input_frame = MatrixInputFrame(self.root, self.sort_matrix)
        self.input_frame.pack(pady=10)
        self.display_frame = MatrixDisplayFrame(self.root)
        self.display_frame.pack(pady=10)
    def sort_matrix(self):
        '''
        Sorts the matrix based on the sum of its rows and updates the display.
        '''
        matrix = self.input_frame.get_matrix()
        if matrix is None:
            return
        sorted_matrix = sorted(matrix, key=sum)
        self.display_frame.update_display(sorted_matrix)
    def run(self):
        '''
        Runs the main loop of the application.
        '''
        self.root.mainloop()
class MatrixInputFrame(tk.Frame):
    '''
    Frame for inputting the matrix.
    '''
    def __init__(self, master, sort_callback):
        super().__init__(master)
        self.sort_callback = sort_callback
        self.label = tk.Label(self, text="Enter matrix (comma-separated rows):")
        self.label.pack()
        self.text_area = tk.Text(self, height=10, width=30)
        self.text_area.pack()
        self.sort_button = tk.Button(self, text="Sort Matrix", command=self.sort_matrix)
        self.sort_button.pack()
    def get_matrix(self):
        '''
        Retrieves the matrix input from the user.
        '''
        try:
            input_text = self.text_area.get("1.0", tk.END).strip()
            matrix = [list(map(int, row.split(','))) for row in input_text.splitlines() if row]
            return matrix
        except ValueError:
            messagebox.showerror("Input Error", "Please enter a valid matrix.")
            return None
    def sort_matrix(self):
        '''
        Calls the sort callback function after validating the matrix input.
        '''
        matrix = self.get_matrix()
        if matrix is None or len(matrix) == 0:
            messagebox.showerror("Input Error", "Please enter a non-empty matrix.")
            return
        self.sort_callback()
class MatrixDisplayFrame(tk.Frame):
    '''
    Frame for displaying the sorted matrix.
    '''

        super().__init__(master)

        self.label.pack()
        self.display_area = tk.Text(self, height=10, width=30)

        '''

import tkinter as tk
from tkinter import messagebox
class MatrixSorterApp:
    '''
    Main application class for sorting a matrix.
    '''
    def __init__(self):
        self.root = tk.Tk()
        self.root.title("Matrix Sorter")
        self.input_frame = MatrixInputFrame(self.root, self.sort_matrix)
        self.input_frame.pack(pady=10)
        self.display_frame = MatrixDisplayFrame(self.root)
        self.display_frame.pack(pady=10)
    def sort_matrix(self):
        '''
        Sorts the matrix based on the sum of its rows and updates the display.
        '''
        matrix = self.input_frame.get_matrix()
        if matrix is None:
            return
        sorted_matrix = sorted(matrix, key=sum)
        self.display_frame.update_display(sorted_matrix)
    def run(self):
        '''
        Runs the main loop of the application.
        '''
        self.root.mainloop()
class MatrixInputFrame(tk.Frame):
    '''
    Frame for inputting the matrix.
    '''
    def __init__(self, master, sort_callback):
        super().__init__(master)
        self.sort_callback = sort_callback
        self.label = tk.Label(self, text="Enter matrix (comma-separated rows):")
        self.label.pack()
        self.text_area = tk.Text(self, height=10, width=30)
        self.text_area.pack()
        self.sort_button = tk.Button(self, text="Sort Matrix", command=self.sort_matrix)
        self.sort_button.pack()
    def get_matrix(self):
        '''
        Retrieves the matrix input from the user.
        '''
        try:
            input_text = self.text_area.get("1.0", tk.END).strip()
            matrix = [list(map(int, row.split(','))) for row in input_text.splitlines() if row]
            return matrix
        except ValueError:
            messagebox.showerror("Input Error", "Please enter a valid matrix.")
            return None
    def sort_matrix(self):
        '''
        Calls the sort callback function after validating the matrix input.
        '''
        matrix = self.get_matrix()
        if matrix is None or len(matrix) == 0:
            messagebox.showerror("Input Error", "Please enter a non-empty matrix.")
            return
        self.sort_callback()
class MatrixDisplayFrame(tk.Frame):

import tkinter as tk
from tkinter import messagebox
class MatrixSorterApp:
    '''
    Main application class for sorting a matrix.
    '''
    def __init__(self):
        self.root = tk.Tk()
        self.root.title("Matrix Sorter")
        self.input_frame = MatrixInputFrame(self.root, self.sort_matrix)
        self.input_frame.pack(pady=10)
        self.display_frame = MatrixDisplayFrame(self.root)
        self.display_frame.pack(pady=10)
    def sort_matrix(self):
        '''
        Sorts the matrix based on the sum of its rows and updates the display.
        '''
        matrix = self.input_frame.get_matrix()
        if matrix is None:
            return
        sorted_matrix = sorted(matrix, key=sum)
        self.display_frame.update_display(sorted_matrix)
    def run(self):
        '''
        Runs the main loop of the application.
        '''
        self.root.mainloop()
class MatrixInputFrame(tk.Frame):
    '''
    Frame for inputting the matrix.
    '''
    def __init__(self, master, sort_callback):
        super().__init__(master)
        self.sort_callback = sort_callback
        self.label = tk.Label(self, text="Enter matrix (comma-separated rows):")
        self.label.pack()
        self.text_area = tk.Text(self, height=10, width=30)
        self.text_area.pack()
        self.sort_button = tk.Button(self, text="Sort Matrix", command=self.sort_matrix)
        self.sort_button.pack()
    def get_matrix(self):
        '''
        Retrieves the matrix input from the user.
        '''
        try:
            input_text = self.text_area.get("1.0", tk.END).strip()
            matrix = [list(map(int, row.split(','))) for row in input_text.splitlines() if row]
            return matrix
        except ValueError:
            messagebox.showerror("Input Error", "Please enter a valid matrix.")
            return None
    def sort_matrix(self):
        '''
        Calls the sort callback function after validating the matrix input.
        '''
        matrix = self.get_matrix()
        if matrix is None or len(matrix) == 0:
            messagebox.showerror("Input Error", "Please enter a non-empty matrix.")
            return
        self.sort_callback()
class MatrixDisplayFrame(tk.Frame):
    '''
    Frame for displaying the sorted matrix.
    '''
    def __init__(self, master):
        super().__init__(master)
        self.label = tk.Label(self, text="Sorted Matrix:")
        self.label.pack()
        self.display_area = tk.Text(self, height=10, width=30)
        self.display_area.pack()
    def update_display(self, sorted_matrix):
        '''
        Updates the display with the sorted matrix.
        '''
        self.display_area.delete("1.0", tk.END)
        for row in sorted_matrix:
            self.display_area.insert(tk.END, ', '.join(map(str, row)) + '\n')

'''

import tkinter as tk
from tkinter import messagebox
class MatrixSorterApp:
    '''
    Main application class for sorting a matrix.
    '''
    def __init__(self):
        self.root = tk.Tk()
        self.root.title("Matrix Sorter")
        self.input_frame = MatrixInputFrame(self.root, self.sort_matrix)
        self.input_frame.pack(pady=10)
        self.display_frame = MatrixDisplayFrame(self.root)
        self.display_frame.pack(pady=10)
    def sort_matrix(self):
        '''
        Sorts the matrix based on the sum of its rows and updates the display.
        '''
        matrix = self.input_frame.get_matrix()
        if matrix is None:
            return
        sorted_matrix = sorted(matrix, key=sum)
        self.display_frame.update_display(sorted_matrix)
    def run(self):
        '''
        Runs the main loop of the application.
        '''
        self.root.mainloop()
class MatrixInputFrame(tk.Frame):
    '''
    Frame for inputting the matrix.
    '''
    def __init__(self, master, sort_callback):
        super().__init__(master)
        self.sort_callback = sort_callback
        self.label = tk.Label(self, text="Enter matrix (comma-separated rows):")
        self.label.pack()
        self.text_area = tk.Text(self, height=10, width=30)
        self.text_area.pack()
        self.sort_button = tk.Button(self, text="Sort Matrix", command=self.sort_matrix)
        self.sort_button.pack()
    def get_matrix(self):
        '''
        Retrieves the matrix input from the user.
        '''
        try:
            input_text = self.text_area.get("1.0", tk.END).strip()
            matrix = [list(map(int, row.split(','))) for row in input_text.splitlines() if row]
            return matrix
        except ValueError:
            messagebox.showerror("Input Error", "Please enter a valid matrix.")
            return None
    def sort_matrix(self):
        '''
        Calls the sort callback function after validating the matrix input.
        '''
        matrix = self.get_matrix()
        if matrix is None or len(matrix) == 0:
            messagebox.showerror("Input Error", "Please enter a non-empty matrix.")
            return
        self.sort_callback()
class MatrixDisplayFrame(tk.Frame):

    Frame for displaying the sorted matrix.
    '''

        self.label.pack()
        self.display_area = tk.Text(self, height=10, width=30)
        self.display_area.pack()

        '''
        Updates the display with the sorted matrix.
        '''
        self.display_area.delete("1.0", tk.END)

import tkinter as tk
from tkinter import messagebox
class MatrixSorterApp:
    '''
    Main application class for sorting a matrix.
    '''
    def __init__(self):
        self.root = tk.Tk()
        self.root.title("Matrix Sorter")
        self.input_frame = MatrixInputFrame(self.root, self.sort_matrix)
        self.input_frame.pack(pady=10)
        self.display_frame = MatrixDisplayFrame(self.root)
        self.display_frame.pack(pady=10)
    def sort_matrix(self):
        '''
        Sorts the matrix based on the sum of its rows and updates the display.
        '''
        matrix = self.input_frame.get_matrix()
        if matrix is None:
            return
        sorted_matrix = sorted(matrix, key=sum)
        self.display_frame.update_display(sorted_matrix)
    def run(self):
        '''
        Runs the main loop of the application.
        '''
        self.root.mainloop()
class MatrixInputFrame(tk.Frame):
    '''
    Frame for inputting the matrix.
    '''
    def __init__(self, master, sort_callback):
        super().__init__(master)
        self.sort_callback = sort_callback
        self.label = tk.Label(self, text="Enter matrix (comma-separated rows):")
        self.label.pack()
        self.text_area = tk.Text(self, height=10, width=30)
        self.text_area.pack()
        self.sort_button = tk.Button(self, text="Sort Matrix", command=self.sort_matrix)
        self.sort_button.pack()
    def get_matrix(self):
        '''
        Retrieves the matrix input from the user.
        '''
        try:
            input_text = self.text_area.get("1.0", tk.END).strip()
            matrix = [list(map(int, row.split(','))) for row in input_text.splitlines() if row]
            return matrix
        except ValueError:
            messagebox.showerror("Input Error", "Please enter a valid matrix.")
            return None
    def sort_matrix(self):
        '''
        Calls the sort callback function after validating the matrix input.
        '''
        matrix = self.get_matrix()
        if matrix is None or len(matrix) == 0:

        self.sort_callback()

'''
This module contains classes to handle matrix sorting and GUI display.
'''
import tkinter as tk
from tkinter import messagebox
class MatrixSorterApp:
    '''
    Main application class for sorting a matrix.
    '''
    def __init__(self):
        self.root = tk.Tk()
        self.root.title("Matrix Sorter")
        self.input_frame = MatrixInputFrame(self.root, self.sort_matrix)
        self.input_frame.pack(pady=10)
        self.display_frame = MatrixDisplayFrame(self.root)
        self.display_frame.pack(pady=10)
    def sort_matrix(self):
        '''
        Sorts the matrix based on the sum of its rows and updates the display.
        '''
        matrix = self.input_frame.get_matrix()
        if matrix is None:
            return
        sorted_matrix = sorted(matrix, key=sum)
        self.display_frame.update_display(sorted_matrix)
    def run(self):
        '''
        Runs the main loop of the application.
        '''
        self.root.mainloop()
class MatrixInputFrame(tk.Frame):
    '''
    Frame for inputting the matrix.
    '''
    def __init__(self, master, sort_callback):
        super().__init__(master)
        self.sort_callback = sort_callback
        self.label = tk.Label(self, text="Enter matrix (comma-separated rows):")
        self.label.pack()
        self.text_area = tk.Text(self, height=10, width=30)
        self.text_area.pack()
        self.sort_button = tk.Button(self, text="Sort Matrix", command=self.sort_matrix)
        self.sort_button.pack()
    def get_matrix(self):
        '''
        Retrieves the matrix input from the user.
        '''
        try:
            input_text = self.text_area.get("1.0", tk.END).strip()
            matrix = [list(map(int, row.split(','))) for row in input_text.splitlines() if row]
            return matrix
        except ValueError:
            messagebox.showerror("Input Error", "Please enter a valid matrix.")
            return None
    def sort_matrix(self):
        '''
        Calls the sort callback function after validating the matrix input.
        '''
        matrix = self.get_matrix()
        if matrix is None or len(matrix) == 0:
            messagebox.showerror("Input Error", "Please enter a non-empty matrix.")
            return
        self.sort_callback()
class MatrixDisplayFrame(tk.Frame):
    '''
    Frame for displaying the sorted matrix.
    '''
    def __init__(self, master):
        super().__init__(master)
        self.label = tk.Label(self, text="Sorted Matrix:")
        self.label.pack()
        self.display_area = tk.Text(self, height=10, width=30)
        self.display_area.pack()
    def update_display(self, sorted_matrix):
        '''
        Updates the display with the sorted matrix.
        '''
        self.display_area.delete("1.0", tk.END)
        for row in sorted_matrix:
            self.display_area.insert(tk.END, ', '.join(map(str, row)) + '\n')
