import tkinter as tk
from tkinter import messagebox

class WordLengthCheckerGUI:
    def __init__(self, master):
        self.master = master
        self.master.title("Word Length Checker")

        self.label = tk.Label(master, text="Enter a word:")
        self.label.pack()

        self.entry = tk.Entry(master)
        self.entry.pack()

        self.check_button = tk.Button(master, text="Check", command=self.check_word_length)
        self.check_button.pack()

    def check_word_length(self):
        word = self.entry.get()
        result = "Odd" if len(word) % 2 != 0 else "Even"
        messagebox.showinfo("Result", f"The length of the word '{word}' is {result}.")
