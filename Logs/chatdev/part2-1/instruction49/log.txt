'''
Main application file for finding the list with minimum length.
'''

from tkinter import Tk, Label, Entry, Button, Text, END
from min_length import MinLengthFinder

class MinLengthApp:
    '''
    Main application class for the GUI.
    '''
    def __init__(self, master):
        self.master = master
        master.title("Minimum Length List Finder")

        self.label = Label(master, text="Enter lists (comma-separated):")
        self.label.pack()

        self.input_text = Text(master, height=10, width=50)
        self.input_text.pack()

        self.find_button = Button(master, text="Find Minimum Length List", command=self.find_min_length)
        self.find_button.pack()

        self.result_label = Label(master, text="Result:")
        self.result_label.pack()

        self.result_text = Text(master, height=5, width=50)
        self.result_text.pack()

    def find_min_length(self):
        '''
        Finds the list with the minimum length and displays it.
        '''
        input_data = self.get_input()
        finder = MinLengthFinder()
        min_list = finder.find_min_length(input_data)
        self.display_result(min_list)

    def get_input(self):
        '''
        Retrieves and processes user input from the text box.
        '''
        input_str = self.input_text.get("1.0", END).strip()
        lists = input_str.splitlines()
        return [eval(lst) for lst in lists if lst]

    def display_result(self, min_list):
        '''
        Displays the result in the GUI.
        '''
        self.result_text.delete("1.0", END)
        self.result_text.insert(END, str(min_list))

if __name__ == "__main__":
    root = Tk()
    app = MinLengthApp(root)
    root.mainloop()
```

class MinLengthFinder:
    '''
    Class to find the list with minimum length.
    '''
    def find_min_length(self, lists):
        '''
        Finds the list with the minimum length using a lambda function.
        '''
        if not lists:
            return None
        return min(lists, key=lambda x: len(x))
