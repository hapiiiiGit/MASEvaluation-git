```python
'''
Main application file for the character addition GUI.
'''

import tkinter as tk
from tkinter import messagebox
from utils import get_character_from_string

class MainApp:
    '''
    Main application class to create the GUI for character addition.
    '''

    def __init__(self, master):
        '''
        Initializes the main window and its components.
        '''
        self.master = master
        master.title("Character Addition App")

        self.label = tk.Label(master, text="Enter a string:")
        self.label.pack()

        self.entry = tk.Entry(master)
        self.entry.pack()

        self.calculate_button = tk.Button(master, text="Calculate Character", command=self.calculate_character)
        self.calculate_button.pack()

        self.result_label = tk.Label(master, text="")
        self.result_label.pack()

    def calculate_character(self):
        '''
        Retrieves the input string, computes the character, and displays the result.
        '''
        input_string = self.entry.get()
        if not input_string:
            messagebox.showerror("Input Error", "Please enter a string.")
            return
        
        result = get_character_from_string(input_string)
        self.result_label.config(text=f"Resulting Character: {result}")

if __name__ == "__main__":
    root = tk.Tk()
    app = MainApp(root)
    root.mainloop()
```

    '''
    total = sum(ord(char) for char in input_string)
    return chr(total)
