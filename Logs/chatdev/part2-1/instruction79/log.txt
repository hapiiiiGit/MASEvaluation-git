'''
This module contains the TetrahedralCalculator class which provides
a method to calculate the nth tetrahedral number.
'''

class TetrahedralCalculator:
    def calculate(self, n):
        """
        Calculate the nth tetrahedral number using the formula:
        T(n) = n * (n + 1) * (n + 2) / 6
        """
        if n < 1:
            return 0
        return n * (n + 1) * (n + 2) // 6

class TetrahedralCalculator:
    def calculate(self, n):
        """
        Calculate the nth tetrahedral number using the formula:
        T(n) = n * (n + 1) * (n + 2) / 6

        """

        self.label.pack()
        self.entry = tk.Entry(master)

        self.calculate_button = tk.Button(master, text="Calculate", command=self.calculate_tetrahedral)

        try:

            self.result_label.config(text=f"Tetrahedral number T({n}) = {result}")

'''

a method to calculate the nth tetrahedral number.

class TetrahedralCalculator:
    def calculate(self, n):
        """
        Calculate the nth tetrahedral number using the formula:
        T(n) = n * (n + 1) * (n + 2) / 6
        Raise ValueError if n is less than 1.
        """
        if n < 1:
            raise ValueError("Input must be a positive integer.")
        return n * (n + 1) * (n + 2) // 6
