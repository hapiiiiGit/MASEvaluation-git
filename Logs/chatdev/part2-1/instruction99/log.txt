import tkinter as tk

class NextPalindromeApp:
    def __init__(self, master):
        self.master = master
        self.master.title("Next Palindrome Finder")

        self.label = tk.Label(master, text="Enter a number:")
        self.label.pack()

        self.entry = tk.Entry(master)
        self.entry.pack()

        self.result_label = tk.Label(master, text="")
        self.result_label.pack()

        self.find_button = tk.Button(master, text="Find Next Palindrome", command=self.find_next_palindrome)
        self.find_button.pack()

    def find_next_palindrome(self):
        input_number = int(self.entry.get())
        next_palindrome = find_next_smallest_palindrome(input_number)
        self.result_label.config(text=f"Next Palindrome: {next_palindrome}")

def find_next_smallest_palindrome(number):
    number += 1
    while str(number) != str(number)[::-1]:
        number += 1
    return number

if __name__ == "__main__":
    root = tk.Tk()
    app = NextPalindromeApp(root)
    root.mainloop()

class NextPalindromeApp:
    def __init__(self, master):
        self.master = master

    number += 1
