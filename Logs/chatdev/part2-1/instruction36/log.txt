```python
'''
Main application file that initializes the GUI and handles user interactions.
'''

from tkinter import Tk, StringVar, Button, Label, Frame
from input_frame import InputFrame
from output_frame import OutputFrame

class MainApp:
    def __init__(self, master):
        self.master = master
        master.title("Mixed List Sorter")

        self.input_frame = InputFrame(master, self.sort_list)
        self.input_frame.pack()

        self.output_frame = OutputFrame(master)
        self.output_frame.pack()

    def sort_list(self):
        mixed_list = self.input_frame.get_input()
        sorted_list = sorted(mixed_list, key=lambda x: (isinstance(x, str), x))
        self.output_frame.display_output(sorted_list)

if __name__ == "__main__":
    root = Tk()
    app = MainApp(root)
    root.mainloop()
```

'''
InputFrame class that handles the input area for the user to enter their mixed list.

from tkinter import Frame, Entry, Button, Label, StringVar

class InputFrame(Frame):
    def __init__(self, master, sort_callback):
        super().__init__(master)
        self.sort_callback = sort_callback

        self.label = Label(self, text="Enter mixed list (e.g., [1, 'apple', 3, 'banana']):")
        self.label.pack()

        self.input_var = StringVar()
        self.input_entry = Entry(self, textvariable=self.input_var, width=50)
        self.input_entry.pack()

        self.sort_button = Button(self, text="Sort List", command=self.sort_callback)
        self.sort_button.pack()

    def get_input(self):
        input_text = self.input_var.get()
        try:
            mixed_list = eval(input_text)
            if not isinstance(mixed_list, list):
                raise ValueError("Input must be a list.")
            return mixed_list
        except Exception as e:
            return []

```

        self.label = Label(self, text="Sorted Output:")
        self.label.pack()

        self.output_label = Label(self, text="")
        self.output_label.pack()

    def display_output(self, sorted_list):
        self.output_label.config(text=str(sorted_list))

        self.label = Label(self, text="Enter mixed list (e.g., [1, 'apple', 3, 'banana']):")
        self.label.pack()

        self.input_entry = Entry(self, textvariable=self.input_var, width=50)
        self.input_entry.pack()
        self.sort_button = Button(self, text="Sort List", command=self.sort_callback)
        self.sort_button.pack()
    def get_input(self):
        input_text = self.input_var.get()
        try:
            mixed_list = eval(input_text)

                raise ValueError("Input must be a list.")
            return mixed_list

'''
InputFrame class that handles the input area for the user to enter their mixed list.
'''
from tkinter import Frame, Entry, Button, Label, StringVar
class InputFrame(Frame):
    def __init__(self, master, sort_callback):
        super().__init__(master)
        self.sort_callback = sort_callback
        self.label = Label(self, text="Enter mixed list (e.g., [1, 'apple', 3, 'banana']):")
        self.label.pack()
        self.input_var = StringVar()
        self.input_entry = Entry(self, textvariable=self.input_var, width=50)
        self.input_entry.pack()
        self.sort_button = Button(self, text="Sort List", command=self.sort_callback)
        self.sort_button.pack()
    def get_input(self):
        input_text = self.input_var.get()
        try:
            mixed_list = eval(input_text)
            if not isinstance(mixed_list, list):
                raise ValueError("Input must be a list.")
            for item in mixed_list:
                if not isinstance(item, (int, str)):
                    raise ValueError("List must contain only integers and strings.")
            return mixed_list
        except Exception as e:
            return None  # Return None on error

        super().__init__(master)
        self.label = Label(self, text="Sorted Output:")
        self.label.pack()
        self.output_label = Label(self, text="")
        self.output_label.pack()
    def display_output(self, sorted_list):
