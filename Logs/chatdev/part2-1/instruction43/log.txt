import tkinter as tk
from word_matcher import WordMatcher
def match_word():
    input_string = entry_string.get()
    input_word = entry_word.get()
    word_matcher = WordMatcher()
    result = word_matcher.match_word_at_beginning(input_string, input_word)
    if result:
        result_label.config(text="Word matches at the beginning!")
    else:
        result_label.config(text="Word does not match at the beginning.")
# GUI setup
root = tk.Tk()
root.title("Word Matcher")
label_string = tk.Label(root, text="Enter a string:")
label_string.pack()
entry_string = tk.Entry(root)
entry_string.pack()
label_word = tk.Label(root, text="Enter a word:")
label_word.pack()
entry_word = tk.Entry(root)
entry_word.pack()
match_button = tk.Button(root, text="Match Word", command=match_word)
match_button.pack()
result_label = tk.Label(root, text="")
result_label.pack()
root.mainloop()

'''
This file contains the WordMatcher class responsible for matching a word at the beginning of a string.

class WordMatcher:
    def __init__(self):
        pass
    def match_word_at_beginning(self, string, word):
        return string.startswith(word)
