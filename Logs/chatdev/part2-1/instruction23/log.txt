class BinaryToDecimalConverter:
    def convert(self, binary_str):
        if not all(char in '01' for char in binary_str):

        decimal_value = 0
        for index, digit in enumerate(reversed(binary_str)):

        return decimal_value

'''
Main file to run the binary to decimal converter application.

from tkinter import Tk, Label, Entry, Button, StringVar
from converter import BinaryToDecimalConverter
class App:
    def __init__(self, master):
        self.master = master
        master.title("Binary to Decimal Converter")
        self.label = Label(master, text="Enter a binary number:")
        self.label.pack()
        self.binary_input = StringVar()
        self.entry = Entry(master, textvariable=self.binary_input)
        self.entry.pack()
        self.convert_button = Button(master, text="Convert", command=self.convert_and_display)
        self.convert_button.pack()
        self.result_label = Label(master, text="")
        self.result_label.pack()
        self.converter = BinaryToDecimalConverter()
    def convert_and_display(self):
        binary_str = self.binary_input.get()
        try:
            decimal_value = self.converter.convert(binary_str)
            self.result_label.config(text=f"Decimal: {decimal_value}")
        except ValueError as e:
            self.result_label.config(text=str(e))
if __name__ == "__main__":
    root = Tk()
    app = App(root)
    root.mainloop()

'''
Module for converting binary numbers to decimal.
'''
class BinaryToDecimalConverter:
    def convert(self, binary_str):
        if not all(char in '01' for char in binary_str):
            raise ValueError("Invalid binary number. Please enter a valid binary string.")
        decimal_value = 0
        for index, digit in enumerate(reversed(binary_str)):
            decimal_value += int(digit) * (2 ** index)
        return decimal_value
