def check_tuples_length(tuples):
    '''
    Checks if all tuples in the given list have the same length.
    
    Parameters:
    tuples (list of tuples): The list of tuples to check.

    Returns:
    bool: True if all tuples have the same length, False otherwise.
    '''

        return True  # An empty list is considered to have equal lengths

    first_length = len(tuples[0])
    return all(len(t) == first_length for t in tuples)

class TupleLengthCheckerApp:

        Initializes the GUI components.

        self.label.pack()
        self.entry = tk.Entry(master, width=50)
        self.entry.pack()

        self.check_button.pack()
    def on_check_button_click(self):
        '''

        '''
        input_text = self.entry.get()

            # Convert input text to a list of tuples
            tuples = eval(input_text)

                raise ValueError("Input must be a list or tuple of tuples.")

        except Exception as e:
            messagebox.showerror("Error", str(e))

def check_tuples_length(tuples):
    '''
    Checks if all tuples in the given list have the same length.
    Parameters:
    tuples (list of tuples): The list of tuples to check.
    Returns:
    bool: True if all tuples have the same length, False otherwise.
    '''
    if not tuples:
        return True  # An empty list is considered to have equal lengths
    first_length = len(tuples[0])
    return all(len(t) == first_length for t in tuples)

    '''
    Checks if all tuples in the given list have the same length.
