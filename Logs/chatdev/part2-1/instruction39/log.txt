   - **`display_result`**: Displays the calculated frequency in the GUI.

'''
Main application file for the Frequency Counter GUI.
'''

from tkinter import Tk, Label, Entry, Button, Text, END
from frequency_counter import find_frequency

class FrequencyCounterApp:
    def __init__(self, master):
        self.master = master
        master.title("Frequency Counter")

        self.label = Label(master, text="Enter a list of lists (e.g., [[1, 2], [2, 3], [1]])")
        self.label.pack()

        self.entry = Entry(master, width=50)
        self.entry.pack()

        self.calculate_button = Button(master, text="Calculate Frequency", command=self.calculate_frequency)
        self.calculate_button.pack()

        self.result_text = Text(master, height=10, width=50)
        self.result_text.pack()

    def calculate_frequency(self):
        input_data = self.entry.get()
        try:
            # Evaluate the input string to convert it into a list of lists
            list_of_lists = eval(input_data)
            frequency = find_frequency(list_of_lists)
            self.display_result(frequency)
        except Exception as e:
            self.result_text.delete(1.0, END)
            self.result_text.insert(END, f"Error: {str(e)}")

    def display_result(self, frequency):
        self.result_text.delete(1.0, END)
        for item, count in frequency.items():
            self.result_text.insert(END, f"{item}: {count}\n")

if __name__ == "__main__":
    root = Tk()
    app = FrequencyCounterApp(root)
    root.mainloop()
```

#### 2. `frequency_counter.py`

```python
'''
Utility module for calculating frequency of elements in a list of lists.
'''

from collections import Counter

def find_frequency(list_of_lists):
    '''
    Calculate the frequency of elements in a given list of lists.
    
    Args:
        list_of_lists (list): A list containing sublists of elements.
    
    Returns:
        dict: A dictionary with elements as keys and their frequencies as values.
    '''
    # Flatten the list of lists and count frequencies
    flat_list = [item for sublist in list_of_lists for item in sublist]
    return dict(Counter(flat_list))
```

from collections import Counter
def find_frequency(list_of_lists):
    '''
    Calculate the frequency of elements in a given list of lists.
    Args:
        list_of_lists (list): A list containing sublists of elements.
    Returns:
        dict: A dictionary with elements as keys and their frequencies as values.
    '''
    # Flatten the list of lists and count frequencies
    flat_list = [item for sublist in list_of_lists for item in sublist]
    return dict(Counter(flat_list))

'''
Utility module for calculating frequency of elements in a list of lists.
'''
from collections import Counter
def find_frequency(list_of_lists):
    '''
    Calculate the frequency of elements in a given list of lists.
    Args:
        list_of_lists (list): A list containing sublists of elements.
    Returns:
        dict: A dictionary with elements as keys and their frequencies as values.
    '''
    # Flatten the list of lists and count frequencies
    flat_list = [item for sublist in list_of_lists for item in sublist]
    return dict(Counter(flat_list))

from tkinter import Tk, Label, Entry, Button, Text, END
from frequency_counter import find_frequency
class FrequencyCounterApp:
    def __init__(self, master):
        self.master = master
        master.title("Frequency Counter")
        self.label = Label(master, text="Enter a list of lists (e.g., [[1, 2], [2, 3], [1]])")
        self.label.pack()
        self.entry = Entry(master, width=50)
        self.entry.pack()
        self.calculate_button = Button(master, text="Calculate Frequency", command=self.calculate_frequency)
        self.calculate_button.pack()
        self.result_text = Text(master, height=10, width=50)
        self.result_text.pack()
    def calculate_frequency(self):
        input_data = self.entry.get()
        try:
            # Evaluate the input string to convert it into a list of lists
            list_of_lists = eval(input_data)
            frequency = find_frequency(list_of_lists)
            self.display_result(frequency)
        except Exception as e:
            self.result_text.delete(1.0, END)

        self.result_text.delete(1.0, END)

'''
Utility module for calculating frequency of elements in a list of lists.
'''
from collections import Counter
def find_frequency(list_of_lists):
    '''
    Calculate the frequency of elements in a given list of lists.
    Args:
        list_of_lists (list): A list containing sublists of elements.
    Returns:
        dict: A dictionary with elements as keys and their frequencies as values.
    '''
    # Flatten the list of lists and count frequencies
    flat_list = [item for sublist in list_of_lists for item in sublist]
    return dict(Counter(flat_list))
