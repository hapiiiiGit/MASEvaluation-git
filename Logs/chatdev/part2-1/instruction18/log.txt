#### 1. main.py

```python
'''
Main application file for checking duplicates in an array of integers.
'''

from tkinter import Tk
from input_frame import InputFrame
from result_frame import ResultFrame

class DuplicateCheckerApp:
    def __init__(self, master):
        self.master = master
        self.master.title("Duplicate Checker")
        
        self.input_frame = InputFrame(master=self.master, app=self)
        self.input_frame.pack()

        self.result_frame = ResultFrame(master=self.master)
        self.result_frame.pack()

    def check_duplicates(self, input_array):
        """
        Check if the input array contains duplicates.
        """
        has_duplicates = len(input_array) != len(set(input_array))
        self.result_frame.display_result(has_duplicates)

if __name__ == "__main__":
    root = Tk()
    app = DuplicateCheckerApp(master=root)
    root.mainloop()
```

#### 2. input_frame.py

```python
'''
Input frame for user to enter an array of integers.
'''

from tkinter import Frame, Label, Entry, Button

class InputFrame(Frame):
    def __init__(self, master, app):
        super().__init__(master)
        self.app = app
        
        self.label = Label(self, text="Enter integers (comma-separated):")
        self.label.pack()
        
        self.entry = Entry(self)
        self.entry.pack()
        
        self.check_button = Button(self, text="Check Duplicates", command=self.check_duplicates)
        self.check_button.pack()

    def check_duplicates(self):
        """
        Retrieve input and check for duplicates.
        """
        input_text = self.entry.get()
        input_array = [int(x.strip()) for x in input_text.split(",") if x.strip().isdigit()]
        self.app.check_duplicates(input_array)
```

#### 3. result_frame.py

```python
'''
Result frame for displaying the result of the duplicate check.
'''

from tkinter import Frame, Label

class ResultFrame(Frame):
    def __init__(self, master):
        super().__init__(master)
        self.result_label = Label(self, text="")
        self.result_label.pack()

    def display_result(self, has_duplicates):
        """
        Display the result of the duplicate check.
        """
        if has_duplicates:
            self.result_label.config(text="Duplicates found!")
        else:
            self.result_label.config(text="No duplicates found.")
```

from tkinter import Frame, Label, Entry, Button

    def __init__(self, master, app):
        super().__init__(master)

from tkinter import Frame, Label, Entry, Button
class InputFrame(Frame):
    def __init__(self, master, app):
        super().__init__(master)
        self.app = app
        self.label = Label(self, text="Enter integers (comma-separated):")
        self.label.pack()
        self.entry = Entry(self)
        self.entry.pack()
        self.check_button = Button(self, text="Check Duplicates", command=self.check_duplicates)
        self.check_button.pack()
    def check_duplicates(self):
        """
        Retrieve input and check for duplicates.
        """
        input_text = self.entry.get()
        if not input_text.strip():  # Check for empty input
            self.app.result_frame.display_result(None)  # Indicate invalid input
            return
        try:
            input_array = [int(x.strip()) for x in input_text.split(",")]
            self.app.check_duplicates(input_array)
        except ValueError:
            self.app.result_frame.display_result(None)  # Indicate invalid input

from tkinter import Frame, Label, Entry, Button
class InputFrame(Frame):
    def __init__(self, master, app):
        super().__init__(master)
        self.app = app
        self.label = Label(self, text="Enter integers (comma-separated):")
        self.label.pack()
        self.entry = Entry(self)
        self.entry.pack()
        self.check_button = Button(self, text="Check Duplicates", command=self.check_duplicates)
        self.check_button.pack()
    def check_duplicates(self):
        """
        Retrieve input and check for duplicates.

            self.app.result_frame.display_result(None)  # Indicate invalid input
            return

            input_array = [int(x.strip()) for x in input_text.split(",")]
            self.app.check_duplicates(input_array)
        except ValueError:
            self.app.result_frame.display_result(None)  # Indicate invalid input

'''
Input frame for user to enter an array of integers.
'''
from tkinter import Frame, Label, Entry, Button
class InputFrame(Frame):
    def __init__(self, master, app):
        super().__init__(master)
        self.app = app
        self.label = Label(self, text="Enter integers (comma-separated):")
        self.label.pack()
        self.entry = Entry(self)
        self.entry.pack()
        self.check_button = Button(self, text="Check Duplicates", command=self.check_duplicates)
        self.check_button.pack()
    def check_duplicates(self):
        """
        Retrieve input and check for duplicates.
        """
        input_text = self.entry.get()
        if not input_text.strip():  # Check for empty input
            self.app.result_frame.display_result(None)  # Indicate invalid input
            return
        try:
            input_array = [int(x.strip()) for x in input_text.split(",")]
            self.app.check_duplicates(input_array)
        except ValueError:
            self.app.result_frame.display_result(None)  # Indicate invalid input
