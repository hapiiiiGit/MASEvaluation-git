'''
This module contains the App class which creates the GUI for the Parallelogram Area Calculator.

from tkinter import Frame, Label, Entry, Button, StringVar, messagebox

        Initializes the GUI components.

        master (Tk): The main window of the application.

class ParallelogramAreaCalculator:
    def calculate_area(self, base, height):
        '''
        Calculate the area of a parallelogram given the base and height.
        Parameters:
        base (float): The length of the base of the parallelogram.
        height (float): The height of the parallelogram.
        Returns:
        float: The area of the parallelogram.
        '''
        if base < 0 or height < 0:
            raise ValueError("Base and height must be positive numbers.")
        return base * height
