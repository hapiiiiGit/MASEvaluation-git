```python
'''
Module that implements the counting sort algorithm.
'''

def counting_sort(arr):
    """
    Sorts an array using the counting sort algorithm.
    
    Parameters:
    arr (list): The list of non-negative integers to be sorted.

    Returns:
    list: A new sorted list.
    """
    if not arr:
        return []

    max_val = max(arr)
    count = [0] * (max_val + 1)

    for num in arr:
        count[num] += 1

    sorted_arr = []
    for i, cnt in enumerate(count):
        sorted_arr.extend([i] * cnt)

    return sorted_arr
```

'''
Module that implements the counting sort algorithm.
'''
def counting_sort(arr):
    """
    Sorts an array using the counting sort algorithm.
    Parameters:
    arr (list): The list of non-negative integers to be sorted.
    Returns:
    list: A new sorted list.
    """
    if not arr:
        return []
    max_val = max(arr)
    count = [0] * (max_val + 1)
    for num in arr:
        count[num] += 1
    sorted_arr = []
    for i, cnt in enumerate(count):
        sorted_arr.extend([i] * cnt)
    return sorted_arr
```

def counting_sort(arr):
    """
    Sorts an array using the counting sort algorithm.
    Parameters:
    arr (list): The list of non-negative integers to be sorted.
    Returns:
    list: A new sorted list.
    """
    if not arr:
        return []
    max_val = max(arr)
    count = [0] * (max_val + 1)
    for num in arr:
        if num < 0:
            raise ValueError("Please enter only non-negative integers.")
        count[num] += 1
    sorted_arr = []
    for i, cnt in enumerate(count):
        sorted_arr.extend([i] * cnt)
    return sorted_arr

'''
Module that implements the counting sort algorithm.
'''
def counting_sort(arr):
    """
    Sorts an array using the counting sort algorithm.
    Parameters:
    arr (list): The list of non-negative integers to be sorted.
    Returns:
    list: A new sorted list.
    """
    if not arr:
        return []
    max_val = max(arr)
    count = [0] * (max_val + 1)
    for num in arr:
        if num < 0:
            raise ValueError("Please enter only non-negative integers.")
        count[num] += 1
    sorted_arr = []
    for i, cnt in enumerate(count):
        sorted_arr.extend([i] * cnt)
    return sorted_arr
