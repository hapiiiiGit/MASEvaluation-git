```python
'''
Main application file for the Undulating Number Checker.
'''

import tkinter as tk
from tkinter import messagebox
from number_checker import NumberChecker

class UndulatingNumberCheckerApp:
    '''
    Class to create the GUI for the Undulating Number Checker application.
    '''

    def __init__(self, master):
        self.master = master
        master.title("Undulating Number Checker")

        self.label = tk.Label(master, text="Enter a number:")
        self.label.pack()

        self.entry = tk.Entry(master)
        self.entry.pack()

        self.check_button = tk.Button(master, text="Check", command=self.check_number)
        self.check_button.pack()

        self.result_label = tk.Label(master, text="")
        self.result_label.pack()

    def check_number(self):
        '''
        Retrieve the input number, validate it, and display the result.
        '''
        number = self.entry.get()
        if number.isdigit():
            checker = NumberChecker()
            is_undulating = checker.is_undulating(int(number))
            result_text = f"{number} is undulating." if is_undulating else f"{number} is not undulating."
            self.result_label.config(text=result_text)
        else:
            messagebox.showerror("Invalid input", "Please enter a valid number.")

if __name__ == "__main__":
    root = tk.Tk()
    app = UndulatingNumberCheckerApp(root)
    root.mainloop()
```

```python
'''
Module containing the logic to check if a number is undulating.
'''

class NumberChecker:
    '''
    Class to check if a number is undulating.
    '''

    def is_undulating(self, number):
        '''
        Check if the given number is undulating.
        '''
        digits = [int(d) for d in str(number)]
        
        if len(digits) < 2:
            return False  # A single digit cannot be undulating

        # Check for undulating pattern
        increasing = None
        for i in range(1, len(digits)):
            if digits[i] > digits[i - 1]:
                if increasing is None:
                    increasing = True
                elif increasing:
                    return False
                else:
                    increasing = True
            elif digits[i] < digits[i - 1]:
                if increasing is None:
                    increasing = False
                elif not increasing:
                    return False
                else:
                    increasing = False
            else:
                return False  # Two consecutive digits are the same

        return True
```

class NumberChecker:

    def is_undulating(self, number):
        '''

        '''
        digits = [int(d) for d in str(number)]
        if len(digits) < 3:
            return False  # A number with less than 3 digits cannot be undulating

        increasing = digits[0] < digits[1]
        for i in range(2, len(digits)):
            if (digits[i] < digits[i - 1]) == increasing:
                increasing = not increasing
            else:
                return False

        return True
