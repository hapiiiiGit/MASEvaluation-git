Main application file for finding the largest prime factor of a given number.

import tkinter as tk
from tkinter import messagebox
from prime_factor import largest_prime_factor

class PrimeFactorApp:
    def __init__(self, master):
        self.master = master
        master.title("Largest Prime Factor Finder")

        self.label = tk.Label(master, text="Enter a number:")
        self.label.pack()

        self.entry = tk.Entry(master)
        self.entry.pack()

        self.find_button = tk.Button(master, text="Find Largest Prime Factor", command=self.find_largest_prime_factor)
        self.find_button.pack()

        self.result_label = tk.Label(master, text="")
        self.result_label.pack()

    def find_largest_prime_factor(self):
        try:
            number = int(self.entry.get())
            if number < 1:
                raise ValueError("Number must be greater than 0.")
            largest_factor = largest_prime_factor(number)
            self.result_label.config(text=f"Largest Prime Factor: {largest_factor}")
        except ValueError as e:
            messagebox.showerror("Input Error", str(e))

if __name__ == "__main__":
    root = tk.Tk()
    app = PrimeFactorApp(root)
    root.mainloop()

'''
Module containing functions to find the largest prime factor of a number.
'''

def largest_prime_factor(n):
    """Return the largest prime factor of the given number n."""
    # Start with the smallest prime factor
    prime_factor = 1

    # Check for number of 2s that divide n
    while n % 2 == 0:
        prime_factor = 2
        n //= 2

    # n must be odd at this point, so we can skip even numbers
    for i in range(3, int(n**0.5) + 1, 2):
        while n % i == 0:
            prime_factor = i
            n //= i

    # This condition is to check if n is a prime number greater than 2
    if n > 2:
        prime_factor = n

    return prime_factor

def is_prime(num):
    """Check if a number is prime."""
    if num <= 1:
        return False
    for i in range(2, int(num**0.5) + 1):
        if num % i == 0:
            return False
    return True
```

def largest_prime_factor(n):
    """Return the largest prime factor of the given number n."""
    # Start with the smallest prime factor
    prime_factor = 1
    # Check for number of 2s that divide n
    while n % 2 == 0:
        prime_factor = 2
        n //= 2
    # n must be odd at this point, so we can skip even numbers
    for i in range(3, int(n**0.5) + 1, 2):
        while n % i == 0:
            prime_factor = i
            n //= i
    # This condition is to check if n is a prime number greater than 2
    if n > 2:
        prime_factor = n
    return prime_factor
def is_prime(num):
    """Check if a number is prime."""
    if num <= 1:
        return False
    for i in range(2, int(num**0.5) + 1):
        if num % i == 0:
            return False
    return True

    """Return the largest prime factor of the given number n."""
    # Start with the smallest prime factor
    prime_factor = 1
    # Check for number of 2s that divide n
    while n % 2 == 0:

        n //= 2

            n //= i

    while n % 2 == 0:

'''
Module containing functions to find the largest prime factor of a number.
'''
def largest_prime_factor(n):
    """Return the largest prime factor of the given number n."""
    # Start with the smallest prime factor
    prime_factor = 1
    # Check for number of 2s that divide n
    while n % 2 == 0:
        prime_factor = 2
        n //= 2
    # n must be odd at this point, so we can skip even numbers
    for i in range(3, int(n**0.5) + 1, 2):
        while n % i == 0:
            prime_factor = i
            n //= i
    # This condition is to check if n is a prime number greater than 2
    if n > 2:
        prime_factor = n
    return prime_factor

def is_prime(num):
    """Check if a number is prime."""
    if num <= 1:
        return False
    for i in range(2, int(num**0.5) + 1):
        if num % i == 0:
            return False
    return True

def largest_prime_factor(n):
    """Return the largest prime factor of the given number n."""
    # Start with the smallest prime factor
    prime_factor = 1
    # Check for number of 2s that divide n
    while n % 2 == 0:
        prime_factor = 2
        n //= 2
    # n must be odd at this point, so we can skip even numbers
    for i in range(3, int(n**0.5) + 1, 2):
        while n % i == 0:
            prime_factor = i
            n //= i
    # This condition is to check if n is a prime number greater than 2
    if n > 2:
        prime_factor = n
    return prime_factor
