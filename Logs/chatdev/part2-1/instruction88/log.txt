Main application file for finding the closest smaller number than n.

import tkinter as tk
from tkinter import messagebox
from number_finder import find_closest_smaller_number

class MainApp:
    def __init__(self, master):
        self.master = master
        master.title("Closest Smaller Number Finder")

        self.label = tk.Label(master, text="Enter a number:")
        self.label.pack()

        self.entry = tk.Entry(master)
        self.entry.pack()

        self.find_button = tk.Button(master, text="Find Closest Smaller", command=self.find_closest_smaller)
        self.find_button.pack()

        self.result_label = tk.Label(master, text="")
        self.result_label.pack()

    def find_closest_smaller(self):
        try:
            n = int(self.entry.get())
            result = find_closest_smaller_number(n)
            self.result_label.config(text=f"Closest smaller number: {result}")
        except ValueError:
            messagebox.showerror("Input Error", "Please enter a valid integer.")

if __name__ == "__main__":

    app = MainApp(root)
    root.mainloop()

import tkinter as tk
from tkinter import messagebox

class MainApp:
