import tkinter as tk

class SubstringCheckerApp:
    def __init__(self, root):
        self.root = root
        self.root.title("Substring Checker")

        self.create_widgets()

    def create_widgets(self):
        self.input_label = tk.Label(self.root, text="Enter Substring:")
        self.input_label.pack()

        self.input_entry = tk.Entry(self.root)
        self.input_entry.pack()

        self.text_label = tk.Label(self.root, text="Enter List of Strings (comma-separated):")
        self.text_label.pack()

        self.text_entry = tk.Entry(self.root)
        self.text_entry.pack()

        self.check_button = tk.Button(self.root, text="Check Substring", command=self.check_substring)
        self.check_button.pack()

        self.result_label = tk.Label(self.root, text="")
        self.result_label.pack()

    def check_substring(self):
        substring = self.input_entry.get()
        strings = self.text_entry.get().split(',')

        for string in strings:
            if substring in string:
                self.result_label.config(text=f"Substring '{substring}' found in the list of strings.")
                return

        self.result_label.config(text=f"Substring '{substring}' not found in the list of strings.")

if __name__ == "__main__":
    root = tk.Tk()
    app = SubstringCheckerApp(root)
    root.mainloop()
```

'''
This file contains the main GUI class for the substring checking application.
'''
import tkinter as tk
class SubstringCheckerApp:
    def __init__(self, root):
        '''
        Initializes the SubstringCheckerApp with the given root window.
        Sets up the user interface components.
        '''
        self.root = root
        self.root.title("Substring Checker")
        self.create_widgets()
    def create_widgets(self):
        '''
        Creates and packs the widgets for the user interface.
        '''
        self.input_label = tk.Label(self.root, text="Enter Substring:")
        self.input_label.pack()
        self.input_entry = tk.Entry(self.root)
        self.input_entry.pack()
        self.text_label = tk.Label(self.root, text="Enter List of Strings (comma-separated):")
        self.text_label.pack()
        self.text_entry = tk.Entry(self.root)
        self.text_entry.pack()
        self.check_button = tk.Button(self.root, text="Check Substring", command=self.check_substring)
        self.check_button.pack()
        self.result_label = tk.Label(self.root, text="")
        self.result_label.pack()
    def check_substring(self):
        '''
        Checks if the entered substring is present in the list of strings provided by the user.
        Updates the result label with the appropriate message.
        '''
        substring = self.input_entry.get()
        strings_input = self.text_entry.get()
        if not substring or not strings_input:
            self.result_label.config(text="Please enter both a substring and a list of strings.")
            return
        strings = strings_input.split(',')
        strings = [s.strip() for s in strings]  # Strip whitespace from each string
        if not strings:  # Check if the list is empty after stripping
            self.result_label.config(text="Error: The list of strings cannot be empty.")
            return
        if not all(isinstance(s, str) for s in strings):
            self.result_label.config(text="Error: All items in the list must be strings.")
            return
        if not any(substring in string for string in strings):
            self.result_label.config(text=f"Substring '{substring}' not found in the list of strings.")
        else:
            self.result_label.config(text=f"Substring '{substring}' found in the list of strings.")
if __name__ == "__main__":
    root = tk.Tk()
    app = SubstringCheckerApp(root)
    root.mainloop()

'''
This file contains the main GUI class for the substring checking application.

import tkinter as tk
class SubstringCheckerApp:
    def __init__(self, root):
        '''
        Initializes the SubstringCheckerApp with the given root window.
        Sets up the user interface components.
        '''
        self.root = root
        self.root.title("Substring Checker")
        self.create_widgets()
    def create_widgets(self):
        '''
        Creates and packs the widgets for the user interface.
        '''
        self.input_label = tk.Label(self.root, text="Enter Substring:")
        self.input_label.pack()
        self.input_entry = tk.Entry(self.root)
        self.input_entry.pack()
        self.text_label = tk.Label(self.root, text="Enter List of Strings (comma-separated):")
        self.text_label.pack()
        self.text_entry = tk.Entry(self.root)
        self.text_entry.pack()
        self.check_button = tk.Button(self.root, text="Check Substring", command=self.check_substring)
        self.check_button.pack()
        self.result_label = tk.Label(self.root, text="")
        self.result_label.pack()
    def check_substring(self):
        '''
        Checks if the entered substring is present in the list of strings provided by the user.
        Updates the result label with the appropriate message.
        '''
        substring = self.input_entry.get()
        strings_input = self.text_entry.get()
        if not substring or not strings_input:
            self.result_label.config(text="Please enter both a substring and a list of strings.")
            return
        strings = strings_input.split(',')
        strings = [s.strip() for s in strings]  # Strip whitespace from each string
        if not strings:  # Check if the list is empty after stripping
            self.result_label.config(text="Error: The list of strings cannot be empty.")
            return
        if not all(isinstance(s, str) for s in strings):
            self.result_label.config(text="Error: All items in the list must be strings.")
            return
        if not any(substring in string for string in strings):
            self.result_label.config(text=f"Substring '{substring}' not found in the list of strings.")
        else:
            self.result_label.config(text=f"Substring '{substring}' found in the list of strings.")
if __name__ == "__main__":
    root = tk.Tk()
    app = SubstringCheckerApp(root)
    root.mainloop()
