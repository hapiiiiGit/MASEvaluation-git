def count_substrings_in_string(s):
    count = 0
    char_count = {}
    # Count occurrences of each character
    for char in s:
        if char in char_count:
            char_count[char] += 1

    # Calculate the number of substrings for each character
    for c in char_count:

    return count

'''
Utility file containing the function to count all substrings starting and ending with the same characters.
'''
def count_substrings_in_string(s):
    count = 0
    char_count = {}
    # Count occurrences of each character
    for char in s:
        if char in char_count:
            char_count[char] += 1
        else:
            char_count[char] = 1
    # Calculate the number of substrings for each character
    for c in char_count:
        n = char_count[c]
        count += (n * (n + 1)) // 2  # Combination formula for substrings
    return count
```

def count_substrings_in_string(s):
    count = 0
    char_count = {}
    # Count occurrences of each character
    for char in s:
        if char in char_count:
            char_count[char] += 1
        else:
            char_count[char] = 1
    # Calculate the number of substrings for each character
    for c in char_count:
        n = char_count[c]
        count += (n * (n + 1)) // 2  # Combination formula for substrings
    return count

'''
Utility file containing the function to count all substrings starting and ending with the same characters.
'''
def count_substrings_in_string(s):
    count = 0
    char_count = {}
    # Count occurrences of each character
    for char in s:
        if char in char_count:
            char_count[char] += 1
        else:
            char_count[char] = 1
    # Calculate the number of substrings for each character
    for c in char_count:
        n = char_count[c]
        count += (n * (n + 1)) // 2  # Combination formula for substrings
    return count
```

from tkinter import Tk, Label, Entry, Button, StringVar
from substring_counter import count_substrings_in_string
class SubstringCounterApp:
    def __init__(self, master):
        self.master = master
        master.title("Substring Counter")
        self.label = Label(master, text="Enter a string:")
        self.label.pack()
        self.input_string = StringVar()
        self.entry = Entry(master, textvariable=self.input_string)
        self.entry.pack()
        self.count_button = Button(master, text="Count Substrings", command=self.count_substrings)
        self.count_button.pack()
        self.result_label = Label(master, text="")
        self.result_label.pack()
    def count_substrings(self):
        input_str = self.input_string.get()
        # Validate input to ensure it is a non-empty string
        if not isinstance(input_str, str) or not input_str:
            self.display_result("Please enter a valid non-empty string.")
            return
        count = count_substrings_in_string(input_str)
        self.display_result(count)
    def display_result(self, count):
        # Display the result of the substring count
        self.result_label.config(text=f"Count of substrings: {count}")
if __name__ == "__main__":
    root = Tk()

'''
Utility file containing the function to count all substrings starting and ending with the same characters.
'''
def count_substrings_in_string(s):
    count = 0
    char_count = {}
    # Count occurrences of each character
    for char in s:
        if char in char_count:
            char_count[char] += 1
        else:
            char_count[char] = 1
    # Calculate the number of substrings for each character
    for c in char_count:
        n = char_count[c]
        count += (n * (n + 1)) // 2  # Combination formula for substrings
    return count
