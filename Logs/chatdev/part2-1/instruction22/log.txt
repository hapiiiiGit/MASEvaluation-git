Main application file for the MaxSumApp.

import tkinter as tk
from tkinter import messagebox
from utils import find_max_sum

class MaxSumApp:
    '''
    Class to create the main application window and handle user interactions.
    '''

    def __init__(self, master):
        '''
        Initializes the GUI components.
        '''
        self.master = master
        master.title("Maximum Sum Finder")

        self.label = tk.Label(master, text="Enter a list of lists (e.g. [[1,2],[3,4],[5]])")
        self.label.pack()

        self.input_text = tk.Entry(master)
        self.input_text.pack()

        self.calculate_button = tk.Button(master, text="Calculate Max Sum", command=self.calculate_max_sum)
        self.calculate_button.pack()

        self.result_label = tk.Label(master, text="")
        self.result_label.pack()

    def calculate_max_sum(self):
        '''
        Retrieves input, calculates the maximum sum, and displays the result.
        '''
        try:
            input_data = eval(self.input_text.get())
            if not isinstance(input_data, list) or not all(isinstance(i, list) for i in input_data):
                raise ValueError("Input must be a list of lists.")
            max_sum = find_max_sum(input_data)
            self.result_label.config(text=f"Maximum Sum: {max_sum}")
        except Exception as e:
            messagebox.showerror("Input Error", str(e))

if __name__ == "__main__":
    root = tk.Tk()

    root.mainloop()

def find_max_sum(list_of_lists):
    '''
    Finds the maximum sum of elements of lists in a list of lists.
    
    Parameters:
        list_of_lists (list): A list containing multiple lists of numbers.
        
    Returns:
        int: The maximum sum of the inner lists.
    '''
    max_sum = 0
    for inner_list in list_of_lists:
        current_sum = sum(inner_list)
        if current_sum > max_sum:
            max_sum = current_sum
    return max_sum

from tkinter import messagebox
from utils import find_max_sum
class MaxSumApp:
    '''
    Class to create the main application window and handle user interactions.
    '''
    def __init__(self, master):
        '''
        Initializes the GUI components.
        '''
        self.master = master
        master.title("Maximum Sum Finder")
        self.label = tk.Label(master, text="Enter a list of lists (e.g. [[1,2],[3,4],[5]])")
        self.label.pack()
        self.input_text = tk.Entry(master)
        self.input_text.pack()
        self.calculate_button = tk.Button(master, text="Calculate Max Sum", command=self.calculate_max_sum)
        self.calculate_button.pack()
        self.result_label = tk.Label(master, text="")
        self.result_label.pack()
    def calculate_max_sum(self):
        '''
        Retrieves input, calculates the maximum sum, and displays the result.
        '''
        try:
            input_data = eval(self.input_text.get())
            if not isinstance(input_data, list) or not all(isinstance(i, list) for i in input_data):
                raise ValueError("Input must be a list of lists.")
            max_sum = find_max_sum(input_data)
            self.result_label.config(text=f"Maximum Sum: {max_sum}")
        except Exception as e:
            messagebox.showerror("Input Error", str(e))

def find_max_sum(list_of_lists):
    '''
    Finds the maximum sum of elements of lists in a list of lists.
    Parameters:
        list_of_lists (list): A list containing multiple lists of numbers.
    Returns:
        int: The maximum sum of the inner lists.
    '''
    max_sum = float('-inf')  # Initialize to negative infinity to handle negative sums
    for inner_list in list_of_lists:
        current_sum = sum(inner_list)
        if current_sum > max_sum:
            max_sum = current_sum
    return max_sum

Main application file for the MaxSumApp.

import tkinter as tk
from tkinter import messagebox
import json
from utils import find_max_sum
class MaxSumApp:
    '''
    Class to create the main application window and handle user interactions.
    '''
    def __init__(self, master):
        '''
        Initializes the GUI components.
        '''
        self.master = master
        master.title("Maximum Sum Finder")
        self.label = tk.Label(master, text="Enter a list of lists (e.g. [[1,2],[3,4],[5]])")
        self.label.pack()
        self.input_text = tk.Entry(master)
        self.input_text.pack()
        self.calculate_button = tk.Button(master, text="Calculate Max Sum", command=self.calculate_max_sum)
        self.calculate_button.pack()
        self.result_label = tk.Label(master, text="")
        self.result_label.pack()
    def calculate_max_sum(self):
        '''
        Retrieves input, calculates the maximum sum, and displays the result.
        '''
        try:
            input_data = json.loads(self.input_text.get())
            if not isinstance(input_data, list) or not all(isinstance(i, list) for i in input_data):
                raise ValueError("Input must be a list of lists.")
            max_sum = find_max_sum(input_data)
            self.result_label.config(text=f"Maximum Sum: {max_sum}")
        except json.JSONDecodeError:
            messagebox.showerror("Input Error", "Invalid JSON format. Please enter a valid list of lists.")
        except Exception as e:
            messagebox.showerror("Input Error", str(e))
if __name__ == "__main__":
    root = tk.Tk()
    app = MaxSumApp(root)
    root.mainloop()

def find_max_sum(list_of_lists):
    '''
    Finds the maximum sum of elements of lists in a list of lists.
    Parameters:
        list_of_lists (list): A list containing multiple lists of numbers.
    Returns:
        int: The maximum sum of the inner lists.
    '''
    max_sum = float('-inf')  # Initialize to negative infinity to handle negative sums
    for inner_list in list_of_lists:
        current_sum = sum(inner_list)
        if current_sum > max_sum:
            max_sum = current_sum
    return max_sum

'''
Main application file for the MaxSumApp.

import tkinter as tk
from tkinter import messagebox
import json
from utils import find_max_sum
class MaxSumApp:
    '''
    Class to create the main application window and handle user interactions.
    '''
    def __init__(self, master):
        '''
        Initializes the GUI components.
        '''
        self.master = master
        master.title("Maximum Sum Finder")
        self.label = tk.Label(master, text="Enter a list of lists (e.g. [[1,2],[3,4],[5]])")
        self.label.pack()
        self.input_text = tk.Entry(master)
        self.input_text.pack()
        self.calculate_button = tk.Button(master, text="Calculate Max Sum", command=self.calculate_max_sum)
        self.calculate_button.pack()
        self.result_label = tk.Label(master, text="")
        self.result_label.pack()
    def calculate_max_sum(self):
        '''
        Retrieves input, calculates the maximum sum, and displays the result.
        '''
        try:
            input_data = json.loads(self.input_text.get())
            if not isinstance(input_data, list) or not all(isinstance(i, list) for i in input_data):
                raise ValueError("Input must be a list of lists.")
            max_sum = find_max_sum(input_data)
            self.result_label.config(text=f"Maximum Sum: {max_sum}")
        except json.JSONDecodeError:
            messagebox.showerror("Input Error", "Invalid JSON format. Please enter a valid list of lists.")
        except Exception as e:
            messagebox.showerror("Input Error", str(e))
if __name__ == "__main__":
    root = tk.Tk()
    app = MaxSumApp(root)
    root.mainloop()

'''
Utility functions for the MaxSumApp.

def find_max_sum(list_of_lists):
    '''
    Finds the maximum sum of elements of lists in a list of lists.
    Parameters:
        list_of_lists (list): A list containing multiple lists of numbers.
    Returns:
        int: The maximum sum of the inner lists.
    '''
    max_sum = float('-inf')  # Initialize to negative infinity to handle negative sums
    for inner_list in list_of_lists:
        current_sum = sum(inner_list)
        if current_sum > max_sum:
            max_sum = current_sum
    return max_sum
