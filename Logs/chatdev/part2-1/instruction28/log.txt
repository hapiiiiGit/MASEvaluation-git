Main application file that initializes the GUI and runs the application.
'''

from tkinter import Tk, Label, Entry, Button, StringVar
from odd_occurrence_finder import OddOccurrenceFinder

class App:
    def __init__(self, master):
        self.master = master
        master.title("Odd Occurrence Finder")

        self.label = Label(master, text="Enter numbers separated by commas:")
        self.label.pack()

        self.input_var = StringVar()
        self.entry = Entry(master, textvariable=self.input_var)
        self.entry.pack()

        self.find_button = Button(master, text="Find Odd Occurrence", command=self.find_odd)
        self.find_button.pack()

        self.result_label = Label(master, text="")
        self.result_label.pack()

    def find_odd(self):
        input_text = self.input_var.get()
        try:
            numbers = list(map(int, input_text.split(',')))
            finder = OddOccurrenceFinder()
            result = finder.find_odd_occurrences(numbers)
            self.result_label.config(text=f"Element occurring odd number of times: {result}")
        except ValueError:
            self.result_label.config(text="Please enter valid integers.")

if __name__ == "__main__":
    root = Tk()
    app = App(root)
    root.mainloop()
```

'''
Module that contains the logic to find the element occurring an odd number of times.
'''

class OddOccurrenceFinder:
    def find_odd_occurrences(self, arr):
        '''
        Finds the element that occurs an odd number of times in the given list.
        
        Parameters:
        arr (list): A list of integers.

        Returns:
        int: The element that occurs an odd number of times.
        '''
        count_map = {}
        for num in arr:
            count_map[num] = count_map.get(num, 0) + 1
        
        for num, count in count_map.items():
            if count % 2 != 0:
                return num
        return None
```

class OddOccurrenceFinder:
    def find_odd_occurrences(self, arr):
        '''
        Finds the element that occurs an odd number of times in the given list.

        arr (list): A list of integers.

class OddOccurrenceFinder:
    def find_odd_occurrences(self, arr):

                return num
        return None

'''
Module that contains the logic to find the element occurring an odd number of times.

class OddOccurrenceFinder:
    def find_odd_occurrences(self, arr):
        '''
        Finds the element that occurs an odd number of times in the given list.
        Parameters:
        arr (list): A list of integers.
        Returns:
        int: The element that occurs an odd number of times, or None if no such element exists.
        '''
        count_map = {}
        for num in arr:
            count_map[num] = count_map.get(num, 0) + 1
        for num, count in count_map.items():
            if count % 2 != 0:
                return num
        return None

'''
Module that contains the logic to find the element occurring an odd number of times.
'''
class OddOccurrenceFinder:
    def find_odd_occurrences(self, arr):
        '''
        Finds the element that occurs an odd number of times in the given list.
        Parameters:
        arr (list): A list of integers.
        Returns:
        int: The element that occurs an odd number of times, or None if no such element exists.
        '''
        count_map = {}
        for num in arr:
            count_map[num] = count_map.get(num, 0) + 1
        for num, count in count_map.items():
            if count % 2 != 0:
                return num
        return None
