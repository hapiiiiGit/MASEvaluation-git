class SublistChecker:
    def check_sublist(self, lst, sublist):
        if len(sublist) == 0:
            return True
        if len(sublist) > len(lst):
            return False

        for i in range(len(lst) - len(sublist) + 1):
            if lst[i:i + len(sublist)] == sublist:
                return True

        return False

'''
This file contains the SublistChecker class for checking whether a list contains a given sublist or not.
'''
class SublistChecker:
    def check_sublist(self, lst, sublist):
        if len(sublist) == 0:
            return True
        if len(sublist) > len(lst):
            return False
        for i in range(len(lst) - len(sublist) + 1):
            if lst[i:i + len(sublist)] == sublist:
                return True
        return False

class SublistChecker:
    def check_sublist(self, lst, sublist):
        if len(sublist) == 0:
            return True
        if len(sublist) > len(lst):
            return False
        for i in range(len(lst) - len(sublist) + 1):
            if lst[i:i + len(sublist)] == sublist:
                return True
        return False

class SublistChecker:
    def check_sublist(self, lst, sublist):
        if len(sublist) == 0:

'''
This file contains the SublistChecker class for checking whether a list contains a given sublist or not.
'''
class SublistChecker:
    def check_sublist(self, lst, sublist):
        if len(sublist) == 0:
            return True
        if len(sublist) > len(lst):
            return False
        for i in range(len(lst) - len(sublist) + 1):
            if lst[i:i + len(sublist)] == sublist:
                return True
        return False

import tkinter as tk
class GUI:
    def __init__(self):
        self.root = tk.Tk()

        label.pack()
    def get_user_input(self, prompt):
        entry_label = tk.Label(self.root, text=prompt)
        entry_label.pack()
        entry = tk.Entry(self.root)
        entry.pack()
        user_input = None
        def submit_input():
            nonlocal user_input
            user_input = entry.get()
            self.root.quit()
        submit_button = tk.Button(self.root, text="Submit", command=submit_input)
        submit_button.pack()
        self.root.mainloop()
        return user_input
