'''
Main application file for the Tuple Checker GUI.
'''

import tkinter as tk
from tkinter import messagebox
from tuple_checker import check_tuples

class App:
    def __init__(self, master):
        self.master = master
        master.title("Tuple Checker")

        self.label = tk.Label(master, text="Enter a list of tuples (e.g., [(1,2), (3,4)]):")
        self.label.pack()

        self.tuple_input = tk.Entry(master)
        self.tuple_input.pack()

        self.k_label = tk.Label(master, text="Enter the number of elements (k):")
        self.k_label.pack()

        self.k_input = tk.Entry(master)
        self.k_input.pack()

        self.check_button = tk.Button(master, text="Check Tuples", command=self.get_input)
        self.check_button.pack()

    def get_input(self):
        try:
            tuple_list = eval(self.tuple_input.get())
            k = int(self.k_input.get())
            result = check_tuples(tuple_list, k)
            self.display_result(result)
        except Exception as e:
            messagebox.showerror("Input Error", str(e))

    def display_result(self, result):
        if result:
            messagebox.showinfo("Result", "All tuples have exactly k elements.")
        else:
            messagebox.showinfo("Result", "Not all tuples have k elements.")

if __name__ == "__main__":
    root = tk.Tk()
    app = App(root)
    root.mainloop()
```

def check_tuples(tuple_list, k):
    '''
    Check if all tuples in the given list have exactly k elements.

    Parameters:
    tuple_list (list of tuples): The list of tuples to check.
    k (int): The number of elements each tuple should have.

    Returns:
    bool: True if all tuples have k elements, False otherwise.
    '''
    return all(len(tup) == k for tup in tuple_list)

def check_tuples(tuple_list, k):

'''

import tkinter as tk
from tkinter import messagebox
from tuple_checker import check_tuples
class App:
    def __init__(self, master):
        self.master = master
        master.title("Tuple Checker")
        self.label = tk.Label(master, text="Enter a list of tuples (e.g., [(1,2), (3,4)]):")
        self.label.pack()
        self.tuple_input = tk.Entry(master)
        self.tuple_input.pack()
        self.k_label = tk.Label(master, text="Enter the number of elements (k):")
        self.k_label.pack()
        self.k_input = tk.Entry(master)
        self.k_input.pack()
        self.check_button = tk.Button(master, text="Check Tuples", command=self.get_input)
        self.check_button.pack()
    def get_input(self):
        try:
            tuple_list = eval(self.tuple_input.get())
            k = int(self.k_input.get())
            result = check_tuples(tuple_list, k)
            self.display_result(result)
        except Exception as e:
            messagebox.showerror("Input Error", str(e))
    def display_result(self, result):
        if result:
            messagebox.showinfo("Result", "All tuples have exactly k elements.")
        else:
            messagebox.showinfo("Result", "Not all tuples have k elements.")
if __name__ == "__main__":
    root = tk.Tk()
    app = App(root)
    root.mainloop()

'''
Module for checking if all tuples in a list have k elements.
'''
def check_tuples(tuple_list, k):
    '''
    Check if all tuples in the given list have exactly k elements.
    Parameters:
    tuple_list (list of tuples): The list of tuples to check.
    k (int): The number of elements each tuple should have.
    Returns:
    bool: True if all tuples have k elements, False otherwise.
    '''
    return all(len(tup) == k for tup in tuple_list)

To run the Tuple Checker application, ensure you have Python installed on your machine. You can install the required dependencies using pip:

```bash

```

The Tuple Checker application allows users to input a list of tuples and a number (k) to verify if all tuples contain exactly k elements. This can be useful for data validation and ensuring consistency in tuple structures.

## 🛠️ Main Functions

- **Input List of Tuples**: Users can enter a list of tuples in the format `[(1,2), (3,4)]`.
- **Input Number of Elements (k)**: Users specify the number of elements each tuple should have.
- **Check Tuples**: The application checks if all tuples in the list have exactly k elements and displays the result.

## 📦 How to Use the Application

1. **Run the Application**: Execute the `main.py` file to launch the Tuple Checker GUI.

   ```bash
   python main.py
   ```

2. **Enter a List of Tuples**: In the input field, type your list of tuples. For example:
   ```
   [(1, 2), (3, 4)]
   ```

3. **Enter the Number of Elements (k)**: In the second input field, enter the integer value for k. For example:
   ```
   2
   ```

4. **Check Tuples**: Click the "Check Tuples" button to validate the tuples.

5. **View Results**: A message box will display whether all tuples have exactly k elements or not.
