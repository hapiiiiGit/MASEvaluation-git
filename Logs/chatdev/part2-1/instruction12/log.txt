'''
Main application file that initializes the GUI and runs the word counter.
'''

from tkinter import Tk, Label, Text, Button, Scrollbar, END
from word_counter import WordCounter

class App:
    def __init__(self, master):
        self.master = master
        master.title("Word Counter")

        self.label = Label(master, text="Enter text:")
        self.label.pack()

        self.text_area = Text(master, wrap='word', height=10, width=50)
        self.text_area.pack()

        self.scrollbar = Scrollbar(master, command=self.text_area.yview)
        self.scrollbar.pack(side='right', fill='y')
        self.text_area.config(yscrollcommand=self.scrollbar.set)

        self.count_button = Button(master, text="Count Words", command=self.run_counter)
        self.count_button.pack()

        self.result_label = Label(master, text="Most Common Words:")
        self.result_label.pack()

        self.result_area = Text(master, wrap='word', height=10, width=50)
        self.result_area.pack()

    def run_counter(self):
        text = self.text_area.get("1.0", END)
        word_counter = WordCounter(text)
        results = word_counter.get_most_common(10)  # Get top 10 most common words
        self.display_results(results)

    def display_results(self, results):
        self.result_area.delete("1.0", END)  # Clear previous results
        for word, count in results:
            self.result_area.insert(END, f"{word}: {count}\n")

if __name__ == "__main__":
    root = Tk()
    app = App(root)
    root.mainloop()
```

'''
Module that contains the WordCounter class for counting words in a given text.

from collections import Counter

class WordCounter:
    def __init__(self, text):
        self.text = text

    def count_words(self):
        words = self.text.split()
        return Counter(words)

    def get_most_common(self, n):
        word_counts = self.count_words()
        return word_counts.most_common(n)

from collections import Counter
class WordCounter:
    def __init__(self, text):
        self.text = text
    def count_words(self):
        words = self.text.split()
        return Counter(words)
    def get_most_common(self, n):
        word_counts = self.count_words()
        return word_counts.most_common(n)

from collections import Counter
import re
class WordCounter:
    def __init__(self, text):
        self.text = text
    def count_words(self):
        # Use regex to find words and ignore punctuation
        words = re.findall(r'\b\w+\b', self.text.lower())  # Convert to lower case for case-insensitivity
        return Counter(words)
    def get_most_common(self, n):

'''
Module that contains the WordCounter class for counting words in a given text.
'''
from collections import Counter
import re
class WordCounter:
    def __init__(self, text):
        self.text = text
    def count_words(self):
        # Use regex to find words and ignore punctuation
        words = re.findall(r'\b\w+\b', self.text.lower())  # Convert to lower case for case-insensitivity
        return Counter(words)
    def get_most_common(self, n):
        word_counts = self.count_words()
        return word_counts.most_common(n)
