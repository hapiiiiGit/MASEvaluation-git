'''
This file contains a function to find the number of divisors of a given integer.
The function handles cases where the input integer is less than 1.
'''
def count_divisors(num):
    if num < 1:
        return 0  # Return 0 for non-positive integers
    count = 0
    for i in range(1, int(num**0.5) + 1):  # Optimize by checking up to the square root
        if num % i == 0:
            count += 1
            if i != num // i:  # Count the complementary divisor
                count += 1
    return count

'''
This file contains a function to find the number of divisors of a given integer.
The function handles cases where the input integer is less than 1.
'''
def count_divisors(num):
    if num < 1:
        return 0  # Return 0 for non-positive integers
    count = 0
    for i in range(1, int(num**0.5) + 1):  # Optimize by checking up to the square root
        if num % i == 0:
            count += 1
            if i != num // i:  # Count the complementary divisor
                count += 1
    return count
