def min_rotations_to_same_string(input_string):
    # Function to find the minimum number of rotations required to get the same string
    if len(input_string) == 0:
        return 0
    
    rotations = 1
    original_string = input_string
    while True:
        input_string = input_string[-1] + input_string[:-1]  # Perform one rotation
        if input_string == original_string:
            return rotations
        rotations += 1

# Example usage

'''
This file contains the main function to find the minimum number of rotations required to get the same string.
'''
def min_rotations_to_same_string(input_string):
    # Function to find the minimum number of rotations required to get the same string
    if len(input_string) <= 1:  # Handle empty string or single character
        return 0
    rotations = 1
    original_string = input_string
    while True:
        input_string = input_string[-1] + input_string[:-1]  # Perform one rotation
        if input_string == original_string:
            return rotations
        rotations += 1
        if rotations > len(original_string):  # Prevent infinite loop
            return -1  # Indicate that it cannot return to original
# Example usage
if __name__ == "__main__":
    input_str = "programming"
    result = min_rotations_to_same_string(input_str)
    print(f"Minimum rotations required to get the same string: {result}")
