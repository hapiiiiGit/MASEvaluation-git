import tkinter as tk

class TriangleCheckGUI:
    def __init__(self):
        self.root = tk.Tk()
        self.root.title("Triangle Equilateral Checker")

        self.side1_label = tk.Label(self.root, text="Enter side 1:")
        self.side1_label.pack()
        self.side1_entry = tk.Entry(self.root)
        self.side1_entry.pack()

        self.side2_label = tk.Label(self.root, text="Enter side 2:")
        self.side2_label.pack()
        self.side2_entry = tk.Entry(self.root)
        self.side2_entry.pack()

        self.side3_label = tk.Label(self.root, text="Enter side 3:")
        self.side3_label.pack()
        self.side3_entry = tk.Entry(self.root)
        self.side3_entry.pack()

        self.check_button = tk.Button(self.root, text="Check Equilateral", command=self.check_triangle)
        self.check_button.pack()

    def check_triangle(self):
        side1 = float(self.side1_entry.get())
        side2 = float(self.side2_entry.get())
        side3 = float(self.side3_entry.get())

        result = "Equilateral" if side1 == side2 == side3 else "Not Equilateral"
        result_label = tk.Label(self.root, text=f"The triangle is {result}")
        result_label.pack()

'''
This file contains the GUI implementation for checking if a triangle is equilateral or not.

import tkinter as tk
from triangle_operations import is_equilateral
class TriangleCheckGUI:
    def __init__(self):
        self.root = tk.Tk()
        self.root.title("Triangle Equilateral Checker")
        self.side1_label = tk.Label(self.root, text="Enter side 1:")
        self.side1_label.pack()
        self.side1_entry = tk.Entry(self.root)
        self.side1_entry.pack()
        self.side2_label = tk.Label(self.root, text="Enter side 2:")
        self.side2_label.pack()
        self.side2_entry = tk.Entry(self.root)
        self.side2_entry.pack()
        self.side3_label = tk.Label(self.root, text="Enter side 3:")
        self.side3_label.pack()
        self.side3_entry = tk.Entry(self.root)
        self.side3_entry.pack()
        self.check_button = tk.Button(self.root, text="Check Equilateral", command=self.check_triangle)
        self.check_button.pack()
    def check_triangle(self):
        try:
            side1 = float(self.side1_entry.get())
            side2 = float(self.side2_entry.get())
            side3 = float(self.side3_entry.get())
            if side1 + side2 > side3 and side1 + side3 > side2 and side2 + side3 > side1:
                result = "Equilateral" if is_equilateral(side1, side2, side3) else "Not Equilateral"
                result_label = tk.Label(self.root, text=f"The triangle is {result}")
                result_label.pack()
            else:
                error_label = tk.Label(self.root, text="Invalid triangle sides. Please enter sides that form a valid triangle.")
                error_label.pack()
        except ValueError:
            error_label = tk.Label(self.root, text="Please enter valid numbers for all sides.")
            error_label.pack()
if __name__ == "__main__":
    gui = TriangleCheckGUI()
    gui.root.mainloop()
