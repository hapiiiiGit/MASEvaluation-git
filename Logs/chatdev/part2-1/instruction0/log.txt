'''
CostMatrix class to handle the cost matrix and calculate the minimum cost path.
'''

class CostMatrix:
    def __init__(self, cost):
        self.cost = cost
        self.rows = len(cost)
        self.cols = len(cost[0]) if self.rows > 0 else 0

    def min_cost_path(self, m, n):
        if m >= self.rows or n >= self.cols:
            raise ValueError("Target coordinates are out of bounds.")
        
        dp = [[0 for _ in range(self.cols)] for _ in range(self.rows)]
        
        dp[0][0] = self.cost[0][0]

        for i in range(1, self.rows):
            dp[i][0] = dp[i-1][0] + self.cost[i][0]

        for j in range(1, self.cols):
            dp[0][j] = dp[0][j-1] + self.cost[0][j]

        for i in range(1, self.rows):
            for j in range(1, self.cols):
                dp[i][j] = self.cost[i][j] + min(dp[i-1][j], dp[i][j-1])

        return dp[m][n]
```

def create_input_frame(master, calculate_callback):
    frame = tk.Frame(master)
    frame.pack(pady=10)

    tk.Label(frame, text="Cost Matrix (comma-separated rows):").grid(row=0, column=0)
    cost_matrix_entry = tk.Entry(frame, width=50)
    cost_matrix_entry.grid(row=0, column=1)

    tk.Label(frame, text="Target Row (m):").grid(row=1, column=0)
    m_entry = tk.Entry(frame, width=10)
    m_entry.grid(row=1, column=1)

    tk.Label(frame, text="Target Column (n):").grid(row=2, column=0)
    n_entry = tk.Entry(frame, width=10)
    n_entry.grid(row=2, column=1)

    def on_calculate():
        cost_matrix_str = cost_matrix_entry.get()
        cost_matrix = [list(map(int, row.split(','))) for row in cost_matrix_str.split(';')]
        m = m_entry.get()
        n = n_entry.get()
        calculate_callback(cost_matrix, m, n)

    calculate_button = tk.Button(frame, text="Calculate Minimum Cost Path", command=on_calculate)
    calculate_button.grid(row=3, columnspan=2)

    return {'cost_matrix': cost_matrix_entry, 'm': m_entry, 'n': n_entry}

    frame.pack(pady=10)

'''
CostMatrix class to handle the cost matrix and calculate the minimum cost path.
'''
class CostMatrix:
    def __init__(self, cost):
        self.cost = cost
        self.rows = len(cost)
        self.cols = len(cost[0]) if self.rows > 0 else 0
    def min_cost_path(self, m, n):
        if m >= self.rows or n >= self.cols:
            raise ValueError("Target coordinates are out of bounds.")
        dp = [[0 for _ in range(self.cols)] for _ in range(self.rows)]
        dp[0][0] = self.cost[0][0]
        for i in range(1, self.rows):
            dp[i][0] = dp[i-1][0] + self.cost[i][0]
        for j in range(1, self.cols):
            dp[0][j] = dp[0][j-1] + self.cost[0][j]
        for i in range(1, self.rows):
            for j in range(1, self.cols):
                dp[i][j] = self.cost[i][j] + min(dp[i-1][j], dp[i][j-1])
        return dp[m][n]

    n_entry = tk.Entry(frame, width=10)
    n_entry.grid(row=2, column=1)
    def on_calculate():
        cost_matrix_str = cost_matrix_entry.get()
        cost_matrix = [list(map(int, row.split(','))) for row in cost_matrix_str.split(';')]
        m = m_entry.get()
        n = n_entry.get()
        calculate_callback(cost_matrix, m, n)
    calculate_button = tk.Button(frame, text="Calculate Minimum Cost Path", command=on_calculate)
    calculate_button.grid(row=3, columnspan=2)
    return {'cost_matrix': cost_matrix_entry, 'm': m_entry, 'n': n_entry}

'''
CostMatrix class to handle the cost matrix and calculate the minimum cost path.
'''
class CostMatrix:
    def __init__(self, cost):
        self.cost = cost
        self.rows = len(cost)
        self.cols = len(cost[0]) if self.rows > 0 else 0
    def min_cost_path(self, m, n):
        if m >= self.rows or n >= self.cols:
            raise ValueError("Target coordinates are out of bounds.")
        dp = [[0 for _ in range(self.cols)] for _ in range(self.rows)]
        dp[0][0] = self.cost[0][0]
        for i in range(1, self.rows):
            dp[i][0] = dp[i-1][0] + self.cost[i][0]
        for j in range(1, self.cols):
            dp[0][j] = dp[0][j-1] + self.cost[0][j]
        for i in range(1, self.rows):
            for j in range(1, self.cols):
                dp[i][j] = self.cost[i][j] + min(dp[i-1][j], dp[i][j-1])
        return dp[m][n]

'''
CostMatrix class to handle the cost matrix and calculate the minimum cost path.
'''
class CostMatrix:
    def __init__(self, cost):
        self.cost = cost
        self.rows = len(cost)
        self.cols = len(cost[0]) if self.rows > 0 else 0
    def min_cost_path(self, m, n):
        if m >= self.rows or n >= self.cols:
            raise ValueError("Target coordinates are out of bounds.")
        dp = [[0 for _ in range(self.cols)] for _ in range(self.rows)]
        dp[0][0] = self.cost[0][0]
        for i in range(1, self.rows):
            dp[i][0] = dp[i-1][0] + self.cost[i][0]
        for j in range(1, self.cols):
            dp[0][j] = dp[0][j-1] + self.cost[0][j]
        for i in range(1, self.rows):
            for j in range(1, self.cols):
                dp[i][j] = self.cost[i][j] + min(dp[i-1][j], dp[i][j-1])
        return dp[m][n]

'''
CostMatrix class to handle the cost matrix and calculate the minimum cost path.
'''
class CostMatrix:
    def __init__(self, cost):
        self.cost = cost
        self.rows = len(cost)
        self.cols = len(cost[0]) if self.rows > 0 else 0
    def min_cost_path(self, m, n):
        if m >= self.rows or n >= self.cols:
            raise ValueError("Target coordinates are out of bounds.")
        dp = [[0 for _ in range(self.cols)] for _ in range(self.rows)]
        dp[0][0] = self.cost[0][0]
        for i in range(1, self.rows):
            dp[i][0] = dp[i-1][0] + self.cost[i][0]
        for j in range(1, self.cols):
            dp[0][j] = dp[0][j-1] + self.cost[0][j]
        for i in range(1, self.rows):
            for j in range(1, self.cols):
                dp[i][j] = self.cost[i][j] + min(dp[i-1][j], dp[i][j-1])
        return dp[m][n]

        dp = [[0 for _ in range(self.cols)] for _ in range(self.rows)]
        dp[0][0] = self.cost[0][0]
        for i in range(1, self.rows):
            dp[i][0] = dp[i-1][0] + self.cost[i][0]
        for j in range(1, self.cols):
            dp[0][j] = dp[0][j-1] + self.cost[0][j]
        for i in range(1, self.rows):
            for j in range(1, self.cols):
                dp[i][j] = self.cost[i][j] + min(dp[i-1][j], dp[i][j-1])
        return dp[m][n]

'''
CostMatrix class to handle the cost matrix and calculate the minimum cost path.
'''
class CostMatrix:
    def __init__(self, cost):
        self.cost = cost
        self.rows = len(cost)
        self.cols = len(cost[0]) if self.rows > 0 else 0
    def min_cost_path(self, m, n):
        if m >= self.rows or n >= self.cols:
            raise ValueError("Target coordinates are out of bounds.")
        dp = [[0 for _ in range(self.cols)] for _ in range(self.rows)]
        dp[0][0] = self.cost[0][0]
        for i in range(1, self.rows):
            dp[i][0] = dp[i-1][0] + self.cost[i][0]
        for j in range(1, self.cols):
            dp[0][j] = dp[0][j-1] + self.cost[0][j]
        for i in range(1, self.rows):
            for j in range(1, self.cols):
                dp[i][j] = self.cost[i][j] + min(dp[i-1][j], dp[i][j-1])
        return dp[m][n]

'''
CostMatrix class to handle the cost matrix and calculate the minimum cost path.
'''
class CostMatrix:
    def __init__(self, cost):
        self.cost = cost
        self.rows = len(cost)
        self.cols = len(cost[0]) if self.rows > 0 else 0
    def min_cost_path(self, m, n):
        if m >= self.rows or n >= self.cols:
            raise ValueError("Target coordinates are out of bounds.")
        dp = [[0 for _ in range(self.cols)] for _ in range(self.rows)]
        dp[0][0] = self.cost[0][0]
        for i in range(1, self.rows):
            dp[i][0] = dp[i-1][0] + self.cost[i][0]
        for j in range(1, self.cols):
            dp[0][j] = dp[0][j-1] + self.cost[0][j]
        for i in range(1, self.rows):
            for j in range(1, self.cols):
                dp[i][j] = self.cost[i][j] + min(dp[i-1][j], dp[i][j-1])
        return dp[m][n]

'''
CostMatrix class to handle the cost matrix and calculate the minimum cost path.
'''
class CostMatrix:
    def __init__(self, cost):
        self.cost = cost
        self.rows = len(cost)
        self.cols = len(cost[0]) if self.rows > 0 else 0

    def min_cost_path(self, m, n):
        if m >= self.rows or n >= self.cols:
            raise ValueError("Target coordinates are out of bounds.")
        dp = [[0 for _ in range(self.cols)] for _ in range(self.rows)]
        dp[0][0] = self.cost[0][0]
        for i in range(1, self.rows):
            dp[i][0] = dp[i-1][0] + self.cost[i][0]
        for j in range(1, self.cols):
            dp[0][j] = dp[0][j-1] + self.cost[0][j]
        for i in range(1, self.rows):
            for j in range(1, self.cols):
                dp[i][j] = self.cost[i][j] + min(dp[i-1][j], dp[i][j-1])
        return dp[m][n]

'''
CostMatrix class to handle the cost matrix and calculate the minimum cost path.
'''
class CostMatrix:
    def __init__(self, cost):
        self.cost = cost
        self.rows = len(cost)
        self.cols = len(cost[0]) if self.rows > 0 else 0
    def min_cost_path(self, m, n):
        if m >= self.rows or n >= self.cols:
            raise ValueError("Target coordinates are out of bounds.")
        dp = [[0 for _ in range(self.cols)] for _ in range(self.rows)]
        dp[0][0] = self.cost[0][0]
        for i in range(1, self.rows):
            dp[i][0] = dp[i-1][0] + self.cost[i][0]
        for j in range(1, self.cols):
            dp[0][j] = dp[0][j-1] + self.cost[0][j]
        for i in range(1, self.rows):
            for j in range(1, self.cols):
                dp[i][j] = self.cost[i][j] + min(dp[i-1][j], dp[i][j-1])
        return dp[m][n]

import tkinter as tk

def create_input_frame(master, calculate_callback):
    frame = tk.Frame(master)
    frame.pack(pady=10)
    tk.Label(frame, text="Cost Matrix (comma-separated rows):").grid(row=0, column=0)
    cost_matrix_entry = tk.Entry(frame, width=50)
    cost_matrix_entry.grid(row=0, column=1)
    tk.Label(frame, text="Target Row (m):").grid(row=1, column=0)
    m_entry = tk.Entry(frame, width=10)
    m_entry.grid(row=1, column=1)
    tk.Label(frame, text="Target Column (n):").grid(row=2, column=0)
    n_entry = tk.Entry(frame, width=10)
    n_entry.grid(row=2, column=1)
    def on_calculate():
        try:
            cost_matrix_str = cost_matrix_entry.get()
            cost_matrix = [list(map(int, row.split(','))) for row in cost_matrix_str.split(',')]
            m = int(m_entry.get())
            n = int(n_entry.get())
            calculate_callback(cost_matrix, m, n)
        except ValueError as e:
            messagebox.showerror("Error", "Invalid input: " + str(e))
    calculate_button = tk.Button(frame, text="Calculate Minimum Cost Path", command=on_calculate)
    calculate_button.grid(row=3, columnspan=2)
    return {'cost_matrix': cost_matrix_entry, 'm': m_entry, 'n': n_entry}

'''
CostMatrix class to handle the cost matrix and calculate the minimum cost path.
'''
class CostMatrix:
    def __init__(self, cost):
        self.cost = cost
        self.rows = len(cost)
        self.cols = len(cost[0]) if self.rows > 0 else 0
    def min_cost_path(self, m, n):
        if m >= self.rows or n >= self.cols:
            raise ValueError("Target coordinates are out of bounds.")
        dp = [[0 for _ in range(self.cols)] for _ in range(self.rows)]
        dp[0][0] = self.cost[0][0]
        for i in range(1, self.rows):
            dp[i][0] = dp[i-1][0] + self.cost[i][0]
        for j in range(1, self.cols):
            dp[0][j] = dp[0][j-1] + self.cost[0][j]
        for i in range(1, self.rows):
            for j in range(1, self.cols):
                dp[i][j] = self.cost[i][j] + min(dp[i-1][j], dp[i][j-1])
        return dp[m][n]
