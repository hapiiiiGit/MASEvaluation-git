'''
Module containing the function to split a string at lowercase letters.

def split_at_lowercase(input_string):
    """
    Splits the input string at each lowercase letter and returns a list of substrings.
    
    Args:
        input_string (str): The string to be split.
        
    Returns:
        list: A list of substrings split at lowercase letters.
    """
    result = []
    current_substring = ""

    for char in input_string:
        if char.islower():
            if current_substring:  # If there's a current substring, add it to the result
                result.append(current_substring)
                current_substring = ""
        current_substring += char

    if current_substring:  # Add the last substring if it exists
        result.append(current_substring)

    return result

def split_at_lowercase(input_string):
    """
    Splits the input string at each lowercase letter and returns a list of substrings.
    Args:
        input_string (str): The string to be split.
    Returns:
        list: A list of substrings split at lowercase letters.
    """
    result = []

                result.append(current_substring)
                current_substring = ""
        current_substring += char
    if current_substring:  # Add the last substring if it exists
        result.append(current_substring)
    return result

'''
Module containing the function to split a string at lowercase letters.
'''
def split_at_lowercase(input_string):
    """
    Splits the input string at each lowercase letter and returns a list of substrings.
    Args:
        input_string (str): The string to be split.
    Returns:
        list: A list of substrings split at lowercase letters.
    """
    result = []
    current_substring = ""
    for char in input_string:
        if char.islower():
            if current_substring:  # If there's a current substring, add it to the result
                result.append(current_substring)
                current_substring = ""
        current_substring += char
    if current_substring:  # Add the last substring if it exists
        result.append(current_substring)
    return result
