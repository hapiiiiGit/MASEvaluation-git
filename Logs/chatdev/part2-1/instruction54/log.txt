'''
This file contains the GeometricSeriesCalculator class responsible for calculating the t-nth term of a geometric series.
'''
class GeometricSeriesCalculator:
    def calculate_nth_term(self, a, r, n):
        """
        Calculate the t-nth term of a geometric series.
        Parameters:
        a (float): The first term of the series.
        r (float): The common ratio.
        n (int): The term number to calculate.
        Returns:
        float: The t-nth term of the geometric series.
        """
        return a * (r ** (n - 1))

'''
This file contains the GeometricSeriesCalculator class responsible for calculating the t-nth term of a geometric series.
'''
class GeometricSeriesCalculator:
    def calculate_nth_term(self, a, r, n):
        """
        Calculate the t-nth term of a geometric series.
        Parameters:
        a (float): The first term of the series.
        r (float): The common ratio.
        n (int): The term number to calculate.
        Returns:
        float: The t-nth term of the geometric series.
        """
        return a * (r ** (n - 1))

'''
This file contains the GeometricSeriesGUI class responsible for setting up the graphical user interface for the geometric series calculator.
'''
import tkinter as tk
from geometric_series_calculator import GeometricSeriesCalculator
class GeometricSeriesGUI:
    @staticmethod
    def display_gui():
        """
        Set up the graphical user interface for the geometric series calculator.
        """
        root = tk.Tk()
        root.title("Geometric Series Calculator")
        label_a = tk.Label(root, text="Enter the first term (a):")
        label_a.pack()
        entry_a = tk.Entry(root)
        entry_a.pack()
        label_r = tk.Label(root, text="Enter the common ratio (r):")
        label_r.pack()
        entry_r = tk.Entry(root)
        entry_r.pack()
        label_n = tk.Label(root, text="Enter the term number (n):")
        label_n.pack()
        entry_n = tk.Entry(root)
        entry_n.pack()
        result_label = tk.Label(root, text="Result:")
        result_label.pack()
        result_var = tk.StringVar()
        result_display = tk.Label(root, textvariable=result_var)
        result_display.pack()
        calculate_button = tk.Button(root, text="Calculate", command=lambda: GeometricSeriesGUI.handle_calculation(entry_a.get(), entry_r.get(), entry_n.get(), result_var))
        calculate_button.pack()
        root.mainloop()
    @staticmethod
    def handle_calculation(a, r, n, result_var):
        """
        Handle the calculation of the t-nth term and update the result label.
        Parameters:
        a (str): The first term input from the user.
        r (str): The common ratio input from the user.
        n (str): The term number input from the user.
        result_var (StringVar): The variable to update the result label.
        """
        try:
            a = float(a)
            r = float(r)
            n = int(n)
            result = GeometricSeriesCalculator().calculate_nth_term(a, r, n)
            result_var.set(str(result))
        except ValueError:
            result_var.set("Invalid input. Please enter numeric values.")
