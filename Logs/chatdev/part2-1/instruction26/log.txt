'''
Main application file for the Digit Remover application.
'''

import tkinter as tk
from tkinter import messagebox
from utils import remove_digits_from_list

class DigitRemoverApp:
    '''
    GUI application to remove digits from a list of strings.
    '''

    def __init__(self, master):
        self.master = master
        master.title("Digit Remover")

        self.label = tk.Label(master, text="Enter strings (comma-separated):")
        self.label.pack()

        self.input_text = tk.Entry(master, width=50)
        self.input_text.pack()

        self.remove_button = tk.Button(master, text="Remove Digits", command=self.remove_digits)
        self.remove_button.pack()

        self.result_label = tk.Label(master, text="Result:")
        self.result_label.pack()

        self.result_text = tk.Text(master, height=10, width=50)
        self.result_text.pack()

    def remove_digits(self):
        '''
        Removes digits from the input strings and displays the result.
        '''
        input_str = self.input_text.get()
        string_list = [s.strip() for s in input_str.split(',')]
        result = remove_digits_from_list(string_list)
        self.result_text.delete(1.0, tk.END)  # Clear previous results
        self.result_text.insert(tk.END, '\n'.join(result))

if __name__ == "__main__":
    root = tk.Tk()
    app = DigitRemoverApp(root)
    root.mainloop()
```

'''

def remove_digits_from_list(string_list):
    '''
    Removes all digits from each string in the provided list.
    
    Parameters:
    string_list (list): A list of strings from which digits will be removed.

    Returns:
    list: A new list of strings with digits removed.
    '''
    return [''.join(filter(lambda x: not x.isdigit(), s)) for s in string_list]
```

'''
Main application file for the Digit Remover application.
'''
import tkinter as tk
from tkinter import messagebox
from utils import remove_digits_from_list
class DigitRemoverApp:
    '''
    GUI application to remove digits from a list of strings.
    '''
    def __init__(self, master):
        self.master = master
        master.title("Digit Remover")
        self.label = tk.Label(master, text="Enter strings (comma-separated):")
        self.label.pack()
        self.input_text = tk.Entry(master, width=50)
        self.input_text.pack()
        self.remove_button = tk.Button(master, text="Remove Digits", command=self.remove_digits)
        self.remove_button.pack()
        self.result_label = tk.Label(master, text="Result:")
        self.result_label.pack()
        self.result_text = tk.Text(master, height=10, width=50)
        self.result_text.pack()
    def remove_digits(self):
        '''
        Removes digits from the input strings and displays the result.
        '''
        input_str = self.input_text.get()
        string_list = [s.strip() for s in input_str.split(',')]
        result = remove_digits_from_list(string_list)
        self.result_text.delete(1.0, tk.END)  # Clear previous results
        self.result_text.insert(tk.END, '\n'.join(result))
if __name__ == "__main__":
    root = tk.Tk()
    app = DigitRemoverApp(root)
    root.mainloop()

Utility functions for the Digit Remover application.

def remove_digits_from_list(string_list):
    '''
    Removes all digits from each string in the provided list.
    Parameters:
    string_list (list): A list of strings from which digits will be removed.
    Returns:
    list: A new list of strings with digits removed.
    '''
    return [''.join(filter(lambda x: not x.isdigit(), s)) for s in string_list]
