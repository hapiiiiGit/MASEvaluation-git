### Core Classes and Functions

1. **Main Application Class (`SmallestNumberApp`)**: This class will handle the main application window and user interactions.
   - **Methods**:
     - `__init__`: Initializes the application and sets up the GUI components.
     - `find_smallest`: Retrieves the input from the user, processes it, and displays the smallest number.

```python
'''
Main application file for finding the smallest number in a list using a GUI.
'''

import tkinter as tk
from tkinter import messagebox
from utils import validate_input

class SmallestNumberApp:
    '''
    Class to create the GUI application for finding the smallest number in a list.
    '''

    def __init__(self, master):
        '''
        Initializes the main application window and its components.
        '''
        self.master = master
        master.title("Find Smallest Number")

        self.label = tk.Label(master, text="Enter numbers separated by commas:")
        self.label.pack()

        self.entry = tk.Entry(master)
        self.entry.pack()

        self.find_button = tk.Button(master, text="Find Smallest", command=self.find_smallest)
        self.find_button.pack()

        self.result_label = tk.Label(master, text="")
        self.result_label.pack()

    def find_smallest(self):
        '''
        Retrieves the input, finds the smallest number, and displays the result.
        '''
        user_input = self.entry.get()
        numbers = validate_input(user_input)

        if numbers is not None:
            smallest_number = min(numbers)
            self.result_label.config(text=f"The smallest number is: {smallest_number}")
        else:
            messagebox.showerror("Input Error", "Please enter a valid list of numbers.")

if __name__ == "__main__":
    root = tk.Tk()
    app = SmallestNumberApp(root)
    root.mainloop()
```

### File: `utils.py`

```python
'''
Utility functions for the application, including input validation.
'''

def validate_input(user_input):
    '''
    Validates the user input and converts it into a list of numbers.
    Returns a list of numbers if valid, otherwise returns None.
    '''
    try:
        # Split the input string by commas and convert to float
        numbers = [float(num.strip()) for num in user_input.split(',')]
        return numbers
    except ValueError:
        return None
```

'''
Main application file for finding the smallest number in a list using a GUI.

import tkinter as tk
from tkinter import messagebox
from utils import validate_input
class SmallestNumberApp:
    '''
    Class to create the GUI application for finding the smallest number in a list.
    '''
    def __init__(self, master):
        '''
        Initializes the main application window and its components.
        '''
        self.master = master
        master.title("Find Smallest Number")
        self.label = tk.Label(master, text="Enter numbers separated by commas:")
        self.label.pack()
        self.entry = tk.Entry(master)
        self.entry.pack()
        self.find_button = tk.Button(master, text="Find Smallest", command=self.find_smallest)
        self.find_button.pack()
        self.result_label = tk.Label(master, text="")
        self.result_label.pack()
    def find_smallest(self):
        '''
        Retrieves the input, finds the smallest number, and displays the result.
        '''
        user_input = self.entry.get()
        numbers = validate_input(user_input)
        if numbers is not None:
            smallest_number = min(numbers)
            self.result_label.config(text=f"The smallest number is: {smallest_number}")
        else:
            messagebox.showerror("Input Error", "Please enter a valid list of numbers.")
if __name__ == "__main__":
    root = tk.Tk()
    app = SmallestNumberApp(root)
    root.mainloop()

def validate_input(user_input):
    '''
    Validates the user input and converts it into a list of numbers.
    Returns a list of numbers if valid, otherwise returns None.
    '''
    if not user_input:
        return None
    try:
        # Split the input string by commas, check for numeric values, and convert to float
        numbers = [float(num.strip()) for num in user_input.split(',') if num.strip().replace('.', '', 1).replace('-', '', 1).replace(' ', '').replace('.', '', 1).isdigit()]
        return numbers
    except ValueError:
        return None
