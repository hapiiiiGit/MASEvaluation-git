'''

This file contains the implementation of the HeapQueue class for finding the largest integers using the heap queue algorithm.

'''

import heapq

class HeapQueue:
    def heapify(arr, n, i):
        largest = i
        l = 2 * i + 1
        r = 2 * i + 2

        if l < n and arr[i] < arr[l]:
            largest = l

        if r < n and arr[largest] < arr[r]:
            largest = r

        if largest != i:
            arr[i], arr[largest] = arr[largest], arr[i]
            heapify(arr, n, largest)

    def find_largest_integers(numbers):
        heap = []
        for num in numbers:
            heapq.heappush(heap, num)
            if len(heap) > 3:
                heapq.heappop(heap)
        
        return sorted(heap, reverse=True)
```
