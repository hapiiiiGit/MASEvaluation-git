'''
Main application file for the Smallest Items GUI.

from tkinter import Tk, Label, Entry, Button, Text, END
from smallest_items import get_n_smallest_items

class SmallestItemsApp:
    '''
    Main application class for the Smallest Items GUI.
    '''
    
    def __init__(self, master):
        self.master = master
        master.title("Get n Smallest Items")

        self.label = Label(master, text="Enter dataset (comma-separated):")
        self.label.pack()

        self.dataset_entry = Entry(master)
        self.dataset_entry.pack()

        self.n_label = Label(master, text="Enter n (number of smallest items):")
        self.n_label.pack()

        self.n_entry = Entry(master)
        self.n_entry.pack()

        self.get_button = Button(master, text="Get Smallest Items", command=self.get_input_data)
        self.get_button.pack()

        self.result_text = Text(master, height=10, width=50)
        self.result_text.pack()

    def get_input_data(self):
        '''
        Retrieve user input from the GUI and display results.
        '''
        dataset = self.dataset_entry.get()
        n = self.n_entry.get()

        try:
            dataset = list(map(int, dataset.split(',')))
            n = int(n)
            smallest_items = get_n_smallest_items(dataset, n)
            self.display_results(smallest_items)
        except ValueError:
            self.result_text.delete(1.0, END)
            self.result_text.insert(END, "Invalid input. Please enter integers only.")

    def display_results(self, smallest_items):
        '''
        Display the results in the GUI.
        '''
        self.result_text.delete(1.0, END)
        self.result_text.insert(END, f"The {len(smallest_items)} smallest items are: {smallest_items}")

if __name__ == "__main__":
    root = Tk()
    app = SmallestItemsApp(root)
    root.mainloop()

'''
Main application file for the Smallest Items GUI.

from tkinter import Tk, Label, Entry, Button, Text, END
from smallest_items import get_n_smallest_items
class SmallestItemsApp:
    '''
    Main application class for the Smallest Items GUI.
    '''
    def __init__(self, master):
        self.master = master
        master.title("Get n Smallest Items")
        self.label = Label(master, text="Enter dataset (comma-separated):")
        self.label.pack()
        self.dataset_entry = Entry(master)
        self.dataset_entry.pack()
        self.n_label = Label(master, text="Enter n (number of smallest items):")
        self.n_label.pack()
        self.n_entry = Entry(master)
        self.n_entry.pack()
        self.get_button = Button(master, text="Get Smallest Items", command=self.get_input_data)
        self.get_button.pack()
        self.result_text = Text(master, height=10, width=50)
        self.result_text.pack()
    def get_input_data(self):
        '''
        Retrieve user input from the GUI and display results.
        '''
        dataset = self.dataset_entry.get()
        n = self.n_entry.get()
        try:
            dataset = list(map(int, dataset.split(',')))
            n = int(n)
            smallest_items = get_n_smallest_items(dataset, n)
            self.display_results(smallest_items)
        except ValueError:
            self.result_text.delete(1.0, END)
            self.result_text.insert(END, "Invalid input. Please enter integers only.")
    def display_results(self, smallest_items):
        '''
        Display the results in the GUI.
        '''
        self.result_text.delete(1.0, END)
        self.result_text.insert(END, f"The {len(smallest_items)} smallest items are: {smallest_items}")
if __name__ == "__main__":
    root = Tk()
    app = SmallestItemsApp(root)
    root.mainloop()

import heapq

    Parameters:
    data (list): The dataset from which to retrieve the smallest items.

    Returns:
    list: A list containing the n smallest items.
    '''
    return heapq.nsmallest(n, data)

from tkinter import Tk, Label, Entry, Button, Text, END
from smallest_items import get_n_smallest_items
class SmallestItemsApp:
    '''
    Main application class for the Smallest Items GUI.
    '''
    def __init__(self, master):
        self.master = master
        master.title("Get n Smallest Items")
        self.label = Label(master, text="Enter dataset (comma-separated):")
        self.label.pack()
        self.dataset_entry = Entry(master)
        self.dataset_entry.pack()
        self.n_label = Label(master, text="Enter n (number of smallest items):")
        self.n_label.pack()
        self.n_entry = Entry(master)
        self.n_entry.pack()
        self.get_button = Button(master, text="Get Smallest Items", command=self.get_input_data)
        self.get_button.pack()
        self.result_text = Text(master, height=10, width=50)
        self.result_text.pack()
    def get_input_data(self):
        '''
        Retrieve user input from the GUI and display results.
        '''
        dataset = self.dataset_entry.get()
        n = self.n_entry.get()
        try:
            dataset = list(map(int, dataset.split(',')))
            n = int(n)
            smallest_items = get_n_smallest_items(dataset, n)
            self.display_results(smallest_items)
        except ValueError:
            self.result_text.delete(1.0, END)
            self.result_text.insert(END, "Invalid input. Please enter integers only.")
    def display_results(self, smallest_items):
        '''
        Display the results in the GUI.
        '''
        self.result_text.delete(1.0, END)
        self.result_text.insert(END, f"The {len(smallest_items)} smallest items are: {smallest_items}")
if __name__ == "__main__":
    root = Tk()
    app = SmallestItemsApp(root)
    root.mainloop()
