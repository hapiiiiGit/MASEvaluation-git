'''
Module to count the number of squares in a rectangle.

class RectangleSquareCounter:
    def count_squares(self, width, height):
        '''
        Calculate the total number of squares that can fit in a rectangle
        of given width and height.
        '''
        total_squares = 0
        # Count squares of size 1x1 to min(width, height)xmin(width, height)
        for size in range(1, min(width, height) + 1):
            total_squares += (width - size + 1) * (height - size + 1)
        return total_squares

class RectangleSquareCounter:

        pass
    def count_squares(self, width, height):
        '''
        Calculate the total number of squares that can fit in a rectangle
        of given width and height.

        for size in range(1, min(width, height) + 1):
            total_squares += (width - size + 1) * (height - size + 1)
        return total_squares

from tkinter import Tk, Label, Entry, Button, StringVar
from rectangle_square_counter import RectangleSquareCounter
class App:
    def __init__(self, master):
        self.master = master
        master.title("Rectangle Square Counter")
        self.label_width = Label(master, text="Width of Rectangle:")
        self.label_width.pack()

        self.label_height.pack()

        self.result_label.pack()

            self.result_var.set("Please enter valid integers.")

'''
Module to count the number of squares in a rectangle.
'''
class RectangleSquareCounter:
    def __init__(self):
        pass
    def count_squares(self, width, height):
        '''
        Calculate the total number of squares that can fit in a rectangle
        of given width and height.
        '''
        total_squares = 0
        # Count squares of size 1x1 to min(width, height)xmin(width, height)
        for size in range(1, min(width, height) + 1):
            total_squares += (width - size + 1) * (height - size + 1)
        return total_squares
