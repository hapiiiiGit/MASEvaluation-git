'''
This is the main file for the Even Number Filter application.
It uses tkinter to create a GUI that allows users to input a list of numbers,
filters the even numbers using a lambda function, and displays the results.
'''

import tkinter as tk
from tkinter import messagebox

class EvenNumberFilterApp:
    def __init__(self, master):
        self.master = master
        master.title("Even Number Filter")

        self.label = tk.Label(master, text="Enter numbers separated by commas:")
        self.label.pack()

        self.entry = tk.Entry(master)
        self.entry.pack()

        self.filter_button = tk.Button(master, text="Filter Even Numbers", command=self.filter_even_numbers)
        self.filter_button.pack()

        self.result_label = tk.Label(master, text="")
        self.result_label.pack()

    def filter_even_numbers(self):
        input_text = self.entry.get()
        try:
            # Convert input string to a list of integers
            numbers = list(map(int, input_text.split(',')))
            # Filter even numbers using a lambda function
            even_numbers = list(filter(lambda x: x % 2 == 0, numbers))
            # Display the result
            self.result_label.config(text=f"Even Numbers: {even_numbers}")
        except ValueError:
            messagebox.showerror("Input Error", "Please enter valid integers separated by commas.")

if __name__ == "__main__":
    root = tk.Tk()
    app = EvenNumberFilterApp(root)
    root.mainloop()
```

import tkinter as tk
from tkinter import messagebox
class EvenNumberFilterApp:
    def __init__(self, master):
        self.master = master
        master.title("Even Number Filter")
        self.label = tk.Label(master, text="Enter numbers separated by commas:")
        self.label.pack()

'''
This is the main file for the Even Number Filter application.
It uses tkinter to create a GUI that allows users to input a list of numbers,
filters the even numbers using a lambda function, and displays the results.
'''
import tkinter as tk
from tkinter import messagebox
class EvenNumberFilterApp:
    def __init__(self, master):
        self.master = master
        master.title("Even Number Filter")
        self.label = tk.Label(master, text="Enter numbers separated by commas:")
        self.label.pack()
        self.entry = tk.Entry(master)
        self.entry.pack()
        self.filter_button = tk.Button(master, text="Filter Even Numbers", command=self.filter_even_numbers)
        self.filter_button.pack()
        self.result_label = tk.Label(master, text="")
        self.result_label.pack()
    def filter_even_numbers(self):
        input_text = self.entry.get()
        try:
            # Convert input string to a list of integers
            numbers = list(map(int, input_text.split(',')))
            # Filter even numbers using a lambda function
            even_numbers = list(filter(lambda x: x % 2 == 0, numbers))
            # Display the result
            self.result_label.config(text=f"Even Numbers: {even_numbers}")
        except ValueError:
            messagebox.showerror("Input Error", "Please enter valid integers separated by commas.")
if __name__ == "__main__":
    root = tk.Tk()
    app = EvenNumberFilterApp(root)
    root.mainloop()
