'''
Main application file for finding the maximum length of the subsequence
with a difference between adjacent elements in a given array.
'''

from tkinter import Tk, Label, Entry, Button, StringVar, messagebox
from utils import is_valid_input

class MaxSubsequenceApp:
    def __init__(self, master):
        self.master = master
        master.title("Max Length Subsequence Finder")

        self.label = Label(master, text="Enter a list of integers (comma-separated):")
        self.label.pack()

        self.input_var = StringVar()
        self.input_entry = Entry(master, textvariable=self.input_var)
        self.input_entry.pack()

        self.calculate_button = Button(master, text="Calculate", command=self.calculate_max_length)
        self.calculate_button.pack()

        self.result_label = Label(master, text="")
        self.result_label.pack()

    def calculate_max_length(self):
        input_str = self.input_var.get()
        input_list = input_str.split(',')
        
        if not is_valid_input(input_list):
            messagebox.showerror("Invalid Input", "Please enter a valid list of integers.")
            return
        
        input_list = list(map(int, input_list))
        max_length = self.find_max_length_subsequence(input_list)
        self.result_label.config(text=f"Maximum Length of Subsequence: {max_length}")

    def find_max_length_subsequence(self, arr):
        if not arr:
            return 0
        
        n = len(arr)
        max_length = 1
        current_length = 1
        
        for i in range(1, n):
            if abs(arr[i] - arr[i - 1]) == 1:
                current_length += 1
            else:
                max_length = max(max_length, current_length)
                current_length = 1
        
        max_length = max(max_length, current_length)
        return max_length

if __name__ == "__main__":
    root = Tk()
    app = MaxSubsequenceApp(root)
    root.mainloop()
```

from utils import is_valid_input
class MaxSubsequenceApp:
    def __init__(self, master):
        self.master = master
        master.title("Max Length Subsequence Finder")
        self.label = Label(master, text="Enter a list of integers (comma-separated):")
        self.label.pack()
        self.input_var = StringVar()
        self.input_entry = Entry(master, textvariable=self.input_var)
        self.input_entry.pack()
        self.calculate_button = Button(master, text="Calculate", command=self.calculate_max_length)
        self.calculate_button.pack()

        self.result_label.pack()

        if not is_valid_input(input_list):
            messagebox.showerror("Invalid Input", "Please enter a valid list of integers.")
            return
        input_list = list(map(int, input_list))
        max_length = self.find_max_length_subsequence(input_list)

        if not arr:

        n = len(arr)
        max_length = 1
        current_length = 1
        for i in range(1, n):
            if abs(arr[i] - arr[i - 1]) == 1:
                current_length += 1
            else:
                max_length = max(max_length, current_length)
                current_length = 1
        max_length = max(max_length, current_length)
        return max_length

from tkinter import Tk, Label, Entry, Button, StringVar, messagebox  # Added messagebox import
from utils import is_valid_input
class MaxSubsequenceApp:
    def __init__(self, master):
        self.master = master
        master.title("Max Length Subsequence Finder")
        self.label = Label(master, text="Enter a list of integers (comma-separated):")
        self.label.pack()
        self.input_var = StringVar()
        self.input_entry = Entry(master, textvariable=self.input_var)
        self.input_entry.pack()
        self.calculate_button = Button(master, text="Calculate", command=self.calculate_max_length)
        self.calculate_button.pack()
        self.result_label = Label(master, text="")
        self.result_label.pack()
    def calculate_max_length(self):
        input_str = self.input_var.get()
        input_list = input_str.split(',')
        if not is_valid_input(input_list):
            messagebox.showerror("Invalid Input", "Please enter a valid list of integers.")
            return
        input_list = list(map(int, input_list))
        max_length = self.find_max_length_subsequence(input_list)
        self.result_label.config(text=f"Maximum Length of Subsequence: {max_length}")
    def find_max_length_subsequence(self, arr):
        if not arr:
            return 0
        n = len(arr)
        max_length = 1
        current_length = 1
        for i in range(1, n):
            if abs(arr[i] - arr[i - 1]) == 1:
                current_length += 1
            else:
                max_length = max(max_length, current_length)
                current_length = 1
        max_length = max(max_length, current_length)
        return max_length
if __name__ == "__main__":
    root = Tk()
    app = MaxSubsequenceApp(root)
    root.mainloop()

'''
Main application file for finding the maximum length of the subsequence
with a difference between adjacent elements in a given array.
'''
from tkinter import Tk, Label, Entry, Button, StringVar, messagebox  # Added messagebox import
from utils import is_valid_input
class MaxSubsequenceApp:
    def __init__(self, master):
        self.master = master
        master.title("Max Length Subsequence Finder")
        self.label = Label(master, text="Enter a list of integers (comma-separated):")
        self.label.pack()
        self.input_var = StringVar()
        self.input_entry = Entry(master, textvariable=self.input_var)
        self.input_entry.pack()
        self.calculate_button = Button(master, text="Calculate", command=self.calculate_max_length)
        self.calculate_button.pack()
        self.result_label = Label(master, text="")
        self.result_label.pack()
    def calculate_max_length(self):
        input_str = self.input_var.get()
        if not input_str:
            messagebox.showerror("Empty Input", "Please enter a list of integers.")
            return
        input_list = input_str.split(',')
        if not is_valid_input(input_list):
            messagebox.showerror("Invalid Input", "Please enter a valid list of integers.")
            return
        input_list = list(map(int, input_list))
        max_length = self.find_max_length_subsequence(input_list)
        self.result_label.config(text=f"Maximum Length of Subsequence: {max_length}")
    def find_max_length_subsequence(self, arr):
        if not arr:
            return 0
        n = len(arr)
        max_length = 1
        current_length = 1
        for i in range(1, n):
            if abs(arr[i] - arr[i - 1]) == 1:
                current_length += 1
            else:
                max_length = max(max_length, current_length)
                current_length = 1
        max_length = max(max_length, current_length)
        return max_length
if __name__ == "__main__":
    root = Tk()
    app = MaxSubsequenceApp(root)
    root.mainloop()

'''
Main application file for finding the maximum length of the subsequence
with a difference between adjacent elements in a given array.
'''
from tkinter import Tk, Label, Entry, Button, StringVar, messagebox  # Added messagebox import
from utils import is_valid_input
class MaxSubsequenceApp:
    def __init__(self, master):
        self.master = master
        master.title("Max Length Subsequence Finder")
        self.label = Label(master, text="Enter a list of integers (comma-separated):")
        self.label.pack()
        self.input_var = StringVar()
        self.input_entry = Entry(master, textvariable=self.input_var)
        self.input_entry.pack()
        self.calculate_button = Button(master, text="Calculate", command=self.calculate_max_length)
        self.calculate_button.pack()
        self.result_label = Label(master, text="")
        self.result_label.pack()
    def calculate_max_length(self):
        input_str = self.input_var.get()
        if not input_str:
            messagebox.showerror("Empty Input", "Please enter a list of integers.")
            return
        input_list = input_str.split(',')
        if not is_valid_input(input_list):
            messagebox.showerror("Invalid Input", "Please enter a valid list of integers.")
            return
        input_list = list(map(int, input_list))
        max_length = self.find_max_length_subsequence(input_list)
        self.result_label.config(text=f"Maximum Length of Subsequence: {max_length}")
    def find_max_length_subsequence(self, arr):
        if not arr:
            return 0
        n = len(arr)
        max_length = 1
        current_length = 1
        for i in range(1, n):
            if abs(arr[i] - arr[i - 1]) == 1:
                current_length += 1
            else:
                max_length = max(max_length, current_length)
                current_length = 1
        max_length = max(max_length, current_length)
        return max_length
if __name__ == "__main__":
    root = Tk()
    app = MaxSubsequenceApp(root)
    root.mainloop()
