'''
Module that contains the logic for finding the first duplicate in an array of integers.
'''
class DuplicateFinder:
    def __init__(self, array):
        self.array = array
    def first_duplicate(self):
        """
        Finds the first duplicate number in the array.
        Uses a set to track seen numbers for efficient lookup.
        Returns the first duplicate number found, or None if no duplicates exist.
        """
        seen = set()
        for number in self.array:
            if number in seen:
                return number  # Return the first duplicate found
            seen.add(number)  # Add the number to the set if not seen
        return None
