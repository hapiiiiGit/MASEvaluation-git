'''
Utility functions for the application, including the calculation of the sum of repeated elements.

def sum_of_repeated_elements(arr):
    element_count = {}
    repeated_sum = 0
    
    # Count occurrences of each element
    for num in arr:
        if num in element_count:
            element_count[num] += 1
        else:
            element_count[num] = 1
    
    # Calculate the sum of repeated elements
    for num, count in element_count.items():
        if count > 1:
            repeated_sum += num * count
            
    return repeated_sum

def sum_of_repeated_elements(arr):

def sum_of_repeated_elements(arr):
    element_count = {}
    repeated_sum = 0
    # Count occurrences of each element
    for num in arr:
        if num in element_count:
            element_count[num] += 1
        else:
            element_count[num] = 1
    # Calculate the sum of repeated elements
    for num, count in element_count.items():
        if count > 1:
            repeated_sum += num * count
    return repeated_sum

def sum_of_repeated_elements(arr):

    # Calculate the sum of repeated elements

        self.input_var = StringVar()

            self.result_label.config(text="Please enter valid integers, separated by commas.")

'''
Utility functions for the application, including the calculation of the sum of repeated elements.
'''
def sum_of_repeated_elements(arr):
    element_count = {}
    repeated_sum = 0
    # Count occurrences of each element
    for num in arr:
        if num in element_count:
            element_count[num] += 1
        else:
            element_count[num] = 1
    # Calculate the sum of repeated elements
    for num, count in element_count.items():
        if count > 1:
            repeated_sum += num * count
    return repeated_sum
