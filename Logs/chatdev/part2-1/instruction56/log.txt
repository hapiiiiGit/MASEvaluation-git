import tkinter as tk
from tkinter import messagebox
from helper import find_largest_number

class LargestNumberFinderApp:
    def __init__(self, master):
        self.master = master

import tkinter as tk

class LargestNumberFinderApp:

        if digits_list:

        for digit in digits_str.split():
            if digit.isdigit():  # Check if the digit is a non-negative integer
                digits_list.append(int(digit))
            else:
                messagebox.showerror("Error", f"Invalid input: '{digit}'. Please enter only non-negative integers.")
                return  # Exit the method if invalid input is found
        if digits_list:
            largest_number = find_largest_number(digits_list)
            messagebox.showinfo("Largest Number", f"The largest number that can be formed is: {largest_number}")
        else:
            messagebox.showerror("Error", "Please enter valid digits.")

'''
This file contains the helper function to find the largest number that can be formed with given digits.
'''
def find_largest_number(digits):
    '''
    This function takes a list of non-negative integers (digits) and returns the largest number
    that can be formed by arranging those digits in descending order.
    Parameters:
    digits (list): A list of non-negative integers.
    Returns:
    int: The largest number that can be formed.
    '''
    digits.sort(reverse=True)
    largest_number = ''.join(map(str, digits))
    return int(largest_number)
