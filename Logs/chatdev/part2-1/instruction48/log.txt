class ExtractorApp:

class ExtractorApp:

'''
This file contains the main logic for extracting elements from a two-dimensional list.
'''
class ExtractorApp:
    @staticmethod
    def extract_elements(data, index=None):
        extracted_elements = []
        for sublist in data:
            if index is not None and 0 <= index < len(sublist):
                extracted_elements.append(sublist[index])
            else:
                extracted_elements.append(sublist[0])
        return extracted_elements

'''
This file contains the GUI implementation for the ExtractorApp.
'''
import tkinter as tk
from tkinter import messagebox
import json
class ExtractorGUI:
    def __init__(self):
        self.root = tk.Tk()
        self.root.title("Extractor App")
        self.data_entry = tk.Entry(self.root, width=50)
        self.index_entry = tk.Entry(self.root, width=10)
        self.extract_button = tk.Button(self.root, text="Extract", command=self.handle_extract)
        self.result_label = tk.Label(self.root, text="Extracted Elements:")
        self.result_text = tk.Text(self.root, height=10, width=50)
    def run(self):
        self.data_entry.pack()
        self.index_entry.pack()
        self.extract_button.pack()
        self.result_label.pack()
        self.result_text.pack()
        self.root.mainloop()
    def handle_extract(self):
        data_str = self.data_entry.get()
        index_str = self.index_entry.get()
        try:
            data = json.loads(data_str)
            index = int(index_str) if index_str else None
            extracted_elements = ExtractorApp.extract_elements(data, index)
            self.result_text.delete(1.0, tk.END)
            self.result_text.insert(tk.END, str(extracted_elements))
        except json.JSONDecodeError:
            messagebox.showerror("Error", "Invalid input format. Please enter a valid two-dimensional list.")
        except Exception as e:
            messagebox.showerror("Error", f"An error occurred: {str(e)}")

'''
This file contains the main logic for extracting elements from a two-dimensional list.
'''
class ExtractorApp:
    @staticmethod
    def extract_elements(data, index=None):
        extracted_elements = []
        for sublist in data:
            if index is not None and 0 <= index < len(sublist):
                extracted_elements.append(sublist[index])
            else:
                extracted_elements.append(sublist[0])
        return extracted_elements

'''

import tkinter as tk
from tkinter import messagebox
import json
class ExtractorGUI:
    def __init__(self):
        self.root = tk.Tk()
        self.root.title("Extractor App")
        self.data_entry = tk.Entry(self.root, width=50)
        self.index_entry = tk.Entry(self.root, width=10)
        self.extract_button = tk.Button(self.root, text="Extract", command=self.handle_extract)
        self.result_label = tk.Label(self.root, text="Extracted Elements:")
        self.result_text = tk.Text(self.root, height=10, width=50)
    def run(self):
        self.data_entry.pack()
        self.index_entry.pack()
        self.extract_button.pack()
        self.result_label.pack()
        self.result_text.pack()
        self.root.mainloop()
    def handle_extract(self):
        data_str = self.data_entry.get()
        index_str = self.index_entry.get()
        try:
            data = json.loads(data_str)
            index = int(index_str) if index_str else None
            extracted_elements = ExtractorApp.extract_elements(data, index)
            self.result_text.delete(1.0, tk.END)
            self.result_text.insert(tk.END, str(extracted_elements))
        except json.JSONDecodeError:
            messagebox.showerror("Error", "Invalid input format. Please enter a valid two-dimensional list.")
        except Exception as e:
            messagebox.showerror("Error", f"An error occurred: {str(e)}")
