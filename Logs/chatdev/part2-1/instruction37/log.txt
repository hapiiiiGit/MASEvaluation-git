'''
Main application file for the division of first even and odd numbers.
'''

from tkinter import Tk, Label, Entry, Button, StringVar
from utils import find_first_even, find_first_odd

class MainApp:
    def __init__(self):
        self.root = Tk()
        self.root.title("Even and Odd Division Calculator")

        self.input_label = Label(self.root, text="Enter a list of numbers (comma-separated):")
        self.input_label.pack()

        self.input_entry = Entry(self.root)
        self.input_entry.pack()

        self.result_var = StringVar()
        self.result_label = Label(self.root, textvariable=self.result_var)
        self.result_label.pack()

        self.calculate_button = Button(self.root, text="Calculate Division", command=self.calculate_division)
        self.calculate_button.pack()

        self.root.mainloop()

    def calculate_division(self):
        input_text = self.input_entry.get()
        try:
            numbers = list(map(int, input_text.split(',')))
            even_number = find_first_even(numbers)
            odd_number = find_first_odd(numbers)

            if even_number is None or odd_number is None:
                self.result_var.set("Even or Odd number not found in the list.")
            else:
                division_result = even_number / odd_number
                self.result_var.set(f"Division of first even ({even_number}) and odd ({odd_number}) is: {division_result}")
        except ValueError:
            self.result_var.set("Please enter valid integers.")

if __name__ == "__main__":
    MainApp()
```

    return None

    Returns the first odd number from the list.
    '''

        if number % 2 != 0:
            return number
    return None

'''
Main application file for the division of first even and odd numbers.

from tkinter import Tk, Label, Entry, Button, StringVar
from utils import find_first_even, find_first_odd
class MainApp:
    def __init__(self):
        self.root = Tk()
        self.root.title("Even and Odd Division Calculator")
        self.input_label = Label(self.root, text="Enter a list of numbers (comma-separated):")
        self.input_label.pack()
        self.input_entry = Entry(self.root)
        self.input_entry.pack()
        self.result_var = StringVar()
        self.result_label = Label(self.root, textvariable=self.result_var)
        self.result_label.pack()
        self.calculate_button = Button(self.root, text="Calculate Division", command=self.calculate_division)
        self.calculate_button.pack()
        self.root.mainloop()
    def calculate_division(self):
        input_text = self.input_entry.get()
        try:
            numbers = list(map(int, input_text.split(',')))
            even_number = find_first_even(numbers)
            odd_number = find_first_odd(numbers)
            if even_number is None or odd_number is None:
                self.result_var.set("Even or Odd number not found in the list.")
            else:
                division_result = even_number / odd_number
                self.result_var.set(f"Division of first even ({even_number}) and odd ({odd_number}) is: {division_result:.2f}")
        except ValueError:
            self.result_var.set("Please enter valid integers.")
if __name__ == "__main__":
    MainApp()

def find_first_even(numbers):
    '''
    Returns the first even number from the list.
    '''
    for number in numbers:
        if number % 2 == 0:
            return number
    return None
def find_first_odd(numbers):
    '''
    Returns the first odd number from the list.
    '''
    for number in numbers:
        if number % 2 != 0:
            return number
    return None

'''
Main application file for the division of first even and odd numbers.

from tkinter import Tk, Label, Entry, Button, StringVar
from utils import find_first_even, find_first_odd
class MainApp:
    def __init__(self):
        self.root = Tk()
        self.root.title("Even and Odd Division Calculator")
        self.input_label = Label(self.root, text="Enter a list of numbers (comma-separated):")
        self.input_label.pack()
        self.input_entry = Entry(self.root)
        self.input_entry.pack()
        self.result_var = StringVar()
        self.result_label = Label(self.root, textvariable=self.result_var)
        self.result_label.pack()
        self.calculate_button = Button(self.root, text="Calculate Division", command=self.calculate_division)
        self.calculate_button.pack()
        self.root.mainloop()
    def calculate_division(self):
        input_text = self.input_entry.get()
        try:
            numbers = list(map(int, input_text.split(',')))
            even_number = find_first_even(numbers)
            odd_number = find_first_odd(numbers)
            if even_number is None or odd_number is None:
                self.result_var.set("Even or Odd number not found in the list.")
            else:
                division_result = even_number / odd_number
                self.result_var.set(f"Division of first even ({even_number}) and odd ({odd_number}) is: {division_result:.2f}")
        except ValueError:
            self.result_var.set("Please enter valid integers.")
if __name__ == "__main__":
    MainApp()

def find_first_even(numbers):
    '''
    Returns the first even number from the list.
    '''
    for number in numbers:
        if number % 2 == 0:
            return number
    return None
def find_first_odd(numbers):
    '''
    Returns the first odd number from the list.
    '''
    for number in numbers:
        if number % 2 != 0:
            return number
    return None
