import tkinter as tk

class CombSortGUI:
    def __init__(self, root):
        self.root = root
        self.root.title("Comb Sort GUI")

        self.input_label = tk.Label(self.root, text="Enter numbers separated by space:")
        self.input_label.pack()

        self.input_entry = tk.Entry(self.root)
        self.input_entry.pack()

        self.sort_button = tk.Button(self.root, text="Sort", command=self.sort_list)
        self.sort_button.pack()

        self.output_label = tk.Label(self.root, text="Sorted List:")
        self.output_label.pack()

        self.output_text = tk.Text(self.root, height=10, width=50)
        self.output_text.pack()

    def sort_list(self):
        input_list = list(map(int, self.input_entry.get().split()))
        sorted_list = self.comb_sort(input_list)
        self.update_display(sorted_list)

    def comb_sort(self, arr):
        gap = len(arr)
        shrink = 1.3
        swapped = True

        while gap > 1 or swapped:
            gap = max(1, int(gap / shrink))
            swapped = False

            for i in range(len(arr) - gap):
                if arr[i] > arr[i + gap]:
                    arr[i], arr[i + gap] = arr[i + gap], arr[i]
                    swapped = True

        return arr

    def update_display(self, sorted_list):
        self.output_text.delete(1.0, tk.END)
        self.output_text.insert(tk.END, ' '.join(map(str, sorted_list)))
```

import tkinter as tk
class CombSortGUI:
    def __init__(self, root):
        self.root = root
        self.root.title("Comb Sort GUI")
        self.input_label = tk.Label(self.root, text="Enter numbers separated by space:")
        self.input_label.pack()
        self.input_entry = tk.Entry(self.root)
        self.input_entry.pack()
        self.sort_button = tk.Button(self.root, text="Sort", command=self.sort_list)
        self.sort_button.pack()
        self.output_label = tk.Label(self.root, text="Sorted List:")
        self.output_label.pack()
        self.output_text = tk.Text(self.root, height=10, width=50)
        self.output_text.pack()
    def sort_list(self):
        input_list = list(map(int, self.input_entry.get().split()))
        sorted_list = self.comb_sort(input_list)
        self.update_display(sorted_list)
    def comb_sort(self, arr):
        gap = len(arr)
        shrink = 1.3
        swapped = True
        while gap > 1 or swapped:
            gap = max(1, int(gap / shrink))
            swapped = False
            for i in range(len(arr) - gap):
                if arr[i] > arr[i + gap]:
                    arr[i], arr[i + gap] = arr[i + gap], arr[i]
                    swapped = True
        return arr  # Ensure the sorted list is returned correctly
    def update_display(self, sorted_list):
        self.output_text.delete(1.0, tk.END)
        self.output_text.insert(tk.END, ' '.join(map(str, sorted_list)))

import tkinter as tk
from tkinter import messagebox
class CombSortGUI:
    def __init__(self, root):
        self.root = root
        self.root.title("Comb Sort GUI")
        self.input_label = tk.Label(self.root, text="Enter numbers separated by space:")
        self.input_label.pack()
        self.input_entry = tk.Entry(self.root)
        self.input_entry.pack()
        self.sort_button = tk.Button(self.root, text="Sort", command=self.sort_list)
        self.sort_button.pack()
        self.output_label = tk.Label(self.root, text="Sorted List:")
        self.output_label.pack()
        self.output_text = tk.Text(self.root, height=10, width=50)
        self.output_text.pack()
    def sort_list(self):
        try:
            input_list = list(map(int, self.input_entry.get().split()))
            sorted_list = self.comb_sort(input_list)
            self.update_display(sorted_list)
        except ValueError:
            # Handle the case where non-integer values are entered
            messagebox.showerror("Error", "Please enter valid integers separated by space.")
        except Exception as e:
            # Handle any other unexpected errors
            messagebox.showerror("Error", f"An unexpected error occurred: {str(e)}")
    def comb_sort(self, arr):
        gap = len(arr)
        shrink = 1.3
        swapped = True
        while gap > 1 or swapped:
            gap = max(1, int(gap / shrink))
            swapped = False
            for i in range(len(arr) - gap):
                if arr[i] > arr[i + gap]:
                    arr[i], arr[i + gap] = arr[i + gap], arr[i]
                    swapped = True
        return arr  # Ensure the sorted list is returned correctly
    def update_display(self, sorted_list):
        self.output_text.delete(1.0, tk.END)
        self.output_text.insert(tk.END, ' '.join(map(str, sorted_list)))

import tkinter as tk
from tkinter import messagebox
class CombSortGUI:
    def __init__(self, root):
        self.root = root
        self.root.title("Comb Sort GUI")
        self.input_label = tk.Label(self.root, text="Enter numbers separated by space:")
        self.input_label.pack()
        self.input_entry = tk.Entry(self.root)
        self.input_entry.pack()
        self.sort_button = tk.Button(self.root, text="Sort", command=self.sort_list)
        self.sort_button.pack()
        self.output_label = tk.Label(self.root, text="Sorted List:")
        self.output_label.pack()
        self.output_text = tk.Text(self.root, height=10, width=50)
        self.output_text.pack()
    def sort_list(self):
        try:
            input_list = list(map(int, self.input_entry.get().split()))
            sorted_list = self.comb_sort(input_list)
            self.update_display(sorted_list)
        except ValueError:
            # Handle the case where non-integer values are entered
            messagebox.showerror("Error", "Please enter valid integers separated by space.")
        except Exception as e:
            # Handle any other unexpected errors
            messagebox.showerror("Error", f"An unexpected error occurred: {str(e)}")
    def comb_sort(self, arr):
        gap = len(arr)
        shrink = 1.3
        swapped = True
        while gap > 1 or swapped:
            gap = max(1, int(gap / shrink))
            swapped = False
            for i in range(len(arr) - gap):
                if arr[i] > arr[i + gap]:
                    arr[i], arr[i + gap] = arr[i + gap], arr[i]
                    swapped = True
        return arr  # Ensure the sorted list is returned correctly
    def update_display(self, sorted_list):
        self.output_text.delete(1.0, tk.END)
        self.output_text.insert(tk.END, ' '.join(map(str, sorted_list)))
