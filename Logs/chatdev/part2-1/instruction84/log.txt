import tkinter as tk
from sphere import Sphere
class SphereSurfaceAreaApp:
    def __init__(self, master):
        self.master = master
        self.master.title("Sphere Surface Area Calculator")
        self.radius_label = tk.Label(master, text="Enter the radius of the sphere:")
        self.radius_label.pack()
        self.radius_entry = tk.Entry(master)
        self.radius_entry.pack()
        self.calculate_button = tk.Button(master, text="Calculate Surface Area", command=self.calculate_surface_area)
        self.calculate_button.pack()
        self.result_label = tk.Label(master, text="")
        self.result_label.pack()
    def calculate_surface_area(self):
        try:
            radius = float(self.radius_entry.get())
            if radius < 0:
                raise ValueError("Radius cannot be negative.")
            surface_area = Sphere.calculate_surface_area(radius)
            self.result_label.config(text=f"Surface Area: {surface_area:.2f}")
        except ValueError as e:
            self.result_label.config(text=f"Error: {e}")
if __name__ == "__main__":
    root = tk.Tk()
    app = SphereSurfaceAreaApp(root)
    root.mainloop()
