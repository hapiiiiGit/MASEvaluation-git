import tkinter as tk
from tkinter import messagebox
from newman_conway import newman_conway

class NewmanConwayApp:
    '''
    GUI application for calculating the n-th number in the Newman-Conway sequence.
    '''
    
    def __init__(self, master):
        self.master = master
        master.title("Newman-Conway Sequence Calculator")

        self.label = tk.Label(master, text="Enter a number (n):")
        self.label.pack()

        self.entry = tk.Entry(master)
        self.entry.pack()

        self.calculate_button = tk.Button(master, text="Calculate", command=self.calculate)
        self.calculate_button.pack()

        self.result_label = tk.Label(master, text="")
        self.result_label.pack()

    def calculate(self):
        '''
        Calculate the n-th number in the Newman-Conway sequence and display the result.
        '''
        try:
            n = int(self.entry.get())
            if n < 1:
                raise ValueError("n must be a positive integer.")
            result = newman_conway(n)
            self.result_label.config(text=f"The {n}-th number in the Newman-Conway sequence is: {result}")
        except ValueError as e:
            messagebox.showerror("Input Error", str(e))

if __name__ == "__main__":
    root = tk.Tk()
    app = NewmanConwayApp(root)
    root.mainloop()

'''
Module containing the function to calculate the n-th number in the Newman-Conway sequence.
'''

def newman_conway(n):
    '''
    Calculate the n-th number in the Newman-Conway sequence.
    
    The Newman-Conway sequence is defined as:
    - P(1) = 1
    - P(2) = 1
    - P(n) = P(P(n-1)) + P(n-P(n-1)) for n > 2
    '''
    if n < 1:
        raise ValueError("n must be a positive integer.")
    
    # Initialize the sequence with the first two values
    p = [0, 1, 1]  # p[0] is unused, p[1] = 1, p[2] = 1

    for i in range(3, n + 1):
        p.append(p[p[i - 1]] + p[i - p[i - 1]])

    return p[n]
```

def newman_conway(n):
    '''
    Calculate the n-th number in the Newman-Conway sequence.
    The Newman-Conway sequence is defined as:
    - P(1) = 1
    - P(2) = 1
    - P(n) = P(P(n-1)) + P(n-P(n-1)) for n > 2
    '''
    if n < 1:
        raise ValueError("n must be a positive integer.")
    # Initialize the sequence with the first two values
    p = [0, 1, 1]  # p[0] is unused, p[1] = 1, p[2] = 1
    for i in range(3, n + 1):
        p.append(p[p[i - 1]] + p[i - p[i - 1]])
    return p[n]

def newman_conway(n):
    '''
    Calculate the n-th number in the Newman-Conway sequence.
    The Newman-Conway sequence is defined as:
    - P(1) = 1
    - P(2) = 1
    - P(n) = P(P(n-1)) + P(n-P(n-1)) for n > 2
    '''
    if n < 1:
        raise ValueError("n must be a positive integer.")
    # Initialize the sequence with the first two values
    p = [0, 1, 1]  # p[0] is unused, p[1] = 1, p[2] = 1
    for i in range(3, n + 1):
        p.append(p[p[i - 1]] + p[i - p[i - 1]])
    return p[n]

def newman_conway(n):
    '''
    Calculate the n-th number in the Newman-Conway sequence.
    The Newman-Conway sequence is defined as:
    - P(1) = 1
    - P(2) = 1
    - P(n) = P(P(n-1)) + P(n-P(n-1)) for n > 2
    '''
    if n < 1:
        raise ValueError("n must be a positive integer.")
    # Initialize the sequence with the first two values
    p = [0, 1, 1]  # p[0] is unused, p[1] = 1, p[2] = 1
    for i in range(3, n + 1):
        p.append(p[p[i - 1]] + p[i - p[i - 1]])
    return p[n]

'''
Module containing the function to calculate the n-th number in the Newman-Conway sequence.
'''
def newman_conway(n):
    '''
    Calculate the n-th number in the Newman-Conway sequence.
    The Newman-Conway sequence is defined as:
    - P(1) = 1
    - P(2) = 1
    - P(n) = P(P(n-1)) + P(n-P(n-1)) for n > 2
    '''
    if n < 1:
        raise ValueError("n must be a positive integer.")
    if n == 1:
        return 1
    if n == 2:
        return 1
    p = [0, 1, 1]  # p[0] is unused, p[1] = 1, p[2] = 1
    for i in range(3, n + 1):
        p.append(p[p[i - 1]] + p[i - p[i - 1]])
    return p[n]

'''
Module containing the function to calculate the n-th number in the Newman-Conway sequence.
'''
def newman_conway(n):
    '''
    Calculate the n-th number in the Newman-Conway sequence.
    The Newman-Conway sequence is defined as:
    - P(1) = 1
    - P(2) = 1
    - P(n) = P(P(n-1)) + P(n-P(n-1)) for n > 2
    '''
    if n < 1:
        raise ValueError("n must be a positive integer.")
    if n == 1:
        return 1
    if n == 2:
        return 1
    p = [0, 1, 1]  # p[0] is unused, p[1] = 1, p[2] = 1
    for i in range(3, n + 1):
        p.append(p[p[i - 1]] + p[i - p[i - 1]])
    return p[n]
