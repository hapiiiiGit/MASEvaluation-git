import tkinter as tk
from string_processor import StringProcessor

class MainApp:
    def __init__(self, root):
        self.root = root
        self.root.title("Character Remover")

        # Input fields
        self.label1 = tk.Label(root, text="Enter first string:")
        self.label1.pack()
        self.input1 = tk.Entry(root)
        self.input1.pack()

        self.label2 = tk.Label(root, text="Enter second string:")
        self.label2.pack()
        self.input2 = tk.Entry(root)
        self.input2.pack()

        # Remove button
        self.remove_button = tk.Button(root, text="Remove Characters", command=self.remove_characters)
        self.remove_button.pack()

        # Output label
        self.output_label = tk.Label(root, text="")
        self.output_label.pack()

    def remove_characters(self):
        str1 = self.input1.get()
        str2 = self.input2.get()
        result = StringProcessor.remove_chars(str1, str2)
        self.output_label.config(text=f"Result: {result}")

if __name__ == "__main__":
    root = tk.Tk()
    app = MainApp(root)
    root.mainloop()
