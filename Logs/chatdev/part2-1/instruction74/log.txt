'''
Main application file for the Divisible Tuples application.
'''

import tkinter as tk
from tkinter import messagebox
from utils import find_divisible_tuples

class DivisibleTuplesApp:
    '''
    Main application class for the Divisible Tuples GUI.
    '''
    
    def __init__(self, master):
        self.master = master
        master.title("Divisible Tuples Finder")

        self.label = tk.Label(master, text="Enter tuples (e.g., (1, 2), (3, 4), (5, 6)):")
        self.label.pack()

        self.tuple_input = tk.Entry(master)
        self.tuple_input.pack()

        self.k_label = tk.Label(master, text="Enter k:")
        self.k_label.pack()

        self.k_input = tk.Entry(master)
        self.k_input.pack()

        self.submit_button = tk.Button(master, text="Submit", command=self.on_submit)
        self.submit_button.pack()

        self.clear_button = tk.Button(master, text="Clear", command=self.clear_results)
        self.clear_button.pack()

        self.result_label = tk.Label(master, text="Results:")
        self.result_label.pack()

        self.result_text = tk.Text(master, height=10, width=50)
        self.result_text.pack()

    def on_submit(self):
        '''
        Handle the submission of user input.
        '''
        try:
            tuples_input = eval(self.tuple_input.get())
            k = int(self.k_input.get())
            if not isinstance(tuples_input, tuple) or not all(isinstance(t, tuple) for t in tuples_input):
                raise ValueError("Input must be a tuple of tuples.")
            result = find_divisible_tuples(tuples_input, k)
            self.result_text.delete(1.0, tk.END)
            self.result_text.insert(tk.END, str(result))
        except Exception as e:
            messagebox.showerror("Error", str(e))

    def clear_results(self):
        '''
        Clear the results displayed in the GUI.
        '''
        self.result_text.delete(1.0, tk.END)
        self.tuple_input.delete(0, tk.END)
        self.k_input.delete(0, tk.END)

if __name__ == "__main__":
    root = tk.Tk()
    app = DivisibleTuplesApp(root)
    root.mainloop()
```

'''

def find_divisible_tuples(tuples_list, k):
    '''
    Find tuples in the list where all elements are divisible by k.
    
    :param tuples_list: List of tuples to check.
    :param k: The divisor.
    :return: List of tuples where all elements are divisible by k.
    '''
    return [tup for tup in tuples_list if all(elem % k == 0 for elem in tup)]

def find_divisible_tuples(tuples_list, k):
    '''
    Find tuples in the list where all elements are divisible by k.
    :param tuples_list: List of tuples to check.
    :param k: The divisor.
    :return: List of tuples where all elements are divisible by k.
    '''
    return [tup for tup in tuples_list if all(elem % k == 0 for elem in tup)]

def find_divisible_tuples(tuples_list, k):
    '''
    Find tuples in the list where all elements are divisible by k.
    :param tuples_list: List of tuples to check.
    :param k: The divisor.
    :return: List of tuples where all elements are divisible by k.
    '''
    return [tup for tup in tuples_list if all(elem % k == 0 for elem in tup)]

'''
Main application file for the Divisible Tuples application.

import tkinter as tk
from tkinter import messagebox
from utils import find_divisible_tuples
class DivisibleTuplesApp:
    '''
    Main application class for the Divisible Tuples GUI.
    '''
    def __init__(self, master):
        self.master = master
        master.title("Divisible Tuples Finder")
        self.label = tk.Label(master, text="Enter tuples (e.g., (1, 2), (3, 4), (5, 6)):")
        self.label.pack()
        self.tuple_input = tk.Entry(master)
        self.tuple_input.pack()
        self.k_label = tk.Label(master, text="Enter k:")
        self.k_label.pack()
        self.k_input = tk.Entry(master)
        self.k_input.pack()
        self.submit_button = tk.Button(master, text="Submit", command=self.on_submit)
        self.submit_button.pack()
        self.clear_button = tk.Button(master, text="Clear", command=self.clear_results)
        self.clear_button.pack()
        self.result_label = tk.Label(master, text="Results:")
        self.result_label.pack()
        self.result_text = tk.Text(master, height=10, width=50)
        self.result_text.pack()
    def on_submit(self):
        '''
        Handle the submission of user input.
        '''
        try:
            tuples_input = eval(self.tuple_input.get())
            k = int(self.k_input.get())
            if not isinstance(tuples_input, tuple) or not all(isinstance(t, tuple) for t in tuples_input):
                raise ValueError("Input must be a tuple of tuples.")
            result = find_divisible_tuples(tuples_input, k)
            self.result_text.delete(1.0, tk.END)
            self.result_text.insert(tk.END, str(result))
        except Exception as e:
            messagebox.showerror("Error", str(e))
    def clear_results(self):
        '''
        Clear the results displayed in the GUI.
        '''
        self.result_text.delete(1.0, tk.END)
        self.tuple_input.delete(0, tk.END)
        self.k_input.delete(0, tk.END)
if __name__ == "__main__":
    root = tk.Tk()
    app = DivisibleTuplesApp(root)
    root.mainloop()

'''
Utility file containing the function to find tuples with all elements divisible by k.
'''
def find_divisible_tuples(tuples_list, k):
    '''
    Find tuples in the list where all elements are divisible by k.
    :param tuples_list: List of tuples to check.
    :param k: The divisor.
    :return: List of tuples where all elements are divisible by k.
    '''
    return [tup for tup in tuples_list if all(elem % k == 0 for elem in tup)]

    '''

    '''
    def __init__(self, master):
        self.master = master
        master.title("Divisible Tuples Finder")
        self.label = tk.Label(master, text="Enter tuples (e.g., (1, 2), (3, 4), (5, 6)):")
        self.label.pack()
        self.tuple_input = tk.Entry(master)
        self.tuple_input.pack()
        self.k_label = tk.Label(master, text="Enter k:")
        self.k_label.pack()
        self.k_input = tk.Entry(master)
        self.k_input.pack()

        self.submit_button.pack()
        self.clear_button = tk.Button(master, text="Clear", command=self.clear_results)
        self.clear_button.pack()
        self.result_label = tk.Label(master, text="Results:")
        self.result_label.pack()
        self.result_text = tk.Text(master, height=10, width=50)
        self.result_text.pack()
    def on_submit(self):
        '''
        Handle the submission of user input.
        '''

            tuples_input = eval(self.tuple_input.get())
            k = int(self.k_input.get())

            result = find_divisible_tuples(tuples_input, k)

            self.result_text.insert(tk.END, str(result))

'''
Utility file containing the function to find tuples with all elements divisible by k.
'''
def find_divisible_tuples(tuples_list, k):
    '''
    Find tuples in the list where all elements are divisible by k.
    :param tuples_list: List of tuples to check.
    :param k: The divisor.
    :return: List of tuples where all elements are divisible by k.
    '''
    return [tup for tup in tuples_list if all(elem % k == 0 for elem in tup)]
