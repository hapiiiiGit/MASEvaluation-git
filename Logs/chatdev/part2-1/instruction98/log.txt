from tkinter import messagebox

class BinaryConverterGUI:
    def __init__(self):
        self.root = tk.Tk()
        self.root.title("Decimal to Binary Converter")
        self.label = tk.Label(self.root, text="Enter a decimal number:")
        self.label.pack()
        self.entry = tk.Entry(self.root)
        self.entry.pack()
        self.convert_button = tk.Button(self.root, text="Convert", command=self.handle_convert_button_click)
        self.convert_button.pack()

    def setup_gui(self):
        self.root.mainloop()

    def handle_convert_button_click(self):
        try:
            decimal_num = int(self.entry.get())
            binary_num = self.convert_decimal_to_binary(decimal_num)
            messagebox.showinfo("Binary Result", f"The binary equivalent is: {binary_num}")
        except ValueError:
            messagebox.showerror("Error", "Please enter a valid decimal number.")

    def convert_decimal_to_binary(self, decimal_num):
        return bin(decimal_num).replace("0b", "")

'''
This file contains the GUI implementation for the binary converter application.

import tkinter as tk
from tkinter import messagebox
from utils import convert_decimal_to_binary
class BinaryConverterGUI:
    def __init__(self):
        self.root = tk.Tk()
        self.root.title("Decimal to Binary Converter")
        self.label = tk.Label(self.root, text="Enter a decimal number:")
        self.label.pack()
        self.entry = tk.Entry(self.root)
        self.entry.pack()
        self.convert_button = tk.Button(self.root, text="Convert", command=self.handle_convert_button_click)
        self.convert_button.pack()
    def setup_gui(self):
        self.root.mainloop()
    def handle_convert_button_click(self):
        '''
        This method handles the click event of the convert button.
        It converts the entered decimal number to binary and displays the result.
        '''
        try:
            decimal_num = int(self.entry.get())
            if decimal_num < 0:
                messagebox.showerror("Error", "Input decimal number must be non-negative.")
            else:
                binary_num = convert_decimal_to_binary(decimal_num)
                messagebox.showinfo("Binary Result", f"The binary equivalent is: {binary_num}")
        except ValueError:
            messagebox.showerror("Error", "Please enter a valid decimal number.")

def convert_decimal_to_binary(decimal_num):
    '''
    This function converts a decimal number to its binary equivalent.
    Parameters:
    decimal_num (int): The decimal number to be converted.
    Returns:
    str: The binary equivalent of the input decimal number.
    '''
    if decimal_num < 0:
        raise ValueError("Input decimal number must be non-negative.")
    return bin(decimal_num).replace("0b", "")

    decimal_num (int): The decimal number to be converted.
    Returns:
    str: The binary equivalent of the input decimal number.
    '''
    if decimal_num < 0:
        raise ValueError("Input decimal number must be non-negative.")
    return bin(decimal_num).replace("0b", "")
