```python
'''
Main application file for converting decimal numbers to binary using tkinter.
'''

import tkinter as tk
from tkinter import messagebox
from converter import convert_to_binary

class DecimalToBinaryApp:
    '''
    Class to create the GUI for the decimal to binary conversion application.
    '''

    def __init__(self, root):
        '''
        Initializes the main window and its components.
        '''
        self.root = root
        self.root.title("Decimal to Binary Converter")

        self.label = tk.Label(root, text="Enter a Decimal Number:")
        self.label.pack()

        self.entry = tk.Entry(root)
        self.entry.pack()

        self.convert_button = tk.Button(root, text="Convert", command=self.convert)
        self.convert_button.pack()

        self.result_label = tk.Label(root, text="")
        self.result_label.pack()

    def convert(self):
        '''
        Converts the input decimal number to binary and displays the result.
        '''
        decimal_number = self.entry.get()
        if validate_input(decimal_number):
            binary_number = convert_to_binary(int(decimal_number))
            self.result_label.config(text=f"Binary: {binary_number}")
        else:
            messagebox.showerror("Invalid Input", "Please enter a valid decimal number.")

def validate_input(input_value):
    '''
    Validates if the input is a valid decimal number.
    '''
    try:
        int(input_value)
        return True
    except ValueError:
        return False

if __name__ == "__main__":
    root = tk.Tk()
    app = DecimalToBinaryApp(root)
    root.mainloop()
```

'''
Module for conversion logic from decimal to binary.

def convert_to_binary(decimal_number):
    '''
    Converts a decimal number to its binary representation.
    '''
    if decimal_number < 0:
        raise ValueError("Negative numbers are not supported.")
    return bin(decimal_number).replace("0b", "")
```

import tkinter as tk
from tkinter import messagebox
from converter import convert_to_binary
class DecimalToBinaryApp:
    '''
    Class to create the GUI for the decimal to binary conversion application.
    '''
    def __init__(self, root):
        '''
        Initializes the main window and its components.
        '''
        self.root = root
        self.root.title("Decimal to Binary Converter")
        self.label = tk.Label(root, text="Enter a Decimal Number:")
        self.label.pack()
        self.entry = tk.Entry(root)
        self.entry.pack()
        self.convert_button = tk.Button(root, text="Convert", command=self.convert)
        self.convert_button.pack()
        self.result_label = tk.Label(root, text="")
        self.result_label.pack()
    def convert(self):
        '''
        Converts the input decimal number to binary and displays the result.
        '''
        decimal_number = self.entry.get()
        if validate_input(decimal_number):
            binary_number = convert_to_binary(int(decimal_number))
            if isinstance(binary_number, str) and "Negative numbers are not supported." in binary_number:
                messagebox.showerror("Invalid Input", binary_number)
            else:
                self.result_label.config(text=f"Binary: {binary_number}")
        else:
            messagebox.showerror("Invalid Input", "Please enter a valid decimal number.")
def validate_input(input_value):
    '''
    Validates if the input is a valid decimal number.
    '''
    try:
        int(input_value)
        return True
    except ValueError:
        return False
if __name__ == "__main__":
    root = tk.Tk()
    app = DecimalToBinaryApp(root)

'''

import tkinter as tk
from tkinter import messagebox
from converter import convert_to_binary
class DecimalToBinaryApp:
    '''
    Class to create the GUI for the decimal to binary conversion application.
    '''
    def __init__(self, root):
        '''
        Initializes the main window and its components.
        '''
        self.root = root
        self.root.title("Decimal to Binary Converter")
        self.label = tk.Label(root, text="Enter a Decimal Number:")
        self.label.pack()
        self.entry = tk.Entry(root)
        self.entry.pack()
        self.convert_button = tk.Button(root, text="Convert", command=self.convert)
        self.convert_button.pack()
        self.result_label = tk.Label(root, text="")
        self.result_label.pack()
    def convert(self):
        '''
        Converts the input decimal number to binary and displays the result.
        '''
        decimal_number = self.entry.get()
        if validate_input(decimal_number):
            binary_number = convert_to_binary(int(decimal_number))
            if isinstance(binary_number, str) and "Negative numbers are not supported." in binary_number:
                messagebox.showerror("Invalid Input", binary_number)
            else:
                self.result_label.config(text=f"Binary: {binary_number}")
        else:
            messagebox.showerror("Invalid Input", "Please enter a valid decimal number.")
def validate_input(input_value):
    '''
    Validates if the input is a valid decimal number.
    '''
    try:
        int(input_value)
        return True
    except ValueError:
        return False
if __name__ == "__main__":
    root = tk.Tk()

def convert_to_binary(decimal_number):
    '''
    Converts a decimal number to its binary representation.
    '''
    if decimal_number < 0:
        return "Negative numbers are not supported."
    return bin(abs(decimal_number))[2:]
