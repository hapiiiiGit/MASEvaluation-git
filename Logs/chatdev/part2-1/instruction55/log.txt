'''
Main application file for checking if a given number is one less than twice its reverse.
'''

import tkinter as tk
from tkinter import messagebox
from utils import reverse_number

class NumberCheckApp:
    def __init__(self, master):
        self.master = master
        master.title("Number Check Application")

        self.label = tk.Label(master, text="Enter a number:")
        self.label.pack()

        self.entry = tk.Entry(master)
        self.entry.pack()

        self.check_button = tk.Button(master, text="Check", command=self.check_number)
        self.check_button.pack()

        self.result_label = tk.Label(master, text="")
        self.result_label.pack()

    def check_number(self):
        try:
            number = int(self.entry.get())
            reversed_num = reverse_number(number)
            if number == 2 * reversed_num - 1:
                self.result_label.config(text=f"{number} is one less than twice its reverse.")
            else:
                self.result_label.config(text=f"{number} is NOT one less than twice its reverse.")
        except ValueError:
            messagebox.showerror("Invalid input", "Please enter a valid integer.")

if __name__ == "__main__":
    root = tk.Tk()
    app = NumberCheckApp(root)
    root.mainloop()
```

#### 2. `utils.py`

```python
'''
Utility functions for the Number Check Application.
'''

def reverse_number(n):
    '''
    Reverses the digits of the given number.
    
    Parameters:
    n (int): The number to be reversed.

    Returns:
    int: The reversed number.
    '''
    return int(str(n)[::-1])
```

```python
'''
Main application file for checking if a given number is one less than twice its reverse.
'''

import tkinter as tk
from tkinter import messagebox
from utils import reverse_number

class NumberCheckApp:
    def __init__(self, master):
        self.master = master
        master.title("Number Check Application")

        self.label = tk.Label(master, text="Enter a number:")
        self.label.pack()

        self.entry = tk.Entry(master)
        self.entry.pack()

        self.check_button = tk.Button(master, text="Check", command=self.check_number)
        self.check_button.pack()

        self.result_label = tk.Label(master, text="")
        self.result_label.pack()

    def check_number(self):
        try:
            number = int(self.entry.get())
            reversed_num = reverse_number(number)
            if number == 2 * reversed_num - 1:
                self.result_label.config(text=f"{number} is one less than twice its reverse.")
            else:
                self.result_label.config(text=f"{number} is NOT one less than twice its reverse.")
        except ValueError:
            messagebox.showerror("Invalid input", "Please enter a valid integer.")

if __name__ == "__main__":
    root = tk.Tk()
    app = NumberCheckApp(root)
    root.mainloop()
```

#### 2. `utils.py`

```python
'''
Utility functions for the Number Check Application.
'''

def reverse_number(n):
    '''
    Reverses the digits of the given number.
    
    Parameters:
    n (int): The number to be reversed.

    Returns:
    int: The reversed number.
    '''
    return int(str(n)[::-1])
```

import tkinter as tk
from tkinter import messagebox
from utils import reverse_number
class NumberCheckApp:
    def __init__(self, master):

                self.result_label.config(text=f"{number} is one less than twice its reverse.")
            else:
                self.result_label.config(text=f"{number} is NOT one less than twice its reverse.")
        except ValueError:
            messagebox.showerror("Invalid input", "Please enter a valid integer.")
if __name__ == "__main__":

    sign = -1 if n < 0 else 1

'''
Main application file for checking if a given number is one less than twice its reverse.
'''
import tkinter as tk
from tkinter import messagebox
from utils import reverse_number
class NumberCheckApp:
    def __init__(self, master):
        self.master = master
        master.title("Number Check Application")
        self.label = tk.Label(master, text="Enter a number:")
        self.label.pack()
        self.entry = tk.Entry(master)
        self.entry.pack()
        self.check_button = tk.Button(master, text="Check", command=self.check_number)
        self.check_button.pack()
        self.result_label = tk.Label(master, text="")
        self.result_label.pack()
    def check_number(self):
        try:
            number = int(self.entry.get())
            reversed_num = reverse_number(number)
            # Check if the number is one less than twice its reverse
            if number == 2 * reversed_num - 1:
                self.result_label.config(text=f"{number} is one less than twice its reverse.")
            else:
                self.result_label.config(text=f"{number} is NOT one less than twice its reverse.")
        except ValueError:
            messagebox.showerror("Invalid input", "Please enter a valid integer.")
if __name__ == "__main__":
    root = tk.Tk()
    app = NumberCheckApp(root)
    root.mainloop()

def reverse_number(n):
    '''
    Reverses the digits of the given number.
    Parameters:
    n (int): The number to be reversed.
    Returns:
    int: The reversed number.
    '''
    sign = -1 if n < 0 else 1
    n = abs(n)
    return sign * int(str(n)[::-1])
