from tkinter import Tk, Label, Entry, Button, StringVar
from utils import count_set_bits
class MainApp:
    def __init__(self):
        self.root = Tk()
        self.root.title("Odd Set Bits Counter")
        self.label = Label(self.root, text="Enter a number:")
        self.label.pack()
        self.input_var = StringVar()
        self.input_entry = Entry(self.root, textvariable=self.input_var)
        self.input_entry.pack()
        self.result_label = Label(self.root, text="")
        self.result_label.pack()
        self.count_button = Button(self.root, text="Count Odd Set Bits", command=self.count_odd_set_bits)
        self.count_button.pack()
        self.root.mainloop()
    def count_odd_set_bits(self):
        try:
            number = int(self.input_var.get())
            if number < 0:
                raise ValueError("Please enter a positive integer.")
            odd_count = sum(1 for i in range(number + 1) if count_set_bits(i) % 2 != 0)
            self.result_label.config(text=f"Count of integers with odd set bits: {odd_count}")
        except ValueError as e:
            self.result_label.config(text=str(e))
if __name__ == "__main__":
    app = MainApp()

def count_set_bits(n):
    '''
    Count the number of set bits in the binary representation of n.
    '''
    count = 0
    n = abs(n)  # Convert negative numbers to positive
    while n:
        count += n & 1
        n >>= 1
    return count

from tkinter import Tk, Label, Entry, Button, StringVar

    def __init__(self):

        self.root.title("Odd Set Bits Counter")

def count_set_bits(n):
    '''
    Count the number of set bits in the binary representation of n.
    '''
    count = 0
    n = abs(n)  # Convert negative numbers to positive
    while n:
        count += n & 1
        n >>= 1
    return count
