'''
Main application file for multiplying numbers in a list and dividing by the length of the list.
'''
import tkinter as tk
from tkinter import messagebox
from utils import validate_input, calculate_result
class MainApp:
    def __init__(self, master):
        self.master = master
        master.title("Multiply and Divide Application")
        self.label = tk.Label(master, text="Enter numbers separated by commas:")
        self.label.pack()
        self.entry = tk.Entry(master)
        self.entry.pack()
        self.calculate_button = tk.Button(master, text="Calculate", command=self.multiply_and_divide)
        self.calculate_button.pack()
        self.result_label = tk.Label(master, text="")
        self.result_label.pack()
    def multiply_and_divide(self):
        user_input = self.entry.get()
        if validate_input(user_input):
            numbers = list(map(float, user_input.split(',')))
            result = calculate_result(numbers)
            self.result_label.config(text=f"Result: {result:.2f}")  # Formatting result to 2 decimal places
        else:
            messagebox.showerror("Input Error", "Please enter valid numbers separated by commas.")
if __name__ == "__main__":
    root = tk.Tk()
    app = MainApp(root)
    root.mainloop()

'''
Utility functions for input validation and calculation.
'''
def validate_input(user_input):
    '''
    Validates the user input to ensure it contains only numbers separated by commas.
    '''
    try:
        numbers = [float(num) for num in user_input.split(',')]
        return True
    except ValueError:
        return False
def calculate_result(numbers):
    '''
    Multiplies all numbers in the list and divides by the length of the list.
    '''
    if not numbers:
        return 0
    product = 1
    for number in numbers:
        product *= number
    return product / len(numbers) if len(numbers) > 0 else 0

Utility functions for input validation and calculation.

def validate_input(user_input):
    '''
    Validates the user input to ensure it contains only numbers separated by commas.
    '''
    try:
        numbers = [float(num) for num in user_input.split(',')]
        return True
    except ValueError:
        return False
def calculate_result(numbers):
    '''
    Multiplies all numbers in the list and divides by the length of the list.
    '''
    if not numbers:
        return 0
    product = 1
    for number in numbers:
        product *= number
    return product / len(numbers) if len(numbers) > 0 else 0
