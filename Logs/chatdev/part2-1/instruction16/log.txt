'''
Main file to run the Square Perimeter Calculator application.
'''

from tkinter import Tk
from calculator import SquarePerimeterCalculator

class App:
    def __init__(self, master):
        self.master = master
        master.title("Square Perimeter Calculator")

        self.label = Label(master, text="Enter the side length of the square:")
        self.label.pack()

        self.entry = Entry(master)
        self.entry.pack()

        self.calculate_button = Button(master, text="Calculate Perimeter", command=self.calculate_perimeter)
        self.calculate_button.pack()

        self.result_label = Label(master, text="")
        self.result_label.pack()

    def calculate_perimeter(self):
        side_length = float(self.entry.get())
        calculator = SquarePerimeterCalculator()
        perimeter = calculator.calculate_perimeter(side_length)
        self.result_label.config(text=f"The perimeter of the square is: {perimeter}")

if __name__ == "__main__":
    root = Tk()
    app = App(root)
    root.mainloop()
```

'''
Main file to run the Square Perimeter Calculator application.
'''
from tkinter import Tk, Label, Entry, Button
from calculator import SquarePerimeterCalculator

class App:
    def __init__(self, master):
        self.master = master
        master.title("Square Perimeter Calculator")
        self.label = Label(master, text="Enter the side length of the square:")
        self.label.pack()
        self.entry = Entry(master)
        self.entry.pack()
        self.calculate_button = Button(master, text="Calculate Perimeter", command=self.calculate_perimeter)
        self.calculate_button.pack()
        self.result_label = Label(master, text="")
        self.result_label.pack()

    def calculate_perimeter(self):
        side_length = float(self.entry.get())
        calculator = SquarePerimeterCalculator()
        perimeter = calculator.calculate_perimeter(side_length)
        self.result_label.config(text=f"The perimeter of the square is: {perimeter}")

if __name__ == "__main__":
    root = Tk()
    app = App(root)
    root.mainloop()
```

from tkinter import Tk, Label, Entry, Button
from calculator import SquarePerimeterCalculator
class App:
    def __init__(self, master):
        self.master = master
        master.title("Square Perimeter Calculator")
        self.label = Label(master, text="Enter the side length of the square:")
        self.label.pack()
        self.entry = Entry(master)
        self.entry.pack()
        self.calculate_button = Button(master, text="Calculate Perimeter", command=self.calculate_perimeter)
        self.calculate_button.pack()
        self.result_label = Label(master, text="")
        self.result_label.pack()
    def calculate_perimeter(self):
        try:
            side_length = float(self.entry.get())
            calculator = SquarePerimeterCalculator()
            perimeter = calculator.calculate_perimeter(side_length)
            self.result_label.config(text=f"The perimeter of the square is: {perimeter}")
        except ValueError:
            self.result_label.config(text="Please enter a valid number.")
