import tkinter as tk
from centered_hexagonal_number import find_centered_hexagonal_number

'''

import tkinter as tk
from centered_hexagonal_number import find_centered_hexagonal_number
class GUIApp:
    def __init__(self):
        self.root = tk.Tk()
        self.root.title("Centered Hexagonal Number Finder")
        self.label = tk.Label(self.root, text="Enter the value of n:")
        self.label.pack()
        self.entry = tk.Entry(self.root)
        self.entry.pack()
        self.result_label = tk.Label(self.root, text="")
        self.result_label.pack()
        self.button = tk.Button(self.root, text="Find", command=self.find_number)
        self.button.pack()
    def find_number(self):
        try:
            n = int(self.entry.get())
            if n <= 0:
                self.result_label.config(text="Please enter a positive integer.")
            else:
                result = find_centered_hexagonal_number(n)
                self.result_label.config(text=f"The {n}th centered hexagonal number is: {result}")
        except ValueError:
            self.result_label.config(text="Invalid input. Please enter a valid integer.")
    def run(self):
        self.root.mainloop()
if __name__ == "__main__":
    app = GUIApp()
    app.run()

The formula to calculate the nth centered hexagonal number is:

- For \( n = 1 \): The result is 1.

  \[
