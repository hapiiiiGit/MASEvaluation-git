'''
Main application file for the String Splitter GUI.

import tkinter as tk
from tkinter import messagebox
from string_splitter import split_string

class StringSplitterApp:
    def __init__(self, master):
        self.master = master
        master.title("String Splitter")

        self.label = tk.Label(master, text="Enter String:")
        self.label.pack()

        self.input_string = tk.Entry(master, width=50)
        self.input_string.pack()

        self.delimiter_label = tk.Label(master, text="Enter Delimiters (e.g., ,;|):")
        self.delimiter_label.pack()

        self.delimiters = tk.Entry(master, width=50)
        self.delimiters.pack()

        self.split_button = tk.Button(master, text="Split String", command=self.split_string)
        self.split_button.pack()

        self.clear_button = tk.Button(master, text="Clear", command=self.clear_fields)
        self.clear_button.pack()

        self.result_label = tk.Label(master, text="Result:")
        self.result_label.pack()

        self.result_text = tk.Text(master, height=10, width=50)
        self.result_text.pack()

    def split_string(self):
        input_str = self.input_string.get()
        delimiters = self.delimiters.get()
        if not input_str or not delimiters:
            messagebox.showerror("Input Error", "Please enter both a string and delimiters.")
            return
        
        try:
            result = split_string(input_str, delimiters)
            self.result_text.delete(1.0, tk.END)  # Clear previous results
            self.result_text.insert(tk.END, "\n".join(result))
        except Exception as e:
            messagebox.showerror("Error", str(e))

    def clear_fields(self):
        self.input_string.delete(0, tk.END)
        self.delimiters.delete(0, tk.END)
        self.result_text.delete(1.0, tk.END)

if __name__ == "__main__":
    root = tk.Tk()
    app = StringSplitterApp(root)
    root.mainloop()

'''

import re

def split_string(input_str, delimiters):
    '''
    Splits the input string using the provided delimiters.
    
    Parameters:
    input_str (str): The string to be split.
    delimiters (str): A string containing all delimiters.
    
    Returns:
    list: A list of split strings.
    '''
    # Create a regex pattern from the delimiters
    pattern = '|'.join(map(re.escape, delimiters))
    return re.split(pattern, input_str)

import re
def split_string(input_str, delimiters):
    '''
    Splits the input string using the provided delimiters.
    Parameters:
    input_str (str): The string to be split.
    delimiters (str): A string containing all delimiters.
    Returns:
    list: A list of split strings.
    '''
    # Create a regex pattern from the delimiters
    pattern = '|'.join(map(re.escape, delimiters))
    return re.split(pattern, input_str)
