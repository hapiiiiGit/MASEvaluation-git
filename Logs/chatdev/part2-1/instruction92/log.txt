import tkinter as tk
from tkinter import messagebox

class PowerCalculatorApp:
    def __init__(self):
        self.window = tk.Tk()
        self.window.title("Power Calculator")

        # Create input fields
        self.label_a = tk.Label(self.window, text="Enter base (a):")
        self.label_a.pack()
        self.entry_a = tk.Entry(self.window)
        self.entry_a.pack()

        self.label_b = tk.Label(self.window, text="Enter exponent (b):")
        self.label_b.pack()
        self.entry_b = tk.Entry(self.window)
        self.entry_b.pack()

        # Create calculate button
        self.calculate_button = tk.Button(self.window, text="Calculate a^b", command=self.calculate_power)
        self.calculate_button.pack()

        # Create result label
        self.result_label = tk.Label(self.window, text="")
        self.result_label.pack()

    def calculate_power(self):
        try:
            a = float(self.entry_a.get())
            b = float(self.entry_b.get())
            result = a ** b
            self.result_label.config(text=f"Result: {result}")
        except ValueError:
            messagebox.showerror("Input Error", "Please enter valid numbers for a and b.")

    def run(self):
        self.window.mainloop()

class PowerCalculatorApp:
    def __init__(self):
        self.window = tk.Tk()
        self.window.title("Power Calculator")
        # Create input fields
        self.label_a = tk.Label(self.window, text="Enter base (a):")
        self.label_a.pack()
        self.entry_a = tk.Entry(self.window)
        self.entry_a.pack()
        self.label_b = tk.Label(self.window, text="Enter exponent (b):")
        self.label_b.pack()
        self.entry_b = tk.Entry(self.window)
        self.entry_b.pack()
        # Create calculate button
        self.calculate_button = tk.Button(self.window, text="Calculate a^b", command=self.calculate_power)
        self.calculate_button.pack()
        # Create result label
        self.result_label = tk.Label(self.window, text="")
        self.result_label.pack()
    def calculate_power(self):
        try:
            a = float(self.entry_a.get())
            b = float(self.entry_b.get())
            if not (isinstance(a, (int, float)) and isinstance(b, (int, float))):
                raise ValueError
            result = a ** b
            self.result_label.config(text=f"Result: {result}")
        except ValueError:
            messagebox.showerror("Input Error", "Please enter valid numbers for a and b.")
    def run(self):
        self.window.mainloop()

import tkinter as tk
from tkinter import messagebox
class PowerCalculatorApp:
    def __init__(self):
        self.window = tk.Tk()
        self.window.title("Power Calculator")
        # Create input fields
        self.label_a = tk.Label(self.window, text="Enter base (a):")
        self.label_a.pack()

        self.entry_a.pack()
        self.label_b = tk.Label(self.window, text="Enter exponent (b):")

        self.calculate_button.pack()
        # Create result label
        self.result_label = tk.Label(self.window, text="")
        self.result_label.pack()

        self.window.mainloop()
