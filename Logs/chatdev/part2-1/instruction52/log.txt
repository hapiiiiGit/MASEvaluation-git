'''
This file contains the main application class for checking if the first and last characters of a string are equal.
'''

import tkinter as tk
from tkinter import messagebox

class StringCheckerApp:
    '''
    A class to create a GUI application that checks if the first and last characters of a string are equal.
    '''

    def __init__(self, master):
        '''
        Initializes the GUI components.
        '''
        self.master = master
        master.title("String Character Checker")

        self.label = tk.Label(master, text="Enter a string:")
        self.label.pack()

        self.entry = tk.Entry(master)
        self.entry.pack()

        self.check_button = tk.Button(master, text="Check", command=self.check_characters)
        self.check_button.pack()

        self.result_label = tk.Label(master, text="")
        self.result_label.pack()

    def check_characters(self):
        '''
        Checks if the first and last characters of the input string are equal.
        '''
        input_string = self.entry.get()
        if len(input_string) == 0:
            messagebox.showwarning("Input Error", "Please enter a string.")
            return
        
        first_char = input_string[0]
        last_char = input_string[-1]

        if first_char == last_char:
            self.result_label.config(text="The first and last characters are equal.")
        else:
            self.result_label.config(text="The first and last characters are not equal.")

def main():
    '''
    The main function to run the application.
    '''
    root = tk.Tk()
    app = StringCheckerApp(root)
    root.mainloop()

if __name__ == "__main__":
    main()
```

'''
This file contains the main application class for checking if the first and last characters of a string are equal.

import tkinter as tk
from tkinter import messagebox

class StringCheckerApp:
    '''
    A class to create a GUI application that checks if the first and last characters of a string are equal.
    '''
    def __init__(self, master):
        '''
        Initializes the GUI components.
        '''
        self.master = master
        master.title("String Character Checker")
        self.label = tk.Label(master, text="Enter a string:")
        self.label.pack()
        self.entry = tk.Entry(master)
        self.entry.pack()
        self.check_button = tk.Button(master, text="Check", command=self.check_characters)
        self.check_button.pack()
        self.result_label = tk.Label(master, text="")
        self.result_label.pack()

    def check_characters(self):
        '''
        Checks if the first and last characters of the input string are equal.
        '''
        input_string = self.entry.get()
        if len(input_string) == 0:
            messagebox.showwarning("Input Error", "Please enter a string.")
            return
        elif len(input_string) == 1:
            self.result_label.config(text="The string has only one character.")
            return
        first_char = input_string[0]
        last_char = input_string[-1]
        if first_char == last_char:
            self.result_label.config(text="The first and last characters are equal.")
        else:
            self.result_label.config(text="The first and last characters are not equal.")

def main():
    '''
    The main function to run the application.
    '''
    root = tk.Tk()
    app = StringCheckerApp(root)
    root.mainloop()

if __name__ == "__main__":
    main()

'''
This file contains the main application class for checking if the first and last characters of a string are equal.
'''
import tkinter as tk
from tkinter import messagebox
class StringCheckerApp:
    '''
    A class to create a GUI application that checks if the first and last characters of a string are equal.
    '''
    def __init__(self, master):
        '''
        Initializes the GUI components.
        '''
        self.master = master
        master.title("String Character Checker")
        self.label = tk.Label(master, text="Enter a string:")
        self.label.pack()
        self.entry = tk.Entry(master)
        self.entry.pack()
        self.check_button = tk.Button(master, text="Check", command=self.check_characters)
        self.check_button.pack()
        self.result_label = tk.Label(master, text="")
        self.result_label.pack()
    def check_characters(self):
        '''
        Checks if the first and last characters of the input string are equal.
        '''
        input_string = self.entry.get()
        if len(input_string) == 0:
            messagebox.showwarning("Input Error", "Please enter a string.")
            return
        elif len(input_string) == 1:
            self.result_label.config(text="The string has only one character.")
            return
        first_char = input_string[0]
        last_char = input_string[-1]
        if first_char == last_char:
            self.result_label.config(text="The first and last characters are equal.")
        else:
            self.result_label.config(text="The first and last characters are not equal.")
def main():
    '''
    The main function to run the application.
    '''
    root = tk.Tk()
    app = StringCheckerApp(root)
    root.mainloop()
if __name__ == "__main__":
    main()
