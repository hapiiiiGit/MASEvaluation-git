'''
Main application file for the Max Difference Calculator.

from tkinter import Tk, Label, Entry, Button, StringVar, messagebox
from utils import find_max_difference

class MaxDifferenceApp:
    def __init__(self, master):
        self.master = master
        master.title("Max Difference Calculator")

        self.label = Label(master, text="Enter tuples (e.g., (1, 2), (3, 4)):")
        self.label.pack()

        self.tuple_input = StringVar()
        self.entry = Entry(master, textvariable=self.tuple_input)
        self.entry.pack()

        self.calculate_button = Button(master, text="Calculate Max Difference", command=self.on_calculate)
        self.calculate_button.pack()

        self.result_label = Label(master, text="")
        self.result_label.pack()

        self.clear_button = Button(master, text="Clear", command=self.clear_input)
        self.clear_button.pack()

    def on_calculate(self):
        try:
            # Convert input string to a list of tuples
            input_tuples = eval(self.tuple_input.get())
            if not isinstance(input_tuples, list) or not all(isinstance(t, tuple) for t in input_tuples):
                raise ValueError("Input must be a list of tuples.")
            max_diff = find_max_difference(input_tuples)
            self.result_label.config(text=f"Max Difference: {max_diff}")
        except Exception as e:
            messagebox.showerror("Input Error", str(e))

    def clear_input(self):
        self.tuple_input.set("")
        self.result_label.config(text="")

if __name__ == "__main__":
    root = Tk()
    app = MaxDifferenceApp(root)
    root.mainloop()
```

'''
Utility file containing the function to find the maximum difference
between pairs in a list of tuples.
'''

def find_max_difference(tuple_list):
    if not tuple_list:
        return 0  # Return 0 if the list is empty

    max_diff = float('-inf')  # Initialize to negative infinity
    for i in range(len(tuple_list)):
        for j in range(i + 1, len(tuple_list)):
            diff = abs(tuple_list[i][0] - tuple_list[j][0]) + abs(tuple_list[i][1] - tuple_list[j][1])
            if diff > max_diff:
                max_diff = diff
    return max_diff
```

def find_max_difference(tuple_list):
    if not tuple_list:
        return 0  # Return 0 if the list is empty
    max_diff = float('-inf')  # Initialize to negative infinity
    for i in range(len(tuple_list)):
        for j in range(i + 1, len(tuple_list)):
            diff = abs(tuple_list[i][0] - tuple_list[j][0]) + abs(tuple_list[i][1] - tuple_list[j][1])
            if diff > max_diff:
                max_diff = diff
    return max_diff

'''
Utility file containing the function to find the maximum difference
between pairs in a list of tuples.
'''
def find_max_difference(tuple_list):
    if not tuple_list:
        return 0  # Return 0 if the list is empty
    max_diff = float('-inf')  # Initialize to negative infinity
    for i in range(len(tuple_list)):
        for j in range(i + 1, len(tuple_list)):
            diff = abs(tuple_list[i][0] - tuple_list[j][0]) + abs(tuple_list[i][1] - tuple_list[j][1])
            if diff > max_diff:
                max_diff = diff
    return max_diff

    if not tuple_list:
        return 0  # Return 0 if the list is empty

    for i in range(len(tuple_list)):
        for j in range(i + 1, len(tuple_list)):
            if len(tuple_list[i]) != 2 or len(tuple_list[j]) != 2:
                continue  # Skip pairs that do not have exactly 2 elements
            diff = abs(tuple_list[i][0] - tuple_list[j][0]) + abs(tuple_list[i][1] - tuple_list[j][1])
            if diff > max_diff:
                max_diff = diff
    return max_diff

def find_max_difference(tuple_list):
    if not tuple_list:
        return 0  # Return 0 if the list is empty
    
    # Initialize min and max for both dimensions
    min_x = min_y = float('inf')
    max_x = max_y = float('-inf')

        min_x = min(min_x, t[0])
        max_x = max(max_x, t[0])
        min_y = min(min_y, t[1])

    # Calculate the maximum differences
    max_diff_x = max_x - min_x
    max_diff_y = max_y - min_y
    
    # Return the maximum of the two differences

'''
Utility file containing the function to find the maximum difference
between pairs in a list of tuples.
'''
def find_max_difference(tuple_list):
    if not tuple_list:
        return 0  # Return 0 if the list is empty
    max_diff = float('-inf')  # Initialize to negative infinity
    for i in range(len(tuple_list)):
        for j in range(i + 1, len(tuple_list)):
            if len(tuple_list[i]) != 2 or len(tuple_list[j]) != 2:
                continue  # Skip pairs that do not have exactly 2 elements
            diff = abs(tuple_list[i][0] - tuple_list[j][0]) + abs(tuple_list[i][1] - tuple_list[j][1])
            if diff > max_diff:
                max_diff = diff
    return max_diff

from tkinter import Tk, Label, Entry, Button, StringVar, messagebox
from utils import find_max_difference
import ast  # Importing ast for safe evaluation
class MaxDifferenceApp:
    def __init__(self, master):
        self.master = master
        master.title("Max Difference Calculator")
        self.label = Label(master, text="Enter tuples (e.g., (1, 2), (3, 4)):")
        self.label.pack()
        self.tuple_input = StringVar()
        self.entry = Entry(master, textvariable=self.tuple_input)
        self.entry.pack()
        self.calculate_button = Button(master, text="Calculate Max Difference", command=self.on_calculate)
        self.calculate_button.pack()
        self.result_label = Label(master, text="")
        self.result_label.pack()
        self.clear_button = Button(master, text="Clear", command=self.clear_input)
        self.clear_button.pack()
    def on_calculate(self):
        try:
            # Convert input string to a list of tuples safely
            input_tuples = ast.literal_eval(self.tuple_input.get())
            if not isinstance(input_tuples, list) or not all(isinstance(t, tuple) for t in input_tuples) or not all(len(t) == 2 for t in input_tuples):
                raise ValueError("Input must be a list of tuples with exactly 2 elements.")
            max_diff = find_max_difference(input_tuples)
            self.result_label.config(text=f"Max Difference: {max_diff}")
        except Exception as e:
            messagebox.showerror("Input Error", str(e))
    def clear_input(self):
        self.tuple_input.set("")
        self.result_label.config(text="")

'''
Utility file containing the function to find the maximum difference
between pairs in a list of tuples.
'''
def find_max_difference(tuple_list):
    if not tuple_list:
        return 0  # Return 0 if the list is empty
    max_diff = float('-inf')  # Initialize to negative infinity
    for i in range(len(tuple_list)):
        for j in range(i + 1, len(tuple_list)):
            if len(tuple_list[i]) != 2 or len(tuple_list[j]) != 2:
                continue  # Skip pairs that do not have exactly 2 elements
            diff = abs(tuple_list[i][0] - tuple_list[j][0]) + abs(tuple_list[i][1] - tuple_list[j][1])
            if diff > max_diff:
                max_diff = diff
    return max_diff
