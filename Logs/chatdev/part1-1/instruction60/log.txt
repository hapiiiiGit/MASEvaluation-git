```python
'''
Main application file for the sum calculator GUI.
'''

import tkinter as tk
from tkinter import messagebox
from sum_function import sum_to_n

class SumApp:
    '''
    Class to create the main application window and handle user interactions.
    '''
    
    def __init__(self, master):
        '''
        Initializes the GUI components.
        '''
        self.master = master
        master.title("Sum Calculator")

        self.label = tk.Label(master, text="Enter a number:")
        self.label.pack()

        self.entry = tk.Entry(master)
        self.entry.pack()

        self.calculate_button = tk.Button(master, text="Calculate Sum", command=self.calculate_sum)
        self.calculate_button.pack()

        self.result_label = tk.Label(master, text="")
        self.result_label.pack()

    def calculate_sum(self):
        '''
        Retrieves user input, computes the sum, and displays the result.
        '''
        try:
            n = int(self.entry.get())
            result = sum_to_n(n)
            self.result_label.config(text=f"The sum from 1 to {n} is: {result}")
        except ValueError:
            messagebox.showerror("Invalid input", "Please enter a valid integer.")

if __name__ == "__main__":
    root = tk.Tk()
    app = SumApp(root)
    root.mainloop()
```

'''

def sum_to_n(n: int) -> int:
    '''
    sum_to_n is a function that sums numbers from 1 to n.
    >>> sum_to_n(30)

    >>> sum_to_n(100)
    5050

    15

    '''

import tkinter as tk
from tkinter import messagebox
from sum_function import sum_to_n
class SumApp:
    '''
    Class to create the main application window and handle user interactions.
    '''
    def __init__(self, master):
        '''
        Initializes the GUI components.
        '''
        self.master = master
        master.title("Sum Calculator")
        self.label = tk.Label(master, text="Enter a number:")
        self.label.pack()
        self.entry = tk.Entry(master)
        self.entry.pack()
        self.calculate_button = tk.Button(master, text="Calculate Sum", command=self.calculate_sum)
        self.calculate_button.pack()
        self.result_label = tk.Label(master, text="")
        self.result_label.pack()
    def calculate_sum(self):
        '''
        Retrieves user input, computes the sum, and displays the result.
        '''
        try:
            n = int(self.entry.get())
            result = sum_to_n(n)
            self.result_label.config(text=f"The sum from 1 to {n} is: {result}")
        except ValueError:
            messagebox.showerror("Invalid input", "Please enter a valid integer.")
if __name__ == "__main__":
    root = tk.Tk()
    app = SumApp(root)
    root.mainloop()

```python
'''
Main application file for the sum calculator GUI.
'''
import tkinter as tk
from tkinter import messagebox
from sum_function import sum_to_n
class SumApp:
    '''
    Class to create the main application window and handle user interactions.
    '''
    def __init__(self, master):
        '''
        Initializes the GUI components.
        '''
        self.master = master
        master.title("Sum Calculator")
        self.label = tk.Label(master, text="Enter a number:")
        self.label.pack()
        self.entry = tk.Entry(master)
        self.entry.pack()
        self.calculate_button = tk.Button(master, text="Calculate Sum", command=self.calculate_sum)
        self.calculate_button.pack()
        self.result_label = tk.Label(master, text="")
        self.result_label.pack()
    def calculate_sum(self):
        '''
        Retrieves user input, computes the sum, and displays the result.
        '''
        try:
            n = int(self.entry.get())
            result = sum_to_n(n)
            self.result_label.config(text=f"The sum from 1 to {n} is: {result}")
        except ValueError:
            messagebox.showerror("Invalid input", "Please enter a valid integer.")
if __name__ == "__main__":
    root = tk.Tk()
    app = SumApp(root)
    root.mainloop()

'''
Module containing the function to sum numbers from 1 to n.
'''
def sum_to_n(n: int) -> int:
    '''
    sum_to_n is a function that sums numbers from 1 to n.
    >>> sum_to_n(30)
    465
    >>> sum_to_n(100)
    5050
    >>> sum_to_n(5)
    15
    >>> sum_to_n(10)
    55
    >>> sum_to_n(1)
    1
    >>> sum_to_n(0)
    0
    >>> sum_to_n(-5)
    0
    '''
    if n < 1:
        return 0  # Return 0 for non-positive integers
    return n * (n + 1) // 2
