'''
Module for counting prime numbers.
'''

class PrimeCounter:
    @staticmethod
    def count_up_to(n):
        """Returns a list of prime numbers less than n."""
        if n < 2:
            return []
        primes = []
        for num in range(2, n):
            is_prime = True
            for i in range(2, int(num**0.5) + 1):
                if num % i == 0:
                    is_prime = False
                    break
            if is_prime:
                primes.append(num)
        return primes
```

```python
'''
Module for the graphical user interface of the application.
'''

import tkinter as tk
from tkinter import messagebox
from prime_counter import PrimeCounter

class App:
    def __init__(self):
        """Initializes the GUI components."""
        self.root = tk.Tk()
        self.root.title("Prime Number Counter")
        
        self.label = tk.Label(self.root, text="Enter a non-negative integer:")
        self.label.pack()
        
        self.entry = tk.Entry(self.root)
        self.entry.pack()
        
        self.button = tk.Button(self.root, text="Calculate Primes", command=self.calculate_primes)
        self.button.pack()
        
        self.result_label = tk.Label(self.root, text="")
        self.result_label.pack()

    def calculate_primes(self):
        """Handles the button click event to calculate primes."""
        try:
            n = int(self.entry.get())
            if n < 0:
                raise ValueError("Please enter a non-negative integer.")
            primes = PrimeCounter.count_up_to(n)
            self.result_label.config(text=f"Primes less than {n}: {primes}")
        except ValueError as e:
            messagebox.showerror("Input Error", str(e))

    def run(self):
        """Runs the main loop of the application."""
        self.root.mainloop()
```

class PrimeCounter:
    @staticmethod
    def count_up_to(n):
        """Returns a list of prime numbers less than n."""
        if n < 2:
            return []
        primes = []
        for num in range(2, n):
            is_prime = True
            for i in range(2, int(num**0.5) + 1):
                if num % i == 0:
                    is_prime = False
                    break
            if is_prime:
                primes.append(num)
        return primes

'''
Module for counting prime numbers.

class PrimeCounter:
    @staticmethod
    def count_up_to(n):
        """Returns a list of the first n prime numbers less than n."""
        if n < 2:
            return []
        primes = []
        for num in range(2, n):
            is_prime = True
            for i in range(2, int(num**0.5) + 1):
                if num % i == 0:
                    is_prime = False
                    break
            if is_prime:
                primes.append(num)
            if len(primes) == n:  # Stop when we have the first n primes
                break
        return primes

'''
Module for counting prime numbers.
'''
class PrimeCounter:
    @staticmethod
    def count_up_to(n):
        """Returns a list of the prime numbers less than n."""
        if n < 2:
            return []
        primes = []
        for num in range(2, n):
            is_prime = True
            for i in range(2, int(num**0.5) + 1):
                if num % i == 0:
                    is_prime = False
                    break
            if is_prime:
                primes.append(num)
        return primes

'''
Module for the graphical user interface of the application.
'''
import tkinter as tk
from tkinter import messagebox
from prime_counter import PrimeCounter
class App:
    def __init__(self):
        """Initializes the GUI components."""
        self.root = tk.Tk()
        self.root.title("Prime Number Counter")
        self.label = tk.Label(self.root, text="Enter a non-negative integer:")
        self.label.pack()
        self.entry = tk.Entry(self.root)
        self.entry.pack()
        self.button = tk.Button(self.root, text="Calculate Primes", command=self.calculate_primes)
        self.button.pack()
        self.result_label = tk.Label(self.root, text="")
        self.result_label.pack()
    def calculate_primes(self):
        """Handles the button click event to calculate primes."""
        try:
            n = int(self.entry.get())
            if n < 0:
                raise ValueError("Please enter a non-negative integer.")
            primes = PrimeCounter.count_up_to(n)
            self.result_label.config(text=f"Primes less than {n}: {primes}")
        except ValueError as e:
            messagebox.showerror("Input Error", str(e))
    def run(self):
        """Runs the main loop of the application."""
        self.root.mainloop()

'''
Module for counting prime numbers.
'''
class PrimeCounter:
    @staticmethod
    def count_up_to(n):
        """Returns a list of the prime numbers less than n."""
        if n < 2:
            return []
        primes = []
        for num in range(2, n):
            is_prime = True
            for i in range(2, int(num**0.5) + 1):
                if num % i == 0:
                    is_prime = False
                    break
            if is_prime:
                primes.append(num)
        return primes
