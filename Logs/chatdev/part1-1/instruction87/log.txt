```python
'''
Core functionality for finding coordinates of a specified integer in a nested list.
'''

def get_row(lst, x):
    """
    Given a 2-dimensional data as nested lists, find integers x in the list,
    and return a list of tuples, [(x1, y1), (x2, y2) ...] such that
    each tuple is a coordinate - (row, column), starting with 0.
    Sort coordinates initially by rows in ascending order.
    Also, sort coordinates of the row by columns in descending order.
    """
    coordinates = []
    
    for row_index, row in enumerate(lst):
        for col_index, value in enumerate(row):
            if value == x:
                coordinates.append((row_index, col_index))
    
    # Sort by row index ascending, and by column index descending
    coordinates.sort(key=lambda coord: (coord[0], -coord[1]))
    
    return coordinates
```

'''
Core functionality for finding coordinates of a specified integer in a nested list.
'''

def get_row(lst, x):
    """
    Given a 2-dimensional data as nested lists, find integers x in the list,
    and return a list of tuples, [(x1, y1), (x2, y2) ...] such that
    each tuple is a coordinate - (row, column), starting with 0.
    Sort coordinates initially by rows in ascending order.
    Also, sort coordinates of the row by columns in descending order.
    """
    coordinates = []
    
    for row_index, row in enumerate(lst):
        for col_index, value in enumerate(row):
            if value == x:
                coordinates.append((row_index, col_index))
    
    # Sort by row index ascending, and by column index descending
    coordinates.sort(key=lambda coord: (coord[0], -coord[1]))
    
    return coordinates
```

    Given a 2-dimensional data as nested lists, find integers x in the list,

    each tuple is a coordinate - (row, column), starting with 0.
    Sort coordinates initially by rows in ascending order.

def get_row(lst, x):
    """
    Given a 2-dimensional data as nested lists, find integers x in the list,
    and return a list of tuples, [(x1, y1), (x2, y2) ...] such that
    each tuple is a coordinate - (row, column), starting with 0.
    Sort coordinates initially by rows in ascending order.
    Also, sort coordinates of the row by columns in descending order.
    """
    coordinates = []
    for row_index, row in enumerate(lst):
        for col_index, value in enumerate(row):
            if value == x:
                coordinates.append((row_index, col_index))
    # Sort by row index ascending, and by column index descending
    coordinates.sort(key=lambda coord: (coord[0], -coord[1]))
    return coordinates

def get_row(lst, x):
    """
    Given a 2-dimensional data as nested lists, find integers x in the list,
    and return a list of tuples, [(x1, y1), (x2, y2) ...] such that
    each tuple is a coordinate - (row, column), starting with 0.
    Sort coordinates initially by rows in ascending order.
    Also, sort coordinates of the row by columns in descending order.
    """
    coordinates = []
    for row_index, row in enumerate(lst):
        for col_index, value in enumerate(row):
            if value == x:
                coordinates.append((row_index, col_index))
    # Sort by row index ascending, and by column index descending
    coordinates.sort(key=lambda coord: (coord[0], -coord[1]))
    return coordinates

'''
Core functionality for finding coordinates of a specified integer in a nested list.
'''
def get_row(lst, x):
    """
    Given a 2-dimensional data as nested lists, find integers x in the list,
    and return a list of tuples, [(x1, y1), (x2, y2) ...] such that
    each tuple is a coordinate - (row, column), starting with 0.
    Sort coordinates initially by rows in ascending order.
    Also, sort coordinates of the row by columns in descending order.
    """
    coordinates = []
    for row_index, row in enumerate(lst):
        for col_index, value in enumerate(row):
            if value == x:
                coordinates.append((row_index, col_index))
    # Sort by row index ascending, and by column index descending
    coordinates.sort(key=lambda coord: (coord[0], -coord[1]))
    return coordinates

'''
Core functionality for finding coordinates of a specified integer in a nested list.
'''
def get_row(lst, x):
    """
    Given a 2-dimensional data as nested lists, find integers x in the list,
    and return a list of tuples, [(x1, y1), (x2, y2) ...] such that
    each tuple is a coordinate - (row, column), starting with 0.
    Sort coordinates initially by rows in ascending order.
    Also, sort coordinates of the row by columns in descending order.
    """
    coordinates = []
    for row_index, row in enumerate(lst):
        for col_index, value in enumerate(row):
            if value == x:
                coordinates.append((row_index, col_index))
    # Sort by row index ascending, and by column index descending
    coordinates.sort(key=lambda coord: (coord[0], -coord[1]))
    return coordinates
