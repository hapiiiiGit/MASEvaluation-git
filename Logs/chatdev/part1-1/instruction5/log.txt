'''
File containing the intersperse function.
'''

from typing import List

def intersperse(numbers: List[int], delimiter: int) -> List[int]:
    """
    Insert a number 'delimiter' between every two consecutive elements of input list `numbers'
    """
    result = []
    for num in numbers:
        result.extend([num, delimiter])
    result.pop()  # Remove the last delimiter
    return result
```

Main file to run the IntersperseApp GUI application.

from tkinter import Tk, Label, Entry, Button
from tkinter import messagebox
from intersperse_functions import intersperse
class IntersperseApp:
    def __init__(self, master):
        self.master = master
        master.title("Intersperse App")
        self.label = Label(master, text="Enter numbers separated by commas:")
        self.label.pack()
        self.entry = Entry(master)
        self.entry.pack()
        self.label_delimiter = Label(master, text="Enter delimiter:")
        self.label_delimiter.pack()
        self.entry_delimiter = Entry(master)
        self.entry_delimiter.pack()
        self.result_label = Label(master, text="")
        self.result_label.pack()
        self.intersperse_button = Button(master, text="Intersperse", command=self.intersperse_numbers)
        self.intersperse_button.pack()
    def intersperse_numbers(self):
        numbers = list(map(int, self.entry.get().split(',')))  # Convert input to integers
        delimiter = int(self.entry_delimiter.get())
        try:
            result = intersperse(numbers, delimiter)
            self.result_label.config(text=f"Result: {result}")
        except ValueError:
            messagebox.showerror("Error", "Please enter valid numbers.")
if __name__ == "__main__":
    root = Tk()
    app = IntersperseApp(root)
    root.mainloop()

'''

'''
from typing import List
def intersperse(numbers: List[int], delimiter: int) -> List[int]:
    """
    Insert a number 'delimiter' between every two consecutive elements of input list `numbers'
    """
    result = []
    for i in range(len(numbers)):
        result.append(numbers[i])
        if i < len(numbers) - 1:
            result.append(delimiter)
    return result

Main file to run the IntersperseApp GUI application.

from tkinter import Tk, Label, Entry, Button
from tkinter import messagebox
from intersperse_functions import intersperse
class IntersperseApp:
    def __init__(self, master):
        self.master = master
        master.title("Intersperse App")
        self.label = Label(master, text="Enter numbers separated by commas:")
        self.label.pack()
        self.entry = Entry(master)
        self.entry.pack()
        self.label_delimiter = Label(master, text="Enter delimiter:")
        self.label_delimiter.pack()
        self.entry_delimiter = Entry(master)
        self.entry_delimiter.pack()
        self.result_label = Label(master, text="")
        self.result_label.pack()
        self.intersperse_button = Button(master, text="Intersperse", command=self.intersperse_numbers)
        self.intersperse_button.pack()
    def intersperse_numbers(self):
        numbers = list(map(int, self.entry.get().split(',')))  # Convert input to integers
        delimiter = int(self.entry_delimiter.get())
        try:
            result = intersperse(numbers, delimiter)
            self.result_label.config(text=f"Result: {result}")
        except ValueError:
            messagebox.showerror("Error", "Please enter valid numbers.")
if __name__ == "__main__":
    root = Tk()
    app = IntersperseApp(root)
    root.mainloop()
