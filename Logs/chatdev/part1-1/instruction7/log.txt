'''
Main application file for the String Filter application.

from tkinter import Tk, Label, Entry, Button, Listbox, END
from filter import filter_by_substring

class StringFilterApp:
    def __init__(self, master):
        self.master = master
        master.title("String Filter Application")

        self.label = Label(master, text="Enter strings (comma-separated):")
        self.label.pack()

        self.entry = Entry(master, width=50)
        self.entry.pack()

        self.label_substring = Label(master, text="Enter substring to filter:")
        self.label_substring.pack()

        self.substring_entry = Entry(master, width=50)
        self.substring_entry.pack()

        self.filter_button = Button(master, text="Filter", command=self.filter_strings)
        self.filter_button.pack()

        self.result_listbox = Listbox(master, width=50)
        self.result_listbox.pack()

    def filter_strings(self):
        strings_input = self.entry.get().split(',')
        substring = self.substring_entry.get()
        filtered_strings = filter_by_substring([s.strip() for s in strings_input], substring)
        
        self.result_listbox.delete(0, END)  # Clear previous results
        for string in filtered_strings:
            self.result_listbox.insert(END, string)

if __name__ == "__main__":
    root = Tk()
    app = StringFilterApp(root)
    root.mainloop()
```

```python
'''
Module containing the filter function to filter strings by substring.
'''

from typing import List

def filter_by_substring(strings: List[str], substring: str) -> List[str]:
    """ 
    Filter an input list of strings only for ones that contain given substring.
    
    >>> filter_by_substring([], 'a')
    []
    >>> filter_by_substring(['abc', 'bacd', 'cde', 'array'], 'a')
    ['abc', 'bacd', 'array']
    """
    return [s for s in strings if substring in s]
```

'''
Main application file for the String Filter application.
'''
from tkinter import Tk, Label, Entry, Button, Listbox, END
from filter import filter_by_substring
class StringFilterApp:
    def __init__(self, master):
        self.master = master
        master.title("String Filter Application")
        self.label = Label(master, text="Enter strings (comma-separated):")
        self.label.pack()
        self.entry = Entry(master, width=50)
        self.entry.pack()
        self.label_substring = Label(master, text="Enter substring to filter:")
        self.label_substring.pack()
        self.substring_entry = Entry(master, width=50)
        self.substring_entry.pack()
        self.filter_button = Button(master, text="Filter", command=self.filter_strings)
        self.filter_button.pack()
        self.result_listbox = Listbox(master, width=50)
        self.result_listbox.pack()
    def filter_strings(self):
        strings_input = self.entry.get().split(',')
        substring = self.substring_entry.get()
        filtered_strings = filter_by_substring([s.strip() for s in strings_input], substring)
        self.result_listbox.delete(0, END)  # Clear previous results
        for string in filtered_strings:
            self.result_listbox.insert(END, string)
if __name__ == "__main__":
    root = Tk()
    app = StringFilterApp(root)
    root.mainloop()

from typing import List
def filter_by_substring(strings: List[str], substring: str) -> List[str]:
    """

    >>> filter_by_substring([], 'a')
    []
    >>> filter_by_substring(['abc', 'bacd', 'cde', 'array'], 'a')
    ['abc', 'bacd', 'array']
    """
    return [s for s in strings if substring in s]

from typing import List

    >>> filter_by_substring([], 'a')
    []
    >>> filter_by_substring(['abc', 'bacd', 'cde', 'array'], 'a')
    ['abc', 'bacd', 'array']
    """
    if strings is None or substring is None:
        return []
    return [s for s in strings if substring in s]

A simple application to filter strings based on a specified substring.

The String Filter Application allows users to:

- Input a list of strings separated by commas.
- Specify a substring to filter the strings.
- Display the filtered results in a list format.

### Environment Dependencies

pip install tk

    ```bash
    python main.py
    ```
