'''
Module for sorting an array based on specific conditions.
'''

def sort_array(array):
    """
    Given an array of non-negative integers, return a copy of the given array after sorting,
    you will sort the given array in ascending order if the sum( first index value, last index value) is odd,
    or sort it in descending order if the sum( first index value, last index value) is even.

    Note:
    * don't change the given array.

    Examples:
    * sort_array([]) => []
    * sort_array([5]) => [5]
    * sort_array([2, 4, 3, 0, 1, 5]) => [0, 1, 2, 3, 4, 5]
    * sort_array([2, 4, 3, 0, 1, 5, 6]) => [6, 5, 4, 3, 2, 1, 0]
    """
    if not array:  # Handle empty array
        return []
    
    first = array[0]
    last = array[-1]
    total = first + last
    
    # Sort based on the sum of the first and last elements
    if total % 2 == 0:
        return sorted(array, reverse=True)  # Descending order
    else:
        return sorted(array)  # Ascending order
```

'''
Module for the graphical user interface of the sorting application.

import tkinter as tk
from tkinter import messagebox
from sorter import sort_array

class App:
    """
    Main application class for the sorting GUI.
    """

    def __init__(self):
        self.window = tk.Tk()
        self.window.title("Array Sorter")
        
        self.label = tk.Label(self.window, text="Enter non-negative integers separated by commas:")
        self.label.pack()
        
        self.entry = tk.Entry(self.window, width=50)
        self.entry.pack()
        
        self.sort_button = tk.Button(self.window, text="Sort", command=self.sort_array)
        self.sort_button.pack()
        
        self.result_label = tk.Label(self.window, text="")
        self.result_label.pack()

    def sort_array(self):
        """
        Retrieve input, sort the array, and display the result.
        """
        input_text = self.entry.get()
        try:
            array = list(map(int, input_text.split(',')))
            if any(num < 0 for num in array):
                raise ValueError("All numbers must be non-negative.")
            sorted_array = sort_array(array)
            self.result_label.config(text=f"Sorted Array: {sorted_array}")
        except ValueError as e:
            messagebox.showerror("Input Error", str(e))

    def run(self):
        """
        Start the Tkinter main loop.
        """
        self.window.mainloop()

def sort_array(array):
    """
    Given an array of non-negative integers, return a copy of the given array after sorting,
    you will sort the given array in ascending order if the sum( first index value, last index value) is odd,
    or sort it in descending order if the sum( first index value, last index value) is even.
    Note:
    * don't change the given array.
    Examples:
    * sort_array([]) => []
    * sort_array([5]) => [5]
    * sort_array([2, 4, 3, 0, 1, 5]) => [0, 1, 2, 3, 4, 5]
    * sort_array([2, 4, 3, 0, 1, 5, 6]) => [6, 5, 4, 3, 2, 1, 0]
    """
    if not array:  # Handle empty array
        return []
    first = array[0]
    last = array[-1]
    total = first + last
    # Sort based on the sum of the first and last elements
    if total % 2 == 0:
        return sorted(array, reverse=True)  # Descending order
    else:
        return sorted(array)  # Ascending order

'''

def sort_array(array):
    """
    Given an array of non-negative integers, return a copy of the given array after sorting,
    you will sort the given array in ascending order if the sum( first index value, last index value) is odd,
    or sort it in descending order if the sum( first index value, last index value) is even.
    Note:
    * don't change the given array.
    Examples:
    * sort_array([]) => []
    * sort_array([5]) => [5]
    * sort_array([2, 4, 3, 0, 1, 5]) => [0, 1, 2, 3, 4, 5]
    * sort_array([2, 4, 3, 0, 1, 5, 6]) => [6, 5, 4, 3, 2, 1, 0]
    """
    if not array:  # Handle empty array
        return []
    first = array[0]
    last = array[-1]
    total = first + last
    # Sort based on the sum of the first and last elements
    if total % 2 == 0:
        return sorted(array, reverse=True)  # Descending order
    else:
        return sorted(array)  # Ascending order

def sort_array(array):
    """
    Given an array of non-negative integers, return a copy of the given array after sorting,
    you will sort the given array in ascending order if the sum( first index value, last index value) is odd,
    or sort it in descending order if the sum( first index value, last index value) is even.
    Note:
    * don't change the given array.
    Examples:
    * sort_array([]) => []
    * sort_array([5]) => [5]
    * sort_array([2, 4, 3, 0, 1, 5]) => [0, 1, 2, 3, 4, 5]
    * sort_array([2, 4, 3, 0, 1, 5, 6]) => [6, 5, 4, 3, 2, 1, 0]
    """
    if not array:  # Handle empty array
        return []
    first = array[0]
    last = array[-1]
    total = first + last
    # Sort based on the sum of the first and last elements
    if total % 2 == 0:
        return sorted(array, reverse=True)  # Descending order
    else:
        return sorted(array)  # Ascending order

'''

def sort_array(array):
    """
    Given an array of non-negative integers, return a copy of the given array after sorting,
    you will sort the given array in ascending order if the sum( first index value, last index value) is odd,
    or sort it in descending order if the sum( first index value, last index value) is even.
    Note:
    * don't change the given array.
    Examples:
    * sort_array([]) => []
    * sort_array([5]) => [5]
    * sort_array([2, 4, 3, 0, 1, 5]) => [0, 1, 2, 3, 4, 5]
    * sort_array([2, 4, 3, 0, 1, 5, 6]) => [6, 5, 4, 3, 2, 1, 0]
    """
    if not array:  # Handle empty array
        return []
    first = array[0]
    last = array[-1]
    total = first + last
    # Sort based on the sum of the first and last elements
    if total % 2 == 0:
        return sorted(array, reverse=True)  # Descending order
    else:
        return sorted(array)  # Ascending order

'''

def sort_array(array):
    """
    Given an array of non-negative integers, return a copy of the given array after sorting,
    you will sort the given array in ascending order if the sum( first index value, last index value) is odd,
    or sort it in descending order if the sum( first index value, last index value) is even.
    Note:
    * don't change the given array.
    Examples:
    * sort_array([]) => []
    * sort_array([5]) => [5]
    * sort_array([2, 4, 3, 0, 1, 5]) => [0, 1, 2, 3, 4, 5]
    * sort_array([2, 4, 3, 0, 1, 5, 6]) => [6, 5, 4, 3, 2, 1, 0]
    """
    if not array:  # Handle empty array
        return []
    first = array[0]
    last = array[-1]
    total = first + last
    # Sort based on the sum of the first and last elements
    if total % 2 == 0:
        return sorted(array, reverse=True)  # Descending order
    else:
        return sorted(array)  # Ascending order

def sort_array(array):
    """
    Given an array of non-negative integers, return a copy of the given array after sorting,
    you will sort the given array in ascending order if the sum( first index value, last index value) is odd,
    or sort it in descending order if the sum( first index value, last index value) is even.
    Note:
    * don't change the given array.
    Examples:
    * sort_array([]) => []
    * sort_array([5]) => [5]
    * sort_array([2, 4, 3, 0, 1, 5]) => [0, 1, 2, 3, 4, 5]
    * sort_array([2, 4, 3, 0, 1, 5, 6]) => [6, 5, 4, 3, 2, 1, 0]
    """
    if not array:  # Handle empty array
        return []
    first = array[0]
    last = array[-1]
    total = first + last
    # Sort based on the sum of the first and last elements
    if total % 2 == 0:
        return sorted(array, reverse=True)  # Descending order
    else:
        return sorted(array)  # Ascending order

'''

def sort_array(array):
    """
    Given an array of non-negative integers, return a copy of the given array after sorting,
    you will sort the given array in ascending order if the sum( first index value, last index value) is odd,
    or sort it in descending order if the sum( first index value, last index value) is even.
    Note:
    * don't change the given array.
    Examples:
    * sort_array([]) => []
    * sort_array([5]) => [5]
    * sort_array([2, 4, 3, 0, 1, 5]) => [0, 1, 2, 3, 4, 5]
    * sort_array([2, 4, 3, 0, 1, 5, 6]) => [6, 5, 4, 3, 2, 1, 0]
    """
    if not array:  # Handle empty array
        return []
    first = array[0]
    last = array[-1]
    total = first + last
    # Sort based on the sum of the first and last elements
    if total % 2 == 0:
        return sorted(array, reverse=True)  # Descending order
    else:
        return sorted(array)  # Ascending order

import tkinter as tk
from tkinter import messagebox
from sorter import sort_array
class App:
    """
    Main application class for the sorting GUI.
    """
    def __init__(self):
        self.window = tk.Tk()
        self.window.title("Array Sorter")
        self.label = tk.Label(self.window, text="Enter non-negative integers separated by commas:")
        self.label.pack()
        self.entry = tk.Entry(self.window, width=50)
        self.entry.pack()
        self.sort_button = tk.Button(self.window, text="Sort", command=self.sort_array)
        self.sort_button.pack()
        self.result_label = tk.Label(self.window, text="")
        self.result_label.pack()
    def sort_array(self):
        """
        Retrieve input, sort the array, and display the result.
        """
        input_text = self.entry.get()
        try:
            array = list(map(int, input_text.split(',')))
            if any(num < 0 for num in array):
                raise ValueError("All numbers must be non-negative.")
            sorted_array = sort_array(array)
            self.result_label.config(text=f"Sorted Array: {sorted_array}")
        except ValueError as e:
            messagebox.showerror("Input Error", str(e))
    def run(self):
        """
        Start the Tkinter main loop.
        """
        self.window.mainloop()

'''
Module for sorting an array based on specific conditions.
'''
def sort_array(array):
    """
    Given an array of non-negative integers, return a copy of the given array after sorting,
    you will sort the given array in ascending order if the sum( first index value, last index value) is odd,
    or sort it in descending order if the sum( first index value, last index value) is even.
    Note:
    * don't change the given array.
    Examples:
    * sort_array([]) => []
    * sort_array([5]) => [5]
    * sort_array([2, 4, 3, 0, 1, 5]) => [0, 1, 2, 3, 4, 5]
    * sort_array([2, 4, 3, 0, 1, 5, 6]) => [6, 5, 4, 3, 2, 1, 0]
    """
    if not array:  # Handle empty array
        return []
    first = array[0]
    last = array[-1]
    total = first + last
    # Sort based on the sum of the first and last elements
    if total % 2 == 0:
        return sorted(array, reverse=True)  # Descending order
    else:
        return sorted(array)  # Ascending order
