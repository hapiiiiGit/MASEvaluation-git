'''
Main application file for the encryption GUI.
'''

import tkinter as tk
from tkinter import messagebox
from encryptor import Encryptor

class MainApp:
    def __init__(self, root):
        self.root = root
        self.root.title("String Encryptor")
        
        self.label = tk.Label(root, text="Enter a string to encrypt:")
        self.label.pack(pady=10)

        self.input_text = tk.Entry(root, width=50)
        self.input_text.pack(pady=10)

        self.encrypt_button = tk.Button(root, text="Encrypt", command=self.encrypt_string)
        self.encrypt_button.pack(pady=10)

        self.result_label = tk.Label(root, text="")
        self.result_label.pack(pady=10)

    def encrypt_string(self):
        text = self.input_text.get()
        if text:
            encryptor = Encryptor()
            encrypted_text = encryptor.encrypt(text)
            self.result_label.config(text=f"Encrypted: {encrypted_text}")
        else:
            messagebox.showwarning("Input Error", "Please enter a string to encrypt.")

if __name__ == "__main__":
    root = tk.Tk()
    app = MainApp(root)
    root.mainloop()

```python
'''
Module containing the Encryptor class for string encryption.
'''

class Encryptor:
    def encrypt(self, s):
        """
        Encrypts the input string by rotating the alphabet down by four places.
        """
        encrypted_string = ""
        for char in s:
            if char.isalpha():  # Check if the character is a letter
                # Calculate the new character
                new_char = chr((ord(char) - ord('a') + 4) % 26 + ord('a')) if char.islower() else chr((ord(char) - ord('A') + 4) % 26 + ord('A'))
                encrypted_string += new_char
            else:
                encrypted_string += char  # Non-alphabetic characters remain unchanged
        return encrypted_string
```

class Encryptor:
    def encrypt(self, s):
        """
        Encrypts the input string by rotating the alphabet down by four places.
        """
        encrypted_string = ""
        for char in s:
            if char.isalpha():  # Check if the character is a letter
                # Calculate the new character
                new_char = chr((ord(char) - ord('a') + 4) % 26 + ord('a')) if char.islower() else chr((ord(char) - ord('A') + 4) % 26 + ord('A'))
                encrypted_string += new_char
            else:
                encrypted_string += char  # Non-alphabetic characters remain unchanged
        return encrypted_string

class Encryptor:
    def encrypt(self, s):
        """
        Encrypts the input string by rotating the alphabet down by four places.
        """
        encrypted_string = ""
        for char in s:
            if char.isalpha():  # Check if the character is a letter
                # Calculate the new character with a shift of 4
                shift = 4
                if char.islower():
                    new_char = chr((ord(char) - ord('a') + shift) % 26 + ord('a'))
                else:
                    new_char = chr((ord(char) - ord('A') + shift) % 26 + ord('A'))
                encrypted_string += new_char
            else:
                encrypted_string += char  # Non-alphabetic characters remain unchanged
        return encrypted_string

'''
Module containing the Encryptor class for string encryption.
'''
class Encryptor:
    def encrypt(self, s):
        """
        Encrypts the input string by rotating the alphabet down by two places.
        """
        encrypted_string = ""
        for char in s:
            if char.isalpha():  # Check if the character is a letter
                shift = 2  # Correct shift as per the user's requirement
                if char.islower():
                    new_char = chr((ord(char) - ord('a') + shift) % 26 + ord('a'))
                else:
                    new_char = chr((ord(char) - ord('A') + shift) % 26 + ord('A'))
                encrypted_string += new_char
            else:
                encrypted_string += char  # Non-alphabetic characters remain unchanged
        return encrypted_string

# String Encryptor

A simple application for encrypting strings by rotating the alphabet.

## Overview

The String Encryptor is a Python-based GUI application that allows users to input a string and receive an encrypted version of that string. The encryption is performed by shifting each letter down the alphabet by two positions. For example, the letter 'a' becomes 'c', 'b' becomes 'd', and so on. Non-alphabetic characters remain unchanged.

## Main Functions

- **Encrypt String**: Input a string and click the "Encrypt" button to receive the encrypted output.
- **User-Friendly Interface**: Built with Tkinter for a simple and intuitive user experience.

   ```bash
   python main.py
