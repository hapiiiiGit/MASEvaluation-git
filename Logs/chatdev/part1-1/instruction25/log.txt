'''
Main application file for the prime factorization GUI.

import tkinter as tk
from tkinter import messagebox
from factorizer import Factorizer

class App:
    '''
    Main application class that sets up the GUI.
    '''
    
    def __init__(self):
        self.root = tk.Tk()
        self.root.title("Prime Factorization")

        self.result_label.pack()
        
        self.factorizer = Factorizer()

        '''
        Retrieves the input number, calls the factorization method, and displays the result.
        '''
        try:
            number = int(self.entry.get())
            factors = self.factorizer.factorize(number)
            self.result_label.config(text=f"Factors: {factors}")

            messagebox.showerror("Input Error", "Please enter a valid integer.")

if __name__ == "__main__":
    App()

```python
'''
Contains the Factorizer class for prime factorization logic.
'''

from typing import List

class Factorizer:
    '''
    Class to handle the factorization of integers into prime factors.
    '''
    
    def factorize(self, n: int) -> List[int]:
        '''
        Return list of prime factors of given integer in the order from smallest to largest.
        Each of the factors should be listed number of times corresponding to how many times it appears in factorization.
        Input number should be equal to the product of all factors.
        
        >>> factorize(8)
        [2, 2, 2]
        >>> factorize(25)
        [5, 5]
        >>> factorize(70)
        [2, 5, 7]
        '''
        factors = []
        divisor = 2
        
        while n > 1:
            while n % divisor == 0:
                factors.append(divisor)
                n //= divisor
            divisor += 1
        
        return factors
```

'''
Contains the Factorizer class for prime factorization logic.
'''
from typing import List
class Factorizer:
    '''
    Class to handle the factorization of integers into prime factors.
    '''
    def factorize(self, n: int) -> List[int]:
        '''
        Return list of prime factors of given integer in the order from smallest to largest.
        Each of the factors should be listed number of times corresponding to how many times it appears in factorization.
        Input number should be equal to the product of all factors.
        >>> factorize(8)
        [2, 2, 2]
        >>> factorize(25)
        [5, 5]
        >>> factorize(70)
        [2, 5, 7]
        '''
        factors = []
        divisor = 2
        while n > 1:
            while n % divisor == 0:
                factors.append(divisor)
                n //= divisor
            divisor += 1
        return factors

from typing import List
class Factorizer:
    '''
    Class to handle the factorization of integers into prime factors.
    '''
    def factorize(self, n: int) -> List[int]:
        '''

        [5, 5]

        '''

        return factors

from typing import List
class Factorizer:
    '''
    Class to handle the factorization of integers into prime factors.
    '''
    def factorize(self, n: int) -> List[int]:
        '''
        Return list of prime factors of given integer in the order from smallest to largest.
        Each of the factors should be listed number of times corresponding to how many times it appears in factorization.
        Input number should be equal to the product of all factors.
        >>> factorize(8)
        [2, 2, 2]
        >>> factorize(25)
        [5, 5]
        >>> factorize(70)
        [2, 5, 7]
        '''
        factors = []
        divisor = 2
        while n > 1:
            while n % divisor == 0:
                factors.append(divisor)
                n //= divisor
            divisor += 1
        return factors

'''
Main application file for the prime factorization GUI.

import tkinter as tk
from tkinter import messagebox
from factorizer import Factorizer
class App:
    '''
    Main application class that sets up the GUI.
    '''
    def __init__(self):
        self.root = tk.Tk()
        self.root.title("Prime Factorization")
        self.label = tk.Label(self.root, text="Enter an integer:")
        self.label.pack()
        self.entry = tk.Entry(self.root)
        self.entry.pack()
        self.button = tk.Button(self.root, text="Factorize", command=self.factorize_number)
        self.button.pack()
        self.result_label = tk.Label(self.root, text="")
        self.result_label.pack()
        self.factorizer = Factorizer()
        self.root.mainloop()
    def factorize_number(self):
        '''
        Retrieves the input number, calls the factorization method, and displays the result.
        '''
        try:
            number = int(self.entry.get())
            factors = self.factorizer.factorize(number)
            self.result_label.config(text=f"Factors: {factors}")
        except ValueError:
            messagebox.showerror("Input Error", "Please enter a valid integer.")
if __name__ == "__main__":
    App()

'''
Contains the Factorizer class for prime factorization logic.
'''
from typing import List
class Factorizer:
    '''
    Class to handle the factorization of integers into prime factors.
    '''
    def factorize(self, n: int) -> List[int]:
        '''
        Return list of prime factors of given integer in the order from smallest to largest.
        Each of the factors should be listed number of times corresponding to how many times it appears in factorization.
        Input number should be equal to the product of all factors.
        >>> factorize(8)
        [2, 2, 2]
        >>> factorize(25)
        [5, 5]
        >>> factorize(70)
        [2, 5, 7]
        '''
        factors = []
        divisor = 2
        while n > 1:
            while n % divisor == 0:
                factors.append(divisor)
                n //= divisor
            divisor += 1
        return factors

'''
Contains the Factorizer class for prime factorization logic.
'''
from typing import List
class Factorizer:
    '''
    Class to handle the factorization of integers into prime factors.
    '''
    def factorize(self, n: int) -> List[int]:
        '''
        Return list of prime factors of given integer in the order from smallest to largest.
        Each of the factors should be listed number of times corresponding to how many times it appears in factorization.
        Input number should be equal to the product of all factors.
        >>> factorize(8)
        [2, 2, 2]
        >>> factorize(25)
        [5, 5]
        >>> factorize(70)
        [2, 5, 7]
        '''
        factors = []
        divisor = 2
        while n > 1:
            while n % divisor == 0:
                factors.append(divisor)
                n //= divisor
            divisor += 1
        return factors
