Main application file for the Common Elements Finder.

from tkinter import Tk, Label, Entry, Button, Text, END
from common import common

class CommonElementsApp:
    def __init__(self, master):
        self.master = master
        master.title("Common Elements Finder")

        self.label1 = Label(master, text="Enter first list (comma-separated):")
        self.label1.pack()

        self.entry1 = Entry(master, width=50)
        self.entry1.pack()

        self.label2 = Label(master, text="Enter second list (comma-separated):")
        self.label2.pack()

        self.entry2 = Entry(master, width=50)
        self.entry2.pack()

        self.find_button = Button(master, text="Find Common Elements", command=self.find_common)
        self.find_button.pack()

        self.result_label = Label(master, text="Common Elements:")
        self.result_label.pack()

        self.result_text = Text(master, height=10, width=50)
        self.result_text.pack()

        self.clear_button = Button(master, text="Clear", command=self.clear_fields)
        self.clear_button.pack()

    def find_common(self):
        try:
            list1 = list(map(int, self.entry1.get().split(',')))
            list2 = list(map(int, self.entry2.get().split(',')))
            result = common(list1, list2)
            self.result_text.delete(1.0, END)
            self.result_text.insert(END, str(result))
        except ValueError:
            self.result_text.delete(1.0, END)
            self.result_text.insert(END, "Please enter valid integers.")

    def clear_fields(self):
        self.entry1.delete(0, END)
        self.entry2.delete(0, END)
        self.result_text.delete(1.0, END)

if __name__ == "__main__":
    root = Tk()
    app = CommonElementsApp(root)
    root.mainloop()

    [2, 3]
    """
    return sorted(set(l1) & set(l2))

def common(l1: list, l2: list):
    """Return sorted unique common elements for two lists.
    >>> common([1, 4, 3, 34, 653, 2, 5], [5, 7, 1, 5, 9, 653, 121])
    [1, 5, 653]
    >>> common([5, 3, 2, 8], [3, 2])
    [2, 3]
    """
    return sorted(set(l1) & set(l2))

Module containing the function to find common elements in two lists.

def common(l1: list, l2: list):
    """Return sorted unique common elements for two lists.
    >>> common([1, 4, 3, 34, 653, 2, 5], [5, 7, 1, 5, 9, 653, 121])
    [1, 5, 653]
    >>> common([5, 3, 2, 8], [3, 2])
    [2, 3]
    """
    return sorted(set(l1) & set(l2))
