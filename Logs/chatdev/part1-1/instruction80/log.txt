'''
This module contains the HappyStringChecker class which provides a method
to check if a given string is happy based on specific criteria.
'''

class HappyStringChecker:
    @staticmethod
    def is_happy(s):
        """Check if the string is happy."""
        if len(s) < 3:
            return False
        for i in range(len(s) - 2):
            if len(set(s[i:i + 3])) < 3:
                return False
        return True
```

import tkinter as tk

from happy_string_checker import HappyStringChecker

class App:
    def __init__(self, master):
        self.master = master
        master.title("Happy String Checker")

        self.label = tk.Label(master, text="Enter a string:")
        self.label.pack()

        self.entry = tk.Entry(master)
        self.entry.pack()

        self.check_button = tk.Button(master, text="Check", command=self.check_string)
        self.check_button.pack()

        self.result_label = tk.Label(master, text="")
        self.result_label.pack()

        input_string = self.entry.get()
        if HappyStringChecker.is_happy(input_string):
            result = "The string is happy!"
        else:
            result = "The string is not happy."
        self.result_label.config(text=result)

if __name__ == "__main__":
    root = tk.Tk()

to check if a given string is happy based on specific criteria.

class HappyStringChecker:
    @staticmethod
    def is_happy(s):
        """Check if the string is happy by ensuring every 3 consecutive characters are distinct."""
        if len(s) < 3:
            return False
        for i in range(len(s) - 2):
            # Check if the three consecutive characters are distinct
            if s[i] == s[i + 1] or s[i] == s[i + 2] or s[i + 1] == s[i + 2]:
                return False
        return True

class HappyStringChecker:
    @staticmethod
    def is_happy(s):
        """Check if the string is happy by ensuring every 3 consecutive characters are distinct."""
        if len(s) < 3:
