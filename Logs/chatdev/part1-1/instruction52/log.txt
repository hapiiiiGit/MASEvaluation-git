'''
Main application file for the Threshold Checker GUI.
'''

from tkinter import Tk, Label, Entry, Button, StringVar
from threshold_checker import below_threshold

class ThresholdCheckerApp:
    def __init__(self, master):
        self.master = master
        master.title("Threshold Checker")

        self.label = Label(master, text="Enter numbers (comma-separated):")
        self.label.pack()

        self.numbers_entry = Entry(master)
        self.numbers_entry.pack()

        self.threshold_label = Label(master, text="Enter threshold:")
        self.threshold_label.pack()

        self.threshold_entry = Entry(master)
        self.threshold_entry.pack()

        self.result_var = StringVar()
        self.result_label = Label(master, textvariable=self.result_var)
        self.result_label.pack()

        self.check_button = Button(master, text="Check", command=self.check_threshold)
        self.check_button.pack()

        self.clear_button = Button(master, text="Clear", command=self.clear_fields)
        self.clear_button.pack()

    def check_threshold(self):
        numbers = self.numbers_entry.get().split(',')
        numbers = [int(num.strip()) for num in numbers]
        threshold = int(self.threshold_entry.get())
        result = below_threshold(numbers, threshold)
        self.result_var.set(f"All numbers below threshold: {result}")

    def clear_fields(self):
        self.numbers_entry.delete(0, 'end')
        self.threshold_entry.delete(0, 'end')
        self.result_var.set("")

if __name__ == "__main__":
    root = Tk()
    app = ThresholdCheckerApp(root)
    root.mainloop()
```

    False
    """

    >>> below_threshold([1, 2, 4, 10], 100)

    >>> below_threshold([1, 20, 4, 10], 5)
    False
    """

def below_threshold(l: list, t: int):

    True
    >>> below_threshold([1, 20, 4, 10], 5)
    False
    """
    return all(x < t for x in l)

'''
Main application file for the Threshold Checker GUI.
'''
from tkinter import Tk, Label, Entry, Button, StringVar
from threshold_checker import below_threshold
class ThresholdCheckerApp:
    def __init__(self, master):
        self.master = master
        master.title("Threshold Checker")
        self.label = Label(master, text="Enter numbers (comma-separated):")
        self.label.pack()
        self.numbers_entry = Entry(master)
        self.numbers_entry.pack()
        self.threshold_label = Label(master, text="Enter threshold:")
        self.threshold_label.pack()
        self.threshold_entry = Entry(master)
        self.threshold_entry.pack()
        self.result_var = StringVar()
        self.result_label = Label(master, textvariable=self.result_var)
        self.result_label.pack()
        self.check_button = Button(master, text="Check", command=self.check_threshold)
        self.check_button.pack()
        self.clear_button = Button(master, text="Clear", command=self.clear_fields)
        self.clear_button.pack()
    def check_threshold(self):
        try:
            numbers = self.numbers_entry.get().split(',')
            numbers = [int(num.strip()) for num in numbers]
            threshold = int(self.threshold_entry.get())
            result = below_threshold(numbers, threshold)
            self.result_var.set(f"All numbers below threshold: {result}")
        except ValueError:
            self.result_var.set("Please enter valid integers.")
    def clear_fields(self):
        self.numbers_entry.delete(0, 'end')
        self.threshold_entry.delete(0, 'end')
        self.result_var.set("")
if __name__ == "__main__":
    root = Tk()
    app = ThresholdCheckerApp(root)
    root.mainloop()

'''
Module containing the function to check if all numbers are below a given threshold.
'''
def below_threshold(l: list, t: int):
    """Return True if all numbers in the list l are below threshold t.
    >>> below_threshold([1, 2, 4, 10], 100)
    True
    >>> below_threshold([1, 20, 4, 10], 5)
    False
    """
    return all(x < t for x in l)
