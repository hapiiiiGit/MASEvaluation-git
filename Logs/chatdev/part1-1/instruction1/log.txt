'''

This file contains the ParenGroupSeparator class responsible for separating groups of nested parentheses.

'''

from typing import List

class ParenGroupSeparator:
    def separate_paren_groups(self, paren_string: str) -> List[str]:
        groups = []
        group = ""
        open_braces = 0

        for char in paren_string:
            if char == '(':
                open_braces += 1
            elif char == ')':
                open_braces -= 1

            group += char

            if open_braces == 0:
                groups.append(group)
                group = ""

        return groups

from typing import List
class ParenGroupSeparator:
    def separate_paren_groups(self, paren_string: str) -> List[str]:

            group += char
            if open_braces == 0:
                groups.append(group)
                group = ""
        return groups

'''
This file contains the ParenGroupSeparator class responsible for separating groups of nested parentheses.

from typing import List

class ParenGroupSeparator:
    def separate_paren_groups(self, paren_string: str) -> List[str]:
        # Remove spaces from the input string
        paren_string = paren_string.replace(" ", "")
        
        def separate_helper(s, start):
            groups = []
            stack = 0
            for i, char in enumerate(s[start:], start):
                if char == '(':
                    stack += 1
                elif char == ')':
                    stack -= 1
                    if stack == 0:
                        groups.append(s[start:i + 1])
                        start = i + 1  # Move start to the next character after the current closing parenthesis
            return groups
        
        groups = []
        start = 0
        while start < len(paren_string):
            if paren_string[start] == '(':
                groups.extend(separate_helper(paren_string, start))
                # Move start to the next character after the last processed group
                start += 1
            else:
                start += 1  # Skip non-parenthesis characters (if any)
        
        return groups

from typing import List
class ParenGroupSeparator:
    def separate_paren_groups(self, paren_string: str) -> List[str]:
        # Remove spaces from the input string
        paren_string = paren_string.replace(" ", "")
        def separate_helper(s, start):
            groups = []
            stack = 0
            for i, char in enumerate(s[start:], start):
                if char == '(':
                    stack += 1
                elif char == ')':
                    stack -= 1
                    if stack == 0:
                        groups.append(s[start:i + 1])
                        start = i + 1  # Move start to the next character after the current closing parenthesis
            return groups
        groups = []
        start = 0
        while start < len(paren_string):
            if paren_string[start] == '(':
                groups.extend(separate_helper(paren_string, start))
                # Move start to the next character after the last processed group
                start += 1
            else:
                start += 1  # Skip non-parenthesis characters (if any)
        return groups

'''
This file contains the ParenGroupSeparator class responsible for separating groups of nested parentheses.
'''
from typing import List
class ParenGroupSeparator:
    def separate_paren_groups(self, paren_string: str) -> List[str]:
        # Remove spaces from the input string
        paren_string = paren_string.replace(" ", "")
        def separate_helper(s, start):
            groups = []
            stack = 0
            for i, char in enumerate(s[start:], start):
                if char == '(':
                    stack += 1
                elif char == ')':
                    stack -= 1
                    if stack == 0:
                        groups.append(s[start:i + 1])
                        start = i + 1  # Move start to the next character after the current closing parenthesis
            return groups
        groups = []
        start = 0
        while start < len(paren_string):
            if paren_string[start] == '(':
                groups.extend(separate_helper(paren_string, start))
                # Move start to the next character after the last processed group
                start += 1
            else:
                start += 1  # Skip non-parenthesis characters (if any)
        return groups
