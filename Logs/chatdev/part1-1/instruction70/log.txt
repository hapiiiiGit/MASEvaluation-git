Main entry point for the Strange Sort application.

import tkinter as tk
from tkinter import messagebox
from sorting import strange_sort_list

class StrangeSortApp:
    '''
    GUI application for strange sorting of a list of integers.
    '''
    
    def __init__(self, master):
        self.master = master
        master.title("Strange Sort List")

        self.label = tk.Label(master, text="Enter a list of integers (comma-separated):")
        self.label.pack()

        self.entry = tk.Entry(master)
        self.entry.pack()

        self.sort_button = tk.Button(master, text="Sort", command=self.get_input)
        self.sort_button.pack()

        self.result_label = tk.Label(master, text="")
        self.result_label.pack()

    def get_input(self):
        '''
        Retrieve input from the user, process it, and display the result.
        '''
        input_text = self.entry.get()
        try:
            # Convert input string to a list of integers
            input_list = list(map(int, input_text.split(',')))
            sorted_list = strange_sort_list(input_list)
            self.display_result(sorted_list)
        except ValueError:
            messagebox.showerror("Input Error", "Please enter a valid list of integers.")

    def display_result(self, sorted_list):
        '''
        Display the sorted list in the GUI.
        '''
        self.result_label.config(text=f"Sorted List: {sorted_list}")

if __name__ == "__main__":
    root = tk.Tk()
    app = StrangeSortApp(root)
    root.mainloop()

'''
Module containing the strange_sort_list function.
'''

def strange_sort_list(lst):
    '''
    Given list of integers, return list in strange order.
    Strange sorting, is when you start with the minimum value,
    then maximum of the remaining integers, then minimum and so on.

    Examples:
    strange_sort_list([1, 2, 3, 4]) == [1, 4, 2, 3]
    strange_sort_list([5, 5, 5, 5]) == [5, 5, 5, 5]
    strange_sort_list([]) == []
    '''
    if not lst:
        return []

    sorted_lst = []
    lst.sort()
    
    while lst:
        sorted_lst.append(lst.pop(0))  # Add the minimum
        if lst:  # Check if there's still elements left
            sorted_lst.append(lst.pop(-1))  # Add the maximum

    return sorted_lst
```

def strange_sort_list(lst):
    '''
    Given list of integers, return list in strange order.
    Strange sorting, is when you start with the minimum value,
    then maximum of the remaining integers, then minimum and so on.
    Examples:
    strange_sort_list([1, 2, 3, 4]) == [1, 4, 2, 3]
    strange_sort_list([5, 5, 5, 5]) == [5, 5, 5, 5]
    strange_sort_list([]) == []
    '''
    if not lst:
        return []
    sorted_lst = []
    lst.sort()

        sorted_lst.append(lst.pop(0))  # Add the minimum
        if lst:  # Check if there's still elements left
            sorted_lst.append(lst.pop(-1))  # Add the maximum
    return sorted_lst

def strange_sort_list(lst):
    '''
    Given list of integers, return list in strange order.
    Strange sorting, is when you start with the minimum value,
    then maximum of the remaining integers, then minimum and so on.
    Examples:
    strange_sort_list([1, 2, 3, 4]) == [1, 4, 2, 3]
    strange_sort_list([5, 5, 5, 5]) == [5, 5, 5, 5]
    strange_sort_list([]) == []
    '''
    if not lst:
        return []
    sorted_lst = []
    lst.sort()

        sorted_lst.append(lst.pop(0))  # Add the minimum
        if lst:  # Check if there's still elements left
            sorted_lst.append(lst.pop(-1))  # Add the maximum
    return sorted_lst

def strange_sort_list(lst):
    '''
    Given list of integers, return list in strange order.
    Strange sorting, is when you start with the minimum value,
    then maximum of the remaining integers, then minimum and so on.
    Examples:
    strange_sort_list([1, 2, 3, 4]) == [1, 4, 2, 3]
    strange_sort_list([5, 5, 5, 5]) == [5, 5, 5, 5]
    strange_sort_list([]) == []
    '''
    if not lst:
        return []
    sorted_lst = []
    lst_copy = sorted(lst)  # Create a sorted copy of the list
    while lst_copy:
        sorted_lst.append(lst_copy.pop(0))  # Add the minimum
        if lst_copy:  # Check if there's still elements left
            sorted_lst.append(lst_copy.pop(-1))  # Add the maximum
    return sorted_lst

import tkinter as tk

class StrangeSortApp:
    '''
    GUI application for strange sorting of a list of integers.
    '''
    def __init__(self, master):
        self.master = master
        master.title("Strange Sort List")
        self.label = tk.Label(master, text="Enter a list of integers (comma-separated):")
        self.label.pack()
        self.entry = tk.Entry(master)
        self.entry.pack()
        self.sort_button = tk.Button(master, text="Sort", command=self.get_input)
        self.sort_button.pack()
        self.result_label = tk.Label(master, text="")
        self.result_label.pack()
    def get_input(self):
        '''
        Retrieve input from the user, process it, and display the result.
        '''
        input_text = self.entry.get()
        try:
            # Convert input string to a list of integers
            input_list = list(map(int, input_text.split(',')))
            sorted_list = strange_sort_list(input_list)
            self.display_result(sorted_list)
        except ValueError:
            messagebox.showerror("Input Error", "Please enter a valid list of integers.")
    def display_result(self, sorted_list):
        '''
        Display the sorted list in the GUI.
        '''
        self.result_label.config(text=f"Sorted List: {sorted_list}")

    app = StrangeSortApp(root)

def strange_sort_list(lst):
    '''
    Given list of integers, return list in strange order.
    Strange sorting, is when you start with the minimum value,
    then maximum of the remaining integers, then minimum and so on.
    Examples:
    strange_sort_list([1, 2, 3, 4]) == [1, 4, 2, 3]
    strange_sort_list([5, 5, 5, 5]) == [5, 5, 5, 5]
    strange_sort_list([]) == []
    '''
    if not lst:
        return []
    sorted_lst = []
    lst_copy = sorted(lst)  # Create a sorted copy of the list
    while lst_copy:
        sorted_lst.append(lst_copy.pop(0))  # Add the minimum
        if lst_copy:  # Check if there's still elements left
            sorted_lst.append(lst_copy.pop(-1))  # Add the maximum
    return sorted_lst

    '''

        self.master = master
        master.title("Strange Sort List")
        self.label = tk.Label(master, text="Enter a list of integers (comma-separated):")
        self.label.pack()
        self.entry = tk.Entry(master)
        self.entry.pack()
        self.sort_button = tk.Button(master, text="Sort", command=self.get_input)
        self.sort_button.pack()
        self.result_label = tk.Label(master, text="")
        self.result_label.pack()
    def get_input(self):
        '''
        Retrieve input from the user, process it, and display the result.
        '''
        input_text = self.entry.get()
        try:
            # Convert input string to a list of integers
            input_list = list(map(int, input_text.split(',')))
            sorted_list = strange_sort_list(input_list)
            self.display_result(sorted_list)
        except ValueError:
            messagebox.showerror("Input Error", "Please enter a valid list of integers.")
    def display_result(self, sorted_list):
        '''
        Display the sorted list in the GUI.
        '''
        self.result_label.config(text=f"Sorted List: {sorted_list}")

    app = StrangeSortApp(root)

'''
Module containing the strange_sort_list function.
'''
def strange_sort_list(lst):
    '''
    Given list of integers, return list in strange order.
    Strange sorting, is when you start with the minimum value,
    then maximum of the remaining integers, then minimum and so on.
    Examples:
    strange_sort_list([1, 2, 3, 4]) == [1, 4, 2, 3]
    strange_sort_list([5, 5, 5, 5]) == [5, 5, 5, 5]
    strange_sort_list([]) == []
    '''
    if not lst:
        return []
    sorted_lst = []
    lst_copy = sorted(lst)  # Create a sorted copy of the list
    while lst_copy:
        sorted_lst.append(lst_copy.pop(0))  # Add the minimum
        if lst_copy:  # Check if there's still elements left
            sorted_lst.append(lst_copy.pop(-1))  # Add the maximum
    return sorted_lst
