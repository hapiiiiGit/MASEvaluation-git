from tkinter import Tk, Label, Entry, Button, StringVar
from max_element import max_element

class MaxElementApp:
    def __init__(self, master):
        self.master = master
        master.title("Max Element Finder")

        self.label = Label(master, text="Enter a list of numbers (comma-separated):")
        self.label.pack()

        self.input_var = StringVar()
        self.entry = Entry(master, textvariable=self.input_var)
        self.entry.pack()

        self.find_button = Button(master, text="Find Max", command=self.find_max)
        self.find_button.pack()

        self.result_label = Label(master, text="")
        self.result_label.pack()

    def find_max(self):
        input_text = self.input_var.get()
        try:
            # Convert input string to a list of integers
            input_list = [int(x.strip()) for x in input_text.split(',')]
            max_value = max_element(input_list)
            self.result_label.config(text=f"Maximum Element: {max_value}")
        except ValueError:
            self.result_label.config(text="Please enter valid integers.")

if __name__ == "__main__":
    root = Tk()
    app = MaxElementApp(root)
    root.mainloop()
```

def max_element(l: list):
    """Return maximum element in the list.

    3

    if not l:

    max_val = l[0]
    for item in l:
        if item > max_val:
            max_val = item
    return max_val

class MaxElementApp:
    def __init__(self, master):

        self.label = Label(master, text="Enter a list of numbers (comma-separated):")
        self.label.pack()

        self.find_button = Button(master, text="Find Max", command=self.find_max)
        self.find_button.pack()

    def find_max(self):
        input_text = self.input_var.get()

            # Convert input string to a list of integers

            max_value = max_element(input_list)
            self.result_label.config(text=f"Maximum Element: {max_value}")
        except ValueError:
            self.result_label.config(text="Please enter valid integers.")

    root = Tk()

    root.mainloop()

'''
Main application file for finding the maximum element in a list.
'''
from tkinter import Tk, Label, Entry, Button, StringVar
from max_element import max_element
class MaxElementApp:
    def __init__(self, master):
        self.master = master
        master.title("Max Element Finder")
        self.label = Label(master, text="Enter a list of numbers (comma-separated):")
        self.label.pack()
        self.input_var = StringVar()
        self.entry = Entry(master, textvariable=self.input_var)
        self.entry.pack()
        self.find_button = Button(master, text="Find Max", command=self.find_max)
        self.find_button.pack()
        self.result_label = Label(master, text="")
        self.result_label.pack()
    def find_max(self):
        input_text = self.input_var.get()
        try:
            # Convert input string to a list of integers
            input_list = [int(x.strip()) for x in input_text.split(',')]
            max_value = max_element(input_list)
            self.result_label.config(text=f"Maximum Element: {max_value}")
        except ValueError:
            self.result_label.config(text="Please enter valid integers.")
if __name__ == "__main__":
    root = Tk()
    app = MaxElementApp(root)
    root.mainloop()

```python
'''
Module containing the function to find the maximum element in a list.
'''
def max_element(l: list):
    """Return maximum element in the list.
    >>> max_element([1, 2, 3])
    3
    >>> max_element([5, 3, -5, 2, -3, 3, 9, 0, 123, 1, -10])
    123
    """
    if not l:
        raise ValueError("The list is empty.")
    max_val = l[0]
    for item in l:
        if item > max_val:
            max_val = item
    return max_val

'''
Main application file for finding the maximum element in a list.
'''
from tkinter import Tk, Label, Entry, Button, StringVar
from max_element import max_element
class MaxElementApp:
    def __init__(self, master):
        self.master = master
        master.title("Max Element Finder")
        self.label = Label(master, text="Enter a list of numbers (comma-separated):")
        self.label.pack()
        self.input_var = StringVar()
        self.entry = Entry(master, textvariable=self.input_var)
        self.entry.pack()
        self.find_button = Button(master, text="Find Max", command=self.find_max)
        self.find_button.pack()
        self.result_label = Label(master, text="")
        self.result_label.pack()
    def find_max(self):
        input_text = self.input_var.get()
        try:
            # Convert input string to a list of integers
            input_list = [int(x.strip()) for x in input_text.split(',')]
            max_value = max_element(input_list)
            self.result_label.config(text=f"Maximum Element: {max_value}")
        except ValueError:
            self.result_label.config(text="Please enter valid integers.")
if __name__ == "__main__":
    root = Tk()
    app = MaxElementApp(root)
    root.mainloop()

'''
Module containing the function to find the maximum element in a list.
'''
def max_element(l: list):
    """Return maximum element in the list.
    >>> max_element([1, 2, 3])
    3
    >>> max_element([5, 3, -5, 2, -3, 3, 9, 0, 123, 1, -10])
    123
    """
    if not l:
        raise ValueError("The list is empty.")
    max_val = l[0]
    for item in l:
        if item > max_val:
            max_val = item
    return max_val
