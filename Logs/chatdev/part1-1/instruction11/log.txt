def string_xor(a: str, b: str) -> str:
    """ Input are two strings a and b consisting only of 1s and 0s.
    Perform binary XOR on these inputs and return result also as a string.
    >>> string_xor('010', '110')
    '100'

'''
This file contains the logic for performing XOR operation on two binary strings.
'''
def string_xor(a: str, b: str) -> str:
    """ Input are two strings a and b consisting only of 1s and 0s.
    Perform binary XOR on these inputs and return result also as a string.
    >>> string_xor('010', '110')
    '100'
    """
    if len(a) != len(b):
        raise ValueError("Input strings must have the same length")
    # Validate that both strings contain only '0' and '1'
    if not all(char in '01' for char in a) or not all(char in '01' for char in b):
        raise ValueError("Input strings must consist only of '0's and '1's")
    result = ""
    for i in range(len(a)):
        if a[i] == b[i]:
            result += '0'
        else:
            result += '1'
    return result

'''
This file contains the logic for performing XOR operation on two binary strings.
'''
def string_xor(a: str, b: str) -> str:
    """ Input are two strings a and b consisting only of 1s and 0s.
    Perform binary XOR on these inputs and return result also as a string.
    >>> string_xor('010', '110')
    '100'
    """
    if len(a) != len(b):
        raise ValueError("Input strings must have the same length")
    # Validate that both strings contain only '0' and '1'
    if not all(char in '01' for char in a) or not all(char in '01' for char in b):
        raise ValueError("Input strings must consist only of '0's and '1's")
    result = ""
    for i in range(len(a)):
        if a[i] == b[i]:
            result += '0'
        else:
            result += '1'
    return result

'''
This file contains the logic for performing XOR operation on two binary strings.
'''
def string_xor(a: str, b: str) -> str:
    """ Input are two strings a and b consisting only of 1s and 0s.
    Perform binary XOR on these inputs and return result also as a string.
    >>> string_xor('010', '110')
    '100'
    """
    if len(a) != len(b):
        raise ValueError("Input strings must have the same length")
    # Validate that both strings contain only '0' and '1'
    if not all(char in '01' for char in a) or not all(char in '01' for char in b):
        raise ValueError("Input strings must consist only of '0's and '1's")
    result = ""
    for i in range(len(a)):
        if a[i] == b[i]:
            result += '0'
        else:
            result += '1'
    return result

def string_xor(a: str, b: str) -> str:
    """ Input are two strings a and b consisting only of 1s and 0s.
    Perform binary XOR on these inputs and return result also as a string.
    >>> string_xor('010', '110')
    '100'
    """
    if len(a) != len(b):
        raise ValueError("Input strings must have the same length")
    # Validate that both strings contain only '0' and '1'
    if not all(char in '01' for char in a) or not all(char in '01' for char in b):
        raise ValueError("Input strings must consist only of '0's and '1's")
    result = ""
    for i in range(len(a)):
        if a[i] == b[i]:
            result += '0'
        else:
            result += '1'
    return result

'''
This file contains the logic for performing XOR operation on two binary strings.
'''
def string_xor(a: str, b: str) -> str:
    """ Input are two strings a and b consisting only of 1s and 0s.
    Perform binary XOR on these inputs and return result also as a string.
    >>> string_xor('010', '110')
    '100'
    """
    if len(a) != len(b):
        raise ValueError("Input strings must have the same length")
    # Validate that both strings contain only '0' and '1'
    if not all(char in '01' for char in a) or not all(char in '01' for char in b):
        raise ValueError("Input strings must consist only of '0's and '1's")
    result = ""
    for i in range(len(a)):
        result += '1' if a[i] != b[i] else '0'
    return result

'''
This file contains the logic for performing XOR operation on two binary strings.
'''
def string_xor(a: str, b: str) -> str:
    """ Input are two strings a and b consisting only of 1s and 0s.
    Perform binary XOR on these inputs and return result also as a string.
    >>> string_xor('010', '110')
    '100'
    """
    if len(a) != len(b):
        raise ValueError("Input strings must have the same length")
    # Validate that both strings contain only '0' and '1'
    if not all(char in '01' for char in a) or not all(char in '01' for char in b):
        raise ValueError("Input strings must consist only of '0's and '1's")
    result = ""
    for i in range(len(a)):
        result += '1' if a[i] != b[i] else '0'
    return result
