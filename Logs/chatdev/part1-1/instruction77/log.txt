Main application file for the Cube Checker GUI.

from tkinter import Tk, Label, Entry, Button, StringVar
from cube_checker import iscube

class CubeCheckerApp:
    '''
    Class to create the Cube Checker application.
    '''

    def __init__(self, master):
        '''
        Initializes the GUI components.
        '''
        self.master = master
        master.title("Cube Checker")

        self.result_label = Label(master, text="")
        self.result_label.pack()

        Retrieves input, checks if it's a cube, and displays the result.

            number = int(self.input_value.get())

            self.result_label.config(text="Please enter a valid integer.")

if __name__ == "__main__":

def iscube(a):
    '''
    Write a function that takes an integer a and returns True 
    if this integer is a cube of some integer number.
    Note: you may assume the input is always valid.
    Examples:
    iscube(1) ==> True
    iscube(2) ==> False
    iscube(-1) ==> True
    iscube(64) ==> True
    iscube(0) ==> True
    iscube(180) ==> False
    '''
    # Calculate the cube root and round it to the nearest integer

    # Check if the cube of the rounded cube root equals the original number
    return cube_root ** 3 == a

class CubeCheckerApp:

'''
Main application file for the Cube Checker GUI.

from tkinter import Tk, Label, Entry, Button, StringVar
from cube_checker import iscube
class CubeCheckerApp:
    '''
    Class to create the Cube Checker application.
    '''
    def __init__(self, master):
        '''
        Initializes the GUI components.
        '''
        self.master = master
        master.title("Cube Checker")
        self.label = Label(master, text="Enter an integer:")
        self.label.pack()
        self.input_value = StringVar()
        self.entry = Entry(master, textvariable=self.input_value)
        self.entry.pack()
        self.check_button = Button(master, text="Check Cube", command=self.check_cube)
        self.check_button.pack()
        self.result_label = Label(master, text="")
        self.result_label.pack()
    def check_cube(self):
        '''
        Retrieves input, checks if it's a cube, and displays the result.
        '''
        try:
            number = int(self.input_value.get())
            if iscube(number):
                self.result_label.config(text=f"{number} is a perfect cube.")
            else:
                self.result_label.config(text=f"{number} is not a perfect cube.")
        except ValueError:
            self.result_label.config(text="Please enter a valid integer.")
if __name__ == "__main__":
    root = Tk()
    app = CubeCheckerApp(root)
    root.mainloop()

'''
Module containing the function to check if a number is a perfect cube.
'''
def iscube(a):
    '''
    Write a function that takes an integer a and returns True 
    if this integer is a cube of some integer number.
    Note: you may assume the input is always valid.
    Examples:
    iscube(1) ==> True
    iscube(2) ==> False
    iscube(-1) ==> True
    iscube(64) ==> True
    iscube(0) ==> True
    iscube(180) ==> False
    '''
    # Calculate the cube root and round it to the nearest integer
    cube_root = round(a ** (1/3))
    # Check if the cube of the rounded cube root equals the original number
    return cube_root ** 3 == a

from tkinter import Tk, Label, Entry, Button, StringVar
from cube_checker import iscube
class CubeCheckerApp:
    '''
    Class to create the Cube Checker application.
    '''
    def __init__(self, master):
        '''
        Initializes the GUI components.
        '''
        self.master = master
        master.title("Cube Checker")
        self.label = Label(master, text="Enter an integer:")
        self.label.pack()
        self.input_value = StringVar()
        self.entry = Entry(master, textvariable=self.input_value)
        self.entry.pack()
        self.check_button = Button(master, text="Check Cube", command=self.check_cube)
        self.check_button.pack()
        self.result_label = Label(master, text="")
        self.result_label.pack()
    def check_cube(self):
        '''
        Retrieves input, checks if it's a cube, and displays the result.
        '''
        try:
            number = int(self.input_value.get())
            if iscube(number):
                self.result_label.config(text=f"{number} is a perfect cube.")
            else:
                self.result_label.config(text=f"{number} is not a perfect cube.")
        except ValueError:
            self.result_label.config(text="Please enter a valid integer.")
if __name__ == "__main__":
    root = Tk()
    app = CubeCheckerApp(root)
    root.mainloop()

'''
Module containing the function to check if a number is a perfect cube.
'''
def iscube(a):
    '''
    Write a function that takes an integer a and returns True 
    if this integer is a cube of some integer number.
    Note: you may assume the input is always valid.
    Examples:
    iscube(1) ==> True
    iscube(2) ==> False
    iscube(-1) ==> True
    iscube(64) ==> True
    iscube(0) ==> True
    iscube(180) ==> False
    '''
    # Calculate the cube root and round it to the nearest integer
    cube_root = round(a ** (1/3))
    # Check if the cube of the rounded cube root equals the original number
    return cube_root ** 3 == a
