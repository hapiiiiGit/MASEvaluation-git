import tkinter as tk
from tkinter import messagebox
from utils import same_chars

class MainApp:
    def __init__(self, master):
        self.master = master
        master.title("Same Characters Checker")

        self.label1 = tk.Label(master, text="Enter first string:")
        self.label1.pack()

        self.entry1 = tk.Entry(master)
        self.entry1.pack()

        self.label2 = tk.Label(master, text="Enter second string:")
        self.label2.pack()

        self.entry2 = tk.Entry(master)
        self.entry2.pack()

        self.check_button = tk.Button(master, text="Check", command=self.check_same_chars)
        self.check_button.pack()

        self.result_label = tk.Label(master, text="")
        self.result_label.pack()

        s0 = self.entry1.get()
        s1 = self.entry2.get()

        if result:
            messagebox.showinfo("Result", "The strings have the same characters.")

if __name__ == "__main__":

'''
Utility file containing the function to check if two words have the same characters.
'''

def same_chars(s0: str, s1: str) -> bool:
    """
    Check if two words have the same characters.
    >>> same_chars('eabcdzzzz', 'dddzzzzzzzddeddabc')
    True
    >>> same_chars('abcd', 'dddddddabc')
    True
    >>> same_chars('dddddddabc', 'abcd')
    True
    >>> same_chars('eabcd', 'dddddddabc')
    False
    >>> same_chars('abcd', 'dddddddabce')
    False
    >>> same_chars('eabcdzzzz', 'dddzzzzzzzddddabc')
    False
    """
    return set(s0) == set(s1)

import tkinter as tk
from tkinter import messagebox
from utils import same_chars
class MainApp:
    def __init__(self, master):
        self.master = master
        master.title("Same Characters Checker")
        self.label1 = tk.Label(master, text="Enter first string:")
        self.label1.pack()
        self.entry1 = tk.Entry(master)
        self.entry1.pack()

        self.label2.pack()

        self.check_button.pack()
        self.result_label = tk.Label(master, text="")

    def check_same_chars(self):
        s0 = self.entry1.get()

        if result:
            messagebox.showinfo("Result", "The strings have the same characters.")
        else:
            messagebox.showinfo("Result", "The strings do not have the same characters.")
if __name__ == "__main__":
    root = tk.Tk()
    app = MainApp(root)
    root.mainloop()

def same_chars(s0: str, s1: str) -> bool:
    """
    Check if two words have the same unique characters.
    >>> same_chars('eabcdzzzz', 'dddzzzzzzzddeddabc')
    True
    >>> same_chars('abcd', 'dddddddabc')
    True
    >>> same_chars('dddddddabc', 'abcd')
    True
    >>> same_chars('eabcd', 'dddddddabc')
    False
    >>> same_chars('abcd', 'dddddddabce')
    False
    >>> same_chars('eabcdzzzz', 'dddzzzzzzzddddabc')
    False
    >>> same_chars('aabb', 'abab')
    True
    """
    return set(s0) == set(s1)
