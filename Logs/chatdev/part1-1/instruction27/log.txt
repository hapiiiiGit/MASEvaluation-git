'''
This module contains the function to flip the case of characters in a string.

def flip_case(string: str) -> str:
    """ For a given string, flip lowercase characters to uppercase and uppercase to lowercase.
    >>> flip_case('Hello')
    'hELLO'
    """
    return string.swapcase()

import tkinter as tk
from tkinter import messagebox
from flip_case import flip_case

class FlipCaseApp:
    def __init__(self, root):
        self.root = root
        self.root.title("Flip Case Application")

        # Input Label
        self.input_label = tk.Label(root, text="Enter String:")
        self.input_label.pack()

        # Input Entry
        self.input_entry = tk.Entry(root, width=50)
        self.input_entry.pack()

        # Flip Button
        self.flip_button = tk.Button(root, text="Flip Case", command=self.on_flip)
        self.flip_button.pack()

        # Output Label
        self.output_label = tk.Label(root, text="Flipped String:")
        self.output_label.pack()

        # Output Entry
        self.output_entry = tk.Entry(root, width=50)
        self.output_entry.pack()

    def on_flip(self):
        input_string = self.input_entry.get()
        if input_string:
            flipped_string = flip_case(input_string)
            self.output_entry.delete(0, tk.END)  # Clear previous output
            self.output_entry.insert(0, flipped_string)  # Insert new output
        else:
            messagebox.showwarning("Input Error", "Please enter a string to flip.")

if __name__ == "__main__":
    root = tk.Tk()
    app = FlipCaseApp(root)
    root.mainloop()

def flip_case(string: str) -> str:
    """ For a given string, flip lowercase characters to uppercase and uppercase to lowercase.
    >>> flip_case('Hello')
    'hELLO'

```python
'''
This is the main application file that creates the GUI for the Flip Case application.
'''
import tkinter as tk
from tkinter import messagebox
from flip_case import flip_case
class FlipCaseApp:
    def __init__(self, root):
        self.root = root
        self.root.title("Flip Case Application")
        # Input Label
        self.input_label = tk.Label(root, text="Enter String:")
        self.input_label.pack()
        # Input Entry
        self.input_entry = tk.Entry(root, width=50)
        self.input_entry.pack()
        # Flip Button
        self.flip_button = tk.Button(root, text="Flip Case", command=self.on_flip)
        self.flip_button.pack()
        # Output Label
        self.output_label = tk.Label(root, text="Flipped String:")
        self.output_label.pack()
        # Output Entry
        self.output_entry = tk.Entry(root, width=50)
        self.output_entry.pack()
    def on_flip(self):
        input_string = self.input_entry.get()
        if input_string:
            flipped_string = flip_case(input_string)
            self.output_entry.delete(0, tk.END)  # Clear previous output
            self.output_entry.insert(0, flipped_string)  # Insert new output
        else:
            messagebox.showwarning("Input Error", "Please enter a string to flip.")
if __name__ == "__main__":
    root = tk.Tk()
    app = FlipCaseApp(root)
    root.mainloop()
