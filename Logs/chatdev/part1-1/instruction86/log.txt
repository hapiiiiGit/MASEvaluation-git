Main application file for the AntiShuffle application.
'''

from tkinter import Tk, Label, Entry, Button, StringVar
from anti_shuffle import anti_shuffle

class AntiShuffleApp:
    '''
    GUI application for anti-shuffling words in a string.
    '''

    def __init__(self):
        self.root = Tk()
        self.root.title("Anti Shuffle Application")

        self.input_label = Label(self.root, text="Enter a string:")
        self.input_label.pack()

        self.input_text = StringVar()
        self.input_entry = Entry(self.root, textvariable=self.input_text, width=50)
        self.input_entry.pack()

        self.submit_button = Button(self.root, text="Submit", command=self.on_submit)
        self.submit_button.pack()

        self.result_label = Label(self.root, text="")
        self.result_label.pack()

        self.root.mainloop()

    def anti_shuffle(self, s):
        '''
        Sorts characters in each word of the input string in ascending order.
        '''
        words = s.split(' ')
        ordered_words = [''.join(sorted(word)) for word in words]
        return ' '.join(ordered_words)

    def on_submit(self):
        '''
        Handles the submit button click event.
        '''
        input_string = self.input_text.get()
        result = self.anti_shuffle(input_string)
        self.result_label.config(text=result)

if __name__ == "__main__":
    AntiShuffleApp()
```

'''
Main application file for the AntiShuffle application.
'''
from tkinter import Tk, Label, Entry, Button, StringVar
from anti_shuffle import anti_shuffle
class AntiShuffleApp:
    '''
    GUI application for anti-shuffling words in a string.
    '''
    def __init__(self):
        self.root = Tk()
        self.root.title("Anti Shuffle Application")
        self.input_label = Label(self.root, text="Enter a string:")
        self.input_label.pack()
        self.input_text = StringVar()
        self.input_entry = Entry(self.root, textvariable=self.input_text, width=50)
        self.input_entry.pack()
        self.submit_button = Button(self.root, text="Submit", command=self.on_submit)
        self.submit_button.pack()
        self.result_label = Label(self.root, text="")
        self.result_label.pack()
        self.root.mainloop()
    def anti_shuffle(self, s):
        '''
        Sorts characters in each word of the input string in ascending order.
        '''
        words = s.split(' ')
        ordered_words = [''.join(sorted(word)) for word in words]
        return ' '.join(ordered_words)
    def on_submit(self):
        '''
        Handles the submit button click event.
        '''
        input_string = self.input_text.get()
        result = self.anti_shuffle(input_string)
        self.result_label.config(text=result)
if __name__ == "__main__":
    AntiShuffleApp()

'''

from tkinter import Tk, Label, Entry, Button, StringVar
from anti_shuffle import anti_shuffle
class AntiShuffleApp:
    '''
    GUI application for anti-shuffling words in a string.
    '''
    def __init__(self):
        self.root = Tk()
        self.root.title("Anti Shuffle Application")
        self.input_label = Label(self.root, text="Enter a string:")
        self.input_label.pack()
        self.input_text = StringVar()
        self.input_entry = Entry(self.root, textvariable=self.input_text, width=50)
        self.input_entry.pack()
        self.submit_button = Button(self.root, text="Submit", command=self.on_submit)
        self.submit_button.pack()
        self.result_label = Label(self.root, text="")
        self.result_label.pack()
        self.root.mainloop()
    def anti_shuffle(self, s):
        '''
        Sorts characters in each word of the input string in ascending order.
        '''
        words = s.split(' ')
        ordered_words = [''.join(sorted(word)) for word in words]
        return ' '.join(ordered_words)
    def on_submit(self):
        '''
        Handles the submit button click event.
        '''
        input_string = self.input_text.get()
        result = self.anti_shuffle(input_string)
        self.result_label.config(text=result)
if __name__ == "__main__":
    AntiShuffleApp()
