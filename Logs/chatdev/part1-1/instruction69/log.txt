'''
Module containing the search function to find the greatest integer 
with the specified frequency condition.
'''

def search(lst):
    frequency = {}
    
    # Count the frequency of each number in the list
    for number in lst:
        if number in frequency:
            frequency[number] += 1
        else:
            frequency[number] = 1
    
    # Find the greatest integer that meets the frequency condition
    result = -1
    for number in frequency:
        if number > 0 and frequency[number] >= number:
            result = max(result, number)
    
    return result
```

#### gui.py

```python
'''
Module to create the graphical user interface for the application.
'''

import tkinter as tk
from tkinter import messagebox
from search import search

class App:
    '''
    Main application class that handles the GUI and user interactions.
    '''
    
    def __init__(self):
        self.window = tk.Tk()
        self.window.title("Greatest Integer Search")
        self.create_widgets()

    def create_widgets(self):
        '''
        Create and arrange the GUI components.
        '''
        self.label = tk.Label(self.window, text="Enter a list of positive integers (comma-separated):")
        self.label.pack()

        self.entry = tk.Entry(self.window, width=50)
        self.entry.pack()

        self.button = tk.Button(self.window, text="Search", command=self.perform_search)
        self.button.pack()

        self.result_label = tk.Label(self.window, text="")
        self.result_label.pack()

    def perform_search(self):
        '''
        Perform the search operation and display the result.
        '''
        input_text = self.entry.get()
        try:
            lst = list(map(int, input_text.split(',')))
            result = search(lst)
            if result == -1:
                self.result_label.config(text="No such value exists.")
            else:
                self.result_label.config(text=f"The greatest integer is: {result}")
        except ValueError:
            messagebox.showerror("Input Error", "Please enter valid integers separated by commas.")

    def run(self):
        '''
        Start the application.
        '''
        self.window.mainloop()
```

'''
Module containing the search function to find the greatest integer 
with the specified frequency condition.
'''
def search(lst):
    frequency = {}
    # Count the frequency of each number in the list
    for number in lst:
        if number in frequency:
            frequency[number] += 1
        else:
            frequency[number] = 1
    # Find the greatest integer that meets the frequency condition
    result = -1
    for number in frequency:
        if number > 0 and frequency[number] >= number:
            result = max(result, number)
    return result

        self.button = tk.Button(self.window, text="Search", command=self.perform_search)
        self.button.pack()
        self.result_label = tk.Label(self.window, text="")
        self.result_label.pack()
    def perform_search(self):
        '''
        Perform the search operation and display the result.
        '''
        input_text = self.entry.get()
        try:
            lst = list(map(int, input_text.split(',')))
            result = search(lst)
            if result == -1:
                self.result_label.config(text="No such value exists.")
            else:
                self.result_label.config(text=f"The greatest integer is: {result}")
        except ValueError:
            messagebox.showerror("Input Error", "Please enter valid integers separated by commas.")
    def run(self):
        '''
        Start the application.
        '''
        self.window.mainloop()

def search(lst):

'''
Module containing the search function to find the greatest integer 
with the specified frequency condition.
'''
def search(lst):
    frequency = {}
    # Count the frequency of each number in the list
    for number in lst:
        if number in frequency:
            frequency[number] += 1
        else:
            frequency[number] = 1
    # Find the greatest integer that meets the frequency condition
    result = -1
    for number in frequency:
        if number > 0 and frequency[number] >= number:
            result = max(result, number)
    return result
```

class App:
    '''
    Main application class that handles the GUI and user interactions.
    '''

        except ValueError:

    def run(self):
        '''
        Start the application.
        '''
        self.window.mainloop()

'''
Module containing the search function to find the greatest integer 
with the specified frequency condition.
'''
def search(lst):
    frequency = {}
    # Count the frequency of each number in the list
    for number in lst:
        if number in frequency:
            frequency[number] += 1
        else:
            frequency[number] = 1
    # Find the greatest integer that meets the frequency condition
    result = -1
    for number in frequency:
        if number > 0 and frequency[number] >= number:
            result = max(result, number)
    return result

        '''

                self.result_label.config(text="No such value exists.")

        except ValueError:
