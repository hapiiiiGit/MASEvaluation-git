'''
Module containing the closest_integer function.
'''

def closest_integer(value):
    '''
    Create a function that takes a value (string) representing a number
    and returns the closest integer to it. If the number is equidistant
    from two integers, round it away from zero.

    Examples
    >>> closest_integer("10")
    10
    >>> closest_integer("15.3")
    15
    >>> closest_integer("14.5")
    15
    >>> closest_integer("-14.5")
    -15
    '''
    num = float(value)
    if num > 0:
        return int(num + 0.5)
    else:
        return int(num - 0.5)
```

### File: gui.py

```python
'''
Module to create the GUI for the closest integer application using Tkinter.
'''

import tkinter as tk
from tkinter import messagebox
from closest_integer import closest_integer

class MainApp:
    '''
    Main application class to handle the GUI.
    '''
    def __init__(self):
        self.root = tk.Tk()
        self.root.title("Closest Integer Finder")
        self.input_frame = InputFrame(self.root, self.calculate)
        self.input_frame.pack()
        self.output_frame = OutputFrame(self.root)
        self.output_frame.pack()

    def calculate(self):
        '''
        Calculate the closest integer and update the output frame.
        '''
        value = self.input_frame.get_input()
        try:
            result = closest_integer(value)
            self.output_frame.update_output(result)
        except ValueError:
            messagebox.showerror("Invalid Input", "Please enter a valid number.")

    def run(self):
        '''
        Run the Tkinter main loop.
        '''
        self.root.mainloop()

class InputFrame(tk.Frame):
    '''
    Frame for user input.
    '''
    def __init__(self, master, calculate_callback):
        super().__init__(master)
        self.label = tk.Label(self, text="Enter a number:")
        self.label.pack()
        self.entry = tk.Entry(self)
        self.entry.pack()
        self.calculate_button = tk.Button(self, text="Calculate", command=calculate_callback)
        self.calculate_button.pack()

    def get_input(self):
        '''
        Get the input from the entry field.
        '''
        return self.entry.get()

class OutputFrame(tk.Frame):
    '''
    Frame for displaying output.
    '''
    def __init__(self, master):
        super().__init__(master)
        self.label = tk.Label(self, text="Closest Integer:")
        self.label.pack()
        self.output_label = tk.Label(self, text="")
        self.output_label.pack()

    def update_output(self, result):
        '''
        Update the output label with the result.
        '''
        self.output_label.config(text=str(result))
```

def closest_integer(value):
    '''
    Create a function that takes a value (string) representing a number
    and returns the closest integer to it. If the number is equidistant
    from two integers, round it away from zero.
    Examples
    >>> closest_integer("10")
    10

    15

    15

    -15

        return int(num + 0.5)
    else:
        return int(num - 0.5)

def closest_integer(value):
    '''
    Create a function that takes a value (string) representing a number
    and returns the closest integer to it. If the number is equidistant
    from two integers, round it away from zero.
    Examples
    >>> closest_integer("10")
    10
    >>> closest_integer("15.3")
    15
    >>> closest_integer("14.5")
    15
    >>> closest_integer("-14.5")
    -15
    '''
    num = float(value)
    if num > 0:
        return int(num + 0.5)
    else:
        return int(num - 0.5)

def closest_integer(value):
    '''
    Create a function that takes a value (string) representing a number
    and returns the closest integer to it. If the number is equidistant
    from two integers, round it away from zero.
    Examples
    >>> closest_integer("10")
    10
    >>> closest_integer("15.3")
    15
    >>> closest_integer("14.5")
    15
    >>> closest_integer("-14.5")
    -15
    '''
    num = float(value)
    if num > 0:
        return int(num + 0.5)
    else:
        return int(num - 0.5)

def closest_integer(value):
    '''
    Create a function that takes a value (string) representing a number
    and returns the closest integer to it. If the number is equidistant
    from two integers, round it away from zero.
    Examples
    >>> closest_integer("10")
    10
    >>> closest_integer("15.3")
    15
    >>> closest_integer("14.5")
    15
    >>> closest_integer("-14.5")
    -15
    '''
    num = float(value)
    if num.is_integer():
        return int(num)
    elif num > 0:
        return int(num + 0.5)
    else:
        return int(num - 0.5)

def closest_integer(value):
    '''
    Create a function that takes a value (string) representing a number
    and returns the closest integer to it. If the number is equidistant
    from two integers, round it away from zero.
    Examples
    >>> closest_integer("10")
    10
    >>> closest_integer("15.3")
    15
    >>> closest_integer("14.5")
    15
    >>> closest_integer("-14.5")
    -15
    '''
    num = float(value)
    if num.is_integer():
        return int(num)
    elif num > 0:
        return int(num + 0.5)
    else:
        return int(num - 0.5)

'''

def closest_integer(value):
    '''
    Create a function that takes a value (string) representing a number
    and returns the closest integer to it. If the number is equidistant
    from two integers, round it away from zero.
    Examples
    >>> closest_integer("10")
    10
    >>> closest_integer("15.3")
    15
    >>> closest_integer("14.5")
    15
    >>> closest_integer("-14.5")
    -15
    '''
    num = float(value)
    if num.is_integer():
        return int(num)
    elif num > 0:
        return int(num + 0.5)
    else:
        return int(num - 0.5)

'''
Module to create the GUI for the closest integer application using Tkinter.

import tkinter as tk
from tkinter import messagebox
from closest_integer import closest_integer
class MainApp:
    '''
    Main application class to handle the GUI.
    '''
    def __init__(self):
        self.root = tk.Tk()
        self.root.title("Closest Integer Finder")
        self.input_frame = InputFrame(self.root, self.calculate)
        self.input_frame.pack()
        self.output_frame = OutputFrame(self.root)
        self.output_frame.pack()
    def calculate(self):
        '''
        Calculate the closest integer and update the output frame.
        '''
        value = self.input_frame.get_input()
        try:
            result = closest_integer(value)
            self.output_frame.update_output(result)
        except ValueError:
            messagebox.showerror("Invalid Input", "Please enter a valid number.")
    def run(self):
        '''
        Run the Tkinter main loop.
        '''
        self.root.mainloop()
class InputFrame(tk.Frame):
    '''
    Frame for user input.
    '''
    def __init__(self, master, calculate_callback):
        super().__init__(master)
        self.label = tk.Label(self, text="Enter a number:")
        self.label.pack()
        self.entry = tk.Entry(self)
        self.entry.pack()
        self.calculate_button = tk.Button(self, text="Calculate", command=calculate_callback)
        self.calculate_button.pack()
    def get_input(self):
        '''
        Get the input from the entry field.
        '''
        return self.entry.get()
class OutputFrame(tk.Frame):
    '''
    Frame for displaying output.
    '''
    def __init__(self, master):
        super().__init__(master)
        self.label = tk.Label(self, text="Closest Integer:")
        self.label.pack()
        self.output_label = tk.Label(self, text="")
        self.output_label.pack()
    def update_output(self, result):
        '''
        Update the output label with the result.
        '''
        self.output_label.config(text=str(result))

def closest_integer(value):
    '''
    Create a function that takes a value (string) representing a number
    and returns the closest integer to it. If the number is equidistant
    from two integers, round it away from zero.
    Examples
    >>> closest_integer("10")
    10
    >>> closest_integer("15.3")
    15
    >>> closest_integer("14.5")
    15
    >>> closest_integer("-14.5")
    -15
    '''
    num = float(value)
    if num.is_integer():
        return int(num)
    elif num > 0:
        return int(num + 0.5)
    else:
        return int(num - 0.5)

# Closest Integer Finder

A simple application to find the closest integer to a given number represented as a string. The application rounds away from zero in cases where the number is equidistant from two integers.

- **closest_integer(value)**: This function takes a string input representing a number and returns the closest integer. If the number is equidistant from two integers, it rounds away from zero.

- `closest_integer("14.5")` returns `15`
- `closest_integer("-14.5")` returns `-15`
