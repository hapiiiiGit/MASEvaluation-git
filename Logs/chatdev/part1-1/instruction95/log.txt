```python
'''
Main application file for the dictionary case checker.
'''

from tkinter import Tk, Label, Entry, Button, StringVar
from dict_checker import check_dict_case

class MainApp:
    def __init__(self, master):
        self.master = master
        master.title("Dictionary Case Checker")

        self.label = Label(master, text="Enter a dictionary (e.g., {'a': 'apple', 'b': 'banana'}):")
        self.label.pack()

        self.input_var = StringVar()
        self.entry = Entry(master, textvariable=self.input_var, width=50)
        self.entry.pack()

        self.check_button = Button(master, text="Check Case", command=self.get_input)
        self.check_button.pack()

        self.result_label = Label(master, text="")
        self.result_label.pack()

    def get_input(self):
        user_input = self.input_var.get()
        try:
            # Evaluate the input string as a dictionary
            user_dict = eval(user_input)
            result = check_dict_case(user_dict)
            self.display_result(result)
        except Exception as e:
            self.display_result(f"Error: {str(e)}")

    def display_result(self, result):
        if isinstance(result, bool):
            self.result_label.config(text=f"Result: {result}")
        else:
            self.result_label.config(text=result)

if __name__ == "__main__":
    root = Tk()
    app = MainApp(root)
    root.mainloop()
```

```python
'''
Module containing the function to check the case of dictionary keys.
'''

def check_dict_case(d):
    """
    Given a dictionary, return True if all keys are strings in lower 
    case or all keys are strings in upper case, else return False.
    The function should return False if the given dictionary is empty.
    """
    if not d:
        return False

    keys = d.keys()
    
    if all(isinstance(key, str) for key in keys):
        if all(key.islower() for key in keys):
            return True
        if all(key.isupper() for key in keys):
            return True
    return False
```

This structure allows for a clean separation of concerns, making the application easier to maintain and extend in the future.

'''
Module containing the function to check the case of dictionary keys.
'''
def check_dict_case(d):
    """
    Given a dictionary, return True if all keys are strings in lower 
    case or all keys are strings in upper case, else return False.
    The function should return False if the given dictionary is empty.
    """
    if not d:
        return False
    keys = d.keys()
    if all(isinstance(key, str) for key in keys):
        if all(key.islower() for key in keys):
            return True
        if all(key.isupper() for key in keys):
            return True
    return False

class MainApp:
    def __init__(self, master):
        self.master = master
        master.title("Dictionary Case Checker")
        self.label = Label(master, text="Enter a dictionary (e.g., {'a': 'apple', 'b': 'banana'}):")
        self.label.pack()
        self.input_var = StringVar()
        self.entry = Entry(master, textvariable=self.input_var, width=50)
        self.entry.pack()
        self.check_button = Button(master, text="Check Case", command=self.get_input)
        self.check_button.pack()
        self.result_label = Label(master, text="")
        self.result_label.pack()
    def get_input(self):
        user_input = self.input_var.get()
        try:
            # Safely evaluate the input string as a dictionary
            user_dict = ast.literal_eval(user_input)
            result = check_dict_case(user_dict)
            self.display_result(result)
        except Exception as e:
            self.display_result(f"Error: {str(e)}")
    def display_result(self, result):
        if isinstance(result, bool):
            self.result_label.config(text=f"Result: {result}")
        else:
            self.result_label.config(text=result)
if __name__ == "__main__":
    root = Tk()
    app = MainApp(root)
    root.mainloop()

'''
Module containing the function to check the case of dictionary keys.
'''
def check_dict_case(d):
    """
    Given a dictionary, return True if all keys are strings in lower 
    case or all keys are strings in upper case, else return False.
    The function should return False if the given dictionary is empty.
    """
    if not d:
        return False
    keys = d.keys()
    if all(isinstance(key, str) for key in keys):
        if all(key.islower() for key in keys) or all(key.isupper() for key in keys):
            return True
    return False

def check_dict_case(d):
    """
    Given a dictionary, return True if all keys are strings in lower 
    case or all keys are strings in upper case, else return False.
    The function should return False if the given dictionary is empty.
    """
    if not d:
        return False
    keys = d.keys()
    if all(isinstance(key, str) for key in keys):
        if all(key.islower() for key in keys) or all(key.isupper() for key in keys):
            return True
    return False
