'''
Main application file for the Duplicate Remover GUI.
'''

import tkinter as tk
from tkinter import messagebox
from duplicate_remover import remove_duplicates

class MainApp:
    def __init__(self, master):
        self.master = master
        master.title("Duplicate Remover")

        self.label = tk.Label(master, text="Enter a list of integers (comma-separated):")
        self.label.pack()

        self.entry = tk.Entry(master)
        self.entry.pack()

        self.remove_button = tk.Button(master, text="Remove Duplicates", command=self.remove_duplicates)
        self.remove_button.pack()

        self.result_label = tk.Label(master, text="Result:")
        self.result_label.pack()

        self.result_display = tk.Label(master, text="")
        self.result_display.pack()

    def remove_duplicates(self):
        input_text = self.entry.get()
        try:
            numbers = list(map(int, input_text.split(',')))
            result = remove_duplicates(numbers)
            self.result_display.config(text=str(result))
        except ValueError:
            messagebox.showerror("Input Error", "Please enter a valid list of integers.")

if __name__ == "__main__":
    root = tk.Tk()
    app = MainApp(root)
    root.mainloop()

```python
'''
Module to remove duplicates from a list of integers.
'''

from typing import List

def remove_duplicates(numbers: List[int]) -> List[int]:
    """ From a list of integers, remove all elements that occur more than once.
    Keep order of elements left the same as in the input.
    >>> remove_duplicates([1, 2, 3, 2, 4])
    [1, 3, 4]
    """
    count = {}
    for number in numbers:
        count[number] = count.get(number, 0) + 1

    return [number for number in numbers if count[number] == 1]
```

from typing import List
def remove_duplicates(numbers: List[int]) -> List[int]:
    """ From a list of integers, remove all elements that occur more than once.
    Keep order of elements left the same as in the input.
    >>> remove_duplicates([1, 2, 3, 2, 4])

from typing import List
def remove_duplicates(numbers: List[int]) -> List[int]:
    """ From a list of integers, remove all elements that occur more than once.
    Keep order of elements left the same as in the input.
    >>> remove_duplicates([1, 2, 3, 2, 4])
    [1, 3, 4]
    """
    count = {}
    for number in numbers:
        count[number] = count.get(number, 0) + 1
    return [number for number in numbers if count[number] == 1]

'''
Module to remove duplicates from a list of integers.
'''
from typing import List
def remove_duplicates(numbers: List[int]) -> List[int]:
    """ From a list of integers, remove all elements that occur more than once.
    Keep order of elements left the same as in the input.
    >>> remove_duplicates([1, 2, 3, 2, 4])
    [1, 3, 4]
    """
    count = {}

    return [number for number in numbers if count[number] == 1]

'''

'''
from typing import List
def remove_duplicates(numbers: List[int]) -> List[int]:
    """ From a list of integers, remove all elements that occur more than once.
    Keep order of elements left the same as in the input.
    >>> remove_duplicates([1, 2, 3, 2, 4])
    [1, 3, 4]
    """
    count = {}
    for number in numbers:
        count[number] = count.get(number, 0) + 1
    return [number for number in numbers if count[number] == 1]

import tkinter as tk

        self.result_label.pack()
        self.result_display = tk.Label(master, text="")
        self.result_display.pack()
    def remove_duplicates(self):

        try:
            numbers = list(map(int, input_text.split(',')))
            result = remove_duplicates(numbers)
            self.result_display.config(text=str(result))
        except ValueError:
            messagebox.showerror("Input Error", "Please enter a valid list of integers.")
if __name__ == "__main__":

from typing import List
def remove_duplicates(numbers: List[int]) -> List[int]:
    """ From a list of integers, remove all elements that occur more than once.
    Keep order of elements left the same as in the input.
    >>> remove_duplicates([1, 2, 3, 2, 4])
    [1, 3, 4]
    """
    count = {}
    for number in numbers:
        count[number] = count.get(number, 0) + 1
    return [number for number in numbers if count[number] == 1]
