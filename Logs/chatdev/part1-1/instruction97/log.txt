Main application file for the unit digit multiplication application.
'''

import tkinter as tk
from tkinter import messagebox
from multiply import multiply

class MainApp:
    '''
    Main application class that creates the GUI for user interaction.
    '''
    def __init__(self, master):
        self.master = master
        master.title("Unit Digit Multiplier")

        self.label_a = tk.Label(master, text="Enter first integer:")
        self.label_a.pack()

        self.entry_a = tk.Entry(master)
        self.entry_a.pack()

        self.label_b = tk.Label(master, text="Enter second integer:")
        self.label_b.pack()

        self.entry_b = tk.Entry(master)
        self.entry_b.pack()

        self.multiply_button = tk.Button(master, text="Multiply Unit Digits", command=self.multiply_digits)
        self.multiply_button.pack()

        self.result_label = tk.Label(master, text="")
        self.result_label.pack()

    def multiply_digits(self):
        '''
        Method to retrieve input, call the multiply function, and display the result.
        '''
        try:
            a = int(self.entry_a.get())
            b = int(self.entry_b.get())
            result = multiply(a, b)
            self.result_label.config(text=f"Result: {result}")
        except ValueError:
            messagebox.showerror("Input Error", "Please enter valid integers.")

if __name__ == "__main__":
    root = tk.Tk()
    app = MainApp(root)
    root.mainloop()

        self.master = master

Main application file for the unit digit multiplication application.
'''
import tkinter as tk
from tkinter import messagebox
from multiply import multiply
class MainApp:
    '''
    Main application class that creates the GUI for user interaction.
    '''
    def __init__(self, master):
        self.master = master
        master.title("Unit Digit Multiplier")
        self.label_a = tk.Label(master, text="Enter first integer:")
        self.label_a.pack()
        self.entry_a = tk.Entry(master)
        self.entry_a.pack()
        self.label_b = tk.Label(master, text="Enter second integer:")
        self.label_b.pack()
        self.entry_b = tk.Entry(master)
        self.entry_b.pack()
        self.multiply_button = tk.Button(master, text="Multiply Unit Digits", command=self.multiply_digits)
        self.multiply_button.pack()
        self.result_label = tk.Label(master, text="")
        self.result_label.pack()
    def multiply_digits(self):
        '''
        Method to retrieve input, call the multiply function, and display the result.
        '''
        try:
            a = int(self.entry_a.get())
            b = int(self.entry_b.get())
            result = multiply(a, b)
            self.result_label.config(text=f"Result: {result}")
        except ValueError:
            messagebox.showerror("Input Error", "Please enter valid integers.")
if __name__ == "__main__":
    root = tk.Tk()
    app = MainApp(root)
    root.mainloop()

'''
Module containing the function to multiply the unit digits of two integers.
'''
def multiply(a, b):
    '''
    Function that takes two integers and returns the product of their unit digits.
    This function handles negative integers by taking their absolute values.
    Examples:
    multiply(148, 412) should return 16.
    multiply(19, 28) should return 72.
    multiply(2020, 1851) should return 0.
    multiply(14, -15) should return 20.
    multiply(-14, -15) should return 20.  # Added example for negative integers
    '''
    unit_digit_a = abs(a) % 10  # Get unit digit of first number
    unit_digit_b = abs(b) % 10  # Get unit digit of second number
    return unit_digit_a * unit_digit_b  # Return the product of unit digits
