import tkinter as tk
from fibonacci import fib

class FibonacciApp:
    def __init__(self, root):
        self.root = root
        self.root.title("Fibonacci Number Calculator")

        self.label = tk.Label(root, text="Enter a number:")
        self.label.pack()

        self.entry = tk.Entry(root)
        self.entry.pack()

        self.result_label = tk.Label(root, text="")
        self.result_label.pack()

        self.calculate_button = tk.Button(root, text="Calculate", command=self.calculate_fibonacci)
        self.calculate_button.pack()

    def calculate_fibonacci(self):
        try:
            n = int(self.entry.get())
            result = fib(n)
            self.result_label.config(text=f"The {n}-th Fibonacci number is: {result}")
        except ValueError:
            self.result_label.config(text="Please enter a valid integer.")

if __name__ == "__main__":
    root = tk.Tk()
    app = FibonacciApp(root)
    root.mainloop()
```

def fib(n: int):
    """Return n-th Fibonacci number."""
    if n <= 0:
        return 0
    elif n == 1:
        return 1
    else:
        return fib(n-1) + fib(n-2)

import tkinter as tk
from fibonacci import fib
class FibonacciApp:
    def __init__(self, root):
        """Initialize the main application window."""
        self.root = root
        self.root.title("Fibonacci Number Calculator")

            self.result_label.config(text=str(e))
if __name__ == "__main__":

def fib(n: int):
    """Return n-th Fibonacci number."""
    if n < 0:
        raise ValueError("Input value must be a non-negative integer.")
    if n == 0:
        return 0
    elif n == 1:
        return 1
    else:
        # Iterative approach to calculate Fibonacci number
        a, b = 0, 1
        for _ in range(2, n + 1):
            a, b = b, a + b
        return b

   ```bash
   sudo apt-get install python3-tk
   ```

   ```bash
   python main.py
   ```
