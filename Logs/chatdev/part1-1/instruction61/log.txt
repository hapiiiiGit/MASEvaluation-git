Main application file for the Bracketing Checker.

from tkinter import Tk, Label, Entry, Button, StringVar
from bracketing import correct_bracketing

class BracketingApp:
    def __init__(self, master):
        self.master = master
        master.title("Bracketing Checker")

        self.label = Label(master, text="Enter a string of brackets:")
        self.label.pack()

        self.bracket_input = StringVar()
        self.entry = Entry(master, textvariable=self.bracket_input)
        self.entry.pack()

        self.check_button = Button(master, text="Check Brackets", command=self.check_brackets)
        self.check_button.pack()

        self.result_label = Label(master, text="")
        self.result_label.pack()

    def check_brackets(self):
        brackets = self.bracket_input.get()
        result = correct_bracketing(brackets)
        self.result_label.config(text="Correctly Balanced: " + str(result))

if __name__ == "__main__":
    root = Tk()
    app = BracketingApp(root)
    root.mainloop()
```

'''
Module containing the function to check correct bracketing.
'''

def correct_bracketing(brackets: str) -> bool:
    """ 
    brackets is a string of "(" and ")". 
    return True if every opening bracket has a corresponding closing bracket.

    >>> correct_bracketing("(")
    False
    >>> correct_bracketing("()")
    True
    >>> correct_bracketing("(()())")
    True
    >>> correct_bracketing(")(()")
    False
    """
    balance = 0
    for char in brackets:
        if char == '(':
            balance += 1
        elif char == ')':
            balance -= 1
        if balance < 0:
            return False
    return balance == 0
```

def correct_bracketing(brackets: str) -> bool:
    """ 
    brackets is a string of "(" and ")". 
    return True if every opening bracket has a corresponding closing bracket.
    >>> correct_bracketing("(")
    False
    >>> correct_bracketing("()")
    True
    >>> correct_bracketing("(()())")
    True
    >>> correct_bracketing(")(()")
    False
    """
    balance = 0
    for char in brackets:
        if char == '(':
            balance += 1
        elif char == ')':
            balance -= 1
        if balance < 0:
            return False
    return balance == 0

def correct_bracketing(brackets: str) -> bool:
    """ 
    brackets is a string of "(" and ")". 
    return True if every opening bracket has a corresponding closing bracket.

    False

def correct_bracketing(brackets: str) -> bool:
    """ 
    brackets is a string of "(" and ")". 
    return True if every opening bracket has a corresponding closing bracket.
    >>> correct_bracketing("(")
    False
    >>> correct_bracketing("()")
    True
    >>> correct_bracketing("(()())")
    True
    >>> correct_bracketing(")(()")
    False
    """
    balance = 0
    for char in brackets:
        if char == '(':
            balance += 1
        elif char == ')':
            balance -= 1
        if balance < 0:
            return False
    return balance == 0

def correct_bracketing(brackets: str) -> bool:
    """ 
    brackets is a string of "(" and ")". 
    return True if every opening bracket has a corresponding closing bracket.
    >>> correct_bracketing("(")
    False
    >>> correct_bracketing("()")
    True
    >>> correct_bracketing("(()())")
    True
    >>> correct_bracketing(")(()")
    False
    """
    balance = 0
    for char in brackets:
        if char == '(':
            balance += 1
        elif char == ')':
            balance -= 1
        if balance < 0:
            return False
    return balance == 0

def correct_bracketing(brackets: str) -> bool:
    """ 
    brackets is a string of "(" and ")". 
    return True if every opening bracket has a corresponding closing bracket.
    >>> correct_bracketing("(")
    False
    >>> correct_bracketing("()")
    True
    >>> correct_bracketing("(()())")
    True
    >>> correct_bracketing(")(()")
    False
    """

    for char in brackets:

            balance += 1
        elif char == ')':
            balance -= 1
        if balance < 0:
            return False
    return balance == 0

'''
Module containing the function to check correct bracketing.
'''
def correct_bracketing(brackets: str) -> bool:
    """ 
    brackets is a string of "(" and ")". 
    return True if every opening bracket has a corresponding closing bracket.
    >>> correct_bracketing("(")
    False
    >>> correct_bracketing("()")
    True
    >>> correct_bracketing("(()())")
    True
    >>> correct_bracketing(")(()")
    False
    """
    balance = 0
    for char in brackets:
        if char == '(':
            balance += 1
        elif char == ')':
            balance -= 1
        if balance < 0:
            return False
    return balance == 0
