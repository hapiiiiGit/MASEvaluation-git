'''
Module containing the function to check if one number is the sum of the other two.
'''

def any_int(x, y, z):
    '''
    Create a function that takes 3 numbers.
    Returns true if one of the numbers is equal to the sum of the other two, and all numbers are integers.
    Returns false in any other cases.
    
    Examples:
    any_int(5, 2, 7) ➞ True
    any_int(3, 2, 2) ➞ False
    any_int(3, -2, 1) ➞ True
    any_int(3.6, -2.2, 2) ➞ False
    '''
    if all(isinstance(i, int) for i in (x, y, z)):
        return x == y + z or y == x + z or z == x + y
    return False
```

#### gui.py

```python
'''
Module to create the graphical user interface for the application.
'''

import tkinter as tk
from tkinter import messagebox
from calculator import any_int

class App:
    '''
    Class to define the GUI application.
    '''

    def __init__(self):
        self.window = tk.Tk()
        self.window.title("Any Int Checker")
        self.create_widgets()

    def create_widgets(self):
        '''
        Create and place the widgets in the window.
        '''
        self.label_x = tk.Label(self.window, text="Enter first number (x):")
        self.label_x.pack()

        self.entry_x = tk.Entry(self.window)
        self.entry_x.pack()

        self.label_y = tk.Label(self.window, text="Enter second number (y):")
        self.label_y.pack()

        self.entry_y = tk.Entry(self.window)
        self.entry_y.pack()

        self.label_z = tk.Label(self.window, text="Enter third number (z):")
        self.label_z.pack()

        self.entry_z = tk.Entry(self.window)
        self.entry_z.pack()

        self.check_button = tk.Button(self.window, text="Check", command=self.check_numbers)
        self.check_button.pack()

    def check_numbers(self):
        '''
        Retrieve input values, check them using any_int, and display the result.
        '''
        try:
            x = int(self.entry_x.get())
            y = int(self.entry_y.get())
            z = int(self.entry_z.get())
            result = any_int(x, y, z)
            messagebox.showinfo("Result", f"Result: {result}")
        except ValueError:
            messagebox.showerror("Input Error", "Please enter valid integers.")

    def run(self):
        '''
        Start the Tkinter main loop.
        '''
        self.window.mainloop()
```

def any_int(x, y, z):
    '''
    Create a function that takes 3 numbers.
    Returns true if one of the numbers is equal to the sum of the other two, and all numbers are integers.
    Returns false in any other cases.
    Examples:
    any_int(5, 2, 7) ➞ True
    any_int(3, 2, 2) ➞ False
    any_int(3, -2, 1) ➞ True
    any_int(3.6, -2.2, 2) ➞ False

        return x == y + z or y == x + z or z == x + y

'''
Module containing the function to check if one number is the sum of the other two.
'''
def any_int(x, y, z):
    '''
    Create a function that takes 3 numbers.
    Returns true if one of the numbers is equal to the sum of the other two, and all numbers are integers.
    Returns false in any other cases.
    Examples:
    any_int(5, 2, 7) ➞ True
    any_int(3, 2, 2) ➞ False
    any_int(3, -2, 1) ➞ True
    any_int(3.6, -2.2, 2) ➞ False
    '''
    if all(isinstance(i, int) for i in (x, y, z)):
        return x == y + z or y == x + z or z == x + y
    return False

def any_int(x, y, z):
    '''
    Create a function that takes 3 numbers.
    Returns true if one of the numbers is equal to the sum of the other two, and all numbers are integers.
    Returns false in any other cases.
    Examples:
    any_int(5, 2, 7) ➞ True
    any_int(3, 2, 2) ➞ False
    any_int(3, -2, 1) ➞ True
    any_int(3.6, -2.2, 2) ➞ False

'''
Module containing the function to check if one number is the sum of the other two.
'''
def any_int(x, y, z):
    '''
    Create a function that takes 3 numbers.
    Returns true if one of the numbers is equal to the sum of the other two, and all numbers are integers.
    Returns false in any other cases.
    Examples:
    any_int(5, 2, 7) ➞ True
    any_int(3, 2, 2) ➞ False
    any_int(3, -2, 1) ➞ True
    any_int(3.6, -2.2, 2) ➞ False
    '''
    if all(isinstance(i, int) for i in (x, y, z)):
        return x == y + z or y == x + z or z == x + y
    return False
