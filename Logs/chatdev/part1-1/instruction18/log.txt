def count_overlapping_substrings(string, substring):
    if not string or not substring:
        return 0

    count = start = 0
    while True:
        start = string.find(substring, start) + 1
        if start > 0:
            count += 1
        else:
            break

    return count
```

        else:

def count_overlapping_substrings(string: str, substring: str) -> int:
    """ Count how many times a substring appears in a string, including overlaps. """
    if not string or not substring:
        return 0
    count = start = 0
    while True:
        start = string.find(substring, start)
        if start != -1:  # Check if the substring was found
            count += 1
            start += 1  # Move to the next character for overlapping
        else:
            break
    return count

def count_overlapping_substrings(string: str, substring: str) -> int:
    """ Count how many times a substring appears in a string, including overlaps. """
    if not string or not substring:
        return 0
    count = start = 0
    while True:
        start = string.find(substring, start)
        if start != -1:  # Check if the substring was found
            count += 1
            start += 1  # Move to the next character for overlapping
        else:
            break
    return count

def count_overlapping_substrings(string: str, substring: str) -> int:
    """ Count how many times a substring appears in a string, including overlaps. """
    if not string or not substring:
        return 0
    count = start = 0
    while True:
        start = string.find(substring, start)
        if start != -1:  # Check if the substring was found
            count += 1
            start += 1  # Move to the next character for overlapping
        else:
            break
    return count
