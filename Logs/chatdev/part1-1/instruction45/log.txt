```python
'''
This module contains the TriangleAreaCalculator class which provides a method to calculate the area of a triangle.
'''

class TriangleAreaCalculator:
    def triangle_area(self, a, h):
        """
        Given length of a side and height, return area for a triangle.
        >>> triangle_area(5, 3)
        7.5
        """
        return 0.5 * a * h
```

```python
'''
This is the main application file that creates the GUI for calculating the area of a triangle.
'''

import tkinter as tk
from tkinter import messagebox
from triangle_calculator import TriangleAreaCalculator

class App:
    def __init__(self, root):
        self.root = root
        self.root.title("Triangle Area Calculator")

        # Create input fields
        self.label_a = tk.Label(root, text="Length of side (a):")
        self.label_a.pack()
        self.entry_a = tk.Entry(root)
        self.entry_a.pack()

        self.label_h = tk.Label(root, text="Height (h):")
        self.label_h.pack()
        self.entry_h = tk.Entry(root)
        self.entry_h.pack()

        # Create calculate button
        self.calculate_button = tk.Button(root, text="Calculate Area", command=self.calculate_area)
        self.calculate_button.pack()

        # Create result label
        self.result_label = tk.Label(root, text="")
        self.result_label.pack()

    def calculate_area(self):
        try:
            a = float(self.entry_a.get())
            h = float(self.entry_h.get())
            calculator = TriangleAreaCalculator()
            area = calculator.triangle_area(a, h)
            self.result_label.config(text=f"Area of Triangle: {area}")
        except ValueError:
            messagebox.showerror("Input Error", "Please enter valid numbers for a and h.")

if __name__ == "__main__":
    root = tk.Tk()
    app = App(root)
    root.mainloop()
```

import tkinter as tk
from tkinter import messagebox
from triangle_calculator import TriangleAreaCalculator
class App:
    def __init__(self, root):
        self.root = root
        self.root.title("Triangle Area Calculator")
        # Create input fields
        self.label_a = tk.Label(root, text="Length of side (a):")

        # Create calculate button
        self.calculate_button = tk.Button(root, text="Calculate Area", command=self.calculate_area)
        self.calculate_button.pack()
        # Create result label
        self.result_label = tk.Label(root, text="")
