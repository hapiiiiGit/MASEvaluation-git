```python
'''
This module contains the BracketingChecker class which checks for correct bracketing.
'''

class BracketingChecker:
    def correct_bracketing(self, brackets: str) -> bool:
        """
        Check if every opening bracket has a corresponding closing bracket.
        
        :param brackets: A string of "<" and ">".
        :return: True if every opening bracket has a corresponding closing bracket, False otherwise.
        """
        balance = 0
        for char in brackets:
            if char == '<':
                balance += 1
            elif char == '>':
                balance -= 1
            if balance < 0:
                return False
        return balance == 0
```

'''
This module contains the Application class which creates the GUI for the Bracketing Checker.
'''

from tkinter import Frame, Label, Entry, Button, StringVar
from bracketing_checker import BracketingChecker

class Application(Frame):
    def __init__(self, master):
        '''
        Initializes the main window and its components.
        '''
        super().__init__(master)
        self.master = master
        self.master.title("Bracketing Checker")
        self.pack()

        self.label = Label(self, text="Enter brackets:")
        self.label.pack()

        self.bracket_input = StringVar()
        self.entry = Entry(self, textvariable=self.bracket_input)
        self.entry.pack()

        self.check_button = Button(self, text="Check", command=self.check_brackets)
        self.check_button.pack()

        self.result_label = Label(self, text="")
        self.result_label.pack()

        self.checker = BracketingChecker()

    def check_brackets(self):
        '''
        Retrieves input from the user, checks the brackets, and displays the result.
        '''
        brackets = self.bracket_input.get()
        result = self.checker.correct_bracketing(brackets)
        if result:
            self.result_label.config(text="Correct Bracketing!", fg="green")
        else:
            self.result_label.config(text="Incorrect Bracketing!", fg="red")

'''
This module contains the BracketingChecker class which checks for correct bracketing.
'''
class BracketingChecker:
    def correct_bracketing(self, brackets: str) -> bool:
        """
        Check if every opening bracket has a corresponding closing bracket.
        :param brackets: A string of "<" and ">".
        :return: True if every opening bracket has a corresponding closing bracket, False otherwise.
        """
        balance = 0
        for char in brackets:
            if char == '<':
                balance += 1
            elif char == '>':
                balance -= 1
            if balance < 0:
                return False
        return balance == 0

'''
This module contains the BracketingChecker class which checks for correct bracketing.
'''
class BracketingChecker:
    def correct_bracketing(self, brackets: str) -> bool:
        """
        Check if every opening bracket has a corresponding closing bracket.
        :param brackets: A string of "<" and ">".
        :return: True if every opening bracket has a corresponding closing bracket, False otherwise.
        """
        balance = 0
        for char in brackets:
            if char == '<':
                balance += 1
            elif char == '>':
                balance -= 1
            if balance < 0:
                return False
        return balance == 0

'''
This module contains the BracketingChecker class which checks for correct bracketing.
'''
class BracketingChecker:
    def correct_bracketing(self, brackets: str) -> bool:
        """
        Check if every opening bracket has a corresponding closing bracket.
        :param brackets: A string of "<" and ">".
        :return: True if every opening bracket has a corresponding closing bracket, False otherwise.
        """
        balance = 0
        for char in brackets:
            if char == '<':
                balance += 1
            elif char == '>':
                balance -= 1
            if balance < 0:
                return False
        return balance == 0

'''
This module contains the Application class which creates the GUI for the Bracketing Checker.
'''
from tkinter import Frame, Label, Entry, Button, StringVar
from bracketing_checker import BracketingChecker
class Application(Frame):
    def __init__(self, master):
        '''
        Initializes the main window and its components.
        '''
        super().__init__(master)
        self.master = master
        self.master.title("Bracketing Checker")
        self.pack()
        self.label = Label(self, text="Enter brackets:")
        self.label.pack()
        self.bracket_input = StringVar()
        self.entry = Entry(self, textvariable=self.bracket_input)
        self.entry.pack()
        self.check_button = Button(self, text="Check", command=self.check_brackets)
        self.check_button.pack()
        self.result_label = Label(self, text="")
        self.result_label.pack()
        self.checker = BracketingChecker()
    def check_brackets(self):
        '''
        Retrieves input from the user, checks the brackets, and displays the result.
        '''
        brackets = self.bracket_input.get()
        result = self.checker.correct_bracketing(brackets)
        if result:
            self.result_label.config(text="Correct Bracketing!", fg="green")
        else:
            self.result_label.config(text="Incorrect Bracketing!", fg="red")

'''
This module contains the BracketingChecker class which checks for correct bracketing.
'''
class BracketingChecker:
    def correct_bracketing(self, brackets: str) -> bool:
        """
        Check if every opening bracket has a corresponding closing bracket.
        :param brackets: A string of "<" and ">".
        :return: True if every opening bracket has a corresponding closing bracket, False otherwise.
        """
        balance = 0
        for char in brackets:
            if char == '<':
                balance += 1
            elif char == '>':
                balance -= 1
            if balance < 0:
                return False
        return balance == 0

from tkinter import Frame, Label, Entry, Button, StringVar
