```python
'''
This module contains the Fib4Calculator class which computes the n-th element of the Fib4 sequence.
'''

class Fib4Calculator:
    @staticmethod
    def fib4(n: int) -> int:
        if n == 0:
            return 0
        elif n == 1:
            return 0
        elif n == 2:
            return 2
        elif n == 3:
            return 0
        
        fib4_sequence = [0, 0, 2, 0]  # Initial values for fib4(0), fib4(1), fib4(2), fib4(3)
        
        for i in range(4, n + 1):
            next_value = fib4_sequence[i - 1] + fib4_sequence[i - 2] + fib4_sequence[i - 3] + fib4_sequence[i - 4]
            fib4_sequence.append(next_value)
        
        return fib4_sequence[n]
```

This module contains the Fib4App class which provides a GUI for the Fib4 calculator.

import tkinter as tk
from tkinter import messagebox
from fib4_calculator import Fib4Calculator

class Fib4App:
    def __init__(self, master):
        self.master = master
        master.title("Fib4 Calculator")

        self.label = tk.Label(master, text="Enter n:")
        self.label.pack()

        self.entry = tk.Entry(master)
        self.entry.pack()

        self.calculate_button = tk.Button(master, text="Calculate", command=self.calculate_fib4)
        self.calculate_button.pack()

        self.result_label = tk.Label(master, text="")
        self.result_label.pack()

    def calculate_fib4(self):
        try:
            n = int(self.entry.get())
            result = Fib4Calculator.fib4(n)
            self.result_label.config(text=f"fib4({n}) = {result}")
        except ValueError:
            messagebox.showerror("Input Error", "Please enter a valid integer.")

'''
This module contains the Fib4Calculator class which computes the n-th element of the Fib4 sequence.
'''
class Fib4Calculator:
    @staticmethod
    def fib4(n: int) -> int:
        if n == 0:
            return 0
        elif n == 1:
            return 0
        elif n == 2:
            return 2
        elif n == 3:
            return 0
        fib4_sequence = [0, 0, 2, 0]  # Initial values for fib4(0), fib4(1), fib4(2), fib4(3)
        for i in range(4, n + 1):
            next_value = fib4_sequence[i - 1] + fib4_sequence[i - 2] + fib4_sequence[i - 3] + fib4_sequence[i - 4]
            fib4_sequence.append(next_value)
        return fib4_sequence[n]

'''
This module contains the Fib4Calculator class which computes the n-th element of the Fib4 sequence.
'''
class Fib4Calculator:
    @staticmethod
    def fib4(n: int) -> int:
        if n == 0:
            return 0
        elif n == 1:
            return 0
        elif n == 2:
            return 2
        elif n == 3:
            return 0
        fib4_sequence = [0, 0, 2, 0]  # Initial values for fib4(0), fib4(1), fib4(2), fib4(3)
        for i in range(4, n + 1):
            next_value = fib4_sequence[i - 1] + fib4_sequence[i - 2] + fib4_sequence[i - 3] + fib4_sequence[i - 4]
            fib4_sequence.append(next_value)
        return fib4_sequence[n]

```python
'''
This module contains the Fib4Calculator class which computes the n-th element of the Fib4 sequence.
'''
class Fib4Calculator:
    @staticmethod
    def fib4(n: int) -> int:
        if n == 0:
            return 0
        elif n == 1:
            return 0
        elif n == 2:
            return 2
        elif n == 3:
            return 0
        fib4_values = [0] * (n + 1)
        fib4_values[0] = 0
        fib4_values[1] = 0
        fib4_values[2] = 2
        fib4_values[3] = 0
        for i in range(4, n + 1):
            fib4_values[i] = fib4_values[i - 1] + fib4_values[i - 2] + fib4_values[i - 3] + fib4_values[i - 4]
        return fib4_values[n]

'''
This module contains the Fib4App class which provides a GUI for the Fib4 calculator.
'''
import tkinter as tk
from tkinter import messagebox
from fib4_calculator import Fib4Calculator
class Fib4App:
    def __init__(self, master):
        self.master = master
        master.title("Fib4 Calculator")
        self.label = tk.Label(master, text="Enter n:")
        self.label.pack()
        self.entry = tk.Entry(master)
        self.entry.pack()
        self.calculate_button = tk.Button(master, text="Calculate", command=self.calculate_fib4)
        self.calculate_button.pack()
        self.result_label = tk.Label(master, text="")
        self.result_label.pack()
    def calculate_fib4(self):
        try:
            n = int(self.entry.get())
            result = Fib4Calculator.fib4(n)
            self.result_label.config(text=f"fib4({n}) = {result}")
        except ValueError:
            messagebox.showerror("Input Error", "Please enter a valid integer.")

```python
'''
This module contains the Fib4Calculator class which computes the n-th element of the Fib4 sequence.
'''
class Fib4Calculator:
    @staticmethod
    def fib4(n: int) -> int:
        if n == 0:
            return 0
        elif n == 1:
            return 0
        elif n == 2:
            return 2
        elif n == 3:
            return 0
        fib4_values = [0] * (n + 1)
        fib4_values[0] = 0
        fib4_values[1] = 0
        fib4_values[2] = 2
        fib4_values[3] = 0
        for i in range(4, n + 1):
            fib4_values[i] = fib4_values[i - 1] + fib4_values[i - 2] + fib4_values[i - 3] + fib4_values[i - 4]
        return fib4_values[n]
