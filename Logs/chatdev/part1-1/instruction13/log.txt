'''
This module contains the logic for calculating the greatest common divisor (GCD) of two integers.
'''

class GCDCalculator:
    @staticmethod
    def greatest_common_divisor(a: int, b: int) -> int:
        """ Return a greatest common divisor of two integers a and b
        >>> GCDCalculator.greatest_common_divisor(3, 5)
        1
        >>> GCDCalculator.greatest_common_divisor(25, 15)
        5
        """
        while b:
            a, b = b, a % b
        return abs(a)

'''
This module contains the logic for calculating the greatest common divisor (GCD) of two integers.
'''
class GCDCalculator:
    @staticmethod
    def greatest_common_divisor(a: int, b: int) -> int:
        """ Return a greatest common divisor of two integers a and b
        >>> GCDCalculator.greatest_common_divisor(3, 5)
        1

        5
        """
        while b:
            a, b = b, a % b

from tkinter import messagebox
from gcd_calculator import GCDCalculator
class GCDApp:
    def __init__(self, master):
        self.master = master
        master.title("GCD Calculator")
        self.label_a = tk.Label(master, text="Enter first integer:")
        self.label_a.pack()
        self.entry_a = tk.Entry(master)
        self.entry_a.pack()
        self.label_b = tk.Label(master, text="Enter second integer:")
        self.label_b.pack()
        self.entry_b = tk.Entry(master)
        self.entry_b.pack()
        self.calculate_button = tk.Button(master, text="Calculate GCD", command=self.calculate_gcd)
        self.calculate_button.pack()
        self.result_label = tk.Label(master, text="")
        self.result_label.pack()
    def calculate_gcd(self):
        try:
            a = int(self.entry_a.get())
            b = int(self.entry_b.get())
            gcd = GCDCalculator.greatest_common_divisor(a, b)
            self.result_label.config(text=f"GCD: {gcd}")
        except ValueError:
            messagebox.showerror("Input error", "Please enter valid integers.")

'''
This module contains the logic for calculating the greatest common divisor (GCD) of two integers.
'''
class GCDCalculator:
    @staticmethod
    def greatest_common_divisor(a: int, b: int) -> int:
        """ Return a greatest common divisor of two integers a and b
        >>> GCDCalculator.greatest_common_divisor(3, 5)
        1
        >>> GCDCalculator.greatest_common_divisor(25, 15)
        5
        >>> GCDCalculator.greatest_common_divisor(-25, 15)
        5
        >>> GCDCalculator.greatest_common_divisor(0, 5)
        5
        >>> GCDCalculator.greatest_common_divisor(0, 0)
        Traceback (most recent call last):
            ...
        ValueError: GCD is not defined for 0 and 0
        """
        if a == 0 and b == 0:
            raise ValueError("GCD is not defined for 0 and 0")
        while b:
            a, b = b, a % b
        return abs(a)
