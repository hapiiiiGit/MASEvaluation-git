'''
Main application file to run the String Filter application.
'''

import tkinter as tk
from tkinter import scrolledtext
from filter import filter_by_prefix

class StringFilterApp:
    def __init__(self, root):
        self.root = root
        self.root.title("String Filter by Prefix")

        # Input for prefix
        self.prefix_label = tk.Label(root, text="Enter Prefix:")
        self.prefix_label.pack()

        self.prefix_entry = tk.Entry(root)
        self.prefix_entry.pack()

        # Button to filter
        self.filter_button = tk.Button(root, text="Filter", command=self.filter_strings)
        self.filter_button.pack()

        # Text area for displaying results
        self.result_area = scrolledtext.ScrolledText(root, width=40, height=10)
        self.result_area.pack()

    def filter_strings(self):
        # Example list of strings to filter
        strings = ['abc', 'bcd', 'cde', 'array']
        prefix = self.prefix_entry.get()
        filtered_strings = filter_by_prefix(strings, prefix)
        self.result_area.delete(1.0, tk.END)  # Clear previous results
        self.result_area.insert(tk.END, '\n'.join(filtered_strings))  # Display results

if __name__ == "__main__":
    root = tk.Tk()
    app = StringFilterApp(root)
    root.mainloop()
```

### File: filter.py

```python
'''
Module to filter a list of strings based on a given prefix.
'''

from typing import List

def filter_by_prefix(strings: List[str], prefix: str) -> List[str]:
    """ 
    Filter an input list of strings only for ones that start with a given prefix.
    
    >>> filter_by_prefix([], 'a')
    []
    >>> filter_by_prefix(['abc', 'bcd', 'cde', 'array'], 'a')
    ['abc', 'array']
    """
    return [s for s in strings if s.startswith(prefix)]
```
