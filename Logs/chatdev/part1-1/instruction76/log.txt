from tkinter import Tk, Label, Entry, Button, StringVar

'''
Module to check if a number x is a simple power of n.
'''

def is_simple_power(x, n):
    """Returns true if x is a simple power of n, otherwise false."""
    if n <= 1:
        return False if n == 1 else (x == 1)
    power = 1
    while power < x:
        power *= n
    return power == x
```

def is_simple_power(x, n):

Module to check if a number x is a simple power of n.

def is_simple_power(x, n):
    """Returns true if x is a simple power of n, otherwise false."""
    if n <= 1:
        return False if n == 1 else (x == 1)
    power = 1
    while power < x:
        power *= n
    return power == x

def is_simple_power(x, n):
    """Returns true if x is a simple power of n, otherwise false."""
    if n <= 1:
        return False if n == 1 else (x == 1)
    power = 1
    while power < x:
        power *= n
    return power == x

def is_simple_power(x, n):
    """Returns true if x is a simple power of n, otherwise false."""
    if n <= 1:
        return False if n == 1 else (x == 1)
    power = 1
    while power < x:
        power *= n
    return power == x

'''

'''
def is_simple_power(x, n):
    """Returns true if x is a simple power of n, otherwise false."""
    if n <= 1:
        return False if n == 1 else (x == 1)
    power = 1
    while power < x:
        power *= n
    return power == x
