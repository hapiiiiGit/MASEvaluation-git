'''
Module for calculating the FibFib number sequence.
'''

class FibFibCalculator:
    def __init__(self):
        self.memo = {}

    def fibfib(self, n: int) -> int:
        """Computes the n-th FibFib number."""
        if n in self.memo:
            return self.memo[n]
        if n == 0:
            return 0
        elif n == 1:
            return 0
        elif n == 2:
            return 1
        else:
            result = self.fibfib(n - 1) + self.fibfib(n - 2) + self.fibfib(n - 3)
            self.memo[n] = result
            return result
```

import tkinter as tk
from tkinter import messagebox
from fibfib_calculator import FibFibCalculator

class FibFibApp:
    def __init__(self):
        self.calculator = FibFibCalculator()
        self.window = tk.Tk()
        self.window.title("FibFib Number Sequence Calculator")

        self.label = tk.Label(self.window, text="Enter n:")
        self.label.pack()

        self.entry = tk.Entry(self.window)
        self.entry.pack()

        self.calculate_button = tk.Button(self.window, text="Calculate", command=self.calculate_fibfib)
        self.calculate_button.pack()

        self.result_label = tk.Label(self.window, text="")
        self.result_label.pack()

    def calculate_fibfib(self):
        """Retrieves user input, computes the FibFib number, and displays the result."""
        try:
            n = int(self.entry.get())
            result = self.calculator.fibfib(n)
            self.result_label.config(text=f"FibFib({n}) = {result}")
        except ValueError:
            messagebox.showerror("Input Error", "Please enter a valid integer.")

    def run(self):
        """Runs the GUI application."""
        self.window.mainloop()

'''
Module for calculating the FibFib number sequence.
'''
class FibFibCalculator:
    def __init__(self):
        self.memo = {}
    def fibfib(self, n: int) -> int:
        """Computes the n-th FibFib number."""
        if n in self.memo:
            return self.memo[n]
        if n == 0:
            return 0
        elif n == 1:
            return 0
        elif n == 2:
            return 1
        else:
            result = self.fibfib(n - 1) + self.fibfib(n - 2) + self.fibfib(n - 3)
            self.memo[n] = result
            return result

class FibFibCalculator:
    def __init__(self):
        self.memo = {}
    def fibfib(self, n: int) -> int:
        """Computes the n-th FibFib number.
        The FibFib number sequence is a sequence similar to the Fibonacci sequence that's defined as follows:
        fibfib(0) == 0
        fibfib(1) == 0
        fibfib(2) == 1
        fibfib(n) == fibfib(n-1) + fibfib(n-2) + fibfib(n-3).
        Please write a function to efficiently compute the n-th element of the fibfib number sequence.
        >>> fibfib(1)
        0
        >>> fibfib(3)
        1
        >>> fibfib(4)
        2
        >>> fibfib(5)
        4
        >>> fibfib(6)
        7
        >>> fibfib(7)
        13
        >>> fibfib(8)
        24
        """
        if n in self.memo:
            return self.memo[n]
        if n == 0:
            return 0
        elif n == 1:
            return 0
        elif n == 2:
            return 1
        else:
            result = self.fibfib(n - 1) + self.fibfib(n - 2) + self.fibfib(n - 3)
            self.memo[n] = result
            return result

'''
Module for calculating the FibFib number sequence.
'''
class FibFibCalculator:
    def __init__(self):
        self.memo = {}
    def fibfib(self, n: int) -> int:
        """Computes the n-th FibFib number.
        The FibFib number sequence is a sequence similar to the Fibonacci sequence that's defined as follows:
        fibfib(0) == 0
        fibfib(1) == 0
        fibfib(2) == 1
        fibfib(n) == fibfib(n-1) + fibfib(n-2) + fibfib(n-3).
        Please write a function to efficiently compute the n-th element of the fibfib number sequence.
        >>> fibfib(1)
        0
        >>> fibfib(3)
        1
        >>> fibfib(4)
        2
        >>> fibfib(5)
        4
        >>> fibfib(6)
        7
        >>> fibfib(7)
        13
        >>> fibfib(8)
        24
        """
        if n in self.memo:
            return self.memo[n]
        if n == 0:
            return 0
        elif n == 1:
            return 0
        elif n == 2:
            return 1
        else:
            result = self.fibfib(n - 1) + self.fibfib(n - 2) + self.fibfib(n - 3)
            self.memo[n] = result
            return result

'''
Module for calculating the FibFib number sequence.
'''
class FibFibCalculator:
    def __init__(self):
        self.memo = {}
    def fibfib(self, n: int) -> int:
        """Computes the n-th FibFib number.
        The FibFib number sequence is a sequence similar to the Fibonacci sequence that's defined as follows:
        fibfib(0) == 0
        fibfib(1) == 0
        fibfib(2) == 1
        fibfib(n) == fibfib(n-1) + fibfib(n-2) + fibfib(n-3).
        >>> fibfib(1)
        0
        >>> fibfib(3)
        1
        >>> fibfib(4)
        2
        >>> fibfib(5)
        4
        >>> fibfib(6)
        7
        >>> fibfib(7)
        13
        >>> fibfib(8)
        24
        """
        if n in self.memo:
            return self.memo[n]
        if n == 0:
            return 0
        elif n == 1:
            return 0
        elif n == 2:
            return 1
        else:
            result = self.fibfib(n - 1) + self.fibfib(n - 2) + self.fibfib(n - 3)
            self.memo[n] = result
            return result

'''
Module for calculating the FibFib number sequence.
'''
class FibFibCalculator:
    def __init__(self):
        self.memo = {}
    def fibfib(self, n: int) -> int:
        """Computes the n-th FibFib number.
        The FibFib number sequence is a sequence similar to the Fibonacci sequence that's defined as follows:
        fibfib(0) == 0
        fibfib(1) == 0
        fibfib(2) == 1
        fibfib(n) == fibfib(n-1) + fibfib(n-2) + fibfib(n-3).
        >>> fibfib(1)
        0
        >>> fibfib(3)
        1
        >>> fibfib(4)
        2
        >>> fibfib(5)
        4
        >>> fibfib(6)
        7
        >>> fibfib(7)
        13
        >>> fibfib(8)
        24
        """
        if n in self.memo:
            return self.memo[n]
        if n == 0:
            return 0
        elif n == 1:
            return 0
        elif n == 2:
            return 1
        else:
            result = self.fibfib(n - 1) + self.fibfib(n - 2) + self.fibfib(n - 3)
            self.memo[n] = result
            return result

'''
Module for calculating the FibFib number sequence.
'''
class FibFibCalculator:
    def __init__(self):
        self.memo = {}
    def fibfib(self, n: int) -> int:
        """Computes the n-th FibFib number.
        The FibFib number sequence is a sequence similar to the Fibonacci sequence that's defined as follows:
        fibfib(0) == 0
        fibfib(1) == 0
        fibfib(2) == 1
        fibfib(n) == fibfib(n-1) + fibfib(n-2) + fibfib(n-3).
        >>> fibfib(1)
        0
        >>> fibfib(3)
        1
        >>> fibfib(4)
        2
        >>> fibfib(5)
        4
        >>> fibfib(6)
        7
        >>> fibfib(7)
        13
        >>> fibfib(8)
        24
        """
        if n in self.memo:
            return self.memo[n]
        if n == 0:
            return 0
        elif n == 1:
            return 0
        elif n == 2:
            return 1
        else:
            result = self.fibfib(n - 1) + self.fibfib(n - 2) + self.fibfib(n - 3)
            self.memo[n] = result
            return result

    def __init__(self):
        self.calculator = FibFibCalculator()

        self.window.title("FibFib Number Sequence Calculator")
        self.label = tk.Label(self.window, text="Enter n:")
        self.label.pack()
        self.entry = tk.Entry(self.window)
        self.entry.pack()
        self.calculate_button = tk.Button(self.window, text="Calculate", command=self.calculate_fibfib)
        self.calculate_button.pack()
        self.result_label = tk.Label(self.window, text="")
        self.result_label.pack()
    def calculate_fibfib(self):
        """Retrieves user input, computes the FibFib number, and displays the result."""
        try:
            n = int(self.entry.get())
            result = self.calculator.fibfib(n)
            self.result_label.config(text=f"FibFib({n}) = {result}")
        except ValueError:
            messagebox.showerror("Input Error", "Please enter a valid integer.")
    def run(self):
        """Runs the GUI application."""
        self.window.mainloop()

'''
Module for calculating the FibFib number sequence.
'''
class FibFibCalculator:
    def __init__(self):
        self.memo = {}
    def fibfib(self, n: int) -> int:
        """Computes the n-th FibFib number.
        The FibFib number sequence is a sequence similar to the Fibonacci sequence that's defined as follows:
        fibfib(0) == 0
        fibfib(1) == 0
        fibfib(2) == 1
        fibfib(n) == fibfib(n-1) + fibfib(n-2) + fibfib(n-3).
        >>> fibfib(1)
        0
        >>> fibfib(3)
        1
        >>> fibfib(4)
        2
        >>> fibfib(5)
        4
        >>> fibfib(6)
        7
        >>> fibfib(7)
        13
        >>> fibfib(8)
        24
        """
        if n in self.memo:
            return self.memo[n]
        if n == 0:
            return 0
        elif n == 1:
            return 0
        elif n == 2:
            return 1
        else:
            result = self.fibfib(n - 1) + self.fibfib(n - 2) + self.fibfib(n - 3)
            self.memo[n] = result
            return result
