   - **`calculate_sum`**: Computes the sum of the digits of \( N \) and converts it to binary.
   - **`display_result`**: Displays the result in the GUI.

```python
'''
Main application file for calculating the sum of digits in binary.
'''

import tkinter as tk
from tkinter import messagebox
from utils import sum_of_digits

class SumDigitsApp:
    '''
    GUI application to calculate the sum of digits of a number in binary.
    '''

    def __init__(self, master):
        '''
        Initializes the GUI components.
        '''
        self.master = master
        master.title("Sum of Digits in Binary")

        self.label = tk.Label(master, text="Enter a positive integer (0 ≤ N ≤ 10000):")
        self.label.pack()

        self.entry = tk.Entry(master)
        self.entry.pack()

        self.calculate_button = tk.Button(master, text="Calculate", command=self.calculate_sum)
        self.calculate_button.pack()

        self.result_label = tk.Label(master, text="")
        self.result_label.pack()

    def calculate_sum(self):
        '''
        Calculates the sum of digits and displays the result in binary.
        '''
        try:
            N = int(self.entry.get())
            if N < 0 or N > 10000:
                raise ValueError("Input must be between 0 and 10000.")
            digit_sum = sum_of_digits(N)
            binary_result = bin(digit_sum)[2:]  # Convert to binary and remove '0b'
            self.display_result(binary_result)
        except ValueError as e:
            messagebox.showerror("Invalid input", str(e))

    def display_result(self, result):
        '''
        Displays the result in the GUI.
        '''
        self.result_label.config(text=f"Sum of digits in binary: {result}")

if __name__ == "__main__":
    root = tk.Tk()
    app = SumDigitsApp(root)
    root.mainloop()
```

'''
Utility functions for the Sum of Digits application.

def sum_of_digits(N):
    '''
    Returns the sum of the digits of the given integer N.
    '''
    return sum(int(digit) for digit in str(N))
```

import tkinter as tk

class SumDigitsApp:
    '''
    GUI application to calculate the sum of digits of a number in binary.
    '''
    def __init__(self, master):
        '''
        Initializes the GUI components.
        '''
        self.master = master
        master.title("Sum of Digits in Binary")
        self.label = tk.Label(master, text="Enter a positive integer (0 ≤ N ≤ 10000):")
        self.label.pack()
        self.entry = tk.Entry(master)
        self.entry.pack()
        self.calculate_button = tk.Button(master, text="Calculate", command=self.calculate_sum)
        self.calculate_button.pack()
        self.result_label = tk.Label(master, text="")
        self.result_label.pack()
    def calculate_sum(self):
        '''
        Calculates the sum of digits and displays the result in binary.
        '''
        try:
            N = int(self.entry.get())
            if N < 0 or N > 10000:
                raise ValueError("Input must be between 0 and 10000.")
            digit_sum = sum_of_digits(N)
            binary_result = bin(digit_sum)[2:]  # Convert to binary and remove '0b'
            self.display_result(binary_result)
        except ValueError as e:
            messagebox.showerror("Invalid input", str(e))
    def display_result(self, result):
        '''
        Displays the result in the GUI.
        '''
        self.result_label.config(text=f"Sum of digits in binary: {result}")

    root = tk.Tk()
    app = SumDigitsApp(root)

import tkinter as tk
from tkinter import messagebox
from utils import sum_of_digits
class SumDigitsApp:
    '''
    GUI application to calculate the sum of digits of a number in binary.
    '''
    def __init__(self, master):
        '''
        Initializes the GUI components.
        '''
        self.master = master
        master.title("Sum of Digits in Binary")
        self.label = tk.Label(master, text="Enter a positive integer (1 ≤ N ≤ 10000):")
        self.label.pack()
        self.entry = tk.Entry(master)
        self.entry.pack()
        self.calculate_button = tk.Button(master, text="Calculate", command=self.calculate_sum)
        self.calculate_button.pack()
        self.result_label = tk.Label(master, text="")
        self.result_label.pack()
    def calculate_sum(self):
        '''
        Calculates the sum of digits and displays the result in binary.
        '''
        try:
            N = int(self.entry.get())
            if N <= 0 or N > 10000:  # Ensure N is a positive integer
                raise ValueError("Input must be a positive integer between 1 and 10000.")
            digit_sum = sum_of_digits(N)
            binary_result = bin(digit_sum)[2:]  # Convert to binary and remove '0b'
            self.display_result(binary_result)
        except ValueError as e:
            messagebox.showerror("Invalid input", str(e))
    def display_result(self, result):
        '''
        Displays the result in the GUI.
        '''
        self.result_label.config(text=f"Sum of digits in binary: {result}")
if __name__ == "__main__":
    root = tk.Tk()

'''
Main application file for calculating the sum of digits in binary.
'''
import tkinter as tk
from tkinter import messagebox
from utils import sum_of_digits
class SumDigitsApp:
    '''
    GUI application to calculate the sum of digits of a number in binary.
    '''
    def __init__(self, master):
        '''
        Initializes the GUI components.
        '''
        self.master = master
        master.title("Sum of Digits in Binary")
        self.label = tk.Label(master, text="Enter a positive integer (1 ≤ N ≤ 10000):")
        self.label.pack()
        self.entry = tk.Entry(master)
        self.entry.pack()
        self.calculate_button = tk.Button(master, text="Calculate", command=self.calculate_sum)
        self.calculate_button.pack()
        self.result_label = tk.Label(master, text="")
        self.result_label.pack()
    def calculate_sum(self):
        '''
        Calculates the sum of digits and displays the result in binary.
        '''
        try:
            N = int(self.entry.get())
            if N < 1 or N > 10000:  # Ensure N is a positive integer
                raise ValueError("Input must be a positive integer between 1 and 10000.")
            digit_sum = sum_of_digits(N)
            binary_result = bin(digit_sum)[2:]  # Convert to binary and remove '0b'
            self.display_result(binary_result)
        except ValueError as e:
            messagebox.showerror("Invalid input", str(e))
    def display_result(self, result):
        '''
        Displays the result in the GUI.
        '''
        self.result_label.config(text=f"Sum of digits in binary: {result}")
if __name__ == "__main__":
    root = tk.Tk()
    app = SumDigitsApp(root)
    root.mainloop()
