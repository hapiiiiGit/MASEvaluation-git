'''
Module to count prime hexadecimal digits in a given hexadecimal string.
'''

class HexCounter:
    @staticmethod
    def count_primes(num):
        prime_digits = {'2', '3', '5', '7', 'B', 'D'}
        count = 0
        
        for digit in num:
            if digit in prime_digits:
                count += 1
                
        return count
```

import tkinter as tk
from tkinter import messagebox
from hex_counter import HexCounter

class HexApp:
    def __init__(self):
        self.window = tk.Tk()
        self.window.title("Hexadecimal Prime Counter")
        
        self.label = tk.Label(self.window, text="Enter Hexadecimal Number:")
        self.label.pack()
        
        self.entry = tk.Entry(self.window)
        self.entry.pack()
        
        self.button = tk.Button(self.window, text="Count Primes", command=self.count_hex)
        self.button.pack()
        
        self.result_label = tk.Label(self.window, text="")
        self.result_label.pack()

    def count_hex(self):
        hex_input = self.entry.get().strip()
        if hex_input == "":
            messagebox.showwarning("Input Error", "Please enter a hexadecimal number.")
            return
        
        count = HexCounter.count_primes(hex_input)
        self.result_label.config(text=f"Number of prime digits: {count}")

    def run(self):
        self.window.mainloop()

'''
Module to count prime hexadecimal digits in a given hexadecimal string.
'''
class HexCounter:
    @staticmethod
    def count_primes(num):
        prime_digits = {'2', '3', '5', '7', 'B', 'D'}
        count = 0
        for digit in num:
            if digit in prime_digits:
                count += 1
        return count

'''
Module to count prime hexadecimal digits in a given hexadecimal string.
'''
class HexCounter:
    @staticmethod
    def count_primes(num):
        prime_digits = {'2', '3', '5', '7', 'B', 'D'}
        count = 0
        for digit in num:
            if digit in prime_digits:
                count += 1
        return count
```

import tkinter as tk
from tkinter import messagebox
from hex_counter import HexCounter
class HexApp:
    def __init__(self):
        self.window = tk.Tk()
        self.window.title("Hexadecimal Prime Counter")
        self.label = tk.Label(self.window, text="Enter Hexadecimal Number:")
        self.label.pack()

        self.entry.pack()

        hex_input = self.entry.get().strip()

        count = HexCounter.count_primes(hex_input)
        self.result_label.config(text=f"Number of prime digits: {count}")
    def run(self):
        self.window.mainloop()

'''
Module to count prime hexadecimal digits in a given hexadecimal string.
'''
class HexCounter:
    @staticmethod
    def count_primes(num):
        prime_digits = {'2', '3', '5', '7', 'B', 'D'}
        count = 0
        for digit in num:
            if digit in prime_digits:
                count += 1
        return count

    def count_hex(self):
        hex_input = self.entry.get().strip()
        if hex_input == "":

            return
        count = HexCounter.count_primes(hex_input)
        self.result_label.config(text=f"Number of prime digits: {count}")
