Main file to run the concatenation application.

from tkinter import Tk, Label, Entry, Button, Text, END
from concatenator import Concatenator

class App:
    def __init__(self):
        self.window = Tk()
        self.window.title("String Concatenator")

        self.label = Label(self.window, text="Enter strings separated by commas:")
        self.label.pack()

        self.entry = Entry(self.window, width=50)
        self.entry.pack()

        self.button = Button(self.window, text="Concatenate", command=self.concatenate_strings)
        self.button.pack()

        self.result_text = Text(self.window, height=5, width=50)
        self.result_text.pack()

    def concatenate_strings(self):
        input_text = self.entry.get()
        strings = [s.strip() for s in input_text.split(',')]
        concatenator = Concatenator()
        result = concatenator.concatenate(strings)
        self.result_text.delete(1.0, END)
        self.result_text.insert(END, result)

    def run(self):
        self.window.mainloop()

if __name__ == "__main__":
    app = App()
    app.run()
```

from tkinter import Tk, Label, Entry, Button, Text, END
from concatenator import Concatenator
class App:
    def __init__(self):
        self.window = Tk()
        self.window.title("String Concatenator")
        self.label = Label(self.window, text="Enter strings separated by commas:")
        self.label.pack()
        self.entry = Entry(self.window, width=50)
        self.entry.pack()
        self.button = Button(self.window, text="Concatenate", command=self.concatenate_strings)
        self.button.pack()
        self.result_text = Text(self.window, height=5, width=50)
        self.result_text.pack()
    def concatenate_strings(self):
        input_text = self.entry.get()
        strings = [s.strip() for s in input_text.split(',')]
        concatenator = Concatenator()
        result = concatenator.concatenate(strings)
        self.result_text.delete(1.0, END)
        self.result_text.insert(END, result)
    def run(self):
        self.window.mainloop()
if __name__ == "__main__":
    app = App()
    app.run()

'''
Main file to run the concatenation application.
'''
from tkinter import Tk, Label, Entry, Button, Text, END
from concatenator import Concatenator
class App:
    def __init__(self):
        self.window = Tk()
        self.window.title("String Concatenator")
        self.label = Label(self.window, text="Enter strings separated by commas:")
        self.label.pack()
        self.entry = Entry(self.window, width=50)
        self.entry.pack()
        self.button = Button(self.window, text="Concatenate", command=self.concatenate_strings)
        self.button.pack()
        self.result_text = Text(self.window, height=5, width=50)
        self.result_text.pack()
    def concatenate_strings(self):
        input_text = self.entry.get()
        strings = [s.strip() for s in input_text.split(',')]
        concatenator = Concatenator()
        result = concatenator.concatenate(strings)
        self.result_text.delete(1.0, END)
        self.result_text.insert(END, result)
    def run(self):
        self.window.mainloop()
if __name__ == "__main__":
    app = App()
    app.run()

'''
Module to handle string concatenation logic.
'''
from typing import List
class Concatenator:
    def concatenate(self, strings: List[str]) -> str:
        """ Concatenate list of strings into a single string
        >>> concatenate([])
        ''
        >>> concatenate(['a', 'b', 'c'])
        'abc'
        """
        return ''.join(strings)
