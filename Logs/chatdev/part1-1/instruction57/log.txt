'''
This module contains the MonotonicChecker class which provides a method to check
if a list of numbers is monotonically increasing or decreasing.
'''

class MonotonicChecker:
    def is_monotonic(self, l: list) -> bool:
        """Return True if list elements are monotonically increasing or decreasing."""
        if not l:
            return True
        
        increasing = decreasing = True
        
        for i in range(1, len(l)):
            if l[i] > l[i - 1]:
                decreasing = False
            elif l[i] < l[i - 1]:
                increasing = False
        
        return increasing or decreasing
```

```python
'''
This module contains the main application for checking if a list is monotonically
increasing or decreasing using a graphical user interface.
'''

import tkinter as tk
from tkinter import messagebox
from monotonic_checker import MonotonicChecker

class App:
    def __init__(self, master):
        self.master = master
        master.title("Monotonic List Checker")

        self.label = tk.Label(master, text="Enter a list of numbers (comma separated):")
        self.label.pack()

        self.entry = tk.Entry(master)
        self.entry.pack()

        self.check_button = tk.Button(master, text="Check Monotonicity", command=self.check_monotonicity)
        self.check_button.pack()

        self.result_label = tk.Label(master, text="")
        self.result_label.pack()

    def check_monotonicity(self):
        input_text = self.entry.get()
        try:
            number_list = [float(x.strip()) for x in input_text.split(',')]
            checker = MonotonicChecker()
            result = checker.is_monotonic(number_list)
            if result:
                self.result_label.config(text="The list is monotonically increasing or decreasing.")
            else:
                self.result_label.config(text="The list is not monotonically increasing or decreasing.")
        except ValueError:
            messagebox.showerror("Input Error", "Please enter valid numbers separated by commas.")

def main():
    root = tk.Tk()
    app = App(root)
    root.mainloop()

if __name__ == "__main__":
    main()

This module contains the MonotonicChecker class which provides a method to check
if a list of numbers is monotonically increasing or decreasing.
'''
class MonotonicChecker:
    def is_monotonic(self, l: list) -> bool:
        """Return True if list elements are monotonically increasing or decreasing."""
        if not l:
            return True
        increasing = decreasing = True
        for i in range(1, len(l)):
            if l[i] > l[i - 1]:
                decreasing = False
            elif l[i] < l[i - 1]:
                increasing = False
        return increasing or decreasing

'''
This module contains the MonotonicChecker class which provides a method to check
if a list of numbers is monotonically increasing or decreasing.
'''
class MonotonicChecker:
    def is_monotonic(self, l: list) -> bool:
        """Return True if list elements are monotonically increasing or decreasing."""
        if not l:
            return True
        increasing = decreasing = True
        for i in range(1, len(l)):
            if l[i] > l[i - 1]:
                decreasing = False
            elif l[i] < l[i - 1]:
                increasing = False
        return increasing or decreasing

'''
This module contains the MonotonicChecker class which provides a method to check
if a list of numbers is monotonically increasing or decreasing.
'''
class MonotonicChecker:
    def is_monotonic(self, l: list) -> bool:
        """Return True if list elements are monotonically increasing or decreasing."""
        if not l:
            return True
        increasing = decreasing = True
        for i in range(1, len(l)):
            if l[i] > l[i - 1]:
                decreasing = False
            elif l[i] < l[i - 1]:
                increasing = False
        return increasing or decreasing

    def __init__(self, master):
