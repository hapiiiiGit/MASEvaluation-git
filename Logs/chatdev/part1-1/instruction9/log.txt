from typing import List
from collections import deque
def rolling_max(numbers: List[int]) -> List[int]:
    """ From a given list of integers, generate a list of rolling maximum element found until given moment
    in the sequence.
    >>> rolling_max([1, 2, 3, 2, 3, 4, 2])
    [1, 2, 3, 3, 3, 4, 4]
    """
    rolling_max_list = []
    window = deque()
    for i, num in enumerate(numbers):
        # Remove indices that are out of the current window
        if window and window[0] < i - len(numbers):
            window.popleft()
        # Maintain the deque in decreasing order
        while window and numbers[window[-1]] <= num:
            window.pop()
        window.append(i)
        # The maximum for the current position is at the front of the deque
        rolling_max_list.append(numbers[window[0]])
    return rolling_max_list

from collections import deque
def rolling_max(numbers: List[int]) -> List[int]:
    """ From a given list of integers, generate a list of rolling maximum element found until given moment
    in the sequence.
    >>> rolling_max([1, 2, 3, 2, 3, 4, 2])
    [1, 2, 3, 3, 3, 4, 4]
    """
    rolling_max_list = []
    window = deque()
    for i, num in enumerate(numbers):
        # Remove indices that are out of the current window
        if window and window[0] < i - len(numbers):
            window.popleft()
        # Maintain the deque in decreasing order
        while window and numbers[window[-1]] <= num:
            window.pop()
        window.append(i)
        # The maximum for the current position is at the front of the deque
        rolling_max_list.append(numbers[window[0]])
    return rolling_max_list

'''

'''
from typing import List
from collections import deque
def rolling_max(numbers: List[int]) -> List[int]:
    """ From a given list of integers, generate a list of rolling maximum element found until given moment
    in the sequence.
    >>> rolling_max([1, 2, 3, 2, 3, 4, 2])
    [1, 2, 3, 3, 3, 4, 4]
    """
    rolling_max_list = []
    window = deque()
    for i, num in enumerate(numbers):
        # Remove indices that are out of the current window
        if window and window[0] < i - (len(rolling_max_list) - 1):
            window.popleft()
        # Maintain the deque in decreasing order
        while window and numbers[window[-1]] <= num:
            window.pop()
        window.append(i)
        # The maximum for the current position is at the front of the deque
        rolling_max_list.append(numbers[window[0]])
    return rolling_max_list

'''
This file contains the implementation of the rolling_max function.
'''
from typing import List
from collections import deque
def rolling_max(numbers: List[int]) -> List[int]:
    """ From a given list of integers, generate a list of rolling maximum element found until given moment
    in the sequence.
    >>> rolling_max([1, 2, 3, 2, 3, 4, 2])
    [1, 2, 3, 3, 3, 4, 4]
    """
    rolling_max_list = []
    window = deque()
    for i, num in enumerate(numbers):
        # Remove indices that are out of the current window
        if window and window[0] < i - (len(rolling_max_list) - 1):
            window.popleft()
        # Maintain the deque in decreasing order
        while window and numbers[window[-1]] <= num:
            window.pop()
        window.append(i)
        # The maximum for the current position is at the front of the deque
        rolling_max_list.append(numbers[window[0]])
    return rolling_max_list
