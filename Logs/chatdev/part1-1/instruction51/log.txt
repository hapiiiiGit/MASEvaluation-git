'''
Module containing the function to remove vowels from a given text.
'''

def remove_vowels(text):
    """ 
    remove_vowels is a function that takes string and returns string without vowels.
    >>> remove_vowels('')
    ''
    >>> remove_vowels("abcdef\nghijklm")
    'bcdf\nghjklm'
    >>> remove_vowels('abcdef')
    'bcdf'
    >>> remove_vowels('aaaaa')
    ''
    >>> remove_vowels('aaBAA')
    'B'
    >>> remove_vowels('zbcd')
    'zbcd'
    """
    vowels = "aeiouAEIOU"
    return ''.join(char for char in text if char not in vowels)
```

### File: gui.py

```python
'''
Module defining the graphical user interface for the Vowel Remover application.
'''

import tkinter as tk
from tkinter import scrolledtext
from vowel_remover import remove_vowels

class VowelRemoverApp:
    '''
    Class to create the Vowel Remover application GUI.
    '''
    
    def __init__(self):
        self.window = tk.Tk()
        self.window.title("Vowel Remover")
        self.create_gui()

    def create_gui(self):
        '''
        Create the GUI components.
        '''
        self.label = tk.Label(self.window, text="Enter text:")
        self.label.pack()

        self.text_area = scrolledtext.ScrolledText(self.window, width=40, height=10)
        self.text_area.pack()

        self.remove_button = tk.Button(self.window, text="Remove Vowels", command=self.remove_vowels)
        self.remove_button.pack()

        self.result_label = tk.Label(self.window, text="Result:")
        self.result_label.pack()

        self.result_area = scrolledtext.ScrolledText(self.window, width=40, height=10, state='disabled')
        self.result_area.pack()

    def remove_vowels(self):
        '''
        Handle the button click to remove vowels from the input text.
        '''
        input_text = self.text_area.get("1.0", tk.END)
        result = remove_vowels(input_text)
        self.result_area.config(state='normal')
        self.result_area.delete("1.0", tk.END)
        self.result_area.insert(tk.END, result)
        self.result_area.config(state='disabled')

    def run(self):
        '''
        Run the application.
        '''
        self.window.mainloop()
```

'''
Module containing the function to remove vowels from a given text.
'''
def remove_vowels(text):
    """ 
    remove_vowels is a function that takes string and returns string without vowels.
    >>> remove_vowels('')
    ''
    >>> remove_vowels("abcdef\nghijklm")
    'bcdf\nghjklm'
    >>> remove_vowels('abcdef')
    'bcdf'
    >>> remove_vowels('aaaaa')
    ''
    >>> remove_vowels('aaBAA')
    'B'
    >>> remove_vowels('zbcd')
    'zbcd'
    """
    vowels = "aeiouAEIOU"
    return ''.join(char for char in text if char not in vowels)

'''
Module containing the function to remove vowels from a given text.
'''
def remove_vowels(text):
    """ 
    remove_vowels is a function that takes string and returns string without vowels.
    >>> remove_vowels('')
    ''
    >>> remove_vowels("abcdef\nghijklm")
    'bcdf\nghjklm'
    >>> remove_vowels('abcdef')
    'bcdf'
    >>> remove_vowels('aaaaa')
    ''
    >>> remove_vowels('aaBAA')
    'B'
    >>> remove_vowels('zbcd')
    'zbcd'
    """
    vowels = "aeiouAEIOU"
    return ''.join(char for char in text if char not in vowels)

'''
Module containing the function to remove vowels from a given text.
'''
def remove_vowels(text):
    """ 
    remove_vowels is a function that takes string and returns string without vowels.
    >>> remove_vowels('')
    ''
    >>> remove_vowels("abcdef\nghijklm")
    'bcdf\nghjklm'
    >>> remove_vowels('abcdef')
    'bcdf'
    >>> remove_vowels('aaaaa')
    ''
    >>> remove_vowels('aaBAA')
    'B'
    >>> remove_vowels('zbcd')
    'zbcd'
    """
    vowels = "aeiouAEIOU"
    return ''.join(char for char in text if char not in vowels)

    '''
    Class to create the Vowel Remover application GUI.
    '''

        self.window = tk.Tk()
        self.window.title("Vowel Remover")
        self.create_gui()
    def create_gui(self):
        '''
        Create the GUI components.
        '''
        self.label = tk.Label(self.window, text="Enter text:")
        self.label.pack()
        self.text_area = scrolledtext.ScrolledText(self.window, width=40, height=10)
        self.text_area.pack()
        self.remove_button = tk.Button(self.window, text="Remove Vowels", command=self.remove_vowels)
        self.remove_button.pack()
        self.result_label = tk.Label(self.window, text="Result:")
        self.result_label.pack()
        self.result_area = scrolledtext.ScrolledText(self.window, width=40, height=10, state='disabled')
        self.result_area.pack()
    def remove_vowels(self):
        '''
        Handle the button click to remove vowels from the input text.
        '''
        input_text = self.text_area.get("1.0", tk.END).strip()  # Strip to remove any trailing newline
        result = remove_vowels(input_text)
        self.result_area.config(state='normal')
        self.result_area.delete("1.0", tk.END)
        self.result_area.insert(tk.END, result)
        self.result_area.config(state='disabled')
    def run(self):
        '''
        Run the application.
        '''
        self.window.mainloop()

'''
Module containing the function to remove vowels from a given text.
'''
def remove_vowels(text):
    """ 
    remove_vowels is a function that takes string and returns string without vowels.
    >>> remove_vowels('')
    ''
    >>> remove_vowels("abcdef\nghijklm")
    'bcdf\nghjklm'
    >>> remove_vowels('abcdef')
    'bcdf'
    >>> remove_vowels('aaaaa')
    ''
    >>> remove_vowels('aaBAA')
    'B'
    >>> remove_vowels('zbcd')
    'zbcd'
    """
    vowels = "aeiouAEIOU"
    return ''.join(char for char in text if char not in vowels)

'''
Module containing the function to remove vowels from a given text.
'''
def remove_vowels(text):
    """ 
    remove_vowels is a function that takes string and returns string without vowels.
    >>> remove_vowels('')
    ''
    >>> remove_vowels("abcdef\nghijklm")
    'bcdf\nghjklm'
    >>> remove_vowels('abcdef')
    'bcdf'
    >>> remove_vowels('aaaaa')
    ''
    >>> remove_vowels('aaBAA')
    'B'
    >>> remove_vowels('zbcd')
    'zbcd'
    """
    vowels = "aeiouAEIOU"
    return ''.join(char for char in text if char not in vowels)

def remove_vowels(text):
    """ 
    remove_vowels is a function that takes string and returns string without vowels.
    >>> remove_vowels('')
    ''

    'B'

'''
Module containing the function to remove vowels from a given text.
'''
def remove_vowels(text):
    """ 
    remove_vowels is a function that takes string and returns string without vowels.
    >>> remove_vowels('')
    ''
    >>> remove_vowels("abcdef\nghijklm")
    'bcdf\nghjklm'
    >>> remove_vowels('abcdef')
    'bcdf'
    >>> remove_vowels('aaaaa')
    ''
    >>> remove_vowels('aaBAA')
    'B'
    >>> remove_vowels('zbcd')
    'zbcd'
    """
    vowels = "aeiouAEIOU"
    return ''.join(char for char in text if char not in vowels)

'''
Module defining the graphical user interface for the Vowel Remover application.
'''
import tkinter as tk
from tkinter import scrolledtext
from vowel_remover import remove_vowels
class VowelRemoverApp:
    '''
    Class to create the Vowel Remover application GUI.
    '''
    def __init__(self):
        self.window = tk.Tk()
        self.window.title("Vowel Remover")
        self.create_gui()
    def create_gui(self):
        '''
        Create the GUI components.
        '''
        self.label = tk.Label(self.window, text="Enter text:")
        self.label.pack()
        self.text_area = scrolledtext.ScrolledText(self.window, width=40, height=10)
        self.text_area.pack()
        self.remove_button = tk.Button(self.window, text="Remove Vowels", command=self.remove_vowels)
        self.remove_button.pack()
        self.result_label = tk.Label(self.window, text="Result:")
        self.result_label.pack()
        self.result_area = scrolledtext.ScrolledText(self.window, width=40, height=10, state='disabled')
        self.result_area.pack()
    def remove_vowels(self):
        '''
        Handle the button click to remove vowels from the input text.
        '''
        input_text = self.text_area.get("1.0", tk.END).strip()  # Strip to remove any trailing newline
        result = remove_vowels(input_text)
        self.result_area.config(state='normal')
        self.result_area.delete("1.0", tk.END)
        self.result_area.insert(tk.END, result)
        self.result_area.config(state='disabled')
    def run(self):
        '''
        Run the application.
        '''
        self.window.mainloop()
```
