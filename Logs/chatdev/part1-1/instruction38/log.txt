Main application file for the Cyclic Encoder application.

from tkinter import Tk, Label, Entry, Button, Text, Scrollbar, END
from cyclic_encoder import encode_cyclic, decode_cyclic

class CyclicEncoderApp:
    '''
    GUI application for encoding and decoding strings using cyclic encoding.
    '''

    def __init__(self, master):
        self.master = master
        master.title("Cyclic Encoder")

        self.label = Label(master, text="Enter text:")
        self.label.pack()

        self.input_text = Text(master, height=10, width=50)
        self.input_text.pack()

        self.encode_button = Button(master, text="Encode", command=self.encode_text)
        self.encode_button.pack()

        self.decode_button = Button(master, text="Decode", command=self.decode_text)
        self.decode_button.pack()

        self.result_label = Label(master, text="Result:")
        self.result_label.pack()

        self.result_text = Text(master, height=10, width=50)
        self.result_text.pack()

        self.scrollbar = Scrollbar(master, command=self.result_text.yview)
        self.scrollbar.pack(side='right', fill='y')
        self.result_text.config(yscrollcommand=self.scrollbar.set)

    def encode_text(self):
        '''
        Encodes the input text and displays the result.
        '''
        input_str = self.input_text.get("1.0", END).strip()
        encoded_str = encode_cyclic(input_str)
        self.result_text.delete("1.0", END)
        self.result_text.insert(END, encoded_str)

    def decode_text(self):
        '''
        Decodes the input text and displays the result.
        '''
        input_str = self.input_text.get("1.0", END).strip()
        decoded_str = decode_cyclic(input_str)
        self.result_text.delete("1.0", END)
        self.result_text.insert(END, decoded_str)

if __name__ == "__main__":
    root = Tk()
    app = CyclicEncoderApp(root)
    root.mainloop()

'''
Module containing functions for cyclic encoding and decoding of strings.
'''

def encode_cyclic(s: str):
    '''
    Returns encoded string by cycling groups of three characters.
    '''
    # Split string into groups of length 3.
    groups = [s[(3 * i):min((3 * i + 3), len(s))] for i in range((len(s) + 2) // 3)]
    # Cycle elements in each group unless group has fewer elements than 3.
    groups = [(group[1:] + group[0]) if len(group) == 3 else group for group in groups]
    return "".join(groups)

def decode_cyclic(s: str):
    '''
    Takes as input string encoded with encode_cyclic function. Returns decoded string.
    '''
    # Split string into groups of length 3.
    groups = [s[(3 * i):min((3 * i + 3), len(s))] for i in range((len(s) + 2) // 3)]
    # Reverse the cycling for each group unless group has fewer elements than 3.
    groups = [(group[-1] + group[:-1]) if len(group) == 3 else group for group in groups]
    return "".join(groups)
```

'''
Module containing functions for cyclic encoding and decoding of strings.
'''
def encode_cyclic(s: str):
    '''
    Returns encoded string by cycling groups of three characters.
    '''
    # Split string into groups of length 3.
    groups = [s[(3 * i):min((3 * i + 3), len(s))] for i in range((len(s) + 2) // 3)]
    # Cycle elements in each group unless group has fewer elements than 3.
    groups = [(group[1:] + group[0]) if len(group) == 3 else group for group in groups]
    return "".join(groups)
def decode_cyclic(s: str):
    '''
    Takes as input string encoded with encode_cyclic function. Returns decoded string.
    '''
    # Split string into groups of length 3.
    groups = [s[(3 * i):min((3 * i + 3), len(s))] for i in range((len(s) + 2) // 3)]
    # Reverse the cycling for each group unless group has fewer elements than 3.
    groups = [(group[-1] + group[:-1]) if len(group) == 3 else group for group in groups]
    return "".join(groups)

def encode_cyclic(s: str):
    '''
    Returns encoded string by cycling groups of three characters.
    '''
    # Split string into groups of length 3.
    groups = [s[(3 * i):min((3 * i + 3), len(s))] for i in range((len(s) + 2) // 3)]
    # Cycle elements in each group unless group has fewer elements than 3.
    groups = [(group[1:] + group[0]) if len(group) == 3 else group for group in groups]
    return "".join(groups)
def decode_cyclic(s: str):
    '''
    Takes as input string encoded with encode_cyclic function. Returns decoded string.
    '''
    # Split string into groups of length 3.
    groups = [s[(3 * i):min((3 * i + 3), len(s))] for i in range((len(s) + 2) // 3)]
    # Reverse the cycling for each group unless group has fewer elements than 3.
    groups = [(group[-1] + group[:-1]) if len(group) == 3 else group for group in groups]
    return "".join(groups)

def encode_cyclic(s: str):
    '''
    Returns encoded string by cycling groups of three characters.
    '''
    # Split string into groups of length 3.
    groups = [s[(3 * i):min((3 * i + 3), len(s))] for i in range((len(s) + 2) // 3)]
    # Cycle elements in each group unless group has fewer elements than 3.
    groups = [(group[1:] + group[0]) if len(group) == 3 else group for group in groups]
    return "".join(groups)
def decode_cyclic(s: str):
    '''
    Takes as input string encoded with encode_cyclic function. Returns decoded string.
    '''
    # Split string into groups of length 3.
    groups = [s[(3 * i):min((3 * i + 3), len(s))] for i in range((len(s) + 2) // 3)]
    # Reverse the cycling for each group unless group has fewer elements than 3.
    groups = [(group[-1] + group[:-1]) if len(group) == 3 else group for group in groups]
    return "".join(groups)

    Returns encoded string by cycling groups of three characters.

def encode_cyclic(s: str):
    '''
    Returns encoded string by cycling groups of three characters.
    '''
    # Split string into groups of length 3.
    groups = [s[(3 * i):min((3 * i + 3), len(s))] for i in range((len(s) + 2) // 3)]
    # Cycle elements in each group unless group has fewer elements than 3.
    groups = [(group[1:] + group[0]) if len(group) == 3 else group for group in groups]
    return "".join(groups)

def decode_cyclic(s: str):
    '''
    Takes as input string encoded with encode_cyclic function. Returns decoded string.
    '''
    # Split string into groups of length 3.
    groups = [s[(3 * i):min((3 * i + 3), len(s))] for i in range((len(s) + 2) // 3)]
    # Reverse the cycling for each group unless group has fewer elements than 3.
    groups = [(group[-1] + group[:-1]) if len(group) == 3 else group for group in groups]
    return "".join(groups)

'''
Module containing functions for cyclic encoding and decoding of strings.
'''
def encode_cyclic(s: str):
    '''
    Returns encoded string by cycling groups of three characters.
    '''
    # Split string into groups of length 3.
    groups = [s[(3 * i):min((3 * i + 3), len(s))] for i in range((len(s) + 2) // 3)]
    # Cycle elements in each group unless group has fewer elements than 3.
    groups = [(group[1:] + group[0]) if len(group) == 3 else group for group in groups]
    return "".join(groups)

def decode_cyclic(s: str):
    '''
    Takes as input string encoded with encode_cyclic function. Returns decoded string.
    '''
    # Split string into groups of length 3.
    groups = [s[(3 * i):min((3 * i + 3), len(s))] for i in range((len(s) + 2) // 3)]
    # Reverse the cycling for each group unless group has fewer elements than 3.
    groups = [(group[-1] + group[:-1]) if len(group) == 3 else group for group in groups]
    return "".join(groups)

'''
Module containing functions for cyclic encoding and decoding of strings.
'''
def encode_cyclic(s: str):
    '''
    Returns encoded string by cycling groups of three characters.
    '''
    # Split string into groups of length 3.
    groups = [s[(3 * i):min((3 * i + 3), len(s))] for i in range((len(s) + 2) // 3)]
    # Cycle elements in each group unless group has fewer elements than 3.
    groups = [(group[1:] + group[0]) if len(group) == 3 else group for group in groups]
    return "".join(groups)
def decode_cyclic(s: str):
    '''
    Takes as input string encoded with encode_cyclic function. Returns decoded string.
    '''
    # Split string into groups of length 3.
    groups = [s[(3 * i):min((3 * i + 3), len(s))] for i in range((len(s) + 2) // 3)]
    # Reverse the cycling for each group unless group has fewer elements than 3.
    groups = [(group[-1] + group[:-1]) if len(group) == 3 else group for group in groups]
    return "".join(groups)

        Encodes the input text and displays the result.

'''
Module containing functions for cyclic encoding and decoding of strings.
'''
def encode_cyclic(s: str):
    '''
    Returns encoded string by cycling groups of three characters.
    '''
    # Split string into groups of length 3.
    groups = [s[(3 * i):min((3 * i + 3), len(s))] for i in range((len(s) + 2) // 3)]
    # Cycle elements in each group unless group has fewer elements than 3.
    groups = [(group[1:] + group[0]) if len(group) == 3 else group for group in groups]
    return "".join(groups)
def decode_cyclic(s: str):
    '''
    Takes as input string encoded with encode_cyclic function. Returns decoded string.
    '''
    # Split string into groups of length 3.
    groups = [s[(3 * i):min((3 * i + 3), len(s))] for i in range((len(s) + 2) // 3)]
    # Reverse the cycling for each group unless group has fewer elements than 3.
    groups = [(group[-1] + group[:-1]) if len(group) == 3 else group for group in groups]
    return "".join(groups)
