'''
This module contains the function to check for pairs that sum to zero.
'''

def pairs_sum_to_zero(l):
    """
    pairs_sum_to_zero takes a list of integers as an input.
    it returns True if there are two distinct elements in the list that
    sum to zero, and False otherwise.
    >>> pairs_sum_to_zero([1, 3, 5, 0])
    False
    >>> pairs_sum_to_zero([1, 3, -2, 1])
    False
    >>> pairs_sum_to_zero([1, 2, 3, 7])
    False
    >>> pairs_sum_to_zero([2, 4, -5, 3, 5, 7])
    True
    >>> pairs_sum_to_zero([1])
    False
    """
    seen = set()
    for number in l:
        if -number in seen:
            return True
        seen.add(number)
    return False
```

import tkinter as tk
from tkinter import messagebox
from pairs import pairs_sum_to_zero

class App:
    '''
    The main application class that creates the GUI.
    '''
    
    def __init__(self):
        self.window = tk.Tk()
        self.window.title("Pairs Sum to Zero Checker")
        
        self.label = tk.Label(self.window, text="Enter a list of integers (comma-separated):")
        self.label.pack()
        
        self.entry = tk.Entry(self.window, width=50)
        self.entry.pack()
        
        self.check_button = tk.Button(self.window, text="Check Pairs", command=self.check_pairs)
        self.check_button.pack()
        
        self.result_label = tk.Label(self.window, text="")
        self.result_label.pack()
    
    def check_pairs(self):
        '''
        Check the pairs in the input list and display the result.
        '''
        input_text = self.entry.get()
        try:
            input_list = list(map(int, input_text.split(',')))
            result = pairs_sum_to_zero(input_list)
            self.result_label.config(text=f"Result: {result}")
        except ValueError:
            messagebox.showerror("Input Error", "Please enter a valid list of integers.")
    
    def run(self):
        '''
        Run the application.
        '''
        self.window.mainloop()

'''
This module contains the function to check for pairs that sum to zero.
'''
def pairs_sum_to_zero(l):
    """
    pairs_sum_to_zero takes a list of integers as an input.
    it returns True if there are two distinct elements in the list that
    sum to zero, and False otherwise.
    >>> pairs_sum_to_zero([1, 3, 5, 0])
    False
    >>> pairs_sum_to_zero([1, 3, -2, 1])
    False
    >>> pairs_sum_to_zero([1, 2, 3, 7])
    False
    >>> pairs_sum_to_zero([2, 4, -5, 3, 5, 7])
    True
    >>> pairs_sum_to_zero([1])
    False
    """
    seen = set()
    for number in l:
        if -number in seen:
            return True
        seen.add(number)
    return False

def pairs_sum_to_zero(l):
    """
    pairs_sum_to_zero takes a list of integers as an input.
    it returns True if there are two distinct elements in the list that
    sum to zero, and False otherwise.
    >>> pairs_sum_to_zero([1, 3, 5, 0])
    False
    >>> pairs_sum_to_zero([1, 3, -2, 1])
    False
    >>> pairs_sum_to_zero([1, 2, 3, 7])
    False
    >>> pairs_sum_to_zero([2, 4, -5, 3, 5, 7])
    True
    >>> pairs_sum_to_zero([1])
    False

    seen = set()
    for number in l:
        if -number in seen:
            return True
        seen.add(number)
    return False

'''
This module contains the function to check for pairs that sum to zero.
'''
def pairs_sum_to_zero(l):
    """
    pairs_sum_to_zero takes a list of integers as an input.
    it returns True if there are two distinct elements in the list that
    sum to zero, and False otherwise.
    >>> pairs_sum_to_zero([1, 3, 5, 0])
    False
    >>> pairs_sum_to_zero([1, 3, -2, 1])
    False
    >>> pairs_sum_to_zero([1, 2, 3, 7])
    False
    >>> pairs_sum_to_zero([2, 4, -5, 3, 5, 7])
    True
    >>> pairs_sum_to_zero([1])
    False
    >>> pairs_sum_to_zero([])
    False
    >>> pairs_sum_to_zero([-1, 1])
    True
    """
    seen = set()
    for number in l:
        if -number in seen:
            return True
        seen.add(number)
    return False

def pairs_sum_to_zero(l):
    """
    pairs_sum_to_zero takes a list of integers as an input.
    it returns True if there are two distinct elements in the list that
    sum to zero, and False otherwise.
    >>> pairs_sum_to_zero([1, 3, 5, 0])
    False
    >>> pairs_sum_to_zero([1, 3, -2, 1])
    False
    >>> pairs_sum_to_zero([1, 2, 3, 7])
    False

    True
    >>> pairs_sum_to_zero([1])
    False
    >>> pairs_sum_to_zero([])
    False
    >>> pairs_sum_to_zero([-1, 1])
    True

def pairs_sum_to_zero(l):
    """
    pairs_sum_to_zero takes a list of integers as an input.
    it returns True if there are two distinct elements in the list that
    sum to zero, and False otherwise.
    >>> pairs_sum_to_zero([1, 3, 5, 0])
    False
    >>> pairs_sum_to_zero([1, 3, -2, 1])
    False
    >>> pairs_sum_to_zero([1, 2, 3, 7])
    False
    >>> pairs_sum_to_zero([2, 4, -5, 3, 5, 7])
    True
    >>> pairs_sum_to_zero([1])
    False
    >>> pairs_sum_to_zero([])
    False
    >>> pairs_sum_to_zero([-1, 1])
    True
    """
    seen = set()
    for number in l:
        if -number in seen:
            return True
        seen.add(number)
    return False

'''
This module contains the function to check for pairs that sum to zero.
'''
def pairs_sum_to_zero(l):
    """
    pairs_sum_to_zero takes a list of integers as an input.
    it returns True if there are two distinct elements in the list that
    sum to zero, and False otherwise.
    >>> pairs_sum_to_zero([1, 3, 5, 0])
    False
    >>> pairs_sum_to_zero([1, 3, -2, 1])
    False
    >>> pairs_sum_to_zero([1, 2, 3, 7])
    False
    >>> pairs_sum_to_zero([2, 4, -5, 3, 5, 7])
    True
    >>> pairs_sum_to_zero([1])
    False
    >>> pairs_sum_to_zero([])
    False
    >>> pairs_sum_to_zero([-1, 1])
    True
    """
    seen = set()
    for number in l:
        if -number in seen:
            return True
        seen.add(number)
    return False

import tkinter as tk

class App:

        self.window = tk.Tk()
        self.window.title("Pairs Sum to Zero Checker")

        self.label.pack()
        self.entry = tk.Entry(self.window, width=50)
        self.entry.pack()
        self.check_button = tk.Button(self.window, text="Check Pairs", command=self.check_pairs)
        self.check_button.pack()
        self.result_label = tk.Label(self.window, text="")
        self.result_label.pack()
    def check_pairs(self):
        '''
        Check the pairs in the input list and display the result.
        '''
        input_text = self.entry.get()
        try:
            input_list = list(map(int, input_text.split(',')))
            result = pairs_sum_to_zero(input_list)
            self.result_label.config(text=f"Result: {result}")
        except ValueError:
            messagebox.showerror("Input Error", "Please enter a valid list of integers.")
    def run(self):
        '''
        Run the application.
        '''
        self.window.mainloop()

'''
This module contains the function to check for pairs that sum to zero.
'''
def pairs_sum_to_zero(l):
    """
    pairs_sum_to_zero takes a list of integers as an input.
    it returns True if there are two distinct elements in the list that
    sum to zero, and False otherwise.
    >>> pairs_sum_to_zero([1, 3, 5, 0])
    False
    >>> pairs_sum_to_zero([1, 3, -2, 1])
    False
    >>> pairs_sum_to_zero([1, 2, 3, 7])
    False
    >>> pairs_sum_to_zero([2, 4, -5, 3, 5, 7])
    True
    >>> pairs_sum_to_zero([1])
    False
    >>> pairs_sum_to_zero([])
    False
    >>> pairs_sum_to_zero([-1, 1])
    True
    """
    seen = set()
    for number in l:
        if -number in seen:
            return True
        seen.add(number)
    return False

'''
This module handles the graphical user interface for the application.

import tkinter as tk
from tkinter import messagebox
from pairs import pairs_sum_to_zero
class App:
    '''
    The main application class that creates the GUI.
    '''
    def __init__(self):
        self.window = tk.Tk()
        self.window.title("Pairs Sum to Zero Checker")
        self.label = tk.Label(self.window, text="Enter a list of integers (comma-separated):")
        self.label.pack()
        self.entry = tk.Entry(self.window, width=50)
        self.entry.pack()
        self.check_button = tk.Button(self.window, text="Check Pairs", command=self.check_pairs)
        self.check_button.pack()
        self.result_label = tk.Label(self.window, text="")
        self.result_label.pack()
    def check_pairs(self):
        '''
        Check the pairs in the input list and display the result.
        '''
        input_text = self.entry.get()
        try:
            input_list = list(map(int, input_text.split(',')))
            result = pairs_sum_to_zero(input_list)
            self.result_label.config(text=f"Result: {result}")
        except ValueError:
            messagebox.showerror("Input Error", "Please enter a valid list of integers.")
    def run(self):
        '''
        Run the application.
        '''
        self.window.mainloop()
