from tkinter import Tk, Label, Entry, Button, StringVar
from divisor import largest_divisor

class App:
    def __init__(self, master):
        self.master = master
        master.title("Largest Divisor Calculator")

        self.label = Label(master, text="Enter a number:")
        self.label.pack()

        self.number_var = StringVar()
        self.entry = Entry(master, textvariable=self.number_var)
        self.entry.pack()

        self.calculate_button = Button(master, text="Calculate", command=self.calculate)

        try:
            number = int(self.number_var.get())
            if number <= 1:
                raise ValueError("Number must be greater than 1.")
            result = largest_divisor(number)
            self.result_label.config(text=f"Largest divisor: {result}")
        except ValueError as e:
            self.result_label.config(text=f"Error: {e}")

if __name__ == "__main__":

'''

def largest_divisor(n: int) -> int:
    """ For a given number n, find the largest number that divides n evenly, smaller than n
    >>> largest_divisor(15)
    5
    """
    for i in range(n // 2, 0, -1):
        if n % i == 0:
            return i
    return 1  # In case n is 1, the largest divisor is 1

'''
Module for calculating the largest divisor of a number.
'''
def largest_divisor(n: int) -> int:
    """ For a given number n, find the largest number that divides n evenly, smaller than n
    >>> largest_divisor(15)
    5
    >>> largest_divisor(1)
    1
    >>> largest_divisor(28)
    14
    >>> largest_divisor(2)
    1
    >>> largest_divisor(100)
    50
    """
    if n <= 1:
        return 1  # In case n is 1 or less, the largest divisor is 1
    for i in range(n // 2, 0, -1):
        if n % i == 0:
            return i
    return 1  # Fallback, should not be reached

'''

from tkinter import Tk, Label, Entry, Button, StringVar
from divisor import largest_divisor
class App:
    def __init__(self, master):
        self.master = master
        master.title("Largest Divisor Calculator")
        self.label = Label(master, text="Enter a number:")
        self.label.pack()
        self.number_var = StringVar()
        self.entry = Entry(master, textvariable=self.number_var)
        self.entry.pack()
        self.calculate_button = Button(master, text="Calculate", command=self.calculate)
        self.calculate_button.pack()
        self.result_label = Label(master, text="")
        self.result_label.pack()
    def calculate(self):
        try:
            number = int(self.number_var.get())
            if number <= 1:
                raise ValueError("Number must be greater than 1.")
            result = largest_divisor(number)
            self.result_label.config(text=f"Largest divisor: {result}")
        except ValueError as e:
            self.result_label.config(text=f"Error: {e}")
if __name__ == "__main__":
    root = Tk()
    app = App(root)
    root.mainloop()

'''
Module for calculating the largest divisor of a number.
'''
def largest_divisor(n: int) -> int:
    """ For a given number n, find the largest number that divides n evenly, smaller than n
    >>> largest_divisor(15)
    5
    >>> largest_divisor(1)
    1
    >>> largest_divisor(28)
    14
    >>> largest_divisor(2)
    1
    >>> largest_divisor(100)
    50
    """
    if n <= 1:
        return 1  # In case n is 1 or less, the largest divisor is 1
    for i in range(n // 2, 0, -1):
        if n % i == 0:
            return i
    return 1  # Fallback, should not be reached
