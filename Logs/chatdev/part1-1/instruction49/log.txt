def modp(n, p):
    """Return 2^n modulo p.
    Parameters:
    n (int): The exponent.
    p (int): The modulus.
    Returns:
    int: The result of 2^n modulo p.
    """
    if p == 1:
        return 0  # Since any number mod 1 is 0
    return pow(2, n, p)
