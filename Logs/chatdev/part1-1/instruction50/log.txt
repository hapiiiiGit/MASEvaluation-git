'''
Main application file for the Shift Encoder GUI.
'''

from tkinter import Tk, Label, Entry, Button, Text, END
from shift_functions import encode_shift, decode_shift

class ShiftEncoderApp:
    def __init__(self, master):
        self.master = master
        master.title("Shift Encoder/Decoder")

        self.label = Label(master, text="Enter text:")
        self.label.pack()

        self.input_text = Entry(master, width=50)
        self.input_text.pack()

        self.encode_button = Button(master, text="Encode", command=self.encode_text)
        self.encode_button.pack()

        self.decode_button = Button(master, text="Decode", command=self.decode_text)
        self.decode_button.pack()

        self.clear_button = Button(master, text="Clear", command=self.clear_text)
        self.clear_button.pack()

        self.output_label = Label(master, text="Output:")
        self.output_label.pack()

        self.output_text = Text(master, height=5, width=50)
        self.output_text.pack()

    def encode_text(self):
        input_str = self.input_text.get()
        encoded_str = encode_shift(input_str)
        self.output_text.delete(1.0, END)
        self.output_text.insert(END, encoded_str)

    def decode_text(self):
        input_str = self.input_text.get()
        decoded_str = decode_shift(input_str)
        self.output_text.delete(1.0, END)
        self.output_text.insert(END, decoded_str)

    def clear_text(self):
        self.input_text.delete(0, END)
        self.output_text.delete(1.0, END)

if __name__ == "__main__":
    root = Tk()
    app = ShiftEncoderApp(root)
    root.mainloop()
```

'''
Contains functions for encoding and decoding strings using a shift cipher.
'''

def encode_shift(s: str) -> str:
    """Returns encoded string by shifting every character by 5 in the alphabet."""
    return "".join([chr(((ord(ch) - ord("a") + 5) % 26) + ord("a")) for ch in s])

def decode_shift(s: str) -> str:
    """Takes as input string encoded with encode_shift function. Returns decoded string."""
    return "".join([chr(((ord(ch) - ord("a") - 5) % 26) + ord("a")) for ch in s])
```

class ShiftEncoderApp:
    def __init__(self, master):
        self.master = master
        master.title("Shift Encoder/Decoder")
        self.label = Label(master, text="Enter text:")
        self.label.pack()
        self.input_text = Entry(master, width=50)
        self.input_text.pack()
        self.encode_button = Button(master, text="Encode", command=self.encode_text)
        self.encode_button.pack()
        self.decode_button = Button(master, text="Decode", command=self.decode_text)
        self.decode_button.pack()
        self.clear_button = Button(master, text="Clear", command=self.clear_text)
        self.clear_button.pack()
        self.output_label = Label(master, text="Output:")
        self.output_label.pack()
        self.output_text = Text(master, height=5, width=50)
        self.output_text.pack()
    def encode_text(self):
        input_str = self.input_text.get()
        encoded_str = encode_shift(input_str)
        self.output_text.delete(1.0, END)

        input_str = self.input_text.get()

        self.output_text.delete(1.0, END)
        self.output_text.insert(END, decoded_str)

'''
Contains functions for encoding and decoding strings using a shift cipher.
'''
def encode_shift(s: str) -> str:
    """Returns encoded string by shifting every character by 5 in the alphabet."""
    return "".join([chr(((ord(ch) - ord("a") + 5) % 26) + ord("a")) for ch in s])
def decode_shift(s: str) -> str:
    """Takes as input string encoded with encode_shift function. Returns decoded string."""
    return "".join([chr(((ord(ch) - ord("a") - 5) % 26) + ord("a")) for ch in s])

'''
Contains functions for encoding and decoding strings using a shift cipher.
'''
def encode_shift(s: str) -> str:
    """Returns encoded string by shifting every character by 5 in the alphabet."""
    return "".join([chr(((ord(ch) - ord("a") + 5) % 26) + ord("a")) for ch in s])
def decode_shift(s: str) -> str:
    """Takes as input string encoded with encode_shift function. Returns decoded string."""
    return "".join([chr(((ord(ch) - ord("a") - 5) % 26) + ord("a")) for ch in s])
