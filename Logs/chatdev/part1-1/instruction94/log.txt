'''
Utility functions for prime number operations.
'''

def is_prime(num):
    """Check if a number is prime."""
    if num <= 1:
        return False
    for i in range(2, int(num**0.5) + 1):
        if num % i == 0:
            return False
    return True

def sum_of_digits(num):
    """Calculate the sum of the digits of a number."""
    return sum(int(digit) for digit in str(num))

def largest_prime(lst):
    """Find the largest prime number in a list."""
    primes = [num for num in lst if is_prime(num)]
    return max(primes) if primes else None
```

    if num <= 1:

'''
Utility functions for prime number operations.
'''
def is_prime(num):
    """Check if a number is prime."""
    if num <= 1:
        return False
    for i in range(2, int(num**0.5) + 1):
        if num % i == 0:
            return False
    return True
def sum_of_digits(num):
    """Calculate the sum of the digits of a number."""
    return sum(int(digit) for digit in str(num))
def largest_prime(lst):
    """Find the largest prime number in a list."""
    primes = [num for num in lst if is_prime(num)]
    return max(primes) if primes else None

def is_prime(num):
    """Check if a number is prime."""
    if num <= 1:
        return False
    for i in range(2, int(num**0.5) + 1):
        if num % i == 0:
            return False
    return True
def sum_of_digits(num):
    """Calculate the sum of the digits of a number."""
    return sum(int(digit) for digit in str(num))
def largest_prime(lst):
    """Find the largest prime number in a list."""
    primes = [num for num in lst if is_prime(num)]
    if not primes:  # Check if the list of primes is empty
        return None  # Return None if no primes are found
    return max(primes)

def is_prime(num):
    """Check if a number is prime."""
    if num <= 1:
        return False
    for i in range(2, int(num**0.5) + 1):
        if num % i == 0:
            return False
    return True
def sum_of_digits(num):
    """Calculate the sum of the digits of a number."""
    return sum(int(digit) for digit in str(num))
def largest_prime(lst):
    """Find the largest prime number in a list."""
    primes = [num for num in lst if is_prime(num)]
    if not primes:  # Check if the list of primes is empty
        return None  # Return None if no primes are found
    return max(primes)

'''
Utility functions for prime number operations.
'''
def is_prime(num):
    """Check if a number is prime."""
    if num <= 1:
        return False
    for i in range(2, int(num**0.5) + 1):
        if num % i == 0:
            return False
    return True
def sum_of_digits(num):
    """Calculate the sum of the digits of a number."""
    return sum(int(digit) for digit in str(num))
def largest_prime(lst):
    """Find the largest prime number in a list."""
    primes = [num for num in lst if is_prime(num)]
    if not primes:  # Check if the list of primes is empty
        return None  # Return None if no primes are found
    return max(primes)

from tkinter import Tk, Label, Entry, Button, StringVar
from gui import create_gui

'''
Utility functions for prime number operations.
'''
def is_prime(num):
    """Check if a number is prime."""
    if num <= 1:
        return False
    for i in range(2, int(num**0.5) + 1):
        if num % i == 0:
            return False
    return True
def sum_of_digits(num):
    """Calculate the sum of the digits of a number."""
    return sum(int(digit) for digit in str(num))
def largest_prime(lst):
    """Find the largest prime number in a list."""
    primes = [num for num in lst if is_prime(num)]
    if not primes:  # Check if the list of primes is empty
        return None  # Return None if no primes are found
    return max(primes)
