'''
Main application file for the Positive Number Filter application.
'''

import tkinter as tk
from tkinter import messagebox
from filter import get_positive

class PositiveNumberFilterApp:
    def __init__(self, master):
        self.master = master
        master.title("Positive Number Filter")

        self.label = tk.Label(master, text="Enter a list of numbers (comma-separated):")
        self.label.pack()

        self.entry = tk.Entry(master, width=50)
        self.entry.pack()

        self.submit_button = tk.Button(master, text="Submit", command=self.on_submit)
        self.submit_button.pack()

        self.result_label = tk.Label(master, text="")
        self.result_label.pack()

    def on_submit(self):
        input_text = self.entry.get()
        try:
            input_list = [int(x.strip()) for x in input_text.split(',')]
            positive_numbers = get_positive(input_list)
            self.display_result(positive_numbers)
        except ValueError:
            messagebox.showerror("Input Error", "Please enter valid integers separated by commas.")

    def display_result(self, positive_numbers):
        if positive_numbers:
            self.result_label.config(text=f"Positive Numbers: {positive_numbers}")
        else:
            self.result_label.config(text="No positive numbers found.")

if __name__ == "__main__":
    root = tk.Tk()
    app = PositiveNumberFilterApp(root)
    root.mainloop()
```

```python
'''
Main application file for the Positive Number Filter application.
'''
import tkinter as tk
from tkinter import messagebox
from filter import get_positive
class PositiveNumberFilterApp:
    def __init__(self, master):
        self.master = master
        master.title("Positive Number Filter")
        self.label = tk.Label(master, text="Enter a list of numbers (comma-separated):")
        self.label.pack()
        self.entry = tk.Entry(master, width=50)
        self.entry.pack()
        self.submit_button = tk.Button(master, text="Submit", command=self.on_submit)
        self.submit_button.pack()
        self.result_label = tk.Label(master, text="")
        self.result_label.pack()
    def on_submit(self):
        input_text = self.entry.get()
        try:
            input_list = [eval(x.strip()) for x in input_text.split(',')]
            positive_numbers = get_positive(input_list)
            self.display_result(positive_numbers)
        except (ValueError, SyntaxError):
            messagebox.showerror("Input Error", "Please enter valid integers separated by commas.")
    def display_result(self, positive_numbers):
        if positive_numbers:
            self.result_label.config(text=f"Positive Numbers: {positive_numbers}")
        else:
            self.result_label.config(text="No positive numbers found.")
if __name__ == "__main__":
    root = tk.Tk()
    app = PositiveNumberFilterApp(root)
    root.mainloop()

'''
Main application file for the Positive Number Filter application.

import tkinter as tk
from tkinter import messagebox
from filter import get_positive
class PositiveNumberFilterApp:
    def __init__(self, master):
        self.master = master
        master.title("Positive Number Filter")
        self.label = tk.Label(master, text="Enter a list of numbers (comma-separated):")
        self.label.pack()
        self.entry = tk.Entry(master, width=50)
        self.entry.pack()
        self.submit_button = tk.Button(master, text="Submit", command=self.on_submit)
        self.submit_button.pack()
        self.result_label = tk.Label(master, text="")
        self.result_label.pack()
    def on_submit(self):
        input_text = self.entry.get()
        try:
            input_list = [eval(x.strip()) for x in input_text.split(',')]
            positive_numbers = get_positive(input_list)
            self.display_result(positive_numbers)
        except (ValueError, SyntaxError):
            messagebox.showerror("Input Error", "Please enter valid integers separated by commas.")
    def display_result(self, positive_numbers):
        if positive_numbers:
            self.result_label.config(text=f"Positive Numbers: {positive_numbers}")
        else:
            self.result_label.config(text="No positive numbers found.")
if __name__ == "__main__":
    root = tk.Tk()
    app = PositiveNumberFilterApp(root)
    root.mainloop()

'''
Module for filtering positive numbers from a list.
'''
def get_positive(l: list):
    """Return only positive numbers in the list.
    >>> get_positive([-1, 2, -4, 5, 6])
    [2, 5, 6]
    >>> get_positive([5, 3, -5, 2, -3, 3, 9, 0, 123, 1, -10])
    [5, 3, 2, 3, 9, 123, 1]
    >>> get_positive(['a', 2, 3.5, -1, 0])
    [2, 3.5]
    """
    return [x for x in l if isinstance(x, (int, float)) and x > 0]
