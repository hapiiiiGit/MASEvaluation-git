'''

from typing import List

def sort_numbers(numbers: str) -> str:
    """
    Input is a space-delimited string of numberals from 'zero' to 'nine'.
    Valid choices are 'zero', 'one', 'two', 'three', 'four', 'five', 'six', 'seven', 'eight' and 'nine'.
    Return the string with numbers sorted from smallest to largest
    >>> sort_numbers('three one five')
    'one three five'
    """
    num_dict = {'zero': 0, 'one': 1, 'two': 2, 'three': 3, 'four': 4, 'five': 5, 'six': 6, 'seven': 7, 'eight': 8, 'nine': 9}
    numbers_list = numbers.split()
    numbers_list.sort(key=lambda x: num_dict[x])
    return ' '.join(numbers_list)

from typing import List
def sort_numbers(numbers: str) -> str:
    """ 
    Input is a space-delimited string of numerals from 'zero' to 'nine'.
    Valid choices are 'zero', 'one', 'two', 'three', 'four', 'five', 
    'six', 'seven', 'eight' and 'nine'.
    Return the string with numbers sorted from smallest to largest.
    >>> sort_numbers('three one five')
    'one three five'
    """
    num_dict = {
        'zero': 0,
        'one': 1,
        'two': 2,
        'three': 3,
        'four': 4,
        'five': 5,
        'six': 6,
        'seven': 7,
        'eight': 8,
        'nine': 9
    }
    # Split the input string into words
    words = numbers.split()
    # Convert words to their corresponding numeric values
    numeric_values = [num_dict[word] for word in words]
    # Sort the numeric values
    numeric_values.sort()
    # Convert back to words
    sorted_words = [list(num_dict.keys())[list(num_dict.values()).index(num)] for num in numeric_values]
    # Join the sorted words into a single string
    return ' '.join(sorted_words)
import tkinter as tk
def sort_numbers_gui():
    def sort():
        input_numbers = entry.get()
        sorted_numbers = sort_numbers(input_numbers)
        result_label.config(text=f'Sorted Numbers: {sorted_numbers}')
    root = tk.Tk()
    root.title("Number Sorting App")
    input_label = tk.Label(root, text="Enter numbers (e.g., 'three one five'): ")
    input_label.pack()
    entry = tk.Entry(root, width=50)
    entry.pack()
    sort_button = tk.Button(root, text="Sort Numbers", command=sort)
    sort_button.pack()
    result_label = tk.Label(root, text="")
    result_label.pack()
    root.mainloop()
if __name__ == "__main__":
    sort_numbers_gui()

from typing import List
def sort_numbers(numbers: str) -> str:
    """ 
    Input is a space-delimited string of numerals from 'zero' to 'nine'.
    Valid choices are 'zero', 'one', 'two', 'three', 'four', 'five', 
    'six', 'seven', 'eight' and 'nine'.
    Return the string with numbers sorted from smallest to largest.
    >>> sort_numbers('three one five')
    'one three five'
    """
    num_dict = {
        'zero': 0,
        'one': 1,
        'two': 2,
        'three': 3,
        'four': 4,
        'five': 5,
        'six': 6,
        'seven': 7,
        'eight': 8,
        'nine': 9
    }
    # Create a reverse mapping from numeric values to words
    reverse_num_dict = {v: k for k, v in num_dict.items()}
    # Split the input string into words
    words = numbers.split()
    # Convert words to their corresponding numeric values
    numeric_values = [num_dict[word] for word in words]
    # Sort the numeric values
    numeric_values.sort()
    # Convert back to words using the reverse mapping
    sorted_words = [reverse_num_dict[num] for num in numeric_values]
    # Join the sorted words into a single string
    return ' '.join(sorted_words)

from typing import List
def sort_numbers(numbers: str) -> str:
    """ 
    Input is a space-delimited string of numerals from 'zero' to 'nine'.
    Valid choices are 'zero', 'one', 'two', 'three', 'four', 'five', 
    'six', 'seven', 'eight' and 'nine'.
    Return the string with numbers sorted from smallest to largest.
    >>> sort_numbers('three one five')
    'one three five'
    """
    num_dict = {
        'zero': 0,
        'one': 1,
        'two': 2,
        'three': 3,
        'four': 4,
        'five': 5,
        'six': 6,
        'seven': 7,
        'eight': 8,
        'nine': 9
    }
    # Create a reverse mapping from numeric values to words
    reverse_num_dict = {v: k for k, v in num_dict.items()}
    # Split the input string into words
    words = numbers.split()
    # Convert words to their corresponding numeric values
    numeric_values = [num_dict[word] for word in words]
    # Sort the numeric values
    numeric_values.sort()
    # Convert back to words using the reverse mapping
    sorted_words = [reverse_num_dict[num] for num in numeric_values]
    # Join the sorted words into a single string
    return ' '.join(sorted_words)
import tkinter as tk
def sort_numbers_gui():
    def sort():
        input_numbers = entry.get()
        sorted_numbers = sort_numbers(input_numbers)
        result_label.config(text=f'Sorted Numbers: {sorted_numbers}')
    root = tk.Tk()
    root.title("Number Sorting App")
    input_label = tk.Label(root, text="Enter numbers (e.g., 'three one five'): ")
    input_label.pack()
    entry = tk.Entry(root, width=50)
    entry.pack()
    sort_button = tk.Button(root, text="Sort Numbers", command=sort)
    sort_button.pack()
    result_label = tk.Label(root, text="")
    result_label.pack()
    root.mainloop()
if __name__ == "__main__":
    sort_numbers_gui()

from typing import List
def sort_numbers(numbers: str) -> str:
    """ 
    Input is a space-delimited string of numerals from 'zero' to 'nine'.
    Valid choices are 'zero', 'one', 'two', 'three', 'four', 'five', 
    'six', 'seven', 'eight' and 'nine'.
    Return the string with numbers sorted from smallest to largest.
    >>> sort_numbers('three one five')
    'one three five'
    """
    num_dict = {
        'zero': 0,
        'one': 1,
        'two': 2,
        'three': 3,
        'four': 4,
        'five': 5,
        'six': 6,
        'seven': 7,
        'eight': 8,
        'nine': 9
    }
    # Create a reverse mapping from numeric values to words
    reverse_num_dict = {v: k for k, v in num_dict.items()}
    # Split the input string into words
    words = numbers.split()
    # Convert words to their corresponding numeric values
    numeric_values = [num_dict[word] for word in words]
    # Sort the numeric values
    numeric_values.sort()
    # Convert back to words using the reverse mapping
    sorted_words = [reverse_num_dict[num] for num in numeric_values]
    # Join the sorted words into a single string
    return ' '.join(sorted_words)
import tkinter as tk
def sort_numbers_gui():
    def sort():
        input_numbers = entry.get()
        sorted_numbers = sort_numbers(input_numbers)
        result_label.config(text=f'Sorted Numbers: {sorted_numbers}')
    root = tk.Tk()
    root.title("Number Sorting App")
    input_label = tk.Label(root, text="Enter numbers (e.g., 'three one five'): ")
    input_label.pack()
    entry = tk.Entry(root, width=50)
    entry.pack()
    sort_button = tk.Button(root, text="Sort Numbers", command=sort)
    sort_button.pack()
    result_label = tk.Label(root, text="")
    result_label.pack()
    root.mainloop()
if __name__ == "__main__":
    sort_numbers_gui()
