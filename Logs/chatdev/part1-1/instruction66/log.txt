'''
Main application file for the Digit Sum Calculator.

from tkinter import Tk, Label, Entry, Button, StringVar
from digit_sum import digitSum
class DigitSumApp:
    '''
    GUI application to calculate the sum of ASCII values of uppercase characters in a string.
    '''
    def __init__(self, master):
        self.master = master
        master.title("Digit Sum Calculator")
        self.label = Label(master, text="Enter a string:")
        self.label.pack()
        self.input_string = StringVar()
        self.entry = Entry(master, textvariable=self.input_string)
        self.entry.pack()
        self.calculate_button = Button(master, text="Calculate", command=self.calculate_sum)
        self.calculate_button.pack()
        self.result_label = Label(master, text="")
        self.result_label.pack()
    def calculate_sum(self):
        '''
        Retrieves the input string, calculates the digit sum, and displays the result.
        '''
        input_text = self.input_string.get()
        if not isinstance(input_text, str):
            self.result_label.config(text="Please enter a valid string.")
            return
        result = digitSum(input_text)
        self.result_label.config(text=f"Sum of ASCII values of uppercase characters: {result}")
if __name__ == "__main__":
    root = Tk()
    app = DigitSumApp(root)
    root.mainloop()

    '''
    Calculate the sum of ASCII values of uppercase characters in the input string.
    Parameters:
    s (str): The input string.
    Returns:

    '''
    return sum(ord(char) for char in s if char.isupper())

from tkinter import Tk, Label, Entry, Button, StringVar

class DigitSumApp:

        self.master = master
        master.title("Digit Sum Calculator")
        self.label = Label(master, text="Enter a string:")

        self.entry.pack()

```python
'''
Main application file for the Digit Sum Calculator.
'''
from tkinter import Tk, Label, Entry, Button, StringVar
from digit_sum import digitSum
class DigitSumApp:
    '''
    GUI application to calculate the sum of ASCII values of uppercase characters in a string.
    '''
    def __init__(self, master):
        self.master = master
        master.title("Digit Sum Calculator")
        self.label = Label(master, text="Enter a string:")
        self.label.pack()
        self.input_string = StringVar()
        self.entry = Entry(master, textvariable=self.input_string)
        self.entry.pack()
        self.calculate_button = Button(master, text="Calculate", command=self.calculate_sum)
        self.calculate_button.pack()
        self.result_label = Label(master, text="")
        self.result_label.pack()
    def calculate_sum(self):
        '''
        Retrieves the input string, calculates the digit sum, and displays the result.
        '''
        input_text = self.input_string.get()
        if not isinstance(input_text, str):
            self.result_label.config(text="Please enter a valid string.")
            return
        result = digitSum(input_text)
        self.result_label.config(text=f"Sum of ASCII values of uppercase characters: {result}")
if __name__ == "__main__":
    root = Tk()
    app = DigitSumApp(root)
    root.mainloop()

def digitSum(s):
    '''
    Calculate the sum of ASCII values of uppercase characters in the input string.
    Parameters:
    s (str): The input string.
    Returns:
    int: The sum of ASCII values of uppercase characters.
    '''
    return sum(ord(char) for char in s if char.isupper())
