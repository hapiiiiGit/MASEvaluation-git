from tkinter import Tk, Label, Entry, Button, StringVar
from string_length import strlen

class StringLengthApp:
    def __init__(self, master):
        self.master = master
        master.title("String Length Calculator")

        self.label = Label(master, text="Enter a string:")
        self.label.pack()

        self.input_string = StringVar()
        self.entry = Entry(master, textvariable=self.input_string)
        self.entry.pack()

        self.calculate_button.pack()

    def calculate_length(self):

        length = strlen(input_str)
        self.result_label.config(text=f"Length: {length}")

if __name__ == "__main__":
    root = Tk()

'''
Main application file that initializes and runs the String Length application.
'''
from tkinter import Tk, Label, Entry, Button, StringVar
from string_length import strlen
class StringLengthApp:
    def __init__(self, master):
        self.master = master
        master.title("String Length Calculator")
        self.label = Label(master, text="Enter a string:")
        self.label.pack()
        self.input_string = StringVar()
        self.entry = Entry(master, textvariable=self.input_string)
        self.entry.pack()
        self.calculate_button = Button(master, text="Calculate Length", command=self.calculate_length)
        self.calculate_button.pack()
        self.result_label = Label(master, text="")
        self.result_label.pack()
    def calculate_length(self):
        input_str = self.input_string.get()
        length = strlen(input_str)
        self.result_label.config(text=f"Length: {length}")
if __name__ == "__main__":
    root = Tk()
    app = StringLengthApp(root)
    root.mainloop()
