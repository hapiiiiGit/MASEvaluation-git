'''
Module to calculate Fibonacci numbers and check for prime numbers.
'''

class FibonacciPrime:
    @staticmethod
    def is_prime(num):
        """Check if a number is prime."""
        if num <= 1:
            return False
        for i in range(2, int(num**0.5) + 1):
            if num % i == 0:
                return False
        return True

    @staticmethod
    def fibonacci(n):
        """Return the n-th Fibonacci number."""
        a, b = 0, 1
        for _ in range(n):
            a, b = b, a + b
        return a

    @classmethod
    def prime_fib(cls, n):
        """Return the n-th Fibonacci number that is also prime."""
        count = 0
        i = 0
        while True:
            fib_num = cls.fibonacci(i)
            if cls.is_prime(fib_num):
                count += 1
                if count == n:
                    return fib_num
            i += 1
```

class FibonacciPrime:
    @staticmethod
    def is_prime(num):
        """Check if a number is prime."""
        if num <= 1:
            return False
        for i in range(2, int(num**0.5) + 1):
            if num % i == 0:
                return False
        return True
    @staticmethod
    def fibonacci(n):
        """Return the n-th Fibonacci number."""
        a, b = 0, 1
        for _ in range(n):
            a, b = b, a + b
        return a
    @classmethod
    def prime_fib(cls, n):
        """Return the n-th Fibonacci number that is also prime."""
        count = 0
        i = 0
        while True:
            fib_num = cls.fibonacci(i)
            if cls.is_prime(fib_num):
                count += 1
                if count == n:
                    return fib_num
            i += 1

class FibonacciPrime:
    @staticmethod
    def is_prime(num):
        """Check if a number is prime."""
        if num <= 1:
            return False
        for i in range(2, int(num**0.5) + 1):
            if num % i == 0:
                return False
        return True

    @staticmethod
    def fibonacci(n):
        """Return the n-th Fibonacci number."""
        a, b = 0, 1
        for _ in range(n):
            a, b = b, a + b
        return a

    @classmethod
    def prime_fib(cls, n):
        """Return the n-th Fibonacci number that is also prime."""
        count = 0
        i = 0
        while True:
            fib_num = cls.fibonacci(i)
            if cls.is_prime(fib_num):
                count += 1
                if count == n:
                    return fib_num
            i += 1

class FibonacciPrime:
    @staticmethod
    def is_prime(num):
        """Check if a number is prime."""
        if num <= 1:
            return False
        for i in range(2, int(num**0.5) + 1):
            if num % i == 0:
                return False
        return True
    @staticmethod
    def fibonacci(n):
        """Return the n-th Fibonacci number."""
        a, b = 0, 1
        for _ in range(n):
            a, b = b, a + b
        return a
    @classmethod
    def prime_fib(cls, n):
        """Return the n-th Fibonacci number that is also prime."""
        count = 0
        i = 0
        while True:
            fib_num = cls.fibonacci(i)
            if cls.is_prime(fib_num):
                count += 1
                if count == n:
                    return fib_num
            i += 1

class FibonacciPrime:
    @staticmethod
    def is_prime(num):
        """Check if a number is prime."""
        if num <= 1:

            if num % i == 0:
                return False
        return True
    @staticmethod
    def fibonacci(n):
        """Return the n-th Fibonacci number."""
        a, b = 0, 1
        for _ in range(n):
            a, b = b, a + b
        return a

        i = 0
        while True:
            fib_num = cls.fibonacci(i)

                    return fib_num

'''
Module to calculate Fibonacci numbers and check for prime numbers.
'''
class FibonacciPrime:
    @staticmethod
    def is_prime(num):
        """Check if a number is prime."""
        if num <= 1:
            return False
        for i in range(2, int(num**0.5) + 1):
            if num % i == 0:
                return False
        return True
    @staticmethod
    def fibonacci(n):
        """Return the n-th Fibonacci number."""
        a, b = 0, 1
        for _ in range(n):
            a, b = b, a + b
        return a
    @classmethod
    def prime_fib(cls, n):
        """Return the n-th Fibonacci number that is also prime."""
        count = 0
        i = 0
        while True:
            fib_num = cls.fibonacci(i)
            if cls.is_prime(fib_num):
                count += 1
                if count == n:
                    return fib_num
            i += 1

'''
Module to calculate Fibonacci numbers and check for prime numbers.
'''
class FibonacciPrime:
    @staticmethod
    def is_prime(num):
        """Check if a number is prime."""
        if num <= 1:
            return False
        for i in range(2, int(num**0.5) + 1):
            if num % i == 0:
                return False
        return True
    @staticmethod
    def fibonacci(n):
        """Return the n-th Fibonacci number."""
        a, b = 0, 1
        for _ in range(n):
            a, b = b, a + b
        return a
    @classmethod
    def prime_fib(cls, n):
        """Return the n-th Fibonacci number that is also prime."""
        count = 0
        i = 0
        while True:
            fib_num = cls.fibonacci(i)
            if cls.is_prime(fib_num):
                count += 1
                if count == n:
                    return fib_num
            i += 1

'''
Module to create the graphical user interface for the application.
'''
import tkinter as tk
from tkinter import messagebox
from fibonacci_prime import FibonacciPrime
class App:
    def __init__(self):
        """Initialize the GUI components."""
        self.root = tk.Tk()
        self.root.title("Prime Fibonacci Finder")
        self.label = tk.Label(self.root, text="Enter n:")
        self.label.pack()
        self.entry = tk.Entry(self.root)
        self.entry.pack()
        self.button = tk.Button(self.root, text="Find Prime Fibonacci", command=self.get_prime_fib)
        self.button.pack()
        self.result_label = tk.Label(self.root, text="")
        self.result_label.pack()
    def get_prime_fib(self):
        """Retrieve the n-th prime Fibonacci number based on user input."""
        try:
            n = int(self.entry.get())
            if n < 1:
                raise ValueError("n must be a positive integer.")
            result = FibonacciPrime.prime_fib(n)
            self.result_label.config(text=f"The {n}-th prime Fibonacci number is: {result}")
        except ValueError as e:
            messagebox.showerror("Input Error", str(e))
    def run(self):
        """Run the application."""
        self.root.mainloop()
