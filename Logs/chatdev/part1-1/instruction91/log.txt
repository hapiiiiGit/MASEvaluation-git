```python
'''
Main application file for the Boredom Counter.
'''

from tkinter import Tk, Label, Entry, Button, StringVar
from boredom_counter import is_bored

class BoredomCounterApp:
    '''
    A class to create the GUI for counting boredoms.
    '''

    def __init__(self, master):
        '''
        Initializes the GUI components.
        '''
        self.master = master
        master.title("Boredom Counter")

        self.label = Label(master, text="Enter your text:")
        self.label.pack()

        self.text_input = StringVar()
        self.entry = Entry(master, textvariable=self.text_input, width=50)
        self.entry.pack()

        self.count_button = Button(master, text="Count Boredoms", command=self.count_boredoms)
        self.count_button.pack()

        self.result_label = Label(master, text="")
        self.result_label.pack()

    def count_boredoms(self):
        '''
        Retrieves input, counts boredoms, and displays the result.
        '''
        input_text = self.text_input.get()
        boredom_count = is_bored(input_text)
        self.result_label.config(text=f"Boredoms: {boredom_count}")

if __name__ == "__main__":
    root = Tk()
    app = BoredomCounterApp(root)
    root.mainloop()
```

#### boredom_counter.py

```python
'''
Module to count the number of sentences starting with "I".
'''

def is_bored(S):
    '''
    Counts the number of sentences that start with "I".
    '''
    # Split the input string into sentences based on delimiters
    sentences = [s.strip() for s in S.split('.') + S.split('?') + S.split('!')]
    # Count sentences that start with "I"
    boredom_count = sum(1 for sentence in sentences if sentence.startswith("I"))
    return boredom_count
```

from tkinter import Tk, Label, Entry, Button, StringVar
from boredom_counter import is_bored

class BoredomCounterApp:
    '''
    A class to create the GUI for counting boredoms.
    '''
    def __init__(self, master):
        '''
        Initializes the GUI components.

        self.master = master
        master.title("Boredom Counter")
        self.label = Label(master, text="Enter your text:")

        '''

        '''
        input_text = self.text_input.get()
        boredom_count = is_bored(input_text)
        self.result_label.config(text=f"Boredoms: {boredom_count}")

if __name__ == "__main__":

    root.mainloop()

'''

def is_bored(S):
    '''
    Counts the number of sentences that start with "I".
    '''
    # Split the input string into sentences based on delimiters
    sentences = [s.strip() for s in S.replace('!', '.').replace('?', '.').split('.')]
    # Count sentences that start with "I"
    boredom_count = sum(1 for sentence in sentences if sentence.startswith("I") and sentence)
    return boredom_count

```python
'''
Main application file for the Boredom Counter.
'''
from tkinter import Tk, Label, Entry, Button, StringVar
from boredom_counter import is_bored
class BoredomCounterApp:
    '''
    A class to create the GUI for counting boredoms.
    '''
    def __init__(self, master):
        '''
        Initializes the GUI components.
        '''
        self.master = master
        master.title("Boredom Counter")
        self.label = Label(master, text="Enter your text:")
        self.label.pack()
        self.text_input = StringVar()
        self.entry = Entry(master, textvariable=self.text_input, width=50)
        self.entry.pack()
        self.count_button = Button(master, text="Count Boredoms", command=self.count_boredoms)
        self.count_button.pack()
        self.result_label = Label(master, text="")
        self.result_label.pack()
    def count_boredoms(self):
        '''
        Retrieves input, counts boredoms, and displays the result.
        '''
        input_text = self.text_input.get()
        boredom_count = is_bored(input_text)
        self.result_label.config(text=f"Boredoms: {boredom_count}")
if __name__ == "__main__":
    root = Tk()
    app = BoredomCounterApp(root)
    root.mainloop()

'''
Module to count the number of sentences starting with "I".
'''
import re
def is_bored(S):
    '''
    Counts the number of sentences that start with "I".
    '''
    # Split the input string into sentences based on delimiters
    sentences = re.split(r'[.!?]', S)
    # Count sentences that start with "I"
    boredom_count = sum(1 for sentence in sentences if sentence.strip().lower().startswith("i"))
    return boredom_count

'''

from tkinter import Tk, Label, Entry, Button, StringVar
from boredom_counter import is_bored
class BoredomCounterApp:
    '''
    A class to create the GUI for counting boredoms.
    '''
    def __init__(self, master):
        '''
        Initializes the GUI components.
        '''
        self.master = master
        master.title("Boredom Counter")
        self.label = Label(master, text="Enter your text:")
        self.label.pack()
        self.text_input = StringVar()
        self.entry = Entry(master, textvariable=self.text_input, width=50)
        self.entry.pack()
        self.count_button = Button(master, text="Count Boredoms", command=self.count_boredoms)
        self.count_button.pack()
        self.result_label = Label(master, text="")
        self.result_label.pack()
    def count_boredoms(self):
        '''
        Retrieves input, counts boredoms, and displays the result.
        '''
        input_text = self.text_input.get()
        boredom_count = is_bored(input_text)
        self.result_label.config(text=f"Boredoms: {boredom_count}")
if __name__ == "__main__":
    root = Tk()
    app = BoredomCounterApp(root)
    root.mainloop()

'''
Module to count the number of sentences starting with "I".
'''
import re
def is_bored(S):
    '''
    Counts the number of sentences that start with "I".
    '''
    # Split the input string into sentences based on delimiters
    sentences = re.split(r'[.!?]', S)
    # Count sentences that start with "I"
    boredom_count = sum(1 for sentence in sentences if sentence.strip().lower().startswith("i"))
    return boredom_count
