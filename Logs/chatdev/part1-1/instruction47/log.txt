'''
Module to calculate the median of a list of numbers.
'''

class MedianCalculator:
    def calculate_median(self, l: list):
        """Return median of elements in the list l.
        >>> median([3, 1, 2, 4, 5])
        3
        >>> median([-10, 4, 6, 1000, 10, 20])
        15.0
        """
        if not l:
            raise ValueError("List is empty")
        
        l.sort()
        n = len(l)
        mid = n // 2
        
        if n % 2 == 0:
            return (l[mid - 1] + l[mid]) / 2
        else:
            return l[mid]
```

    def calculate_median(self, l: list):

        >>> median([3, 1, 2, 4, 5])
        3

        15.0

        if not l:
            raise ValueError("List is empty")
        l.sort()
        n = len(l)
        mid = n // 2
        if n % 2 == 0:
            return (l[mid - 1] + l[mid]) / 2
        else:
            return l[mid]

'''
Module to calculate the median of a list of numbers.
'''
class MedianCalculator:
    def calculate_median(self, l: list):
        """Return median of elements in the list l.
        >>> median([3, 1, 2, 4, 5])
        3
        >>> median([-10, 4, 6, 1000, 10, 20])
        15.0
        """
        if not l:
            raise ValueError("List is empty")
        l.sort()
        n = len(l)
        mid = n // 2
        if n % 2 == 0:
            return (l[mid - 1] + l[mid]) / 2
        else:
            return l[mid]

from tkinter import Tk, Label, Entry, Button, StringVar
from median_calculator import MedianCalculator
class App:
    def __init__(self, master):
        self.master = master
        master.title("Median Calculator")
        self.label = Label(master, text="Enter numbers separated by commas:")
        self.label.pack()
        self.entry = Entry(master)
        self.entry.pack()
        self.result_var = StringVar()
        self.result_label = Label(master, textvariable=self.result_var)
        self.result_label.pack()
        self.calculate_button = Button(master, text="Calculate Median", command=self.calculate)
        self.calculate_button.pack()
    def calculate(self):
        input_text = self.entry.get()
        try:

        except ValueError:
            self.result_var.set("Please enter valid numbers.")
if __name__ == "__main__":

'''
Module to calculate the median of a list of numbers.
'''
class MedianCalculator:
    def calculate_median(self, l: list):
        """Return median of elements in the list l.
        >>> median([3, 1, 2, 4, 5])
        3
        >>> median([-10, 4, 6, 1000, 10, 20])
        15.0
        """
        if not l:
            raise ValueError("List is empty")
        l.sort()
        n = len(l)
        mid = n // 2
        if n % 2 == 0:
            return (l[mid - 1] + l[mid]) / 2
        else:
            return l[mid]
