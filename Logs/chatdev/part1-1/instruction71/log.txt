'''
This module contains the TriangleAreaCalculator class which handles the logic
for calculating the area of a triangle based on the lengths of its sides.
'''

import math

class TriangleAreaCalculator:
    @staticmethod
    def is_valid_triangle(a, b, c):
        '''Check if the sides form a valid triangle.'''
        return (a + b > c) and (a + c > b) and (b + c > a)

    @staticmethod
    def calculate_area(a, b, c):
        '''Calculate the area of the triangle using Heron's formula.'''
        s = (a + b + c) / 2
        area = math.sqrt(s * (s - a) * (s - b) * (s - c))
        return round(area, 2)

    def get_area(self, a, b, c):
        '''Return the area of the triangle or -1 if invalid.'''
        if self.is_valid_triangle(a, b, c):
            return self.calculate_area(a, b, c)
        else:
            return -1
```

'''
This module contains the TriangleAreaCalculator class which handles the logic
for calculating the area of a triangle based on the lengths of its sides.
'''
import math
class TriangleAreaCalculator:
    @staticmethod
    def is_valid_triangle(a, b, c):
        '''Check if the sides form a valid triangle.'''
        return (a + b > c) and (a + c > b) and (b + c > a)
    @staticmethod
    def calculate_area(a, b, c):
        '''Calculate the area of the triangle using Heron's formula.'''
        s = (a + b + c) / 2
        area = math.sqrt(s * (s - a) * (s - b) * (s - c))
        return round(area, 2)
    def get_area(self, a, b, c):
        '''Return the area of the triangle or -1 if invalid.'''
        if self.is_valid_triangle(a, b, c):
            return self.calculate_area(a, b, c)
        else:
            return -1

'''
This module contains the TriangleAreaCalculator class which handles the logic
for calculating the area of a triangle based on the lengths of its sides.
'''
import math

class TriangleAreaCalculator:
    @staticmethod
    def is_valid_triangle(a, b, c):
        '''Check if the sides form a valid triangle.'''
        return (a + b > c) and (a + c > b) and (b + c > a)

    @staticmethod
    def calculate_area(a, b, c):
        '''Calculate the area of the triangle using Heron's formula.'''
        s = (a + b + c) / 2
        area = math.sqrt(s * (s - a) * (s - b) * (s - c))
        return round(area, 2)

    def get_area(self, a, b, c):
        '''Return the area of the triangle or -1 if invalid.'''
        if self.is_valid_triangle(a, b, c):
            return self.calculate_area(a, b, c)
        else:
            return -1

import math
class TriangleAreaCalculator:
    @staticmethod
    def is_valid_triangle(a, b, c):
        '''Check if the sides form a valid triangle.'''
        return (a + b > c) and (a + c > b) and (b + c > a)
    @staticmethod
    def calculate_area(a, b, c):
        '''Calculate the area of the triangle using Heron's formula.'''
        s = (a + b + c) / 2
        area = math.sqrt(s * (s - a) * (s - b) * (s - c))
        return round(area, 2)
    def get_area(self, a, b, c):
        '''Return the area of the triangle or -1 if invalid.'''
        if self.is_valid_triangle(a, b, c):
            return self.calculate_area(a, b, c)
        else:
            return -1

import math
class TriangleAreaCalculator:

'''
This module contains the TriangleAreaCalculator class which handles the logic
for calculating the area of a triangle based on the lengths of its sides.
'''
import math
class TriangleAreaCalculator:
    @staticmethod
    def is_valid_triangle(a, b, c):
        '''Check if the sides form a valid triangle.'''
        return (a + b > c) and (a + c > b) and (b + c > a)
    @staticmethod
    def calculate_area(a, b, c):
        '''Calculate the area of the triangle using Heron's formula.'''
        s = (a + b + c) / 2
        area = math.sqrt(s * (s - a) * (s - b) * (s - c))
        return round(area, 2)
    def get_area(self, a, b, c):
        '''Return the area of the triangle or -1 if invalid.'''
        if self.is_valid_triangle(a, b, c):
            return self.calculate_area(a, b, c)
        else:
            return -1
