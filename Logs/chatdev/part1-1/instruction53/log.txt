'''
Main application for adding two numbers using a GUI.
'''

def add(x: int, y: int):
    """Add two numbers x and y.
    
    >>> add(2, 3)
    5
    >>> add(5, 7)
    12
    """
    return x + y

import tkinter as tk
from tkinter import messagebox

class App:
    """Main application class for the GUI."""
    
    def __init__(self, master):
        """Initialize the GUI components."""
        self.master = master
        master.title("Addition App")

        self.label1 = tk.Label(master, text="Enter first number:")
        self.label1.pack()

        self.entry1 = tk.Entry(master)
        self.entry1.pack()

        self.label2 = tk.Label(master, text="Enter second number:")
        self.label2.pack()

        self.entry2 = tk.Entry(master)
        self.entry2.pack()

        self.add_button = tk.Button(master, text="Add", command=self.add_numbers)
        self.add_button.pack()

        self.result_label = tk.Label(master, text="")
        self.result_label.pack()

    def add_numbers(self):
        """Retrieve input values, call add function, and display result."""
        try:
            x = int(self.entry1.get())
            y = int(self.entry2.get())
            result = add(x, y)
            self.result_label.config(text=f"Result: {result}")
        except ValueError:
            messagebox.showerror("Input Error", "Please enter valid integers.")

if __name__ == "__main__":
    root = tk.Tk()
    app = App(root)
    root.mainloop()
```

    5

import tkinter as tk
from tkinter import messagebox
class App:
    """Main application class for the GUI."""
    def __init__(self, master):
        """Initialize the GUI components."""
        self.master = master
        master.title("Addition App")
        # Create and pack the first number input
        self.label1 = tk.Label(master, text="Enter first number:")
        self.label1.pack()
        self.entry1 = tk.Entry(master)
        self.entry1.pack()
        # Create and pack the second number input

        self.label2.pack()

        self.entry2.pack()

        self.add_button.pack()
