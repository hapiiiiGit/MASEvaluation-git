'''
Main application file for the Largest Prime Factor Finder.
'''

import tkinter as tk
from tkinter import messagebox
from prime_factor import largest_prime_factor

class MainApp:
    def __init__(self, root):
        self.root = root
        self.root.title("Largest Prime Factor Finder")
        
        self.label = tk.Label(root, text="Enter a number greater than 1:")
        self.label.pack(pady=10)
        
        self.entry = tk.Entry(root)
        self.entry.pack(pady=10)
        
        self.button = tk.Button(root, text="Find Largest Prime Factor", command=self.find_largest_prime_factor)
        self.button.pack(pady=10)
        
        self.result_label = tk.Label(root, text="")
        self.result_label.pack(pady=10)

    def find_largest_prime_factor(self):
        user_input = self.entry.get()
        try:
            n = validate_input(user_input)
            result = largest_prime_factor(n)
            self.result_label.config(text=f"Largest Prime Factor: {result}")
        except ValueError as e:
            messagebox.showerror("Input Error", str(e))

def validate_input(user_input):
    '''Validate the user input to ensure it is an integer greater than 1.'''
    n = int(user_input)
    if n <= 1:
        raise ValueError("Input must be an integer greater than 1.")
    return n

if __name__ == "__main__":
    root = tk.Tk()
    app = MainApp(root)
    root.mainloop()
```

'''
Module containing the function to find the largest prime factor of a number.
'''

def largest_prime_factor(n: int):
    '''Return the largest prime factor of n. Assume n > 1 and is not a prime.'''
    # Start with the smallest prime factor
    prime_factor = 2
    
    # Divide n by 2 until it becomes odd
    while n % prime_factor == 0:
        n //= prime_factor
    
    # Check for odd factors from 3 onwards
    prime_factor = 3
    while prime_factor * prime_factor <= n:
        while n % prime_factor == 0:
            n //= prime_factor
        prime_factor += 2
    
    # If n becomes a prime number greater than 2
    if n > 2:
        return n
    return prime_factor
```

def largest_prime_factor(n: int):
    '''Return the largest prime factor of n. Assume n > 1 and is not a prime.'''
    # Start with the smallest prime factor
    prime_factor = 2
    # Divide n by 2 until it becomes odd
    while n % prime_factor == 0:
        n //= prime_factor
    # Check for odd factors from 3 onwards
    prime_factor = 3
    while prime_factor * prime_factor <= n:
        while n % prime_factor == 0:
            n //= prime_factor
        prime_factor += 2
    # If n becomes a prime number greater than 2
    if n > 2:
        return n
    return prime_factor

def largest_prime_factor(n: int):
    '''Return the largest prime factor of n. Assume n > 1 and is not a prime.'''
    # Start with the smallest prime factor
    prime_factor = 2
    # Divide n by 2 until it becomes odd
    while n % prime_factor == 0:
        n //= prime_factor
    # Check for odd factors from 3 onwards
    prime_factor = 3
    while prime_factor * prime_factor <= n:
        while n % prime_factor == 0:
            n //= prime_factor
        prime_factor += 2
    # If n becomes a prime number greater than 2
    if n > 2:
        return n
    return prime_factor

'''
Main application file for the Largest Prime Factor Finder.

import tkinter as tk
from tkinter import messagebox
from prime_factor import largest_prime_factor
class MainApp:
    def __init__(self, root):
        self.root = root
        self.root.title("Largest Prime Factor Finder")
        self.label = tk.Label(root, text="Enter a number greater than 1:")
        self.label.pack(pady=10)
        self.entry = tk.Entry(root)
        self.entry.pack(pady=10)
        self.button = tk.Button(root, text="Find Largest Prime Factor", command=self.find_largest_prime_factor)
        self.button.pack(pady=10)
        self.result_label = tk.Label(root, text="")
        self.result_label.pack(pady=10)
    def find_largest_prime_factor(self):
        user_input = self.entry.get()
        try:
            n = validate_input(user_input)
            result = largest_prime_factor(n)
            self.result_label.config(text=f"Largest Prime Factor: {result}")
        except ValueError as e:
            messagebox.showerror("Input Error", str(e))
def validate_input(user_input):
    '''Validate the user input to ensure it is an integer greater than 1.'''
    n = int(user_input)
    if n <= 1:
        raise ValueError("Input must be an integer greater than 1.")
    return n
if __name__ == "__main__":
    root = tk.Tk()
    app = MainApp(root)
    root.mainloop()

'''
Module containing the function to find the largest prime factor of a number.
'''
def largest_prime_factor(n: int):
    '''Return the largest prime factor of n. Assume n > 1 and is not a prime.'''
    # Start with the smallest prime factor
    prime_factor = 2
    # Divide n by 2 until it becomes odd
    while n % prime_factor == 0:
        n //= prime_factor
    # Check for odd factors from 3 onwards
    prime_factor = 3
    while prime_factor * prime_factor <= n:
        while n % prime_factor == 0:
            n //= prime_factor
        prime_factor += 2
    # If n becomes a prime number greater than 2, return the last prime factor found
    if n > 2:
        return n
    return prime_factor  # This now correctly returns the last prime factor found

'''
Module containing the function to find the largest prime factor of a number.

def largest_prime_factor(n: int):
    '''Return the largest prime factor of n. Assume n > 1 and is not a prime.'''
    # Start with the smallest prime factor
    prime_factor = 2
    # Divide n by 2 until it becomes odd
    while n % prime_factor == 0:
        n //= prime_factor
    # Check for odd factors from 3 onwards
    prime_factor = 3
    while prime_factor * prime_factor <= n:
        while n % prime_factor == 0:
            n //= prime_factor
        prime_factor += 2
    # If n becomes a prime number greater than 2, return the last prime factor found
    if n > 2:
        return n
    return prime_factor  # This now correctly returns the last prime factor found
