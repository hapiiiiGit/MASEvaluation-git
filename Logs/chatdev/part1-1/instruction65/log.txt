'''
Circular Shift Application
This application allows users to perform a circular shift on the digits of an integer.
'''

import tkinter as tk
from tkinter import messagebox

class CircularShiftApp:
    def __init__(self, master):
        self.master = master
        master.title("Circular Shift Application")

        self.label = tk.Label(master, text="Enter an integer:")
        self.label.pack()

        self.input_entry = tk.Entry(master)
        self.input_entry.pack()

        self.shift_label = tk.Label(master, text="Enter shift value:")
        self.shift_label.pack()

        self.shift_entry = tk.Entry(master)
        self.shift_entry.pack()

        self.shift_button = tk.Button(master, text="Shift Digits", command=self.shift_digits)
        self.shift_button.pack()

        self.result_label = tk.Label(master, text="Result:")
        self.result_label.pack()

        self.result_output = tk.Label(master, text="")
        self.result_output.pack()

    def circular_shift(self, x, shift):
        """Circular shift the digits of the integer x, shift the digits right by shift
        and return the result as a string.
        If shift > number of digits, return digits reversed.
        """
        str_x = str(x)
        num_digits = len(str_x)

        if shift >= num_digits:
            return str_x[::-1]  # Return reversed digits
        else:
            shift = shift % num_digits  # Normalize shift
            return str_x[-shift:] + str_x[:-shift]  # Perform circular shift

    def shift_digits(self):
        """Handle the button click event to perform the shift and update the output."""
        try:
            x = int(self.input_entry.get())
            shift = int(self.shift_entry.get())
            result = self.circular_shift(x, shift)
            self.result_output.config(text=result)
        except ValueError:
            messagebox.showerror("Input Error", "Please enter valid integers.")

def main():
    root = tk.Tk()
    app = CircularShiftApp(root)

'''
Circular Shift Application
This application allows users to perform a circular shift on the digits of an integer.
'''
import tkinter as tk
from tkinter import messagebox

class CircularShiftApp:
    def __init__(self, master):
        self.master = master
        master.title("Circular Shift Application")
        
        self.label = tk.Label(master, text="Enter an integer:")
        self.label.pack()
        
        self.input_entry = tk.Entry(master)
        self.input_entry.pack()
        
        self.shift_label = tk.Label(master, text="Enter shift value:")
        self.shift_label.pack()
        
        self.shift_entry = tk.Entry(master)
        self.shift_entry.pack()
        
        self.shift_button = tk.Button(master, text="Shift Digits", command=self.shift_digits)
        self.shift_button.pack()
        
        self.result_label = tk.Label(master, text="Result:")
        self.result_label.pack()
        
        self.result_output = tk.Label(master, text="")
        self.result_output.pack()

    def circular_shift(self, x, shift):
        """Circular shift the digits of the integer x, shift the digits right by shift
        and return the result as a string.
        If shift > number of digits, return digits reversed.
        """
        str_x = str(x)
        num_digits = len(str_x)
        
        if shift >= num_digits:
            return str_x[::-1]  # Return reversed digits
        else:
            shift = shift % num_digits  # Normalize shift
            return str_x[-shift:] + str_x[:-shift]  # Perform circular shift

    def shift_digits(self):
        """Handle the button click event to perform the shift and update the output."""
        try:
            x = int(self.input_entry.get())
            shift = int(self.shift_entry.get())
            result = self.circular_shift(x, shift)
            self.result_output.config(text=result)
        except ValueError:
            messagebox.showerror("Input Error", "Please enter valid integers.")

def main():
    root = tk.Tk()
    app = CircularShiftApp(root)
    root.mainloop()

if __name__ == "__main__":
    main()

'''
Circular Shift Application
This application allows users to perform a circular shift on the digits of an integer.
'''
import tkinter as tk
from tkinter import messagebox
class CircularShiftApp:
    def __init__(self, master):
        self.master = master
        master.title("Circular Shift Application")
        self.label = tk.Label(master, text="Enter an integer:")
        self.label.pack()
        self.input_entry = tk.Entry(master)
        self.input_entry.pack()
        self.shift_label = tk.Label(master, text="Enter shift value:")
        self.shift_label.pack()
        self.shift_entry = tk.Entry(master)
        self.shift_entry.pack()
        self.shift_button = tk.Button(master, text="Shift Digits", command=self.shift_digits)
        self.shift_button.pack()
        self.result_label = tk.Label(master, text="Result:")
        self.result_label.pack()
        self.result_output = tk.Label(master, text="")
        self.result_output.pack()
    def circular_shift(self, x, shift):
        """Circular shift the digits of the integer x, shift the digits right by shift
        and return the result as a string.
        If shift > number of digits, return digits reversed.
        """
        str_x = str(x)
        num_digits = len(str_x)
        if shift >= num_digits:
            return str_x[::-1]  # Return reversed digits
        else:
            shift = shift % num_digits  # Normalize shift
            return str_x[-shift:] + str_x[:-shift]  # Perform circular shift
    def shift_digits(self):
        """Handle the button click event to perform the shift and update the output."""
        try:
            x = int(self.input_entry.get())
            shift = int(self.shift_entry.get())
            if shift < 0:
                messagebox.showerror("Input Error", "Shift value cannot be negative.")
                return
            result = self.circular_shift(x, shift)
            self.result_output.config(text=result)
        except ValueError:
            messagebox.showerror("Input Error", "Please enter valid integers.")
def main():
    root = tk.Tk()
    app = CircularShiftApp(root)
    root.mainloop()
if __name__ == "__main__":
    main()

'''
Circular Shift Application
This application allows users to perform a circular shift on the digits of an integer.
'''
import tkinter as tk
from tkinter import messagebox
class CircularShiftApp:
    def __init__(self, master):
        self.master = master
        master.title("Circular Shift Application")
        self.label = tk.Label(master, text="Enter an integer:")
        self.label.pack()
        self.input_entry = tk.Entry(master)
        self.input_entry.pack()
        self.shift_label = tk.Label(master, text="Enter shift value:")
        self.shift_label.pack()
        self.shift_entry = tk.Entry(master)
        self.shift_entry.pack()
        self.shift_button = tk.Button(master, text="Shift Digits", command=self.shift_digits)
        self.shift_button.pack()
        self.result_label = tk.Label(master, text="Result:")
        self.result_label.pack()
        self.result_output = tk.Label(master, text="")
        self.result_output.pack()
    def circular_shift(self, x, shift):
        """Circular shift the digits of the integer x, shift the digits right by shift
        and return the result as a string.
        If shift > number of digits, return digits reversed.
        """
        str_x = str(x)
        num_digits = len(str_x)
        if shift >= num_digits:
            return str_x[::-1]  # Return reversed digits
        else:
            shift = shift % num_digits  # Normalize shift
            return str_x[-shift:] + str_x[:-shift]  # Perform circular shift
    def shift_digits(self):
        """Handle the button click event to perform the shift and update the output."""
        try:
            x = int(self.input_entry.get())
            shift = int(self.shift_entry.get())
            if shift < 0:
                messagebox.showerror("Input Error", "Shift value cannot be negative.")
                return
            result = self.circular_shift(x, shift)
            self.result_output.config(text=result)
        except ValueError:
            messagebox.showerror("Input Error", "Please enter valid integers.")
def main():
    root = tk.Tk()
    app = CircularShiftApp(root)
    root.mainloop()
if __name__ == "__main__":
    main()
