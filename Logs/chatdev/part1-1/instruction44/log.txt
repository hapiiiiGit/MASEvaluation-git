```python
'''
BaseConverter class for converting numbers from decimal to a specified base.
'''

class BaseConverter:
    @staticmethod
    def change_base(x: int, base: int) -> str:
        """Change numerical base of input number x to base.
        return string representation after the conversion.
        base numbers are less than 10.
        >>> change_base(8, 3)
        '22'
        >>> change_base(8, 2)
        '1000'
        >>> change_base(7, 2)
        '111'
        """
        if base < 2 or base >= 10:
            raise ValueError("Base must be between 2 and 9.")
        
        if x == 0:
            return '0'
        
        digits = []
        while x:
            digits.append(str(x % base))
            x //= base
        
        return ''.join(digits[::-1])
```

'''
BaseConverter class for converting numbers from decimal to a specified base.
'''
class BaseConverter:
    @staticmethod
    def change_base(x: int, base: int) -> str:
        """Change numerical base of input number x to base.
        return string representation after the conversion.
        base numbers are less than 10.
        >>> change_base(8, 3)
        '22'
        >>> change_base(8, 2)
        '1000'
        >>> change_base(7, 2)
        '111'
        """
        if base < 2 or base >= 10:
            raise ValueError("Base must be between 2 and 9.")
        if x == 0:
            return '0'
        digits = []
        while x:
            digits.append(str(x % base))
            x //= base
        return ''.join(digits[::-1])

class BaseConverter:
    @staticmethod
    def change_base(x: int, base: int) -> str:
        """Change numerical base of input number x to base.
        return string representation after the conversion.
        base numbers are less than 10.
        >>> change_base(8, 3)
        '22'
        >>> change_base(8, 2)
        '1000'
        >>> change_base(7, 2)
        '111'
        """
        if base < 2 or base >= 10:
            raise ValueError("Base must be between 2 and 9.")
        if x == 0:
            return '0'
        digits = []
        while x:
            digits.append(str(x % base))
            x //= base
        return ''.join(reversed(digits))  # Fixed the order of digits

'''
BaseConverter class for converting numbers from decimal to a specified base.
'''
class BaseConverter:
    @staticmethod
    def change_base(x: int, base: int) -> str:
        """Change numerical base of input number x to base.
        return string representation after the conversion.
        base numbers are less than 10.
        >>> change_base(8, 3)
        '22'
        >>> change_base(8, 2)
        '1000'
        >>> change_base(7, 2)
        '111'
        >>> change_base(-8, 2)
        ValueError: Input number must be non-negative.
        """
        if base < 2 or base >= 10:
            raise ValueError("Base must be between 2 and 9.")
        if x < 0:
            raise ValueError("Input number must be non-negative.")
        if x == 0:
            return '0'
        digits = []
        while x:
            digits.append(str(x % base))
            x //= base
        return ''.join(reversed(digits))  # Fixed the order of digits

'''
BaseConverter class for converting numbers from decimal to a specified base.
'''
class BaseConverter:
    @staticmethod
    def change_base(x: int, base: int) -> str:
        """Change numerical base of input number x to base.
        return string representation after the conversion.
        base numbers are less than 10.
        >>> change_base(8, 3)
        '22'
        >>> change_base(8, 2)
        '1000'
        >>> change_base(7, 2)
        '111'
        >>> change_base(-8, 2)
        ValueError: Input number must be non-negative.
        """
        if base < 2 or base >= 10:
            raise ValueError("Base must be between 2 and 9.")
        if x < 0:
            raise ValueError("Input number must be non-negative.")
        if x == 0:
            return '0'
        digits = []
        while x:
            digits.append(str(x % base))
            x //= base
        return ''.join(reversed(digits))  # Fixed the order of digits
