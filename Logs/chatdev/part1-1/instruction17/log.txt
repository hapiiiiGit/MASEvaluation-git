'''
This file contains the main function to parse the music string and return a list of integers representing beats.
'''

from typing import List

def parse_music(music_string: str) -> List[int]:
    """
    Input to this function is a string representing musical notes in a special ASCII format.
    Your task is to parse this string and return a list of integers corresponding to how many beats does each note last.

    Here is a legend:
    'o' - whole note, lasts four beats
    'o|' - half note, lasts two beats
    '.|' - quarter note, lasts one beat

    >>> parse_music('o o| .| o| o| .| .| .| .| o o')
    [4, 2, 1, 2, 2, 1, 1, 1, 1, 4, 4]
    """
    beats = []
    notes = music_string.split()
    
    for note in notes:
        if note == 'o':
            beats.append(4)
        elif note == 'o|':
            beats.append(2)
        elif note == '.|':
            beats.append(1)
    
    return beats
```

'''
This file contains the Graphical User Interface to interact with the user.

import tkinter as tk
from typing import List
def parse_music(music_string: str) -> List[int]:
    """
    Parse a string of musical notes and return a list of corresponding beats.
    Args:
        music_string (str): A string representing musical notes in a special ASCII format.
    Returns:
        List[int]: A list of integers corresponding to how many beats each note lasts.
    """
    beats = []
    notes = music_string.replace('|', ' | ').split()  # Split by spaces and handle '|' characters
    for note in notes:
        if note == 'o':
            beats.append(4)  # Whole note
        elif note == 'o|':
            beats.append(2)  # Half note
        elif note == '.|':
            beats.append(1)  # Quarter note
    return beats
def parse_music_gui():
    def get_music_string():
        music_string = entry.get()
        beats = parse_music(music_string)
        result_label.config(text=f"Beats: {beats}")
    root = tk.Tk()
    root.title("Music Parser")
    entry = tk.Entry(root, width=50)
    entry.pack(pady=10)
    parse_button = tk.Button(root, text="Parse Music", command=get_music_string)
    parse_button.pack(pady=10)
    result_label = tk.Label(root, text="")
    result_label.pack(pady=10)
    root.mainloop()
if __name__ == "__main__":
    parse_music_gui()

'''
This file contains the Graphical User Interface to interact with the user.

import tkinter as tk
from typing import List
def parse_music(music_string: str) -> List[int]:
    """
    Parse a string of musical notes and return a list of corresponding beats.
    Args:
        music_string (str): A string representing musical notes in a special ASCII format.
    Returns:
        List[int]: A list of integers corresponding to how many beats each note lasts.
    """
    beats = []
    notes = music_string.replace('|', ' | ').split()  # Split by spaces and handle '|' characters
    for note in notes:
        if note == 'o':
            beats.append(4)  # Whole note
        elif note == 'o|':
            beats.append(2)  # Half note
        elif note == '.|':
            beats.append(1)  # Quarter note
    return beats
def parse_music_gui():
    def get_music_string():
        music_string = entry.get()
        beats = parse_music(music_string)
        result_label.config(text=f"Beats: {beats}")

    root.title("Music Parser")
    entry = tk.Entry(root, width=50)
    entry.pack(pady=10)
    parse_button = tk.Button(root, text="Parse Music", command=get_music_string)
    parse_button.pack(pady=10)
    result_label = tk.Label(root, text="")
    result_label.pack(pady=10)
    root.mainloop()
