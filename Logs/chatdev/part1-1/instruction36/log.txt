'''
Main application file for the FizzBuzz GUI.
'''

from tkinter import Tk, Label, Entry, Button, StringVar
from fizzbuzz import fizz_buzz

class FizzBuzzApp:
    def __init__(self, master):
        self.master = master
        master.title("FizzBuzz Counter")

        self.label = Label(master, text="Enter a number:")
        self.label.pack()

        self.number_var = StringVar()
        self.entry = Entry(master, textvariable=self.number_var)
        self.entry.pack()

        self.calculate_button = Button(master, text="Calculate", command=self.calculate)
        self.calculate_button.pack()

        self.result_label = Label(master, text="")
        self.result_label.pack()

    def calculate(self):
        try:
            n = int(self.number_var.get())
            result = fizz_buzz(n)
            self.result_label.config(text=f"The digit '7' appears {result} times.")
        except ValueError:
            self.result_label.config(text="Please enter a valid integer.")

if __name__ == "__main__":
    root = Tk()
    app = FizzBuzzApp(root)
    root.mainloop()
```

```python
'''
Module containing the fizz_buzz function to count occurrences of the digit '7'.
'''

def fizz_buzz(n: int) -> int:
    count = 0
    for i in range(n):
        if i % 11 == 0 or i % 13 == 0:
            count += str(i).count('7')
    return count
```

'''
Module containing the fizz_buzz function to count occurrences of the digit '7'.
'''
def fizz_buzz(n: int) -> int:
    count = 0
    for i in range(n):
        if i % 11 == 0 or i % 13 == 0:
            # Check each digit of the number
            temp = i
            while temp > 0:
                if temp % 10 == 7:
                    count += 1
                temp //= 10
    return count

'''

from tkinter import Tk, Label, Entry, Button, StringVar
from fizzbuzz import fizz_buzz
class FizzBuzzApp:
    def __init__(self, master):
        self.master = master
        master.title("FizzBuzz Counter")
        self.label = Label(master, text="Enter a number:")
        self.label.pack()
        self.number_var = StringVar()
        self.entry = Entry(master, textvariable=self.number_var)
        self.entry.pack()
        self.calculate_button = Button(master, text="Calculate", command=self.calculate)
        self.calculate_button.pack()
        self.result_label = Label(master, text="")
        self.result_label.pack()
    def calculate(self):
        """Calculate the number of times the digit '7' appears in integers less than the input number."""
        try:
            n = int(self.number_var.get())
            result = fizz_buzz(n)
            self.result_label.config(text=f"The digit '7' appears {result} times.")
        except ValueError:
            self.result_label.config(text="Please enter a valid integer.")
if __name__ == "__main__":
    root = Tk()
    app = FizzBuzzApp(root)
    root.mainloop()

'''
Module containing the fizz_buzz function to count occurrences of the digit '7'.
'''
def fizz_buzz(n: int) -> int:
    count = 0
    for i in range(n):
        if i % 11 == 0 or i % 13 == 0:
            # Check each digit of the number
            temp = i
            while temp > 0:
                if temp % 10 == 7:
                    count += 1
                temp //= 10
    return count
