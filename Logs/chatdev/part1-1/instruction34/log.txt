Main application file for the Unique List application.
'''

from tkinter import Tk, Label, Entry, Button, Text, END
from unique_list_processor import UniqueListProcessor

class UniqueListApp:
    def __init__(self, master):
        self.master = master
        master.title("Unique List Finder")
        
        self.label = Label(master, text="Enter a list of numbers (comma-separated):")
        self.label.pack()
        
        self.entry = Entry(master, width=50)
        self.entry.pack()
        
        self.submit_button = Button(master, text="Submit", command=self.on_submit)
        self.submit_button.pack()
        
        self.result_text = Text(master, height=10, width=50)
        self.result_text.pack()
        
        self.processor = UniqueListProcessor()

    def on_submit(self):
        input_text = self.entry.get()
        try:
            input_list = [int(x.strip()) for x in input_text.split(',')]
            unique_sorted_list = self.processor.unique(input_list)
            self.result_text.delete(1.0, END)
            self.result_text.insert(END, str(unique_sorted_list))
        except ValueError:
            self.result_text.delete(1.0, END)
            self.result_text.insert(END, "Please enter valid integers.")

if __name__ == "__main__":
    root = Tk()
    app = UniqueListApp(root)
    root.mainloop()
```

from tkinter import Tk, Label, Entry, Button, Text, END
from unique_list_processor import UniqueListProcessor
class UniqueListApp:
    def __init__(self, master):
        self.master = master
        master.title("Unique List Finder")
        self.label = Label(master, text="Enter a list of numbers (comma-separated):")
        self.label.pack()
        self.entry = Entry(master, width=50)
        self.entry.pack()
        self.submit_button = Button(master, text="Submit", command=self.on_submit)
        self.submit_button.pack()
        self.result_text = Text(master, height=10, width=50)
        self.result_text.pack()
        self.processor = UniqueListProcessor()
    def on_submit(self):
        input_text = self.entry.get()
        try:
            input_list = [int(x.strip()) for x in input_text.split(',')]
            unique_sorted_list = self.processor.unique(input_list)
            self.result_text.delete(1.0, END)
            self.result_text.insert(END, str(unique_sorted_list))
        except ValueError:
            self.result_text.delete(1.0, END)
            self.result_text.insert(END, "Please enter valid integers.")
if __name__ == "__main__":
    root = Tk()
    app = UniqueListApp(root)
    root.mainloop()
