'''
Main application file that initializes and runs the GUI for finding the longest string.
'''

from tkinter import Tk, Label, Entry, Button, StringVar
from longest_string_finder import LongestStringFinder

class App:
    def __init__(self, master):
        self.master = master
        master.title("Longest String Finder")

        self.label = Label(master, text="Enter strings separated by commas:")
        self.label.pack()

        self.input_string = StringVar()
        self.entry = Entry(master, textvariable=self.input_string, width=50)
        self.entry.pack()

        self.find_button = Button(master, text="Find Longest String", command=self.find_longest)
        self.find_button.pack()

        self.result_label = Label(master, text="")
        self.result_label.pack()

    def find_longest(self):
        input_text = self.input_string.get()
        strings = [s.strip() for s in input_text.split(',')]
        longest_string = LongestStringFinder.longest(strings)
        self.result_label.config(text=f"Longest String: {longest_string if longest_string else 'None'}")

def main():
    root = Tk()
    app = App(root)
    root.mainloop()

if __name__ == "__main__":
    main()
```

'''
Module that contains the logic for finding the longest string in a list.
'''

from typing import List, Optional

class LongestStringFinder:
    @staticmethod
    def longest(strings: List[str]) -> Optional[str]:
        """ Out of list of strings, return the longest one. Return the first one in case of multiple
        strings of the same length. Return None in case the input list is empty.
        >>> longest([]) -> None
        >>> longest(['a', 'b', 'c']) -> 'a'
        >>> longest(['a', 'bb', 'ccc']) -> 'ccc'
        """
        if not strings:
            return None
        longest_string = strings[0]
        for string in strings:
            if len(string) > len(longest_string):
                longest_string = string
        return longest_string
```

'''
Module that contains the logic for finding the longest string in a list.
'''
from typing import List, Optional
class LongestStringFinder:
    @staticmethod
    def longest(strings: List[str]) -> Optional[str]:
        """ Out of list of strings, return the longest one. Return the first one in case of multiple
        strings of the same length. Return None in case the input list is empty.
        >>> longest([]) -> None
        >>> longest(['a', 'b', 'c']) -> 'a'
        >>> longest(['a', 'bb', 'ccc']) -> 'ccc'
        """
        if not strings:
            return None
        longest_string = strings[0]
        for string in strings:
            if len(string) > len(longest_string):
                longest_string = string
        return longest_string

from tkinter import Tk, Label, Entry, Button, StringVar
from longest_string_finder import LongestStringFinder
class App:
    def __init__(self, master):
        self.master = master
        master.title("Longest String Finder")
        self.label = Label(master, text="Enter strings separated by commas:")
        self.label.pack()
        self.input_string = StringVar()
        self.entry = Entry(master, textvariable=self.input_string, width=50)
        self.entry.pack()
        self.find_button = Button(master, text="Find Longest String", command=self.find_longest)
        self.find_button.pack()
        self.result_label = Label(master, text="")
        self.result_label.pack()
    def find_longest(self):
        input_text = self.input_string.get()
        strings = [s.strip() for s in input_text.split(',')]
        longest_string = LongestStringFinder.longest(strings)
        self.result_label.config(text=f"Longest String: {longest_string if longest_string else 'None'}")
def main():
    root = Tk()
    app = App(root)
    root.mainloop()
if __name__ == "__main__":
    main()

from typing import List, Optional
class LongestStringFinder:
    @staticmethod
    def longest(strings: List[str]) -> Optional[str]:
        """ Out of list of strings, return the longest one. Return the first one in case of multiple
        strings of the same length. Return None in case the input list is empty.
        >>> LongestStringFinder.longest([]) -> None
        >>> LongestStringFinder.longest(['a', 'b', 'c']) -> 'a'
        >>> LongestStringFinder.longest(['a', 'bb', 'ccc']) -> 'ccc'
        """
        if not strings:
            return None
        longest_string = strings[0]
        for string in strings:
            if len(string) > len(longest_string):
                longest_string = string
        return longest_string

'''
Main application file that initializes and runs the GUI for finding the longest string.
'''
from tkinter import Tk, Label, Entry, Button, StringVar
from longest_string_finder import LongestStringFinder
class App:
    def __init__(self, master):
        self.master = master
        master.title("Longest String Finder")
        self.label = Label(master, text="Enter strings separated by commas:")
        self.label.pack()
        self.input_string = StringVar()
        self.entry = Entry(master, textvariable=self.input_string, width=50)
        self.entry.pack()
        self.find_button = Button(master, text="Find Longest String", command=self.find_longest)
        self.find_button.pack()
        self.result_label = Label(master, text="")
        self.result_label.pack()
    def find_longest(self):
        input_text = self.input_string.get()
        strings = [s.strip() for s in input_text.split(',')]
        longest_string = LongestStringFinder.longest(strings)
        self.result_label.config(text=f"Longest String: {longest_string if longest_string else 'None'}")
def main():
    root = Tk()
    app = App(root)
    root.mainloop()
if __name__ == "__main__":
    main()

'''
Module that contains the logic for finding the longest string in a list.
'''
from typing import List, Optional
class LongestStringFinder:
    @staticmethod
    def longest(strings: List[str]) -> Optional[str]:
        """ Out of list of strings, return the longest one. Return the first one in case of multiple
        strings of the same length. Return None in case the input list is empty.
        >>> LongestStringFinder.longest([]) -> None
        >>> LongestStringFinder.longest(['a', 'b', 'c']) -> 'a'
        >>> LongestStringFinder.longest(['a', 'bb', 'ccc']) -> 'ccc'
        """
        if not strings:
            return None
        longest_string = strings[0]
        for string in strings:
            if len(string) > len(longest_string):
                longest_string = string
        return longest_string

# Longest String Finder User Manual

Welcome to the Longest String Finder application user manual. This manual will guide you through the main functions of the software, how to install environment dependencies, and how to use the application effectively.

## Main Functions

The Longest String Finder application allows you to input a list of strings separated by commas and find the longest string in the list. If there are multiple strings of the same length, the application will return the first one. In case the input list is empty, the application will return None.

## Installation

To run the Longest String Finder application, you need to install the required dependencies listed in the `requirements.txt` file. You can do this using `pip` by running the following command:

```bash
pip install -r requirements.txt
```

## How to Use

1. Clone the repository containing the `main.py` and `longest_string_finder.py` files.
2. Install the required dependencies as mentioned above.
3. Run the `main.py` file using Python to start the application.
4. A GUI window titled "Longest String Finder" will appear.
5. Enter the strings you want to analyze in the input field, separated by commas.
6. Click on the "Find Longest String" button to find the longest string in the input list.
7. The result will be displayed below the input field, showing the longest string found.

## Example

Suppose you input the following strings: "a, bb, ccc". Upon clicking the "Find Longest String" button, the application will return "ccc" as the longest string in the list.
