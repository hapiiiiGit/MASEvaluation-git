'''
Contains the logic for calculating the number of collisions between cars.
'''

class CollisionCalculator:
    @staticmethod
    def car_race_collision(n: int) -> int:
        """
        Calculate the number of collisions between cars moving in opposite directions.
        
        Args:
            n (int): The number of cars in each direction.
        
        Returns:
            int: The number of collisions.
        """
        # Each car moving left to right will collide with each car moving right to left
        return n * n
```

'''
Handles the graphical user interface for the car race collision application.
'''

import tkinter as tk
from tkinter import messagebox
from collision import CollisionCalculator

class App:
    def __init__(self):
        self.window = tk.Tk()
        self.window.title("Car Race Collision Calculator")

        self.label = tk.Label(self.window, text="Enter the number of cars (n):")
        self.label.pack()

        self.entry = tk.Entry(self.window)
        self.entry.pack()

        self.calculate_button = tk.Button(self.window, text="Calculate Collisions", command=self.calculate_collisions)
        self.calculate_button.pack()

        self.result_label = tk.Label(self.window, text="")
        self.result_label.pack()

    def calculate_collisions(self):
        try:
            n = int(self.entry.get())
            if n < 0:
                raise ValueError("Number of cars must be non-negative.")
            collisions = CollisionCalculator.car_race_collision(n)
            self.result_label.config(text=f"Number of collisions: {collisions}")
        except ValueError as e:
            messagebox.showerror("Input Error", str(e))

    def run(self):
        self.window.mainloop()

class CollisionCalculator:
    @staticmethod
    def car_race_collision(n: int) -> int:
        """
        Calculate the number of collisions between cars moving in opposite directions.
        Args:
            n (int): The number of cars in each direction.
        Returns:
            int: The number of collisions.
        """
        # Each car moving left to right will collide with each car moving right to left

'''
Contains the logic for calculating the number of collisions between cars.
'''
class CollisionCalculator:
    @staticmethod
    def car_race_collision(n: int) -> int:
        """
        Calculate the number of collisions between cars moving in opposite directions.
        Args:
            n (int): The number of cars in each direction.
        Returns:
            int: The number of collisions.
        """
        # Each car moving left to right will collide with each car moving right to left
        return n * n

'''
Handles the graphical user interface for the car race collision application.
'''
import tkinter as tk
from tkinter import messagebox
from collision import CollisionCalculator
class App:
    def __init__(self):
        self.window = tk.Tk()
        self.window.title("Car Race Collision Calculator")
        self.label = tk.Label(self.window, text="Enter the number of cars (n):")
        self.label.pack()
        self.entry = tk.Entry(self.window)
        self.entry.pack()
        self.calculate_button = tk.Button(self.window, text="Calculate Collisions", command=self.calculate_collisions)
        self.calculate_button.pack()
        self.result_label = tk.Label(self.window, text="")
        self.result_label.pack()
    def calculate_collisions(self):
        try:
            n = int(self.entry.get())
            if n < 0:
                raise ValueError("Number of cars must be non-negative.")
            collisions = CollisionCalculator.car_race_collision(n)
            self.result_label.config(text=f"Number of collisions: {collisions}")
        except ValueError as e:
            messagebox.showerror("Input Error", str(e))
    def run(self):
        self.window.mainloop()

class CollisionCalculator:
    @staticmethod
    def car_race_collision(n: int) -> int:
        """
        Calculate the number of collisions between cars moving in opposite directions.
        Args:
            n (int): The number of cars in each direction.
        Returns:
            int: The number of collisions.
        """
        # Each car moving left to right will collide with each car moving right to left
        return n * n
