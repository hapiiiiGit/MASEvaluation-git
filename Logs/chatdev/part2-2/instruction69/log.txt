'''
Module to calculate the count of binary sequences of length 2n where the sum of the first n bits equals the sum of the last n bits.
'''
class BinarySequenceCounter:
    @staticmethod
    def count_sequences(n):
        # Dynamic programming approach to count sequences
        dp = [[0 for _ in range(n + 1)] for _ in range(n + 1)]
        dp[0][0] = 1
        for i in range(1, 2 * n + 1):
            for j in range(min(i, n), -1, -1):
                for k in range(min(i, n), -1, -1):
                    if j > 0:
                        dp[j][k] += dp[j - 1][k]
                    if k > 0:
                        dp[j][k] += dp[j][k - 1]
        return dp[n][n]

'''  
Module to calculate the count of binary sequences of length 2n where the sum of the first n bits equals the sum of the last n bits.  
'''  
class BinarySequenceCounter:  
    @staticmethod  
    def count_sequences(n):  
        """  
        Count binary sequences of length 2n where the sum of the first n bits equals the sum of the last n bits.  
        :param n: Length of half the binary sequence  
        :return: Count of valid binary sequences  
        """  
        # Use dynamic programming to calculate the count of sequences  
        # dp[i][j] represents the number of ways to form a sequence of length i with a sum difference of j  
        dp = [[0 for _ in range(2 * n + 1)] for _ in range(n + 1)]  
        dp[0][n] = 1  # Base case: 0 length, difference is 0 (middle index is n)  
        for i in range(1, n + 1):  
            for j in range(2 * n + 1):  
                if j > 0:  
                    dp[i][j] += dp[i - 1][j - 1]  # Add a 1 to the first half  
                if j < 2 * n:  
                    dp[i][j] += dp[i - 1][j + 1]  # Add a 1 to the second half  
        return dp[n][n]  # The count of sequences with zero difference

'''  
Module to calculate the count of binary sequences of length 2n where the sum of the first n bits equals the sum of the last n bits.  
'''  
class BinarySequenceCounter:  
    @staticmethod  
    def count_sequences(n):  
        """  
        Count binary sequences of length 2n where the sum of the first n bits equals the sum of the last n bits.  
        :param n: Length of half the binary sequence  
        :return: Count of valid binary sequences  
        """  
        # Use dynamic programming to calculate the count of sequences  
        # dp[i][j] represents the number of ways to form a sequence of length i with a sum difference of j  
        dp = [[0 for _ in range(2 * n + 1)] for _ in range(n + 1)]  
        dp[0][n] = 1  # Base case: 0 length, difference is 0 (middle index is n)  
        for i in range(1, n + 1):  
            for j in range(2 * n + 1):  
                if j > 0:  
                    dp[i][j] += dp[i - 1][j - 1]  # Add a 1 to the first half  
                if j < 2 * n:  
                    dp[i][j] += dp[i - 1][j + 1]  # Add a 1 to the second half  
        return dp[n][n]  # The count of sequences with zero difference

'''
Module to calculate the count of binary sequences of length 2n where the sum of the first n bits equals the sum of the last n bits.
'''
class BinarySequenceCounter:
    @staticmethod
    def count_sequences(n):
        """
        Count binary sequences of length 2n where the sum of the first n bits equals the sum of the last n bits.
        :param n: Length of half the binary sequence
        :return: Count of valid binary sequences
        """
        # Use dynamic programming to calculate the count of sequences
        # dp[i][j] represents the number of ways to form a sequence of length i with a sum difference of j
        dp = [[0 for _ in range(2 * n + 1)] for _ in range(n + 1)]
        dp[0][n] = 1  # Base case: 0 length, difference is 0 (middle index is n)
        for i in range(1, n + 1):
            for j in range(2 * n + 1):
                if j > 0:
                    dp[i][j] += dp[i - 1][j - 1]  # Add a 1 to the first half
                if j < 2 * n:
                    dp[i][j] += dp[i - 1][j + 1]  # Add a 1 to the second half
        return dp[n][n]  # The count of sequences with zero difference

'''
Module to calculate the count of binary sequences of length 2n where the sum of the first n bits equals the sum of the last n bits.
'''
class BinarySequenceCounter:
    @staticmethod
    def count_sequences(n):
        """
        Count binary sequences of length 2n where the sum of the first n bits equals the sum of the last n bits.
        :param n: Length of half the binary sequence
        :return: Count of valid binary sequences
        """
        # Use dynamic programming to calculate the count of sequences
        # dp[i][j] represents the number of ways to form a sequence of length i with a sum difference of j
        dp = [[0 for _ in range(2 * n + 1)] for _ in range(n + 1)]
        dp[0][n] = 1  # Base case: 0 length, difference is 0 (middle index is n)
        for i in range(1, n + 1):
            for j in range(2 * n + 1):
                if j > 0:
                    dp[i][j] += dp[i - 1][j - 1]  # Add a 1 to the first half
                if j < 2 * n:
                    dp[i][j] += dp[i - 1][j + 1]  # Add a 1 to the second half
        return dp[n][n]  # The count of sequences with zero difference
