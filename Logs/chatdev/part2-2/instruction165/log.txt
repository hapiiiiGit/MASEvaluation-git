'''

This file contains the main application logic for the Python program that calculates the difference between the sum of cubes of the first n natural numbers and the sum of the first n natural numbers. It includes the GUI implementation using tkinter.

'''

import tkinter as tk
from tkinter import messagebox
from calculator import calculate_difference


class MainApp:
    '''
    This class defines the graphical user interface (GUI) for the application.
    '''
    def __init__(self, root):
        self.root = root
        self.root.title("Sum of Cubes Difference Calculator")
        self.root.geometry("400x300")
        
        # Create GUI elements
        self.label = tk.Label(root, text="Enter a natural number (n):", font=("Arial", 12))
        self.label.pack(pady=10)
        
        self.entry = tk.Entry(root, font=("Arial", 12))
        self.entry.pack(pady=10)
        
        self.calculate_button = tk.Button(root, text="Calculate", command=self.calculate, font=("Arial", 12))
        self.calculate_button.pack(pady=10)
        
        self.result_label = tk.Label(root, text="", font=("Arial", 12), fg="blue")
        self.result_label.pack(pady=10)
    
    def calculate(self):
        '''
        Handles the calculation when the user clicks the "Calculate" button.
        '''
        try:
            n = int(self.entry.get())
            if n <= 0:
                raise ValueError("Please enter a positive integer.")
            
            # Call the calculation function
            result = calculate_difference(n)
            
            # Display the result
            self.result_label.config(text=f"Result: {result}")
        except ValueError as e:
            messagebox.showerror("Invalid Input", str(e))


def run_app():
    '''
    Initializes and runs the GUI application.
    '''
    root = tk.Tk()
    app = MainApp(root)
    root.mainloop()


if __name__ == "__main__":
    run_app()

This file contains the function to calculate the difference between the sum of cubes of the first n natural numbers and the sum of the first n natural numbers.

'''

def calculate_difference(n):
    '''
    Calculates the difference between the sum of cubes of the first n natural numbers
    and the sum of the first n natural numbers.

    Args:
        n (int): The number of natural numbers to consider.

    Returns:
        int: The calculated difference.
    '''
    # Calculate the sum of the first n natural numbers
    sum_n = sum(range(1, n + 1))
    
    # Calculate the sum of cubes of the first n natural numbers
    sum_cubes = sum(i**3 for i in range(1, n + 1))
    
    # Return the difference
    return sum_cubes - sum_n
```

'''

This file contains the main application logic for the Python program that calculates the difference between the sum of cubes of the first n natural numbers and the sum of the first n natural numbers. It includes the GUI implementation using tkinter.

'''

import tkinter as tk
from tkinter import messagebox
from calculator import calculate_difference


class MainApp:
    '''
    This class defines the graphical user interface (GUI) for the application.
    '''
    def __init__(self, root):
        self.root = root
        self.root.title("Sum of Cubes Difference Calculator")
        self.root.geometry("400x300")
        
        # Create GUI elements
        self.label = tk.Label(root, text="Enter a natural number (n):", font=("Arial", 12))
        self.label.pack(pady=10)
        
        self.entry = tk.Entry(root, font=("Arial", 12))
        self.entry.pack(pady=10)
        
        self.calculate_button = tk.Button(root, text="Calculate", command=self.calculate, font=("Arial", 12))
        self.calculate_button.pack(pady=10)
        
        self.result_label = tk.Label(root, text="", font=("Arial", 12), fg="blue")
        self.result_label.pack(pady=10)
    
    def calculate(self):
        '''
        Handles the calculation when the user clicks the "Calculate" button.
        '''
        try:
            n = int(self.entry.get())
            if n <= 0:
                raise ValueError("Please enter a positive integer.")
            
            # Call the calculation function
            result = calculate_difference(n)
            
            # Display the result
            self.result_label.config(text=f"Result: {result}")
        except ValueError as e:
            messagebox.showerror("Invalid Input", str(e))


def run_app():
    '''
    Initializes and runs the GUI application.
    '''
    root = tk.Tk()
    app = MainApp(root)
    root.mainloop()


if __name__ == "__main__":
    run_app()

def calculate_difference(n):
    '''

    # Calculate the sum of cubes of the first n natural numbers

    # Return the difference
    return sum_cubes - sum_n

def calculate_difference(n):
    '''
    Calculates the difference between the sum of cubes of the first n natural numbers
    and the sum of the first n natural numbers.
    Args:
        n (int): The number of natural numbers to consider.
    Returns:
        int: The calculated difference.
    '''
    # Calculate the sum of the first n natural numbers
    sum_n = sum(range(1, n + 1))
    # Calculate the sum of cubes of the first n natural numbers
    sum_cubes = sum(i**3 for i in range(1, n + 1))
    # Return the difference
    return sum_cubes - sum_n

'''
This file contains the main application logic for the Python program that calculates the difference between the sum of cubes of the first n natural numbers and the sum of the first n natural numbers. It includes the GUI implementation using tkinter.
'''
import tkinter as tk
from tkinter import messagebox
from calculator import calculate_difference

class MainApp:
    '''
    This class defines the graphical user interface (GUI) for the application.
    '''
    def __init__(self, root):
        self.root = root
        self.root.title("Sum of Cubes Difference Calculator")
        self.root.geometry("400x300")
        # Create GUI elements
        self.label = tk.Label(root, text="Enter a natural number (n):", font=("Arial", 12))
        self.label.pack(pady=10)
        self.entry = tk.Entry(root, font=("Arial", 12))
        self.entry.pack(pady=10)
        self.calculate_button = tk.Button(root, text="Calculate", command=self.calculate, font=("Arial", 12))
        self.calculate_button.pack(pady=10)
        self.result_label = tk.Label(root, text="", font=("Arial", 12), fg="blue")
        self.result_label.pack(pady=10)

    def calculate(self):
        '''
        Handles the calculation when the user clicks the "Calculate" button.
        '''
        try:
            n = int(self.entry.get())
            if n <= 0:
                raise ValueError("Please enter a positive integer.")
            # Call the calculation function
            result = calculate_difference(n)
            # Display the result
            self.result_label.config(text=f"Result: {result}")
        except ValueError as e:
            messagebox.showerror("Invalid Input", str(e))

def run_app():
    '''
    Initializes and runs the GUI application.
    '''
    root = tk.Tk()
    app = MainApp(root)
    root.mainloop()

if __name__ == "__main__":
    run_app()

'''

def calculate_difference(n):
    '''
    Calculates the difference between the sum of cubes of the first n natural numbers
    and the sum of the first n natural numbers.
    Args:
        n (int): The number of natural numbers to consider.
    Returns:
        int: The calculated difference.
    '''
    # Calculate the sum of the first n natural numbers
    sum_n = sum(range(1, n + 1))
    # Calculate the sum of cubes of the first n natural numbers
    sum_cubes = sum(i**3 for i in range(1, n + 1))
    # Return the difference
    return sum_cubes - sum_n

'''
This file contains the main application logic for the Python program that calculates the difference between the sum of cubes of the first n natural numbers and the sum of the first n natural numbers. It includes the GUI implementation using tkinter.
'''
import tkinter as tk
from tkinter import messagebox
from calculator import calculate_difference
class MainApp:
    '''
    This class defines the graphical user interface (GUI) for the application.
    '''
    def __init__(self, root):
        self.root = root
        self.root.title("Sum of Cubes Difference Calculator")
        self.root.geometry("400x300")
        # Create GUI elements
        self.label = tk.Label(root, text="Enter a natural number (n):", font=("Arial", 12))
        self.label.pack(pady=10)

        self.calculate_button.pack(pady=10)
        self.result_label = tk.Label(root, text="", font=("Arial", 12), fg="blue")
        self.result_label.pack(pady=10)
    def calculate(self):
        '''
        Handles the calculation when the user clicks the "Calculate" button.
        '''
        try:
            n = int(self.entry.get())
            if n <= 0:
                raise ValueError("Please enter a positive integer.")
            # Call the calculation function
            result = calculate_difference(n)
            # Display the result
            self.result_label.config(text=f"Result: {result}")
        except ValueError as e:
            messagebox.showerror("Invalid Input", str(e))
def run_app():
    '''
    Initializes and runs the GUI application.
    '''
    root = tk.Tk()
    app = MainApp(root)
    root.mainloop()

    run_app()

def calculate_difference(n):
    '''
    Calculates the difference between the sum of cubes of the first n natural numbers
    and the sum of the first n natural numbers.
    Args:
        n (int): The number of natural numbers to consider.
    Returns:
        int: The calculated difference.
    '''
    # Calculate the sum of the first n natural numbers
    sum_n = sum(range(1, n + 1))
    # Calculate the sum of cubes of the first n natural numbers
    sum_cubes = sum(i**3 for i in range(1, n + 1))
    # Return the difference
    return sum_cubes - sum_n
