    '''

'''
Main file to run the Triangle Calculator application using Tkinter.

import tkinter as tk
from tkinter import messagebox
def largest_triangle_area(radius):
    '''
    Calculate the area of the largest triangle that can be inscribed in a semicircle with the given radius.
    Parameters:
    radius (float): The radius of the semicircle.
    Returns:
    float: The area of the largest inscribed triangle.
    '''
    # The largest triangle is an isosceles right triangle with hypotenuse as the diameter of the semicircle
    area = (radius * radius)
    return area
class TriangleCalculatorApp:
    '''
    A GUI application to calculate the area of the largest triangle inscribed in a semicircle.
    '''
    def __init__(self, master):
        self.master = master
        master.title("Triangle Area Calculator")
        self.label = tk.Label(master, text="Enter the radius of the semicircle:")
        self.label.pack()
        self.radius_entry = tk.Entry(master)
        self.radius_entry.pack()
        self.calculate_button = tk.Button(master, text="Calculate Area", command=self.calculate_area)
        self.calculate_button.pack()
        self.result_label = tk.Label(master, text="")
        self.result_label.pack()
    def calculate_area(self):
        try:
            radius = float(self.radius_entry.get())
            if radius <= 0:
                raise ValueError("Radius must be positive.")
            area = largest_triangle_area(radius)
            self.display_result(area)
        except ValueError as e:
            messagebox.showerror("Invalid input", str(e))
    def display_result(self, area):
        self.result_label.config(text=f"The area of the largest inscribed triangle is: {area:.2f}")
if __name__ == "__main__":
    root = tk.Tk()
    app = TriangleCalculatorApp(root)
    root.mainloop()

'''
Main file to run the Triangle Calculator application using Tkinter.
'''
import tkinter as tk
from tkinter import messagebox
def largest_triangle_area(radius):
    '''
    Calculate the area of the largest triangle that can be inscribed in a semicircle with the given radius.
    Parameters:
    radius (float): The radius of the semicircle.
    Returns:
    float: The area of the largest inscribed triangle.
    '''
    # The largest triangle is an isosceles right triangle with hypotenuse as the diameter of the semicircle
    area = (radius * radius)
    return area
class TriangleCalculatorApp:
    '''
    A GUI application to calculate the area of the largest triangle inscribed in a semicircle.
    '''
    def __init__(self, master):
        self.master = master
        master.title("Triangle Area Calculator")
        self.label = tk.Label(master, text="Enter the radius of the semicircle:")
        self.label.pack()
        self.radius_entry = tk.Entry(master)
        self.radius_entry.pack()
        self.calculate_button = tk.Button(master, text="Calculate Area", command=self.calculate_area)
        self.calculate_button.pack()
        self.result_label = tk.Label(master, text="")
        self.result_label.pack()
    def calculate_area(self):
        try:
            radius = float(self.radius_entry.get())
            if radius <= 0:
                raise ValueError("Radius must be positive.")
            area = largest_triangle_area(radius)
            self.display_result(area)
        except ValueError as e:
            messagebox.showerror("Invalid input", str(e))
    def display_result(self, area):
        self.result_label.config(text=f"The area of the largest inscribed triangle is: {area:.2f}")
if __name__ == "__main__":
    root = tk.Tk()
    app = TriangleCalculatorApp(root)
    root.mainloop()
