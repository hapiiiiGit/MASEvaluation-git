'''

This module contains the main application code for a GUI that checks if a word contains the letter 'z'.

'''

import tkinter as tk

def check_for_z(word):
    '''
    Check if the given word contains the letter 'z'.

    :param word: The word to check.
    :return: True if the word contains 'z', False otherwise.
    '''
    return 'z' in word

class MainApp:
    '''
    The main application class for the word checker GUI.
    '''

    def __init__(self, root):
        '''
        Initialize the main application window and its components.

        :param root: The root window of the tkinter application.
        '''
        self.root = root
        self.root.title("Word Checker")
        self.create_widgets()

    def create_widgets(self):
        '''
        Create and place the widgets in the application window.
        '''
        self.label = tk.Label(self.root, text="Enter a word:")
        self.label.pack(pady=10)

        self.entry = tk.Entry(self.root)
        self.entry.pack(pady=5)

        self.check_button = tk.Button(self.root, text="Check", command=self.check_word)
        self.check_button.pack(pady=10)

        self.result_label = tk.Label(self.root, text="")
        self.result_label.pack(pady=10)

    def check_word(self):
        '''
        Check if the entered word contains 'z' and update the result label.
        '''
        word = self.entry.get()
        if check_for_z(word):
            self.result_label.config(text="The word contains 'z'.")
        else:
            self.result_label.config(text="The word does not contain 'z'.")

if __name__ == "__main__":
    root = tk.Tk()
    app = MainApp(root)
    root.mainloop()
```

  - An entry widget for user input.
  - A button that triggers the `check_word` method.
  - A result label that displays whether the word contains 'z'.

- The application runs in a loop, waiting for user interaction, and updates the GUI based on the input.

This module contains the main application code for a GUI that checks if a word contains the letter 'z'.

'''

import tkinter as tk

def check_for_z(word):
    '''
    Check if the given word contains the letter 'z'.

    :param word: The word to check.
    :return: True if the word contains 'z', False otherwise.
    '''
    return 'z' in word

class MainApp:
    '''
    The main application class for the word checker GUI.
    '''

    def __init__(self, root):
        '''
        Initialize the main application window and its components.

        :param root: The root window of the tkinter application.
        '''
        self.root = root
        self.root.title("Word Checker")
        self.create_widgets()

    def create_widgets(self):
        '''
        Create and place the widgets in the application window.
        '''
        self.label = tk.Label(self.root, text="Enter a word:")
        self.label.pack(pady=10)

        self.entry = tk.Entry(self.root)
        self.entry.pack(pady=5)

        self.check_button = tk.Button(self.root, text="Check", command=self.check_word)
        self.check_button.pack(pady=10)

        self.result_label = tk.Label(self.root, text="")
        self.result_label.pack(pady=10)

    def check_word(self):
        '''
        Check if the entered word contains 'z' and update the result label.
        '''
        word = self.entry.get()
        if check_for_z(word):
            self.result_label.config(text="The word contains 'z'.")
        else:
            self.result_label.config(text="The word does not contain 'z'.")

if __name__ == "__main__":
    root = tk.Tk()
    app = MainApp(root)
    root.mainloop()
```

### Explanation

- **check_for_z**: This function takes a word as input and returns `True` if the word contains the letter 'z', otherwise `False`.

- **MainApp**: This class sets up the GUI using `tkinter`. It includes:
  - A label prompting the user to enter a word.
  - An entry widget for user input.
  - A button that triggers the `check_word` method.
  - A result label that displays whether the word contains 'z'.

- The application runs in a loop, waiting for user interaction, and updates the GUI based on the input.

This implementation provides a simple and user-friendly interface for checking if a word contains the letter 'z'.
