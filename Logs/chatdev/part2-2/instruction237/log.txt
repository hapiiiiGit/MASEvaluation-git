'''

Main application file for checking no profit and no loss using tkinter GUI.

'''

import tkinter as tk
from tkinter import messagebox

def check_no_profit_no_loss(cost_price, selling_price):
    '''
    Function to check if there is no profit and no loss.
    
    Parameters:
    cost_price (float): The cost price of the item.
    selling_price (float): The selling price of the item.
    
    Returns:
    bool: True if no profit and no loss, False otherwise.
    '''
    return cost_price == selling_price

class MainApplication(tk.Tk):
    '''
    Main application class for the GUI.
    '''
    def __init__(self):
        super().__init__()
        self.title("No Profit No Loss Checker")
        self.geometry("300x200")
        
        # Labels and Entries
        self.cost_label = tk.Label(self, text="Cost Price:")
        self.cost_label.pack(pady=5)
        self.cost_entry = tk.Entry(self)
        self.cost_entry.pack(pady=5)
        
        self.selling_label = tk.Label(self, text="Selling Price:")
        self.selling_label.pack(pady=5)
        self.selling_entry = tk.Entry(self)
        self.selling_entry.pack(pady=5)
        
        # Check Button
        self.check_button = tk.Button(self, text="Check", command=self.check_no_profit_no_loss)
        self.check_button.pack(pady=20)
    
    def check_no_profit_no_loss(self):
        '''
        Method to check if there is no profit and no loss and display the result.
        '''
        try:

import tkinter as tk
from tkinter import messagebox

def check_no_profit_no_loss(cost_price, selling_price):
    '''
    Function to check if there is no profit and no loss.
    
    Parameters:
    cost_price (float): The cost price of the item.
    selling_price (float): The selling price of the item.
    
    Returns:
    bool: True if no profit and no loss, False otherwise.

import tkinter as tk
from tkinter import messagebox

def check_no_profit_no_loss(cost_price, selling_price):
    '''
    Function to check if there is no profit and no loss.
    Parameters:
    cost_price (float): The cost price of the item.
    selling_price (float): The selling price of the item.
    Returns:
    bool: True if no profit and no loss, False otherwise.
    '''
    return cost_price == selling_price

class MainApplication(tk.Tk):
    '''
    Main application class for the GUI.
    '''
    def __init__(self):
        super().__init__()
        self.title("No Profit No Loss Checker")
        self.geometry("300x250")
        self.resizable(False, False)  # Make the window non-resizable
        self.create_widgets()

    def create_widgets(self):
        '''
        Method to create and arrange widgets in the GUI.
        '''
        # Labels and Entries
        self.cost_label = tk.Label(self, text="Cost Price:")
        self.cost_label.pack(pady=5)
        self.cost_entry = tk.Entry(self)
        self.cost_entry.pack(pady=5)

        self.selling_label = tk.Label(self, text="Selling Price:")
        self.selling_label.pack(pady=5)
        self.selling_entry = tk.Entry(self)
        self.selling_entry.pack(pady=5)

        # Check Button
        self.check_button = tk.Button(self, text="Check", command=self.check_no_profit_no_loss)
        self.check_button.pack(pady=10)

        # Clear Button
        self.clear_button = tk.Button(self, text="Clear", command=self.clear_entries)
        self.clear_button.pack(pady=5)

        # Exit Button
        self.exit_button = tk.Button(self, text="Exit", command=self.exit_application)
        self.exit_button.pack(pady=5)

    def check_no_profit_no_loss(self):
        '''
        Method to check if there is no profit and no loss and display the result.
        '''
        try:
            cost_price = float(self.cost_entry.get())
            selling_price = float(self.selling_entry.get())
            if check_no_profit_no_loss(cost_price, selling_price):
                messagebox.showinfo("Result", "No Profit No Loss")
            else:
                if selling_price > cost_price:
                    messagebox.showinfo("Result", "There is a Profit")
                else:
                    messagebox.showinfo("Result", "There is a Loss")
        except ValueError:
            messagebox.showerror("Error", "Please enter valid numbers")

    def clear_entries(self):
        '''
        Method to clear the input fields.
        '''
        self.cost_entry.delete(0, tk.END)
        self.selling_entry.delete(0, tk.END)

    def exit_application(self):
        '''
        Method to exit the application.
        '''
        self.destroy()

if __name__ == "__main__":
    app = MainApplication()
    app.mainloop()

def check_no_profit_no_loss(cost_price, selling_price):
    '''
    Function to check if there is no profit and no loss.
    Parameters:
    cost_price (float): The cost price of the item.
    selling_price (float): The selling price of the item.
    Returns:
    bool: True if no profit and no loss, False otherwise.
    '''

    '''
    Main application class for the GUI.
    '''

def check_no_profit_no_loss(cost_price, selling_price):
    '''

    bool: True if no profit and no loss, False otherwise.

class MainApplication(tk.Tk):
    '''
    Main application class for the GUI.
    '''
    def __init__(self):
        super().__init__()
        self.title("No Profit No Loss Checker")
        self.geometry("300x250")
        self.resizable(False, False)  # Make the window non-resizable
        self.create_widgets()
