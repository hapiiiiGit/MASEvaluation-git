import tkinter as tk

from pattern_matcher import match_pattern

class MainApplication:
    def __init__(self, root):
        self.root = root
        self.root.title("Pattern Matcher")
        
        self.label = tk.Label(root, text="Enter a string:")
        self.label.pack(pady=10)
        
        self.entry = tk.Entry(root, width=50)
        self.entry.pack(pady=10)
        
        self.check_button = tk.Button(root, text="Check Pattern", command=self.check_pattern)
        self.check_button.pack(pady=10)
        
    def check_pattern(self):
        user_input = self.entry.get()
        if match_pattern(user_input):
            messagebox.showinfo("Result", "The string matches the pattern!")
        else:
            messagebox.showinfo("Result", "The string does not match the pattern.")
