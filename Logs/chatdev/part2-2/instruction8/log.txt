'''

main.py

import tkinter as tk
from tkinter import messagebox

def are_all_bits_unset(start, end):
    """
    Check if all bits are unset in the given range.
    
    Parameters:
    start (int): The starting integer of the range.
    end (int): The ending integer of the range.
    
    Returns:
    bool: True if all bits are unset, False otherwise.
    """
    for number in range(start, end + 1):
        if number != 0:
            return False
    return True

class BitCheckerApp:
    """
    A class to create a GUI application for checking if all bits are unset in a given range.
    """
    def __init__(self, master):
        self.master = master
        master.title("Bit Checker")

        self.label = tk.Label(master, text="Enter range (start and end):")
        self.label.pack()

        self.start_entry = tk.Entry(master)
        self.start_entry.pack()
        self.start_entry.insert(0, "0")

        self.end_entry = tk.Entry(master)
        self.end_entry.pack()
        self.end_entry.insert(0, "0")

        self.check_button = tk.Button(master, text="Check Bits", command=self.check_bits)
        self.check_button.pack()

    def check_bits(self):
        try:
            start = int(self.start_entry.get())
            end = int(self.end_entry.get())
            if start > end:
                messagebox.showerror("Error", "Start should be less than or equal to End.")
                return

            result = are_all_bits_unset(start, end)
            if result:
                messagebox.showinfo("Result", "All bits are unset in the given range.")
            else:
                messagebox.showinfo("Result", "Not all bits are unset in the given range.")
        except ValueError:
            messagebox.showerror("Error", "Please enter valid integers for start and end.")

if __name__ == "__main__":
    root = tk.Tk()
    app = BitCheckerApp(root)
    root.mainloop()
```

import tkinter as tk
from tkinter import messagebox

def are_all_bits_unset(start, end):
    """
    Check if all bits are unset in the given range.
    
    Parameters:
    start (int): The starting integer of the range.
    end (int): The ending integer of the range.
    
    Returns:
    bool: True if all bits are unset, False otherwise.
    """
    for number in range(start, end + 1):
        if number != 0:
            return False
    return True

class BitCheckerApp:
    """
    A class to create a GUI application for checking if all bits are unset in a given range.
    """
    def __init__(self, master):
        self.master = master
        master.title("Bit Checker")

        self.label = tk.Label(master, text="Enter range (start and end):")
        self.label.pack()

        self.start_entry = tk.Entry(master)
        self.start_entry.pack()
        self.start_entry.insert(0, "0")

        self.end_entry = tk.Entry(master)
        self.end_entry.pack()
        self.end_entry.insert(0, "0")

        self.check_button = tk.Button(master, text="Check Bits", command=self.check_bits)
        self.check_button.pack()

    def check_bits(self):
        try:
            start = int(self.start_entry.get())
            end = int(self.end_entry.get())
            if start > end:
                messagebox.showerror("Error", "Start should be less than or equal to End.")
                return

            result = are_all_bits_unset(start, end)
            if result:
                messagebox.showinfo("Result", "All bits are unset in the given range.")
            else:
                messagebox.showinfo("Result", "Not all bits are unset in the given range.")
        except ValueError:
            messagebox.showerror("Error", "Please enter valid integers for start and end.")

if __name__ == "__main__":
    root = tk.Tk()
    app = BitCheckerApp(root)
    root.mainloop()

    """

'''
This script creates a GUI application using Tkinter to check if all bits are unset in the binary representation of numbers in a given range.
The function `are_all_bits_unset` checks if all bits in the binary representation of numbers in the specified range are unset.
'''
import tkinter as tk
from tkinter import messagebox
def are_all_bits_unset(start, end):
    """
    Check if all bits in the binary representation of numbers in the given range are unset.
    Parameters:
    start (int): The starting integer of the range.
    end (int): The ending integer of the range.
    Returns:
    bool: True if all bits are unset for all numbers, False otherwise.
    """
    return start == 0 and end == 0
class BitCheckerApp:
    """
    A class to create a GUI application for checking if all bits in the binary representation of numbers in a given range are unset.
    """
    def __init__(self, master):
        self.master = master
        master.title("Bit Checker")
        self.label = tk.Label(master, text="Enter range (start and end):")
        self.label.pack()
        self.start_entry = tk.Entry(master)
        self.start_entry.pack()
        self.start_entry.insert(0, "0")
        self.end_entry = tk.Entry(master)
        self.end_entry.pack()
        self.end_entry.insert(0, "0")
        self.check_button = tk.Button(master, text="Check Bits", command=self.check_bits)
        self.check_button.pack()
    def check_bits(self):
        try:
            start = int(self.start_entry.get())
            end = int(self.end_entry.get())
            if start > end:
                messagebox.showerror("Error", "Start should be less than or equal to End.")
                return
            result = are_all_bits_unset(start, end)
            if result:
                messagebox.showinfo("Result", "All bits are unset in the given range.")
            else:
                messagebox.showinfo("Result", "Not all bits are unset in the given range.")
        except ValueError:
            messagebox.showerror("Error", "Please enter valid integers for start and end.")
if __name__ == "__main__":
    root = tk.Tk()
    app = BitCheckerApp(root)
    root.mainloop()
