This is the main file for the Power of Two application. It initializes the GUI and handles user interactions.

'''

import tkinter as tk
from power_of_two import highest_power_of_two

class PowerOfTwoApp:
    def __init__(self, master):
        self.master = master
        master.title("Highest Power of 2 Finder")

        self.label = tk.Label(master, text="Enter a number:")
        self.label.pack()

        self.entry = tk.Entry(master)
        self.entry.pack()

        self.find_button = tk.Button(master, text="Find", command=self.find_highest_power_of_two)
        self.find_button.pack()

        self.result_label = tk.Label(master, text="")
        self.result_label.pack()

    def find_highest_power_of_two(self):
        try:
            n = int(self.entry.get())
            result = highest_power_of_two(n)
            self.display_result(result)
        except ValueError:
            self.result_label.config(text="Please enter a valid integer.")

    def display_result(self, result):
        self.result_label.config(text=f"The highest power of 2 ≤ n is: {result}")

if __name__ == "__main__":
    root = tk.Tk()
    app = PowerOfTwoApp(root)
    root.mainloop()
```

#### power_of_two.py

```python
'''

This module contains the function to calculate the highest power of 2 less than or equal to a given number.

'''

def highest_power_of_two(n):
    '''
    Calculate the highest power of 2 less than or equal to n.

    :param n: The number to compare against.
    :return: The highest power of 2 less than or equal to n.
    '''
    if n < 1:
        return 0
    power = 1
    while power * 2 <= n:
        power *= 2
    return power
```

'''

This is the main file for the Power of Two application. It initializes the GUI and handles user interactions.

'''

import tkinter as tk
from power_of_two import highest_power_of_two

class PowerOfTwoApp:
    def __init__(self, master):
        self.master = master
        master.title("Highest Power of 2 Finder")

        self.label = tk.Label(master, text="Enter a number:")
        self.label.pack()

        self.entry = tk.Entry(master)
        self.entry.pack()

        self.find_button = tk.Button(master, text="Find", command=self.find_highest_power_of_two)
        self.find_button.pack()

        self.result_label = tk.Label(master, text="")
        self.result_label.pack()

    def find_highest_power_of_two(self):
        try:
            n = int(self.entry.get())
            result = highest_power_of_two(n)
            self.display_result(result)
        except ValueError:
            self.result_label.config(text="Please enter a valid integer.")

    def display_result(self, result):
        self.result_label.config(text=f"The highest power of 2 ≤ n is: {result}")

if __name__ == "__main__":
    root = tk.Tk()
    app = PowerOfTwoApp(root)
    root.mainloop()
```

'''

This module contains the function to calculate the highest power of 2 less than or equal to a given number.

'''

def highest_power_of_two(n):
    '''
    Calculate the highest power of 2 less than or equal to n.

    :param n: The number to compare against.
    :return: The highest power of 2 less than or equal to n.
    '''
    if n < 1:
        return 0
    power = 1
    while power * 2 <= n:
        power *= 2
    return power
```

def highest_power_of_two(n):
    '''
    Calculate the highest power of 2 less than or equal to n.
    :param n: The number to compare against.
    :return: The highest power of 2 less than or equal to n.
    '''
    if n < 1:

    power = 1
    while power * 2 <= n:

    return power

'''
This module contains the function to calculate the highest power of 2 less than or equal to a given number.
'''
def highest_power_of_two(n):
    '''
    Calculate the highest power of 2 less than or equal to n.
    :param n: The number to compare against.
    :return: The highest power of 2 less than or equal to n.
    '''
    if n < 1:
        return 0  # Return 0 for inputs less than 1, as no valid power of 2 exists.
    power = 1
    while power * 2 <= n:
        power *= 2
    return power

'''
This module contains the function to calculate the highest power of 2 less than or equal to a given number.
'''
def highest_power_of_two(n):
    '''
    Calculate the highest power of 2 less than or equal to n.
    :param n: The number to compare against.
    :return: The highest power of 2 less than or equal to n.
    '''
    if n < 1:
        return 0  # Return 0 for inputs less than 1, as no valid power of 2 exists.
    power = 1
    while power * 2 <= n:
        power *= 2
    return power

def highest_power_of_two(n):
    '''
    Calculate the highest power of 2 less than or equal to n.
    :param n: The number to compare against.
    :return: The highest power of 2 less than or equal to n.
    '''
    if n < 1:
        return 0  # Return 0 for inputs less than 1, as no valid power of 2 exists.
    power = 1
    while power * 2 <= n:
        power *= 2
    return power

'''
This module contains the function to calculate the highest power of 2 less than or equal to a given number.
'''
def highest_power_of_two(n):
    '''
    Calculate the highest power of 2 less than or equal to n.
    :param n: The number to compare against.
    :return: The highest power of 2 less than or equal to n.
    '''
    if n < 1:
        return 0  # Return 0 for inputs less than 1, as no valid power of 2 exists.
    power = 1
    while power * 2 <= n:
        power *= 2
    return power
