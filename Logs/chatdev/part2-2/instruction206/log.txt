'''

This module contains the EvenCheckerApp class and the is_even function to create a GUI application that checks if a number is even.

'''

import tkinter as tk

def is_even(number):
    '''
    Determines if the given number is even.

    Parameters:
    number (int): The number to check.

    Returns:
    bool: True if the number is even, False otherwise.
    '''
    return number % 2 == 0

class EvenCheckerApp:
    '''
    A class to create a GUI application for checking if a number is even.
    '''
    def __init__(self, root):
        '''
        Initializes the EvenCheckerApp with GUI components.

        Parameters:
        root (tk.Tk): The root window of the application.
        '''
        self.root = root
        self.root.title("Even Number Checker")

        self.label = tk.Label(root, text="Enter a number:")
        self.label.pack()

        self.entry = tk.Entry(root)
        self.entry.pack()

        self.check_button = tk.Button(root, text="Check", command=self.check_even)
        self.check_button.pack()

        self.result_label = tk.Label(root, text="")
        self.result_label.pack()

    def check_even(self):
        '''
        Checks if the entered number is even and updates the result label.
        '''
        try:
            number = int(self.entry.get())
            if is_even(number):
                self.result_label.config(text=f"{number} is even.")
            else:
                self.result_label.config(text=f"{number} is odd.")
        except ValueError:
            self.result_label.config(text="Please enter a valid integer.")

if __name__ == "__main__":
    root = tk.Tk()
    app = EvenCheckerApp(root)
    root.mainloop()

```
