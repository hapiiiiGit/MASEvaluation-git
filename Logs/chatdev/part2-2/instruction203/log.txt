'''

Function to find the kth element from two sorted arrays.

'''

def find_kth_element(arr1, arr2, k):
    """
    Find the kth element from two sorted arrays.
    
    :param arr1: List[int] - First sorted array
    :param arr2: List[int] - Second sorted array
    :param k: int - The kth position to find
    :return: int - The kth element
    """
    # Ensure arr1 is the smaller array
    if len(arr1) > len(arr2):
        return find_kth_element(arr2, arr1, k)
    
    # Edge cases
    if not arr1:

def find_kth_element(arr1, arr2, k):
    """
    Find the kth element from two sorted arrays.
    :param arr1: List[int] - First sorted array
    :param arr2: List[int] - Second sorted array
    :param k: int - The kth position to find
    :return: int - The kth element
    """
    # Ensure arr1 is the smaller array
    if len(arr1) > len(arr2):
        return find_kth_element(arr2, arr1, k)
    # Edge cases
    if not arr1:
        return arr2[k-1]
    if k == 1:
        return min(arr1[0], arr2[0])
    # Divide k into two parts
    i = min(len(arr1), k // 2)
    j = min(len(arr2), k // 2)
    if arr1[i-1] > arr2[j-1]:
        return find_kth_element(arr1, arr2[j:], k-j)
    else:
        return find_kth_element(arr1[i:], arr2, k-i)

'''
Function to find the kth element from two sorted arrays.
'''
def find_kth_element(arr1, arr2, k):
    """
    Find the kth element from two sorted arrays.
    :param arr1: List[int] - First sorted array
    :param arr2: List[int] - Second sorted array
    :param k: int - The kth position to find
    :return: int - The kth element
    """
    # Check if k is within the valid range
    if k < 1 or k > len(arr1) + len(arr2):
        raise ValueError("k is out of the valid range.")
    # Ensure arr1 is the smaller array
    if len(arr1) > len(arr2):
        return find_kth_element(arr2, arr1, k)
    # Edge cases
    if not arr1:
        return arr2[k-1]
    if k == 1:
        return min(arr1[0], arr2[0])
    # Divide k into two parts
    i = min(len(arr1), k // 2)
    j = min(len(arr2), k // 2)
    if arr1[i-1] > arr2[j-1]:
        return find_kth_element(arr1, arr2[j:], k-j)
    else:
        return find_kth_element(arr1[i:], arr2, k-i)

def find_kth_element(arr1, arr2, k):
    """
    Find the kth element from two sorted arrays.
    :param arr1: List[int] - First sorted array
    :param arr2: List[int] - Second sorted array
    :param k: int - The kth position to find
    :return: int - The kth element
    """
    # Check if k is within the valid range
    if k < 1 or k > len(arr1) + len(arr2):
        raise ValueError("k is out of the valid range.")
    # Ensure arr1 is the smaller array
    if len(arr1) > len(arr2):
        return find_kth_element(arr2, arr1, k)
    # Edge cases
    if not arr1:
        return arr2[k-1]
    if k == 1:
        return min(arr1[0], arr2[0])
    # Divide k into two parts
    i = min(len(arr1), k // 2)
    j = min(len(arr2), k // 2)
    if arr1[i-1] > arr2[j-1]:
        return find_kth_element(arr1, arr2[j:], k-j)
    else:
        return find_kth_element(arr1[i:], arr2, k-i)

def find_kth_element(arr1, arr2, k):
    """
    Find the kth element from two sorted arrays.
    :param arr1: List[int] - First sorted array
    :param arr2: List[int] - Second sorted array
    :param k: int - The kth position to find
    :return: int - The kth element
    """
    # Check if k is within the valid range
    if k < 1 or k > len(arr1) + len(arr2):
        raise ValueError("k is out of the valid range.")
    # Ensure arr1 is the smaller array
    if len(arr1) > len(arr2):
        return find_kth_element(arr2, arr1, k)
    # Edge cases
    if not arr1:
        return arr2[k-1]
    if k == 1:
        return min(arr1[0], arr2[0])
    # Divide k into two parts
    i = min(len(arr1), k // 2)
    j = min(len(arr2), k // 2)
    if arr1[i-1] > arr2[j-1]:
        return find_kth_element(arr1, arr2[j:], k-j)
    else:
        return find_kth_element(arr1[i:], arr2, k-i)

'''
Function to find the kth element from two sorted arrays.
'''
def find_kth_element(arr1, arr2, k):
    """
    Find the kth element from two sorted arrays.
    :param arr1: List[int] - First sorted array
    :param arr2: List[int] - Second sorted array
    :param k: int - The kth position to find
    :return: int - The kth element
    """
    # Check if k is within the valid range
    if k < 1 or k > len(arr1) + len(arr2):
        raise ValueError("k is out of the valid range.")
    # Ensure arr1 is the smaller array
    if len(arr1) > len(arr2):
        return find_kth_element(arr2, arr1, k)
    # Edge cases
    if not arr1:
        return arr2[k-1]
    if k == 1:
        return min(arr1[0], arr2[0])
    # Divide k into two parts
    i = min(len(arr1), k // 2)
    j = min(len(arr2), k // 2)
    if arr1[i-1] > arr2[j-1]:
        return find_kth_element(arr1, arr2[j:], k-j)
    else:
        return find_kth_element(arr1[i:], arr2, k-i)

'''
Function to find the kth element from two sorted arrays.
'''
def find_kth_element(arr1, arr2, k):
    """
    Find the kth element from two sorted arrays.
    :param arr1: List[int] - First sorted array
    :param arr2: List[int] - Second sorted array
    :param k: int - The kth position to find
    :return: int - The kth element
    """
    # Check if k is within the valid range
    if k < 1 or k > len(arr1) + len(arr2):
        raise ValueError("k is out of the valid range.")
    # Ensure arr1 is the smaller array
    if len(arr1) > len(arr2):
        return find_kth_element(arr2, arr1, k)
    # Edge cases
    if not arr1:
        return arr2[k-1]
    if k == 1:
        return min(arr1[0], arr2[0])
    # Divide k into two parts
    i = min(len(arr1), k // 2)
    j = min(len(arr2), k // 2)
    if arr1[i-1] > arr2[j-1]:
        return find_kth_element(arr1, arr2[j:], k-j)
    else:
        return find_kth_element(arr1[i:], arr2, k-i)
