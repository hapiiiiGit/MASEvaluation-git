'''

Main application file for the TupleListApp. This file sets up the GUI and handles user interactions.

'''

import tkinter as tk
from tkinter import messagebox
from utils import add_tuple

class TupleListApp:
    def __init__(self, root):
        self.root = root
        self.root.title("Tuple to List Adder")

        # Initialize list and tuple
        self.my_list = []
        self.my_tuple = ()

        # Create GUI elements
        self.list_label = tk.Label(root, text="Current List: []")
        self.list_label.pack()

        self.tuple_entry = tk.Entry(root)
        self.tuple_entry.pack()
        self.tuple_entry.insert(0, "Enter tuple elements separated by commas")

        self.add_button = tk.Button(root, text="Add Tuple to List", command=self.add_tuple_to_list)
        self.add_button.pack()

    def add_tuple_to_list(self):
        try:
            # Convert entry to tuple
            tuple_elements = self.tuple_entry.get().split(',')
            self.my_tuple = tuple(map(str.strip, tuple_elements))
            
            # Add tuple to list
            add_tuple(self.my_list, self.my_tuple)

            # Update list label
            self.list_label.config(text=f"Current List: {self.my_list}")

            # Clear entry
            self.tuple_entry.delete(0, tk.END)
            messagebox.showinfo("Success", "Tuple added to list successfully!")
        except Exception as e:
            messagebox.showerror("Error", f"An error occurred: {e}")

if __name__ == "__main__":
    root = tk.Tk()
    app = TupleListApp(root)
    root.mainloop()
```

'''

Main application file for the TupleListApp. This file sets up the GUI and handles user interactions.

'''

import tkinter as tk
from tkinter import messagebox
from utils import add_tuple

class TupleListApp:
    def __init__(self, root):
        self.root = root
        self.root.title("Tuple to List Adder")

        # Initialize list and tuple
        self.my_list = []
        self.my_tuple = ()

        # Create GUI elements
        self.list_label = tk.Label(root, text="Current List: []")
        self.list_label.pack()

        self.tuple_entry = tk.Entry(root)
        self.tuple_entry.pack()
        self.tuple_entry.insert(0, "Enter tuple elements separated by commas")

        self.add_button = tk.Button(root, text="Add Tuple to List", command=self.add_tuple_to_list)
        self.add_button.pack()

    def add_tuple_to_list(self):
        try:
            # Convert entry to tuple
            tuple_elements = self.tuple_entry.get().split(',')
            self.my_tuple = tuple(map(str.strip, tuple_elements))
            
            # Add tuple to list
            add_tuple(self.my_list, self.my_tuple)

            # Update list label
            self.list_label.config(text=f"Current List: {self.my_list}")

            # Clear entry
            self.tuple_entry.delete(0, tk.END)
            messagebox.showinfo("Success", "Tuple added to list successfully!")
        except Exception as e:
            messagebox.showerror("Error", f"An error occurred: {e}")

if __name__ == "__main__":
    root = tk.Tk()
    app = TupleListApp(root)
    root.mainloop()

'''
Main application file for the TupleListApp. This file sets up the GUI and handles user interactions.

import tkinter as tk
from tkinter import messagebox
from utils import add_tuple
class TupleListApp:
    def __init__(self, root):
        self.root = root
        self.root.title("Tuple to List Adder")
        # Initialize list and tuple
        self.my_list = []
        self.my_tuple = ()
        # Create GUI elements
        self.list_label = tk.Label(root, text="Current List: []")
        self.list_label.pack()
        self.tuple_format_label = tk.Label(root, text="Enter tuple elements separated by commas (e.g., 'element1, element2')")
        self.tuple_format_label.pack()
        self.tuple_entry = tk.Entry(root)
        self.tuple_entry.pack()
        self.tuple_entry.insert(0, "Enter tuple elements separated by commas")
        self.add_button = tk.Button(root, text="Add Tuple to List", command=self.add_tuple_to_list)
        self.add_button.pack()
    def add_tuple_to_list(self):
        try:
            # Convert entry to tuple
            tuple_elements = self.tuple_entry.get().split(',')
            if not tuple_elements or any(not elem.strip() for elem in tuple_elements):
                raise ValueError("Invalid tuple format. Please enter elements separated by commas.")
            self.my_tuple = tuple(map(str.strip, tuple_elements))
            # Add tuple to list
            add_tuple(self.my_list, self.my_tuple)
            # Update list label
            self.list_label.config(text=f"Current List: {self.my_list}")
            # Clear entry
            self.tuple_entry.delete(0, tk.END)
            messagebox.showinfo("Success", "Tuple added to list successfully!")
        except ValueError as ve:
            messagebox.showerror("Input Error", str(ve))
        except Exception as e:
            messagebox.showerror("Error", f"An error occurred: {e}")
if __name__ == "__main__":
    root = tk.Tk()
    app = TupleListApp(root)
    root.mainloop()

'''

import tkinter as tk
from tkinter import messagebox
from utils import add_tuple
class TupleListApp:
    def __init__(self, root):
        self.root = root
        self.root.title("Tuple to List Adder")
        # Initialize list and tuple
        self.my_list = []
        self.my_tuple = ()
        # Create GUI elements
        self.list_label = tk.Label(root, text="Current List: []")
        self.list_label.pack()
        self.tuple_format_label = tk.Label(root, text="Enter tuple elements separated by commas (e.g., 'element1, element2')")
        self.tuple_format_label.pack()
        self.tuple_entry = tk.Entry(root)
        self.tuple_entry.pack()
        self.tuple_entry.insert(0, "Enter tuple elements separated by commas")
        self.add_button = tk.Button(root, text="Add Tuple to List", command=self.add_tuple_to_list)
        self.add_button.pack()
    def add_tuple_to_list(self):
        try:
            # Convert entry to tuple
            tuple_elements = self.tuple_entry.get().split(',')
            if not tuple_elements or any(not elem.strip() for elem in tuple_elements):
                raise ValueError("Invalid tuple format. Please enter elements separated by commas.")
            self.my_tuple = tuple(map(str.strip, tuple_elements))
            # Add tuple to list
            add_tuple(self.my_list, self.my_tuple)
            # Update list label
            self.list_label.config(text=f"Current List: {self.my_list}")
            # Clear entry
            self.tuple_entry.delete(0, tk.END)
            messagebox.showinfo("Success", "Tuple added to list successfully!")
        except ValueError as ve:
            messagebox.showerror("Input Error", str(ve))
        except Exception as e:
            messagebox.showerror("Error", f"An error occurred: {e}")
if __name__ == "__main__":
    root = tk.Tk()
    app = TupleListApp(root)
    root.mainloop()
