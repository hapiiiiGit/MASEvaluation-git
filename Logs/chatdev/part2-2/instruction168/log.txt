#### utils.py

```python
'''

Utility functions for the application.

'''

def sum_of_per_digit_difference(a, b):
    """
    Calculate the sum of per-digit differences between two integers.
    
    Parameters:
    a (int): First integer.
    b (int): Second integer.
    
    Returns:
    int: Sum of per-digit differences.
    """
    # Convert integers to strings to compare digits
    str_a = str(a)
    str_b = str(b)

    # Pad the shorter string with zeros
    max_length = max(len(str_a), len(str_b))
    str_a = str_a.zfill(max_length)
    str_b = str_b.zfill(max_length)

    # Calculate the sum of per-digit differences
    total_difference = 0
    for digit_a, digit_b in zip(str_a, str_b):
        total_difference += abs(int(digit_a) - int(digit_b))

    return total_difference

def sum_of_per_digit_difference(a, b):
    """
    Calculate the sum of per-digit differences between two integers.
    
    Parameters:
    a (int): First integer.
    b (int): Second integer.
    
    Returns:
    int: Sum of per-digit differences.
    """
    # Convert integers to strings to compare digits
    str_a = str(a)
    str_b = str(b)

    # Pad the shorter string with zeros
    max_length = max(len(str_a), len(str_b))
    str_a = str_a.zfill(max_length)
    str_b = str_b.zfill(max_length)

    # Calculate the sum of per-digit differences
    total_difference = 0
    for digit_a, digit_b in zip(str_a, str_b):
        total_difference += abs(int(digit_a) - int(digit_b))

    return total_difference

def sum_of_per_digit_difference(a, b):
    """
    Calculate the sum of per-digit differences between two integers.
    Parameters:
    a (int): First integer.
    b (int): Second integer.
    Returns:
    int: Sum of per-digit differences.
    """
    # Convert integers to strings to compare digits
    str_a = str(a)
    str_b = str(b)
    # Pad the shorter string with zeros
    max_length = max(len(str_a), len(str_b))
    str_a = str_a.zfill(max_length)
    str_b = str_b.zfill(max_length)
    # Calculate the sum of per-digit differences
    total_difference = 0
    for digit_a, digit_b in zip(str_a, str_b):
        total_difference += abs(int(digit_a) - int(digit_b))
    return total_difference

def sum_of_per_digit_difference(a, b):
    """
    Calculate the sum of per-digit differences between two integers.
    Parameters:
    a (int): First integer.
    b (int): Second integer.
    Returns:
    int: Sum of per-digit differences.
    """
    # Convert integers to strings to compare digits
    str_a = str(a)
    str_b = str(b)
    # Pad the shorter string with zeros
    max_length = max(len(str_a), len(str_b))
    str_a = str_a.zfill(max_length)
    str_b = str_b.zfill(max_length)
    # Calculate the sum of per-digit differences
    total_difference = 0
    for digit_a, digit_b in zip(str_a, str_b):
        total_difference += abs(int(digit_a) - int(digit_b))
    return total_difference

def sum_of_per_digit_difference(a, b):
    """
    Calculate the sum of per-digit differences between two integers.
    Parameters:
    a (int): First integer.
    b (int): Second integer.
    Returns:
    int: Sum of per-digit differences.
    """
    # Convert integers to strings to compare digits
    str_a = str(a)
    str_b = str(b)
    # Pad the shorter string with zeros
    max_length = max(len(str_a), len(str_b))
    str_a = str_a.zfill(max_length)
    str_b = str_b.zfill(max_length)
    # Calculate the sum of per-digit differences
    total_difference = 0
    for digit_a, digit_b in zip(str_a, str_b):
        total_difference += abs(int(digit_a) - int(digit_b))
    return total_difference

'''
Utility functions for the application.

def sum_of_per_digit_difference(a, b):
    """
    Calculate the sum of per-digit differences between two integers.
    Parameters:
    a (int): First integer.
    b (int): Second integer.
    Returns:
    int: Sum of per-digit differences.
    """
    # Convert integers to strings to compare digits
    str_a = str(a)
    str_b = str(b)
    # Pad the shorter string with zeros
    max_length = max(len(str_a), len(str_b))
    str_a = str_a.zfill(max_length)
    str_b = str_b.zfill(max_length)
    # Calculate the sum of per-digit differences
    total_difference = 0
    for digit_a, digit_b in zip(str_a, str_b):
        total_difference += abs(int(digit_a) - int(digit_b))
    return total_difference

# Per-Digit Difference Calculator

A simple Python application with a graphical user interface (GUI) to calculate the sum of per-digit differences between two integers.

## Introduction

The **Per-Digit Difference Calculator** is a user-friendly application designed to compute the sum of the absolute differences between corresponding digits of two integers. This tool is useful for mathematical explorations, educational purposes, or any scenario requiring digit-by-digit comparison.

### Main Functions

1. **Sum of Per-Digit Differences**: 
   - Calculates the sum of absolute differences between corresponding digits of two integers.
   - Handles integers of different lengths by padding the shorter number with leading zeros.

2. **Graphical User Interface (GUI)**:
   - Allows users to input two integers.
   - Displays the calculated result or an error message if invalid inputs are provided.
