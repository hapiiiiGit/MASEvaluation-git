'''

This file contains the main entry point for the application. It sets up the graphical user interface (GUI) using tkinter and integrates the harmonic sum calculation functionality.

'''

import tkinter as tk
from harmonic_calculator import harmonic_sum

class HarmonicSumApp:
    def __init__(self, root):
        self.root = root
        self.root.title("Harmonic Sum Calculator")
        
        # Input Label
        self.label = tk.Label(root, text="Enter an integer (n):")
        self.label.pack()
        
        # Input Entry
        self.entry = tk.Entry(root)
        self.entry.pack()
        
        # Calculate Button
        self.calculate_button = tk.Button(root, text="Calculate Harmonic Sum", command=self.calculate)
        self.calculate_button.pack()
        
        # Result Label
        self.result_label = tk.Label(root, text="")
        self.result_label.pack()
    
    def calculate(self):
        try:
            n = int(self.entry.get())
            if n <= 1:
                self.result_label.config(text="Please enter an integer greater than 1.")
            else:
                result = harmonic_sum(n)
                self.result_label.config(text=f"Harmonic Sum of {n-1}: {result}")
        except ValueError:
            self.result_label.config(text="Invalid input. Please enter a valid integer.")

if __name__ == "__main__":
    root = tk.Tk()
    app = HarmonicSumApp(root)
    root.mainloop()

import tkinter as tk

class HarmonicSumApp:
    def __init__(self, root):
        """
        Initialize the Harmonic Sum Calculator application.
        Args:
            root (tk.Tk): The root window for the application.
        """
        self.root = root
        self.root.title("Harmonic Sum Calculator")
        # Input Label
        self.label = tk.Label(root, text="Enter an integer (n):")
        self.label.pack(pady=5)
        # Input Entry
        self.entry = tk.Entry(root)
        self.entry.pack(pady=5)
        # Calculate Button
        self.calculate_button = tk.Button(root, text="Calculate Harmonic Sum", command=self.calculate)
        self.calculate_button.pack(pady=5)
        # Result Label
        self.result_label = tk.Label(root, text="", fg="blue")
        self.result_label.pack(pady=5)
    def calculate(self):
        """
        Handle the calculation of the harmonic sum based on user input.
        """
        try:
            n = int(self.entry.get())

            self.result_label.config(text="Invalid input. Please enter a valid integer.", fg="red")
if __name__ == "__main__":

    root.mainloop()

def harmonic_sum(n):
    """
    Calculate the harmonic sum of n-1.
    Args:
        n (int): The input integer.
    Returns:
        float: The harmonic sum of n-1.
    """
    if n <= 1:
        raise ValueError("Input must be greater than 1.")
    harmonic_sum = 0
    for i in range(1, n):
        harmonic_sum += 1 / i
    return harmonic_sum

def harmonic_sum(n):
    """
    Calculate the harmonic sum of n-1.
    Args:
        n (int): The input integer.
    Returns:
        float: The harmonic sum of n-1.
    """
    if n <= 1:
        raise ValueError("Input must be greater than 1.")
    harmonic_sum = 0
    for i in range(1, n):
        harmonic_sum += 1 / i
    return harmonic_sum

```

This application provides a clean and intuitive interface for users to input their desired integer and view the calculated harmonic sum instantly.
