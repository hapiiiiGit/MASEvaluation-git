        # Entry fields
        self.entry1 = tk.Entry(root, width=50)

        '''
        Handles the logic to replace the last element of the first list with the elements of the second list.
        '''
        try:
            # Get input from entry fields
            list1 = self.entry1.get().split(',')
            list2 = self.entry2.get().split(',')
            # Strip whitespace from elements
            list1 = [item.strip() for item in list1]
            list2 = [item.strip() for item in list2]
            # Validate input
            if not list1:
                raise ValueError("The first list cannot be empty.")
            # Call the logic function
            result = replace_last_element_in_list(list1, list2)
            # Display the result
            self.result_label.config(text=f"Result: {result}")
        except ValueError as ve:
            messagebox.showerror("Input Error", str(ve))
        except Exception as e:
            messagebox.showerror("Error", f"An error occurred: {e}")
    def clear_inputs(self):
        '''
        Clears the input fields and result label.
        '''
        self.entry1.delete(0, tk.END)
        self.entry2.delete(0, tk.END)
        self.result_label.config(text="Result: ")
if __name__ == "__main__":
    root = tk.Tk()
    app = ListReplacerApp(root)
    root.mainloop()

def replace_last_element_in_list(list1, list2):
    '''
    Replaces the last element of the first list with the elements of the second list.
    Args:
        list1 (list): The first list.
        list2 (list): The second list.

class ListReplacerApp:
    def __init__(self, root):
        '''
        Initializes the GUI application.
        '''
        self.root = root
        self.root.title("List Replacer Tool")

        self.label1 = tk.Label(root, text="Enter First List (comma-separated):")

        '''
        Clears the input fields and result label.
        '''
        self.entry1.delete(0, tk.END)
        self.entry2.delete(0, tk.END)
        self.result_label.config(text="Result: ")
if __name__ == "__main__":
