import tkinter as tk
from tkinter import messagebox
from find_occurrence import find_first_occurrence

class MainApplication:
    def __init__(self, root):
        self.root = root
        self.root.title("Find First Occurrence in Sorted Array")

        self.label_array = tk.Label(root, text="Enter sorted array (comma-separated):")
        self.label_array.pack()

        self.entry_array = tk.Entry(root, width=50)
        self.entry_array.pack()

        self.label_target = tk.Label(root, text="Enter target number:")
        self.label_target.pack()

        self.entry_target = tk.Entry(root, width=50)
        self.entry_target.pack()

        self.button_find = tk.Button(root, text="Find Index", command=self.find_index)
        self.button_find.pack()

    def find_index(self):
        try:
            array_str = self.entry_array.get()
            array = list(map(int, array_str.split(',')))
            target = int(self.entry_target.get())

            index = find_first_occurrence(array, target)
            if index != -1:
                messagebox.showinfo("Result", f"First occurrence of {target} is at index {index}.")
            else:
                messagebox.showinfo("Result", f"{target} not found in the array.")
        except ValueError:
            messagebox.showerror("Error", "Invalid input. Please enter integers only.")

if __name__ == "__main__":
    root = tk.Tk()
    app = MainApplication(root)
    root.mainloop()

#### find_occurrence.py

```python
'''

Function to find the index of the first occurrence of a given number in a sorted array.

'''

def find_first_occurrence(array, target):
    """
    Finds the index of the first occurrence of target in a sorted array.
    
    Parameters:
    array (list of int): The sorted array to search.
    target (int): The number to find in the array.
    
    Returns:
    int: The index of the first occurrence of target, or -1 if not found.
    """
    for index, value in enumerate(array):
        if value == target:
            return index
    return -1
```

def find_first_occurrence(array, target):
    """
    Finds the index of the first occurrence of target in a sorted array.
    
    Parameters:
    array (list of int): The sorted array to search.
    target (int): The number to find in the array.
    
    Returns:

def find_first_occurrence(array, target):
    """
    Finds the index of the first occurrence of target in a sorted array using binary search.
    Parameters:
    array (list of int): The sorted array to search.
    target (int): The number to find in the array.
    Returns:
    int: The index of the first occurrence of target, or -1 if not found.
    """
    left, right = 0, len(array) - 1
    result = -1
    while left <= right:
        mid = left + (right - left) // 2
        if array[mid] == target:
            result = mid
            right = mid - 1  # Continue to search in the left half
        elif array[mid] < target:
            left = mid + 1
        else:
            right = mid - 1
    return result

def find_first_occurrence(array, target):
    """
    Finds the index of the first occurrence of target in a sorted array.
    Parameters:
    array (list of int): The sorted array to search.
    target (int): The number to find in the array.

    for index, value in enumerate(array):

    return -1

'''
Function to find the index of the first occurrence of a given number in a sorted array using binary search.
'''
def find_first_occurrence(array, target):
    """
    Finds the index of the first occurrence of target in a sorted array using binary search.
    Parameters:
    array (list of int): The sorted array to search.
    target (int): The number to find in the array.
    Returns:
    int: The index of the first occurrence of target, or -1 if not found.
    """
    left, right = 0, len(array) - 1
    result = -1
    while left <= right:
        mid = left + (right - left) // 2
        if array[mid] == target:
            result = mid
            right = mid - 1  # Continue to search in the left half
        elif array[mid] < target:
            left = mid + 1
        else:
            right = mid - 1
    return result

Function to find the index of the first occurrence of a given number in a sorted array using binary search.
'''
def find_first_occurrence(array, target):
    """
    Finds the index of the first occurrence of target in a sorted array using binary search.
    Parameters:
    array (list of int): The sorted array to search.
    target (int): The number to find in the array.
    Returns:
    int: The index of the first occurrence of target, or -1 if not found.
    """
    left, right = 0, len(array) - 1
    result = -1
    while left <= right:
        mid = left + (right - left) // 2
        if array[mid] == target:
            result = mid
            right = mid - 1  # Continue to search in the left half
        elif array[mid] < target:
            left = mid + 1
        else:
            right = mid - 1
    return result
