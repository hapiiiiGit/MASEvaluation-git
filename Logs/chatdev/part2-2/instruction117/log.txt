'''

Main application file for tuple extraction GUI.

'''

import tkinter as tk
from tkinter import messagebox
from tuple_operations import TupleOperations

class MainApplication:
    def __init__(self, root):
        self.root = root
        self.root.title("Tuple Extraction Application")
        
        self.label1 = tk.Label(root, text="Enter first tuple (comma-separated):")
        self.label1.pack()
        
        self.entry1 = tk.Entry(root)
        self.entry1.pack()
        
        self.label2 = tk.Label(root, text="Enter second tuple (comma-separated):")
        self.label2.pack()
        
        self.entry2 = tk.Entry(root)
        self.entry2.pack()
        
        self.submit_button = tk.Button(root, text="Submit", command=self.on_submit)
        self.submit_button.pack()
        
        self.result_label = tk.Label(root, text="")
        self.result_label.pack()
        
    def on_submit(self):
        tuple1_str = self.entry1.get()
        tuple2_str = self.entry2.get()
        
        try:
            tuple1 = tuple(map(int, tuple1_str.split(',')))
            tuple2 = tuple(map(int, tuple2_str.split(',')))
            
            operations = TupleOperations()
            elementwise_sum = operations.elementwise_sum(tuple1, tuple2)

class TupleOperations:
    def elementwise_sum(self, tuple1, tuple2):
        '''
        Calculate the elementwise sum of two tuples, handling different lengths by padding with zeros.
        '''
        max_length = max(len(tuple1), len(tuple2))
        tuple1 = tuple1 + (0,) * (max_length - len(tuple1))
        tuple2 = tuple2 + (0,) * (max_length - len(tuple2))
        return tuple(a + b for a, b in zip(tuple1, tuple2))
    def extract_tuples(self, tuple1, tuple2):
        '''
        Extract tuples from the given two tuples, handling different lengths by padding with None.
        '''
        max_length = max(len(tuple1), len(tuple2))
        tuple1 = tuple1 + (None,) * (max_length - len(tuple1))
        tuple2 = tuple2 + (None,) * (max_length - len(tuple2))
        return tuple((a, b) for a, b in zip(tuple1, tuple2))

class TupleOperations:
    def elementwise_sum(self, tuple1, tuple2):
        '''
        Calculate the elementwise sum of two tuples, handling different lengths by padding with zeros.

class TupleOperations:
    def elementwise_sum(self, tuple1, tuple2):
        '''
        Calculate the elementwise sum of two tuples, handling different lengths by padding with zeros.
        '''
        max_length = max(len(tuple1), len(tuple2))
        tuple1 = tuple1 + (0,) * (max_length - len(tuple1))
        tuple2 = tuple2 + (0,) * (max_length - len(tuple2))
        return tuple(a + b for a, b in zip(tuple1, tuple2))
    def extract_tuples(self, tuple1, tuple2):
        '''
        Extract tuples from the given two tuples, handling different lengths by padding with None.

        tuple1 = tuple1 + (None,) * (max_length - len(tuple1))
        tuple2 = tuple2 + (None,) * (max_length - len(tuple2))

class TupleOperations:
    def elementwise_sum(self, tuple1, tuple2):
        '''
        Calculate the elementwise sum of two tuples, handling different lengths by padding with zeros.
        '''
        max_length = max(len(tuple1), len(tuple2))
        tuple1 = tuple1 + (0,) * (max_length - len(tuple1))
        tuple2 = tuple2 + (0,) * (max_length - len(tuple2))
        return tuple(a + b for a, b in zip(tuple1, tuple2))

    def extract_tuples(self, tuple1, tuple2):
        '''
        Extract tuples from the given two tuples, handling different lengths by padding with None.
        '''
        max_length = max(len(tuple1), len(tuple2))
        tuple1 = tuple1 + (None,) * (max_length - len(tuple1))
        tuple2 = tuple2 + (None,) * (max_length - len(tuple2))
        return tuple((a, b) for a, b in zip(tuple1, tuple2))

'''
Tuple operations for elementwise sum and extraction.
'''
class TupleOperations:
    def elementwise_sum(self, tuple1, tuple2):
        '''
        Calculate the elementwise sum of two tuples, handling different lengths by padding with zeros.
        '''
        max_length = max(len(tuple1), len(tuple2))
        tuple1 = tuple1 + (0,) * (max_length - len(tuple1))
        tuple2 = tuple2 + (0,) * (max_length - len(tuple2))
        return tuple(a + b for a, b in zip(tuple1, tuple2))
    def extract_tuples(self, tuple1, tuple2):
        '''
        Extract tuples from the given two tuples, handling different lengths by padding with None.
        '''
        max_length = max(len(tuple1), len(tuple2))
        tuple1 = tuple1 + (None,) * (max_length - len(tuple1))
        tuple2 = tuple2 + (None,) * (max_length - len(tuple2))
        return tuple((a, b) for a, b in zip(tuple1, tuple2))

# Tuple Extraction Application

A Python-based GUI application for performing tuple operations, including elementwise sum and tuple extraction.

## 🤔 What is this?

The Tuple Extraction Application is a user-friendly software designed to perform operations on tuples. It allows users to input two tuples of integers and calculates:

1. **Elementwise Sum**: Adds corresponding elements of the two tuples. If the tuples have different lengths, the shorter tuple is padded with zeros.
2. **Extracted Tuples**: Combines corresponding elements of the two tuples into tuples of pairs. If the tuples have different lengths, the shorter tuple is padded with `None`.

This application is ideal for users who need to perform tuple operations quickly and efficiently without writing code manually.

---

## 🚀 Quick Install

### Prerequisites
Ensure you have Python installed on your system. The application does not require any external dependencies.

### Installation Steps
1. Clone the repository or download the source code files (`main.py`, `tuple_operations.py`, and `requirements.txt`).
2. Open a terminal and navigate to the folder containing the files.
3. Install the required dependencies (if any). Since no external dependencies are needed, you can skip this step.

---

## 📖 How to Use

### Step 1: Run the Application
1. Open a terminal and navigate to the folder containing the `main.py` file.
2. Run the application using the following command:
   ```
   python main.py
   ```

### Step 2: Input Tuples
1. Enter the first tuple in the text box labeled "Enter first tuple (comma-separated):". Example: `1,2,3`.
2. Enter the second tuple in the text box labeled "Enter second tuple (comma-separated):". Example: `4,5,6`.

### Step 3: Submit and View Results
1. Click the "Submit" button.
2. The application will validate the inputs and perform the tuple operations.
3. The results will be displayed below the "Submit" button:
   - **Elementwise Sum**: The sum of corresponding elements from the two tuples.
   - **Extracted Tuples**: Pairs of corresponding elements from the two tuples.

### Example Usage
#### Input:
- First Tuple: `1,2,3`
- Second Tuple: `4,5`

#### Output:
- Elementwise Sum: `(5, 7, 3)`
- Extracted Tuples: `((1, 4), (2, 5), (3, None))`

---

## 🛠 Features

### 1. Elementwise Sum
- Adds corresponding elements of two tuples.
- Handles tuples of different lengths by padding the shorter tuple with zeros.

### 2. Extracted Tuples
- Combines corresponding elements of two tuples into pairs.
- Handles tuples of different lengths by padding the shorter tuple with `None`.

### 3. Input Validation
- Ensures that the user inputs valid tuples of integers.
- Displays error messages for invalid inputs.
