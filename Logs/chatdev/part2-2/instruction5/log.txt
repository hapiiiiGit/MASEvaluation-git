'''

This file contains the MainApplication class which sets up the GUI and handles user interactions to count set bits in a given number.

'''

import tkinter as tk
from tkinter import messagebox

class MainApplication:
    def __init__(self, root):
        self.root = root
        self.root.title("Set Bits Counter")
        
        # Create and place widgets
        self.label = tk.Label(root, text="Enter a number:")
        self.label.pack(pady=10)
        
        self.entry = tk.Entry(root)
        self.entry.pack(pady=5)
        
        self.button = tk.Button(root, text="Calculate", command=self.on_calculate)
        self.button.pack(pady=10)
        
        self.result_label = tk.Label(root, text="")
        self.result_label.pack(pady=10)
    
    @staticmethod
    def count_set_bits(number):
        '''Counts the number of set bits (1s) in the binary representation of the given number.'''
        return bin(number).count('1')
    
    def on_calculate(self):
        '''Handles the calculation and displays the result.'''
        try:
            number = int(self.entry.get())
            count = self.count_set_bits(number)
            self.result_label.config(text=f"Number of set bits: {count}")
        except ValueError:
            messagebox.showerror("Invalid Input", "Please enter a valid integer.")

if __name__ == "__main__":
    root = tk.Tk()
    app = MainApplication(root)
    root.mainloop()

```

- **MainApplication Class**: This class sets up the GUI using `tkinter`. It includes an entry field for the user to input a number, a button to trigger the calculation, and a label to display the result.

- **on_calculate Method**: This method is called when the "Calculate" button is clicked. It retrieves the input, calculates the set bits using `count_set_bits`, and updates the result label. It also handles invalid input using a message box.

This file contains the MainApplication class which sets up the GUI and handles user interactions to count set bits in a given number.

import tkinter as tk
from tkinter import messagebox

class MainApplication:
    def __init__(self, root):
        self.root = root
        self.root.title("Set Bits Counter")
        
        # Create and place widgets
        self.label = tk.Label(root, text="Enter a number:")
        self.label.pack(pady=10)
        
        self.entry = tk.Entry(root)
        self.entry.pack(pady=5)
        
        self.button = tk.Button(root, text="Calculate", command=self.on_calculate)
        self.button.pack(pady=10)
        
        self.result_label = tk.Label(root, text="")
        self.result_label.pack(pady=10)
    
    @staticmethod
    def count_set_bits(number):
        '''Counts the number of set bits (1s) in the binary representation of the given number.'''
        return bin(number).count('1')
    
    def on_calculate(self):
        '''Handles the calculation and displays the result.'''
        try:
            number = int(self.entry.get())
            count = self.count_set_bits(number)
            self.result_label.config(text=f"Number of set bits: {count}")
        except ValueError:
            messagebox.showerror("Invalid Input", "Please enter a valid integer.")

if __name__ == "__main__":
    root = tk.Tk()
    app = MainApplication(root)
    root.mainloop()

'''
This file contains the MainApplication class which sets up the GUI and handles user interactions to count set bits in a given number.

import tkinter as tk
from tkinter import messagebox
class MainApplication:
    def __init__(self, root):
        '''
        Initializes the GUI application, setting up widgets for user interaction.
        '''
        self.root = root
        self.root.title("Set Bits Counter")
        # Create and place widgets
        self.label = tk.Label(root, text="Enter a number:")
        self.label.pack(pady=10)
        self.entry = tk.Entry(root)
        self.entry.pack(pady=5)
        self.button = tk.Button(root, text="Calculate", command=self.on_calculate)
        self.button.pack(pady=10)
        self.result_label = tk.Label(root, text="")
        self.result_label.pack(pady=10)
    @staticmethod
    def count_set_bits(number):
        '''
        Counts the number of set bits (1s) in the binary representation of the given number.
        '''
        if number < 0:
            raise ValueError("Number must be non-negative.")
        return bin(number).count('1')
    def on_calculate(self):
        '''
        Handles the calculation and displays the result.
        '''
        try:
            number = int(self.entry.get())
            if number < 0:
                raise ValueError("Number must be non-negative.")
            count = self.count_set_bits(number)
            self.result_label.config(text=f"Number of set bits: {count}")
        except ValueError as e:
            messagebox.showerror("Invalid Input", str(e))
if __name__ == "__main__":
    root = tk.Tk()
    app = MainApplication(root)
    root.mainloop()
