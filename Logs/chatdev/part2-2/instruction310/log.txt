import tkinter as tk
from tkinter import messagebox
from utils import count_lists_in_input

class MainApplication:
    def __init__(self, master):
        self.master = master
        master.title("List Counter")

        self.label = tk.Label(master, text="Enter lists separated by commas:")
        self.label.pack()

        self.entry = tk.Entry(master, width=50)
        self.entry.pack()

        self.count_button = tk.Button(master, text="Count Lists", command=self.count_lists)
        self.count_button.pack()

        self.result_label = tk.Label(master, text="")
        self.result_label.pack()

    def count_lists(self):
        input_str = self.entry.get()
        try:
            list_count = count_lists_in_input(input_str)
            self.result_label.config(text=f"Number of lists: {list_count}")
        except Exception as e:
            messagebox.showerror("Error", f"Invalid input: {e}")

if __name__ == "__main__":
    root = tk.Tk()
    app = MainApplication(root)
    root.mainloop()
```

#### utils.py

```python
'''

Utility functions for the list counting application.

'''

def count_lists_in_input(input_str):
    """
    Count the number of lists in a given string representation of lists.
    
    Parameters:
    input_str (str): A string containing multiple lists separated by commas.
    
    Returns:
    int: The number of lists in the input string.
    """
    try:
        # Evaluate the input string to convert it into a list of lists
        lists = eval(input_str)
        if isinstance(lists, list):
            return len(lists)
        else:
            raise ValueError("Input is not a list of lists.")
    except Exception as e:
        raise ValueError("Invalid input format. Please enter a valid list of lists.") from e
```

import ast

import ast
def count_lists_in_input(input_str):
    """
    Count the number of lists in a given string representation of lists.
    Parameters:
    input_str (str): A string containing multiple lists separated by semicolons.
    Returns:
    int: The number of lists in the input string.
    """
    try:
        # Split the input by semicolons to separate different lists
        list_strings = input_str.split(';')
        lists = [ast.literal_eval(f"[{lst.strip()}]") for lst in list_strings if lst.strip()]
        return len(lists)
    except Exception as e:
        raise ValueError("Invalid input format. Please enter lists separated by semicolons, e.g., '1,2; 3,4'.") from e
