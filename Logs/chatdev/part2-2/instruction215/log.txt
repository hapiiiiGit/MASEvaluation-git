import tkinter as tk
from tkinter import messagebox

def max_of_nth_column(matrix, n):
    '''
    Returns the maximum value of the nth column in the given matrix.
    
    :param matrix: List of lists representing the matrix.
    :param n: Index of the column to find the maximum value.
    :return: Maximum value in the nth column.
    '''
    try:
        # Extract the nth column
        column = [row[n] for row in matrix]
        # Return the maximum value in the column
        return max(column)
    except IndexError:
        raise ValueError("Column index out of range.")
    except Exception as e:
        raise ValueError(f"An error occurred: {e}")

    A GUI application to find the maximum value of a specified column in a matrix.

        Create and layout the widgets for the application.
        '''

        self.matrix_label.pack()

        self.matrix_entry = tk.Entry(self.root, width=50)
        self.matrix_entry.pack()

        self.column_label = tk.Label(self.root, text="Enter Column Index:")
        self.column_label.pack()

        self.column_entry = tk.Entry(self.root, width=5)
        self.column_entry.pack()

        self.calculate_button = tk.Button(self.root, text="Calculate Max", command=self.get_max_of_column)
        self.calculate_button.pack()

        self.result_label = tk.Label(self.root, text="Result: ")

import tkinter as tk
from tkinter import messagebox
def max_of_nth_column(matrix, n):
    '''
    Returns the maximum value of the nth column in the given matrix.
    :param matrix: List of lists representing the matrix.
    :param n: Index of the column to find the maximum value.
    :return: Maximum value in the nth column.
    '''
    try:
        # Extract the nth column
        column = [row[n] for row in matrix]
        # Return the maximum value in the column
        return max(column)
    except IndexError:
        raise ValueError("Column index out of range.")
    except Exception as e:
        raise ValueError(f"An error occurred: {e}")

    '''
    A GUI application to find the maximum value of a specified column in a matrix.

        self.result_label.pack()

        '''
        Calculate and display the maximum value of the specified column.
        '''
        try:
            # Parse the matrix input
            matrix_input = self.matrix_entry.get()
            matrix = [list(map(int, row.split(','))) for row in matrix_input.split(';')]
            # Parse the column index
            column_index = int(self.column_entry.get())

        except Exception as e:
            messagebox.showerror("Error", f"An unexpected error occurred: {e}")

'''
This script creates a GUI application that allows users to input a matrix and a column index,
then calculates and displays the maximum value of the specified column.
'''
import tkinter as tk
from tkinter import messagebox
def max_of_nth_column(matrix, n):
    '''
    Returns the maximum value of the nth column in the given matrix.
    :param matrix: List of lists representing the matrix.
    :param n: Index of the column to find the maximum value.
    :return: Maximum value in the nth column.
    '''
    try:
        # Extract the nth column
        column = [row[n] for row in matrix]
        # Return the maximum value in the column
        return max(column)
    except IndexError:
        raise ValueError("Column index out of range.")
    except Exception as e:
        raise ValueError(f"An error occurred: {e}")
class MatrixMaxApp:
    '''
    A GUI application to find the maximum value of a specified column in a matrix.
    '''
    def __init__(self, root):
        self.root = root
        self.root.title("Matrix Column Max Finder")
        self.create_widgets()
    def create_widgets(self):
        '''
        Create and layout the widgets for the application.
        '''
        self.matrix_label = tk.Label(self.root, text="Enter Matrix (comma-separated rows):")
        self.matrix_label.pack()
        self.matrix_entry = tk.Entry(self.root, width=50)
        self.matrix_entry.pack()
        self.column_label = tk.Label(self.root, text="Enter Column Index:")
        self.column_label.pack()
        self.column_entry = tk.Entry(self.root, width=5)
        self.column_entry.pack()
        self.calculate_button = tk.Button(self.root, text="Calculate Max", command=self.get_max_of_column)
        self.calculate_button.pack()
        self.result_label = tk.Label(self.root, text="Result: ")
        self.result_label.pack()
    def get_max_of_column(self):
        '''
        Calculate and display the maximum value of the specified column.
        '''
        try:
            # Parse the matrix input
            matrix_input = self.matrix_entry.get()
            matrix = []
            for row in matrix_input.split(';'):
                # Ensure each row contains valid integers
                row_values = row.split(',')
                if not all(value.strip().isdigit() for value in row_values):
                    raise ValueError("Matrix entries must be integers.")
                matrix.append(list(map(int, row_values)))
            # Parse the column index
            column_index = int(self.column_entry.get())
            # Calculate the max value of the specified column
            max_value = max_of_nth_column(matrix, column_index)
            # Display the result
            self.result_label.config(text=f"Result: {max_value}")
        except ValueError as ve:
            messagebox.showerror("Error", str(ve))
        except Exception as e:
            messagebox.showerror("Error", f"An unexpected error occurred: {e}")

'''
This script creates a GUI application that allows users to input a matrix and a column index,
then calculates and displays the maximum value of the specified column.
'''
import tkinter as tk
from tkinter import messagebox
def max_of_nth_column(matrix, n):
    '''
    Returns the maximum value of the nth column in the given matrix.
    :param matrix: List of lists representing the matrix.
    :param n: Index of the column to find the maximum value.
    :return: Maximum value in the nth column.
    '''
    try:
        # Extract the nth column
        column = [row[n] for row in matrix]
        # Return the maximum value in the column
        return max(column)
    except IndexError:
        raise ValueError("Column index out of range.")
    except Exception as e:
        raise ValueError(f"An error occurred: {e}")
class MatrixMaxApp:
    '''
    A GUI application to find the maximum value of a specified column in a matrix.
    '''
    def __init__(self, root):
        self.root = root
        self.root.title("Matrix Column Max Finder")
        self.create_widgets()
    def create_widgets(self):
        '''
        Create and layout the widgets for the application.
        '''
        self.matrix_label = tk.Label(self.root, text="Enter Matrix (comma-separated rows):")
        self.matrix_label.pack()
        self.matrix_entry = tk.Entry(self.root, width=50)
        self.matrix_entry.pack()
        self.column_label = tk.Label(self.root, text="Enter Column Index:")
        self.column_label.pack()
        self.column_entry = tk.Entry(self.root, width=5)
        self.column_entry.pack()
        self.calculate_button = tk.Button(self.root, text="Calculate Max", command=self.get_max_of_column)
        self.calculate_button.pack()
        self.result_label = tk.Label(self.root, text="Result: ")
        self.result_label.pack()
    def get_max_of_column(self):
        '''
        Calculate and display the maximum value of the specified column.
        '''
        try:
            # Parse the matrix input
            matrix_input = self.matrix_entry.get()
            matrix = []
            for row in matrix_input.split(';'):
                # Ensure each row contains valid integers
                row_values = row.split(',')
                if not all(value.strip().isdigit() for value in row_values):
                    raise ValueError("Matrix entries must be integers.")
                matrix.append(list(map(int, row_values)))
            # Parse the column index
            column_index = int(self.column_entry.get())
            # Calculate the max value of the specified column
            max_value = max_of_nth_column(matrix, column_index)
            # Display the result
            self.result_label.config(text=f"Result: {max_value}")
        except ValueError as ve:
            messagebox.showerror("Error", str(ve))
        except Exception as e:
            messagebox.showerror("Error", f"An unexpected error occurred: {e}")
if __name__ == "__main__":
    root = tk.Tk()
    app = MatrixMaxApp(root)
    root.mainloop()

```python
'''
This script creates a GUI application that allows users to input a matrix and a column index,
then calculates and displays the maximum value of the specified column.
'''
import tkinter as tk
from tkinter import messagebox
def max_of_nth_column(matrix, n):
    '''
    Returns the maximum value of the nth column in the given matrix.
    :param matrix: List of lists representing the matrix.
    :param n: Index of the column to find the maximum value.
    :return: Maximum value in the nth column.
    '''
    try:
        # Extract the nth column
        column = [row[n] for row in matrix]
        # Return the maximum value in the column
        return max(column)
    except IndexError:
        raise ValueError("Column index out of range.")
    except Exception as e:
        raise ValueError(f"An error occurred: {e}")
class MatrixMaxApp:
    '''
    A GUI application to find the maximum value of a specified column in a matrix.
    '''
    def __init__(self, root):
        self.root = root
        self.root.title("Matrix Column Max Finder")
        self.create_widgets()
    def create_widgets(self):
        '''
        Create and layout the widgets for the application.
        '''
        self.matrix_label = tk.Label(self.root, text="Enter Matrix (comma-separated rows):")
        self.matrix_label.pack()
        self.matrix_entry = tk.Entry(self.root, width=50)
        self.matrix_entry.pack()
        self.column_label = tk.Label(self.root, text="Enter Column Index:")
        self.column_label.pack()
        self.column_entry = tk.Entry(self.root, width=5)
        self.column_entry.pack()
        self.calculate_button = tk.Button(self.root, text="Calculate Max", command=self.get_max_of_column)
        self.calculate_button.pack()
        self.result_label = tk.Label(self.root, text="Result: ")
        self.result_label.pack()
    def get_max_of_column(self):
        '''
        Calculate and display the maximum value of the specified column.
        '''
        try:
            # Parse the matrix input
            matrix_input = self.matrix_entry.get()
            matrix = []
            for row in matrix_input.split(';'):
                # Ensure each row contains valid integers
                row_values = row.split(',')
                if not all(value.strip().isdigit() for value in row_values):
                    raise ValueError("Matrix entries must be integers.")
                matrix.append(list(map(int, row_values)))
            # Parse the column index
            column_index = int(self.column_entry.get())
            # Calculate the max value of the specified column
            max_value = max_of_nth_column(matrix, column_index)
            # Display the result
            self.result_label.config(text=f"Result: {max_value}")
        except ValueError as ve:
            messagebox.showerror("Error", str(ve))
        except Exception as e:
            messagebox.showerror("Error", f"An unexpected error occurred: {e}")
if __name__ == "__main__":
    root = tk.Tk()
    app = MatrixMaxApp(root)
    root.mainloop()
