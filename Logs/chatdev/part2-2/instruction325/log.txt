```python
'''

This module contains the TupleCheckerApp class which provides a GUI to check if a given tuple contains a specified element.

'''

import tkinter as tk
from tkinter import messagebox

def contains_element(tup, k):
    '''
    Check if the tuple contains the element k.

    Parameters:
    tup (tuple): The tuple to check.
    k (any): The element to find in the tuple.

    Returns:
    bool: True if k is in tup, False otherwise.
    '''
    return k in tup

class TupleCheckerApp:
    '''
    A GUI application to check if a tuple contains a specified element.
    '''

    def __init__(self, master):
        '''
        Initialize the application with the main window and widgets.

        Parameters:
        master (Tk): The root window of the application.
        '''
        self.master = master
        master.title("Tuple Checker")

        self.label = tk.Label(master, text="Enter a tuple (comma-separated):")
        self.label.pack()

        self.tuple_entry = tk.Entry(master)
        self.tuple_entry.pack()

        self.k_label = tk.Label(master, text="Enter the element to check:")
        self.k_label.pack()

        self.k_entry = tk.Entry(master)
        self.k_entry.pack()

        self.check_button = tk.Button(master, text="Check", command=self.check_tuple)
        self.check_button.pack()

        self.result_label = tk.Label(master, text="")
        self.result_label.pack()

    def check_tuple(self):
        '''
        Check if the entered tuple contains the specified element and update the result label.
        '''
        try:
            # Convert the input string to a tuple
            tuple_input = self.tuple_entry.get()
            tup = tuple(map(str.strip, tuple_input.split(',')))

            # Get the element to check
            k = self.k_entry.get().strip()

            # Check if the element is in the tuple
            if contains_element(tup, k):
                self.result_label.config(text=f"The element '{k}' is in the tuple.")
            else:
                self.result_label.config(text=f"The element '{k}' is not in the tuple.")
        except Exception as e:
            messagebox.showerror("Error", f"Invalid input: {e}")

if __name__ == "__main__":
    root = tk.Tk()
    app = TupleCheckerApp(root)
    root.mainloop()
```

This code provides a simple GUI application where users can input a tuple and an element to check if the element is present in the tuple. The `contains_element` function performs the check, and the `TupleCheckerApp` class manages the GUI.

'''

This module contains the TupleCheckerApp class which provides a GUI to check if a given tuple contains a specified element.

'''

import tkinter as tk
from tkinter import messagebox

def contains_element(tup, k):
    '''
    Check if the tuple contains the element k.

    Parameters:
    tup (tuple): The tuple to check.
    k (any): The element to find in the tuple.

    Returns:
    bool: True if k is in tup, False otherwise.
    '''
    return k in tup

class TupleCheckerApp:
    '''
    A GUI application to check if a tuple contains a specified element.
    '''

    def __init__(self, master):
        '''
        Initialize the application with the main window and widgets.

        Parameters:
        master (Tk): The root window of the application.
        '''
        self.master = master
        master.title("Tuple Checker")

        self.label = tk.Label(master, text="Enter a tuple (comma-separated):")
        self.label.pack()

        self.tuple_entry = tk.Entry(master)
        self.tuple_entry.pack()

        self.k_label = tk.Label(master, text="Enter the element to check:")
        self.k_label.pack()

        self.k_entry = tk.Entry(master)
        self.k_entry.pack()

        self.check_button = tk.Button(master, text="Check", command=self.check_tuple)
        self.check_button.pack()

        self.result_label = tk.Label(master, text="")
        self.result_label.pack()

    def check_tuple(self):
        '''
        Check if the entered tuple contains the specified element and update the result label.
        '''
        try:
            # Convert the input string to a tuple
            tuple_input = self.tuple_entry.get()
            tup = tuple(map(str.strip, tuple_input.split(',')))

            # Get the element to check
            k = self.k_entry.get().strip()

            # Check if the element is in the tuple
            if contains_element(tup, k):
                self.result_label.config(text=f"The element '{k}' is in the tuple.")
            else:
                self.result_label.config(text=f"The element '{k}' is not in the tuple.")
        except Exception as e:
            messagebox.showerror("Error", f"Invalid input: {e}")

if __name__ == "__main__":
    root = tk.Tk()
    app = TupleCheckerApp(root)
    root.mainloop()
```

'''
This module contains the TupleCheckerApp class which provides a GUI to check if a given tuple contains a specified element.
'''
import tkinter as tk
from tkinter import messagebox
def contains_element(tup, k):
    '''
    Check if the tuple contains the element k.
    Parameters:
    tup (tuple): The tuple to check.
    k (any): The element to find in the tuple.
    Returns:
    bool: True if k is in tup, False otherwise.
    '''
    return k in tup
class TupleCheckerApp:
    '''
    A GUI application to check if a tuple contains a specified element.
    '''
    def __init__(self, master):
        '''
        Initialize the application with the main window and widgets.
        Parameters:
        master (Tk): The root window of the application.
        '''
        self.master = master
        master.title("Tuple Checker")
        self.label = tk.Label(master, text="Enter a tuple (comma-separated):")
        self.label.pack()
        self.tuple_entry = tk.Entry(master)
        self.tuple_entry.pack()
        self.k_label = tk.Label(master, text="Enter the element to check:")
        self.k_label.pack()
        self.k_entry = tk.Entry(master)
        self.k_entry.pack()
        self.check_button = tk.Button(master, text="Check", command=self.check_tuple)
        self.check_button.pack()
        self.result_label = tk.Label(master, text="")
        self.result_label.pack()
    def check_tuple(self):
        '''
        Check if the entered tuple contains the specified element and update the result label.
        '''
        try:
            # Convert the input string to a tuple
            tuple_input = self.tuple_entry.get()
            tup = tuple(map(str.strip, tuple_input.split(',')))
            # Get the element to check
            k = self.k_entry.get().strip()
            # Check if the element is in the tuple
            if contains_element(tup, k):
                self.result_label.config(text=f"The element '{k}' is in the tuple.")
            else:
                self.result_label.config(text=f"The element '{k}' is not in the tuple.")
        except Exception as e:
            messagebox.showerror("Error", f"Invalid input: {e}")
if __name__ == "__main__":
    root = tk.Tk()
    app = TupleCheckerApp(root)
    root.mainloop()

'''
This module contains the TupleCheckerApp class which provides a GUI to check if a given tuple contains a specified element.
'''
import tkinter as tk
from tkinter import messagebox
def contains_element(tup, k):
    '''
    Check if the tuple contains the element k.
    Parameters:
    tup (tuple): The tuple to check.
    k (any): The element to find in the tuple.
    Returns:
    bool: True if k is in tup, False otherwise.
    '''
    return k in tup
class TupleCheckerApp:
    '''
    A GUI application to check if a tuple contains a specified element.
    '''
    def __init__(self, master):
        '''
        Initialize the application with the main window and widgets.
        Parameters:
        master (Tk): The root window of the application.
        '''
        self.master = master
        master.title("Tuple Checker")
        self.label = tk.Label(master, text="Enter a tuple (comma-separated):")
        self.label.pack()
        self.tuple_entry = tk.Entry(master)
        self.tuple_entry.pack()
        self.k_label = tk.Label(master, text="Enter the element to check:")
        self.k_label.pack()
        self.k_entry = tk.Entry(master)
        self.k_entry.pack()
        self.check_button = tk.Button(master, text="Check", command=self.check_tuple)
        self.check_button.pack()
        self.result_label = tk.Label(master, text="")
        self.result_label.pack()
    def check_tuple(self):
        '''
        Check if the entered tuple contains the specified element and update the result label.
        '''
        try:
            # Convert the input string to a tuple
            tuple_input = self.tuple_entry.get()
            tup = tuple(map(str.strip, tuple_input.split(',')))
            # Get the element to check
            k = self.k_entry.get().strip()
            # Check if the element is in the tuple
            if contains_element(tup, k):
                self.result_label.config(text=f"The element '{k}' is in the tuple.")
            else:
                self.result_label.config(text=f"The element '{k}' is not in the tuple.")
        except Exception as e:
            messagebox.showerror("Error", f"Invalid input: {e}")
if __name__ == "__main__":
    root = tk.Tk()
    app = TupleCheckerApp(root)
    root.mainloop()

'''
This module contains the TupleCheckerApp class which provides a GUI to check if a given tuple contains a specified element.
'''
import tkinter as tk
from tkinter import messagebox
def contains_element(tup, k):
    '''
    Check if the tuple contains the element k.
    Parameters:
    tup (tuple): The tuple to check.
    k (any): The element to find in the tuple.
    Returns:
    bool: True if k is in tup, False otherwise.
    '''
    return k in tup
class TupleCheckerApp:
    '''
    A GUI application to check if a tuple contains a specified element.
    '''
    def __init__(self, master):
        '''
        Initialize the application with the main window and widgets.
        Parameters:
        master (Tk): The root window of the application.
        '''
        self.master = master
        master.title("Tuple Checker")
        self.label = tk.Label(master, text="Enter a tuple (comma-separated):")
        self.label.pack()
        self.tuple_entry = tk.Entry(master)
        self.tuple_entry.pack()
        self.k_label = tk.Label(master, text="Enter the element to check:")
        self.k_label.pack()
        self.k_entry = tk.Entry(master)
        self.k_entry.pack()
        self.check_button = tk.Button(master, text="Check", command=self.check_tuple)
        self.check_button.pack()
        self.result_label = tk.Label(master, text="")
        self.result_label.pack()
    def check_tuple(self):
        '''
        Check if the entered tuple contains the specified element and update the result label.
        '''
        try:
            # Convert the input string to a tuple
            tuple_input = self.tuple_entry.get()
            tup = tuple(map(str.strip, tuple_input.split(',')))
            # Get the element to check
            k = self.k_entry.get().strip()
            # Check if the element is in the tuple
            if contains_element(tup, k):
                self.result_label.config(text=f"The element '{k}' is in the tuple.")
            else:
                self.result_label.config(text=f"The element '{k}' is not in the tuple.")
        except Exception as e:
            messagebox.showerror("Error", f"Invalid input: {e}")
if __name__ == "__main__":
    root = tk.Tk()
    app = TupleCheckerApp(root)
    root.mainloop()
