```python
'''

This file contains the main application code for the GUI-based odd number filter.
It uses tkinter for the graphical user interface and imports the utility function
to filter odd numbers.

'''

import tkinter as tk
from tkinter import messagebox
from filter_utils import filter_odd_numbers


class MainApplication:
    '''
    The MainApplication class initializes and runs the GUI application.
    It allows users to input a list of numbers and displays the filtered odd numbers.
    '''

    def __init__(self, root):
        '''
        Initializes the GUI components.
        :param root: The root window of the tkinter application.
        '''
        self.root = root
        self.root.title("Odd Number Filter")

        # Input Label
        self.label = tk.Label(root, text="Enter a list of numbers (comma-separated):")
        self.label.pack(pady=10)

        # Input Entry
        self.input_entry = tk.Entry(root, width=50)
        self.input_entry.pack(pady=10)

        # Filter Button
        self.filter_button = tk.Button(root, text="Filter Odd Numbers", command=self.filter_odd_numbers)
        self.filter_button.pack(pady=10)

        # Output Label
        self.output_label = tk.Label(root, text="Filtered Odd Numbers:")
        self.output_label.pack(pady=10)

        # Output Text
        self.output_text = tk.Text(root, height=10, width=50, state='disabled')
        self.output_text.pack(pady=10)

    def filter_odd_numbers(self):
        '''
        Filters odd numbers from the input list and displays the result.
        '''
        input_text = self.input_entry.get()
        try:
            # Convert input text to a list of integers
            numbers = list(map(int, input_text.split(',')))
            # Filter odd numbers using the utility function
            odd_numbers = filter_odd_numbers(numbers)
            # Display the result
            self.output_text.config(state='normal')
            self.output_text.delete(1.0, tk.END)
            self.output_text.insert(tk.END, ', '.join(map(str, odd_numbers)))
            self.output_text.config(state='disabled')
        except ValueError:
            # Show an error message if the input is invalid
            messagebox.showerror("Invalid Input", "Please enter a valid list of numbers (comma-separated).")


if __name__ == "__main__":
    # Create the root window
    root = tk.Tk()
    # Create and run the main application
    app = MainApplication(root)
    root.mainloop()
```

#### filter_utils.py

```python
'''

This file contains utility functions for the odd number filter application.
It includes a function to filter odd numbers from a list of integers.

'''


def filter_odd_numbers(numbers):
    '''
    Filters odd numbers from a list of integers.
    :param numbers: A list of integers.
    :return: A list of odd integers.
    '''
    return [num for num in numbers if num % 2 != 0]
```

```python
'''

This file contains the main application code for the GUI-based odd number filter.
It uses tkinter for the graphical user interface and imports the utility function
to filter odd numbers.

'''

import tkinter as tk
from tkinter import messagebox
from filter_utils import filter_odd_numbers


class MainApplication:
    '''
    The MainApplication class initializes and runs the GUI application.
    It allows users to input a list of numbers and displays the filtered odd numbers.
    '''

    def __init__(self, root):
        '''
        Initializes the GUI components.
        :param root: The root window of the tkinter application.
        '''
        self.root = root
        self.root.title("Odd Number Filter")

        # Input Label
        self.label = tk.Label(root, text="Enter a list of numbers (comma-separated):")
        self.label.pack(pady=10)

        # Input Entry
        self.input_entry = tk.Entry(root, width=50)
        self.input_entry.pack(pady=10)

        # Filter Button
        self.filter_button = tk.Button(root, text="Filter Odd Numbers", command=self.filter_odd_numbers)
        self.filter_button.pack(pady=10)

        # Output Label
        self.output_label = tk.Label(root, text="Filtered Odd Numbers:")
        self.output_label.pack(pady=10)

        # Output Text
        self.output_text = tk.Text(root, height=10, width=50, state='disabled')
        self.output_text.pack(pady=10)

    def filter_odd_numbers(self):
        '''
        Filters odd numbers from the input list and displays the result.
        '''
        input_text = self.input_entry.get()
        try:
            # Convert input text to a list of integers
            numbers = list(map(int, input_text.split(',')))
            # Filter odd numbers using the utility function
            odd_numbers = filter_odd_numbers(numbers)
            # Display the result
            self.output_text.config(state='normal')
            self.output_text.delete(1.0, tk.END)
            self.output_text.insert(tk.END, ', '.join(map(str, odd_numbers)))
            self.output_text.config(state='disabled')
        except ValueError:
            # Show an error message if the input is invalid
            messagebox.showerror("Invalid Input", "Please enter a valid list of numbers (comma-separated).")


if __name__ == "__main__":
    # Create the root window
    root = tk.Tk()
    # Create and run the main application
    app = MainApplication(root)
    root.mainloop()
```

'''

It uses tkinter for the graphical user interface and imports the utility function
to filter odd numbers.

import tkinter as tk
from tkinter import messagebox
from filter_utils import filter_odd_numbers
class MainApplication:
    '''
    The MainApplication class initializes and runs the GUI application.
    It allows users to input a list of numbers and displays the filtered odd numbers.
    '''
    def __init__(self, root):
        '''
        Initializes the GUI components.
        :param root: The root window of the tkinter application.
        '''
        self.root = root
        self.root.title("Odd Number Filter")
        # Input Label
        self.label = tk.Label(root, text="Enter a list of numbers (comma-separated):")
        self.label.pack(pady=10)
        # Input Entry
        self.input_entry = tk.Entry(root, width=50)
        self.input_entry.pack(pady=10)
        # Filter Button
        self.filter_button = tk.Button(root, text="Filter Odd Numbers", command=self.filter_odd_numbers)
        self.filter_button.pack(pady=10)
        # Output Label
        self.output_label = tk.Label(root, text="Filtered Odd Numbers:")
        self.output_label.pack(pady=10)
        # Output Text
        self.output_text = tk.Text(root, height=10, width=50, state='disabled')
        self.output_text.pack(pady=10)
    def filter_odd_numbers(self):
        '''
        Filters odd numbers from the input list and displays the result.
        '''
        input_text = self.input_entry.get()
        try:
            # Convert input text to a list of integers
            numbers = list(map(int, input_text.split(',')))
            # Filter odd numbers using the utility function
            odd_numbers = filter_odd_numbers(numbers)
            # Display the result
            self.output_text.config(state='normal')
            self.output_text.delete(1.0, tk.END)
            self.output_text.insert(tk.END, ', '.join(map(str, odd_numbers)))
            self.output_text.config(state='disabled')
        except ValueError:
            # Show an error message if the input is invalid
            messagebox.showerror("Invalid Input", "Please enter a valid list of numbers (comma-separated).")
if __name__ == "__main__":
    # Create the root window
    root = tk.Tk()
    # Create and run the main application
    app = MainApplication(root)
    root.mainloop()

'''
This file contains the main application code for the GUI-based odd number filter.
It uses tkinter for the graphical user interface and imports the utility function
to filter odd numbers.

import tkinter as tk
from tkinter import messagebox
from filter_utils import filter_odd_numbers
class MainApplication:
    '''
    The MainApplication class initializes and runs the GUI application.
    It allows users to input a list of numbers and displays the filtered odd numbers.
    '''
    def __init__(self, root):
        '''
        Initializes the GUI components.
        :param root: The root window of the tkinter application.
        '''
        self.root = root
        self.root.title("Odd Number Filter")
        # Input Label
        self.label = tk.Label(root, text="Enter a list of numbers (comma-separated):")
        self.label.pack(pady=10)
        # Input Entry
        self.input_entry = tk.Entry(root, width=50)
        self.input_entry.pack(pady=10)
        # Filter Button
        self.filter_button = tk.Button(root, text="Filter Odd Numbers", command=self.filter_odd_numbers)
        self.filter_button.pack(pady=10)
        # Output Label
        self.output_label = tk.Label(root, text="Filtered Odd Numbers:")
        self.output_label.pack(pady=10)
        # Output Text
        self.output_text = tk.Text(root, height=10, width=50, state='disabled')
        self.output_text.pack(pady=10)
    def filter_odd_numbers(self):
        '''
        Filters odd numbers from the input list and displays the result.
        '''
        input_text = self.input_entry.get()
        try:
            # Convert input text to a list of integers
            numbers = list(map(int, input_text.split(',')))
            # Filter odd numbers using the utility function
            odd_numbers = filter_odd_numbers(numbers)
            # Display the result
            self.output_text.config(state='normal')
            self.output_text.delete(1.0, tk.END)
            self.output_text.insert(tk.END, ', '.join(map(str, odd_numbers)))
            self.output_text.config(state='disabled')
        except ValueError:
            # Show an error message if the input is invalid
            messagebox.showerror("Invalid Input", "Please enter a valid list of numbers (comma-separated).")
if __name__ == "__main__":
    # Create the root window
    root = tk.Tk()
    # Create and run the main application
    app = MainApplication(root)
    root.mainloop()

    The MainApplication class initializes and runs the GUI application.

        '''
        Initializes the GUI components.

        self.root.title("Odd Number Filter")
        # Input Label
        self.label = tk.Label(root, text="Enter a list of numbers (comma-separated):")
        self.label.pack(pady=10)

        self.input_entry = tk.Entry(root, width=50)
        self.input_entry.pack(pady=10)
        # Filter Button
        self.filter_button = tk.Button(root, text="Filter Odd Numbers", command=self.filter_odd_numbers)
        self.filter_button.pack(pady=10)
        # Output Label
        self.output_label = tk.Label(root, text="Filtered Odd Numbers:")
        self.output_label.pack(pady=10)
        # Output Text

            # Convert input text to a list of integers
            numbers = list(map(int, input_text.split(',')))

            odd_numbers = filter_odd_numbers(numbers)
            # Display the result
            self.output_text.config(state='normal')
            self.output_text.delete(1.0, tk.END)
            self.output_text.insert(tk.END, ', '.join(map(str, odd_numbers)))

This application is perfect for users who need to quickly identify odd numbers from a list without writing any code themselves.

## 📖 Documentation

### Main Functions

- **Input List of Numbers:** Users can enter a list of numbers in a comma-separated format.
- **Filter Odd Numbers:** The application processes the input and filters out odd numbers using a utility function.
- **Display Results:** The filtered odd numbers are displayed in the application interface.

### Installation

1. **Ensure Python is Installed:** Make sure Python is installed on your system. You can download it from [python.org](https://www.python.org/downloads/).
2. **Clone the Repository:** Download or clone the project repository to your local machine.
3. **Run the Application:** Navigate to the directory containing the `main.py` file and execute the following command in your terminal or command prompt:
   ```
   python main.py
   ```

### How to Use

1. **Launch the Application:** Run the `main.py` file to start the application.
2. **Enter Numbers:** In the application window, enter a list of numbers separated by commas (e.g., `1, 2, 3, 4, 5`).
3. **Filter Odd Numbers:** Click the "Filter Odd Numbers" button to process the input.
4. **View Results:** The filtered odd numbers will be displayed in the text area below the button.

### Error Handling

- **Invalid Input:** If the input is not a valid list of numbers, an error message will be displayed. Ensure the numbers are entered correctly and separated by commas.
