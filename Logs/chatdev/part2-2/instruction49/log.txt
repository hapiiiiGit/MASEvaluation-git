'''

This file contains the main application code for the Dictionary Filter App.
It uses tkinter to create a GUI that allows users to input a dictionary and an integer,
and then filters the dictionary to only include entries with values greater than or equal to the integer.

'''

import tkinter as tk
from tkinter import messagebox

def filter_dict(input_dict, n):
    '''

    Filters the dictionary to only include entries with values greater than or equal to n.

    Parameters:
    input_dict (dict): The dictionary to be filtered.
    n (int): The threshold value.

    Returns:
    dict: A new dictionary with filtered entries.

    '''
    return {k: v for k, v in input_dict.items() if v >= n}

class DictionaryFilterApp:
    def __init__(self, master):
        '''

        Initializes the Dictionary Filter App GUI.

        Parameters:
        master (tk.Tk): The root window of the application.

        '''
        self.master = master
        master.title("Dictionary Filter App")

        self.label = tk.Label(master, text="Enter dictionary (e.g., {'a': 1, 'b': 2}):")
        self.label.pack()

        self.dict_entry = tk.Entry(master, width=50)
        self.dict_entry.pack()

        self.label_n = tk.Label(master, text="Enter integer n:")
        self.label_n.pack()

        self.n_entry = tk.Entry(master, width=10)
        self.n_entry.pack()

        self.filter_button = tk.Button(master, text="Filter", command=self.filter_dictionary)
        self.filter_button.pack()

        self.result_label = tk.Label(master, text="Filtered Dictionary:")
        self.result_label.pack()

        self.result_text = tk.Text(master, height=10, width=50)

'''
This file contains the main application code for the Dictionary Filter App.
It uses tkinter to create a GUI that allows users to input a dictionary and an integer,
and then filters the dictionary to only include entries with values greater than or equal to the integer.
'''
import tkinter as tk
from tkinter import messagebox
import ast
def filter_dict(input_dict, n):
    '''
    Filters the dictionary to only include entries with values greater than or equal to n.
    Parameters:
    input_dict (dict): The dictionary to be filtered.
    n (int): The threshold value.
    Returns:
    dict: A new dictionary with filtered entries.
    '''
    return {k: v for k, v in input_dict.items() if v >= n}
class DictionaryFilterApp:
    def __init__(self, master):
        '''
        Initializes the Dictionary Filter App GUI.
        Parameters:
        master (tk.Tk): The root window of the application.
        '''
        self.master = master
        master.title("Dictionary Filter App")
        self.label = tk.Label(master, text="Enter dictionary (e.g., {'a': 1, 'b': 2}):")
        self.label.pack()
        self.dict_entry = tk.Entry(master, width=50)
        self.dict_entry.pack()
        self.label_n = tk.Label(master, text="Enter integer n:")
        self.label_n.pack()
        self.n_entry = tk.Entry(master, width=10)
        self.n_entry.pack()
        self.filter_button = tk.Button(master, text="Filter", command=self.filter_dictionary)
        self.filter_button.pack()
        self.result_label = tk.Label(master, text="Filtered Dictionary:")
        self.result_label.pack()
        self.result_text = tk.Text(master, height=10, width=50)
        self.result_text.pack()
    def filter_dictionary(self):
        '''
        Filters the dictionary based on user input and displays the result.
        '''
        try:
            input_dict = ast.literal_eval(self.dict_entry.get())
            n = int(self.n_entry.get())
            if not isinstance(input_dict, dict):
                raise ValueError("Input is not a valid dictionary.")
            filtered_dict = filter_dict(input_dict, n)
            self.display_result(filtered_dict)
        except Exception as e:
            messagebox.showerror("Error", f"Invalid input: {e}")
    def display_result(self, result):
        '''
        Displays the filtered dictionary in the text widget.
        Parameters:
        result (dict): The filtered dictionary to display.
        '''
        self.result_text.delete(1.0, tk.END)
        self.result_text.insert(tk.END, str(result))
if __name__ == "__main__":
    root = tk.Tk()
    app = DictionaryFilterApp(root)
    root.mainloop()

'''
This file contains the main application code for the Dictionary Filter App.
It uses tkinter to create a GUI that allows users to input a dictionary and an integer,
and then filters the dictionary to only include entries with values greater than or equal to the integer.
'''
import tkinter as tk
from tkinter import messagebox
import ast
def filter_dict(input_dict, n):
    '''
    Filters the dictionary to only include entries with values greater than or equal to n.
    Parameters:
    input_dict (dict): The dictionary to be filtered.
    n (int): The threshold value.
    Returns:
    dict: A new dictionary with filtered entries.
    '''
    return {k: v for k, v in input_dict.items() if v >= n}
class DictionaryFilterApp:
    def __init__(self, master):
        '''
        Initializes the Dictionary Filter App GUI.
        Parameters:
        master (tk.Tk): The root window of the application.
        '''
        self.master = master
        master.title("Dictionary Filter App")
        self.label = tk.Label(master, text="Enter dictionary (e.g., {'a': 1, 'b': 2}):")
        self.label.pack()
        self.dict_entry = tk.Entry(master, width=50)
        self.dict_entry.pack()
        self.label_n = tk.Label(master, text="Enter integer n:")
        self.label_n.pack()
        self.n_entry = tk.Entry(master, width=10)
        self.n_entry.pack()
        self.filter_button = tk.Button(master, text="Filter", command=self.filter_dictionary)
        self.filter_button.pack()
        self.result_label = tk.Label(master, text="Filtered Dictionary:")
        self.result_label.pack()
        self.result_text = tk.Text(master, height=10, width=50)
        self.result_text.pack()
    def filter_dictionary(self):
        '''
        Filters the dictionary based on user input and displays the result.
        '''
        try:
            input_dict = ast.literal_eval(self.dict_entry.get())
            n = int(self.n_entry.get())
            if not isinstance(input_dict, dict):
                raise ValueError("Input is not a valid dictionary.")
            filtered_dict = filter_dict(input_dict, n)
            self.display_result(filtered_dict)
        except Exception as e:
            messagebox.showerror("Error", f"Invalid input: {e}")
    def display_result(self, result):
        '''
        Displays the filtered dictionary in the text widget.
        Parameters:
        result (dict): The filtered dictionary to display.
        '''
        self.result_text.delete(1.0, tk.END)
        self.result_text.insert(tk.END, str(result))
if __name__ == "__main__":
    root = tk.Tk()
    app = DictionaryFilterApp(root)
    root.mainloop()
