'''

Main file for the dictionary depth calculator application. Initializes the GUI and handles user interactions.

'''

import tkinter as tk
from tkinter import messagebox
from utils import find_dict_depth

class App:
    def __init__(self, root):
        '''
        Initializes the GUI application.
        '''
        self.root = root
        self.root.title("Dictionary Depth Calculator")
        
        # Input label and text box
        self.label = tk.Label(root, text="Enter a dictionary:")
        self.label.pack(pady=5)
        
        self.textbox = tk.Text(root, height=10, width=50)
        self.textbox.pack(pady=5)
        
        # Calculate button
        self.calculate_button = tk.Button(root, text="Calculate Depth", command=self.calculate_depth)
        self.calculate_button.pack(pady=10)
        
        # Output label
        self.result_label = tk.Label(root, text="", fg="blue")
        self.result_label.pack(pady=5)
    
    def calculate_depth(self):
        '''
        Handles the calculation of dictionary depth based on user input.
        '''
        try:
            # Get the dictionary from the textbox
            user_input = self.textbox.get("1.0", tk.END).strip()
            user_dict = eval(user_input)  # Convert string to dictionary
            
            if not isinstance(user_dict, dict):
                raise ValueError("Input is not a dictionary.")
            
            # Calculate depth using utility function
            depth = find_dict_depth(user_dict)
            
            # Display the result
            self.result_label.config(text=f"Depth of the dictionary: {depth}")
        except Exception as e:
            messagebox.showerror("Error", f"Invalid input: {e}")

if __name__ == "__main__":
    root = tk.Tk()
    app = App(root)
    root.mainloop()

'''

Utility file containing functions for dictionary depth calculation.

'''

def find_dict_depth(d):
    '''
    Recursively calculates the depth of a dictionary.
    
    Parameters:
        d (dict): The dictionary whose depth is to be calculated.
    
    Returns:
        int: Depth of the dictionary.
    '''
    if not isinstance(d, dict) or not d:
        return 0
    return 1 + max(find_dict_depth(value) if isinstance(value, dict) else 0 for value in d.values())

```

import tkinter as tk
from tkinter import messagebox

class App:
    def __init__(self, root):
        '''
        Initializes the GUI application.
        '''
        self.root = root

def find_dict_depth(d):
    '''
    Recursively calculates the depth of a dictionary.
    
    Parameters:
        d (dict): The dictionary whose depth is to be calculated.
    
    Returns:
        int: Depth of the dictionary.
    '''
    if not isinstance(d, dict) or not d:
        return 0
    return 1 + max(find_dict_depth(value) if isinstance(value, dict) else 0 for value in d.values())

'''

def find_dict_depth(d):
    '''
    Recursively calculates the depth of a dictionary.
    Parameters:
        d (dict): The dictionary whose depth is to be calculated.
    Returns:
        int: Depth of the dictionary.
    '''
    if not isinstance(d, dict) or not d:
        return 0
    return 1 + max(find_dict_depth(value) if isinstance(value, dict) else 0 for value in d.values())

def find_dict_depth(d):
    '''
    Recursively calculates the depth of a dictionary.
    Parameters:
        d (dict): The dictionary whose depth is to be calculated.
    Returns:
        int: Depth of the dictionary.
    '''
    if not isinstance(d, dict) or not d:
        return 0
    return 1 + max(find_dict_depth(value) if isinstance(value, dict) else 0 for value in d.values())
