'''

This file contains the main application logic for the GUI-based Python program. It uses tkinter for the graphical user interface and imports the function to generate combinations with repetition.

'''

import tkinter as tk
from tkinter import messagebox
from itertools import combinations_with_replacement

def generate_combinations(list_elements, n):
    '''
    Generates all combinations (with repetition) of the elements of the list.

    Parameters:
        list_elements (list): The input list of elements.
        n (int): The length of each combination.

    Returns:
        list: A list of tuples representing all combinations.
    '''
    try:
        return list(combinations_with_replacement(list_elements, n))
    except Exception as e:
        raise ValueError(f"Error generating combinations: {e}")

class App:
    '''
    A class to define the GUI application for generating combinations with repetition.
    '''
    def __init__(self, root):
        '''
        Initializes the GUI application.

        Parameters:
            root (tk.Tk): The root tkinter window.
        '''
        self.root = root
        self.root.title("Combination Generator")
        self.root.geometry("400x300")

        # Input fields
        self.label_list = tk.Label(root, text="Enter list elements (comma-separated):")
        self.label_list.pack()
        self.entry_list = tk.Entry(root, width=40)
        self.entry_list.pack()

'''

This file contains the main application logic for the GUI-based Python program. It uses tkinter for the graphical user interface and imports the function to generate combinations with repetition.

'''

import tkinter as tk
from tkinter import messagebox
from itertools import combinations_with_replacement

def generate_combinations(list_elements, n):
    '''
    Generates all combinations (with repetition) of the elements of the list.

    Parameters:
        list_elements (list): The input list of elements.
        n (int): The length of each combination.

    Returns:
        list: A list of tuples representing all combinations.
    '''
    try:
        return list(combinations_with_replacement(list_elements, n))
    except Exception as e:
        raise ValueError(f"Error generating combinations: {e}")

class App:
    '''
    A class to define the GUI application for generating combinations with repetition.
    '''
    def __init__(self, root):
        '''
        Initializes the GUI application.

        Parameters:
            root (tk.Tk): The root tkinter window.
        '''
        self.root = root
        self.root.title("Combination Generator")
        self.root.geometry("400x300")

        # Input fields
        self.label_list = tk.Label(root, text="Enter list elements (comma-separated):")
        self.label_list.pack()

        self.entry_list.pack()

        self.entry_n = tk.Entry(root, width=10)
        self.entry_n.pack()

        # Generate button
        self.button_generate = tk.Button(root, text="Generate Combinations", command=self.generate_combinations)
        self.button_generate.pack()

        # Output field
        self.label_output = tk.Label(root, text="Combinations:")
        self.label_output.pack()
        self.text_output = tk.Text(root, height=10, width=40)
        self.text_output.pack()

    def generate_combinations(self):
        '''
        Handles the button click event to generate combinations and display them in the GUI.
        '''
        try:
            # Get user input
            list_input = self.entry_list.get()
            n_input = self.entry_n.get()

            # Process input
            list_elements = [item.strip() for item in list_input.split(",")]
            n = int(n_input)

            # Generate combinations
            combinations = generate_combinations(list_elements, n)

            # Display results
            self.text_output.delete(1.0, tk.END)
            for combo in combinations:
                self.text_output.insert(tk.END, f"{combo}\n")
        except ValueError as ve:
            messagebox.showerror("Error", str(ve))
        except Exception as e:
            messagebox.showerror("Error", f"An unexpected error occurred: {e}")

if __name__ == "__main__":
    root = tk.Tk()
    app = App(root)
    root.mainloop()

'''
This file contains the main application logic for the GUI-based Python program. It uses tkinter for the graphical user interface and imports the function to generate combinations with repetition.
'''
import tkinter as tk
from tkinter import messagebox
from itertools import combinations_with_replacement
def generate_combinations(list_elements, n):
    '''
    Generates all combinations (with repetition) of the elements of the list.
    Parameters:
        list_elements (list): The input list of elements.
        n (int): The length of each combination.
    Returns:
        list: A list of tuples representing all combinations.
    '''
    try:
        if not isinstance(list_elements, list):
            raise ValueError("The input must be a list.")
        if not isinstance(n, int) or n <= 0:
            raise ValueError("The combination length (n) must be a positive integer.")
        return list(combinations_with_replacement(list_elements, n))
    except Exception as e:
        raise ValueError(f"Error generating combinations: {e}")
class App:
    '''
    A class to define the GUI application for generating combinations with repetition.
    '''
    def __init__(self, root):
        '''
        Initializes the GUI application.
        Parameters:
            root (tk.Tk): The root tkinter window.
        '''
        self.root = root
        self.root.title("Combination Generator")
        self.root.geometry("400x300")
        # Input fields
        self.label_list = tk.Label(root, text="Enter list elements (comma-separated):")
        self.label_list.pack(pady=5)
        self.entry_list = tk.Entry(root, width=40)
        self.entry_list.pack(pady=5)
        self.label_n = tk.Label(root, text="Enter combination length (n):")
        self.label_n.pack(pady=5)
        self.entry_n = tk.Entry(root, width=10)
        self.entry_n.pack(pady=5)
        # Generate button
        self.button_generate = tk.Button(root, text="Generate Combinations", command=self.generate_combinations)
        self.button_generate.pack(pady=10)
        # Output field
        self.label_output = tk.Label(root, text="Combinations:")
        self.label_output.pack(pady=5)
        self.text_output = tk.Text(root, height=10, width=40)
        self.text_output.pack(pady=5)
    def generate_combinations(self):
        '''
        Handles the button click event to generate combinations and display them in the GUI.
        '''
        try:
            # Get user input
            list_input = self.entry_list.get()
            n_input = self.entry_n.get()
            # Validate and process input
            if not list_input.strip():
                raise ValueError("The list of elements cannot be empty.")
            list_elements = [item.strip() for item in list_input.split(",")]
            if not n_input.strip():
                raise ValueError("The combination length (n) cannot be empty.")
            n = int(n_input)
            # Generate combinations
            combinations = generate_combinations(list_elements, n)
            # Display results
            self.text_output.delete(1.0, tk.END)
            for combo in combinations:
                self.text_output.insert(tk.END, f"{combo}\n")
        except ValueError as ve:
            messagebox.showerror("Input Error", str(ve))
        except Exception as e:
            messagebox.showerror("Error", f"An unexpected error occurred: {e}")
if __name__ == "__main__":
    root = tk.Tk()
    app = App(root)
    root.mainloop()

'''
This file contains the main application logic for the GUI-based Python program. It uses tkinter for the graphical user interface and imports the function to generate combinations with repetition.
'''
import tkinter as tk
from tkinter import messagebox
from itertools import combinations_with_replacement
def generate_combinations(list_elements, n):
    '''
    Generates all combinations (with repetition) of the elements of the list.
    Parameters:
        list_elements (list): The input list of elements.
        n (int): The length of each combination.
    Returns:
        list: A list of tuples representing all combinations.
    '''
    try:
        if not isinstance(list_elements, list):
            raise ValueError("The input must be a list.")
        if not isinstance(n, int) or n <= 0:
            raise ValueError("The combination length (n) must be a positive integer.")
        return list(combinations_with_replacement(list_elements, n))
    except Exception as e:
        raise ValueError(f"Error generating combinations: {e}")
class App:
    '''
    A class to define the GUI application for generating combinations with repetition.
    '''
    def __init__(self, root):
        '''
        Initializes the GUI application.
        Parameters:
            root (tk.Tk): The root tkinter window.
        '''
        self.root = root
        self.root.title("Combination Generator")
        self.root.geometry("400x300")
        # Input fields
        self.label_list = tk.Label(root, text="Enter list elements (comma-separated):")
        self.label_list.pack(pady=5)
        self.entry_list = tk.Entry(root, width=40)
        self.entry_list.pack(pady=5)
        self.label_n = tk.Label(root, text="Enter combination length (n):")
        self.label_n.pack(pady=5)
        self.entry_n = tk.Entry(root, width=10)
        self.entry_n.pack(pady=5)
        # Generate button
        self.button_generate = tk.Button(root, text="Generate Combinations", command=self.generate_combinations)
        self.button_generate.pack(pady=10)
        # Output field
        self.label_output = tk.Label(root, text="Combinations:")
        self.label_output.pack(pady=5)
        self.text_output = tk.Text(root, height=10, width=40)
        self.text_output.pack(pady=5)
    def generate_combinations(self):
        '''
        Handles the button click event to generate combinations and display them in the GUI.
        '''
        try:
            # Get user input
            list_input = self.entry_list.get()
            n_input = self.entry_n.get()
            # Validate and process input
            if not list_input.strip():
                raise ValueError("The list of elements cannot be empty.")
            list_elements = [item.strip() for item in list_input.split(",")]
            if not n_input.strip():
                raise ValueError("The combination length (n) cannot be empty.")
            n = int(n_input)
            # Generate combinations
            combinations = generate_combinations(list_elements, n)
            # Display results
            self.text_output.delete(1.0, tk.END)
            for combo in combinations:
                self.text_output.insert(tk.END, f"{combo}\n")
        except ValueError as ve:
            messagebox.showerror("Input Error", str(ve))
        except Exception as e:
            messagebox.showerror("Error", f"An unexpected error occurred: {e}")
if __name__ == "__main__":
    root = tk.Tk()
    app = App(root)
    root.mainloop()
