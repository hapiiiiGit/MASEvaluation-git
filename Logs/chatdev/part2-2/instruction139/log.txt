from tkinter import messagebox
def calculate_loss(cost_price, selling_price):
    '''
    Calculate the loss amount if there is a loss, otherwise return 0.
    Parameters:
    cost_price (float): The cost price of the item.
    selling_price (float): The selling price of the item.
    Returns:
    float: The loss amount if there is a loss, otherwise 0.

    A class to represent the Loss Calculator application with a GUI.
    '''

'''
This file contains the main application code for the Loss Calculator App.
It sets up the GUI and handles user interactions to calculate the loss amount.
'''
import tkinter as tk
from tkinter import messagebox
def calculate_loss(cost_price, selling_price):
    '''
    Calculate the loss amount if there is a loss, otherwise return 0.
    Parameters:
    cost_price (float): The cost price of the item.
    selling_price (float): The selling price of the item.
    Returns:
    float: The loss amount if there is a loss, otherwise 0.
    '''
    loss = cost_price - selling_price
    return loss if loss > 0 else 0
class LossCalculatorApp:
    '''
    A class to represent the Loss Calculator application with a GUI.
    '''
    def __init__(self, root):
        '''
        Initialize the application with the main window and widgets.
        Parameters:
        root (tk.Tk): The main window of the application.
        '''
        self.root = root
        self.root.title("Loss Calculator")
        # Create and place labels and entry widgets
        self.label_cost_price = tk.Label(root, text="Cost Price:")
        self.label_cost_price.grid(row=0, column=0, padx=10, pady=10)
        self.entry_cost_price = tk.Entry(root)
        self.entry_cost_price.grid(row=0, column=1, padx=10, pady=10)
        self.label_selling_price = tk.Label(root, text="Selling Price:")
        self.label_selling_price.grid(row=1, column=0, padx=10, pady=10)
        self.entry_selling_price = tk.Entry(root)
        self.entry_selling_price.grid(row=1, column=1, padx=10, pady=10)
        # Create and place the calculate button
        self.button_calculate = tk.Button(root, text="Calculate Loss", command=self.calculate_loss)
        self.button_calculate.grid(row=2, column=0, columnspan=2, pady=20)
    def calculate_loss(self):
        '''
        Handle the calculation of loss when the button is clicked.
        '''
        try:
            cost_price = float(self.entry_cost_price.get())
            selling_price = float(self.entry_selling_price.get())
            loss = calculate_loss(cost_price, selling_price)
            messagebox.showinfo("Result", f"Loss Amount: {loss}")
        except ValueError:
            messagebox.showerror("Input Error", "Please enter valid numbers for cost and selling prices.")
if __name__ == "__main__":
    root = tk.Tk()
    app = LossCalculatorApp(root)
    root.mainloop()
