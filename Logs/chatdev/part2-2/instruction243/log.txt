def reverse_array_upto_position(array, position):
    """
    Reverse the array up to the given position.
    :param array: List of integers.
    :param position: Integer position up to which the array should be reversed.
    :return: List of integers with elements reversed up to the given position.
    """
    return array[:position][::-1] + array[position:]

    :return: List of integers.
    :raises ValueError: If the input is invalid.
    """
    try:

        return array
    except ValueError:

def reverse_array_upto_position(array, position):
    """
    Reverse the array up to the given position.
    :param array: List of integers.
    :param position: Integer position up to which the array should be reversed.
    :return: List of integers with elements reversed up to the given position.
    """
    return array[:position][::-1] + array[position:]
def validate_input(input_str):
    """
    Validate and convert the input string into a list of integers.
    :param input_str: String input representing array elements.
    :return: List of integers.
    :raises ValueError: If the input is invalid.
    """
    try:
        array = [int(x.strip()) for x in input_str.split(',')]
        return array
    except ValueError:
        raise ValueError("Invalid array input. Please enter integers separated by commas.")
