'''

This file contains the implementation of a Python function to find the first odd number in a list and a GUI application using tkinter to interact with the user.

'''

import tkinter as tk
from tkinter import messagebox

def find_first_odd(numbers):
    '''
    Finds the first odd number in a list of numbers.

    Parameters:
    numbers (list): A list of integers.

    Returns:
    int: The first odd number in the list, or None if no odd number is found.
    '''
    for number in numbers:
        if number % 2 != 0:
            return number
    return None

class OddNumberFinderApp:
    '''
    A GUI application to find the first odd number in a list of numbers.
    '''
    def __init__(self, root):
        self.root = root
        self.root.title("First Odd Number Finder")

        self.label = tk.Label(root, text="Enter numbers separated by commas:")
        self.label.pack()

        self.entry = tk.Entry(root, width=50)
        self.entry.pack()

        self.button = tk.Button(root, text="Find First Odd Number", command=self.find_odd)
        self.button.pack()

    def find_odd(self):
        '''
        Retrieves the list of numbers from the entry widget, finds the first odd number, and displays it.
        '''
        try:
            numbers = list(map(int, self.entry.get().split(',')))
            first_odd = find_first_odd(numbers)
            if first_odd is not None:
                messagebox.showinfo("Result", f"The first odd number is: {first_odd}")
            else:
                messagebox.showinfo("Result", "No odd numbers found.")
        except ValueError:
            messagebox.showerror("Error", "Please enter valid integers separated by commas.")

def run():
    '''
    Initializes and runs the GUI application.
    '''
    root = tk.Tk()
    app = OddNumberFinderApp(root)
    root.mainloop()

if __name__ == "__main__":
    run()
```

This file contains the implementation of a Python function to find the first odd number in a list and a GUI application using tkinter to interact with the user.

'''

import tkinter as tk
from tkinter import messagebox

def find_first_odd(numbers):
    '''
    Finds the first odd number in a list of numbers.

    Parameters:
    numbers (list): A list of integers.

    Returns:
    int: The first odd number in the list, or None if no odd number is found.
    '''
    for number in numbers:
        if number % 2 != 0:
            return number
    return None

class OddNumberFinderApp:
    '''
    A GUI application to find the first odd number in a list of numbers.
    '''
    def __init__(self, root):
        self.root = root
        self.root.title("First Odd Number Finder")

        self.label = tk.Label(root, text="Enter numbers separated by commas:")
        self.label.pack()

        self.entry = tk.Entry(root, width=50)
        self.entry.pack()

        self.button = tk.Button(root, text="Find First Odd Number", command=self.find_odd)
        self.button.pack()

    def find_odd(self):
        '''
        Retrieves the list of numbers from the entry widget, finds the first odd number, and displays it.
        '''
        try:
            numbers = list(map(int, self.entry.get().split(',')))
            first_odd = find_first_odd(numbers)
            if first_odd is not None:
                messagebox.showinfo("Result", f"The first odd number is: {first_odd}")

        except ValueError:
            messagebox.showerror("Error", "Please enter valid integers separated by commas.")

def run():
    '''
    Initializes and runs the GUI application.
    '''
    root = tk.Tk()
    app = OddNumberFinderApp(root)
    root.mainloop()

if __name__ == "__main__":
    run()

'''
This file contains the implementation of a Python function to find the first odd number in a list and a GUI application using tkinter to interact with the user.
'''
import tkinter as tk
from tkinter import messagebox
def find_first_odd(numbers):
    '''
    Finds the first odd number in a list of numbers.
    Parameters:
    numbers (list): A list of integers.
    Returns:
    int: The first odd number in the list, or None if no odd number is found.
    '''
    for number in numbers:
        if number % 2 != 0:
            return number
    return None
class OddNumberFinderApp:
    '''
    A GUI application to find the first odd number in a list of numbers.
    '''
    def __init__(self, root):
        self.root = root
        self.root.title("First Odd Number Finder")
        self.label = tk.Label(root, text="Enter numbers separated by commas:")
        self.label.pack()
        self.entry = tk.Entry(root, width=50)
        self.entry.pack()
        self.button = tk.Button(root, text="Find First Odd Number", command=self.find_odd)
        self.button.pack()
    def find_odd(self):
        '''
        Retrieves the list of numbers from the entry widget, finds the first odd number, and displays it.
        '''
        try:
            numbers = list(map(int, self.entry.get().split(',')))
            first_odd = find_first_odd(numbers)
            if first_odd is not None:
                messagebox.showinfo("Result", f"The first odd number is: {first_odd}")
            else:
                messagebox.showinfo("Result", "No odd numbers found.")
        except ValueError:
            messagebox.showerror("Error", "Please enter valid integers separated by commas.")
def run():
    '''
    Initializes and runs the GUI application.
    '''
    root = tk.Tk()
    app = OddNumberFinderApp(root)
    root.mainloop()
if __name__ == "__main__":
    run()

'''
This file contains the implementation of a Python function to find the first odd number in a list and a GUI application using tkinter to interact with the user.
'''
import tkinter as tk
from tkinter import messagebox
def find_first_odd(numbers):
    '''
    Finds the first odd number in a list of numbers.
    Parameters:
    numbers (list): A list of integers.
    Returns:
    int: The first odd number in the list, or None if no odd number is found.
    '''
    for number in numbers:
        if number % 2 != 0:
            return number
    return None
class OddNumberFinderApp:
    '''
    A GUI application to find the first odd number in a list of numbers.
    '''
    def __init__(self, root):
        self.root = root
        self.root.title("First Odd Number Finder")
        self.label = tk.Label(root, text="Enter numbers separated by commas:")
        self.label.pack()
        self.entry = tk.Entry(root, width=50)
        self.entry.pack()
        self.button = tk.Button(root, text="Find First Odd Number", command=self.find_odd)
        self.button.pack()
    def find_odd(self):
        '''
        Retrieves the list of numbers from the entry widget, finds the first odd number, and displays it.
        '''
        try:
            numbers = list(map(int, self.entry.get().split(',')))
            first_odd = find_first_odd(numbers)
            if first_odd is not None:
                messagebox.showinfo("Result", f"The first odd number is: {first_odd}")
            else:
                messagebox.showinfo("Result", "No odd numbers found.")
        except ValueError:
            messagebox.showerror("Error", "Please enter valid integers separated by commas.")
def run():
    '''
    Initializes and runs the GUI application.
    '''
    root = tk.Tk()
    app = OddNumberFinderApp(root)
    root.mainloop()
if __name__ == "__main__":
    run()
