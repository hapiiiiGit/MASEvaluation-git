```python
'''

This module contains the main application class and utility function to calculate the difference between the largest and smallest values in a list.

'''

import tkinter as tk
from tkinter import messagebox

def calculate_difference(numbers):
    '''
    Calculate the difference between the largest and smallest values in a list.

    :param numbers: List of numbers
    :return: Difference between the largest and smallest values
    '''
    if not numbers:
        return 0
    largest = max(numbers)
    smallest = min(numbers)
    return largest - smallest

class DifferenceCalculatorApp:
    '''
    A tkinter-based application to find the difference between the largest and smallest values in a list.
    '''
    def __init__(self, master):
        self.master = master
        master.title("Difference Calculator")

        self.label = tk.Label(master, text="Enter numbers separated by commas:")
        self.label.pack()

        self.entry = tk.Entry(master)
        self.entry.pack()

        self.calculate_button = tk.Button(master, text="Calculate Difference", command=self.calculate_difference)
        self.calculate_button.pack()

        self.result_label = tk.Label(master, text="")
        self.result_label.pack()

    def calculate_difference(self):
        '''
        Retrieve user input, calculate the difference, and display the result.
        '''
        try:
            numbers = list(map(float, self.entry.get().split(',')))
            difference = calculate_difference(numbers)
            self.result_label.config(text=f"Difference: {difference}")
        except ValueError:
            messagebox.showerror("Invalid input", "Please enter valid numbers separated by commas.")

if __name__ == "__main__":
    root = tk.Tk()
    app = DifferenceCalculatorApp(root)
    root.mainloop()
```

This module contains the main application class and utility function to calculate the difference between the largest and smallest values in a list.

import tkinter as tk
from tkinter import messagebox

def calculate_difference(numbers):
    '''
    Calculate the difference between the largest and smallest values in a list.

        return 0

    A tkinter-based application to find the difference between the largest and smallest values in a list.
    '''

        master.title("Difference Calculator")

        self.label = tk.Label(master, text="Enter numbers separated by commas:")

'''
This module contains the main application class and utility function to calculate the difference between the largest and smallest values in a list.
'''
import tkinter as tk
from tkinter import messagebox
def calculate_difference(numbers):
    '''
    Calculate the difference between the largest and smallest values in a list.
    :param numbers: List of numbers
    :return: Difference between the largest and smallest values
    '''
    if not numbers:
        return 0
    largest = max(numbers)
    smallest = min(numbers)
    return largest - smallest
class DifferenceCalculatorApp:
    '''
    A tkinter-based application to find the difference between the largest and smallest values in a list.
    '''
    def __init__(self, master):
        self.master = master
        master.title("Difference Calculator")
        self.label = tk.Label(master, text="Enter numbers separated by commas:")
        self.label.pack()
        self.entry = tk.Entry(master)
        self.entry.pack()
        self.calculate_button = tk.Button(master, text="Calculate Difference", command=self.calculate_difference)
        self.calculate_button.pack()
        self.result_label = tk.Label(master, text="")
        self.result_label.pack()
    def calculate_difference(self):
        '''
        Retrieve user input, calculate the difference, and display the result.
        '''
        try:
            numbers = list(map(float, self.entry.get().split(',')))
            difference = calculate_difference(numbers)
            self.result_label.config(text=f"Difference: {difference}")
        except ValueError:
            messagebox.showerror("Invalid input", "Please enter valid numbers separated by commas.")
if __name__ == "__main__":
    root = tk.Tk()
    app = DifferenceCalculatorApp(root)
    root.mainloop()
