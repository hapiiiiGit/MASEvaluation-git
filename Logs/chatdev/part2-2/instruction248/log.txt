def calculate_sum_of_digits(base, power):
    '''
    Calculate the sum of all digits of the base raised to the specified power.
    Args:
        base (int): The base number.
        power (int): The power to raise the base to.
    Returns:
        int: The sum of all digits of the result.
    '''
    # Calculate base raised to the power
    result = base ** power
    # Convert the result to a string and calculate the sum of its digits
    digit_sum = sum(int(digit) for digit in str(result))
    return digit_sum

import tkinter as tk
from tkinter import messagebox
from digit_sum_calculator import calculate_sum_of_digits
class DigitSumApp:
    '''
    Main application class for the digit sum calculator.
    '''
    def __init__(self, root):
        self.root = root
        self.root.title("Digit Sum Calculator")
        self.root.geometry("400x300")
        # Create GUI components
        self.base_label = tk.Label(root, text="Enter Base:")
        self.base_label.pack(pady=10)
        self.base_entry = tk.Entry(root)

        self.result_label = tk.Label(root, text="Result: ", font=("Arial", 12))
        self.result_label.pack(pady=10)

            self.result_label.config(text=f"Result: {result}")

            messagebox.showerror("Input Error", "Please enter valid integers for base and power.")
        except Exception as e:
            messagebox.showerror("Error", str(e))
if __name__ == "__main__":

'''
This file contains the helper function to calculate the sum of digits of base raised to the specified power.
'''
def calculate_sum_of_digits(base, power):
    '''
    Calculate the sum of all digits of the base raised to the specified power.
    Args:
        base (int): The base number.
        power (int): The power to raise the base to.
    Returns:
        int: The sum of all digits of the result.
    '''
    # Calculate base raised to the power
    result = base ** power
    # Convert the result to a string and calculate the sum of its digits
    digit_sum = sum(int(digit) for digit in str(result))
    return digit_sum

def calculate_sum_of_digits(base, power):
    '''
    Calculate the sum of all digits of the base raised to the specified power.
    Args:
        base (int): The base number.
        power (int): The power to raise the base to.

        int: The sum of all digits of the result.
