import tkinter as tk
from triangle_calculator import TriangleCalculator

def create_gui():
    '''
    Sets up the graphical user interface for the application.
    '''
    # Initialize the main window
    root = tk.Tk()
    root.title("Right-Angled Triangle Calculator")
    root.geometry("400x300")

    # Create labels and entry fields for inputs

    side_a_entry.grid(row=0, column=1, padx=10, pady=10)

    tk.Label(root, text="Enter second side (b):").grid(row=1, column=0, padx=10, pady=10)
    side_b_entry = tk.Entry(root)

    tk.Label(root, text="Enter hypotenuse (c):").grid(row=2, column=0, padx=10, pady=10)
    hypotenuse_entry = tk.Entry(root)
    hypotenuse_entry.grid(row=2, column=1, padx=10, pady=10)

    # Create a label for the result
    result_label = tk.Label(root, text="Result: ")
    result_label.grid(row=4, column=0, columnspan=2, padx=10, pady=10)

    # Define the calculation logic
    def calculate():
        try:
            side_a = float(side_a_entry.get()) if side_a_entry.get() else None
            side_b = float(side_b_entry.get()) if side_b_entry.get() else None
            hypotenuse = float(hypotenuse_entry.get()) if hypotenuse_entry.get() else None

            calculator = TriangleCalculator()

            if side_a and side_b:
                result = calculator.calculate_hypotenuse(side_a, side_b)
                result_label.config(text=f"Result: Hypotenuse (c) = {result:.2f}")
            elif hypotenuse and side_a:
                result = calculator.calculate_leg(hypotenuse, side_a)
                result_label.config(text=f"Result: Second side (b) = {result:.2f}")
            elif hypotenuse and side_b:
                result = calculator.calculate_leg(hypotenuse, side_b)
                result_label.config(text=f"Result: First side (a) = {result:.2f}")
            else:

            result_label.config(text="Result: Invalid input. Please enter numeric values.")

    # Create a button to trigger the calculation

            result_label.config(text="Result: Invalid input. Please enter numeric values.")

    calculate_button.grid(row=3, column=0, columnspan=2, padx=10, pady=10)

import math
class TriangleCalculator:
    '''
    A class to calculate the third side of a right-angled triangle using the Pythagorean theorem.
    '''
    @staticmethod
    def calculate_hypotenuse(a, b):
        '''
        Calculates the hypotenuse given the two legs of the triangle.
        :param a: First leg of the triangle.
        :param b: Second leg of the triangle.
        :return: Hypotenuse of the triangle.
        '''
        return math.sqrt(a**2 + b**2)
    @staticmethod
    def calculate_leg(c, a_or_b):
        '''
        Calculates one leg given the hypotenuse and the other leg.
        :param c: Hypotenuse of the triangle.
        :param a_or_b: One leg of the triangle.
        :return: The other leg of the triangle.
        '''
        return math.sqrt(c**2 - a_or_b**2)

import math

class TriangleCalculator:
    '''
    A class to calculate the third side of a right-angled triangle using the Pythagorean theorem.
    '''
    @staticmethod
    def calculate_hypotenuse(a, b):
        '''
        Calculates the hypotenuse given the two legs of the triangle.
        :param a: First leg of the triangle.
        :param b: Second leg of the triangle.
        :return: Hypotenuse of the triangle.
        '''
        if a <= 0 or b <= 0:
            raise ValueError("Sides must be positive numbers.")
        return math.sqrt(a**2 + b**2)

    @staticmethod
    def calculate_leg(c, a_or_b):
        '''
        Calculates one leg given the hypotenuse and the other leg.
        :param c: Hypotenuse of the triangle.
        :param a_or_b: One leg of the triangle.
        :return: The other leg of the triangle.
        '''
        if c <= 0 or a_or_b <= 0:
            raise ValueError("Sides must be positive numbers.")
        if c <= a_or_b:
            raise ValueError("Hypotenuse must be greater than the given side.")
        return math.sqrt(c**2 - a_or_b**2)

'''
This file contains the TriangleCalculator class, which provides methods to calculate the third side of a right-angled triangle.
'''
import math
class TriangleCalculator:
    '''
    A class to calculate the third side of a right-angled triangle using the Pythagorean theorem.
    '''
    @staticmethod
    def calculate_hypotenuse(a, b):
        '''
        Calculates the hypotenuse given the two legs of the triangle.
        :param a: First leg of the triangle.
        :param b: Second leg of the triangle.
        :return: Hypotenuse of the triangle.
        '''
        if a <= 0 or b <= 0:
            raise ValueError("Sides must be positive numbers.")
        return math.sqrt(a**2 + b**2)
    @staticmethod
    def calculate_leg(c, a_or_b):
        '''
        Calculates one leg given the hypotenuse and the other leg.
        :param c: Hypotenuse of the triangle.
        :param a_or_b: One leg of the triangle.
        :return: The other leg of the triangle.
        '''
        if c <= 0 or a_or_b <= 0:
            raise ValueError("Sides must be positive numbers.")
        if c <= a_or_b:
            raise ValueError("Hypotenuse must be greater than the given side.")
        return math.sqrt(c**2 - a_or_b**2)

'''
This file contains the TriangleCalculator class, which provides methods to calculate the third side of a right-angled triangle.
'''
import math
class TriangleCalculator:
    '''
    A class to calculate the third side of a right-angled triangle using the Pythagorean theorem.
    '''
    @staticmethod
    def calculate_hypotenuse(a, b):
        '''
        Calculates the hypotenuse given the two legs of the triangle.
        :param a: First leg of the triangle.
        :param b: Second leg of the triangle.
        :return: Hypotenuse of the triangle.
        '''
        if a <= 0 or b <= 0:
            raise ValueError("Sides must be positive numbers.")
        return math.sqrt(a**2 + b**2)
    @staticmethod
    def calculate_leg(c, a_or_b):
        '''
        Calculates one leg given the hypotenuse and the other leg.
        :param c: Hypotenuse of the triangle.
        :param a_or_b: One leg of the triangle.
        :return: The other leg of the triangle.
        '''
        if c <= 0 or a_or_b <= 0:
            raise ValueError("Sides must be positive numbers.")
        if c <= a_or_b:
            raise ValueError("Hypotenuse must be greater than the given side.")
        return math.sqrt(c**2 - a_or_b**2)

'''
This file contains the TriangleCalculator class, which provides methods to calculate the third side of a right-angled triangle.
'''
import math
class TriangleCalculator:
    '''
    A class to calculate the third side of a right-angled triangle using the Pythagorean theorem.
    '''
    @staticmethod
    def calculate_hypotenuse(a, b):
        '''
        Calculates the hypotenuse given the two legs of the triangle.
        :param a: First leg of the triangle.
        :param b: Second leg of the triangle.
        :return: Hypotenuse of the triangle.
        '''
        if a <= 0 or b <= 0:
            raise ValueError("Sides must be positive numbers.")
        return math.sqrt(a**2 + b**2)
    @staticmethod
    def calculate_leg(c, a_or_b):
        '''
        Calculates one leg given the hypotenuse and the other leg.
        :param c: Hypotenuse of the triangle.
        :param a_or_b: One leg of the triangle.
        :return: The other leg of the triangle.
        '''
        if c <= 0 or a_or_b <= 0:
            raise ValueError("Sides must be positive numbers.")
        if c <= a_or_b:
            raise ValueError("Hypotenuse must be greater than the given side.")
        return math.sqrt(c**2 - a_or_b**2)

import math  

class TriangleCalculator:  
    '''  
    A class to calculate the third side of a right-angled triangle using the Pythagorean theorem.  
    '''  

    @staticmethod  
    def calculate_hypotenuse(a, b):  
        '''  
        Calculates the hypotenuse given the two legs of the triangle.  
        :param a: First leg of the triangle.  
        :param b: Second leg of the triangle.  
        :return: Hypotenuse of the triangle.  
        '''  
        if a <= 0 or b <= 0:  
            raise ValueError("Sides must be positive numbers.")  
        return math.sqrt(a**2 + b**2)  

    @staticmethod  
    def calculate_leg(c, a_or_b):  
        '''  
        Calculates one leg given the hypotenuse and the other leg.  
        :param c: Hypotenuse of the triangle.  
        :param a_or_b: One leg of the triangle.  
        :return: The other leg of the triangle.  
        '''  
        if c <= 0 or a_or_b <= 0:  
            raise ValueError("Sides must be positive numbers.")  
        if c <= a_or_b:  
            raise ValueError("Hypotenuse must be greater than the given side.")  
        return math.sqrt(c**2 - a_or_b**2)

'''  
This file contains the TriangleCalculator class, which provides methods to calculate the third side of a right-angled triangle.  
'''  
import math  
class TriangleCalculator:  
    '''  
    A class to calculate the third side of a right-angled triangle using the Pythagorean theorem.  
    '''  
    @staticmethod  
    def calculate_hypotenuse(a, b):  
        '''  
        Calculates the hypotenuse given the two legs of the triangle.  
        :param a: First leg of the triangle.  
        :param b: Second leg of the triangle.  
        :return: Hypotenuse of the triangle.  
        '''  
        if a <= 0 or b <= 0:  
            raise ValueError("Sides must be positive numbers.")  
        return math.sqrt(a**2 + b**2)  
    @staticmethod  
    def calculate_leg(c, a_or_b):  
        '''  
        Calculates one leg given the hypotenuse and the other leg.  
        :param c: Hypotenuse of the triangle.  
        :param a_or_b: One leg of the triangle.  
        :return: The other leg of the triangle.  
        '''  
        if c <= 0 or a_or_b <= 0:  
            raise ValueError("Sides must be positive numbers.")  
        if c <= a_or_b:  
            raise ValueError("Hypotenuse must be greater than the given side.")  
        return math.sqrt(c**2 - a_or_b**2)
