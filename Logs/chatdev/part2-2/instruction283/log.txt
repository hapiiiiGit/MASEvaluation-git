'''

This module contains the main application logic for checking if a given month has 30 days using a GUI.

'''

import tkinter as tk
from tkinter import messagebox

def has_30_days(month):
    '''
    Determines if the given month number has 30 days.

    Parameters:
    month (int): The month number (1-12).

    Returns:
    bool: True if the month has 30 days, otherwise False.
    '''
    # Months with 30 days: April (4), June (6), September (9), November (11)
    return month in [4, 6, 9, 11]

class MonthCheckerApp:
    '''
    A class to create a GUI application for checking if a month has 30 days.
    '''
    def __init__(self, master):
        '''
        Initializes the MonthCheckerApp with GUI components.

        Parameters:
        master (tk.Tk): The root window of the application.
        '''
        self.master = master
        master.title("Month Checker")

        self.label = tk.Label(master, text="Enter month number (1-12):")
        self.label.pack()

        self.entry = tk.Entry(master)
        self.entry.pack()

        self.check_button = tk.Button(master, text="Check", command=self.check_month)
        self.check_button.pack()

        self.result_label = tk.Label(master, text="")
        self.result_label.pack()

    def check_month(self):
        '''
        Checks if the entered month has 30 days and updates the result label.
        '''
        try:
            month = int(self.entry.get())
            if 1 <= month <= 12:
                if has_30_days(month):
                    self.result_label.config(text=f"Month {month} has 30 days.")
                else:
                    self.result_label.config(text=f"Month {month} does not have 30 days.")
            else:
                messagebox.showerror("Invalid Input", "Please enter a valid month number between 1 and 12.")
        except ValueError:
            messagebox.showerror("Invalid Input", "Please enter a valid integer.")

'''

This module contains the main application logic for checking if a given month has 30 days using a GUI.

'''

import tkinter as tk
from tkinter import messagebox

def has_30_days(month):
    '''
    Determines if the given month number has 30 days.

    Parameters:
    month (int): The month number (1-12).

    Returns:
    bool: True if the month has 30 days, otherwise False.
    '''
    # Months with 30 days: April (4), June (6), September (9), November (11)
    return month in [4, 6, 9, 11]

class MonthCheckerApp:
    '''
    A class to create a GUI application for checking if a month has 30 days.
    '''
    def __init__(self, master):
        '''
        Initializes the MonthCheckerApp with GUI components.

        Parameters:
        master (tk.Tk): The root window of the application.
        '''
        self.master = master
        master.title("Month Checker")

        self.label = tk.Label(master, text="Enter month number (1-12):")
        self.label.pack()

        self.entry = tk.Entry(master)
        self.entry.pack()

        self.check_button = tk.Button(master, text="Check", command=self.check_month)
        self.check_button.pack()

        self.result_label = tk.Label(master, text="")
        self.result_label.pack()

    def check_month(self):
        '''
        Checks if the entered month has 30 days and updates the result label.
        '''
        try:
            month = int(self.entry.get())
            if 1 <= month <= 12:
                if has_30_days(month):
                    self.result_label.config(text=f"Month {month} has 30 days.")
                else:
                    self.result_label.config(text=f"Month {month} does not have 30 days.")
            else:
                messagebox.showerror("Invalid Input", "Please enter a valid month number between 1 and 12.")
        except ValueError:
            messagebox.showerror("Invalid Input", "Please enter a valid integer.")

if __name__ == "__main__":
    root = tk.Tk()
    app = MonthCheckerApp(root)
    root.mainloop()

import tkinter as tk
from tkinter import messagebox
def has_30_days(month):
    '''
    Determines if the given month number has 30 days.
    Parameters:
    month (int): The month number (1-12).
    Returns:
    bool: True if the month has 30 days, otherwise False.

    # Months with 30 days: April (4), June (6), September (9), November (11)

    A class to create a GUI application for checking if a month has 30 days.

        '''

                    self.result_label.config(text=f"Month {month} does not have 30 days.")

                messagebox.showerror("Invalid Input", "Please enter a valid month number between 1 and 12.")
        except ValueError:
            messagebox.showerror("Invalid Input", "Please enter a valid integer.")

This module contains the main application logic for checking if a given month has 30 days using a GUI.
'''
import tkinter as tk
from tkinter import messagebox
def has_30_days(month):
    '''
    Determines if the given month number has 30 days.
    Parameters:
    month (int): The month number (1-12).
    Returns:
    bool: True if the month has 30 days, otherwise False.
    '''
    # Months with 30 days: April (4), June (6), September (9), November (11)
    return month in [4, 6, 9, 11]
class MonthCheckerApp:
    '''
    A class to create a GUI application for checking if a month has 30 days.
    '''
    def __init__(self, master):
        '''
        Initializes the MonthCheckerApp with GUI components.
        Parameters:
        master (tk.Tk): The root window of the application.
        '''
        self.master = master
        master.title("Month Checker")
        self.label = tk.Label(master, text="Enter month number (1-12):")
        self.label.pack()
        self.entry = tk.Entry(master)
        self.entry.pack()
        self.check_button = tk.Button(master, text="Check", command=self.check_month)
        self.check_button.pack()
        self.result_label = tk.Label(master, text="")
        self.result_label.pack()
    def check_month(self):
        '''
        Checks if the entered month has 30 days and updates the result label.
        '''
        try:
            month = int(self.entry.get())
            if 1 <= month <= 12:
                if has_30_days(month):
                    self.result_label.config(text=f"Month {month} has 30 days.")
                else:
                    self.result_label.config(text=f"Month {month} does not have 30 days.")
            else:
                messagebox.showerror("Invalid Input", "Please enter a valid month number between 1 and 12.")
        except ValueError:
            messagebox.showerror("Invalid Input", "Please enter a valid integer.")
if __name__ == "__main__":
    root = tk.Tk()
    app = MonthCheckerApp(root)
    root.mainloop()

'''
This module contains the main application logic for checking if a given month has 30 days using a GUI.
'''
import tkinter as tk
from tkinter import messagebox
def has_30_days(month):
    '''
    Determines if the given month number has 30 days.
    Parameters:
    month (int): The month number (1-12).
    Returns:
    bool: True if the month has 30 days, otherwise False.
    '''
    # Months with 30 days: April (4), June (6), September (9), November (11)
    return month in [4, 6, 9, 11]
class MonthCheckerApp:
    '''
    A class to create a GUI application for checking if a month has 30 days.
    '''
    def __init__(self, master):
        '''
        Initializes the MonthCheckerApp with GUI components.
        Parameters:
        master (tk.Tk): The root window of the application.
        '''
        self.master = master
        master.title("Month Checker")
        self.label = tk.Label(master, text="Enter month number (1-12):")
        self.label.pack()
        self.entry = tk.Entry(master)
        self.entry.pack()
        self.check_button = tk.Button(master, text="Check", command=self.check_month)
        self.check_button.pack()
        self.result_label = tk.Label(master, text="")
        self.result_label.pack()
    def check_month(self):
        '''
        Checks if the entered month has 30 days and updates the result label.
        '''
        try:
            month = int(self.entry.get())
            if 1 <= month <= 12:
                if has_30_days(month):
                    self.result_label.config(text=f"Month {month} has 30 days.")
                else:
                    self.result_label.config(text=f"Month {month} does not have 30 days.")
            else:
                messagebox.showerror("Invalid Input", "Please enter a valid month number between 1 and 12.")
        except ValueError:
            messagebox.showerror("Invalid Input", "Please enter a valid integer.")
if __name__ == "__main__":
    root = tk.Tk()
    app = MonthCheckerApp(root)
    root.mainloop()
