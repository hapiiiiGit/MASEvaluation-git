'''

Main application file for the WordFilterApp using tkinter.

'''

import tkinter as tk
from tkinter import messagebox
from utils import filter_words_starting_with_p

class WordFilterApp:
    def __init__(self, root):
        '''
        Initializes the WordFilterApp GUI components.
        '''
        self.root = root
        self.root.title("Word Filter App")
        
        self.label = tk.Label(root, text="Enter a list of words separated by commas:")
        self.label.pack()
        
        self.entry = tk.Entry(root, width=50)
        self.entry.pack()
        
        self.button = tk.Button(root, text="Filter Words", command=self.filter_words)
        self.button.pack()
        
        self.result_label = tk.Label(root, text="")
        self.result_label.pack()
    
    def filter_words(self):
        '''
        Filters the input list to find words starting with 'p' and displays the result.
        '''
        words = self.entry.get().split(',')
        filtered_words = filter_words_starting_with_p(words)
        if filtered_words:
            self.result_label.config(text=f"Filtered Words: {', '.join(filtered_words)}")
        else:
            messagebox.showinfo("Result", "No words starting with 'p' found.")

if __name__ == "__main__":
    root = tk.Tk()
    app = WordFilterApp(root)
    root.mainloop()

#### utils.py

```python
'''

Utility functions for word filtering.

'''

def filter_words_starting_with_p(words):
    '''
    Takes a list of words and returns two words starting with 'p'.
    
    :param words: List of words to filter.
    :return: List of up to two words starting with 'p'.
    '''
    filtered_words = [word.strip() for word in words if word.strip().lower().startswith('p')]
    return filtered_words[:2]
```

Main application file for the WordFilterApp using tkinter.

import tkinter as tk
from tkinter import messagebox
from utils import filter_words_starting_with_p

class WordFilterApp:
    def __init__(self, root):
        '''
        Initializes the WordFilterApp GUI components.
        '''
        self.root = root
        self.root.title("Word Filter App")
        
        self.label = tk.Label(root, text="Enter a list of words separated by commas:")
        self.label.pack()
        
        self.entry = tk.Entry(root, width=50)
        self.entry.pack()
        
        self.button = tk.Button(root, text="Filter Words", command=self.filter_words)
        self.button.pack()
        
        self.result_label = tk.Label(root, text="")
        self.result_label.pack()
    
    def filter_words(self):
        '''
        Filters the input list to find words starting with 'p' and displays the result.
        '''
        words = self.entry.get().split(',')
        filtered_words = filter_words_starting_with_p(words)
        if filtered_words:
            self.result_label.config(text=f"Filtered Words: {', '.join(filtered_words)}")
        else:
            messagebox.showinfo("Result", "No words starting with 'p' found.")

if __name__ == "__main__":

'''
Main application file for the WordFilterApp using tkinter.

import tkinter as tk
from tkinter import messagebox
from utils import filter_words_starting_with_p
class WordFilterApp:
    def __init__(self, root):
        '''
        Initializes the WordFilterApp GUI components.
        '''
        self.root = root
        self.root.title("Word Filter App")
        self.label = tk.Label(root, text="Enter a list of words separated by commas:")
        self.label.pack()
        self.entry = tk.Entry(root, width=50)
        self.entry.pack()
        self.button = tk.Button(root, text="Filter Words", command=self.filter_words)
        self.button.pack()
        self.result_label = tk.Label(root, text="")
        self.result_label.pack()
    def filter_words(self):
        '''
        Filters the input list to find words starting with 'p' and displays the result.
        '''
        words = self.entry.get().split(',')
        filtered_words = filter_words_starting_with_p(words)
        if filtered_words:
            self.result_label.config(text=f"Filtered Words: {', '.join(filtered_words)}")
        else:
            messagebox.showinfo("Result", "No words starting with 'p' found.")
if __name__ == "__main__":
    root = tk.Tk()
    app = WordFilterApp(root)
    root.mainloop()

A simple and intuitive application to filter words starting with the letter 'p' from a user-provided list.

## Introduction

The **Word Filter App** is a Python-based application with a graphical user interface (GUI) built using the `tkinter` library. The app allows users to input a list of words and filters out up to two words that start with the letter 'p'. It is designed to be lightweight, user-friendly, and requires no external dependencies.

### Main Features

- **Input List of Words**: Users can input a list of words separated by commas.
- **Filter Words Starting with 'p'**: The app filters and displays up to two words starting with the letter 'p'.
- **Interactive GUI**: Built with `tkinter`, the app provides an intuitive interface for user interaction.
- **Error Handling**: Displays a message if no words starting with 'p' are found.

---

## Installation Guide

### Prerequisites

Ensure you have Python installed on your system. The app is compatible with Python 3.6 or later. You can download Python from [python.org](https://www.python.org/).

### Installation Steps

1. **Clone or Download the Repository**:
   - Clone the repository using Git:
     ```bash
     git clone <repository-url>
     ```
   - Or download the ZIP file and extract it to your desired location.

2. **Navigate to the Project Directory**:
   ```bash
   cd <project-directory>
   ```

3. **Install Dependencies**:
   - No external dependencies are required for this project. Ensure Python's standard library is available.

---

## User Guide

### How to Use the App

1. **Launch the Application**:
   - Run the `main.py` file to start the app:
     ```bash
     python main.py
     ```

2. **Input Words**:
   - Enter a list of words separated by commas in the input field. For example:
     ```
     apple, pear, pineapple, banana, peach
     ```

3. **Filter Words**:
   - Click the **Filter Words** button. The app will process the input and display up to two words starting with the letter 'p'.

4. **View Results**:
   - If words starting with 'p' are found, they will be displayed below the button. For example:
     ```
     Filtered Words: pear, pineapple
     ```
   - If no words starting with 'p' are found, a message box will appear stating:
     ```
     No words starting with 'p' found.
     ```

potato, tomato, pepper, apple, peach

### Output:

Filtered Words: potato, pepper
```
