'''

Main application file for counting characters with vowels as neighbors in a given string using a GUI.

'''

import tkinter as tk
from tkinter import messagebox

def count_vowel_neighbors(s):
    '''
    Count characters in the string that have vowels as their neighbors.

    :param s: Input string
    :return: Count of characters with vowels as neighbors
    '''
    vowels = set('aeiouAEIOU')
    count = 0
    for i in range(1, len(s) - 1):
        if s[i - 1] in vowels or s[i + 1] in vowels:
            count += 1
    return count

class MainApplication:
    '''
    Main application class for the GUI.
    '''
    def __init__(self, root):
        '''
        Initialize the main application with GUI components.

        :param root: Tkinter root window
        '''
        self.root = root
        self.root.title("Vowel Neighbor Counter")

        # Input label and entry
        self.label = tk.Label(root, text="Enter a string:")
        self.label.pack(pady=10)

        self.entry = tk.Entry(root, width=50)
        self.entry.pack(pady=10)

        # Submit button
        self.submit_button = tk.Button(root, text="Count", command=self.on_submit)
        self.submit_button.pack(pady=10)

        # Result label
        self.result_label = tk.Label(root, text="")
        self.result_label.pack(pady=10)

    def on_submit(self):
        '''
        Handle the submit button click event.
        '''
        input_string = self.entry.get()
        if not input_string:
            messagebox.showwarning("Input Error", "Please enter a string.")
            return

        count = count_vowel_neighbors(input_string)
        self.result_label.config(text=f"Characters with vowels as neighbors: {count}")

if __name__ == "__main__":
    root = tk.Tk()
    app = MainApplication(root)
    root.mainloop()
```

This code sets up a simple GUI application using `tkinter` where the user can input a string, and upon clicking the "Count" button, it calculates and displays the number of characters that have vowels as their neighbors. The `count_vowel_neighbors` function performs the logic of counting based on the given criteria.

```python
'''

Main application file for counting characters with vowels as neighbors in a given string using a GUI.

'''

import tkinter as tk
from tkinter import messagebox

def count_vowel_neighbors(s):
    '''
    Count characters in the string that have vowels as their neighbors.

    :param s: Input string
    :return: Count of characters with vowels as neighbors
    '''
    vowels = set('aeiouAEIOU')
    count = 0
    for i in range(1, len(s) - 1):
        if s[i - 1] in vowels or s[i + 1] in vowels:
            count += 1
    return count

class MainApplication:
    '''
    Main application class for the GUI.
    '''
    def __init__(self, root):
        '''
        Initialize the main application with GUI components.

        :param root: Tkinter root window
        '''
        self.root = root
        self.root.title("Vowel Neighbor Counter")

        # Input label and entry
        self.label = tk.Label(root, text="Enter a string:")
        self.label.pack(pady=10)

        self.entry = tk.Entry(root, width=50)
        self.entry.pack(pady=10)

        # Submit button
        self.submit_button = tk.Button(root, text="Count", command=self.on_submit)
        self.submit_button.pack(pady=10)

        # Result label
        self.result_label = tk.Label(root, text="")
        self.result_label.pack(pady=10)

    def on_submit(self):
        '''
        Handle the submit button click event.
        '''
        input_string = self.entry.get()
        if not input_string:
            messagebox.showwarning("Input Error", "Please enter a string.")
            return

        count = count_vowel_neighbors(input_string)
        self.result_label.config(text=f"Characters with vowels as neighbors: {count}")

if __name__ == "__main__":
    root = tk.Tk()
    app = MainApplication(root)
    root.mainloop()
```

'''
Main application file for counting characters with vowels as neighbors in a given string using a GUI.
'''
import tkinter as tk
from tkinter import messagebox

def count_vowel_neighbors(s):
    '''
    Count characters in the string that have vowels as their neighbors.
    :param s: Input string
    :return: Count of characters with vowels as neighbors
    '''
    vowels = set('aeiouAEIOU')
    count = 0
    for i in range(1, len(s) - 1):
        if (s[i - 1] in vowels) or (s[i + 1] in vowels):
            count += 1
    return count

class MainApplication:
    '''
    Main application class for the GUI.
    '''
    def __init__(self, root):
        '''
        Initialize the main application with GUI components.
        :param root: Tkinter root window
        '''
        self.root = root
        self.root.title("Vowel Neighbor Counter")
        
        # Input label and entry
        self.label = tk.Label(root, text="Enter a string:")
        self.label.pack(pady=10)
        
        self.entry = tk.Entry(root, width=50)
        self.entry.pack(pady=10)
        
        # Submit button
        self.submit_button = tk.Button(root, text="Count", command=self.on_submit)
        self.submit_button.pack(pady=10)
        
        # Result label
        self.result_label = tk.Label(root, text="")
        self.result_label.pack(pady=10)

    def on_submit(self):
        '''
        Handle the submit button click event.
        '''
        input_string = self.entry.get()
        if not input_string:
            messagebox.showwarning("Input Error", "Please enter a string.")
            return
        count = count_vowel_neighbors(input_string)
        self.result_label.config(text=f"Characters with vowels as neighbors: {count}")

if __name__ == "__main__":
    root = tk.Tk()
    app = MainApplication(root)
    root.mainloop()
```

'''
Main application file for counting characters with vowels as neighbors in a given string using a GUI.
'''
import tkinter as tk
from tkinter import messagebox
def count_vowel_neighbors(s):
    '''
    Count characters in the string that have vowels as their neighbors.
    :param s: Input string
    :return: Count of characters with vowels as neighbors
    '''
    vowels = set('aeiouAEIOU')
    count = 0
    for i in range(1, len(s) - 1):
        if (s[i - 1] in vowels) or (s[i + 1] in vowels):
            count += 1
    return count
class MainApplication:
    '''
    Main application class for the GUI.
    '''
    def __init__(self, root):
        '''
        Initialize the main application with GUI components.
        :param root: Tkinter root window
        '''
        self.root = root
        self.root.title("Vowel Neighbor Counter")
        # Input label and entry
        self.label = tk.Label(root, text="Enter a string:")
        self.label.pack(pady=10)
        self.entry = tk.Entry(root, width=50)

        # Submit button

        if not input_string:
