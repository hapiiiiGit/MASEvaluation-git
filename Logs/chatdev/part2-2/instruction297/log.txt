'''

This module contains utility functions for processing lists.

'''

def pack_consecutive_duplicates(lst):
    '''
    Packs consecutive duplicates of list elements into sublists.

    Parameters:
    lst (list): The input list with potential consecutive duplicates.

    Returns:
    list: A list where consecutive duplicates are packed into sublists.
    '''
    if not lst:
        return []

    packed_list = []
    current_sublist = [lst[0]]

    for item in lst[1:]:
        if item == current_sublist[-1]:
            current_sublist.append(item)
        else:
            packed_list.append(current_sublist)
            current_sublist = [item]

    packed_list.append(current_sublist)
    return packed_list
```

```python
'''

This module contains utility functions for processing lists.

'''

def pack_consecutive_duplicates(lst):
    '''
    Packs consecutive duplicates of list elements into sublists.

    Parameters:
    lst (list): The input list with potential consecutive duplicates.

    Returns:
    list: A list where consecutive duplicates are packed into sublists.
    '''
    if not lst:
        return []

    packed_list = []
    current_sublist = [lst[0]]

    for item in lst[1:]:
        if item == current_sublist[-1]:
            current_sublist.append(item)
        else:
            packed_list.append(current_sublist)
            current_sublist = [item]

    packed_list.append(current_sublist)
    return packed_list

def pack_consecutive_duplicates(lst):
    '''
    Packs consecutive duplicates of list elements into sublists.
    Parameters:
    lst (list): The input list with potential consecutive duplicates.
    Returns:
    list: A list where consecutive duplicates are packed into sublists.

'''
This module contains utility functions for processing lists.
'''
def pack_consecutive_duplicates(lst):
    '''
    Packs consecutive duplicates of list elements into sublists.
    Parameters:
    lst (list): The input list with potential consecutive duplicates.
    Returns:
    list: A list where consecutive duplicates are packed into sublists.
    '''
    if not lst:
        return []
    packed_list = []
    current_sublist = [lst[0]]
    for item in lst[1:]:
        if item == current_sublist[-1]:
            current_sublist.append(item)
        else:
            packed_list.append(current_sublist)
            current_sublist = [item]
    packed_list.append(current_sublist)
    return packed_list

def pack_consecutive_duplicates(lst):
    '''
    Packs consecutive duplicates of list elements into sublists.
    Parameters:
    lst (list): The input list with potential consecutive duplicates.
    Returns:
    list: A list where consecutive duplicates are packed into sublists.

'''
This module contains utility functions for processing lists.
'''
def pack_consecutive_duplicates(lst):
    '''
    Packs consecutive duplicates of list elements into sublists.
    Parameters:
    lst (list): The input list with potential consecutive duplicates.
    Returns:
    list: A list where consecutive duplicates are packed into sublists.
    '''
    if not lst:
        return []
    packed_list = []
    current_sublist = [lst[0]]
    for item in lst[1:]:
        if item == current_sublist[-1]:
            current_sublist.append(item)
        else:
            packed_list.append(current_sublist)
            current_sublist = [item]
    packed_list.append(current_sublist)
    return packed_list

class App:
    def __init__(self, root):
        '''

        root (tk.Tk): The root window of the Tkinter application.
        '''
        self.root = root
        self.root.title("Pack Consecutive Duplicates")
        self.create_widgets()
    def create_widgets(self):
        '''
        Creates and places the widgets in the application window.
        '''
        self.input_label = tk.Label(self.root, text="Enter a list (comma separated):")
        self.input_label.pack()
        self.input_entry = tk.Entry(self.root, width=50)
        self.input_entry.pack()
        self.pack_button = tk.Button(self.root, text="Pack Duplicates", command=self.pack_duplicates)
        self.pack_button.pack()
        self.output_label = tk.Label(self.root, text="Output:")
        self.output_label.pack()
        self.output_text = tk.Text(self.root, height=10, width=50)
        self.output_text.pack()
    def pack_duplicates(self):
        '''
        Handles the button click event to pack duplicates and display the result.
        '''
        input_text = self.input_entry.get()
        try:
            # Safely evaluate the input to convert it into a list
            input_list = ast.literal_eval(input_text)
            if not isinstance(input_list, list):
                raise ValueError("Input is not a list.")
            packed_list = pack_consecutive_duplicates(input_list)
            self.output_text.delete(1.0, tk.END)
            self.output_text.insert(tk.END, str(packed_list))
        except Exception as e:
            messagebox.showerror("Error", f"Invalid input: {e}")

    list: A list where consecutive duplicates are packed into sublists.
    '''
    if not lst:
        return []
    packed_list = []
    current_sublist = [lst[0]]
    for item in lst[1:]:
        if item == current_sublist[-1]:
            current_sublist.append(item)
        else:
            packed_list.append(current_sublist)
            current_sublist = [item]
    packed_list.append(current_sublist)
    return packed_list

    def __init__(self, root):

        Initializes the application with a root window.

        root (tk.Tk): The root window of the Tkinter application.

    def create_widgets(self):
        '''
        Creates and places the widgets in the application window.
        '''
        self.input_label = tk.Label(self.root, text="Enter a list (comma separated):")
        self.input_label.pack()
        self.input_entry = tk.Entry(self.root, width=50)
        self.input_entry.pack()
        self.pack_button = tk.Button(self.root, text="Pack Duplicates", command=self.pack_duplicates)
        self.pack_button.pack()
        self.output_label = tk.Label(self.root, text="Output:")
        self.output_label.pack()
        self.output_text = tk.Text(self.root, height=10, width=50)
        self.output_text.pack()
    def pack_duplicates(self):
        '''
        Handles the button click event to pack duplicates and display the result.
        '''
        input_text = self.input_entry.get()
        try:
            # Safely evaluate the input to convert it into a list
            input_list = ast.literal_eval(input_text)
            if not isinstance(input_list, list):
                raise ValueError("Input is not a list.")
            packed_list = pack_consecutive_duplicates(input_list)
            self.output_text.delete(1.0, tk.END)
            self.output_text.insert(tk.END, str(packed_list))
        except Exception as e:
            messagebox.showerror("Error", f"Invalid input: {e}")
if __name__ == "__main__":
    root = tk.Tk()
