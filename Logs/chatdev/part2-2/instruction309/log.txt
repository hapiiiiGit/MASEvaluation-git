        self.process_button = tk.Button(root, text="Process", command=self.process_input)

'''

Module for processing tuples.

'''

class TupleProcessor:
    def remove_tuples(self, input_tuple, tuples_to_remove):
        '''
        Removes specified tuples from the input tuple.

        :param input_tuple: The original tuple from which tuples will be removed.
        :param tuples_to_remove: A list of tuples to be removed from the input tuple.
        :return: A new tuple with the specified tuples removed.
        '''
        return tuple(item for item in input_tuple if item not in tuples_to_remove)
```

        self.remove_entry.pack()
        self.process_button = tk.Button(root, text="Process", command=self.process_input)

        self.result_label.pack()
    def process_input(self):
        try:
            input_tuple = self.parse_tuple(self.input_entry.get())
            tuples_to_remove = self.parse_tuple(self.remove_entry.get())
            processor = TupleProcessor()
            result = processor.remove_tuples(input_tuple, tuples_to_remove)
            self.result_label.config(text=f"Result: {result}")
        except ValueError as e:
            self.result_label.config(text=f"Error: {str(e)}")
    def parse_tuple(self, input_str):
        # Remove whitespace and check if the input starts and ends with parentheses
        input_str = input_str.strip()
        if not (input_str.startswith('(') and input_str.endswith(')')):
            raise ValueError("Input must be a tuple.")
        # Split the input string by commas and convert to a tuple
        elements = input_str[1:-1].split(',')
        return tuple(element.strip() for element in elements if element.strip())
if __name__ == "__main__":
    root = tk.Tk()
    app = TupleGUI(root)
    root.mainloop()

class TupleProcessor:
    def remove_tuples(self, input_tuple, tuples_to_remove):
        '''

        :param input_tuple: The original tuple from which tuples will be removed.
        :param tuples_to_remove: A list of tuples to be removed from the input tuple.
        :return: A new tuple with the specified tuples removed.
        '''
        return tuple(item for item in input_tuple if item not in tuples_to_remove)

def parse_tuple(self, input_str):
    # Remove whitespace and check if the input starts and ends with parentheses
    input_str = input_str.strip()
    if not (input_str.startswith('(') and input_str.endswith(')')):
        raise ValueError("Input must be a tuple.")
    
    # Use a stack to parse nested tuples
    stack = []
    current = []
    for char in input_str:
        if char == '(':
            stack.append(current)
            current = []
        elif char == ')':
            if stack:
                last = stack.pop()
                last.append(tuple(current))
                current = last
        elif char == ',':
            if current:
                current.append(''.join(current).strip())
                current = []
        else:
            current.append(char)
    
    if stack:
        raise ValueError("Mismatched parentheses in input.")

'''

import tkinter as tk
from tuple_processor import TupleProcessor
class TupleGUI:
    def __init__(self, root):
        self.root = root
        self.root.title("Tuple Processor")
        self.input_label = tk.Label(root, text="Enter a tuple:")
        self.input_label.pack()
        self.input_entry = tk.Entry(root)
        self.input_entry.pack()
        self.remove_label = tk.Label(root, text="Enter tuples to remove (comma-separated):")
        self.remove_label.pack()
        self.remove_entry = tk.Entry(root)
        self.remove_entry.pack()
        self.process_button = tk.Button(root, text="Process", command=self.process_input)
        self.process_button.pack()
        self.result_label = tk.Label(root, text="")
        self.result_label.pack()
    def process_input(self):
        try:
            input_tuple = self.parse_tuple(self.input_entry.get())
            tuples_to_remove = self.parse_tuple(self.remove_entry.get())
            processor = TupleProcessor()
            result = processor.remove_tuples(input_tuple, tuples_to_remove)
            self.result_label.config(text=f"Result: {result}")
        except ValueError as e:
            self.result_label.config(text=f"Error: {str(e)}")
    def parse_tuple(self, input_str):
        # Remove whitespace and check if the input starts and ends with parentheses
        input_str = input_str.strip()
        if not (input_str.startswith('(') and input_str.endswith(')')):
            raise ValueError("Input must be a tuple.")
        # Use a stack to parse nested tuples
        stack = []
        current = []
        temp = ""
        for char in input_str:
            if char == '(':
                if temp:
                    current.append(temp.strip())
                    temp = ""
                stack.append(current)
                current = []
            elif char == ')':
                if temp:
                    current.append(temp.strip())
                    temp = ""
                if stack:
                    last = stack.pop()
                    last.append(tuple(current))
                    current = last
            elif char == ',':
                if temp:
                    current.append(temp.strip())
                    temp = ""
            else:
                temp += char
        if stack:
            raise ValueError("Mismatched parentheses in input.")
        return tuple(current)
if __name__ == "__main__":
    root = tk.Tk()
    app = TupleGUI(root)
    root.mainloop()

'''
Module for processing tuples.

class TupleProcessor:
    def remove_tuples(self, input_tuple, tuples_to_remove):
        '''
        Removes specified tuples from the input tuple.
        :param input_tuple: The original tuple from which tuples will be removed.
        :param tuples_to_remove: A list of tuples to be removed from the input tuple.
        :return: A new tuple with the specified tuples removed.
        '''
        return tuple(item for item in input_tuple if item not in tuples_to_remove)
