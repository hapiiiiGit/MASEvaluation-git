        '''  
        Calculate the sum of numbers in the list within the specified range.  
        '''  
        try:  
            # Parse the input list  
            input_list = self.entry_list.get()  
            numbers = parse_input_list(input_list)  
            # Get the start and end indices  
            start_index = int(self.entry_start.get())  
            end_index = int(self.entry_end.get())  
            # Validate indices  
            if start_index < 0 or end_index >= len(numbers) or start_index > end_index:  
                raise ValueError("Invalid indices. Ensure 0 <= start_index <= end_index < len(list).")  
            # Calculate the sum  
            result = sum(numbers[start_index:end_index + 1])  
            # Display the result  
            self.label_result.config(text=f"Result: {result}")  
        except ValueError as e:  
            # Show an error message if input is invalid  
            messagebox.showerror("Error", str(e))

    '''  
    Parse a comma-separated string into a list of numbers.  
    '''  
    try:  
        # Split the string by commas and convert to integers  
        return [int(num.strip()) for num in input_str.split(",")]  
    except ValueError:  
        raise ValueError("Invalid list format. Please enter a comma-separated list of numbers.")

'''
This file contains the main application logic for a Python GUI program that calculates the sum of numbers in a list within a range specified by two indices.

import tkinter as tk
from tkinter import messagebox
class MainApp:
    '''
    Main application class to create and manage the GUI.
    '''
    def __init__(self, root):
        '''
        Initialize the GUI layout and widgets.
        '''
        self.root = root
        self.root.title("Sum of Numbers in a Range")
        # Input for the list of numbers
        self.label_list = tk.Label(root, text="Enter a list of numbers (comma-separated):")
        self.label_list.pack()
        self.entry_list = tk.Entry(root, width=50)
        self.entry_list.pack()
        # Input for the start index
        self.label_start = tk.Label(root, text="Enter the start index:")
        self.label_start.pack()
        self.entry_start = tk.Entry(root, width=10)
        self.entry_start.pack()
        # Input for the end index
        self.label_end = tk.Label(root, text="Enter the end index:")
        self.label_end.pack()
        self.entry_end = tk.Entry(root, width=10)
        self.entry_end.pack()
        # Button to calculate the sum
        self.button_calculate = tk.Button(root, text="Calculate Sum", command=self.calculate_sum)
        self.button_calculate.pack()
        # Label to display the result
        self.label_result = tk.Label(root, text="Result: ")
        self.label_result.pack()
    def calculate_sum(self):
        '''
        Calculate the sum of numbers in the list within the specified range.
        '''
        try:
            # Parse the input list
            input_list = self.entry_list.get()
            numbers = parse_input_list(input_list)
            # Get the start and end indices
            start_index = int(self.entry_start.get())
            end_index = int(self.entry_end.get())
            # Validate indices
            if start_index < 0 or end_index >= len(numbers) or start_index > end_index:
                raise ValueError("Invalid indices. Ensure 0 <= start_index <= end_index < len(list).")
            # Calculate the sum
            result = sum(numbers[start_index:end_index + 1])
            # Display the result
            self.label_result.config(text=f"Result: {result}")
        except ValueError as e:
            # Show an error message if input is invalid
            messagebox.showerror("Error", str(e))
def parse_input_list(input_str):
    '''
    Parse a comma-separated string into a list of numbers.
    '''
    try:
        # Split the string by commas and convert to integers
        return [int(num.strip()) for num in input_str.split(",")]
    except ValueError:
        raise ValueError("Invalid list format. Please enter a comma-separated list of numbers.")
