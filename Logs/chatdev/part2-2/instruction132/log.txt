Main application file for the Tuple Trimmer GUI.

'''

import tkinter as tk
from tkinter import messagebox
from tuple_trimmer import trim_tuples

class TupleTrimmerApp:
    def __init__(self, root):
        self.root = root
        self.root.title("Tuple Trimmer")

        self.label = tk.Label(root, text="Enter tuples list (e.g., [(1,2,3), (4,5,6)]):")
        self.label.pack()

        self.tuples_entry = tk.Entry(root, width=50)
        self.tuples_entry.pack()

        self.k_label = tk.Label(root, text="Enter k:")
        self.k_label.pack()

        self.k_entry = tk.Entry(root, width=10)
        self.k_entry.pack()

        self.trim_button = tk.Button(root, text="Trim Tuples", command=self.trim_tuples)
        self.trim_button.pack()

        self.result_label = tk.Label(root, text="")
        self.result_label.pack()

    def trim_tuples(self):
        try:
            tuples_list = eval(self.tuples_entry.get())
            k = int(self.k_entry.get())
            trimmed_list = trim_tuples(tuples_list, k)
            self.result_label.config(text=f"Trimmed Tuples: {trimmed_list}")
        except Exception as e:
            messagebox.showerror("Error", f"Invalid input: {e}")

if __name__ == "__main__":
    root = tk.Tk()
    app = TupleTrimmerApp(root)
    root.mainloop()

'''
Function to trim each tuple in the list by k elements.

def trim_tuples(tuples_list, k):
    """
    Trims each tuple in the given list by k elements.
    Parameters:
    tuples_list (list): List of tuples to be trimmed.
    k (int): Number of elements to trim from each tuple.
    Returns:
    list: List of trimmed tuples.
    """
    return [t[:-k] if k < len(t) else () for t in tuples_list]
