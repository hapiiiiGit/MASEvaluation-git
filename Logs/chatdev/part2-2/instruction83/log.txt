'''

Main application file for converting multiple lists to a nested dictionary using a GUI.

'''

import tkinter as tk
from tkinter import messagebox

def convert_lists_to_nested_dict(keys, values):
    '''
    Convert two lists into a nested dictionary.
    
    :param keys: List of keys for the dictionary.
    :param values: List of values for the dictionary.
    :return: Nested dictionary created from the lists.
    '''
    if len(keys) != len(values):
        raise ValueError("Keys and values lists must have the same length.")
    
    nested_dict = {}
    for key, value in zip(keys, values):
        current_level = nested_dict
        for part in key[:-1]:
            current_level = current_level.setdefault(part, {})
        current_level[key[-1]] = value
    
    return nested_dict

class MainApp:
    def __init__(self, root):
        '''
        Initialize the main application window.
        
        :param root: The root window object.
        '''
        self.root = root
        self.root.title("List to Nested Dictionary Converter")
        self.create_widgets()

    def create_widgets(self):
        '''
        Create and place the GUI widgets.
        '''
        self.keys_label = tk.Label(self.root, text="Enter Keys (comma-separated):")
        self.keys_label.pack()

        self.keys_entry = tk.Entry(self.root, width=50)
        self.keys_entry.pack()

        self.values_label = tk.Label(self.root, text="Enter Values (comma-separated):")
        self.values_label.pack()

        self.values_entry = tk.Entry(self.root, width=50)
        self.values_entry.pack()

        self.convert_button = tk.Button(self.root, text="Convert", command=self.convert_to_dict)
        self.convert_button.pack()

        self.result_label = tk.Label(self.root, text="Result:")
        self.result_label.pack()

        self.result_text = tk.Text(self.root, height=10, width=50)
        self.result_text.pack()

    def convert_to_dict(self):
        '''
        Convert the input lists into a nested dictionary and display the result.
        '''
        keys_input = self.keys_entry.get()
        values_input = self.values_entry.get()

        try:
            keys = [tuple(k.strip().split('.')) for k in keys_input.split(',')]
            values = [v.strip() for v in values_input.split(',')]
            result = convert_lists_to_nested_dict(keys, values)
            self.display_result(result)
        except Exception as e:
            messagebox.showerror("Error", str(e))

    def display_result(self, result):
        '''
        Display the resulting nested dictionary in the GUI.

```python
'''

Main application file for converting multiple lists to a nested dictionary using a GUI.

'''

import tkinter as tk
from tkinter import messagebox

def convert_lists_to_nested_dict(keys, values):
    '''
    Convert two lists into a nested dictionary.
    
    :param keys: List of keys for the dictionary.
    :param values: List of values for the dictionary.
    :return: Nested dictionary created from the lists.
    '''
    if len(keys) != len(values):
        raise ValueError("Keys and values lists must have the same length.")
    
    nested_dict = {}
    for key, value in zip(keys, values):
        current_level = nested_dict
        for part in key[:-1]:
            current_level = current_level.setdefault(part, {})
        current_level[key[-1]] = value
    
    return nested_dict

class MainApp:
    def __init__(self, root):
        '''
        Initialize the main application window.
        
        :param root: The root window object.
        '''
        self.root = root
        self.root.title("List to Nested Dictionary Converter")
        self.create_widgets()

    def create_widgets(self):
        '''
        Create and place the GUI widgets.
        '''
        self.keys_label = tk.Label(self.root, text="Enter Keys (comma-separated):")
        self.keys_label.pack()

        self.keys_entry = tk.Entry(self.root, width=50)
        self.keys_entry.pack()

        self.values_label = tk.Label(self.root, text="Enter Values (comma-separated):")
        self.values_label.pack()

        self.values_entry = tk.Entry(self.root, width=50)
        self.values_entry.pack()

        self.convert_button = tk.Button(self.root, text="Convert", command=self.convert_to_dict)
        self.convert_button.pack()

        self.result_label = tk.Label(self.root, text="Result:")
        self.result_label.pack()

        self.result_text = tk.Text(self.root, height=10, width=50)
        self.result_text.pack()

    def convert_to_dict(self):
        '''
        Convert the input lists into a nested dictionary and display the result.
        '''
        keys_input = self.keys_entry.get()
        values_input = self.values_entry.get()

        try:
            keys = [tuple(k.strip().split('.')) for k in keys_input.split(',')]
            values = [v.strip() for v in values_input.split(',')]
            result = convert_lists_to_nested_dict(keys, values)
            self.display_result(result)
        except Exception as e:
            messagebox.showerror("Error", str(e))

    def display_result(self, result):
        '''
        Display the resulting nested dictionary in the GUI.
        
        :param result: The nested dictionary to display.
        '''
        self.result_text.delete(1.0, tk.END)
        self.result_text.insert(tk.END, str(result))

if __name__ == "__main__":
    root = tk.Tk()
    app = MainApp(root)
    root.mainloop()
```

def convert_lists_to_nested_dict(keys, values):
    '''
    Convert two lists into a nested dictionary.
    :param keys: List of keys for the dictionary.
    :param values: List of values for the dictionary.
    :return: Nested dictionary created from the lists.
    '''
    if len(keys) != len(values):
        raise ValueError("Keys and values lists must have the same length.")
    nested_dict = {}
    for key, value in zip(keys, values):
        current_level = nested_dict
        for part in key[:-1]:
            current_level = current_level.setdefault(part, {})
        current_level[key[-1]] = value
    return nested_dict

import tkinter as tk
from tkinter import messagebox
def convert_lists_to_nested_dict(keys, values):
    '''
    Convert two lists into a nested dictionary.
    :param keys: List of keys for the dictionary.
    :param values: List of values for the dictionary.
    :return: Nested dictionary created from the lists.
    '''
    if len(keys) != len(values):
        raise ValueError("Keys and values lists must have the same length.")
    nested_dict = {}
    for key, value in zip(keys, values):
        current_level = nested_dict
        for part in key[:-1]:
            current_level = current_level.setdefault(part, {})
        current_level[key[-1]] = value
    return nested_dict

'''
Main application file for converting multiple lists to a nested dictionary using a GUI.
'''
import tkinter as tk
from tkinter import messagebox
def convert_lists_to_nested_dict(keys, values):
    '''
    Convert two lists into a nested dictionary.
    :param keys: List of keys for the dictionary.
    :param values: List of values for the dictionary.
    :return: Nested dictionary created from the lists.
    '''
    if len(keys) != len(values):
        raise ValueError("Keys and values lists must have the same length.")
    nested_dict = {}
    for key, value in zip(keys, values):
        current_level = nested_dict
        for part in key[:-1]:
            current_level = current_level.setdefault(part, {})
        current_level[key[-1]] = value
    return nested_dict
class MainApp:
    def __init__(self, root):
        '''
        Initialize the main application window.
        :param root: The root window object.
        '''
        self.root = root
        self.root.title("List to Nested Dictionary Converter")
        self.create_widgets()
    def create_widgets(self):
        '''
        Create and place the GUI widgets.
        '''
        self.keys_label = tk.Label(self.root, text="Enter Keys (comma-separated):")
        self.keys_label.pack()
        self.keys_entry = tk.Entry(self.root, width=50)
        self.keys_entry.pack()
        # Add example format for keys input
        self.keys_example = tk.Label(self.root, text="Example: key1.subkey1,key2.subkey2")
        self.keys_example.pack()
        self.values_label = tk.Label(self.root, text="Enter Values (comma-separated):")
        self.values_label.pack()
        self.values_entry = tk.Entry(self.root, width=50)
        self.values_entry.pack()
        self.convert_button = tk.Button(self.root, text="Convert", command=self.convert_to_dict)
        self.convert_button.pack()
        self.result_label = tk.Label(self.root, text="Result:")
        self.result_label.pack()
        self.result_text = tk.Text(self.root, height=10, width=50)
        self.result_text.pack()
    def convert_to_dict(self):
        '''
        Convert the input lists into a nested dictionary and display the result.
        '''
        keys_input = self.keys_entry.get().strip()
        values_input = self.values_entry.get().strip()
        if not keys_input or not values_input:
            messagebox.showerror("Error", "Keys and values cannot be empty.")
            return
        try:
            keys = [tuple(k.strip().split('.')) for k in keys_input.split(',')]
            values = [v.strip() for v in values_input.split(',')]
            if len(keys) != len(values):
                messagebox.showerror("Error", "Keys and values lists must have the same length.")
                return
            result = convert_lists_to_nested_dict(keys, values)
            self.display_result(result)
        except Exception as e:
            messagebox.showerror("Error", f"An error occurred: {str(e)}")
    def display_result(self, result):
        '''
        Display the resulting nested dictionary in the GUI.
        :param result: The nested dictionary to display.
        '''
        self.result_text.delete(1.0, tk.END)
        self.result_text.insert(tk.END, str(result))

    root = tk.Tk()

'''
Main application file for converting multiple lists to a nested dictionary using a GUI.

import tkinter as tk
from tkinter import messagebox
def convert_lists_to_nested_dict(keys, values):
    '''
    Convert two lists into a nested dictionary.
    :param keys: List of keys for the dictionary.
    :param values: List of values for the dictionary.
    :return: Nested dictionary created from the lists.
    '''
    if len(keys) != len(values):
        raise ValueError("Keys and values lists must have the same length.")
    nested_dict = {}
    for key, value in zip(keys, values):
        current_level = nested_dict
        for part in key[:-1]:
            current_level = current_level.setdefault(part, {})
        current_level[key[-1]] = value
    return nested_dict
class MainApp:
    def __init__(self, root):
        '''
        Initialize the main application window.
        :param root: The root window object.
        '''
        self.root = root
        self.root.title("List to Nested Dictionary Converter")
        self.create_widgets()
    def create_widgets(self):
        '''
        Create and place the GUI widgets.
        '''
        self.keys_label = tk.Label(self.root, text="Enter Keys (comma-separated):")
        self.keys_label.pack()
        self.keys_entry = tk.Entry(self.root, width=50)
        self.keys_entry.pack()
        # Add example format for keys input
        self.keys_example = tk.Label(self.root, text="Example: key1.subkey1,key2.subkey2")
        self.keys_example.pack()
        self.values_label = tk.Label(self.root, text="Enter Values (comma-separated):")
        self.values_label.pack()
        self.values_entry = tk.Entry(self.root, width=50)
        self.values_entry.pack()
        self.convert_button = tk.Button(self.root, text="Convert", command=self.convert_to_dict)
        self.convert_button.pack()
        self.result_label = tk.Label(self.root, text="Result:")
        self.result_label.pack()
        self.result_text = tk.Text(self.root, height=10, width=50)
        self.result_text.pack()
    def convert_to_dict(self):
        '''
        Convert the input lists into a nested dictionary and display the result.
        '''
        keys_input = self.keys_entry.get().strip()
        values_input = self.values_entry.get().strip()
        if not keys_input or not values_input:
            messagebox.showerror("Error", "Keys and values cannot be empty.")
            return
        try:
            keys = [tuple(k.strip().split('.')) for k in keys_input.split(',')]
            values = [v.strip() for v in values_input.split(',')]
            if len(keys) != len(values):
                messagebox.showerror("Error", "Keys and values lists must have the same length.")
                return
            result = convert_lists_to_nested_dict(keys, values)
            self.display_result(result)
        except Exception as e:
            messagebox.showerror("Error", f"An error occurred: {str(e)}")
    def display_result(self, result):
        '''
        Display the resulting nested dictionary in the GUI.
        :param result: The nested dictionary to display.
        '''
        self.result_text.delete(1.0, tk.END)
        self.result_text.insert(tk.END, str(result))
if __name__ == "__main__":
    root = tk.Tk()
    app = MainApp(root)
    root.mainloop()
