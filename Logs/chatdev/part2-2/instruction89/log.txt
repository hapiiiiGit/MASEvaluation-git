```python
'''

This script creates a simple GUI application to find the median of three numbers using tkinter.

'''

import tkinter as tk
from tkinter import messagebox

def calculate_median(a, b, c):
    '''
    Calculate the median of three numbers.
    
    Parameters:
    a (float): The first number.
    b (float): The second number.
    c (float): The third number.
    
    Returns:
    float: The median of the three numbers.
    '''
    numbers = [a, b, c]
    numbers.sort()
    return numbers[1]

class MedianFinderApp:
    '''
    A class to create a GUI application for finding the median of three numbers.
    '''
    def __init__(self, master):
        '''
        Initialize the GUI components.
        
        Parameters:
        master (tk.Tk): The root window of the application.
        '''
        self.master = master
        master.title("Median Finder")
        
        self.create_widgets()

    def create_widgets(self):
        '''
        Create and layout the widgets in the GUI.
        '''
        self.label1 = tk.Label(self.master, text="Enter first number:")
        self.label1.grid(row=0, column=0, padx=10, pady=5)

        self.entry1 = tk.Entry(self.master)
        self.entry1.grid(row=0, column=1, padx=10, pady=5)

        self.label2 = tk.Label(self.master, text="Enter second number:")
        self.label2.grid(row=1, column=0, padx=10, pady=5)

        self.entry2 = tk.Entry(self.master)
        self.entry2.grid(row=1, column=1, padx=10, pady=5)

        self.label3 = tk.Label(self.master, text="Enter third number:")
        self.label3.grid(row=2, column=0, padx=10, pady=5)

        self.entry3 = tk.Entry(self.master)
        self.entry3.grid(row=2, column=1, padx=10, pady=5)

        self.calculate_button = tk.Button(self.master, text="Calculate Median", command=self.find_median)
        self.calculate_button.grid(row=3, column=0, columnspan=2, pady=10)

    def find_median(self):
        '''
        Retrieve the numbers from the entries, calculate the median, and display it.
        '''
        try:
            a = float(self.entry1.get())
            b = float(self.entry2.get())
            c = float(self.entry3.get())
            median = calculate_median(a, b, c)
            messagebox.showinfo("Median Result", f"The median is: {median}")
        except ValueError:
            messagebox.showerror("Input Error", "Please enter valid numbers.")

if __name__ == "__main__":
    root = tk.Tk()
    app = MedianFinderApp(root)
    root.mainloop()

'''

This script creates a simple GUI application to find the median of three numbers using tkinter.

'''

import tkinter as tk
from tkinter import messagebox

def calculate_median(a, b, c):
    '''
    Calculate the median of three numbers.
    
    Parameters:
    a (float): The first number.
    b (float): The second number.
    c (float): The third number.
    
    Returns:
    float: The median of the three numbers.
    '''
    numbers = [a, b, c]
    numbers.sort()
    return numbers[1]

class MedianFinderApp:
    '''
    A class to create a GUI application for finding the median of three numbers.
    '''
    def __init__(self, master):
        '''
        Initialize the GUI components.
        
        Parameters:
        master (tk.Tk): The root window of the application.
        '''
        self.master = master
        master.title("Median Finder")
        
        self.create_widgets()

    def create_widgets(self):
        '''
        Create and layout the widgets in the GUI.
        '''
        self.label1 = tk.Label(self.master, text="Enter first number:")
        self.label1.grid(row=0, column=0, padx=10, pady=5)

        self.entry1 = tk.Entry(self.master)
        self.entry1.grid(row=0, column=1, padx=10, pady=5)

        self.label2 = tk.Label(self.master, text="Enter second number:")
        self.label2.grid(row=1, column=0, padx=10, pady=5)

        self.entry2 = tk.Entry(self.master)
        self.entry2.grid(row=1, column=1, padx=10, pady=5)

        self.label3 = tk.Label(self.master, text="Enter third number:")
        self.label3.grid(row=2, column=0, padx=10, pady=5)

        self.entry3 = tk.Entry(self.master)
        self.entry3.grid(row=2, column=1, padx=10, pady=5)

        self.calculate_button = tk.Button(self.master, text="Calculate Median", command=self.find_median)
        self.calculate_button.grid(row=3, column=0, columnspan=2, pady=10)

    def find_median(self):
        '''
        Retrieve the numbers from the entries, calculate the median, and display it.
        '''
        try:
            a = float(self.entry1.get())
            b = float(self.entry2.get())
            c = float(self.entry3.get())
            median = calculate_median(a, b, c)
            messagebox.showinfo("Median Result", f"The median is: {median}")
        except ValueError:
            messagebox.showerror("Input Error", "Please enter valid numbers.")

if __name__ == "__main__":
    root = tk.Tk()
    app = MedianFinderApp(root)
    root.mainloop()

    '''
