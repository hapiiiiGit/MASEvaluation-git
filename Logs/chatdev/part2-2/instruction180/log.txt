import tkinter as tk
from tkinter import messagebox
from utils import calculate_product_of_unique_numbers

class ProductCalculatorApp:
    def __init__(self, root):
        self.root = root
        self.root.title("Product Calculator")
        
        self.label = tk.Label(root, text="Enter numbers separated by commas:")
        self.label.pack()
        
        self.entry = tk.Entry(root, width=50)
        self.entry.pack()
        
        self.calculate_button = tk.Button(root, text="Calculate Product", command=self.on_calculate_button_click)
        self.calculate_button.pack()
        
        self.result_label = tk.Label(root, text="")
        self.result_label.pack()
    
    def on_calculate_button_click(self):
        input_text = self.entry.get()
        try:
            numbers = list(map(int, input_text.split(',')))
            product = calculate_product_of_unique_numbers(numbers)
            self.result_label.config(text=f"Product of unique numbers: {product}")
        except ValueError:
            messagebox.showerror("Invalid Input", "Please enter a valid list of numbers separated by commas.")
    
def main():
    root = tk.Tk()
    app = ProductCalculatorApp(root)
    root.mainloop()

if __name__ == "__main__":
    main()

### utils.py

```python
'''

Utility functions for the Product Calculator App.

'''

def calculate_product_of_unique_numbers(numbers):
    '''
    Calculate the product of unique numbers in a given list.
    
    Parameters:
    numbers (list): A list of integers.
    
    Returns:
    int: The product of unique numbers.
    '''
    unique_numbers = set(numbers)
    product = 1
    for number in unique_numbers:
        product *= number
    return product
```

    def __init__(self, root):
        self.root = root
        self.root.title("Product Calculator")

            numbers = list(map(int, input_text.split(',')))

            self.result_label.config(text=f"Product of unique numbers: {product}")

            messagebox.showerror("Invalid Input", "Please enter a valid list of numbers separated by commas.")

    root = tk.Tk()

def calculate_product_of_unique_numbers(numbers):
    '''
    Calculate the product of unique numbers in a given list.
    
    Parameters:
    numbers (list): A list of integers.
    
    Returns:
    int: The product of unique numbers.
    '''
    unique_numbers = set(numbers)
    product = 1
    for number in unique_numbers:
        product *= number
    return product

Main application file for the Product Calculator App using tkinter.

import tkinter as tk
from tkinter import messagebox
from utils import calculate_product_of_unique_numbers

class ProductCalculatorApp:
    def __init__(self, root):
        self.root = root
        self.root.title("Product Calculator")
        self.label = tk.Label(root, text="Enter numbers separated by commas:")
        self.label.pack()
        self.entry = tk.Entry(root, width=50)
        self.entry.pack()
        self.calculate_button = tk.Button(root, text="Calculate Product", command=self.on_calculate_button_click)
        self.calculate_button.pack()
        self.result_label = tk.Label(root, text="")
        self.result_label.pack()

    def on_calculate_button_click(self):
        input_text = self.entry.get()
        try:
            numbers = list(map(int, input_text.split(',')))
            product = calculate_product_of_unique_numbers(numbers)
            self.result_label.config(text=f"Product of unique numbers: {product}")
        except ValueError:
            messagebox.showerror("Invalid Input", "Please enter a valid list of numbers separated by commas.")

def main():
    root = tk.Tk()
    app = ProductCalculatorApp(root)
    root.mainloop()

if __name__ == "__main__":
    main()
