'''

Main application file for the Cube Average Calculator using Tkinter.

'''

import tkinter as tk
from tkinter import messagebox
from utils import calculate_average_of_cubes

class CubeAverageApp:
    def __init__(self, master):
        self.master = master
        master.title("Cube Average Calculator")

        self.label = tk.Label(master, text="Enter a natural number:")
        self.label.pack()

        self.entry = tk.Entry(master)
        self.entry.pack()

        self.calculate_button = tk.Button(master, text="Calculate Average", command=self.calculate_average)
        self.calculate_button.pack()

        self.result_label = tk.Label(master, text="")
        self.result_label.pack()

    def calculate_average(self):
        try:
            n = int(self.entry.get())
            if n <= 0:
                raise ValueError("The number must be a positive integer.")
            average = calculate_average_of_cubes(n)
            self.display_result(average)
        except ValueError as e:
            messagebox.showerror("Invalid Input", str(e))

    def display_result(self, result):
        self.result_label.config(text=f"Average of cubes: {result:.2f}")

if __name__ == "__main__":
    root = tk.Tk()
    app = CubeAverageApp(root)
    root.mainloop()

Main application file for the Cube Average Calculator using Tkinter.

import tkinter as tk
from tkinter import messagebox
from utils import calculate_average_of_cubes

class CubeAverageApp:
    def __init__(self, master):
        self.master = master
        master.title("Cube Average Calculator")
        self.label = tk.Label(master, text="Enter a natural number:")
        self.label.pack()
        self.entry = tk.Entry(master)
        self.entry.pack()
        self.calculate_button = tk.Button(master, text="Calculate Average", command=self.calculate_average)
        self.calculate_button.pack()
        self.result_label = tk.Label(master, text="")
        self.result_label.pack()

    def calculate_average(self):
        try:
            n = int(self.entry.get())
            if n <= 0:
                raise ValueError("The number must be a positive integer.")
            average = calculate_average_of_cubes(n)
            self.display_result(average)
        except ValueError as e:
            messagebox.showerror("Invalid Input", str(e))

    def display_result(self, result):
        self.result_label.config(text=f"Average of cubes: {result:.2f}")

if __name__ == "__main__":
    root = tk.Tk()
    app = CubeAverageApp(root)
    root.mainloop()
