'''

This is the main file that initializes and runs the Cube Calculator application.
It creates a GUI for the user to input the side length of a cube and calculates its lateral surface area.

'''

import tkinter as tk
from tkinter import messagebox
from calculator import calculate_lateral_surface_area


class CubeCalculatorApp:
    '''
    A class to create a GUI application for calculating the lateral surface area of a cube.
    '''

    def __init__(self, root):
        '''
        Initialize the CubeCalculatorApp with a root window.
        '''
        self.root = root
        self.root.title("Cube Lateral Surface Area Calculator")

        # Create and place widgets
        self.create_widgets()

    def create_widgets(self):
        '''
        Create and place the widgets in the GUI.
        '''
        # Label for side length input
        self.label = tk.Label(self.root, text="Enter the side length of the cube:")
        self.label.pack(pady=10)

        # Entry for side length input
        self.side_length_entry = tk.Entry(self.root, width=20)
        self.side_length_entry.pack(pady=5)

        # Button to calculate lateral surface area
        self.calculate_button = tk.Button(
            self.root, text="Calculate", command=self.calculate_area
        )
        self.calculate_button.pack(pady=10)

        # Label to display the result
        self.result_label = tk.Label(self.root, text="", fg="blue")
        self.result_label.pack(pady=10)

    def calculate_area(self):
        '''
        Calculate the lateral surface area based on the input side length.
        '''
        try:
            # Get the side length from the entry
            side_length = float(self.side_length_entry.get())

            if side_length <= 0:
                raise ValueError("Side length must be positive.")

            # Calculate the lateral surface area
            area = calculate_lateral_surface_area(side_length)

            # Display the result
            self.result_label.config(
                text=f"Lateral Surface Area: {area:.2f} square units"
            )
        except ValueError as e:
            # Show an error message if the input is invalid
            messagebox.showerror("Invalid Input", str(e))


if __name__ == "__main__":
    # Create the main application window
    root = tk.Tk()
    app = CubeCalculatorApp(root)
    root.mainloop()

import tkinter as tk
from tkinter import messagebox
from calculator import calculate_lateral_surface_area
class CubeCalculatorApp:
    '''
    A class to create a GUI application for calculating the lateral surface area of a cube.
    '''
    def __init__(self, root):
        '''
        Initialize the CubeCalculatorApp with a root window.
        '''
        self.root = root
        self.root.title("Cube Lateral Surface Area Calculator")
        # Create and place widgets
        self.create_widgets()
    def create_widgets(self):
        '''
        Create and place the widgets in the GUI.
        '''
        # Label for side length input
        self.label = tk.Label(self.root, text="Enter the side length of the cube:")
        self.label.pack(pady=10)
        # Entry for side length input
        self.side_length_entry = tk.Entry(self.root, width=20)
        self.side_length_entry.pack(pady=5)
        # Button to calculate lateral surface area
        self.calculate_button = tk.Button(
            self.root, text="Calculate", command=self.calculate_area
        )
        self.calculate_button.pack(pady=10)
        # Label to display the result
        self.result_label = tk.Label(self.root, text="", fg="blue")
        self.result_label.pack(pady=10)
    def calculate_area(self):
        '''
        Calculate the lateral surface area based on the input side length.
        '''
        try:
            # Get the side length from the entry
            side_length = float(self.side_length_entry.get())
            if side_length <= 0:
                raise ValueError("Side length must be positive.")
            # Calculate the lateral surface area
            area = calculate_lateral_surface_area(side_length)
            # Display the result
            self.result_label.config(
                text=f"Lateral Surface Area: {area:.2f} square units"
            )
        except ValueError as e:
            # Show an error message if the input is invalid
            messagebox.showerror("Invalid Input", str(e))
if __name__ == "__main__":
    # Create the main application window
    root = tk.Tk()
    app = CubeCalculatorApp(root)
    root.mainloop()

'''
This is the main file that initializes and runs the Cube Calculator application.
It creates a GUI for the user to input the side length of a cube and calculates its lateral surface area.
'''
import tkinter as tk
from tkinter import messagebox
from calculator import calculate_lateral_surface_area
class CubeCalculatorApp:
    '''
    A class to create a GUI application for calculating the lateral surface area of a cube.
    '''
    def __init__(self, root):
        '''
        Initialize the CubeCalculatorApp with a root window.
        '''
        self.root = root
        self.root.title("Cube Lateral Surface Area Calculator")
        # Create and place widgets
        self.create_widgets()
    def create_widgets(self):
        '''
        Create and place the widgets in the GUI.
        '''
        # Label for side length input
        self.label = tk.Label(self.root, text="Enter the side length of the cube:")
        self.label.pack(pady=10)
        # Entry for side length input
        self.side_length_entry = tk.Entry(self.root, width=20)
        self.side_length_entry.pack(pady=5)
        # Button to calculate lateral surface area
        self.calculate_button = tk.Button(
            self.root, text="Calculate", command=self.calculate_area
        )
        self.calculate_button.pack(pady=10)
        # Label to display the result
        self.result_label = tk.Label(self.root, text="", fg="blue")
        self.result_label.pack(pady=10)
    def calculate_area(self):
        '''
        Calculate the lateral surface area based on the input side length.
        '''
        try:
            # Get the side length from the entry
            side_length = float(self.side_length_entry.get())
            if side_length <= 0:
                raise ValueError("Side length must be positive.")
            # Calculate the lateral surface area
            area = calculate_lateral_surface_area(side_length)
            # Display the result
            self.result_label.config(
                text=f"Lateral Surface Area: {area:.2f} square units"
            )
        except ValueError as e:
            # Show an error message if the input is invalid
            messagebox.showerror("Invalid Input", str(e))
if __name__ == "__main__":
    # Create the main application window
    root = tk.Tk()
    app = CubeCalculatorApp(root)
    root.mainloop()

import tkinter as tk
from tkinter import messagebox
from calculator import calculate_lateral_surface_area

class CubeCalculatorApp:
    '''
    A class to create a GUI application for calculating the lateral surface area of a cube.
    '''
    def __init__(self, root):
        '''
        Initialize the CubeCalculatorApp with a root window.
        '''
        self.root = root
        self.root.title("Cube Lateral Surface Area Calculator")
        # Create and place widgets
        self.create_widgets()

    def create_widgets(self):
        '''
        Create and place the widgets in the GUI.
        '''
        # Label for side length input
        self.label = tk.Label(self.root, text="Enter the side length of the cube:")
        self.label.pack(pady=10)
        # Entry for side length input
        self.side_length_entry = tk.Entry(self.root, width=20)
        self.side_length_entry.pack(pady=5)
        # Button to calculate lateral surface area
        self.calculate_button = tk.Button(
            self.root, text="Calculate", command=self.calculate_area
        )
        self.calculate_button.pack(pady=10)
        # Label to display the result
        self.result_label = tk.Label(self.root, text="", fg="blue")
        self.result_label.pack(pady=10)

    def calculate_area(self):
        '''
        Calculate the lateral surface area based on the input side length.
        '''
        try:
            # Get the side length from the entry
            side_length = float(self.side_length_entry.get())
            if side_length <= 0:
                raise ValueError("Side length must be positive.")
            # Calculate the lateral surface area
            area = calculate_lateral_surface_area(side_length)
            # Display the result
            self.result_label.config(
                text=f"Lateral Surface Area: {area:.2f} square units"
            )
        except ValueError as e:
            # Show an error message if the input is invalid
            messagebox.showerror("Invalid Input", str(e))

if __name__ == "__main__":
    # Create the main application window
    root = tk.Tk()

    root.mainloop()

'''

It creates a GUI for the user to input the side length of a cube and calculates its lateral surface area.

import tkinter as tk
from tkinter import messagebox
from calculator import calculate_lateral_surface_area
class CubeCalculatorApp:
    '''
    A class to create a GUI application for calculating the lateral surface area of a cube.
    '''
    def __init__(self, root):
        '''
        Initialize the CubeCalculatorApp with a root window.
        '''
        self.root = root
        self.root.title("Cube Lateral Surface Area Calculator")
        # Create and place widgets
        self.create_widgets()
    def create_widgets(self):
        '''
        Create and place the widgets in the GUI.
        '''
        # Label for side length input
        self.label = tk.Label(self.root, text="Enter the side length of the cube:")
        self.label.pack(pady=10)
        # Entry for side length input
        self.side_length_entry = tk.Entry(self.root, width=20)
        self.side_length_entry.pack(pady=5)
        # Button to calculate lateral surface area
        self.calculate_button = tk.Button(
            self.root, text="Calculate", command=self.calculate_area
        )
        self.calculate_button.pack(pady=10)
        # Label to display the result
        self.result_label = tk.Label(self.root, text="", fg="blue")
        self.result_label.pack(pady=10)
    def calculate_area(self):
        '''
        Calculate the lateral surface area based on the input side length.
        '''
        try:
            # Get the side length from the entry
            side_length = float(self.side_length_entry.get())
            if side_length <= 0:
                raise ValueError("Side length must be positive.")
            # Calculate the lateral surface area
            area = calculate_lateral_surface_area(side_length)
            # Display the result
            self.result_label.config(
                text=f"Lateral Surface Area: {area:.2f} square units"
            )
        except ValueError as e:
            # Show an error message if the input is invalid
            messagebox.showerror("Invalid Input", str(e))
if __name__ == "__main__":
    # Create the main application window
    root = tk.Tk()

    root.mainloop()

# Cube Lateral Surface Area Calculator

This application is a simple and user-friendly tool designed to calculate the lateral surface area of a cube based on the side length provided by the user. It features a graphical user interface (GUI) for ease of use and provides instant results.

## 📋 Main Features

1. **Calculate Lateral Surface Area**: 
   - The application computes the lateral surface area of a cube using the formula: `4 * (side_length ** 2)`.
   - The result is displayed in square units with two decimal precision.

2. **User-Friendly GUI**:
   - The application provides an intuitive interface where users can input the side length of the cube and get the result with a single click.

3. **Error Handling**:
   - The application validates user input to ensure the side length is a positive number.
   - If invalid input is provided, an error message is displayed to guide the user.
