import tkinter as tk
from digit_counter import count_digits_in_string

class DigitCounterApp:
    '''
    A class to create a GUI application for counting digits in a string.
    '''

    def __init__(self, root):
        '''
        Initializes the application with a text entry, a button, and a label to display results.
        '''
        self.root = root
        self.root.title("Digit Counter")

        self.label = tk.Label(root, text="Enter a string:")
        self.label.pack()

        self.entry = tk.Entry(root, width=50)
        self.entry.pack()

        self.button = tk.Button(root, text="Count Digits", command=self.count_digits)
        self.button.pack()

        self.result_label = tk.Label(root, text="")
        self.result_label.pack()

    def count_digits(self):
        '''
        Counts the digits in the input string and updates the result label.
        '''
        input_string = self.entry.get()
        digit_count = count_digits_in_string(input_string)
        self.result_label.config(text=f"Number of digits: {digit_count}")

if __name__ == "__main__":
    root = tk.Tk()
    app = DigitCounterApp(root)
    root.mainloop()
```

#### Utility File: `digit_counter.py`

```python
'''

digit_counter.py

'''

def count_digits_in_string(input_string):
    '''
    Counts the number of digits in the given string.

    Parameters:
    input_string (str): The string to be analyzed.

    Returns:
    int: The number of digits in the string.
    '''
    return sum(char.isdigit() for char in input_string)
```

        self.root.title("Digit Counter")

        self.label = tk.Label(root, text="Enter a string:")
        self.label.pack()

        self.entry = tk.Entry(root, width=50)
        self.entry.pack()

        self.button = tk.Button(root, text="Count Digits", command=self.count_digits)
        self.button.pack()

        self.result_label = tk.Label(root, text="")
        self.result_label.pack()

    def count_digits(self):
        '''
        Counts the digits in the input string and updates the result label.
        '''
        input_string = self.entry.get()
        digit_count = count_digits_in_string(input_string)
        self.result_label.config(text=f"Number of digits: {digit_count}")
