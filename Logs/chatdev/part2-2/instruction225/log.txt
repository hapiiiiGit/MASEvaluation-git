'''

Main application file for the PowerApp GUI application.

'''

import tkinter as tk
from tkinter import messagebox
from power_calculator import compute_nth_power

class PowerApp:
    def __init__(self, root):
        self.root = root
        self.root.title("Power Calculator")

        # Input for numbers
        self.label_numbers = tk.Label(root, text="Enter numbers (comma-separated):")
        self.label_numbers.pack()
        self.entry_numbers = tk.Entry(root, width=50)
        self.entry_numbers.pack()

        # Input for power
        self.label_power = tk.Label(root, text="Enter power (n):")
        self.label_power.pack()
        self.entry_power = tk.Entry(root, width=10)
        self.entry_power.pack()

        # Compute button
        self.button_compute = tk.Button(root, text="Compute", command=self.compute_powers)
        self.button_compute.pack()

        # Result display
        self.label_result = tk.Label(root, text="Result:")
        self.label_result.pack()
        self.text_result = tk.Text(root, height=10, width=50)
        self.text_result.pack()

    def compute_powers(self):
        try:
            numbers = list(map(float, self.entry_numbers.get().split(',')))
            n = int(self.entry_power.get())
            result = compute_nth_power(numbers, n)
            self.display_result(result)
        except ValueError as e:
            messagebox.showerror("Input Error", f"Invalid input: {e}")

    def display_result(self, result):
        self.text_result.delete(1.0, tk.END)
        self.text_result.insert(tk.END, ', '.join(map(str, result)))

if __name__ == "__main__":
    root = tk.Tk()
    app = PowerApp(root)
    root.mainloop()
```

#### power_calculator.py

```python
'''

Helper function to compute the n-th power of each number in a list.

'''

def compute_nth_power(numbers, n):
    '''
    Compute the n-th power of each number in the list.

    :param numbers: List of numbers.
    :param n: The power to which each number is raised.
    :return: List of numbers raised to the n-th power.
    '''
    return [x ** n for x in numbers]
```

        self.label_power = tk.Label(root, text="Enter power (n):")

    return [x ** n for x in numbers]

    :param numbers: List of numbers.
