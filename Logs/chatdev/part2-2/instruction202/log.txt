'''

Main application file for the Tuple Size Calculator.

'''

import tkinter as tk
from tkinter import messagebox
from utils import get_tuple_size

class TupleSizeApp:
    def __init__(self, root):
        self.root = root
        self.root.title("Tuple Size Calculator")
        
        self.label = tk.Label(root, text="Enter elements of the tuple separated by commas:")
        self.label.pack(pady=10)
        
        self.entry = tk.Entry(root, width=50)
        self.entry.pack(pady=10)
        
        self.button = tk.Button(root, text="Calculate Size", command=self.calculate_tuple_size)
        self.button.pack(pady=10)
        
        self.result_label = tk.Label(root, text="")
        self.result_label.pack(pady=10)
    
    def calculate_tuple_size(self):
        try:
            # Convert input string to a tuple
            tuple_input = tuple(map(str.strip, self.entry.get().split(',')))
            size = get_tuple_size(tuple_input)
            self.result_label.config(text=f"Size of tuple: {size} bytes")
        except Exception as e:
            messagebox.showerror("Error", f"An error occurred: {e}")
    
    def run(self):
        self.root.mainloop()

if __name__ == "__main__":
    root = tk.Tk()
    app = TupleSizeApp(root)
    app.run()
```

    Returns:
    int: The size of the tuple in bytes.
    '''
    return sys.getsizeof(tpl)

```python
'''

Main application file for the Tuple Size Calculator.

'''

import tkinter as tk
from tkinter import messagebox
from utils import get_tuple_size

class TupleSizeApp:
    def __init__(self, root):
        self.root = root
        self.root.title("Tuple Size Calculator")
        
        self.label = tk.Label(root, text="Enter elements of the tuple separated by commas:")
        self.label.pack(pady=10)
        
        self.entry = tk.Entry(root, width=50)
        self.entry.pack(pady=10)
        
        self.button = tk.Button(root, text="Calculate Size", command=self.calculate_tuple_size)
        self.button.pack(pady=10)
        
        self.result_label = tk.Label(root, text="")
        self.result_label.pack(pady=10)
    
    def calculate_tuple_size(self):
        try:
            # Convert input string to a tuple
            tuple_input = tuple(map(str.strip, self.entry.get().split(',')))
            size = get_tuple_size(tuple_input)
            self.result_label.config(text=f"Size of tuple: {size} bytes")
        except Exception as e:
            messagebox.showerror("Error", f"An error occurred: {e}")
    
    def run(self):
        self.root.mainloop()

if __name__ == "__main__":
    root = tk.Tk()
    app = TupleSizeApp(root)
    app.run()
```

    Parameters:

    Returns:
    int: The size of the tuple in bytes.
    '''

import tkinter as tk
from tkinter import messagebox
from utils import get_tuple_size
class TupleSizeApp:
    def __init__(self, root):
        self.root = root
        self.root.title("Tuple Size Calculator")
        self.label = tk.Label(root, text="Enter elements of the tuple separated by commas:")
        self.label.pack(pady=10)
        self.entry = tk.Entry(root, width=50)
        self.entry.pack(pady=10)
        self.button = tk.Button(root, text="Calculate Size", command=self.calculate_tuple_size)
        self.button.pack(pady=10)
        self.result_label = tk.Label(root, text="")
        self.result_label.pack(pady=10)
    def calculate_tuple_size(self):
        try:
            # Convert input string to a tuple
            tuple_input = tuple(map(str.strip, self.entry.get().split(',')))
            size = get_tuple_size(tuple_input)
            self.result_label.config(text=f"Size of tuple: {size} bytes")
        except Exception as e:
            messagebox.showerror("Error", f"An error occurred: {e}")
    def run(self):
        self.root.mainloop()
if __name__ == "__main__":
    root = tk.Tk()
    app = TupleSizeApp(root)
    app.run()

'''
Main application file for the Tuple Size Calculator.

import tkinter as tk
from tkinter import messagebox
from utils import get_tuple_size
class TupleSizeApp:
    def __init__(self, root):
        self.root = root
        self.root.title("Tuple Size Calculator")
        self.label = tk.Label(root, text="Enter elements of the tuple separated by commas:")
        self.label.pack(pady=10)
        self.entry = tk.Entry(root, width=50)
        self.entry.pack(pady=10)
        self.button = tk.Button(root, text="Calculate Size", command=self.calculate_tuple_size)
        self.button.pack(pady=10)
        self.result_label = tk.Label(root, text="")
        self.result_label.pack(pady=10)
    def calculate_tuple_size(self):
        try:
            # Convert input string to a tuple
            tuple_input = tuple(map(str.strip, self.entry.get().split(',')))
            size = get_tuple_size(tuple_input)
            self.result_label.config(text=f"Size of tuple: {size} bytes")
        except Exception as e:
            messagebox.showerror("Error", f"An error occurred: {e}")
    def run(self):
        self.root.mainloop()
if __name__ == "__main__":
    root = tk.Tk()
    app = TupleSizeApp(root)
    app.run()
