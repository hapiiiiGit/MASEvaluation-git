import tkinter as tk
from tkinter import messagebox
class TupleConverterApp:
    def __init__(self, master):
        self.master = master
        self.master.title("Tuple to Dictionary Converter")
        # Update the label text to provide clearer instructions
        self.label = tk.Label(master, text="Enter a tuple of integers (e.g., 1,2,3,4):")
        self.label.pack()
        self.entry = tk.Entry(master)
        self.entry.pack()
        self.convert_button = tk.Button(master, text="Convert", command=self.on_convert_button_click)
        self.convert_button.pack()
        self.result_label = tk.Label(master, text="")
        self.result_label.pack()
    # Enhance error handling in convert_tuple_to_dict method
    def convert_tuple_to_dict(self, tuple_input):
        try:
            tuple_elements = tuple(map(int, tuple_input.split(',')))
            if len(tuple_elements) % 2 != 0:
                raise ValueError("Tuple must have an even number of elements.")
            return {tuple_elements[i]: tuple_elements[i + 1] for i in range(0, len(tuple_elements), 2)}
        except ValueError as ve:
            messagebox.showerror("Error", f"Value Error: {ve}")
        except Exception as e:
            messagebox.showerror("Error", f"Unexpected Error: {e}")
        return None
    def on_convert_button_click(self):
        tuple_input = self.entry.get()
        result_dict = self.convert_tuple_to_dict(tuple_input)
        if result_dict is not None:
            self.result_label.config(text=f"Result: {result_dict}")

'''
TupleConverterApp class to manage the GUI and tuple conversion logic.
'''
import tkinter as tk
from tkinter import messagebox
class TupleConverterApp:
    def __init__(self, master):
        self.master = master
        self.master.title("Tuple to Dictionary Converter")
        # Update the label text to provide clearer instructions
        self.label = tk.Label(master, text="Enter a tuple of integers (e.g., 1,2,3,4):")
        self.label.pack()
        self.entry = tk.Entry(master)
        self.entry.pack()
        self.convert_button = tk.Button(master, text="Convert", command=self.on_convert_button_click)
        self.convert_button.pack()
        self.result_label = tk.Label(master, text="")
        self.result_label.pack()
    # Enhance error handling in convert_tuple_to_dict method
    def convert_tuple_to_dict(self, tuple_input):
        try:
            # Strip whitespace and split the input string
            tuple_elements = tuple_input.replace(" ", "").split(',')
            # Convert elements to integers and check for non-integer values
            tuple_elements = [int(element) for element in tuple_elements]
            if len(tuple_elements) % 2 != 0:
                raise ValueError("Tuple must have an even number of elements.")
            return {tuple_elements[i]: tuple_elements[i + 1] for i in range(0, len(tuple_elements), 2)}
        except ValueError as ve:
            messagebox.showerror("Error", f"Value Error: {ve}")
        except Exception as e:
            messagebox.showerror("Error", f"Unexpected Error: {e}")
        return None
    def on_convert_button_click(self):
        tuple_input = self.entry.get()
        result_dict = self.convert_tuple_to_dict(tuple_input)
        if result_dict is not None:
            self.result_label.config(text=f"Result: {result_dict}")
