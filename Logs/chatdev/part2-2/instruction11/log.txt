'''

This module contains the main application class and utility function to find the n largest items from a list.

'''

import tkinter as tk
from tkinter import messagebox

def get_n_largest_items(lst, n):
    '''
    Returns the n largest items from the list.

    Parameters:
    lst (list): The list from which to find the largest items.
    n (int): The number of largest items to return.

    Returns:
    list: A list containing the n largest items.
    '''
    if n <= 0:
        return []
    return sorted(lst, reverse=True)[:n]

class App:
    '''
    The main application class for the GUI.
    '''
    def __init__(self, root):
        '''
        Initializes the GUI components.
        
        Parameters:
        root (tk.Tk): The root window of the application.
        '''

        self.root.title("N Largest Items Finder")

        # Input for list
        self.list_label = tk.Label(root, text="Enter a list of numbers (comma-separated):")
        self.list_label.pack()
        self.list_entry = tk.Entry(root, width=50)
        self.list_entry.pack()

        # Input for n
        self.n_label = tk.Label(root, text="Enter the number of largest items to find:")
        self.n_label.pack()
        self.n_entry = tk.Entry(root, width=10)
        self.n_entry.pack()

        # Button to trigger the computation
        self.compute_button = tk.Button(root, text="Find N Largest", command=self.get_n_largest)
        self.compute_button.pack()

        # Output label
        self.result_label = tk.Label(root, text="")
        self.result_label.pack()

    def get_n_largest(self):
        '''
        Retrieves the list and integer n from user input, computes the n largest items, and displays the result.
        '''
        try:
            # Parse the list input

    '''

    '''
    Returns the n largest items from the list.

'''
This module contains the main application class and utility function to find the n largest items from a list.
'''
import tkinter as tk
from tkinter import messagebox
def get_n_largest_items(lst, n):
    '''
    Returns the n largest items from the list.
    Parameters:
    lst (list): The list from which to find the largest items.
    n (int): The number of largest items to return.
    Returns:
    list: A list containing the n largest items.
    '''
    if n <= 0:
        return []
    return sorted(lst, reverse=True)[:n]
class App:
    '''
    The main application class for the GUI.
    '''
    def __init__(self, root):
        '''
        Initializes the GUI components.
        Parameters:
        root (tk.Tk): The root window of the application.
        '''
        self.root = root
        self.root.title("N Largest Items Finder")
        # Input for list
        self.list_label = tk.Label(root, text="Enter a list of numbers (comma-separated):")
        self.list_label.pack()
        self.list_entry = tk.Entry(root, width=50)
        self.list_entry.pack()
        # Input for n
        self.n_label = tk.Label(root, text="Enter the number of largest items to find:")
        self.n_label.pack()
        self.n_entry = tk.Entry(root, width=10)
        self.n_entry.pack()
        # Button to trigger the computation
        self.compute_button = tk.Button(root, text="Find N Largest", command=self.get_n_largest)
        self.compute_button.pack()
        # Output label
        self.result_label = tk.Label(root, text="")
        self.result_label.pack()
    def get_n_largest(self):
        '''
        Retrieves the list and integer n from user input, computes the n largest items, and displays the result.
        '''
        try:
            # Parse the list input
            lst = list(map(int, self.list_entry.get().split(',')))
            # Parse the n input
            n = int(self.n_entry.get())
            # Get the n largest items
            result = get_n_largest_items(lst, n)
            # Display the result
            self.result_label.config(text=f"The {n} largest items are: {result}")
        except ValueError:
            messagebox.showerror("Invalid input", "Please enter valid numbers.")
if __name__ == "__main__":
    root = tk.Tk()
    app = App(root)
    root.mainloop()
