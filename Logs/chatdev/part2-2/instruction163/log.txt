This file contains the main application class FirstDigitFinderApp which creates the GUI and handles user interactions.

import tkinter as tk
from logic import find_first_digit_of_number

class FirstDigitFinderApp:
    def __init__(self, root):
        self.root = root
        self.root.title("First Digit Finder")
        self.create_widgets()

    def create_widgets(self):
        self.label = tk.Label(self.root, text="Enter a number:")
        self.label.pack()

        self.entry = tk.Entry(self.root)
        self.entry.pack()

        self.button = tk.Button(self.root, text="Find First Digit", command=self.find_first_digit)
        self.button.pack()

        self.result_label = tk.Label(self.root, text="")
        self.result_label.pack()

    def find_first_digit(self):
        number = self.entry.get()
        try:
            first_digit = find_first_digit_of_number(int(number))
            self.result_label.config(text=f"First digit: {first_digit}")
        except ValueError:
            self.result_label.config(text="Please enter a valid integer.")

if __name__ == "__main__":
    root = tk.Tk()
    app = FirstDigitFinderApp(root)
    root.mainloop()
```

def find_first_digit_of_number(number):
    '''

    '''
    number_str = str(abs(number))  # Convert to string and handle negative numbers
    return int(number_str[0])

        self.root.title("First Digit Finder")
        self.create_widgets()

    def create_widgets(self):
        self.label = tk.Label(self.root, text="Enter a number:")
        self.label.pack()

        self.entry = tk.Entry(self.root)
        self.entry.pack()

        self.button = tk.Button(self.root, text="Find First Digit", command=self.find_first_digit)
        self.button.pack()

        self.result_label = tk.Label(self.root, text="")
        self.result_label.pack()

    def find_first_digit(self):
        number = self.entry.get()
        try:
            first_digit = find_first_digit_of_number(int(number))
            self.result_label.config(text=f"First digit: {first_digit}")
        except ValueError:
            self.result_label.config(text="Please enter a valid integer.")

if __name__ == "__main__":
    root = tk.Tk()
    app = FirstDigitFinderApp(root)
    root.mainloop()

        self.label.pack()

        self.entry.pack()

        self.button = tk.Button(self.root, text="Find First Digit", command=self.find_first_digit)

        self.result_label.pack()

    def find_first_digit(self):
        number = self.entry.get()
        try:
            first_digit = find_first_digit_of_number(int(number))
            self.result_label.config(text=f"First digit: {first_digit}")
        except ValueError:
            self.result_label.config(text="Please enter a valid integer.")

if __name__ == "__main__":
    root = tk.Tk()

    root.mainloop()

'''
This file contains the main application class FirstDigitFinderApp which creates the GUI and handles user interactions.
'''
import tkinter as tk
from logic import find_first_digit_of_number
class FirstDigitFinderApp:
    def __init__(self, root):
        self.root = root
        self.root.title("First Digit Finder")
        self.create_widgets()
    def create_widgets(self):
        self.label = tk.Label(self.root, text="Enter a number:")
        self.label.pack()
        self.entry = tk.Entry(self.root)
        self.entry.pack()
        self.button = tk.Button(self.root, text="Find First Digit", command=self.find_first_digit)
        self.button.pack()
        self.result_label = tk.Label(self.root, text="")
        self.result_label.pack()
    def find_first_digit(self):
        number = self.entry.get()
        try:
            first_digit = find_first_digit_of_number(int(number))
            self.result_label.config(text=f"First digit: {first_digit}")
        except ValueError:
            self.result_label.config(text="Please enter a valid integer.")
if __name__ == "__main__":
    root = tk.Tk()
    app = FirstDigitFinderApp(root)
    root.mainloop()
