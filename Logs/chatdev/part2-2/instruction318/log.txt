'''

This file contains the MainApplication class which sets up the GUI and handles user interactions,
and the count_equal_numbers function which calculates the number of equal integers from three given inputs.

'''

import tkinter as tk
from tkinter import messagebox

def count_equal_numbers(a, b, c):
    """
    Counts the number of equal numbers among three given integers.
    
    Parameters:
    a (int): First integer
    b (int): Second integer
    c (int): Third integer
    
    Returns:
    int: The count of equal numbers
    """
    if a == b == c:
        return 3
    elif a == b or b == c or a == c:
        return 2
    else:
        return 0

class MainApplication(tk.Tk):
    def __init__(self):
        super().__init__()
        self.title("Equal Numbers Counter")
        
        # Create and place labels and entry widgets
        self.label_a = tk.Label(self, text="Enter first integer:")
        self.label_a.grid(row=0, column=0, padx=10, pady=5)
        
        self.entry_a = tk.Entry(self)
        self.entry_a.grid(row=0, column=1, padx=10, pady=5)
        
        self.label_b = tk.Label(self, text="Enter second integer:")
        self.label_b.grid(row=1, column=0, padx=10, pady=5)
        
        self.entry_b = tk.Entry(self)
        self.entry_b.grid(row=1, column=1, padx=10, pady=5)
        
        self.label_c = tk.Label(self, text="Enter third integer:")
        self.label_c.grid(row=2, column=0, padx=10, pady=5)
        
        self.entry_c = tk.Entry(self)
        self.entry_c.grid(row=2, column=1, padx=10, pady=5)
        
        # Create and place the button
        self.button = tk.Button(self, text="Count Equal Numbers", command=self.count_equal_numbers)
        self.button.grid(row=3, column=0, columnspan=2, pady=10)
        
    def count_equal_numbers(self):
        try:
            a = int(self.entry_a.get())
            b = int(self.entry_b.get())
            c = int(self.entry_c.get())
            result = count_equal_numbers(a, b, c)
            messagebox.showinfo("Result", f"Number of equal integers: {result}")
        except ValueError:
            messagebox.showerror("Invalid input", "Please enter valid integers.")

if __name__ == "__main__":
    app = MainApplication()
    app.mainloop()

```

This code sets up a simple GUI application using `tkinter`. It allows users to input three integers and calculates the number of equal integers using the `count_equal_numbers` function. The result is displayed in a message box.

```python

'''

This file contains the MainApplication class which sets up the GUI and handles user interactions,
and the count_equal_numbers function which calculates the number of equal integers from three given inputs.

'''

import tkinter as tk
from tkinter import messagebox

def count_equal_numbers(a, b, c):
    """
    Counts the number of equal numbers among three given integers.
    
    Parameters:
    a (int): First integer
    b (int): Second integer
    c (int): Third integer
    
    Returns:
    int: The count of equal numbers
    """
    if a == b == c:
        return 3
    elif a == b or b == c or a == c:
        return 2
    else:
        return 0

class MainApplication(tk.Tk):
    def __init__(self):
        super().__init__()
        self.title("Equal Numbers Counter")
        
        # Create and place labels and entry widgets
        self.label_a = tk.Label(self, text="Enter first integer:")
        self.label_a.grid(row=0, column=0, padx=10, pady=5)
        
        self.entry_a = tk.Entry(self)
        self.entry_a.grid(row=0, column=1, padx=10, pady=5)
        
        self.label_b = tk.Label(self, text="Enter second integer:")
        self.label_b.grid(row=1, column=0, padx=10, pady=5)
        
        self.entry_b = tk.Entry(self)
        self.entry_b.grid(row=1, column=1, padx=10, pady=5)
        
        self.label_c = tk.Label(self, text="Enter third integer:")
        self.label_c.grid(row=2, column=0, padx=10, pady=5)
        
        self.entry_c = tk.Entry(self)
        self.entry_c.grid(row=2, column=1, padx=10, pady=5)
        
        # Create and place the button
        self.button = tk.Button(self, text="Count Equal Numbers", command=self.count_equal_numbers)
        self.button.grid(row=3, column=0, columnspan=2, pady=10)
        
    def count_equal_numbers(self):
        try:
            a = int(self.entry_a.get())
            b = int(self.entry_b.get())
            c = int(self.entry_c.get())
            result = count_equal_numbers(a, b, c)
            messagebox.showinfo("Result", f"Number of equal integers: {result}")
        except ValueError:
            messagebox.showerror("Invalid input", "Please enter valid integers.")

if __name__ == "__main__":
    app = MainApplication()
    app.mainloop()

import tkinter as tk
from tkinter import messagebox
def count_equal_numbers(a, b, c):
    """
    Counts the number of equal numbers among three given integers.
    Parameters:
    a (int): First integer
    b (int): Second integer
    c (int): Third integer
    Returns:
    int: The count of equal numbers

    elif a == b or b == c or a == c:
        return 2
    else:
        return 0

        super().__init__()
        self.title("Equal Numbers Counter")
        # Create and place labels and entry widgets
        self.label_a = tk.Label(self, text="Enter first integer:")

'''
This file contains the MainApplication class which sets up the GUI and handles user interactions,
and the count_equal_numbers function which calculates the number of equal integers from three given inputs.
'''
import tkinter as tk
from tkinter import messagebox
def count_equal_numbers(a, b, c):
    """
    Counts the number of equal numbers among three given integers.
    Parameters:
    a (int): First integer
    b (int): Second integer
    c (int): Third integer
    Returns:
    int: The count of equal numbers
    """
    if a == b == c:
        return 3
    elif a == b or b == c or a == c:
        return 2
    else:
        return 0
class MainApplication(tk.Tk):
    def __init__(self):
        super().__init__()
        self.title("Equal Numbers Counter")
        # Create and place labels and entry widgets
        self.label_a = tk.Label(self, text="Enter first integer:")
        self.label_a.grid(row=0, column=0, padx=10, pady=5)
        self.entry_a = tk.Entry(self)
        self.entry_a.grid(row=0, column=1, padx=10, pady=5)
        self.label_b = tk.Label(self, text="Enter second integer:")
        self.label_b.grid(row=1, column=0, padx=10, pady=5)
        self.entry_b = tk.Entry(self)
        self.entry_b.grid(row=1, column=1, padx=10, pady=5)
        self.label_c = tk.Label(self, text="Enter third integer:")
        self.label_c.grid(row=2, column=0, padx=10, pady=5)
        self.entry_c = tk.Entry(self)
        self.entry_c.grid(row=2, column=1, padx=10, pady=5)
        # Create and place the button
        self.button = tk.Button(self, text="Count Equal Numbers", command=self.count_equal_numbers)
        self.button.grid(row=3, column=0, columnspan=2, pady=10)
    def count_equal_numbers(self):
        try:
            a = int(self.entry_a.get())
            b = int(self.entry_b.get())
            c = int(self.entry_c.get())
            result = count_equal_numbers(a, b, c)
            messagebox.showinfo("Result", f"Number of equal integers: {result}")
        except ValueError:
            messagebox.showerror("Invalid input", "Please enter valid integers.")
if __name__ == "__main__":
    app = MainApplication()
    app.mainloop()

## Overview

The Equal Numbers Counter is a Python application that allows users to input three integers and calculates how many of them are equal. This application is built using the Tkinter library for the GUI, providing an easy-to-use interface for users to interact with.

## Main Functions

  - `3` if all three numbers are equal.
  - `2` if any two numbers are equal.
  - `0` if all numbers are different.
