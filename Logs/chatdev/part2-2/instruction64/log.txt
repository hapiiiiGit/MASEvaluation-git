'''

This file contains the main application logic for finding the maximum value in a heterogeneous list.
It includes a GUI built with tkinter and a utility function to compute the maximum value.

'''

import tkinter as tk
from tkinter import messagebox
from utils import find_max_in_list


class MainApp:
    '''
    The MainApp class creates the GUI for the application and handles user interactions.
    '''
    def __init__(self, root):
        '''
        Initializes the GUI components.
        :param root: The root window of the tkinter application.
        '''
        self.root = root
        self.root.title("Find Maximum in Heterogeneous List")

        # Input label and entry
        self.label = tk.Label(root, text="Enter a list (e.g., [1, 'a', 3.5, 2]):")
        self.label.pack(pady=5)

        self.entry = tk.Entry(root, width=50)
        self.entry.pack(pady=5)

        # Find Max button
        self.button = tk.Button(root, text="Find Max", command=self.find_max)
        self.button.pack(pady=10)

        # Result label
        self.result_label = tk.Label(root, text="", fg="blue")
        self.result_label.pack(pady=5)

    def find_max(self):
        '''
        Processes the user input, computes the maximum value, and displays the result.
        '''
        user_input = self.entry.get()
        try:
            # Convert the input string to a Python list
            data_list = eval(user_input)
            if not isinstance(data_list, list):
                raise ValueError("Input is not a list.")

            # Find the maximum value in the list
            max_value = find_max_in_list(data_list)
            self.result_label.config(text=f"Maximum Value: {max_value}")

        except Exception as e:
            messagebox.showerror("Error", f"Invalid input: {e}")
            self.result_label.config(text="")


if __name__ == "__main__":
    root = tk.Tk()
    app = MainApp(root)
    root.mainloop()

def find_max_in_list(data_list):
    '''
    Finds the maximum value in a heterogeneous list.

        # Filter out non-comparable elements
        comparable_elements = [x for x in data_list if isinstance(x, (int, float, str))]
        if not comparable_elements:
            raise ValueError("No comparable elements in the list.")

        return max(comparable_elements)
    except Exception as e:
        raise ValueError(f"Error finding maximum: {e}")

def find_max_in_list(data_list):
    '''
    Finds the maximum value in a heterogeneous list.
    Non-comparable elements are ignored.
    :param data_list: The input list containing elements of various data types.
    :return: The maximum value among comparable elements.

def find_max_in_list(data_list):  
    '''  
    Finds the maximum value in a heterogeneous list.  
    Non-comparable elements are ignored.  
    :param data_list: The input list containing elements of various data types.  
    :return: The maximum value among comparable elements.  
    '''  
    try:  
        # Filter out non-comparable elements  
        comparable_elements = [x for x in data_list if isinstance(x, (int, float, str))]  
        if not comparable_elements:  
            raise ValueError("No comparable elements in the list.")  
        # Find and return the maximum value  
        return max(comparable_elements, key=lambda x: (isinstance(x, str), x))

def find_max_in_list(data_list):  
    '''  
    Finds the maximum value in a heterogeneous list.  
    Non-comparable elements are ignored.  
    :param data_list: The input list containing elements of various data types.  
    :return: The maximum value among comparable elements.  
    '''  
    try:  
        # Filter out non-comparable elements  
        comparable_elements = [x for x in data_list if isinstance(x, (int, float, str))]  
        if not comparable_elements:  
            raise ValueError("No comparable elements in the list.")  
        # Find and return the maximum value  
        return max(comparable_elements, key=lambda x: (isinstance(x, str), x))  
    except Exception as e:  
        raise ValueError(f"Error finding maximum: {e}")

    '''

    '''
    def __init__(self, root):
        '''
        Initializes the GUI components.

        '''
        self.root = root

def find_max_in_list(data_list):
    '''
    Finds the maximum value in a heterogeneous list.
    Non-comparable elements are ignored.
    :param data_list: The input list containing elements of various data types.
    :return: The maximum value among comparable elements.
    '''
    try:
        # Filter out non-comparable elements
        comparable_elements = [x for x in data_list if isinstance(x, (int, float, str))]
        if not comparable_elements:
            raise ValueError("No comparable elements in the list.")
        # Find and return the maximum value
        return max(comparable_elements, key=lambda x: (isinstance(x, str), x))
    except Exception as e:
        raise ValueError(f"Error finding maximum: {e}")
