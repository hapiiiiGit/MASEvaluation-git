'''

import tkinter as tk
from tkinter import messagebox
from utils import replace_spaces_in_string
class ReplaceSpacesApp:
    '''
    The ReplaceSpacesApp class manages the GUI and handles user inputs for replacing spaces in a string.
    '''
    def __init__(self, root):
        self.root = root
        self.root.title("Replace Spaces in String")
        self.root.geometry("400x300")
        # Input String Label and Entry
        self.label_input_string = tk.Label(root, text="Enter String:")
        self.label_input_string.pack(pady=10)
        self.entry_input_string = tk.Entry(root, width=40)
        self.entry_input_string.pack(pady=5)
        # Replacement Character Label and Entry
        self.label_replacement_char = tk.Label(root, text="Enter Replacement Character:")
        self.label_replacement_char.pack(pady=10)
        self.entry_replacement_char = tk.Entry(root, width=5)
        self.entry_replacement_char.pack(pady=5)
        # Replace Button
        self.button_replace = tk.Button(root, text="Replace Spaces", command=self.replace_spaces)
        self.button_replace.pack(pady=20)
        # Output Label
        self.label_output = tk.Label(root, text="Output String:")
        self.label_output.pack(pady=10)
        self.text_output = tk.Text(root, height=5, width=40)
        self.text_output.pack(pady=5)
    def replace_spaces(self):
        '''
        Handles the logic for replacing spaces in the string and updating the output in the GUI.
        '''
        input_string = self.entry_input_string.get()
        replacement_char = self.entry_replacement_char.get()
        if len(replacement_char) != 1:
            messagebox.showerror("Error", "Replacement character must be a single character.")
            return
        output_string = replace_spaces_in_string(input_string, replacement_char)
        self.text_output.delete("1.0", tk.END)
        self.text_output.insert(tk.END, output_string)
if __name__ == "__main__":
    root = tk.Tk()
    app = ReplaceSpacesApp(root)
    root.mainloop()
