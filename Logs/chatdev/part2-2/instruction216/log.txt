This script creates a GUI application using tkinter that processes a list of lists.
The application takes input from the user, processes it to separate the first and second elements
of each sublist into two separate lists, and displays the result.
'''
import tkinter as tk
from tkinter import messagebox
import ast
class ListProcessorApp:
    def __init__(self, master):
        self.master = master
        master.title("List Processor")
        # Input label and text entry
        self.label = tk.Label(master, text="Enter a list of lists (e.g., [[1, 2], [3, 4]]):")
        self.label.pack()
        self.entry = tk.Entry(master, width=50)
        self.entry.pack()
        # Process button
        self.process_button = tk.Button(master, text="Process", command=self.process_input)
        self.process_button.pack()
        # Output label
        self.output_label = tk.Label(master, text="Output:")
        self.output_label.pack()
        self.output_text = tk.Text(master, height=10, width=50)
        self.output_text.pack()
    def process_input(self):
        try:
            # Get input from entry
            input_str = self.entry.get()
            input_list = ast.literal_eval(input_str)  # Safely convert string input to list
            # Validate input
            if not all(isinstance(sublist, list) and len(sublist) == 2 for sublist in input_list):
                raise ValueError("Input must be a list of lists with two elements each.")
            # Process input
            result = self.split_lists(input_list)
            # Display output
            self.output_text.delete(1.0, tk.END)
            self.output_text.insert(tk.END, f"First elements: {result[0]}\nSecond elements: {result[1]}")
        except Exception as e:
            messagebox.showerror("Error", str(e))
    def split_lists(self, input_list):
        first_elements = [sublist[0] for sublist in input_list]
        second_elements = [sublist[1] for sublist in input_list]
        return [first_elements, second_elements]
if __name__ == "__main__":
    root = tk.Tk()
    app = ListProcessorApp(root)
    root.mainloop()

This script creates a GUI application using tkinter that processes a list of lists.
The application takes input from the user, processes it to separate the first and second elements
of each sublist into two separate lists, and displays the result.
'''
import tkinter as tk
from tkinter import messagebox
import ast
class ListProcessorApp:
    def __init__(self, master):
        self.master = master
        master.title("List Processor")
        # Input label and text entry
        self.label = tk.Label(master, text="Enter a list of lists (e.g., [[1, 2], [3, 4]]):")
        self.label.pack()
        self.entry = tk.Entry(master, width=50)
        self.entry.pack()
        # Process button
        self.process_button = tk.Button(master, text="Process", command=self.process_input)
        self.process_button.pack()
        # Output label
        self.output_label = tk.Label(master, text="Output:")
        self.output_label.pack()
        self.output_text = tk.Text(master, height=10, width=50)
        self.output_text.pack()
    def process_input(self):
        try:
            # Get input from entry
            input_str = self.entry.get()
            input_list = ast.literal_eval(input_str)  # Safely convert string input to list
            # Validate input
            if not all(isinstance(sublist, list) and len(sublist) == 2 for sublist in input_list):
                raise ValueError("Input must be a list of lists with two elements each.")
            # Process input
            result = self.split_lists(input_list)
            # Display output
            self.output_text.delete(1.0, tk.END)
            self.output_text.insert(tk.END, f"First elements: {result[0]}\nSecond elements: {result[1]}")
        except Exception as e:
            messagebox.showerror("Error", str(e))
    def split_lists(self, input_list):
        first_elements = [sublist[0] for sublist in input_list]
        second_elements = [sublist[1] for sublist in input_list]
