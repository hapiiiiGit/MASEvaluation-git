'''

This module contains the PatternMatcher class which provides functionality to match a string pattern using regular expressions.

'''

import re

class PatternMatcher:
    def match_pattern(self, input_string):
        """
        Matches a string that has an 'a' followed by one or more 'b's.

        :param input_string: The string to be checked.
        :return: True if the string matches the pattern, False otherwise.
        """
        pattern = r'a+b+'
        return bool(re.match(pattern, input_string))
```

'''
This module contains the PatternMatcher class which provides functionality to match a string pattern using regular expressions.
'''
import re
class PatternMatcher:
    def match_pattern(self, input_string):
        """
        Matches a string that has an 'a' followed by one or more 'b's.
        :param input_string: The string to be checked.
        :return: True if the string matches the pattern, False otherwise.
        """
        pattern = r'ab+'
        return bool(re.search(pattern, input_string))

'''
This module contains the PatternMatcher class which provides functionality to match a string pattern using regular expressions.
'''
import re
class PatternMatcher:
    def match_pattern(self, input_string):
        """
        Matches a string that has an 'a' followed by one or more 'b's.
        :param input_string: The string to be checked.
        :return: True if the string matches the pattern, False otherwise.
        """
        pattern = r'ab+'
        return bool(re.search(pattern, input_string))
