```python
'''

This module contains the main application class for the Capital Space Inserter App.
It uses tkinter for the GUI and provides functionality to insert spaces before capital letters in a given string.

'''

import tkinter as tk

def add_spaces_to_capitals(input_string):
    '''
    This function takes a string as input and returns a new string with spaces inserted before each capital letter.
    
    Parameters:
    input_string (str): The input string to process.
    
    Returns:
    str: The processed string with spaces before capital letters.
    '''
    result = []
    for char in input_string:
        if char.isupper():
            result.append(' ')
        result.append(char)
    return ''.join(result).strip()

class CapitalSpaceInserterApp:
    '''
    This class creates a GUI application to insert spaces before capital letters in a given string.
    '''
    
    def __init__(self, master):
        '''
        Initializes the GUI components.
        
        Parameters:
        master (tk.Tk): The root window of the tkinter application.
        '''
        self.master = master
        master.title("Capital Space Inserter")

        self.label = tk.Label(master, text="Enter your string:")
        self.label.pack()

        self.entry = tk.Entry(master, width=50)
        self.entry.pack()

        self.process_button = tk.Button(master, text="Process", command=self.insert_spaces)
        self.process_button.pack()

        self.result_label = tk.Label(master, text="Processed string:")
        self.result_label.pack()

        self.result = tk.Label(master, text="", wraplength=400)
        self.result.pack()

    def insert_spaces(self):
        '''
        Processes the input string and updates the output with spaces inserted before capital letters.
        '''
        input_string = self.entry.get()
        processed_string = add_spaces_to_capitals(input_string)
        self.result.config(text=processed_string)

if __name__ == "__main__":
    root = tk.Tk()
    app = CapitalSpaceInserterApp(root)
    root.mainloop()
```

```python
'''

This module contains the main application class for the Capital Space Inserter App.
It uses tkinter for the GUI and provides functionality to insert spaces before capital letters in a given string.

'''

import tkinter as tk

def add_spaces_to_capitals(input_string):
    '''
    This function takes a string as input and returns a new string with spaces inserted before each capital letter.
    
    Parameters:
    input_string (str): The input string to process.
    
    Returns:
    str: The processed string with spaces before capital letters.
    '''
    result = []
    for char in input_string:
        if char.isupper():
            result.append(' ')
        result.append(char)
    return ''.join(result).strip()

class CapitalSpaceInserterApp:
    '''
    This class creates a GUI application to insert spaces before capital letters in a given string.
    '''
    
    def __init__(self, master):
        '''
        Initializes the GUI components.
        
        Parameters:
        master (tk.Tk): The root window of the tkinter application.
        '''
        self.master = master
        master.title("Capital Space Inserter")

        self.label = tk.Label(master, text="Enter your string:")
        self.label.pack()

        self.entry = tk.Entry(master, width=50)
        self.entry.pack()

        self.process_button = tk.Button(master, text="Process", command=self.insert_spaces)
        self.process_button.pack()

        self.result_label = tk.Label(master, text="Processed string:")
        self.result_label.pack()

        self.result = tk.Label(master, text="", wraplength=400)
        self.result.pack()

    def insert_spaces(self):
        '''
        Processes the input string and updates the output with spaces inserted before capital letters.
        '''
        input_string = self.entry.get()
        processed_string = add_spaces_to_capitals(input_string)
        self.result.config(text=processed_string)

if __name__ == "__main__":
    root = tk.Tk()
    app = CapitalSpaceInserterApp(root)
    root.mainloop()
```

'''
This module contains the main application class for the Capital Space Inserter App.
It uses tkinter for the GUI and provides functionality to insert spaces before capital letters in a given string.
'''
import tkinter as tk

def add_spaces_to_capitals(input_string):
    '''
    This function takes a string as input and returns a new string with spaces inserted before each capital letter.
    
    Parameters:
    input_string (str): The input string to process.
    
    Returns:
    str: The processed string with spaces before capital letters.
    '''
    result = []
    for char in input_string:
        if char.isupper():
            result.append(' ')
        result.append(char)
    return ''.join(result).strip()

class CapitalSpaceInserterApp:
    '''
    This class creates a GUI application to insert spaces before capital letters in a given string.
    '''
    def __init__(self, master):
        '''
        Initializes the GUI components.
        
        Parameters:
        master (tk.Tk): The root window of the tkinter application.
        '''
        self.master = master
        master.title("Capital Space Inserter")
        self.label = tk.Label(master, text="Enter your string:")
        self.label.pack()
        self.entry = tk.Entry(master, width=50)
        self.entry.pack()
        self.process_button = tk.Button(master, text="Process", command=self.insert_spaces)
        self.process_button.pack()
        self.result_label = tk.Label(master, text="Processed string:")
        self.result_label.pack()
        self.result = tk.Label(master, text="", wraplength=400)
        self.result.pack()

    def insert_spaces(self):
        '''
        Processes the input string and updates the output with spaces inserted before capital letters.
        '''
        input_string = self.entry.get()
        processed_string = add_spaces_to_capitals(input_string)
        self.result.config(text=processed_string)

if __name__ == "__main__":
    root = tk.Tk()
    app = CapitalSpaceInserterApp(root)
    root.mainloop()

'''
This module contains the main application class for the Capital Space Inserter App.
It uses tkinter for the GUI and provides functionality to insert spaces before capital letters in a given string.
'''
import tkinter as tk
def add_spaces_to_capitals(input_string):
    '''
    This function takes a string as input and returns a new string with spaces inserted before each capital letter.
    Parameters:
    input_string (str): The input string to process.
    Returns:
    str: The processed string with spaces before capital letters.
    '''
    result = []
    for char in input_string:
        if char.isupper():
            result.append(' ')
        result.append(char)
    return ''.join(result).strip()
class CapitalSpaceInserterApp:
    '''
    This class creates a GUI application to insert spaces before capital letters in a given string.
    '''
    def __init__(self, master):
        '''
        Initializes the GUI components.
        Parameters:
        master (tk.Tk): The root window of the tkinter application.
        '''
        self.master = master
        master.title("Capital Space Inserter")
        self.label = tk.Label(master, text="Enter your string:")
        self.label.pack()
        self.entry = tk.Entry(master, width=50)
        self.entry.pack()
        self.process_button = tk.Button(master, text="Process", command=self.insert_spaces)
        self.process_button.pack()
        self.result_label = tk.Label(master, text="Processed string:")
        self.result_label.pack()
        self.result = tk.Label(master, text="", wraplength=400)
        self.result.pack()
    def insert_spaces(self):
        '''
        Processes the input string and updates the output with spaces inserted before capital letters.
        '''
        input_string = self.entry.get()
        processed_string = add_spaces_to_capitals(input_string)
        self.result.config(text=processed_string)
if __name__ == "__main__":
    root = tk.Tk()
    app = CapitalSpaceInserterApp(root)
    root.mainloop()
