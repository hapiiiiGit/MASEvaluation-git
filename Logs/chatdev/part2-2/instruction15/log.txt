import tkinter as tk

class TupleCounterApp:
    def __init__(self, root):
        self.root = root
        self.root.title("Tuple Counter Application")
        
        # Input Label

    def process_input(self):
        # Get input from the user

        self.output_label = tk.Label(root, text="Resulting Dictionary:")

'''
def count_tuples(tuple_list):
    """
    Count the occurrences of each unique tuple in the list.
    Args:
        tuple_list (list): A list of tuples.
    Returns:
        dict: A dictionary mapping each unique tuple to its occurrence count.
    """
    tuple_count = {}
    for tup in tuple_list:
        if tup in tuple_count:
            tuple_count[tup] += 1
        else:
            tuple_count[tup] = 1
    return tuple_count

'''
This file contains the function `count_tuples`, which takes a list of tuples as input and returns a dictionary mapping each unique tuple to the number of times it occurs in the list.
'''
def count_tuples(tuple_list):
    """
    Count the occurrences of each unique tuple in the list.
    Args:
        tuple_list (list): A list of tuples.
    Returns:
        dict: A dictionary mapping each unique tuple to its occurrence count.
    """
    tuple_count = {}
    for tup in tuple_list:
        if tup in tuple_count:
            tuple_count[tup] += 1
        else:
            tuple_count[tup] = 1
    return tuple_count

'''
This file contains the main application logic for the Tuple Counter GUI. It uses tkinter to create a graphical user interface that allows users to input a list of tuples, process it, and display the resulting dictionary.
'''
import tkinter as tk
from tkinter import messagebox
from tuple_counter import count_tuples
class TupleCounterApp:
    def __init__(self, root):
        '''
        Initialize the Tuple Counter Application GUI.
        Args:
            root (tk.Tk): The root tkinter window.
        '''
        self.root = root
        self.root.title("Tuple Counter Application")
        # Input Label
        self.input_label = tk.Label(root, text="Enter a list of tuples (e.g., [(1,2), (3,4), (1,2)]:")
        self.input_label.pack(pady=5)
        # Input Field
        self.input_field = tk.Entry(root, width=50)
        self.input_field.pack(pady=5)
        # Process Button
        self.process_button = tk.Button(root, text="Process", command=self.process_input)
        self.process_button.pack(pady=10)
        # Output Label
        self.output_label = tk.Label(root, text="Resulting Dictionary:")
        self.output_label.pack(pady=5)
        # Output Field
        self.output_field = tk.Text(root, height=10, width=50)
        self.output_field.pack(pady=5)
    def process_input(self):
        '''
        Process the user input, validate it, and display the resulting dictionary.
        '''
        # Get input from the user
        user_input = self.input_field.get()
        try:
            # Convert the input string to a list of tuples
            tuple_list = eval(user_input)
            # Ensure the input is a list of tuples
            if not isinstance(tuple_list, list) or not all(isinstance(item, tuple) for item in tuple_list):
                raise ValueError("Input must be a list of tuples.")
            # Process the list of tuples
            result_dict = count_tuples(tuple_list)
            # Display the resulting dictionary
            self.output_field.delete("1.0", tk.END)
            self.output_field.insert(tk.END, str(result_dict))
        except Exception as e:
            # Show error message if input is invalid
            messagebox.showerror("Error", f"Invalid input: {e}")
if __name__ == "__main__":
    root = tk.Tk()
    app = TupleCounterApp(root)
    root.mainloop()

'''
def count_tuples(tuple_list):
    '''
    Count the occurrences of each unique tuple in the list.
    Args:
        tuple_list (list): A list of tuples.
    Returns:
        dict: A dictionary mapping each unique tuple to its occurrence count.
    '''
    tuple_count = {}
    for tup in tuple_list:
        if tup in tuple_count:
            tuple_count[tup] += 1
        else:
            tuple_count[tup] = 1
    return tuple_count

class TupleCounterApp:
    def __init__(self, root):
        '''
        Initialize the Tuple Counter Application GUI.
        Args:
            root (tk.Tk): The root tkinter window.
        '''
        self.root = root
        self.root.title("Tuple Counter Application")
        # Input Label
        self.input_label = tk.Label(root, text="Enter a list of tuples (e.g., [(1,2), (3,4), (1,2)]:")

        self.input_field = tk.Entry(root, width=50)

        self.process_button = tk.Button(root, text="Process", command=self.process_input)
        self.process_button.pack(pady=10)
        # Output Label
        self.output_label = tk.Label(root, text="Resulting Dictionary:")

        Process the user input, validate it, and display the resulting dictionary.

def count_tuples(tuple_list):
    '''
    Count the occurrences of each unique tuple in the list.
    Args:
