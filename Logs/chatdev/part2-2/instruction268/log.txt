'''

This file contains the main application logic for the Sector Area Calculator.
It initializes the GUI and connects it with the calculation logic.

'''

import tkinter as tk
from calculator import SectorAreaCalculator

class SectorAreaApp:
    def __init__(self, master):
        self.master = master
        master.title("Sector Area Calculator")

        self.label_radius = tk.Label(master, text="Radius:")
        self.label_radius.pack()

        self.entry_radius = tk.Entry(master)
        self.entry_radius.pack()

        self.label_angle = tk.Label(master, text="Angle:")
        self.label_angle.pack()

        self.entry_angle = tk.Entry(master)
        self.entry_angle.pack()

        self.calculate_button = tk.Button(master, text="Calculate Area", command=self.calculate_and_display_area)
        self.calculate_button.pack()

        self.result_label = tk.Label(master, text="")
        self.result_label.pack()

    def calculate_and_display_area(self):
        try:
            radius = float(self.entry_radius.get())
            angle = float(self.entry_angle.get())
            calculator = SectorAreaCalculator()
            area = calculator.calculate_area(radius, angle)
            if area is None:
                self.result_label.config(text="Angle must be less than or equal to 360 degrees.")
            else:
                self.result_label.config(text=f"Area of sector: {area:.2f}")

import tkinter as tk
from calculator import SectorAreaCalculator
class SectorAreaApp:
    def __init__(self, master):
        self.master = master
        master.title("Sector Area Calculator")
        self.label_radius = tk.Label(master, text="Radius:")
        self.label_radius.pack()
        self.entry_radius = tk.Entry(master)
        self.entry_radius.pack()
        self.label_angle = tk.Label(master, text="Angle:")
        self.label_angle.pack()
        self.entry_angle = tk.Entry(master)
        self.entry_angle.pack()
        self.calculate_button = tk.Button(master, text="Calculate Area", command=self.calculate_and_display_area)
        self.calculate_button.pack()
        self.result_label = tk.Label(master, text="")
        self.result_label.pack()
    def calculate_and_display_area(self):
        try:
            radius = float(self.entry_radius.get())
            angle = float(self.entry_angle.get())
            calculator = SectorAreaCalculator()
            area = calculator.calculate_area(radius, angle)
            if area is None:
                self.result_label.config(text="Angle must be less than or equal to 360 degrees.")
            else:
                self.result_label.config(text=f"Area of sector: {area:.2f}")
        except ValueError:
            self.result_label.config(text="Please enter valid numbers.")
if __name__ == "__main__":
    root = tk.Tk()
    app = SectorAreaApp(root)
    root.mainloop()

'''
This file contains the logic for calculating the area of a sector.
It defines the SectorAreaCalculator class with a method to perform the calculation.
'''
import math
class SectorAreaCalculator:
    def calculate_area(self, radius, angle):
        '''
        Calculate the area of a sector given the radius and angle.
        Returns None if the angle is larger than 360 degrees.
        '''
        if angle > 360:
            return None
        return (angle / 360) * math.pi * radius ** 2
