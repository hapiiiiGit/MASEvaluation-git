'''

Main application file for the Adverb Finder GUI.

'''

import tkinter as tk
from tkinter import messagebox
from adverb_finder import find_first_adverb

class AdverbFinderApp:
    def __init__(self, master):
        self.master = master
        master.title("Adverb Finder")

        self.label = tk.Label(master, text="Enter a sentence:")
        self.label.pack()

        self.entry = tk.Entry(master, width=50)
        self.entry.pack()

        self.find_button = tk.Button(master, text="Find Adverb", command=self.find_adverb)
        self.find_button.pack()

        self.result_label = tk.Label(master, text="", fg="blue")
        self.result_label.pack()

    def find_adverb(self):
        sentence = self.entry.get()
        adverb, position = find_first_adverb(sentence)
        if adverb:
            self.result_label.config(text=f"First adverb: '{adverb}' at position {position}")
        else:
            self.result_label.config(text="No adverb found.")
            messagebox.showinfo("Result", "No adverb found in the sentence.")

if __name__ == "__main__":
    root = tk.Tk()
    app = AdverbFinderApp(root)
    root.mainloop()
```

#### adverb_finder.py

```python
'''

Module to find the first adverb in a given sentence.

'''

import nltk
from nltk import pos_tag, word_tokenize

# Ensure that the necessary NLTK resources are downloaded
nltk.download('averaged_perceptron_tagger')
nltk.download('punkt')

def find_first_adverb(sentence):
    '''
    Finds the first adverb in the given sentence and returns it along with its position.

    Parameters:
    sentence (str): The sentence to search for an adverb.

    Returns:
    tuple: A tuple containing the first adverb and its position, or (None, None) if no adverb is found.
    '''
    words = word_tokenize(sentence)
    tagged_words = pos_tag(words)

    for index, (word, tag) in enumerate(tagged_words):
        if tag.startswith('RB'):  # 'RB' is the POS tag for adverbs
            return word, index

    return None, None
```

'''

Module to find the first adverb in a given sentence.

'''

import nltk
from nltk import pos_tag, word_tokenize

# Ensure that the necessary NLTK resources are downloaded
nltk.download('averaged_perceptron_tagger')
nltk.download('punkt')

def find_first_adverb(sentence):
    '''
    Finds the first adverb in the given sentence and returns it along with its position.

    Parameters:
    sentence (str): The sentence to search for an adverb.

    Returns:
    tuple: A tuple containing the first adverb and its position, or (None, None) if no adverb is found.
    '''
    words = word_tokenize(sentence)
    tagged_words = pos_tag(words)

    for index, (word, tag) in enumerate(tagged_words):
        if tag.startswith('RB'):  # 'RB' is the POS tag for adverbs
            return word, index

    return None, None

'''
Module to find the first adverb in a given sentence.
'''
import nltk
from nltk import pos_tag, word_tokenize
# Ensure that the necessary NLTK resources are downloaded
nltk.download('averaged_perceptron_tagger', quiet=True)
nltk.download('punkt', quiet=True)
def find_first_adverb(sentence):
    '''
    Finds the first adverb in the given sentence and returns it along with its position.
    Parameters:
    sentence (str): The sentence to search for an adverb.
    Returns:
    tuple: A tuple containing the first adverb and its position, or (None, None) if no adverb is found.
    '''
    words = word_tokenize(sentence)
    tagged_words = pos_tag(words)
    for index, (word, tag) in enumerate(tagged_words):
        if tag.startswith('RB'):  # 'RB' is the POS tag for adverbs
            return word, index
    return None, None

import tkinter as tk
from tkinter import messagebox
from adverb_finder import find_first_adverb
class AdverbFinderApp:
    def __init__(self, master):
        '''
        Initializes the Adverb Finder application GUI.
        Parameters:
        master (Tk): The root window of the application.
        '''
        self.master = master
        master.title("Adverb Finder")
        self.label = tk.Label(master, text="Enter a sentence:")
        self.label.pack()
        self.entry = tk.Entry(master, width=50)
        self.entry.pack()
        self.find_button = tk.Button(master, text="Find Adverb", command=self.find_adverb)

        self.result_label = tk.Label(master, text="", fg="blue")
        self.result_label.pack()

        '''
        Retrieves the sentence from the entry widget, finds the first adverb and its position,

        if adverb:

'''
Module to find the first adverb in a given sentence.
'''
import nltk
from nltk import pos_tag, word_tokenize
# Ensure that the necessary NLTK resources are downloaded
nltk.download('averaged_perceptron_tagger', quiet=True)
nltk.download('punkt', quiet=True)
def find_first_adverb(sentence):
    '''
    Finds the first adverb in the given sentence and returns it along with its position.
    Parameters:
    sentence (str): The sentence to search for an adverb.
    Returns:
    tuple: A tuple containing the first adverb and its position, or (None, None) if no adverb is found.
    '''
    words = word_tokenize(sentence)
    tagged_words = pos_tag(words)
    for index, (word, tag) in enumerate(tagged_words):
        if tag.startswith('RB'):  # 'RB' is the POS tag for adverbs
            return word, index
    return None, None
