def count_sequences(m, n):
    '''
    Computes the number of valid sequences of length n such that each element is
    a positive integer, greater than or equal to twice the previous element, and less than or equal to m.

    Parameters:
        m (int): Maximum value for elements in the sequence.
        n (int): Length of the sequence.

    Returns:
        int: Number of valid sequences.
    '''
    def helper(prev, length):
        '''
        Recursive helper function to compute sequences.

        Parameters:
            prev (int): Previous element in the sequence.
            length (int): Remaining length of the sequence.

        Returns:
            int: Number of valid sequences for the given parameters.
        '''
        if length == 0:
            return 1

        count = 0
        for current in range(max(2 * prev, 1), m + 1):
            count += helper(current, length - 1)

        return count

    return helper(0, n)

def count_sequences(m, n):
    '''
    Computes the number of valid sequences of length n such that each element is
    a positive integer, greater than or equal to twice the previous element, and less than or equal to m.
    Parameters:
        m (int): Maximum value for elements in the sequence.
        n (int): Length of the sequence.
    Returns:
        int: Number of valid sequences.
    '''
    def helper(prev, length):
        '''
        Recursive helper function to compute sequences.
        Parameters:
            prev (int): Previous element in the sequence.
            length (int): Remaining length of the sequence.
        Returns:
            int: Number of valid sequences for the given parameters.
        '''
        if length == 0:
            return 1
        count = 0
        for current in range(max(2 * prev, 1), m + 1):
            count += helper(current, length - 1)
        return count
    return helper(0, n)

def count_sequences(m, n):
    '''
    Computes the number of valid sequences of length n such that each element is
    a positive integer, greater than or equal to twice the previous element, and less than or equal to m.
    Parameters:
        m (int): Maximum value for elements in the sequence.
        n (int): Length of the sequence.
    Returns:
        int: Number of valid sequences.
    '''
    def helper(prev, length):
        '''
        Recursive helper function to compute sequences.
        Parameters:
            prev (int): Previous element in the sequence.
            length (int): Remaining length of the sequence.
        Returns:
            int: Number of valid sequences for the given parameters.
        '''
        if length == 0:
            return 1
        count = 0
        # Ensure the range starts at max(2 * prev, 1) to satisfy the condition
        for current in range(max(2 * prev, 1), m + 1):
            count += helper(current, length - 1)
        return count
    # Start the recursion with prev = 0 to allow the first element to be any valid number
    return helper(0, n)

def count_sequences(m, n):
    '''
    Computes the number of valid sequences of length n such that each element is
    a positive integer, greater than or equal to twice the previous element, and less than or equal to m.
    Parameters:
        m (int): Maximum value for elements in the sequence.
        n (int): Length of the sequence.
    Returns:
        int: Number of valid sequences.
    '''
    def helper(prev, length):
        '''
        Recursive helper function to compute sequences.
        Parameters:
            prev (int): Previous element in the sequence.
            length (int): Remaining length of the sequence.
        Returns:
            int: Number of valid sequences for the given parameters.
        '''
        if length == 0:
            return 1
        count = 0
        # Ensure the range starts at max(2 * prev, 1) to satisfy the condition
        for current in range(max(2 * prev, 1), m + 1):
            count += helper(current, length - 1)
        return count

    # Start the recursion with prev = 0 to allow the first element to be any valid number
    return helper(0, n)

def count_sequences(m, n):
    '''
    Computes the number of valid sequences of length n such that each element is
    a positive integer, greater than or equal to twice the previous element, and less than or equal to m.
    Parameters:
        m (int): Maximum value for elements in the sequence.
        n (int): Length of the sequence.
    Returns:
        int: Number of valid sequences.
    '''
    def helper(prev, length):
        '''
        Recursive helper function to compute sequences.
        Parameters:
            prev (int): Previous element in the sequence.
            length (int): Remaining length of the sequence.
        Returns:
            int: Number of valid sequences for the given parameters.
        '''
        if length == 0:
            return 1
        count = 0
        # Ensure the range starts at max(2 * prev, 1) to satisfy the condition
        for current in range(max(2 * prev, 1), m + 1):
            count += helper(current, length - 1)
        return count
    # Start the recursion with prev = 0 to allow the first element to be any valid number
    return helper(0, n)

'''  
This file contains the helper function to compute the number of valid sequences based on the problem statement.  
'''  
def count_sequences(m, n):  
    '''  
    Computes the number of valid sequences of length n such that each element is  
    a positive integer, greater than or equal to twice the previous element, and less than or equal to m.  
    Parameters:  
        m (int): Maximum value for elements in the sequence.  
        n (int): Length of the sequence.  
    Returns:  
        int: Number of valid sequences.  
    '''  
    def helper(prev, length):  
        '''  
        Recursive helper function to compute sequences.  
        Parameters:  
            prev (int): Previous element in the sequence.  
            length (int): Remaining length of the sequence.  
        Returns:  
            int: Number of valid sequences for the given parameters.  
        '''  
        if length == 0:  
            return 1  
        count = 0  
        # Ensure the range starts at max(2 * prev, 1) to satisfy the condition  
        for current in range(max(2 * prev, 1), m + 1):  
            count += helper(current, length - 1)  
        return count  
    # Start the recursion with prev = 0 to allow the first element to be any valid number  
    return helper(0, n)

```python
'''  
This file contains the helper function to compute the number of valid sequences based on the problem statement.  
'''  
def count_sequences(m, n):  
    '''  
    Computes the number of valid sequences of length n such that each element is  
    a positive integer, greater than or equal to twice the previous element, and less than or equal to m.  
    Parameters:  
        m (int): Maximum value for elements in the sequence.  
        n (int): Length of the sequence.  
    Returns:  
        int: Number of valid sequences.  
    '''  
    def helper(prev, length):  
        '''  
        Recursive helper function to compute sequences.  
        Parameters:  
            prev (int): Previous element in the sequence.  
            length (int): Remaining length of the sequence.  
        Returns:  
            int: Number of valid sequences for the given parameters.  
        '''  
        if length == 0:  
            return 1  
        count = 0  
        # Ensure the range starts at max(2 * prev, 1) to satisfy the condition  
        for current in range(max(2 * prev, 1), m + 1):  
            count += helper(current, length - 1)  
        return count  
    # Start the recursion with prev = 0 to allow the first element to be any valid number  
    return helper(0, n)

## 📖 Introduction

The Sequence Calculator is a Python-based GUI application that calculates the number of possible sequences of length `n` such that:
- Each element in the sequence is a positive integer.
- Each element is greater than or equal to twice the previous element.
- Each element is less than or equal to a maximum value `m`.

This tool is ideal for mathematicians, programmers, and anyone interested in sequence generation based on custom constraints.
