'''

import tkinter as tk
from lucid_numbers import get_lucid_numbers

class LucidNumberApp:
    def __init__(self, root):
        self.root = root
        self.create_widgets()

    def create_widgets(self):
        self.label = tk.Label(self.root, text="Enter a number:")
        self.label.pack()

        self.entry = tk.Entry(self.root)
        self.entry.pack()

        self.button = tk.Button(self.root, text="Find Lucid Numbers", command=self.calculate_lucid_numbers)
        self.button.pack()

        self.result_label = tk.Label(self.root, text="")
        self.result_label.pack()

    def calculate_lucid_numbers(self):
        try:
            limit = int(self.entry.get())
            lucid_numbers = get_lucid_numbers(limit)
            self.result_label.config(text=f"Lucid Numbers: {lucid_numbers}")
        except ValueError:
            self.result_label.config(text="Please enter a valid integer.")

def is_lucid_number(n):
    '''
    Determines if a number is a lucid number.
    For demonstration purposes, a lucid number is defined as a prime number.
    '''
    if n < 2:
        return False
    for i in range(2, int(n ** 0.5) + 1):
        if n % i == 0:
            return False
    return True
def get_lucid_numbers(limit):
    '''
    Returns a list of lucid numbers up to a given limit.
    '''
    lucid_numbers = []
    for i in range(1, limit + 1):
        if is_lucid_number(i):
            lucid_numbers.append(i)
    return lucid_numbers

'''
Module to calculate lucid numbers.
'''
def is_lucid_number(n):
    '''
    Determines if a number is a lucid number.
    For demonstration purposes, a lucid number is defined as a prime number.
    '''
    if n < 2:
        return False
    for i in range(2, int(n ** 0.5) + 1):
        if n % i == 0:
            return False
    return True
def get_lucid_numbers(limit):
    '''
    Returns a list of lucid numbers up to a given limit.
    '''
    lucid_numbers = []
    for i in range(1, limit + 1):
        if is_lucid_number(i):
            lucid_numbers.append(i)
    return lucid_numbers

def is_lucid_number(n):
    '''
    Determines if a number is a lucid number.
    For demonstration purposes, a lucid number is defined as a prime number.
    '''
    if n < 2:
        return False

The Lucid Numbers Finder application is a simple tool that allows users to input a number and find all lucid numbers (prime numbers) up to that number. The application features a graphical user interface (GUI) built using Tkinter, making it easy to use for anyone.
