```python
'''

main.py

'''

import tkinter as tk
from tkinter import messagebox

def is_sorted(lst):
    '''
    Check if the list is sorted in non-decreasing order.
    
    :param lst: List of numbers
    :return: True if sorted, False otherwise
    '''
    return all(lst[i] <= lst[i + 1] for i in range(len(lst) - 1))

class App:
    '''
    GUI Application to check if a list is sorted.
    '''
    def __init__(self, root):
        self.root = root
        self.root.title("Check if List is Sorted")

        self.label = tk.Label(root, text="Enter a list of numbers separated by commas:")
        self.label.pack()

        self.entry = tk.Entry(root, width=50)
        self.entry.pack()

        self.check_button = tk.Button(root, text="Check", command=self.check_sorted)
        self.check_button.pack()

        self.result_label = tk.Label(root, text="")
        self.result_label.pack()

    def check_sorted(self):
        '''
        Check if the entered list is sorted and display the result.
        '''
        try:
            # Get the list from the entry and convert it to a list of numbers
            input_list = list(map(int, self.entry.get().split(',')))
            # Check if the list is sorted
            if is_sorted(input_list):
                self.result_label.config(text="The list is sorted.")
            else:
                self.result_label.config(text="The list is not sorted.")
        except ValueError:
            messagebox.showerror("Invalid input", "Please enter a valid list of numbers separated by commas.")

if __name__ == "__main__":
    root = tk.Tk()
    app = App(root)
    root.mainloop()
```

- **Error Handling**: If the user enters invalid input (non-numeric values), an error message is displayed using `messagebox.showerror`.

This code provides a complete, functional application that allows users to input a list of numbers and check if it is sorted. The GUI is simple and intuitive, making it easy for users to interact with the application.

import tkinter as tk
from tkinter import messagebox

def is_sorted(lst):
    '''
    Check if the list is sorted in non-decreasing order.
    
    :param lst: List of numbers
    :return: True if sorted, False otherwise
    '''

    GUI Application to check if a list is sorted.
    '''
    def __init__(self, root):
        self.root = root
        self.root.title("Check if List is Sorted")

        self.entry = tk.Entry(root, width=50)
        self.entry.pack()

        self.check_button = tk.Button(root, text="Check", command=self.check_sorted)

        self.result_label = tk.Label(root, text="")
        self.result_label.pack()

    def check_sorted(self):

        Check if the entered list is sorted and display the result.
        '''
        try:
            # Get the list from the entry and convert it to a list of numbers

if __name__ == "__main__":
    root = tk.Tk()

'''  
This script creates a GUI application using Tkinter to check if a list of numbers is sorted in non-decreasing order.  
'''  
import tkinter as tk  
from tkinter import messagebox  

def is_sorted(lst):  
    '''  
    Check if the list is sorted in non-decreasing order.  
    :param lst: List of numbers  
    :return: True if sorted, False otherwise  
    '''  
    return all(lst[i] <= lst[i + 1] for i in range(len(lst) - 1))  

class App:  
    '''  
    GUI Application to check if a list is sorted.  
    '''  
    def __init__(self, root):  
        self.root = root  
        self.root.title("Check if List is Sorted")  
        self.label = tk.Label(root, text="Enter a list of numbers separated by commas:")  
        self.label.pack()  
        self.entry = tk.Entry(root, width=50)  
        self.entry.pack()  
        self.check_button = tk.Button(root, text="Check", command=self.check_sorted)  
        self.check_button.pack()  
        self.result_label = tk.Label(root, text="")  
        self.result_label.pack()  

    def check_sorted(self):  
        '''  
        Check if the entered list is sorted and display the result.  
        '''  
        try:  
            # Get the list from the entry and convert it to a list of numbers  
            input_list = list(map(int, self.entry.get().split(',')))  
            # Check if the list is sorted  
            if is_sorted(input_list):  
                self.result_label.config(text="The list is sorted.", fg="green")  
            else:  
                self.result_label.config(text="The list is not sorted.", fg="red")  
        except ValueError:  
            messagebox.showerror("Invalid input", "Please enter a valid list of numbers separated by commas.")  

if __name__ == "__main__":  
    root = tk.Tk()  
    app = App(root)  
    root.mainloop()

'''

import tkinter as tk  
from tkinter import messagebox  
def is_sorted(lst):  
    '''  
    Check if the list is sorted in non-decreasing order.

    '''  
    return all(lst[i] <= lst[i + 1] for i in range(len(lst) - 1))

    '''  
    GUI Application to check if a list is sorted.  
    '''

        except ValueError:  
            messagebox.showerror("Invalid input", "Please enter a valid list of numbers separated by commas.")

'''
This script creates a GUI application using Tkinter to check if a list of numbers is sorted in non-decreasing order.
'''
import tkinter as tk
from tkinter import messagebox

def is_sorted(lst):
    '''
    Check if the list is sorted in non-decreasing order.
    :param lst: List of numbers
    :return: True if sorted, False otherwise
    '''
    return all(lst[i] <= lst[i + 1] for i in range(len(lst) - 1))

class App:
    '''
    GUI Application to check if a list is sorted.
    '''
    def __init__(self, root):
        self.root = root
        self.root.title("Check if List is Sorted")
        self.label = tk.Label(root, text="Enter a list of numbers separated by commas:")
        self.label.pack()
        self.entry = tk.Entry(root, width=50)
        self.entry.pack()
        self.check_button = tk.Button(root, text="Check", command=self.check_sorted)
        self.check_button.pack()
        self.result_label = tk.Label(root, text="")
        self.result_label.pack()

    def check_sorted(self):
        '''
        Check if the entered list is sorted and display the result.
        '''
        try:
            # Get the list from the entry and convert it to a list of numbers
            input_list = list(map(int, self.entry.get().split(',')))
            # Check if the list is sorted
            if is_sorted(input_list):
                self.result_label.config(text="The list is sorted.", fg="green")
            else:
                self.result_label.config(text="The list is not sorted.", fg="red")
        except ValueError:
            messagebox.showerror("Invalid input", "Please enter a valid list of numbers separated by commas.")

if __name__ == "__main__":
    root = tk.Tk()
    app = App(root)
    root.mainloop()

import tkinter as tk
from tkinter import messagebox
def is_sorted(lst):
    '''
    Check if the list is sorted in non-decreasing order.
    :param lst: List of numbers
    :return: True if sorted, False otherwise
    '''
    return all(lst[i] <= lst[i + 1] for i in range(len(lst) - 1))
class App:
    '''
    GUI Application to check if a list is sorted.
    '''
    def __init__(self, root):
        self.root = root
        self.root.title("Check if List is Sorted")
        self.label = tk.Label(root, text="Enter a list of numbers separated by commas:")
        self.label.pack()
        self.entry = tk.Entry(root, width=50)
        self.entry.pack()
        self.check_button = tk.Button(root, text="Check", command=self.check_sorted)
        self.check_button.pack()
        self.result_label = tk.Label(root, text="")
        self.result_label.pack()
    def check_sorted(self):
        '''
        Check if the entered list is sorted and display the result.
        '''
        try:
            # Get the list from the entry and convert it to a list of numbers
            input_list = list(map(int, self.entry.get().split(',')))
            # Check if the list is sorted
            if is_sorted(input_list):
                self.result_label.config(text="The list is sorted.", fg="green")
            else:
                self.result_label.config(text="The list is not sorted.", fg="red")
        except ValueError:
            messagebox.showerror("Invalid input", "Please enter a valid list of numbers separated by commas.")
if __name__ == "__main__":
    root = tk.Tk()

# Check if List is Sorted - User Manual

This application provides a simple and intuitive graphical user interface (GUI) to check whether a list of numbers is sorted in non-decreasing order. It is built using Python and Tkinter, making it lightweight and easy to use.

---

## Main Functions of the Software

### 1. **Check if a List is Sorted**
The core functionality of this application is to determine whether a given list of numbers is sorted in non-decreasing order. Users can input a list of numbers separated by commas, and the application will analyze the list and display the result.

### 2. **Error Handling**
The application includes error handling to ensure that users input valid data. If the input is not a valid list of numbers, an error message will be displayed, guiding the user to correct their input.

---

python main.py
```

---

## How to Use the Application

### Step 1: Launch the Application
Run the `main.py` file using Python. A GUI window will appear.

### Step 2: Input a List of Numbers
- Enter a list of numbers separated by commas (e.g., `1,2,3,4,5`) in the input field.
- Ensure the numbers are integers and properly formatted.

### Step 3: Check if the List is Sorted
- Click the "Check" button.
- The application will analyze the list and display one of the following messages:
  - **"The list is sorted."** (in green) if the list is sorted in non-decreasing order.
  - **"The list is not sorted."** (in red) if the list is not sorted.
