'''

This module contains the main application class and functions to perform bitwise XOR operations on tuples.

'''

import tkinter as tk
from tkinter import messagebox

def perform_xor(tuples):
    '''
    Perform bitwise XOR operation across a list of tuples.
    
    :param tuples: List of tuples containing integers.
    :return: List of results after performing XOR on each tuple.
    '''
    results = []
    for tup in tuples:
        xor_result = 0
        for num in tup:
            xor_result ^= num
        results.append(xor_result)
    return results

class BitwiseXORApp:
    '''
    A GUI application to perform bitwise XOR operation on user-input tuples.
    '''
    
    def __init__(self, root):
        self.root = root
        self.root.title("Bitwise XOR Application")
        
        self.label = tk.Label(root, text="Enter tuples (e.g., (1, 2), (3, 4, 5)):")
        self.label.pack()
        
        self.entry = tk.Entry(root, width=50)
        self.entry.pack()
        
        self.submit_button = tk.Button(root, text="Compute XOR", command=self.compute_xor)
        self.submit_button.pack()
        
        self.result_label = tk.Label(root, text="")
        self.result_label.pack()
    
    def compute_xor(self):
        '''
        Compute the XOR for the input tuples and display the result.
        '''
        try:
            input_text = self.entry.get()
            tuples = eval(input_text)  # Convert string input to tuple
            if not isinstance(tuples, list):
                raise ValueError("Input must be a list of tuples.")
            results = perform_xor(tuples)
            self.result_label.config(text=f"XOR Results: {results}")
        except Exception as e:
            messagebox.showerror("Error", f"Invalid input: {e}")

def create_gui():
    '''
    Create and run the GUI application.
    '''
    root = tk.Tk()

import tkinter as tk
from tkinter import messagebox
import ast
def perform_xor(tuples):
    '''
    Perform bitwise XOR operation across a list of tuples.
    :param tuples: List of tuples containing integers.
    :return: List of results after performing XOR on each tuple.
    '''
    results = []
    for tup in tuples:
        xor_result = 0
        for num in tup:
            xor_result ^= num
        results.append(xor_result)
    return results
class BitwiseXORApp:
    '''
    A GUI application to perform bitwise XOR operation on user-input tuples.
    '''
    def __init__(self, root):
        self.root = root
        self.root.title("Bitwise XOR Application")
        self.label = tk.Label(root, text="Enter tuples (e.g., (1, 2), (3, 4, 5)):")
        self.label.pack()

        self.entry.pack()
        self.submit_button = tk.Button(root, text="Compute XOR", command=self.compute_xor)
        self.submit_button.pack()
        self.result_label = tk.Label(root, text="")
        self.result_label.pack()
    def compute_xor(self):
        '''
        Compute the XOR for the input tuples and display the result.
        '''

'''
This module contains the main application class and functions to perform bitwise XOR operations on tuples.
'''
import tkinter as tk
from tkinter import messagebox
import ast
def perform_xor(tuples):
    '''
    Perform bitwise XOR operation across a list of tuples.
    :param tuples: List of tuples containing integers.
    :return: List of results after performing XOR on each tuple.
    '''
    results = []
    for tup in tuples:
        xor_result = 0
        for num in tup:
            xor_result ^= num
        results.append(xor_result)
    return results
class BitwiseXORApp:
    '''
    A GUI application to perform bitwise XOR operation on user-input tuples.
    '''
    def __init__(self, root):
        self.root = root
        self.root.title("Bitwise XOR Application")
        self.label = tk.Label(root, text="Enter tuples (e.g., (1, 2), (3, 4, 5)):")
        self.label.pack()
        self.entry = tk.Entry(root, width=50)
        self.entry.pack()
        self.submit_button = tk.Button(root, text="Compute XOR", command=self.compute_xor)
        self.submit_button.pack()
        self.result_label = tk.Label(root, text="")
        self.result_label.pack()
    def compute_xor(self):
        '''
        Compute the XOR for the input tuples and display the result.
        '''
        try:
            input_text = self.entry.get()
            tuples = ast.literal_eval(input_text)  # Safely convert string input to tuple
            if not isinstance(tuples, list):
                raise ValueError("Input must be a list of tuples.")
            results = perform_xor(tuples)
            self.result_label.config(text=f"XOR Results: {results}")
        except Exception as e:
            messagebox.showerror("Error", f"Invalid input: {e}")
def create_gui():
    '''
    Create and run the GUI application.
    '''
    root = tk.Tk()
    app = BitwiseXORApp(root)
    root.mainloop()
if __name__ == "__main__":
    create_gui()
