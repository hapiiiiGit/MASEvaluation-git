'''

This module contains the main application logic for a simple GUI application that replaces spaces in a given string with '%20'.

'''

import tkinter as tk
from tkinter import messagebox

class MainApp:
    def __init__(self, root):
        '''
        Initialize the main application with GUI components.
        
        Parameters:
        root (tk.Tk): The root window of the tkinter application.
        '''
        self.root = root
        self.root.title("Space Replacer")
        
        # Create and place the input label and entry
        self.label = tk.Label(root, text="Enter your string:")
        self.label.pack(pady=10)
        
        self.entry = tk.Entry(root, width=50)
        self.entry.pack(pady=10)
        
        # Create and place the submit button
        self.submit_button = tk.Button(root, text="Submit", command=self.on_submit)
        self.submit_button.pack(pady=20)
        
        # Create and place the result label
        self.result_label = tk.Label(root, text="", fg="blue")
        self.result_label.pack(pady=10)
    
    def replace_spaces(self, input_string):
        '''
        Replace all spaces in the input string with '%20'.
        
        Parameters:
        input_string (str): The string to process.
        
        Returns:
        str: The processed string with spaces replaced by '%20'.
        '''
        return input_string.replace(' ', '%20')
    
    def on_submit(self):
        '''
        Handle the submit button click event.
        '''
        input_string = self.entry.get()
        if input_string:
            result = self.replace_spaces(input_string)
            self.result_label.config(text=f"Result: {result}")
        else:
            messagebox.showwarning("Input Error", "Please enter a string.")

def main():
    '''
    The main function to start the application.
    '''
    root = tk.Tk()
    app = MainApp(root)
    root.mainloop()

if __name__ == "__main__":
    main()
```

- **replace_spaces Method**: This method takes a string and replaces all spaces with '%20'.
- **on_submit Method**: This method is triggered when the user clicks the submit button. It retrieves the input string, processes it, and updates the result label.
- **main Function**: This function initializes the `tkinter` root window and starts the application.

This module contains the main application logic for a simple GUI application that replaces spaces in a given string with '%20'.

'''

import tkinter as tk
from tkinter import messagebox

class MainApp:
    def __init__(self, root):
        '''
        Initialize the main application with GUI components.
        
        Parameters:
        root (tk.Tk): The root window of the tkinter application.
        '''
        self.root = root
        self.root.title("Space Replacer")
        
        # Create and place the input label and entry
        self.label = tk.Label(root, text="Enter your string:")
        self.label.pack(pady=10)
        
        self.entry = tk.Entry(root, width=50)
        self.entry.pack(pady=10)
        
        # Create and place the submit button
        self.submit_button = tk.Button(root, text="Submit", command=self.on_submit)
        self.submit_button.pack(pady=20)
        
        # Create and place the result label
        self.result_label = tk.Label(root, text="", fg="blue")
        self.result_label.pack(pady=10)
    
    def replace_spaces(self, input_string):
        '''
        Replace all spaces in the input string with '%20'.
        
        Parameters:
        input_string (str): The string to process.
        
        Returns:
        str: The processed string with spaces replaced by '%20'.
        '''
        return input_string.replace(' ', '%20')
    
    def on_submit(self):
        '''
        Handle the submit button click event.
        '''
        input_string = self.entry.get()
        if input_string:
            result = self.replace_spaces(input_string)
            self.result_label.config(text=f"Result: {result}")
        else:
            messagebox.showwarning("Input Error", "Please enter a string.")

def main():
    '''
    The main function to start the application.
    '''
    root = tk.Tk()
    app = MainApp(root)
    root.mainloop()

if __name__ == "__main__":

This module contains the main application logic for a simple GUI application that replaces spaces in a given string with '%20'.
'''
import tkinter as tk
from tkinter import messagebox

class MainApp:
    def __init__(self, root):
        '''
        Initialize the main application with GUI components.
        Parameters:
        root (tk.Tk): The root window of the tkinter application.
        '''
        self.root = root
        self.root.title("Space Replacer")
        
        # Create and place the input label and entry
        self.label = tk.Label(root, text="Enter your string:")
        self.label.pack(pady=10)
        
        self.entry = tk.Entry(root, width=50)
        self.entry.pack(pady=10)
        
        # Create and place the submit button
        self.submit_button = tk.Button(root, text="Submit", command=self.on_submit)
        self.submit_button.pack(pady=20)
        
        # Create and place the result label
        self.result_label = tk.Label(root, text="", fg="blue")
        self.result_label.pack(pady=10)

    def replace_spaces(self, input_string):
        '''
        Replace all spaces in the input string with '%20'.
        Parameters:
        input_string (str): The string to process.
        Returns:
        str: The processed string with spaces replaced by '%20'.
        '''
        return input_string.replace(' ', '%20')

    def on_submit(self):
        '''
        Handle the submit button click event.
        '''
        input_string = self.entry.get()
        if input_string:
            result = self.replace_spaces(input_string)
            self.result_label.config(text=f"Result: {result}")
        else:
            messagebox.showwarning("Input Error", "Please enter a string.")

def main():
    '''
    The main function to start the application.
    '''
    root = tk.Tk()
    app = MainApp(root)
    root.mainloop()

if __name__ == "__main__":
    main()

import tkinter as tk
from tkinter import messagebox
class MainApp:
    def __init__(self, root):
        '''
        Initialize the main application with GUI components.
        Parameters:
        root (tk.Tk): The root window of the tkinter application.
        '''
        self.root = root
        self.root.title("Space Replacer")
        # Create and place the input label and entry
        self.label = tk.Label(root, text="Enter your string:")
        self.label.pack(pady=10)
        self.entry = tk.Entry(root, width=50)
        self.entry.pack(pady=10)
        # Create and place the submit button
        self.submit_button = tk.Button(root, text="Submit", command=self.on_submit)
        self.submit_button.pack(pady=20)
        # Create and place the result label
        self.result_label = tk.Label(root, text="", fg="blue")
        self.result_label.pack(pady=10)
    def replace_spaces(self, input_string):
        '''
        Replace all spaces in the input string with '%20'.
        Parameters:
        input_string (str): The string to process.
        Returns:
        str: The processed string with spaces replaced by '%20'.
        '''
        return input_string.replace(' ', '%20')
    def on_submit(self):
        '''
        Handle the submit button click event.
        '''
        input_string = self.entry.get()
        if input_string:

            self.result_label.config(text=f"Result: {result}")
        else:
            messagebox.showwarning("Input Error", "Please enter a string.")
def main():
    '''
    The main function to start the application.
    '''
