import tkinter as tk

class MainApplication:
    def __init__(self, root):
        self.root = root
        self.root.title("Insertion Point Finder")

        self.label_list = tk.Label(root, text="Enter a sorted list (comma-separated):")

```python
'''

Utility functions for the insertion point finder application.

'''

def find_insertion_point(sorted_list, value):
    """
    Finds the right insertion point for a specified value in a sorted list.
    
    :param sorted_list: List of sorted integers.
    :param value: Integer value to find the insertion point for.
    :return: Index where the value should be inserted to maintain sorted order.
    """
    low, high = 0, len(sorted_list)
    while low < high:
        mid = (low + high) // 2
        if sorted_list[mid] <= value:
            low = mid + 1
        else:
            high = mid
    return low
```

    def __init__(self, root):

'''

Utility functions for the insertion point finder application.

'''

def find_insertion_point(sorted_list, value):
    """
    Finds the right insertion point for a specified value in a sorted list.
    
    :param sorted_list: List of sorted integers.
    :param value: Integer value to find the insertion point for.
    :return: Index where the value should be inserted to maintain sorted order.
    """
    low, high = 0, len(sorted_list)
    while low < high:
        mid = (low + high) // 2
        if sorted_list[mid] <= value:
            low = mid + 1
        else:
            high = mid
    return low
```

import tkinter as tk
from tkinter import messagebox
from utils import find_insertion_point

class MainApplication:
    def __init__(self, root):
        self.root = root
        self.root.title("Insertion Point Finder")
        
        self.label_list = tk.Label(root, text="Enter a sorted list (comma-separated):")

'''

def find_insertion_point(sorted_list, value):
    """
    Finds the right insertion point for a specified value in a sorted list.
    :param sorted_list: List of sorted integers.
    :param value: Integer value to find the insertion point for.
    :return: Index where the value should be inserted to maintain sorted order.
    """
    low, high = 0, len(sorted_list)
    while low < high:
        mid = (low + high) // 2
        if sorted_list[mid] <= value:
            low = mid + 1
        else:
            high = mid
    return low

def find_insertion_point(sorted_list, value):
    """
    Finds the right insertion point for a specified value in a sorted list.
    :param sorted_list: List of sorted integers.
    :param value: Integer value to find the insertion point for.
    :return: Index where the value should be inserted to maintain sorted order.
    """
    low, high = 0, len(sorted_list)
    while low < high:
        mid = (low + high) // 2
        if sorted_list[mid] <= value:
            low = mid + 1
        else:
            high = mid
    return low

import tkinter as tk
from tkinter import messagebox
from utils import find_insertion_point
class MainApplication:
    def __init__(self, root):
        self.root = root
        self.root.title("Insertion Point Finder")
        self.label_list = tk.Label(root, text="Enter a sorted list (comma-separated):")
        self.label_list.pack()
        self.entry_list = tk.Entry(root, width=50)
        self.entry_list.pack()
        self.label_value = tk.Label(root, text="Enter the value to insert:")
        self.label_value.pack()
        self.entry_value = tk.Entry(root, width=50)
        self.entry_value.pack()
        self.submit_button = tk.Button(root, text="Find Insertion Point", command=self.on_submit)
        self.submit_button.pack()
    def on_submit(self):
        try:
            sorted_list = list(map(int, self.entry_list.get().split(',')))
            value = int(self.entry_value.get())
            index = find_insertion_point(sorted_list, value)
            messagebox.showinfo("Result", f"The right insertion point for {value} is at index {index}.")
        except ValueError:
            messagebox.showerror("Error", "Please enter valid integers.")
if __name__ == "__main__":
    root = tk.Tk()
    app = MainApplication(root)
    root.mainloop()

'''
Utility functions for the insertion point finder application.
'''
def find_insertion_point(sorted_list, value):
    """
    Finds the right insertion point for a specified value in a sorted list.
    :param sorted_list: List of sorted integers.
    :param value: Integer value to find the insertion point for.
    :return: Index where the value should be inserted to maintain sorted order.
    """
    low, high = 0, len(sorted_list)
    while low < high:
        mid = (low + high) // 2
        if sorted_list[mid] <= value:
            low = mid + 1
        else:
            high = mid
    return low

    :param sorted_list: List of sorted integers.
    :param value: Integer value to find the insertion point for.
    :return: Index where the value should be inserted to maintain sorted order.
    """
    low, high = 0, len(sorted_list)
    while low < high:
        mid = (low + high) // 2
        if sorted_list[mid] <= value:
            low = mid + 1
        else:
            high = mid
    return low
