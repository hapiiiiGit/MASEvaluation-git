'''

This module contains the logic to find the nth polite number.

'''

def is_polite(number):
    """
    Check if a number is polite.
    A number is polite if it is not a power of 2.
    """
    return (number & (number - 1)) != 0

def find_nth_polite(n):
    """
    Find the nth polite number.
    """
    count = 0
    number = 1
    while count < n:
        number += 1
        if is_polite(number):
            count += 1
    return number

'''

This module contains the logic to find the nth polite number.

def is_polite(number):
    """
    Check if a number is polite.
    A number is polite if it is not a power of 2.
    """
    return (number & (number - 1)) != 0

def find_nth_polite(n):
    """
    Find the nth polite number.
    """
    count = 0
    number = 1
    while count < n:
        number += 1
        if is_polite(number):
            count += 1
    return number

def is_polite(number):
    """
    Check if a number is polite.
    A number is polite if it is not a power of 2.
    """
    return (number & (number - 1)) != 0
def find_nth_polite(n):

'''
This module contains the logic to find the nth polite number.

def is_polite(number):
    """
    Check if a number is polite.
    A number is polite if it is not a power of 2.
    """
    return (number & (number - 1)) != 0
def find_nth_polite(n):
    """
    Find the nth polite number.
    """
    count = 0
    number = 1
    while count < n:
        number += 1
        if is_polite(number):
            count += 1
    return number
