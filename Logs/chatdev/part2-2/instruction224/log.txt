'''

median_finder.py

Contains the MedianFinder class which provides a method to find the median of two sorted lists.

'''

class MedianFinder:
    @staticmethod
    def find_median(list1, list2):
        """
        Finds the median of two sorted lists of the same size.
        
        :param list1: First sorted list
        :param list2: Second sorted list
        :return: Median of the two lists
        """
        merged_list = sorted(list1 + list2)
        n = len(merged_list)
        mid = n // 2
        if n % 2 == 0:
            return (merged_list[mid - 1] + merged_list[mid]) / 2
        else:
            return merged_list[mid]
```

class MedianFinder:
    @staticmethod
    def find_median(list1, list2):
        """
        Finds the median of two sorted lists of the same size using binary search.
        :param list1: First sorted list
        :param list2: Second sorted list
        :return: Median of the two lists
        """
        n = len(list1)
        if n == 0:
            raise ValueError("Lists must not be empty.")
        
        def get_median(a, b):
            if (a + b) % 2 == 0:
                return (a + b) // 2
            else:
                return (a + b) // 2 + 1

        def median_of_two_sorted_arrays(arr1, arr2):
            if len(arr1) > len(arr2):
                arr1, arr2 = arr2, arr1
            x, y = len(arr1), len(arr2)
            low, high = 0, x
            while low <= high:
                partitionX = (low + high) // 2
                partitionY = (x + y + 1) // 2 - partitionX

                maxX = float('-inf') if partitionX == 0 else arr1[partitionX - 1]
                minX = float('inf') if partitionX == x else arr1[partitionX]

                maxY = float('-inf') if partitionY == 0 else arr2[partitionY - 1]
                minY = float('inf') if partitionY == y else arr2[partitionY]

                if maxX <= minY and maxY <= minX:
                    if (x + y) % 2 == 0:
                        return (max(maxX, maxY) + min(minX, minY)) / 2
                    else:
                        return max(maxX, maxY)
                elif maxX > minY:
                    high = partitionX - 1
                else:
                    low = partitionX + 1

        return median_of_two_sorted_arrays(list1, list2)

class MedianFinder:
    @staticmethod
    def find_median(list1, list2):
        """
        Finds the median of two sorted lists of the same size.
        :param list1: First sorted list
        :param list2: Second sorted list
        :return: Median of the two lists
        """
        merged_list = sorted(list1 + list2)
        n = len(merged_list)
        mid = n // 2
        if n % 2 == 0:
            return (merged_list[mid - 1] + merged_list[mid]) / 2
        else:
            return merged_list[mid]

class MedianFinder:
    @staticmethod
    def find_median(list1, list2):
        """
        Finds the median of two sorted lists of the same size using binary search.
        :param list1: First sorted list
        :param list2: Second sorted list
        :return: Median of the two lists
        """
        n = len(list1)
        if n == 0:
            raise ValueError("Lists must not be empty.")
        def median_of_two_sorted_arrays(arr1, arr2):
            if len(arr1) > len(arr2):
                arr1, arr2 = arr2, arr1
            x, y = len(arr1), len(arr2)
            low, high = 0, x
            while low <= high:
                partitionX = (low + high) // 2
                partitionY = (x + y + 1) // 2 - partitionX
                maxX = float('-inf') if partitionX == 0 else arr1[partitionX - 1]
                minX = float('inf') if partitionX == x else arr1[partitionX]
                maxY = float('-inf') if partitionY == 0 else arr2[partitionY - 1]
                minY = float('inf') if partitionY == y else arr2[partitionY]
                if maxX <= minY and maxY <= minX:
                    if (x + y) % 2 == 0:
                        return (max(maxX, maxY) + min(minX, minY)) / 2
                    else:
                        return max(maxX, maxY)
                elif maxX > minY:
                    high = partitionX - 1
                else:
                    low = partitionX + 1
        return median_of_two_sorted_arrays(list1, list2)

class MedianFinder:
    @staticmethod
    def find_median(list1, list2):
        """
        Finds the median of two sorted lists of the same size using binary search.
        :param list1: First sorted list
        :param list2: Second sorted list
        :return: Median of the two lists
        """
        n = len(list1)
        if n == 0:
            raise ValueError("Lists must not be empty.")
        def median_of_two_sorted_arrays(arr1, arr2):
            if len(arr1) > len(arr2):
                arr1, arr2 = arr2, arr1
            x, y = len(arr1), len(arr2)
            low, high = 0, x
            while low <= high:
                partitionX = (low + high) // 2
                partitionY = (x + y + 1) // 2 - partitionX
                maxX = float('-inf') if partitionX == 0 else arr1[partitionX - 1]
                minX = float('inf') if partitionX == x else arr1[partitionX]
                maxY = float('-inf') if partitionY == 0 else arr2[partitionY - 1]
                minY = float('inf') if partitionY == y else arr2[partitionY]
                if maxX <= minY and maxY <= minX:
                    if (x + y) % 2 == 0:
                        return (max(maxX, maxY) + min(minX, minY)) / 2
                    else:
                        return max(maxX, maxY)
                elif maxX > minY:
                    high = partitionX - 1
                else:
                    low = partitionX + 1
        return median_of_two_sorted_arrays(list1, list2)

class MedianFinder:
    @staticmethod
    def find_median(list1, list2):
        """
        Finds the median of two sorted lists of the same size using binary search.
        :param list1: First sorted list
        :param list2: Second sorted list
        :return: Median of the two lists
        """
        n = len(list1)
        if n == 0:
            raise ValueError("Lists must not be empty.")
        def median_of_two_sorted_arrays(arr1, arr2):
            if len(arr1) > len(arr2):
                arr1, arr2 = arr2, arr1
            x, y = len(arr1), len(arr2)
            low, high = 0, x
            while low <= high:
                partitionX = (low + high) // 2
                partitionY = (x + y + 1) // 2 - partitionX
                maxX = float('-inf') if partitionX == 0 else arr1[partitionX - 1]
                minX = float('inf') if partitionX == x else arr1[partitionX]
                maxY = float('-inf') if partitionY == 0 else arr2[partitionY - 1]
                minY = float('inf') if partitionY == y else arr2[partitionY]
                if maxX <= minY and maxY <= minX:
                    if (x + y) % 2 == 0:
                        return (max(maxX, maxY) + min(minX, minY)) / 2
                    else:
                        return max(maxX, maxY)
                elif maxX > minY:
                    high = partitionX - 1
                else:
                    low = partitionX + 1
        return median_of_two_sorted_arrays(list1, list2)

Contains the MedianFinder class which provides a method to find the median of two sorted lists.
'''
class MedianFinder:
    @staticmethod
    def find_median(list1, list2):
        """
        Finds the median of two sorted lists of the same size using binary search.
        :param list1: First sorted list
        :param list2: Second sorted list
        :return: Median of the two lists
        """
        n = len(list1)
        if n == 0:
            raise ValueError("Lists must not be empty.")
        
        def median_of_two_sorted_arrays(arr1, arr2):
            if len(arr1) > len(arr2):
                arr1, arr2 = arr2, arr1
            x, y = len(arr1), len(arr2)
            low, high = 0, x
            while low <= high:
                partitionX = (low + high) // 2
                partitionY = (x + y + 1) // 2 - partitionX
                maxX = float('-inf') if partitionX == 0 else arr1[partitionX - 1]
                minX = float('inf') if partitionX == x else arr1[partitionX]
                maxY = float('-inf') if partitionY == 0 else arr2[partitionY - 1]
                minY = float('inf') if partitionY == y else arr2[partitionY]
                if maxX <= minY and maxY <= minX:
                    if (x + y) % 2 == 0:
                        return (max(maxX, maxY) + min(minX, minY)) / 2
                    else:
                        return max(maxX, maxY)
                elif maxX > minY:
                    high = partitionX - 1
                else:
                    low = partitionX + 1
        
        return median_of_two_sorted_arrays(list1, list2)

Contains the MedianFinder class which provides a method to find the median of two sorted lists.

class MedianFinder:
    @staticmethod
    def find_median(list1, list2):
        """
        Finds the median of two sorted lists of the same size using binary search.
        :param list1: First sorted list
        :param list2: Second sorted list
        :return: Median of the two lists
        """
        n = len(list1)
        if n == 0:
            raise ValueError("Lists must not be empty.")
        def median_of_two_sorted_arrays(arr1, arr2):
            if len(arr1) > len(arr2):
                arr1, arr2 = arr2, arr1
            x, y = len(arr1), len(arr2)
            low, high = 0, x
            while low <= high:
                partitionX = (low + high) // 2
                partitionY = (x + y + 1) // 2 - partitionX
                maxX = float('-inf') if partitionX == 0 else arr1[partitionX - 1]
                minX = float('inf') if partitionX == x else arr1[partitionX]
                maxY = float('-inf') if partitionY == 0 else arr2[partitionY - 1]
                minY = float('inf') if partitionY == y else arr2[partitionY]
                if maxX <= minY and maxY <= minX:
                    if (x + y) % 2 == 0:
                        return (max(maxX, maxY) + min(minX, minY)) / 2
                    else:
                        return max(maxX, maxY)
                elif maxX > minY:
                    high = partitionX - 1
                else:
                    low = partitionX + 1
        return median_of_two_sorted_arrays(list1, list2)

        Finds the median of two sorted lists of the same size using binary search.
        :param list1: First sorted list
        :param list2: Second sorted list
        :return: Median of the two lists
        """
        n = len(list1)
        if n == 0:
            raise ValueError("Lists must not be empty.")
        def median_of_two_sorted_arrays(arr1, arr2):
            if len(arr1) > len(arr2):
                arr1, arr2 = arr2, arr1
            x, y = len(arr1), len(arr2)
            low, high = 0, x
            while low <= high:
                partitionX = (low + high) // 2
                partitionY = (x + y + 1) // 2 - partitionX
                maxX = float('-inf') if partitionX == 0 else arr1[partitionX - 1]
                minX = float('inf') if partitionX == x else arr1[partitionX]
                maxY = float('-inf') if partitionY == 0 else arr2[partitionY - 1]
                minY = float('inf') if partitionY == y else arr2[partitionY]
                if maxX <= minY and maxY <= minX:
                    if (x + y) % 2 == 0:
                        return (max(maxX, maxY) + min(minX, minY)) / 2
                    else:
                        return max(maxX, maxY)
                elif maxX > minY:
                    high = partitionX - 1
                else:
                    low = partitionX + 1
        return median_of_two_sorted_arrays(list1, list2)
