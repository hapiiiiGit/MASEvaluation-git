'''

Main application file for the Square Pyramid Surface Area Calculator.
This file contains the GUI implementation using tkinter.

'''

import tkinter as tk
from pyramid_calculator import calculate_surface_area

class SquarePyramidApp:
    def __init__(self, root):
        self.root = root
        self.root.title("Square Pyramid Surface Area Calculator")

        # Base edge input
        self.base_edge_label = tk.Label(root, text="Base Edge:")
        self.base_edge_label.pack()
        self.base_edge_entry = tk.Entry(root)
        self.base_edge_entry.pack()

        # Height input
        self.height_label = tk.Label(root, text="Height:")
        self.height_label.pack()
        self.height_entry = tk.Entry(root)
        self.height_entry.pack()

        # Calculate button
        self.calculate_button = tk.Button(root, text="Calculate", command=self.calculate_surface_area)
        self.calculate_button.pack()

        # Result display
        self.result_label = tk.Label(root, text="Surface Area:")
        self.result_label.pack()
        self.result_value = tk.Label(root, text="")
        self.result_value.pack()

    def calculate_surface_area(self):
        try:
            base_edge = float(self.base_edge_entry.get())
            height = float(self.height_entry.get())
            surface_area = calculate_surface_area(base_edge, height)
            self.result_value.config(text=f"{surface_area:.2f}")
        except ValueError:
            self.result_value.config(text="Invalid input. Please enter numeric values.")

if __name__ == "__main__":
    root = tk.Tk()
    app = SquarePyramidApp(root)
    root.mainloop()

#### `pyramid_calculator.py`

```python
'''

Module for calculating the surface area of a square pyramid.

'''

import math

def calculate_surface_area(base_edge, height):
    '''
    Calculate the surface area of a square pyramid.

    Parameters:
    base_edge (float): The length of the base edge of the pyramid.
    height (float): The height of the pyramid.

    Returns:
    float: The surface area of the square pyramid.
    '''
    base_area = base_edge ** 2
    slant_height = math.sqrt((base_edge / 2) ** 2 + height ** 2)
    lateral_area = 2 * base_edge * slant_height
    return base_area + lateral_area

'''

Module for calculating the surface area of a square pyramid.

import math

def calculate_surface_area(base_edge, height):
    '''
    Calculate the surface area of a square pyramid.

    Parameters:
    base_edge (float): The length of the base edge of the pyramid.
    height (float): The height of the pyramid.

    Returns:
    float: The surface area of the square pyramid.
    '''
    base_area = base_edge ** 2
    slant_height = math.sqrt((base_edge / 2) ** 2 + height ** 2)
    lateral_area = 2 * base_edge * slant_height
    return base_area + lateral_area

'''
Module for calculating the surface area of a square pyramid.
'''
import math
def calculate_surface_area(base_edge, height):
    '''
    Calculate the surface area of a square pyramid.
    Parameters:
    base_edge (float): The length of the base edge of the pyramid.
    height (float): The height of the pyramid.
    Returns:
    float: The surface area of the square pyramid.
    '''
    base_area = base_edge ** 2
    slant_height = math.sqrt((base_edge / 2) ** 2 + height ** 2)
    lateral_area = 2 * base_edge * slant_height
    return base_area + lateral_area

'''
Module for calculating the surface area of a square pyramid.
'''
import math
def calculate_surface_area(base_edge, height):
    '''
    Calculate the surface area of a square pyramid.
    Parameters:
    base_edge (float): The length of the base edge of the pyramid.
    height (float): The height of the pyramid.
    Returns:
    float: The surface area of the square pyramid.
    '''
    base_area = base_edge ** 2
    slant_height = math.sqrt((base_edge / 2) ** 2 + height ** 2)
    lateral_area = 2 * base_edge * slant_height
    return base_area + lateral_area

'''
Module for calculating the surface area of a square pyramid.
'''
import math
def calculate_surface_area(base_edge, height):
    '''
    Calculate the surface area of a square pyramid.
    Parameters:
    base_edge (float): The length of the base edge of the pyramid.
    height (float): The height of the pyramid.
    Returns:
    float: The surface area of the square pyramid.
    '''
    base_area = base_edge ** 2
    slant_height = math.sqrt((base_edge / 2) ** 2 + height ** 2)
    lateral_area = 2 * base_edge * slant_height
    return base_area + lateral_area

# Square Pyramid Surface Area Calculator

Welcome to the Square Pyramid Surface Area Calculator! This application allows you to calculate the surface area of a square pyramid given its base edge and height. The application provides a simple graphical user interface (GUI) for easy input and calculation.

## Main Functions

The main function of this software is to calculate the surface area of a square pyramid. The surface area is calculated using the formula:

\[ \text{Surface Area} = \text{Base Area} + \text{Lateral Area} \]

Where:
- Base Area = \( \text{base\_edge}^2 \)
- Lateral Area = \( 2 \times \text{base\_edge} \times \text{slant\_height} \)
- Slant Height = \( \sqrt{\left(\frac{\text{base\_edge}}{2}\right)^2 + \text{height}^2} \)
