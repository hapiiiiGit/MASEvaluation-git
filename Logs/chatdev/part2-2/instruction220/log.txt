'''

main.py

This file contains the main application logic for the Jump Calculator App using tkinter for GUI.

'''

import tkinter as tk
from tkinter import messagebox

def calculate_number_of_jumps(d, jump_length):
    '''
    Calculate the minimum number of jumps required to reach (d, 0) from (0, 0).

    Parameters:
    d (int): The distance to the point (d, 0).
    jump_length (int): The length of each jump.

    Returns:
    int: The minimum number of jumps required.
    '''
    if jump_length <= 0:
        raise ValueError("Jump length must be greater than zero.")
    
    # Calculate the number of jumps required
    jumps = d // jump_length
    if d % jump_length != 0:
        jumps += 1
    return jumps

class JumpCalculatorApp:
    def __init__(self, master):
        '''
        Initialize the Jump Calculator App GUI components.

        Parameters:
        master (tk.Tk): The root window of the application.
        '''
        self.master = master
        master.title("Jump Calculator")

        self.label_d = tk.Label(master, text="Distance (d):")
        self.label_d.pack()

        self.entry_d = tk.Entry(master)
        self.entry_d.pack()

        self.label_jump_length = tk.Label(master, text="Jump Length:")

def calculate_number_of_jumps(d, jump_length):
    '''

    jumps = d // jump_length

        jumps += 1
    return jumps

            d = int(self.entry_d.get())

            messagebox.showerror("Error", str(e))

def calculate_number_of_jumps(d, jump_length):
    '''
    Calculate the minimum number of jumps required to reach (d, 0) from (0, 0).
    Parameters:
    d (int): The distance to the point (d, 0).
    jump_length (int): The length of each jump.
    Returns:
    int: The minimum number of jumps required.
    '''
    if jump_length <= 0:
        raise ValueError("Jump length must be greater than zero.")
    # Calculate the number of jumps required
    jumps = d // jump_length
    if d % jump_length != 0:
        jumps += 1
    return jumps
