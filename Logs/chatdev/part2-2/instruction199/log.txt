'''

This file contains the logic for processing IP addresses to remove leading zeroes.

'''

class IPAddressProcessor:
    def remove_leading_zeroes(self, ip_address):
        # Split the IP address into its components
        parts = ip_address.split('.')
        # Remove leading zeroes from each part
        processed_parts = [str(int(part)) for part in parts]
        # Join the processed parts back into an IP address
        return '.'.join(processed_parts)
```

'''
This file contains the logic for processing IP addresses to remove leading zeroes.
'''
class IPAddressProcessor:
    def remove_leading_zeroes(self, ip_address):
        # Split the IP address into its components
        parts = ip_address.split('.')
        # Remove leading zeroes from each part
        processed_parts = [str(int(part)) for part in parts]
        # Join the processed parts back into an IP address
        return '.'.join(processed_parts)

'''
This file contains the logic for processing IP addresses to remove leading zeroes.
'''
class IPAddressProcessor:
    def remove_leading_zeroes(self, ip_address):
        # Split the IP address into its components
        parts = ip_address.split('.')
        # Remove leading zeroes from each part
        processed_parts = [str(int(part)) for part in parts]
        # Join the processed parts back into an IP address
        return '.'.join(processed_parts)
