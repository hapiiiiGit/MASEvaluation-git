'''

This file initializes the GUI application, allowing users to input two lists and calculate the difference between them.

'''

import tkinter as tk
from tkinter import messagebox
from list_operations import get_difference

def calculate_difference():
    '''
    Fetches the input lists from the GUI, computes the difference using the get_difference function,
    and displays the result in the GUI.
    '''
    try:
        list1 = list(map(str.strip, entry_list1.get().split(',')))
        list2 = list(map(str.strip, entry_list2.get().split(',')))
        result = get_difference(list1, list2)
        result_label.config(text=f"Difference: {result}")
    except Exception as e:
        messagebox.showerror("Error", f"An error occurred: {e}")

def create_gui():
    '''
    Sets up the graphical user interface for the application.
    '''
    global entry_list1, entry_list2, result_label

    root = tk.Tk()
    root.title("List Difference Calculator")

    tk.Label(root, text="Enter List 1 (comma-separated):").grid(row=0, column=0, padx=10, pady=10)
    entry_list1 = tk.Entry(root, width=50)

    tk.Label(root, text="Enter List 2 (comma-separated):").grid(row=1, column=0, padx=10, pady=10)

    tk.Button(root, text="Calculate Difference", command=calculate_difference).grid(row=2, column=0, columnspan=2, pady=10)

    root.mainloop()

if __name__ == "__main__":

#### list_operations.py
```python
'''

This file contains the function to compute the difference between two lists.

'''

def get_difference(list1, list2):
    '''
    Computes the difference between two lists.
    Returns a list containing elements in list1 that are not in list2.
    '''
    return [item for item in list1 if item not in list2]
```

def get_difference(list1, list2):
    '''
    Computes the difference between two lists.
    Returns a list containing elements in list1 that are not in list2.
    '''
    try:
        # Ensure both inputs are lists
        if not isinstance(list1, list) or not isinstance(list2, list):

def get_difference(list1, list2):
    '''
    Computes the difference between two lists.
    Returns a list containing elements in list1 that are not in list2.
    '''
    try:
        # Ensure both inputs are lists
        if not isinstance(list1, list) or not isinstance(list2, list):
            raise ValueError("Both inputs must be lists.")
        # Compute the difference
        return [item for item in list1 if item not in list2]
    except Exception as e:
        # Raise an exception if any error occurs
        raise ValueError(f"Error computing difference: {e}")

def get_difference(list1, list2):
    '''
    Computes the difference between two lists.
    Returns a list containing elements in list1 that are not in list2.
    '''
    try:
        # Ensure both inputs are lists

# List Difference Calculator

A Python-based GUI application to compute the difference between two lists. This tool is designed to simplify the process of comparing two lists and identifying elements that are present in one list but not in the other.

## Main Features

1. **Graphical User Interface (GUI):** 
   - A user-friendly interface built with `tkinter` that allows users to input two lists and compute their difference.

2. **List Difference Calculation:**
   - Computes the difference between two lists, returning elements in the first list that are not present in the second list.

3. **Error Handling:**
   - Provides clear error messages if the input is invalid or if any unexpected issues occur during computation.

---

## Installation Guide

### Prerequisites

Ensure you have Python 3.6 or higher installed on your system. You can download Python from the [official Python website](https://www.python.org/downloads/).

### Install Required Dependencies

This application uses the `tkinter` library, which is included by default in most Python installations. However, if you encounter any issues, you can install it using the following command:

```bash
sudo apt-get install python3-tk
```

For Windows users, `tkinter` is included with Python, so no additional installation is required.

---

## How to Use the Application

### Step 1: Clone or Download the Repository

Download the source code files (`main.py` and `list_operations.py`) from the repository or copy them into a local directory on your system.

### Step 2: Run the Application

Navigate to the directory containing the files and execute the following command in your terminal or command prompt:

```bash
python main.py
```

### Step 3: Input the Lists

1. Enter the first list in the "Enter List 1 (comma-separated)" field. For example: `apple, banana, cherry`.
2. Enter the second list in the "Enter List 2 (comma-separated)" field. For example: `banana, cherry, date`.

### Step 4: Calculate the Difference

Click the "Calculate Difference" button. The application will compute the difference between the two lists and display the result in the "Difference" section.

For the example above, the result will be:

```
Difference: ['apple']
```

### Step 5: Handle Errors

If you enter invalid inputs (e.g., non-list data), the application will display an error message. Ensure that the inputs are comma-separated values.
