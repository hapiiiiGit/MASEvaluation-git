'''

import tkinter as tk
from tkinter import messagebox

def find_minimum_of_three(a, b, c):
    '''
    DOCSTRING: Function to find the minimum of three numbers.
    Parameters:
        a (float): First number.
        b (float): Second number.
        c (float): Third number.
    Returns:
        float: The minimum of the three numbers.
    '''
    return min(a, b, c)

class MinFinderApp:
    '''
    DOCSTRING: Class to create a GUI application for finding the minimum of three numbers.
    '''
    def __init__(self, master):
        '''
        DOCSTRING: Initializes the GUI components.
        Parameters:
            master (tk.Tk): The root window of the application.
        '''
        self.master = master
        master.title("Minimum Finder")

        self.label = tk.Label(master, text="Enter three numbers:")
        self.label.pack()

        self.entry1 = tk.Entry(master)
        self.entry1.pack()

        self.entry2 = tk.Entry(master)
        self.entry2.pack()

        self.entry3 = tk.Entry(master)
        self.entry3.pack()

        self.find_button = tk.Button(master, text="Find Minimum", command=self.find_minimum)
        self.find_button.pack()

        self.result_label = tk.Label(master, text="")
        self.result_label.pack()

    def find_minimum(self):
        '''
        DOCSTRING: Retrieves input values, computes the minimum, and displays the result.
        '''
        try:
            a = float(self.entry1.get())
            b = float(self.entry2.get())
            c = float(self.entry3.get())
            minimum = find_minimum_of_three(a, b, c)
            self.result_label.config(text=f"The minimum is: {minimum}")
        except ValueError:
            messagebox.showerror("Invalid input", "Please enter valid numbers.")

if __name__ == "__main__":
    root = tk.Tk()
    app = MinFinderApp(root)
    root.mainloop()

    '''

import tkinter as tk
from tkinter import messagebox

def find_minimum_of_three(a, b, c):
    '''
    Function to find the minimum of three numbers.
    Parameters:
        a (float): First number.
        b (float): Second number.
        c (float): Third number.
    Returns:
        float: The minimum of the three numbers.
    '''
    return min(a, b, c)

class MinFinderApp:
    '''
    Class to create a GUI application for finding the minimum of three numbers.
    '''
    def __init__(self, master):
        '''
        Initializes the GUI components.
        Parameters:
            master (tk.Tk): The root window of the application.
        '''
        self.master = master
        master.title("Minimum Finder")
        
        self.label = tk.Label(master, text="Enter three numbers:")
        self.label.pack()
        
        self.entry1 = tk.Entry(master)
        self.entry1.pack()
        
        self.entry2 = tk.Entry(master)
        self.entry2.pack()
        
        self.entry3 = tk.Entry(master)
        self.entry3.pack()
        
        self.find_button = tk.Button(master, text="Find Minimum", command=self.find_minimum)
        self.find_button.pack()
        
        self.result_label = tk.Label(master, text="")
        self.result_label.pack()

    def find_minimum(self):
        '''
        Retrieves input values, computes the minimum, and displays the result.
        '''
        try:
            a = float(self.entry1.get())
            b = float(self.entry2.get())
            c = float(self.entry3.get())
            minimum = find_minimum_of_three(a, b, c)
            self.result_label.config(text=f"The minimum is: {minimum}")
        except ValueError:
            messagebox.showerror("Invalid input", "Please enter valid numbers.")

if __name__ == "__main__":
    root = tk.Tk()
    app = MinFinderApp(root)
    root.mainloop()

def find_minimum_of_three(a, b, c):
    '''

def find_minimum_of_three(a, b, c):
    '''

Welcome to the Minimum Finder Application! This software provides a simple graphical user interface (GUI) to find the minimum of three numbers. It is designed to be user-friendly and efficient, making it easy for anyone to use.

## Main Functions of the Software

- **Find Minimum of Three Numbers**: The core functionality of this application is to compute and display the minimum value among three user-input numbers.
- **User-Friendly GUI**: The application features a straightforward interface where users can input numbers and receive results instantly.

## Installation Guide

### Prerequisites

- **Python**: Ensure that Python is installed on your system. You can download it from [python.org](https://www.python.org/downloads/).

### Quick Install

1. **Clone the Repository**: Download or clone the project repository to your local machine.

2. **Navigate to the Project Directory**: Open a terminal or command prompt and navigate to the directory where the project files are located.

3. **Install Dependencies**: The application does not require any additional Python packages beyond the standard library. Therefore, no extra installation steps are necessary.

## How to Use the Application

1. **Run the Application**: Open a terminal or command prompt, navigate to the project directory, and execute the following command to start the application:

   ```bash
   python main.py
   ```

2. **Enter Numbers**: Once the application window opens, you will see three input fields. Enter the three numbers for which you want to find the minimum.

3. **Find Minimum**: Click the "Find Minimum" button. The application will compute the minimum of the three numbers and display the result below the button.

4. **Error Handling**: If you enter invalid inputs (e.g., non-numeric values), the application will display an error message prompting you to enter valid numbers.

- **No Additional Packages Required**: The application is built using Python's standard library, specifically the `tkinter` module for the GUI, which is included with Python installations.

- **Cross-Platform Compatibility**: The application can run on any operating system that supports Python and `tkinter`, including Windows, macOS, and Linux.
