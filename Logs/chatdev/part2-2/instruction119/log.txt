Main GUI application for checking common elements between two lists.

import tkinter as tk
from common_elements import has_common_elements

class MainApplication:
    def __init__(self, root):
        self.root = root
        self.root.title("Common Elements Checker")

        # Labels
        self.label1 = tk.Label(root, text="Enter first list (comma-separated):")
        self.label1.pack()
        self.label2 = tk.Label(root, text="Enter second list (comma-separated):")
        self.label2.pack()

        # Entry fields
        self.entry1 = tk.Entry(root, width=50)
        self.entry1.pack()
        self.entry2 = tk.Entry(root, width=50)
        self.entry2.pack()

        # Button
        self.check_button = tk.Button(root, text="Check Common Elements", command=self.check_common_elements)
        self.check_button.pack()

        # Result label
        self.result_label = tk.Label(root, text="")
        self.result_label.pack()

    def check_common_elements(self):
        list1 = self.entry1.get().split(',')
        list2 = self.entry2.get().split(',')
        result = has_common_elements(list1, list2)
        if result:
            self.result_label.config(text="The lists have common elements.")
        else:
            self.result_label.config(text="The lists do not have common elements.")

if __name__ == "__main__":
    root = tk.Tk()
    app = MainApplication(root)
    root.mainloop()

```python
'''

Function to check if two lists have at least one common element.

'''

def has_common_elements(list1, list2):
    # Convert lists to sets and check for intersection
    set1 = set(list1)
    set2 = set(list2)
    return not set1.isdisjoint(set2)
```

'''

Main GUI application for checking common elements between two lists.

'''

import tkinter as tk
from common_elements import has_common_elements

class MainApplication:
    def __init__(self, root):
        self.root = root
        self.root.title("Common Elements Checker")

        # Labels
        self.label1 = tk.Label(root, text="Enter first list (comma-separated):")
        self.label1.pack()
        self.label2 = tk.Label(root, text="Enter second list (comma-separated):")
        self.label2.pack()

        # Entry fields
        self.entry1 = tk.Entry(root, width=50)
        self.entry1.pack()
        self.entry2 = tk.Entry(root, width=50)
        self.entry2.pack()

        # Button
        self.check_button = tk.Button(root, text="Check Common Elements", command=self.check_common_elements)
        self.check_button.pack()

        # Result label
        self.result_label = tk.Label(root, text="")
        self.result_label.pack()

    def check_common_elements(self):
        list1 = self.entry1.get().split(',')
        list2 = self.entry2.get().split(',')
        result = has_common_elements(list1, list2)
        if result:
            self.result_label.config(text="The lists have common elements.")
        else:
            self.result_label.config(text="The lists do not have common elements.")

if __name__ == "__main__":
    root = tk.Tk()
    app = MainApplication(root)
    root.mainloop()
```

'''

Function to check if two lists have at least one common element.

'''

def has_common_elements(list1, list2):
    # Convert lists to sets and check for intersection
    set1 = set(list1)
    set2 = set(list2)
    return not set1.isdisjoint(set2)

    set2 = set(list2)
    return not set1.isdisjoint(set2)
