'''

This module contains the main application class for the List Counter App and a utility function to count lists within a list.

'''

import tkinter as tk
from tkinter import messagebox

def count_lists_in_list(list_of_lists):
    '''
    Convert each list in the list_of_lists to a tuple and count its occurrences.
    
    Parameters:
    list_of_lists (list): A list containing other lists.
    
    Returns:
    dict: A dictionary with tuples as keys and their occurrence counts as values.
    '''
    count_dict = {}
    for lst in list_of_lists:
        tuple_representation = tuple(lst)
        if tuple_representation in count_dict:
            count_dict[tuple_representation] += 1
        else:
            count_dict[tuple_representation] = 1
    return count_dict

class ListCounterApp:
    '''
    A GUI application to count lists within a list and display the result.
    '''
    def __init__(self, root):

import tkinter as tk
from tkinter import messagebox
import ast
def count_lists_in_list(list_of_lists):
    '''
    Convert each list in the list_of_lists to a tuple and count its occurrences.
    Parameters:
    list_of_lists (list): A list containing other lists.
    Returns:
    dict: A dictionary with tuples as keys and their occurrence counts as values.
    '''
    count_dict = {}
    for lst in list_of_lists:
        tuple_representation = tuple(lst)
        if tuple_representation in count_dict:
            count_dict[tuple_representation] += 1
        else:
            count_dict[tuple_representation] = 1
    return count_dict
class ListCounterApp:
    '''
    A GUI application to count lists within a list and display the result.
    '''

        self.root = root
        self.root.title("List Counter App")

        self.result_text = tk.Text(root, height=10, width=50)
        self.result_text.pack()
    def count_lists(self):
        '''
        Count the lists within the input and display the result.
        '''
        try:
            input_text = self.entry.get()

            if isinstance(list_of_lists, list) and all(isinstance(i, list) for i in list_of_lists):
                result = count_lists_in_list(list_of_lists)
                self.result_text.delete(1.0, tk.END)
                self.result_text.insert(tk.END, str(result))
            else:
                raise ValueError
        except Exception as e:
            messagebox.showerror("Invalid Input", "Please enter a valid list of lists.")
if __name__ == "__main__":

'''
This module contains the main application class for the List Counter App and a utility function to count lists within a list.
'''
import tkinter as tk
from tkinter import messagebox
import ast
def count_lists_in_list(list_of_lists):
    '''
    Convert each list in the list_of_lists to a tuple and count its occurrences.
    Parameters:
    list_of_lists (list): A list containing other lists.
    Returns:
    dict: A dictionary with tuples as keys and their occurrence counts as values.
    '''
    count_dict = {}
    for lst in list_of_lists:
        tuple_representation = tuple(lst)
        if tuple_representation in count_dict:
            count_dict[tuple_representation] += 1
        else:
            count_dict[tuple_representation] = 1
    return count_dict
class ListCounterApp:
    '''
    A GUI application to count lists within a list and display the result.
    '''
    def __init__(self, root):
        self.root = root
        self.root.title("List Counter App")
        self.label = tk.Label(root, text="Enter a list of lists (e.g., [[1, 2], [3, 4], [1, 2]]):")
        self.label.pack()
        self.entry = tk.Entry(root, width=50)
        self.entry.pack()
        self.count_button = tk.Button(root, text="Count Lists", command=self.count_lists)
        self.count_button.pack()
        self.result_label = tk.Label(root, text="Result:")
        self.result_label.pack()
        self.result_text = tk.Text(root, height=10, width=50)
        self.result_text.pack()
    def count_lists(self):
        '''
        Count the lists within the input and display the result.
        '''
        try:
            input_text = self.entry.get()
            list_of_lists = ast.literal_eval(input_text)  # Safely convert string input to a list of lists
            if isinstance(list_of_lists, list) and all(isinstance(i, list) for i in list_of_lists):
                result = count_lists_in_list(list_of_lists)
                self.result_text.delete(1.0, tk.END)
                self.result_text.insert(tk.END, str(result))
            else:
                raise ValueError
        except Exception as e:
            messagebox.showerror("Invalid Input", "Please enter a valid list of lists.")
if __name__ == "__main__":
    root = tk.Tk()
    app = ListCounterApp(root)
    root.mainloop()

- **Count Lists in a List**: The core functionality of the app is to take a list of lists as input and return a dictionary where each list is converted to a tuple, and the value of each tuple is the number of its occurrences in the original list.

- **Graphical User Interface (GUI)**: The app provides a simple and intuitive GUI for users to input their data and view the results.
