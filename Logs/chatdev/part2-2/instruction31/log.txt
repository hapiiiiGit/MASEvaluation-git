def is_prime(n):
    '''
    Checks if a number is prime.
    :param n: Integer to check.
    :return: True if n is prime, False otherwise.
    '''
    if n < 2:
        return False
    for i in range(2, int(n**0.5) + 1):
        if n % i == 0:
            return False
    return True
def count_primes_less_than(n):
    '''
    Counts the number of prime numbers less than a given number.
    :param n: Non-negative integer.
    :return: Count of prime numbers less than n.
    '''
    count = 0
    for i in range(n):
        if is_prime(i):
            count += 1
    return count

def is_prime(n):
    '''
    Checks if a number is prime.
    :param n: Integer to check.
    :return: True if n is prime, False otherwise.
    '''
    if n < 2:

    :param n: Non-negative integer.

def is_prime(n):
    '''
    Checks if a number is prime.
    :param n: Integer to check.
    :return: True if n is prime, False otherwise.
    '''
    if n < 2:

    for i in range(2, int(n**0.5) + 1):

    :param n: Non-negative integer.

def is_prime(n):
    '''
    Checks if a number is prime.
    :param n: Integer to check.
    :return: True if n is prime, False otherwise.
    '''
    if n < 2:
        return False
    for i in range(2, int(n**0.5) + 1):
        if n % i == 0:
            return False
    return True

    '''
    Counts the number of prime numbers less than a given number.
    :param n: Non-negative integer.
    :return: Count of prime numbers less than n.
    '''
    count = 0
    for i in range(n):
        if is_prime(i):
            count += 1
    return count

import tkinter as tk
from tkinter import messagebox
from prime_utils import count_primes_less_than
class PrimeCounterApp:
    '''
    A class to define the GUI application for counting prime numbers less than a given number.
    '''
    def __init__(self, root):
        '''
        Initializes the GUI application with necessary components.
        :param root: The root Tkinter window.
        '''
        self.root = root
        self.root.title("Prime Counter App")
        self.root.geometry("400x200")
        # Label for instruction
        self.label = tk.Label(root, text="Enter a non-negative number:", font=("Arial", 12))
        self.label.pack(pady=10)
        # Entry for user input
        self.entry = tk.Entry(root, font=("Arial", 12))
        self.entry.pack(pady=10)
        # Button to calculate primes
        self.button = tk.Button(root, text="Count Primes", command=self.calculate_primes, font=("Arial", 12))
        self.button.pack(pady=10)
        # Label to display the result
        self.result_label = tk.Label(root, text="", font=("Arial", 12))
        self.result_label.pack(pady=10)
    def calculate_primes(self):
        '''
        Handles the calculation of prime numbers and updates the GUI with the result.
        '''
        try:
            # Get user input
            user_input = self.entry.get()
            number = int(user_input)
            if number < 0:
                raise ValueError("The number must be non-negative.")
            # Calculate the number of primes less than the input
            prime_count = count_primes_less_than(number)
            # Display the result
            self.result_label.config(text=f"Number of primes less than {number}: {prime_count}")
        except ValueError as e:
            # Handle invalid input
            messagebox.showerror("Invalid Input", str(e))
# Main entry point for the application
if __name__ == "__main__":
    root = tk.Tk()
    app = PrimeCounterApp(root)
    root.mainloop()

'''
This file contains utility functions for prime number calculations, including checking if a number is prime and counting primes less than a given number.
'''
def is_prime(n):
    '''
    Checks if a number is prime.
    :param n: Integer to check.
    :return: True if n is prime, False otherwise.
    '''
    if n < 2:
        return False
    for i in range(2, int(n**0.5) + 1):
        if n % i == 0:
            return False
    return True
def count_primes_less_than(n):
    '''
    Counts the number of prime numbers less than a given number.
    :param n: Non-negative integer.
    :return: Count of prime numbers less than n.
    '''
    count = 0
    for i in range(n):
        if is_prime(i):
            count += 1
    return count

'''
This file contains utility functions for prime number calculations, including checking if a number is prime and counting primes less than a given number.
'''
def is_prime(n):
    '''
    Checks if a number is prime.
    :param n: Integer to check.
    :return: True if n is prime, False otherwise.
    '''
    if n < 2:
        return False
    for i in range(2, int(n**0.5) + 1):
        if n % i == 0:
            return False
    return True
def count_primes_less_than(n):
    '''
    Counts the number of prime numbers less than a given number.
    :param n: Non-negative integer.
    :return: Count of prime numbers less than n.
    '''
    count = 0
    for i in range(n):
        if is_prime(i):
            count += 1
    return count
