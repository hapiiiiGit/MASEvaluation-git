def subtract_tuples(tuple1, tuple2):
    '''
    Subtracts the elements of the first tuple by the elements of the second tuple element-wise.
    Args:
        tuple1 (tuple): The first tuple.
        tuple2 (tuple): The second tuple.
    Returns:
        tuple: A tuple containing the result of the subtraction.
    Raises:
        IndexError: If the tuples are not of the same length.
    '''
    if len(tuple1) != len(tuple2):
        raise IndexError("Tuples must have the same length.")
    return tuple(a - b for a, b in zip(tuple1, tuple2))

    '''

    '''

        '''
        Initializes the GUI components.
        '''

        self.entry1.pack()

import tkinter as tk
from tkinter import messagebox
from utils import subtract_tuples
class TupleSubtractionApp:
    '''
    A GUI application for subtracting two tuples element-wise.
    '''
    def __init__(self, root):
        '''
        Initializes the GUI components.
        '''
        self.root = root
        self.root.title("Tuple Subtraction App")
        # Input labels and entry fields
        self.label1 = tk.Label(root, text="Enter the first tuple (comma-separated):")
        self.label1.pack()
        self.entry1 = tk.Entry(root, width=50)
        self.entry1.pack()
        self.label2 = tk.Label(root, text="Enter the second tuple (comma-separated):")
        self.label2.pack()
        self.entry2 = tk.Entry(root, width=50)
        self.entry2.pack()
        # Submit button
        self.submit_button = tk.Button(root, text="Subtract", command=self.subtract_tuples)
        self.submit_button.pack()
        # Result label
        self.result_label = tk.Label(root, text="Result: ", font=("Arial", 12))
        self.result_label.pack()
    def subtract_tuples(self):
        '''
        Handles the subtraction logic and updates the result in the GUI.
        '''
        try:
            # Parse the input tuples
            tuple1 = tuple(map(int, self.entry1.get().strip().split(',')))
            tuple2 = tuple(map(int, self.entry2.get().strip().split(',')))
            # Perform the subtraction
            result = subtract_tuples(tuple1, tuple2)
            # Display the result
            self.result_label.config(text=f"Result: {result}")
        except ValueError:
            messagebox.showerror("Input Error", "Please enter valid tuples with integers.")
        except IndexError:
            messagebox.showerror("Input Error", "Tuples must have the same length.")
        except Exception as e:
            messagebox.showerror("Error", str(e))
if __name__ == "__main__":
    root = tk.Tk()
    app = TupleSubtractionApp(root)
    root.mainloop()

def subtract_tuples(tuple1, tuple2):
    '''
    Subtracts the elements of the first tuple by the elements of the second tuple element-wise.
    Args:
        tuple1 (tuple): The first tuple.
        tuple2 (tuple): The second tuple.
    Returns:
        tuple: A tuple containing the result of the subtraction.
    Raises:
        IndexError: If the tuples are not of the same length.
    '''
    if len(tuple1) != len(tuple2):
        raise IndexError("Tuples must have the same length.")
    return tuple(a - b for a, b in zip(tuple1, tuple2))

    Subtracts the elements of the first tuple by the elements of the second tuple element-wise.
    Args:
        tuple1 (tuple): The first tuple.
        tuple2 (tuple): The second tuple.
    Returns:
        tuple: A tuple containing the result of the subtraction.
    Raises:
        IndexError: If the tuples are not of the same length.
    '''
    if len(tuple1) != len(tuple2):
        raise IndexError("Tuples must have the same length.")
    return tuple(a - b for a, b in zip(tuple1, tuple2))
