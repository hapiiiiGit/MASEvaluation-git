```python
'''

Main application file to create a GUI for computing the sum of digits of each number in a given list.

'''

import tkinter as tk
from tkinter import messagebox
from utils import sum_of_digits

class SumOfDigitsApp:
    def __init__(self, root):
        '''
        Initialize the main application window and setup the GUI layout.
        '''
        self.root = root
        self.root.title("Sum of Digits Calculator")
        
        self.create_widgets()

    def create_widgets(self):
        '''
        Create and place the widgets in the main application window.
        '''
        self.label = tk.Label(self.root, text="Enter a list of numbers separated by commas:")
        self.label.pack(pady=10)

        self.entry = tk.Entry(self.root, width=50)
        self.entry.pack(pady=10)

        self.calculate_button = tk.Button(self.root, text="Calculate", command=self.calculate_sum_of_digits)
        self.calculate_button.pack(pady=10)

        self.result_label = tk.Label(self.root, text="")
        self.result_label.pack(pady=10)

    def calculate_sum_of_digits(self):
        '''
        Calculate the sum of digits for each number in the input list and display the result.
        '''
        input_text = self.entry.get()
        try:
            numbers = [int(num.strip()) for num in input_text.split(',')]
            results = [sum_of_digits(num) for num in numbers]
            self.result_label.config(text=f"Sum of digits: {results}")
        except ValueError:
            messagebox.showerror("Invalid input", "Please enter a valid list of integers separated by commas.")

if __name__ == "__main__":
    root = tk.Tk()
    app = SumOfDigitsApp(root)
    root.mainloop()
```

### utils.py

```python
'''

Utility functions for the Sum of Digits application.

'''

def sum_of_digits(number):
    '''
    Calculate the sum of digits of a given integer.
    
    :param number: Integer whose digits are to be summed.
    :return: Sum of the digits of the number.
    '''
    return sum(int(digit) for digit in str(abs(number)))
```

'''

Main application file to create a GUI for computing the sum of digits of each number in a given list.

'''

import tkinter as tk
from tkinter import messagebox
from utils import sum_of_digits

class SumOfDigitsApp:
    def __init__(self, root):
        '''
        Initialize the main application window and setup the GUI layout.
        '''
        self.root = root
        self.root.title("Sum of Digits Calculator")
        
        self.create_widgets()

    def create_widgets(self):
        '''
        Create and place the widgets in the main application window.
        '''
        self.label = tk.Label(self.root, text="Enter a list of numbers separated by commas:")
        self.label.pack(pady=10)

        self.entry = tk.Entry(self.root, width=50)
        self.entry.pack(pady=10)

        self.calculate_button = tk.Button(self.root, text="Calculate", command=self.calculate_sum_of_digits)
        self.calculate_button.pack(pady=10)

        self.result_label = tk.Label(self.root, text="")
        self.result_label.pack(pady=10)

    def calculate_sum_of_digits(self):
        '''
        Calculate the sum of digits for each number in the input list and display the result.
        '''
        input_text = self.entry.get()
        try:
            numbers = [int(num.strip()) for num in input_text.split(',')]
            results = [sum_of_digits(num) for num in numbers]
            self.result_label.config(text=f"Sum of digits: {results}")
        except ValueError:
            messagebox.showerror("Invalid input", "Please enter a valid list of integers separated by commas.")

if __name__ == "__main__":
    root = tk.Tk()
    app = SumOfDigitsApp(root)
    root.mainloop()

import tkinter as tk
from tkinter import messagebox
from utils import sum_of_digits
class SumOfDigitsApp:
    def __init__(self, root):
        '''
        Initialize the main application window and setup the GUI layout.
        '''
        self.root = root
        self.root.title("Sum of Digits Calculator")
        self.create_widgets()
    def create_widgets(self):
        '''
        Create and place the widgets in the main application window.
        '''
        self.label = tk.Label(self.root, text="Enter a list of numbers separated by commas:")
        self.label.pack(pady=10)
        self.entry = tk.Entry(self.root, width=50)
        self.entry.pack(pady=10)
        self.calculate_button = tk.Button(self.root, text="Calculate", command=self.calculate_sum_of_digits)
        self.calculate_button.pack(pady=10)
        self.result_label = tk.Label(self.root, text="")
        self.result_label.pack(pady=10)
    def calculate_sum_of_digits(self):
        '''
        Calculate the sum of digits for each number in the input list and display the result.
        '''
        input_text = self.entry.get()
        try:
            numbers = [int(num.strip()) for num in input_text.split(',')]
            results = [sum_of_digits(num) for num in numbers]
            self.result_label.config(text=f"Sum of digits: {results}")
        except ValueError:
            messagebox.showerror("Invalid input", "Please enter a valid list of integers separated by commas.")
if __name__ == "__main__":
    root = tk.Tk()
    app = SumOfDigitsApp(root)
    root.mainloop()

'''
Main application file to create a GUI for computing the sum of digits of each number in a given list.
'''
import tkinter as tk
from tkinter import messagebox
from utils import sum_of_digits
class SumOfDigitsApp:
    def __init__(self, root):
        '''
        Initialize the main application window and setup the GUI layout.
        '''
        self.root = root
        self.root.title("Sum of Digits Calculator")
        self.create_widgets()
    def create_widgets(self):
        '''
        Create and place the widgets in the main application window.
        '''
        self.label = tk.Label(self.root, text="Enter a list of numbers separated by commas:")
        self.label.pack(pady=10)
        self.entry = tk.Entry(self.root, width=50)
        self.entry.pack(pady=10)
        self.calculate_button = tk.Button(self.root, text="Calculate", command=self.calculate_sum_of_digits)
        self.calculate_button.pack(pady=10)
        self.result_label = tk.Label(self.root, text="")
        self.result_label.pack(pady=10)
    def calculate_sum_of_digits(self):
        '''
        Calculate the sum of digits for each number in the input list and display the result.
        '''
        input_text = self.entry.get()
        try:
            numbers = [int(num.strip()) for num in input_text.split(',')]
            results = [sum_of_digits(num) for num in numbers]
            self.result_label.config(text=f"Sum of digits: {results}")
        except ValueError:
            messagebox.showerror("Invalid input", "Please enter a valid list of integers separated by commas.")
if __name__ == "__main__":
    root = tk.Tk()
    app = SumOfDigitsApp(root)
    root.mainloop()

'''
Utility functions for the Sum of Digits application.

def sum_of_digits(number):
    '''
    Calculate the sum of digits of a given integer.
    :param number: Integer whose digits are to be summed.
    :return: Sum of the digits of the number.
    '''
    return sum(int(digit) for digit in str(abs(number)))
