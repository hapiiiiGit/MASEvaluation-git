'''

This module contains the main application class for the Tuple Adder application and a utility function to perform index-wise addition of nested tuples.

'''

import tkinter as tk
from tkinter import messagebox

def add_nested_tuples(tuple1, tuple2):
    '''
    Perform index-wise addition of two nested tuples.
    
    Parameters:
    tuple1 (tuple): The first nested tuple.
    tuple2 (tuple): The second nested tuple.
    
    Returns:
    tuple: A new tuple with index-wise addition of the input tuples.
    '''
    if len(tuple1) != len(tuple2):
        raise ValueError("Tuples must be of the same length")
    
    result = tuple(a + b for a, b in zip(tuple1, tuple2))
    return result

        self.entry_tuple2.grid(row=1, column=1, padx=10, pady=10)

        tk.Label(root, text="Enter Tuple 1:").grid(row=0, column=0, padx=10, pady=10)

        # Button to perform addition
        self.add_button = tk.Button(root, text="Add Tuples", command=self.add_tuples)
        self.add_button.grid(row=2, column=0, columnspan=2, pady=10)

        self.result_label.grid(row=3, column=0, columnspan=2, pady=10)

'''

This module contains the main application class for the Tuple Adder application and a utility function to perform index-wise addition of nested tuples.

'''

import tkinter as tk
from tkinter import messagebox

def add_nested_tuples(tuple1, tuple2):
    '''
    Perform index-wise addition of two nested tuples.
    
    Parameters:
    tuple1 (tuple): The first nested tuple.
    tuple2 (tuple): The second nested tuple.
    
    Returns:
    tuple: A new tuple with index-wise addition of the input tuples.
    '''
    if len(tuple1) != len(tuple2):
        raise ValueError("Tuples must be of the same length")
    
    result = tuple(a + b for a, b in zip(tuple1, tuple2))
    return result

    '''
    A class to create a GUI application for adding nested tuples index-wise.
    '''

        self.add_button = tk.Button(root, text="Add Tuples", command=self.add_tuples)
        self.add_button.grid(row=2, column=0, columnspan=2, pady=10)
        
        # Result display
        self.result_label = tk.Label(root, text="Result: ")
        self.result_label.grid(row=3, column=0, columnspan=2, pady=10)
    
    def add_tuples(self):
        '''
        Retrieve tuples from entry fields, perform index-wise addition, and display the result.
        '''
        try:
            # Convert input strings to tuples
            tuple1 = eval(self.entry_tuple1.get())
            tuple2 = eval(self.entry_tuple2.get())
            
            # Validate input
            if not (isinstance(tuple1, tuple) and isinstance(tuple2, tuple)):
                raise ValueError("Inputs must be tuples")
            
            # Perform addition
            result = add_nested_tuples(tuple1, tuple2)
            
            # Display result
            self.result_label.config(text=f"Result: {result}")
        
        except Exception as e:
            messagebox.showerror("Error", str(e))

if __name__ == "__main__":
    root = tk.Tk()
    app = TupleAdderApp(root)
    root.mainloop()

import tkinter as tk
from tkinter import messagebox
import ast
def add_nested_tuples(tuple1, tuple2):
    '''
    Perform index-wise addition of two nested tuples.
    Parameters:
    tuple1 (tuple): The first nested tuple.
    tuple2 (tuple): The second nested tuple.
    Returns:
    tuple: A new tuple with index-wise addition of the input tuples.
    '''
    if len(tuple1) != len(tuple2):
        raise ValueError("Tuples must be of the same length")
    result = tuple(a + b for a, b in zip(tuple1, tuple2))

    '''
    A class to create a GUI application for adding nested tuples index-wise.
    '''

        Initialize the TupleAdderApp with the given root window.
        Parameters:
        root (Tk): The root window of the tkinter application.

        self.root.title("Tuple Adder")

        self.entry_tuple1 = tk.Entry(root, width=50)

        '''
        Retrieve tuples from entry fields, perform index-wise addition, and display the result.

'''
This module contains the main application class for the Tuple Adder application and a utility function to perform index-wise addition of nested tuples.
'''
import tkinter as tk
from tkinter import messagebox
import ast
def add_nested_tuples(tuple1, tuple2):
    '''
    Perform index-wise addition of two nested tuples.
    Parameters:
    tuple1 (tuple): The first nested tuple.
    tuple2 (tuple): The second nested tuple.
    Returns:
    tuple: A new tuple with index-wise addition of the input tuples.
    '''
    if len(tuple1) != len(tuple2):
        raise ValueError("Tuples must be of the same length")
    result = tuple(a + b for a, b in zip(tuple1, tuple2))
    return result
class TupleAdderApp:
    '''
    A class to create a GUI application for adding nested tuples index-wise.
    '''
    def __init__(self, root):
        '''
        Initialize the TupleAdderApp with the given root window.
        Parameters:
        root (Tk): The root window of the tkinter application.
        '''
        self.root = root
        self.root.title("Tuple Adder")
        # Entry fields for tuples
        self.entry_tuple1 = tk.Entry(root, width=50)
        self.entry_tuple1.grid(row=0, column=1, padx=10, pady=10)
        self.entry_tuple2 = tk.Entry(root, width=50)
        self.entry_tuple2.grid(row=1, column=1, padx=10, pady=10)
        # Labels
        tk.Label(root, text="Enter Tuple 1:").grid(row=0, column=0, padx=10, pady=10)
        tk.Label(root, text="Enter Tuple 2:").grid(row=1, column=0, padx=10, pady=10)
        # Button to perform addition
        self.add_button = tk.Button(root, text="Add Tuples", command=self.add_tuples)
        self.add_button.grid(row=2, column=0, columnspan=2, pady=10)
        # Result display
        self.result_label = tk.Label(root, text="Result: ")
        self.result_label.grid(row=3, column=0, columnspan=2, pady=10)
    def add_tuples(self):
        '''
        Retrieve tuples from entry fields, perform index-wise addition, and display the result.
        '''
        try:
            # Convert input strings to tuples using ast.literal_eval for safety
            tuple1 = ast.literal_eval(self.entry_tuple1.get())
            tuple2 = ast.literal_eval(self.entry_tuple2.get())
            # Validate input
            if not (isinstance(tuple1, tuple) and isinstance(tuple2, tuple)):
                raise ValueError("Inputs must be tuples")
            # Perform addition
            result = add_nested_tuples(tuple1, tuple2)
            # Display result
            self.result_label.config(text=f"Result: {result}")
        except Exception as e:
            messagebox.showerror("Error", str(e))
if __name__ == "__main__":
    root = tk.Tk()
    app = TupleAdderApp(root)
    root.mainloop()
