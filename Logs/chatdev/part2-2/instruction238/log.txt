'''

This module contains the WindChillCalculator class which provides a method to calculate the wind chill index.

'''

class WindChillCalculator:
    @staticmethod
    def calculate_wind_chill(temperature, wind_velocity):
        """
        Calculate the wind chill index given temperature in Celsius and wind velocity in km/h.
        The formula used is specific for temperatures below 10°C and wind speeds above 4.8 km/h.
        """
        if temperature > 10 or wind_velocity < 4.8:
            return temperature
        
        wind_chill_index = 13.12 + 0.6215 * temperature - 11.37 * (wind_velocity ** 0.16) + 0.3965 * temperature * (wind_velocity ** 0.16)
        return round(wind_chill_index)
```

'''

This module contains the WindChillCalculator class which provides a method to calculate the wind chill index.

'''

class WindChillCalculator:
    @staticmethod
    def calculate_wind_chill(temperature, wind_velocity):
        """
        Calculate the wind chill index given temperature in Celsius and wind velocity in km/h.
        The formula used is specific for temperatures below 10°C and wind speeds above 4.8 km/h.
        """
        if temperature > 10 or wind_velocity < 4.8:
            return temperature
        
        wind_chill_index = 13.12 + 0.6215 * temperature - 11.37 * (wind_velocity ** 0.16) + 0.3965 * temperature * (wind_velocity ** 0.16)
        return round(wind_chill_index)

'''
This module contains the WindChillCalculator class which provides a method to calculate the wind chill index.
'''
class WindChillCalculator:
    @staticmethod
    def calculate_wind_chill(temperature, wind_velocity):
        """
        Calculate the wind chill index given temperature in Celsius and wind velocity in km/h.
        The formula used is specific for temperatures below 10°C and wind speeds above 4.8 km/h.
        """
        if temperature > 10 or wind_velocity < 4.8:
            return temperature
        wind_chill_index = 13.12 + 0.6215 * temperature - 11.37 * (wind_velocity ** 0.16) + 0.3965 * temperature * (wind_velocity ** 0.16)
        return round(wind_chill_index)

'''
This module contains the WindChillCalculator class which provides a method to calculate the wind chill index.
'''
class WindChillCalculator:
    @staticmethod
    def calculate_wind_chill(temperature, wind_velocity):
        """
        Calculate the wind chill index given temperature in Celsius and wind velocity in km/h.
        The formula used is specific for temperatures below 10°C and wind speeds above 4.8 km/h.
        """
        if temperature > 10 or wind_velocity < 4.8:
            return temperature
        wind_chill_index = 13.12 + 0.6215 * temperature - 11.37 * (wind_velocity ** 0.16) + 0.3965 * temperature * (wind_velocity ** 0.16)
        return round(wind_chill_index)

'''
This module contains the WindChillCalculator class which provides a method to calculate the wind chill index.
'''
class WindChillCalculator:
    @staticmethod
    def calculate_wind_chill(temperature, wind_velocity):
        """
        Calculate the wind chill index given temperature in Celsius and wind velocity in km/h.
        The formula used is specific for temperatures below 10°C and wind speeds above 4.8 km/h.
        """
        if temperature > 10 or wind_velocity < 4.8:
            return temperature
        wind_chill_index = 13.12 + 0.6215 * temperature - 11.37 * (wind_velocity ** 0.16) + 0.3965 * temperature * (wind_velocity ** 0.16)
        return round(wind_chill_index)
