'''

This module contains the MainApp class which creates a GUI application using Tkinter to count occurrences of list elements in a tuple.

'''

import tkinter as tk
from tkinter import messagebox

def count_elements_in_tuple(lst, tpl):
    '''
    Count occurrences of each element in the list within the tuple.

    Parameters:
    lst (list): The list of elements to count.
    tpl (tuple): The tuple in which to count occurrences.

    Returns:
    dict: A dictionary with list elements as keys and their counts in the tuple as values.
    '''
    count_dict = {}
    for element in lst:
        count_dict[element] = tpl.count(element)
    return count_dict

class MainApp:
    '''
    Main application class to create and manage the GUI.
    '''
    def __init__(self, root):
        self.root = root
        self.root.title("List Element Counter in Tuple")
        self.create_widgets()

    def create_widgets(self):
        '''
        Create and place the widgets in the application window.
        '''
        self.label_list = tk.Label(self.root, text="Enter list (comma separated):")
        self.label_list.pack()

        self.entry_list = tk.Entry(self.root, width=50)
        self.entry_list.pack()

        self.label_tuple = tk.Label(self.root, text="Enter tuple (comma separated):")
        self.label_tuple.pack()

        self.entry_tuple = tk.Entry(self.root, width=50)
        self.entry_tuple.pack()

        self.count_button = tk.Button(self.root, text="Count Occurrences", command=self.count_occurrences)
        self.count_button.pack()

        self.result_label = tk.Label(self.root, text="")
        self.result_label.pack()

    def count_occurrences(self):
        '''
        Count occurrences of list elements in the tuple and display the result.
        '''
        try:
            lst = self.entry_list.get().split(',')
            tpl = tuple(self.entry_tuple.get().split(','))
            result = count_elements_in_tuple(lst, tpl)
            result_str = "\n".join([f"{k}: {v}" for k, v in result.items()])
            self.result_label.config(text=result_str)
        except Exception as e:
            messagebox.showerror("Error", f"An error occurred: {e}")

if __name__ == "__main__":
    root = tk.Tk()
    app = MainApp(root)
    root.mainloop()
```

```python
'''

This module contains the MainApp class which creates a GUI application using Tkinter to count occurrences of list elements in a tuple.

'''

import tkinter as tk
from tkinter import messagebox

def count_elements_in_tuple(lst, tpl):
    '''
    Count occurrences of each element in the list within the tuple.

    Parameters:
    lst (list): The list of elements to count.
    tpl (tuple): The tuple in which to count occurrences.

    Returns:
    dict: A dictionary with list elements as keys and their counts in the tuple as values.
    '''
    count_dict = {}
    for element in lst:
        count_dict[element] = tpl.count(element)
    return count_dict

class MainApp:
    '''
    Main application class to create and manage the GUI.
    '''
    def __init__(self, root):
        self.root = root
        self.root.title("List Element Counter in Tuple")
        self.create_widgets()

    def create_widgets(self):
        '''
        Create and place the widgets in the application window.
        '''
        self.label_list = tk.Label(self.root, text="Enter list (comma separated):")
        self.label_list.pack()

        self.entry_list = tk.Entry(self.root, width=50)
        self.entry_list.pack()

        self.label_tuple = tk.Label(self.root, text="Enter tuple (comma separated):")
        self.label_tuple.pack()

        self.entry_tuple = tk.Entry(self.root, width=50)
        self.entry_tuple.pack()

        self.count_button = tk.Button(self.root, text="Count Occurrences", command=self.count_occurrences)
        self.count_button.pack()

        self.result_label = tk.Label(self.root, text="")
        self.result_label.pack()

    def count_occurrences(self):
        '''
        Count occurrences of list elements in the tuple and display the result.
        '''
        try:
            lst = self.entry_list.get().split(',')
            tpl = tuple(self.entry_tuple.get().split(','))
            result = count_elements_in_tuple(lst, tpl)
            result_str = "\n".join([f"{k}: {v}" for k, v in result.items()])
            self.result_label.config(text=result_str)
        except Exception as e:
            messagebox.showerror("Error", f"An error occurred: {e}")

if __name__ == "__main__":
    root = tk.Tk()
    app = MainApp(root)
    root.mainloop()

'''

'''
import tkinter as tk
from tkinter import messagebox
def count_elements_in_tuple(lst, tpl):
    '''
    Count occurrences of each element in the list within the tuple.
    Parameters:
    lst (list): The list of elements to count.
    tpl (tuple): The tuple in which to count occurrences.
    Returns:
    dict: A dictionary with list elements as keys and their counts in the tuple as values.
    '''
    count_dict = {}
    for element in lst:
        count_dict[element] = tpl.count(element)
    return count_dict
class MainApp:
    '''
    Main application class to create and manage the GUI.
    '''
    def __init__(self, root):
        self.root = root
        self.root.title("List Element Counter in Tuple")
        self.create_widgets()
    def create_widgets(self):
        '''
        Create and place the widgets in the application window.
        '''
        self.label_list = tk.Label(self.root, text="Enter list (comma separated):")
        self.label_list.pack()
        self.entry_list = tk.Entry(self.root, width=50)
        self.entry_list.pack()
        self.label_tuple = tk.Label(self.root, text="Enter tuple (comma separated):")
        self.label_tuple.pack()
        self.entry_tuple = tk.Entry(self.root, width=50)
        self.entry_tuple.pack()
        self.count_button = tk.Button(self.root, text="Count Occurrences", command=self.count_occurrences)
        self.count_button.pack()
        self.result_label = tk.Label(self.root, text="")
        self.result_label.pack()
    def count_occurrences(self):
        '''
        Count occurrences of list elements in the tuple and display the result.
        '''
        try:
            lst = self.entry_list.get().split(',')
            tpl = tuple(self.entry_tuple.get().split(','))
            result = count_elements_in_tuple(lst, tpl)
            result_str = "\n".join([f"{k}: {v}" for k, v in result.items()])
            self.result_label.config(text=result_str)
        except Exception as e:
            messagebox.showerror("Error", f"An error occurred: {e}")
if __name__ == "__main__":
    root = tk.Tk()
    app = MainApp(root)
    root.mainloop()

'''
This module contains the MainApp class which creates a GUI application using Tkinter to count occurrences of list elements in a tuple.
'''
import tkinter as tk
from tkinter import messagebox
def count_elements_in_tuple(lst, tpl):
    '''
    Count occurrences of each element in the list within the tuple.
    Parameters:
    lst (list): The list of elements to count.
    tpl (tuple): The tuple in which to count occurrences.
    Returns:
    dict: A dictionary with list elements as keys and their counts in the tuple as values.
    '''
    count_dict = {}
    for element in lst:
        count_dict[element] = tpl.count(element)
    return count_dict
class MainApp:
    '''
    Main application class to create and manage the GUI.
    '''
    def __init__(self, root):
        '''
        Initialize the MainApp class and set up the GUI window.
        Parameters:
        root (Tk): The root window of the Tkinter application.
        '''
        self.root = root
        self.root.title("List Element Counter in Tuple")
        self.create_widgets()
    def create_widgets(self):
        '''
        Create and place the widgets in the application window.
        '''
        self.label_list = tk.Label(self.root, text="Enter list (comma separated):")
        self.label_list.pack()
        self.entry_list = tk.Entry(self.root, width=50)
        self.entry_list.pack()
        self.label_tuple = tk.Label(self.root, text="Enter tuple (comma separated):")
        self.label_tuple.pack()
        self.entry_tuple = tk.Entry(self.root, width=50)
        self.entry_tuple.pack()
        self.count_button = tk.Button(self.root, text="Count Occurrences", command=self.count_occurrences)
        self.count_button.pack()
        self.result_label = tk.Label(self.root, text="")
        self.result_label.pack()
    def count_occurrences(self):
        '''
        Count occurrences of list elements in the tuple and display the result.
        '''
        try:
            lst = [item.strip() for item in self.entry_list.get().split(',')]
            tpl = tuple(item.strip() for item in self.entry_tuple.get().split(','))
            result = count_elements_in_tuple(lst, tpl)
            result_str = "\n".join([f"{k}: {v}" for k, v in result.items()])
            self.result_label.config(text=result_str)
        except Exception as e:
            messagebox.showerror("Error", f"An error occurred: {e}")
if __name__ == "__main__":
    root = tk.Tk()
    app = MainApp(root)
    root.mainloop()

'''
This module contains the MainApp class which creates a GUI application using Tkinter to count occurrences of list elements in a tuple.
'''
import tkinter as tk
from tkinter import messagebox
def count_elements_in_tuple(lst, tpl):
    '''
    Count occurrences of each element in the list within the tuple.
    Parameters:
    lst (list): The list of elements to count.
    tpl (tuple): The tuple in which to count occurrences.
    Returns:
    dict: A dictionary with list elements as keys and their counts in the tuple as values.
    '''
    count_dict = {}
    for element in lst:
        count_dict[element] = tpl.count(element)
    return count_dict
class MainApp:
    '''
    Main application class to create and manage the GUI.
    '''
    def __init__(self, root):
        '''
        Initialize the MainApp class and set up the GUI window.
        Parameters:
        root (Tk): The root window of the Tkinter application.
        '''
        self.root = root
        self.root.title("List Element Counter in Tuple")
        self.create_widgets()
    def create_widgets(self):
        '''
        Create and place the widgets in the application window.
        '''
        self.label_list = tk.Label(self.root, text="Enter list (comma separated):")
        self.label_list.pack()
        self.entry_list = tk.Entry(self.root, width=50)
        self.entry_list.pack()
        self.label_tuple = tk.Label(self.root, text="Enter tuple (comma separated):")
        self.label_tuple.pack()
        self.entry_tuple = tk.Entry(self.root, width=50)
        self.entry_tuple.pack()
        self.count_button = tk.Button(self.root, text="Count Occurrences", command=self.count_occurrences)
        self.count_button.pack()
        self.result_label = tk.Label(self.root, text="")
        self.result_label.pack()
    def count_occurrences(self):
        '''
        Count occurrences of list elements in the tuple and display the result.
        '''
        try:
            lst = [item.strip() for item in self.entry_list.get().split(',')]
            tpl = tuple(item.strip() for item in self.entry_tuple.get().split(','))
            result = count_elements_in_tuple(lst, tpl)
            result_str = "\n".join([f"{k}: {v}" for k, v in result.items()])
            self.result_label.config(text=result_str)
        except Exception as e:
            messagebox.showerror("Error", f"An error occurred: {e}")
if __name__ == "__main__":
    root = tk.Tk()
    app = MainApp(root)
    root.mainloop()
