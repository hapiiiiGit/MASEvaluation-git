'''

Utility functions for calculating the minimum product from a list of tuples.

'''

def find_min_product(tuple_list):
    """
    Calculate the minimum product from the pairs of tuples within a given list.
    
    :param tuple_list: List of tuples, where each tuple contains two integers.
    :return: The minimum product of the pairs of tuples.
    """
    if not tuple_list:
        return None
    
    min_product = float('inf')
    for tup in tuple_list:
        if len(tup) != 2:
            raise ValueError("Each tuple must contain exactly two integers.")
        product = tup[0] * tup[1]
        if product < min_product:
            min_product = product
    
    return min_product
```

Main application file for the GUI to find the minimum product from pairs of tuples.

import tkinter as tk
from tkinter import messagebox
from utils import find_min_product

class MainApp:
    def __init__(self, root):
        self.root = root
        self.root.title("Minimum Product Finder")
        
        # Input label and entry
        self.label = tk.Label(root, text="Enter list of tuples (e.g., [(1, 2), (3, 4)]):")
        self.label.pack(pady=10)
        
        self.entry = tk.Entry(root, width=50)
        self.entry.pack(pady=10)
        
        # Calculate button
        self.calculate_button = tk.Button(root, text="Calculate Minimum Product", command=self.on_calculate)
        self.calculate_button.pack(pady=20)
        
        # Result label
        self.result_label = tk.Label(root, text="")
        self.result_label.pack(pady=10)
    
    def on_calculate(self):
        try:
            # Get input from entry
            input_text = self.entry.get()
            # Convert string to list of tuples
            tuple_list = eval(input_text)
            # Calculate minimum product
            min_product = find_min_product(tuple_list)
            # Display result
            self.result_label.config(text=f"Minimum Product: {min_product}")
        except Exception as e:
            messagebox.showerror("Error", f"Invalid input: {e}")

if __name__ == "__main__":

def find_min_product(tuple_list):
    """
    Calculate the minimum product from the pairs of tuples within a given list.
    
    :param tuple_list: List of tuples, where each tuple contains two integers.
    :return: The minimum product of the pairs of tuples.
    """
    if not tuple_list:
        return None
    
    min_product = float('inf')
    for tup in tuple_list:
        if len(tup) != 2:
            raise ValueError("Each tuple must contain exactly two integers.")
        product = tup[0] * tup[1]
        if product < min_product:
            min_product = product

'''
Utility functions for calculating the minimum product from a list of tuples.
'''
def find_min_product(tuple_list):
    """
    Calculate the minimum product from the pairs of tuples within a given list.
    :param tuple_list: List of tuples, where each tuple contains two integers.
    :return: The minimum product of the pairs of tuples.
    """
    if not tuple_list:
        return None
    min_product = float('inf')
    for tup in tuple_list:
        if len(tup) != 2:
            raise ValueError("Each tuple must contain exactly two integers.")
        product = tup[0] * tup[1]
        if product < min_product:
            min_product = product
    return min_product

'''
Utility functions for calculating the minimum product from a list of tuples.

def find_min_product(tuple_list):
    """
    Calculate the minimum product from the pairs of tuples within a given list.
    :param tuple_list: List of tuples, where each tuple contains two integers.
    :return: The minimum product of the pairs of tuples.
    """
    if not tuple_list:
        return None
    min_product = float('inf')
    for tup in tuple_list:
        if len(tup) != 2:
            raise ValueError("Each tuple must contain exactly two integers.")
        product = tup[0] * tup[1]
        if product < min_product:
            min_product = product
    return min_product

'''
Utility functions for calculating the minimum product from a list of tuples.
'''
def find_min_product(tuple_list):
    """
    Calculate the minimum product from the pairs of tuples within a given list.
    :param tuple_list: List of tuples, where each tuple contains two integers.
    :return: The minimum product of the pairs of tuples.
    """
    if not tuple_list:
        return None
    min_product = float('inf')
    for tup in tuple_list:
        if len(tup) != 2:
            raise ValueError("Each tuple must contain exactly two integers.")
        product = tup[0] * tup[1]
        if product < min_product:
            min_product = product
    return min_product
```

'''
Utility functions for calculating the minimum product from a list of tuples.
'''
def find_min_product(tuple_list):
    """
    Calculate the minimum product from the pairs of tuples within a given list.
    :param tuple_list: List of tuples, where each tuple contains two integers.
    :return: The minimum product of the pairs of tuples.
    """
    if not tuple_list:
        return None
    min_product = float('inf')
    for tup in tuple_list:
        if len(tup) != 2:
            raise ValueError("Each tuple must contain exactly two integers.")
        product = tup[0] * tup[1]
        if product < min_product:
            min_product = product
    return min_product

'''
Main application file for the GUI to find the minimum product from pairs of tuples.

import tkinter as tk
from tkinter import messagebox
from utils import find_min_product
class MainApp:
    def __init__(self, root):
        self.root = root
        self.root.title("Minimum Product Finder")
        # Input label and entry
        self.label = tk.Label(root, text="Enter list of tuples (e.g., [(1, 2), (3, 4)]):")
        self.label.pack(pady=10)
        self.entry = tk.Entry(root, width=50)
        self.entry.pack(pady=10)
        # Calculate button
        self.calculate_button = tk.Button(root, text="Calculate Minimum Product", command=self.on_calculate)
        self.calculate_button.pack(pady=20)
        # Result label
        self.result_label = tk.Label(root, text="")
        self.result_label.pack(pady=10)
    def on_calculate(self):
        try:
            # Get input from entry
            input_text = self.entry.get()
            # Convert string to list of tuples
            tuple_list = eval(input_text)
            # Calculate minimum product
            min_product = find_min_product(tuple_list)
            # Display result
            self.result_label.config(text=f"Minimum Product: {min_product}")
        except Exception as e:
            messagebox.showerror("Error", f"Invalid input: {e}")
if __name__ == "__main__":
    root = tk.Tk()
    app = MainApp(root)
    root.mainloop()
