'''

This is the main file for the Python application that provides a GUI for splitting a list into sublists for every nth element.
The application uses tkinter for the graphical user interface.

'''

import tkinter as tk
from tkinter import messagebox


def split_list_by_n(input_list, n):
    '''
    Splits a list into sublists for every nth element.

    Args:
        input_list (list): The list to be split.
        n (int): The interval at which to split the list.

    Returns:
        list: A list of sublists split at every nth element.
    '''
    if n <= 0:
        raise ValueError("n must be a positive integer.")
    return [input_list[i:i + n] for i in range(0, len(input_list), n)]


class MainApp:
    '''
    The main class for the GUI application.
    '''
    def __init__(self, root):
        '''
        Initializes the GUI components.

        Args:
            root (tk.Tk): The root window of the tkinter application.
        '''
        self.root = root
        self.root.title("List Splitter")

        # Input for the list

    '''

    Args:

import tkinter as tk
from tkinter import messagebox

def split_list_by_n(input_list, n):
    '''
    Splits a list into sublists for every nth element.
    Args:
        input_list (list): The list to be split.
        n (int): The interval at which to split the list.
    Returns:
        list: A list of sublists split at every nth element.
    '''
    if n <= 0:
        raise ValueError("n must be a positive integer.")
    return [input_list[i:i + n] for i in range(0, len(input_list), n)]

'''
This is the main file for the Python application that provides a GUI for splitting a list into sublists for every nth element.
The application uses tkinter for the graphical user interface.
'''
import tkinter as tk
from tkinter import messagebox
def split_list_by_n(input_list, n):
    '''
    Splits a list into sublists for every nth element.
    Args:
        input_list (list): The list to be split.
        n (int): The interval at which to split the list.
    Returns:
        list: A list of sublists split at every nth element.
    '''
    if n <= 0:
        raise ValueError("n must be a positive integer.")
    return [input_list[i:i + n] for i in range(0, len(input_list), n)]
class MainApp:
    '''
    The main class for the GUI application.
    '''
    def __init__(self, root):
        '''
        Initializes the GUI components.
        Args:
            root (tk.Tk): The root window of the tkinter application.
        '''
        self.root = root
        self.root.title("List Splitter")
        self.root.geometry("500x400")  # Set a fixed window size for better layout
        # Input for the list
        self.list_label = tk.Label(root, text="Enter a list (comma-separated):")
        self.list_label.pack(pady=5)
        self.list_entry = tk.Entry(root, width=50)
        self.list_entry.pack(pady=5)
        # Input for the integer n
        self.n_label = tk.Label(root, text="Enter an integer n:")
        self.n_label.pack(pady=5)
        self.n_entry = tk.Entry(root, width=20)
        self.n_entry.pack(pady=5)
        # Button to trigger the split operation
        self.split_button = tk.Button(root, text="Split List", command=self.split_list)
        self.split_button.pack(pady=10)
        # Output area for the result
        self.result_label = tk.Label(root, text="Result:")
        self.result_label.pack(pady=5)
        self.result_text = tk.Text(root, height=10, width=50, state=tk.DISABLED)
        self.result_text.pack(pady=5)
    def split_list(self):
        '''
        Handles the split operation and updates the result in the GUI.
        '''
        try:
            # Get inputs
            raw_list = self.list_entry.get()
            raw_n = self.n_entry.get()
            # Validate and process inputs
            input_list = self.validate_list(raw_list)
            n = self.validate_integer(raw_n)
            # Perform the split operation
            result = split_list_by_n(input_list, n)
            # Display the result
            self.display_result(result)
        except Exception as e:
            messagebox.showerror("Error", str(e))
    def validate_list(self, raw_list):
        '''
        Validates and converts the input string into a list.
        Args:
            raw_list (str): The raw input string representing a list.
        Returns:
            list: The validated and converted list.
        '''
        try:
            # Convert the string to a list
            input_list = [item.strip() for item in raw_list.split(",") if item.strip()]
            if not input_list:
                raise ValueError("The list cannot be empty.")
            return input_list
        except Exception:
            raise ValueError("Invalid list format. Please enter a comma-separated list.")
    def validate_integer(self, raw_n):
        '''
        Validates and converts the input string into an integer.
        Args:
            raw_n (str): The raw input string representing an integer.
        Returns:
            int: The validated and converted integer.
        '''
        try:
            n = int(raw_n)
            if n <= 0:
                raise ValueError
            return n
        except ValueError:
            raise ValueError("Invalid integer. Please enter a positive integer.")
    def display_result(self, result):
        '''
        Displays the result in the GUI.
        Args:
            result (list): The result to be displayed.
        '''
        # Clear the text area
        self.result_text.config(state=tk.NORMAL)
        self.result_text.delete(1.0, tk.END)
        # Insert the result
        self.result_text.insert(tk.END, str(result))
        # Disable editing
        self.result_text.config(state=tk.DISABLED)
if __name__ == "__main__":
    # Create the main window
    root = tk.Tk()
    app = MainApp(root)
    root.mainloop()
