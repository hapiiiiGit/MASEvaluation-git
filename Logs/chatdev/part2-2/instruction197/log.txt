'''

This file contains the main application code for a Python program that swaps the first and last elements of a list. It uses tkinter for the graphical user interface.

'''

import tkinter as tk
from tkinter import messagebox


def swap_elements(lst):
    '''
    This function takes a list as input and interchanges its first and last elements.
    If the list has fewer than two elements, it returns the list unchanged.
    '''
    if len(lst) < 2:
        return lst
    lst[0], lst[-1] = lst[-1], lst[0]
    return lst


class SwapApp:
    '''
    This class represents the GUI application for swapping the first and last elements of a list.
    '''
    def __init__(self, root):
        self.root = root
        self.root.title("Swap First and Last Elements")
        self.root.geometry("400x300")
        
        # Input Label
        self.label = tk.Label(root, text="Enter a list of elements (comma-separated):")
        self.label.pack(pady=10)
        
        # Input Entry
        self.entry = tk.Entry(root, width=50)
        self.entry.pack(pady=10)
        
        # Swap Button
        self.swap_button = tk.Button(root, text="Swap Elements", command=self.swap_action)
        self.swap_button.pack(pady=10)
        
        # Output Label
        self.output_label = tk.Label(root, text="Result:", font=("Arial", 12))
        self.output_label.pack(pady=10)
        
        # Output Text
        self.output_text = tk.Label(root, text="", font=("Arial", 12), fg="blue")
        self.output_text.pack(pady=10)
    
    def swap_action(self):
        '''
        This method handles the swap button click event.
        It reads the input, processes the list, and displays the result.
        '''
        input_text = self.entry.get()
        try:
            # Convert input text to a list
            input_list = [item.strip() for item in input_text.split(",")]
            
            # Swap elements
            swapped_list = swap_elements(input_list)
            
            # Display result
            self.output_text.config(text=", ".join(swapped_list))
        except Exception as e:
            messagebox.showerror("Error", f"An error occurred: {str(e)}")


if __name__ == "__main__":
    root = tk.Tk()
    app = SwapApp(root)
    root.mainloop()

'''

This file contains the main application code for a Python program that swaps the first and last elements of a list. It uses tkinter for the graphical user interface.

'''

import tkinter as tk
from tkinter import messagebox


def swap_elements(lst):
    '''
    This function takes a list as input and interchanges its first and last elements.
    If the list has fewer than two elements, it returns the list unchanged.
    '''
    if len(lst) < 2:
        return lst
    lst[0], lst[-1] = lst[-1], lst[0]
    return lst


class SwapApp:
    '''
    This class represents the GUI application for swapping the first and last elements of a list.
    '''
    def __init__(self, root):
        self.root = root
        self.root.title("Swap First and Last Elements")
        self.root.geometry("400x300")
        
        # Input Label
        self.label = tk.Label(root, text="Enter a list of elements (comma-separated):")
        self.label.pack(pady=10)
        
        # Input Entry
        self.entry = tk.Entry(root, width=50)
        self.entry.pack(pady=10)
        
        # Swap Button
        self.swap_button = tk.Button(root, text="Swap Elements", command=self.swap_action)
        self.swap_button.pack(pady=10)
        
        # Output Label
        self.output_label = tk.Label(root, text="Result:", font=("Arial", 12))
        self.output_label.pack(pady=10)
        
        # Output Text
        self.output_text = tk.Label(root, text="", font=("Arial", 12), fg="blue")
        self.output_text.pack(pady=10)
    
    def swap_action(self):
        '''
        This method handles the swap button click event.
        It reads the input, processes the list, and displays the result.
        '''
        input_text = self.entry.get()
        try:
            # Convert input text to a list
            input_list = [item.strip() for item in input_text.split(",")]
            
            # Swap elements
            swapped_list = swap_elements(input_list)
            
            # Display result
            self.output_text.config(text=", ".join(swapped_list))
        except Exception as e:
            messagebox.showerror("Error", f"An error occurred: {str(e)}")


if __name__ == "__main__":
    root = tk.Tk()
    app = SwapApp(root)
    root.mainloop()

'''
This file contains the main application code for a Python program that swaps the first and last elements of a list. It uses tkinter for the graphical user interface.
'''
import tkinter as tk
from tkinter import messagebox
def swap_elements(lst):
    '''
    This function takes a list as input and interchanges its first and last elements.
    If the list has fewer than two elements, it returns the list unchanged.
    '''
    if len(lst) < 2:
        return lst
    lst[0], lst[-1] = lst[-1], lst[0]
    return lst
class SwapApp:
    '''
    This class represents the GUI application for swapping the first and last elements of a list.
    '''
    def __init__(self, root):
        self.root = root
        self.root.title("Swap First and Last Elements")
        self.root.geometry("400x300")
        self.root.resizable(False, False)
        # Input Label
        self.label = tk.Label(root, text="Enter a list of elements (comma-separated):", font=("Arial", 10))
        self.label.pack(pady=10)
        # Input Entry
        self.entry = tk.Entry(root, width=50, font=("Arial", 10))
        self.entry.pack(pady=10)
        # Swap Button
        self.swap_button = tk.Button(root, text="Swap Elements", command=self.swap_action, font=("Arial", 10))
        self.swap_button.pack(pady=10)
        # Output Label
        self.output_label = tk.Label(root, text="Result:", font=("Arial", 12))
        self.output_label.pack(pady=10)
        # Output Text
        self.output_text = tk.Label(root, text="", font=("Arial", 12), fg="blue")
        self.output_text.pack(pady=10)
    def swap_action(self):
        '''
        This method handles the swap button click event.
        It reads the input, processes the list, and displays the result.
        '''
        input_text = self.entry.get()
        try:
            # Validate input
            if not input_text.strip():
                raise ValueError("Input cannot be empty. Please enter a valid list.")
            # Convert input text to a list
            input_list = [item.strip() for item in input_text.split(",") if item.strip()]
            if len(input_list) < 2:
                raise ValueError("The list must contain at least two elements to swap.")
            # Swap elements
            swapped_list = swap_elements(input_list)
            # Display result
            self.output_text.config(text=", ".join(swapped_list))
        except ValueError as ve:
            messagebox.showerror("Input Error", str(ve))
        except Exception as e:
            messagebox.showerror("Error", f"An unexpected error occurred: {str(e)}")
if __name__ == "__main__":
    root = tk.Tk()
    app = SwapApp(root)
    root.mainloop()

'''
This file contains the main application code for a Python program that swaps the first and last elements of a list. It uses tkinter for the graphical user interface.
'''
import tkinter as tk
from tkinter import messagebox
def swap_elements(lst):
    '''
    This function takes a list as input and interchanges its first and last elements.
    If the list has fewer than two elements, it returns the list unchanged.
    '''
    if len(lst) < 2:
        return lst
    lst[0], lst[-1] = lst[-1], lst[0]
    return lst
class SwapApp:
    '''
    This class represents the GUI application for swapping the first and last elements of a list.
    '''
    def __init__(self, root):
        self.root = root
        self.root.title("Swap First and Last Elements")
        self.root.geometry("400x300")
        self.root.resizable(False, False)
        # Input Label
        self.label = tk.Label(root, text="Enter a list of elements (comma-separated):", font=("Arial", 10))
        self.label.pack(pady=10)
        # Input Entry
        self.entry = tk.Entry(root, width=50, font=("Arial", 10))
        self.entry.pack(pady=10)
        # Swap Button
        self.swap_button = tk.Button(root, text="Swap Elements", command=self.swap_action, font=("Arial", 10))
        self.swap_button.pack(pady=10)
        # Output Label
        self.output_label = tk.Label(root, text="Result:", font=("Arial", 12))
        self.output_label.pack(pady=10)
        # Output Text
        self.output_text = tk.Label(root, text="", font=("Arial", 12), fg="blue")
        self.output_text.pack(pady=10)
    def swap_action(self):
        '''
        This method handles the swap button click event.
        It reads the input, processes the list, and displays the result.
        '''
        input_text = self.entry.get()
        try:
            # Validate input
            if not input_text.strip():
                raise ValueError("Input cannot be empty. Please enter a valid list.")
            # Convert input text to a list
            input_list = [item.strip() for item in input_text.split(",") if item.strip()]
            if len(input_list) < 2:
                raise ValueError("The list must contain at least two elements to swap.")
            # Swap elements
            swapped_list = swap_elements(input_list)
            # Display result
            self.output_text.config(text=", ".join(swapped_list))
        except ValueError as ve:
            messagebox.showerror("Input Error", str(ve))
        except Exception as e:
            messagebox.showerror("Error", f"An unexpected error occurred: {str(e)}")

    root = tk.Tk()
