'''

This file contains the main application logic and GUI implementation for calculating the sum of the product of consecutive binomial coefficients.

'''

import tkinter as tk
from tkinter import messagebox
from math import comb

def calculate_binomial_sum(n):
    '''
    Calculate the sum of the product of consecutive binomial coefficients for a given n.
    
    Parameters:
        n (int): The number of terms in the binomial expansion.
        
    Returns:
        int: The sum of the product of consecutive binomial coefficients.
    '''
    total_sum = 0
    for k in range(n):
        total_sum += comb(n, k) * comb(n, k + 1)
    return total_sum

class MainApp:
    '''
    The main application class for the GUI.
    '''

        '''
        Initialize the GUI components.

        self.root = root
        self.root.title("Binomial Coefficients Calculator")
        
        # Input label and entry
        self.label = tk.Label(root, text="Enter n (number of terms):")
        self.label.pack(pady=5)

        self.entry.pack(pady=5)
        
        # Calculate button
        self.calculate_button = tk.Button(root, text="Calculate", command=self.calculate_and_display)
        self.calculate_button.pack(pady=10)
        
        # Result label
        self.result_label = tk.Label(root, text="", fg="blue")
        self.result_label.pack(pady=10)

        '''

import tkinter as tk

    '''
    Calculate the sum of the product of consecutive binomial coefficients for a given n.

from math import comb

import tkinter as tk
from tkinter import messagebox
from math import comb

def calculate_binomial_sum(n):

    Returns:

'''
This file contains the main application logic and GUI implementation for calculating the sum of the product of consecutive binomial coefficients.
'''
import tkinter as tk
from tkinter import messagebox
from math import comb
def calculate_binomial_sum(n):
    '''
    Calculate the sum of the product of consecutive binomial coefficients for a given n.
    Parameters:
        n (int): The number of terms in the binomial expansion.
    Returns:
        int: The sum of the product of consecutive binomial coefficients.
    '''
    total_sum = 0
    for k in range(n):
        total_sum += comb(n, k) * comb(n, k + 1)
    return total_sum
class MainApp:
    '''
    The main application class for the GUI.
    '''
    def __init__(self, root):
        '''
        Initialize the GUI components.
        Parameters:
            root (tk.Tk): The root window for the GUI.
        '''
        self.root = root
        self.root.title("Binomial Coefficients Calculator")
        # Input label and entry
        self.label = tk.Label(root, text="Enter n (number of terms):")
        self.label.pack(pady=5)
        self.entry = tk.Entry(root)
        self.entry.pack(pady=5)
        # Calculate button
        self.calculate_button = tk.Button(root, text="Calculate", command=self.calculate_and_display)
        self.calculate_button.pack(pady=10)
        # Result label
        self.result_label = tk.Label(root, text="", fg="blue")
        self.result_label.pack(pady=10)
    def calculate_and_display(self):
        '''
        Handle user input, call the calculation function, and display the result.
        '''
        try:
            n = int(self.entry.get())
            if n < 0:
                raise ValueError("n must be a non-negative integer.")
            result = calculate_binomial_sum(n)
            self.result_label.config(text=f"Result: {result}")
        except ValueError as e:
            messagebox.showerror("Invalid Input", str(e))
if __name__ == "__main__":
    root = tk.Tk()
    app = MainApp(root)
