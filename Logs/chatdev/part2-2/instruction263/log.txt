'''

This file contains the main application logic, including the function to check if all characters in a string are the same.

'''

import tkinter as tk
from tkinter import messagebox

def check_all_characters_same(input_string):
    """
    Check if all characters in the input string are the same.

    :param input_string: The string to check.
    :return: True if all characters are the same, False otherwise.
    """
    if not input_string:
        return False
    first_char = input_string[0]
    for char in input_string:
        if char != first_char:
            return False
    return True

class MainApplication(tk.Tk):
    def __init__(self):
        super().__init__()
        self.title("Character Check Application")
        self.geometry("300x150")

        self.label = tk.Label(self, text="Enter a string:")
        self.label.pack(pady=10)

        self.entry = tk.Entry(self)
        self.entry.pack(pady=5)

        self.check_button = tk.Button(self, text="Check", command=self.check_characters)
        self.check_button.pack(pady=10)

    def check_characters(self):
        input_string = self.entry.get()
        if check_all_characters_same(input_string):
            messagebox.showinfo("Result", "All characters are the same.")
        else:
            messagebox.showinfo("Result", "Characters are not the same.")

if __name__ == "__main__":
    app = MainApplication()
    app.mainloop()
```

'''

This file contains the main application logic, including the function to check if all characters in a string are the same.

'''

import tkinter as tk
from tkinter import messagebox

def check_all_characters_same(input_string):
    """
    Check if all characters in the input string are the same.

    :param input_string: The string to check.
    :return: True if all characters are the same, False otherwise.
    """
    if not input_string:
        return False
    first_char = input_string[0]
    for char in input_string:
        if char != first_char:
            return False
    return True

class MainApplication(tk.Tk):
    def __init__(self):
        super().__init__()
        self.title("Character Check Application")
        self.geometry("300x150")

        self.label = tk.Label(self, text="Enter a string:")
        self.label.pack(pady=10)

        self.entry = tk.Entry(self)
        self.entry.pack(pady=5)

        self.check_button = tk.Button(self, text="Check", command=self.check_characters)
        self.check_button.pack(pady=10)

    def check_characters(self):
        input_string = self.entry.get()
        if check_all_characters_same(input_string):
            messagebox.showinfo("Result", "All characters are the same.")
        else:
            messagebox.showinfo("Result", "Characters are not the same.")

if __name__ == "__main__":
    app = MainApplication()
    app.mainloop()

'''
This file contains the main application logic, including the function to check if all characters in a string are the same.

import tkinter as tk
from tkinter import messagebox
def check_all_characters_same(input_string):
    """
    Check if all characters in the input string are the same.
    :param input_string: The string to check.
    :return: True if all characters are the same, False otherwise.
    """
    if not input_string:
        return False
    first_char = input_string[0]
    for char in input_string:
        if char != first_char:
            return False
    return True
class MainApplication(tk.Tk):
    """
    Main application class for the Character Check Application.
    """
    def __init__(self):
        super().__init__()
        self.title("Character Check Application")
        self.geometry("300x150")
        self.label = tk.Label(self, text="Enter a string:")
        self.label.pack(pady=10)
        self.entry = tk.Entry(self)
        self.entry.pack(pady=5)
        self.check_button = tk.Button(self, text="Check", command=self.check_characters)
        self.check_button.pack(pady=10)
    def check_characters(self):
        """
        Retrieve the input string from the entry widget and check if all characters are the same.
        Display the result in a message box.
        """
        input_string = self.entry.get()
        if check_all_characters_same(input_string):
            messagebox.showinfo("Result", "All characters are the same.")
        else:
            messagebox.showinfo("Result", "Characters are not the same.")
if __name__ == "__main__":
    app = MainApplication()
    app.mainloop()

# Character Check Application

The Character Check Application is a simple graphical user interface (GUI) application that allows users to check if all characters in a given string are the same. This application is built using Python and the Tkinter library for the GUI.

## Main Functions

- **Check All Characters Same**: The core function of the application checks whether all characters in the input string are identical. It returns `True` if all characters are the same and `False` otherwise.

- **Graphical User Interface**: The application provides a user-friendly interface where users can input a string and receive immediate feedback on whether all characters in the string are the same.
