from tkinter import Tk, Label, Entry, Button, StringVar
from tuple_checker import check_element
class TupleCheckerApp:
    '''
    A GUI application to check whether an element exists in a tuple.
    '''
    def __init__(self, root):
        '''
        Initializes the GUI components.
        :param root: The root window of the application.
        '''
        self.root = root
        self.root.title("Tuple Checker")
        # Input for tuple
        self.tuple_label = Label(root, text="Enter a tuple (comma-separated):")
        self.tuple_label.grid(row=0, column=0, padx=10, pady=10)
        self.tuple_entry = Entry(root, width=30)
        self.tuple_entry.grid(row=0, column=1, padx=10, pady=10)
        # Input for element
        self.element_label = Label(root, text="Enter an element to check:")
        self.element_label.grid(row=1, column=0, padx=10, pady=10)
        self.element_entry = Entry(root, width=30)
        self.element_entry.grid(row=1, column=1, padx=10, pady=10)
        # Result display
        self.result_var = StringVar()
        self.result_label = Label(root, textvariable=self.result_var, fg="blue")
        self.result_label.grid(row=3, column=0, columnspan=2, padx=10, pady=10)
        # Check button
        self.check_button = Button(root, text="Check", command=self.check_element_in_tuple)
        self.check_button.grid(row=2, column=0, columnspan=2, pady=10)
    def check_element_in_tuple(self):
        '''
        Checks if the element exists in the tuple and updates the result label.
        '''
        tuple_input = self.tuple_entry.get()
        element_input = self.element_entry.get()
        try:
            # Convert the tuple input from string to a tuple
            tuple_data = tuple(map(str.strip, tuple_input.split(',')))
            # Check if the element exists in the tuple
            exists = check_element(tuple_data, element_input)
            if exists:
                self.result_var.set(f"The element '{element_input}' exists in the tuple.")
            else:
                self.result_var.set(f"The element '{element_input}' does not exist in the tuple.")
        except Exception as e:
            self.result_var.set(f"Error: {str(e)}")
if __name__ == "__main__":
    root = Tk()
    app = TupleCheckerApp(root)
    root.mainloop()

    '''

        Initializes the GUI components.

        self.tuple_entry = Entry(root, width=30)
        self.tuple_entry.grid(row=0, column=1, padx=10, pady=10)
        # Input for element
        self.element_label = Label(root, text="Enter an element to check:")
        self.element_label.grid(row=1, column=0, padx=10, pady=10)
        self.element_entry = Entry(root, width=30)
        self.element_entry.grid(row=1, column=1, padx=10, pady=10)
        # Result display
        self.result_var = StringVar()

        self.check_button.grid(row=2, column=0, columnspan=2, pady=10)
    def check_element_in_tuple(self):
        '''
        Checks if the element exists in the tuple and updates the result label.
        '''
        tuple_input = self.tuple_entry.get()
        element_input = self.element_entry.get()
        try:
            # Convert the tuple input from string to a tuple
            tuple_data = tuple(map(str.strip, tuple_input.split(',')))
            # Check if the element exists in the tuple
            exists = check_element(tuple_data, element_input)
            if exists:
                self.result_var.set(f"The element '{element_input}' exists in the tuple.")
            else:
                self.result_var.set(f"The element '{element_input}' does not exist in the tuple.")
        except Exception as e:
            self.result_var.set(f"Error: {str(e)}")
if __name__ == "__main__":
    root = Tk()
    app = TupleCheckerApp(root)
    root.mainloop()

'''
This file contains the main entry point for the Tuple Checker application. It initializes the GUI and handles user interactions.
'''
from tkinter import Tk, Label, Entry, Button, StringVar
from tuple_checker import check_element
class TupleCheckerApp:
    '''
    A GUI application to check whether an element exists in a tuple.
    '''
    def __init__(self, root):
        '''
        Initializes the GUI components.
        :param root: The root window of the application.
        '''
        self.root = root
        self.root.title("Tuple Checker")
        # Input for tuple
        self.tuple_label = Label(root, text="Enter a tuple (comma-separated):")
        self.tuple_label.grid(row=0, column=0, padx=10, pady=10)
        self.tuple_entry = Entry(root, width=30)
        self.tuple_entry.grid(row=0, column=1, padx=10, pady=10)
        # Input for element
        self.element_label = Label(root, text="Enter an element to check:")
        self.element_label.grid(row=1, column=0, padx=10, pady=10)
        self.element_entry = Entry(root, width=30)
        self.element_entry.grid(row=1, column=1, padx=10, pady=10)
        # Result display
        self.result_var = StringVar()
        self.result_label = Label(root, textvariable=self.result_var, fg="blue")
        self.result_label.grid(row=3, column=0, columnspan=2, padx=10, pady=10)
        # Check button
        self.check_button = Button(root, text="Check", command=self.check_element_in_tuple)
        self.check_button.grid(row=2, column=0, columnspan=2, pady=10)
    def check_element_in_tuple(self):
        '''
        Checks if the element exists in the tuple and updates the result label.
        '''
        tuple_input = self.tuple_entry.get()
        element_input = self.element_entry.get()
        try:
            # Convert the tuple input from string to a tuple
            tuple_data = tuple(map(str.strip, tuple_input.split(',')))
            # Check if the element exists in the tuple
            exists = check_element(tuple_data, element_input)
            if exists:
                self.result_var.set(f"The element '{element_input}' exists in the tuple.")
            else:
                self.result_var.set(f"The element '{element_input}' does not exist in the tuple.")
        except Exception as e:
            self.result_var.set(f"Error: {str(e)}")
if __name__ == "__main__":
    root = Tk()
    app = TupleCheckerApp(root)
    root.mainloop()

'''

from tkinter import Tk, Label, Entry, Button, StringVar
from tuple_checker import check_element
class TupleCheckerApp:
    '''
    A GUI application to check whether an element exists in a tuple.
    '''
    def __init__(self, root):
        '''
        Initializes the GUI components.
        :param root: The root window of the application.
        '''
        self.root = root
        self.root.title("Tuple Checker")
        # Input for tuple
        self.tuple_label = Label(root, text="Enter a tuple (comma-separated):")
        self.tuple_label.grid(row=0, column=0, padx=10, pady=10)
        self.tuple_entry = Entry(root, width=30)
        self.tuple_entry.grid(row=0, column=1, padx=10, pady=10)
        # Input for element
        self.element_label = Label(root, text="Enter an element to check:")
        self.element_label.grid(row=1, column=0, padx=10, pady=10)
        self.element_entry = Entry(root, width=30)
        self.element_entry.grid(row=1, column=1, padx=10, pady=10)
        # Result display
        self.result_var = StringVar()
        self.result_label = Label(root, textvariable=self.result_var, fg="blue")
        self.result_label.grid(row=3, column=0, columnspan=2, padx=10, pady=10)
        # Check button
        self.check_button = Button(root, text="Check", command=self.check_element_in_tuple)
        self.check_button.grid(row=2, column=0, columnspan=2, pady=10)
    def check_element_in_tuple(self):
        '''
        Checks if the element exists in the tuple and updates the result label.
        '''
        tuple_input = self.tuple_entry.get()
        element_input = self.element_entry.get()
        try:
            # Convert the tuple input from string to a tuple
            tuple_data = tuple(map(str.strip, tuple_input.split(',')))
            # Check if the element exists in the tuple
            exists = check_element(tuple_data, element_input)
            if exists:
                self.result_var.set(f"The element '{element_input}' exists in the tuple.")
            else:
                self.result_var.set(f"The element '{element_input}' does not exist in the tuple.")
        except Exception as e:
            self.result_var.set(f"Error: {str(e)}")
if __name__ == "__main__":
    root = Tk()
    app = TupleCheckerApp(root)
    root.mainloop()
