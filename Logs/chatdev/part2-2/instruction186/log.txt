'''

Provides functionality to process tuples, specifically to remove uneven elements.

'''

def remove_uneven_elements(nested_tuple):
    """
    Removes uneven elements from a nested mixed tuple.
    
    :param nested_tuple: A tuple potentially containing other tuples and elements.
    :return: A new tuple with all uneven elements removed.
    """
    def is_even(x):
        return isinstance(x, int) and x % 2 == 0

    def process_tuple(t):
        result = []
        for item in t:
            if isinstance(item, tuple):
                result.append(process_tuple(item))
            elif is_even(item):
                result.append(item)
        return tuple(result)

    return process_tuple(nested_tuple)
```

'''

import tkinter as tk
from tkinter import messagebox
from tuple_processor import remove_uneven_elements

class TupleApp(tk.Tk):
    def __init__(self):
        super().__init__()
        self.title("Tuple Processor")
        self.geometry("400x300")
        self.create_widgets()

    def create_widgets(self):
        self.input_label = tk.Label(self, text="Enter a nested tuple:")
        self.input_label.pack()

        self.input_entry = tk.Entry(self, width=50)
        self.input_entry.pack()

        self.process_button = tk.Button(self, text="Process", command=self.process_input)
        self.process_button.pack()

        self.result_label = tk.Label(self, text="Result:")
        self.result_label.pack()

        self.result_text = tk.Text(self, height=10, width=50)
        self.result_text.pack()

    def process_input(self):
        user_input = self.input_entry.get()
        try:
            # Safely evaluate the input as a tuple
            nested_tuple = eval(user_input, {"__builtins__": None}, {})
            if not isinstance(nested_tuple, tuple):
                raise ValueError("Input is not a tuple.")
            
            result = remove_uneven_elements(nested_tuple)
            self.result_text.delete(1.0, tk.END)
            self.result_text.insert(tk.END, str(result))
        except Exception as e:
            messagebox.showerror("Error", f"Invalid input: {e}")

```

'''

Provides functionality to process tuples, specifically to remove uneven elements.

'''

def remove_uneven_elements(nested_tuple):
    """
    Removes uneven elements from a nested mixed tuple.
    
    :param nested_tuple: A tuple potentially containing other tuples and elements.
    :return: A new tuple with all uneven elements removed.
    """
    def is_even(x):
        return isinstance(x, int) and x % 2 == 0

    def process_tuple(t):
        result = []
        for item in t:
            if isinstance(item, tuple):
                result.append(process_tuple(item))
            elif is_even(item):
                result.append(item)
        return tuple(result)

    return process_tuple(nested_tuple)

def remove_uneven_elements(nested_tuple):
    """
    Removes uneven elements from a nested mixed tuple.
    :param nested_tuple: A tuple potentially containing other tuples and elements.
    :return: A new tuple with all uneven elements removed.
    """
    def is_even(x):
        return isinstance(x, int) and x % 2 == 0
    def process_tuple(t):
        result = []
        for item in t:
            if isinstance(item, tuple):
                result.append(process_tuple(item))
            elif is_even(item):
                result.append(item)
        return tuple(result)
    return process_tuple(nested_tuple)

def remove_uneven_elements(nested_tuple):
    """
    Removes uneven elements from a nested mixed tuple.
    :param nested_tuple: A tuple potentially containing other tuples and elements.
    :return: A new tuple with all uneven elements removed.
    """
    def is_even(x):
        return isinstance(x, int) and x % 2 == 0
    def process_tuple(t):
        result = []
        for item in t:
            if isinstance(item, tuple):
                result.append(process_tuple(item))
            elif is_even(item):
                result.append(item)
        return tuple(result)
    return process_tuple(nested_tuple)

'''
Provides functionality to process tuples, specifically to remove uneven elements.
'''
def remove_uneven_elements(nested_tuple):
    """
    Removes uneven elements from a nested mixed tuple.
    :param nested_tuple: A tuple potentially containing other tuples and elements.
    :return: A new tuple with all uneven elements removed.
    """
    def is_even(x):
        return isinstance(x, int) and x % 2 == 0
    def process_tuple(t):
        result = []
        for item in t:
            if isinstance(item, tuple):
                result.append(process_tuple(item))
            elif is_even(item):
                result.append(item)
        return tuple(result)
    return process_tuple(nested_tuple)

'''
Provides functionality to process tuples, specifically to remove uneven elements.
'''
def remove_uneven_elements(nested_tuple):
    """
    Removes uneven elements from a nested mixed tuple.
    :param nested_tuple: A tuple potentially containing other tuples and elements.
    :return: A new tuple with all uneven elements removed.
    """
    def is_even(x):
        # Check if the element is an integer and even
        return isinstance(x, int) and x % 2 == 0

    def process_tuple(t):
        # Recursively process the tuple to remove uneven elements
        result = []
        for item in t:
            if isinstance(item, tuple):
                # Process nested tuples
                result.append(process_tuple(item))
            elif is_even(item):
                # Include only even integers
                result.append(item)
        return tuple(result)

    return process_tuple(nested_tuple)

'''
Provides functionality to process tuples, specifically to remove uneven elements.
'''
def remove_uneven_elements(nested_tuple):
    """
    Removes uneven elements from a nested mixed tuple.
    :param nested_tuple: A tuple potentially containing other tuples and elements.
    :return: A new tuple with all uneven elements removed.
    """
    def is_even(x):
        # Check if the element is an integer and even
        return isinstance(x, int) and x % 2 == 0
    def process_tuple(t):
        # Recursively process the tuple to remove uneven elements
        result = []
        for item in t:
            if isinstance(item, tuple):
                # Process nested tuples
                result.append(process_tuple(item))
            elif is_even(item):
                # Include only even integers
                result.append(item)
        return tuple(result)
    return process_tuple(nested_tuple)
