'''

This file contains the main implementation of the Cylinder Calculator application.
It uses tkinter to create a GUI for calculating the lateral surface area of a cylinder.

'''

import tkinter as tk
from tkinter import messagebox
import math


def calculate_lateral_surface_area(radius, height):
    '''
    Calculate the lateral surface area of a cylinder.
    
    Parameters:
        radius (float): The radius of the cylinder.
        height (float): The height of the cylinder.
    
    Returns:
        float: The lateral surface area of the cylinder.
    '''
    return 2 * math.pi * radius * height


class CylinderCalculatorApp:
    '''
    A class to represent the Cylinder Calculator GUI application.
    '''
    
    def __init__(self, root):
        '''
        Initialize the GUI application.
        
        Parameters:
            root (tk.Tk): The root window of the tkinter application.
        '''
        self.root = root
        self.root.title("Cylinder Lateral Surface Area Calculator")
        
        # Create and place widgets
        self.create_widgets()
    
    def create_widgets(self):
        '''
        Create and place the widgets for the GUI.
        '''
        # Title Label
        title_label = tk.Label(self.root, text="Cylinder Lateral Surface Area Calculator", font=("Arial", 16))
        title_label.grid(row=0, column=0, columnspan=2, pady=10)
        
        # Radius Input
        radius_label = tk.Label(self.root, text="Radius (r):", font=("Arial", 12))
        radius_label.grid(row=1, column=0, padx=10, pady=5, sticky="e")
        self.radius_entry = tk.Entry(self.root, font=("Arial", 12))
        self.radius_entry.grid(row=1, column=1, padx=10, pady=5)
        
        # Height Input
        height_label = tk.Label(self.root, text="Height (h):", font=("Arial", 12))
        height_label.grid(row=2, column=0, padx=10, pady=5, sticky="e")
        self.height_entry = tk.Entry(self.root, font=("Arial", 12))
        self.height_entry.grid(row=2, column=1, padx=10, pady=5)
        
        # Calculate Button
        calculate_button = tk.Button(self.root, text="Calculate", font=("Arial", 12), command=self.calculate)
        calculate_button.grid(row=3, column=0, columnspan=2, pady=10)
        
        # Result Label
        self.result_label = tk.Label(self.root, text="", font=("Arial", 12), fg="green")
        self.result_label.grid(row=4, column=0, columnspan=2, pady=10)
    
    def calculate(self):
        '''
        Handle the calculation of the lateral surface area when the button is clicked.
        '''
        try:
            # Get inputs
            radius = float(self.radius_entry.get())
            height = float(self.height_entry.get())

'''

This file contains the main implementation of the Cylinder Calculator application.
It uses tkinter to create a GUI for calculating the lateral surface area of a cylinder.

'''

import tkinter as tk
from tkinter import messagebox
import math


def calculate_lateral_surface_area(radius, height):
    '''
    Calculate the lateral surface area of a cylinder.
    
    Parameters:
        radius (float): The radius of the cylinder.
        height (float): The height of the cylinder.
    
    Returns:
        float: The lateral surface area of the cylinder.
    '''
    return 2 * math.pi * radius * height


class CylinderCalculatorApp:
    '''
    A class to represent the Cylinder Calculator GUI application.
    '''
    
    def __init__(self, root):
        '''
        Initialize the GUI application.
        
        Parameters:
            root (tk.Tk): The root window of the tkinter application.
        '''
        self.root = root
        self.root.title("Cylinder Lateral Surface Area Calculator")
        
        # Create and place widgets
        self.create_widgets()
    
    def create_widgets(self):
        '''
        Create and place the widgets for the GUI.
        '''
        # Title Label
        title_label = tk.Label(self.root, text="Cylinder Lateral Surface Area Calculator", font=("Arial", 16))
        title_label.grid(row=0, column=0, columnspan=2, pady=10)
        
        # Radius Input
        radius_label = tk.Label(self.root, text="Radius (r):", font=("Arial", 12))
        radius_label.grid(row=1, column=0, padx=10, pady=5, sticky="e")
        self.radius_entry = tk.Entry(self.root, font=("Arial", 12))
        self.radius_entry.grid(row=1, column=1, padx=10, pady=5)
        
        # Height Input
        height_label = tk.Label(self.root, text="Height (h):", font=("Arial", 12))
        height_label.grid(row=2, column=0, padx=10, pady=5, sticky="e")
        self.height_entry = tk.Entry(self.root, font=("Arial", 12))
        self.height_entry.grid(row=2, column=1, padx=10, pady=5)
        
        # Calculate Button
        calculate_button = tk.Button(self.root, text="Calculate", font=("Arial", 12), command=self.calculate)
        calculate_button.grid(row=3, column=0, columnspan=2, pady=10)
        
        # Result Label
        self.result_label = tk.Label(self.root, text="", font=("Arial", 12), fg="green")
        self.result_label.grid(row=4, column=0, columnspan=2, pady=10)
    
    def calculate(self):
        '''
        Handle the calculation of the lateral surface area when the button is clicked.
        '''
        try:
            # Get inputs
            radius = float(self.radius_entry.get())
            height = float(self.height_entry.get())
            
            # Validate inputs
            if radius <= 0 or height <= 0:
                raise ValueError("Radius and height must be positive numbers.")
            
            # Calculate lateral surface area
            lateral_surface_area = calculate_lateral_surface_area(radius, height)
            
            # Display result
            self.result_label.config(text=f"Lateral Surface Area: {lateral_surface_area:.2f}")
        
        except ValueError as e:
            # Show error message
            messagebox.showerror("Input Error", str(e))


if __name__ == "__main__":
    # Create the main application window
    root = tk.Tk()
    app = CylinderCalculatorApp(root)
    root.mainloop()

'''
This file contains the main implementation of the Cylinder Calculator application.
It uses tkinter to create a GUI for calculating the lateral surface area of a cylinder.
'''
import tkinter as tk
from tkinter import messagebox
import math
def calculate_lateral_surface_area(radius, height):
    '''
    Calculate the lateral surface area of a cylinder.
    Parameters:
        radius (float): The radius of the cylinder.
        height (float): The height of the cylinder.
    Returns:
        float: The lateral surface area of the cylinder.
    '''
    return 2 * math.pi * radius * height
class CylinderCalculatorApp:
    '''
    A class to represent the Cylinder Calculator GUI application.
    '''
    def __init__(self, root):
        '''
        Initialize the GUI application.
        Parameters:
            root (tk.Tk): The root window of the tkinter application.
        '''
        self.root = root
        self.root.title("Cylinder Lateral Surface Area Calculator")
        # Create and place widgets
        self.create_widgets()
    def create_widgets(self):
        '''
        Create and place the widgets for the GUI.
        '''
        # Title Label
        title_label = tk.Label(self.root, text="Cylinder Lateral Surface Area Calculator", font=("Arial", 16))
        title_label.grid(row=0, column=0, columnspan=2, pady=10)
        # Radius Input
        radius_label = tk.Label(self.root, text="Radius (r):", font=("Arial", 12))
        radius_label.grid(row=1, column=0, padx=10, pady=5, sticky="e")
        self.radius_entry = tk.Entry(self.root, font=("Arial", 12))
        self.radius_entry.grid(row=1, column=1, padx=10, pady=5)
        # Height Input
        height_label = tk.Label(self.root, text="Height (h):", font=("Arial", 12))
        height_label.grid(row=2, column=0, padx=10, pady=5, sticky="e")
        self.height_entry = tk.Entry(self.root, font=("Arial", 12))
        self.height_entry.grid(row=2, column=1, padx=10, pady=5)
        # Calculate Button
        calculate_button = tk.Button(self.root, text="Calculate", font=("Arial", 12), command=self.calculate)
        calculate_button.grid(row=3, column=0, columnspan=2, pady=10)
        # Result Label
        self.result_label = tk.Label(self.root, text="", font=("Arial", 12), fg="green")
        self.result_label.grid(row=4, column=0, columnspan=2, pady=10)
    def calculate(self):
        '''
        Handle the calculation of the lateral surface area when the button is clicked.
        '''
        try:
            # Get inputs
            radius = float(self.radius_entry.get())
            height = float(self.height_entry.get())
            # Validate inputs
            if radius <= 0 or height <= 0:
                raise ValueError("Radius and height must be positive numbers.")
            # Calculate lateral surface area
            lateral_surface_area = calculate_lateral_surface_area(radius, height)
            # Display result
            self.result_label.config(text=f"Lateral Surface Area: {lateral_surface_area:.2f}")
        except ValueError as e:
            # Show error message
            messagebox.showerror("Input Error", str(e))
if __name__ == "__main__":
    # Create the main application window
    root = tk.Tk()
    app = CylinderCalculatorApp(root)
    root.mainloop()

'''  
This file contains the main implementation of the Cylinder Calculator application.  
It uses tkinter to create a GUI for calculating the lateral surface area of a cylinder.  
'''  
import tkinter as tk  
from tkinter import messagebox  
import math  

def calculate_lateral_surface_area(radius, height):  
    '''  
    Calculate the lateral surface area of a cylinder.  
    Parameters:  
        radius (float): The radius of the cylinder.  
        height (float): The height of the cylinder.  
    Returns:  
        float: The lateral surface area of the cylinder.  
    '''  
    return 2 * math.pi * radius * height  

class CylinderCalculatorApp:  
    '''  
    A class to represent the Cylinder Calculator GUI application.  
    '''  
    def __init__(self, root):  
        '''  
        Initialize the GUI application.  
        Parameters:  
            root (tk.Tk): The root window of the tkinter application.  
        '''  
        self.root = root  
        self.root.title("Cylinder Lateral Surface Area Calculator")  
        # Create and place widgets  
        self.create_widgets()  

    def create_widgets(self):  
        '''  
        Create and place the widgets for the GUI.  
        '''  
        # Title Label  
        title_label = tk.Label(self.root, text="Cylinder Lateral Surface Area Calculator", font=("Arial", 16))  
        title_label.grid(row=0, column=0, columnspan=2, pady=10)  

        # Radius Input  
        radius_label = tk.Label(self.root, text="Radius (r):", font=("Arial", 12))  
        radius_label.grid(row=1, column=0, padx=10, pady=5, sticky="e")  
        self.radius_entry = tk.Entry(self.root, font=("Arial", 12))  
        self.radius_entry.grid(row=1, column=1, padx=10, pady=5)  

        # Height Input  
        height_label = tk.Label(self.root, text="Height (h):", font=("Arial", 12))  
        height_label.grid(row=2, column=0, padx=10, pady=5, sticky="e")  
        self.height_entry = tk.Entry(self.root, font=("Arial", 12))  
        self.height_entry.grid(row=2, column=1, padx=10, pady=5)  

        # Calculate Button  
        calculate_button = tk.Button(self.root, text="Calculate", font=("Arial", 12), command=self.calculate)  
        calculate_button.grid(row=3, column=0, columnspan=2, pady=10)  

        # Result Label  
        self.result_label = tk.Label(self.root, text="", font=("Arial", 12), fg="green")  
        self.result_label.grid(row=4, column=0, columnspan=2, pady=10)  

    def calculate(self):  
        '''  
        Handle the calculation of the lateral surface area when the button is clicked.  
        '''  
        try:  
            # Get inputs  
            radius = float(self.radius_entry.get())  
            height = float(self.height_entry.get())  

            # Validate inputs  
            if radius <= 0 or height <= 0:  
                raise ValueError("Radius and height must be positive numbers.")  

            # Calculate lateral surface area  
            lateral_surface_area = calculate_lateral_surface_area(radius, height)  

            # Display result  
            self.result_label.config(text=f"Lateral Surface Area: {lateral_surface_area:.2f}")  
        except ValueError as e:  
            # Show error message  
            messagebox.showerror("Input Error", str(e))  

if __name__ == "__main__":  
    '''  
    Entry point for the Cylinder Calculator application.  
    Creates the main application window and starts the tkinter main loop.  
    '''  
    # Create the main application window  
    root = tk.Tk()  
    app = CylinderCalculatorApp(root)  
    root.mainloop()

def calculate_lateral_surface_area(radius, height):  
    '''  
    Calculate the lateral surface area of a cylinder.  
    Parameters:  
        radius (float): The radius of the cylinder.  
        height (float): The height of the cylinder.  
    Returns:  
        float: The lateral surface area of the cylinder.  
    '''  
    return 2 * math.pi * radius * height

'''  
This file contains the main implementation of the Cylinder Calculator application.  
It uses tkinter to create a GUI for calculating the lateral surface area of a cylinder.  
'''  
import tkinter as tk  
from tkinter import messagebox  
import math  
def calculate_lateral_surface_area(radius, height):  
    '''  
    Calculate the lateral surface area of a cylinder.  
    Parameters:  
        radius (float): The radius of the cylinder.  
        height (float): The height of the cylinder.  
    Returns:  
        float: The lateral surface area of the cylinder.  
    '''  
    return 2 * math.pi * radius * height  
class CylinderCalculatorApp:  
    '''  
    A class to represent the Cylinder Calculator GUI application.  
    '''  
    def __init__(self, root):  
        '''  
        Initialize the GUI application.  
        Parameters:  
            root (tk.Tk): The root window of the tkinter application.  
        '''  
        self.root = root  
        self.root.title("Cylinder Lateral Surface Area Calculator")  
        # Create and place widgets  
        self.create_widgets()  
    def create_widgets(self):  
        '''  
        Create and place the widgets for the GUI.  
        '''  
        # Title Label  
        title_label = tk.Label(self.root, text="Cylinder Lateral Surface Area Calculator", font=("Arial", 16))  
        title_label.grid(row=0, column=0, columnspan=2, pady=10)  
        # Radius Input  
        radius_label = tk.Label(self.root, text="Radius (r):", font=("Arial", 12))  
        radius_label.grid(row=1, column=0, padx=10, pady=5, sticky="e")  
        self.radius_entry = tk.Entry(self.root, font=("Arial", 12))  
        self.radius_entry.grid(row=1, column=1, padx=10, pady=5)  
        # Height Input  
        height_label = tk.Label(self.root, text="Height (h):", font=("Arial", 12))  
        height_label.grid(row=2, column=0, padx=10, pady=5, sticky="e")  
        self.height_entry = tk.Entry(self.root, font=("Arial", 12))  
        self.height_entry.grid(row=2, column=1, padx=10, pady=5)  
        # Calculate Button  
        calculate_button = tk.Button(self.root, text="Calculate", font=("Arial", 12), command=self.calculate)  
        calculate_button.grid(row=3, column=0, columnspan=2, pady=10)  
        # Result Label  
        self.result_label = tk.Label(self.root, text="", font=("Arial", 12), fg="green")  
        self.result_label.grid(row=4, column=0, columnspan=2, pady=10)  
    def calculate(self):  
        '''  
        Handle the calculation of the lateral surface area when the button is clicked.  
        '''  
        try:  
            # Get inputs  
            radius = float(self.radius_entry.get())  
            height = float(self.height_entry.get())  
            # Validate inputs  
            if radius <= 0 or height <= 0:  
                raise ValueError("Radius and height must be positive numbers.")  
            # Calculate lateral surface area  
            lateral_surface_area = calculate_lateral_surface_area(radius, height)  
            # Display result  
            self.result_label.config(text=f"Lateral Surface Area: {lateral_surface_area:.2f}")  
        except ValueError as e:  
            # Show error message  
            messagebox.showerror("Input Error", str(e))  
if __name__ == "__main__":  
    '''  
    Entry point for the Cylinder Calculator application.  
    Creates the main application window and starts the tkinter main loop.  
    '''  
    # Create the main application window  
    root = tk.Tk()  
    app = CylinderCalculatorApp(root)  
    root.mainloop()

'''
This file contains the main implementation of the Cylinder Calculator application.
It uses tkinter to create a GUI for calculating the lateral surface area of a cylinder.
'''
import tkinter as tk
from tkinter import messagebox
import math

def calculate_lateral_surface_area(radius, height):
    '''
    Calculate the lateral surface area of a cylinder.
    Parameters:
        radius (float): The radius of the cylinder.
        height (float): The height of the cylinder.
    Returns:
        float: The lateral surface area of the cylinder.
    '''
    return 2 * math.pi * radius * height

class CylinderCalculatorApp:
    '''
    A class to represent the Cylinder Calculator GUI application.
    '''
    def __init__(self, root):
        '''
        Initialize the GUI application.
        Parameters:
            root (tk.Tk): The root window of the tkinter application.
        '''
        self.root = root
        self.root.title("Cylinder Lateral Surface Area Calculator")
        # Create and place widgets
        self.create_widgets()

    def create_widgets(self):
        '''
        Create and place the widgets for the GUI.
        '''
        # Title Label
        title_label = tk.Label(self.root, text="Cylinder Lateral Surface Area Calculator", font=("Arial", 16))
        title_label.grid(row=0, column=0, columnspan=2, pady=10)

        # Radius Input
        radius_label = tk.Label(self.root, text="Radius (r):", font=("Arial", 12))
        radius_label.grid(row=1, column=0, padx=10, pady=5, sticky="e")
        self.radius_entry = tk.Entry(self.root, font=("Arial", 12))
        self.radius_entry.grid(row=1, column=1, padx=10, pady=5)

        # Height Input
        height_label = tk.Label(self.root, text="Height (h):", font=("Arial", 12))
        height_label.grid(row=2, column=0, padx=10, pady=5, sticky="e")
        self.height_entry = tk.Entry(self.root, font=("Arial", 12))
        self.height_entry.grid(row=2, column=1, padx=10, pady=5)

        # Calculate Button
        calculate_button = tk.Button(self.root, text="Calculate", font=("Arial", 12), command=self.calculate)
        calculate_button.grid(row=3, column=0, columnspan=2, pady=10)

        # Result Label
        self.result_label = tk.Label(self.root, text="", font=("Arial", 12), fg="green")
        self.result_label.grid(row=4, column=0, columnspan=2, pady=10)

    def calculate(self):
        '''
        Handle the calculation of the lateral surface area when the button is clicked.
        '''
        try:
            # Get inputs
            radius = float(self.radius_entry.get())
            height = float(self.height_entry.get())

            # Validate inputs
            if radius <= 0 or height <= 0:
                raise ValueError("Radius and height must be positive numbers.")

            # Calculate lateral surface area
            lateral_surface_area = calculate_lateral_surface_area(radius, height)

            # Display result
            self.result_label.config(text=f"Lateral Surface Area: {lateral_surface_area:.2f}")
        except ValueError as e:
            # Show error message
            messagebox.showerror("Input Error", str(e))

if __name__ == "__main__":
    '''
    Entry point for the Cylinder Calculator application.
    Creates the main application window and starts the tkinter main loop.
    '''
    # Create the main application window
    root = tk.Tk()
    app = CylinderCalculatorApp(root)
    root.mainloop()

import tkinter as tk

import math
def calculate_lateral_surface_area(radius, height):
    '''
    Calculate the lateral surface area of a cylinder.
    Parameters:
        radius (float): The radius of the cylinder.
        height (float): The height of the cylinder.
    Returns:
        float: The lateral surface area of the cylinder.
    '''
    return 2 * math.pi * radius * height
class CylinderCalculatorApp:
    '''
    A class to represent the Cylinder Calculator GUI application.
    '''

        '''
        Initialize the GUI application.
        Parameters:

        self.root = root
        self.root.title("Cylinder Lateral Surface Area Calculator")
        # Create and place widgets
        self.create_widgets()
    def create_widgets(self):
        '''

        # Title Label

        self.radius_entry.grid(row=1, column=1, padx=10, pady=5)
        # Height Input
        height_label = tk.Label(self.root, text="Height (h):", font=("Arial", 12))

            messagebox.showerror("Input Error", str(e))

'''
This file contains the main implementation of the Cylinder Calculator application.
It uses tkinter to create a GUI for calculating the lateral surface area of a cylinder.
'''
import tkinter as tk
from tkinter import messagebox
import math
def calculate_lateral_surface_area(radius, height):
    '''
    Calculate the lateral surface area of a cylinder.
    Parameters:
        radius (float): The radius of the cylinder.
        height (float): The height of the cylinder.
    Returns:
        float: The lateral surface area of the cylinder.
    '''
    return 2 * math.pi * radius * height
class CylinderCalculatorApp:
    '''
    A class to represent the Cylinder Calculator GUI application.
    '''
    def __init__(self, root):
        '''
        Initialize the GUI application.
        Parameters:
            root (tk.Tk): The root window of the tkinter application.
        '''
        self.root = root
        self.root.title("Cylinder Lateral Surface Area Calculator")
        # Create and place widgets
        self.create_widgets()
    def create_widgets(self):
        '''
        Create and place the widgets for the GUI.
        '''
        # Title Label
        title_label = tk.Label(self.root, text="Cylinder Lateral Surface Area Calculator", font=("Arial", 16))
        title_label.grid(row=0, column=0, columnspan=2, pady=10)
        # Radius Input
        radius_label = tk.Label(self.root, text="Radius (r):", font=("Arial", 12))
        radius_label.grid(row=1, column=0, padx=10, pady=5, sticky="e")
        self.radius_entry = tk.Entry(self.root, font=("Arial", 12))
        self.radius_entry.grid(row=1, column=1, padx=10, pady=5)
        # Height Input
        height_label = tk.Label(self.root, text="Height (h):", font=("Arial", 12))
        height_label.grid(row=2, column=0, padx=10, pady=5, sticky="e")
        self.height_entry = tk.Entry(self.root, font=("Arial", 12))
        self.height_entry.grid(row=2, column=1, padx=10, pady=5)
        # Calculate Button
        calculate_button = tk.Button(self.root, text="Calculate", font=("Arial", 12), command=self.calculate)
        calculate_button.grid(row=3, column=0, columnspan=2, pady=10)
        # Result Label
        self.result_label = tk.Label(self.root, text="", font=("Arial", 12), fg="green")
        self.result_label.grid(row=4, column=0, columnspan=2, pady=10)
    def calculate(self):
        '''
        Handle the calculation of the lateral surface area when the button is clicked.
        '''
        try:
            # Get inputs
            radius = float(self.radius_entry.get())
            height = float(self.height_entry.get())
            # Validate inputs
            if radius <= 0 or height <= 0:
                raise ValueError("Radius and height must be positive numbers.")
            # Calculate lateral surface area
            lateral_surface_area = calculate_lateral_surface_area(radius, height)
            # Display result
            self.result_label.config(text=f"Lateral Surface Area: {lateral_surface_area:.2f}")
        except ValueError as e:
            # Show error message
            messagebox.showerror("Input Error", str(e))
if __name__ == "__main__":
    '''
