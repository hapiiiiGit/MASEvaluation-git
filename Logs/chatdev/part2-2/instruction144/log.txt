'''

This script creates a GUI application using tkinter to find the sublist with the minimum length from a list of lists input by the user.

'''

import tkinter as tk
from tkinter import messagebox

def find_min_length_sublist(lists):
    '''
    Finds the sublist with the minimum length from a list of lists.

    Parameters:
    lists (list of lists): The list containing sublists.

    Returns:
    list: The sublist with the minimum length.
    '''
    if not lists:
        return []
    min_length_sublist = min(lists, key=len)
    return min_length_sublist

def on_submit(entry, result_label):
    '''
    Handles the submit button click event, processes the input, and updates the result label.

    Parameters:
    entry (tk.Entry): The entry widget containing the user's input.
    result_label (tk.Label): The label widget to display the result.
    '''
    try:
        # Parse the input string to a list of lists
        input_str = entry.get()
        lists = eval(input_str)  # Note: eval can be dangerous; ensure input is sanitized in real applications
        if not isinstance(lists, list) or not all(isinstance(i, list) for i in lists):
            raise ValueError("Input must be a list of lists.")
        
        # Find the sublist with the minimum length
        min_sublist = find_min_length_sublist(lists)
        
        # Update the result label
        result_label.config(text=f"Minimum Length Sublist: {min_sublist}")
    except Exception as e:
        messagebox.showerror("Error", f"Invalid input: {e}")

def create_gui():
    '''
    Sets up the GUI components and starts the tkinter main loop.
    '''
    # Create the main window
    root = tk.Tk()
    root.title("Find Minimum Length Sublist")

    # Create and place the input label and entry
    input_label = tk.Label(root, text="Enter a list of lists:")
    input_label.pack(pady=5)
    
    entry = tk.Entry(root, width=50)
    entry.pack(pady=5)

    # Create and place the submit button
    submit_button = tk.Button(root, text="Submit", command=lambda: on_submit(entry, result_label))
    submit_button.pack(pady=5)

    # Create and place the result label
    result_label = tk.Label(root, text="")
    result_label.pack(pady=5)

    # Start the GUI event loop
    root.mainloop()

if __name__ == "__main__":
    create_gui()
```

'''

This script creates a GUI application using tkinter to find the sublist with the minimum length from a list of lists input by the user.

'''

import tkinter as tk
from tkinter import messagebox

def find_min_length_sublist(lists):
    '''
    Finds the sublist with the minimum length from a list of lists.

    Parameters:
    lists (list of lists): The list containing sublists.

    Returns:
    list: The sublist with the minimum length.
    '''
    if not lists:
        return []
    min_length_sublist = min(lists, key=len)
    return min_length_sublist

def on_submit(entry, result_label):
    '''
    Handles the submit button click event, processes the input, and updates the result label.

    Parameters:
    entry (tk.Entry): The entry widget containing the user's input.
    result_label (tk.Label): The label widget to display the result.
    '''
    try:
        # Parse the input string to a list of lists
        input_str = entry.get()
        lists = eval(input_str)  # Note: eval can be dangerous; ensure input is sanitized in real applications
        if not isinstance(lists, list) or not all(isinstance(i, list) for i in lists):
            raise ValueError("Input must be a list of lists.")
        
        # Find the sublist with the minimum length
        min_sublist = find_min_length_sublist(lists)
        
        # Update the result label
        result_label.config(text=f"Minimum Length Sublist: {min_sublist}")
    except Exception as e:
        messagebox.showerror("Error", f"Invalid input: {e}")

def create_gui():
    '''
    Sets up the GUI components and starts the tkinter main loop.
    '''
    # Create the main window
    root = tk.Tk()
    root.title("Find Minimum Length Sublist")

    # Create and place the input label and entry
    input_label = tk.Label(root, text="Enter a list of lists:")
    input_label.pack(pady=5)
    
    entry = tk.Entry(root, width=50)
    entry.pack(pady=5)

    # Create and place the submit button
    submit_button = tk.Button(root, text="Submit", command=lambda: on_submit(entry, result_label))
    submit_button.pack(pady=5)

    # Create and place the result label
    result_label = tk.Label(root, text="")
    result_label.pack(pady=5)

    # Start the GUI event loop
    root.mainloop()

if __name__ == "__main__":
    create_gui()

'''
import tkinter as tk
from tkinter import messagebox
import ast
def find_min_length_sublist(lists):
    '''
    Finds the sublist with the minimum length from a list of lists.
    Parameters:
    lists (list of lists): The list containing sublists.
    Returns:
    list: The sublist with the minimum length.
    '''
    if not lists:
        return []
    min_length_sublist = min(lists, key=len)
    return min_length_sublist
def on_submit(entry, result_label):
    '''
    Handles the submit button click event, processes the input, and updates the result label.
    Parameters:
    entry (tk.Entry): The entry widget containing the user's input.
    result_label (tk.Label): The label widget to display the result.
    '''
    try:
        # Parse the input string to a list of lists
        input_str = entry.get()
        lists = ast.literal_eval(input_str)  # Use ast.literal_eval for safe evaluation
        if not isinstance(lists, list) or not all(isinstance(i, list) for i in lists):
            raise ValueError("Input must be a list of lists.")
        # Find the sublist with the minimum length
        min_sublist = find_min_length_sublist(lists)
        # Update the result label
        result_label.config(text=f"Minimum Length Sublist: {min_sublist}")
    except Exception as e:
        messagebox.showerror("Error", f"Invalid input: {e}")

'''
This script creates a GUI application using tkinter to find the sublist with the minimum length from a list of lists input by the user.
'''
import tkinter as tk
from tkinter import messagebox
import ast
def find_min_length_sublist(lists):
    '''
    Finds the sublist with the minimum length from a list of lists.
    Parameters:
    lists (list of lists): The list containing sublists.
    Returns:
    list: The sublist with the minimum length.
    '''
    if not lists:
        return []
    min_length_sublist = min(lists, key=len)
    return min_length_sublist
def on_submit(entry, result_label):
    '''
    Handles the submit button click event, processes the input, and updates the result label.
    Parameters:
    entry (tk.Entry): The entry widget containing the user's input.
    result_label (tk.Label): The label widget to display the result.
    '''
    try:
        # Parse the input string to a list of lists
        input_str = entry.get()
        lists = ast.literal_eval(input_str)  # Use ast.literal_eval for safe evaluation
        if not isinstance(lists, list) or not all(isinstance(i, list) for i in lists):
            raise ValueError("Input must be a list of lists.")
        # Find the sublist with the minimum length
        min_sublist = find_min_length_sublist(lists)
        # Update the result label
        result_label.config(text=f"Minimum Length Sublist: {min_sublist}")
    except Exception as e:
        messagebox.showerror("Error", f"Invalid input: {e}")
def create_gui():
    '''
    Sets up the GUI components and starts the tkinter main loop.
    '''
    # Create the main window
    root = tk.Tk()

    input_label = tk.Label(root, text="Enter a list of lists:")

    result_label = tk.Label(root, text="")

    # Start the GUI event loop
    root.mainloop()
