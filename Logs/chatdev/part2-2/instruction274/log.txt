'''
Module to calculate the nth Jacobsthal number.

def jacobsthal_number(n):
    '''
    Calculates the nth Jacobsthal number using an iterative approach.
    Parameters:
        n (int): The position of the Jacobsthal number to calculate.
    Returns:
        int: The nth Jacobsthal number.
    '''
    if n == 0:
        return 0
    elif n == 1:
        return 1
    jacobsthal_prev = 0
    jacobsthal_curr = 1
    for i in range(2, n + 1):
        jacobsthal_next = jacobsthal_curr + 2 * jacobsthal_prev
        jacobsthal_prev = jacobsthal_curr
        jacobsthal_curr = jacobsthal_next
    return jacobsthal_curr

import tkinter as tk
from jacobsthal import jacobsthal_number

class JacobsthalApp:

'''
Module to calculate the nth Jacobsthal number.

def jacobsthal_number(n):
    '''
    Calculates the nth Jacobsthal number using an iterative approach.
    Parameters:
        n (int): The position of the Jacobsthal number to calculate.
    Returns:
        int: The nth Jacobsthal number.
    '''
    if n == 0:
        return 0
    elif n == 1:
        return 1
    
    jacobsthal_prev = 0
    jacobsthal_curr = 1
    
    for i in range(2, n + 1):
        jacobsthal_next = jacobsthal_curr + 2 * jacobsthal_prev
        jacobsthal_prev = jacobsthal_curr
        jacobsthal_curr = jacobsthal_next
    
    return jacobsthal_curr

def jacobsthal_number(n):
    '''
    Calculates the nth Jacobsthal number using an iterative approach.
    Parameters:
        n (int): The position of the Jacobsthal number to calculate.
    Returns:
        int: The nth Jacobsthal number.
    '''
    if n == 0:
        return 0
    elif n == 1:
        return 1
    jacobsthal_prev = 0
    jacobsthal_curr = 1
    for i in range(2, n + 1):
        jacobsthal_next = jacobsthal_curr + 2 * jacobsthal_prev

    return jacobsthal_curr
