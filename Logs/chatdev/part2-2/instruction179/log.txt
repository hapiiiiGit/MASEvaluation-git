'''

This module contains the main application class for the Duplicate Remover App.
It sets up the GUI and handles user interactions.

'''

import tkinter as tk
from tkinter import messagebox
from duplicate_remover import remove_duplicates

class DuplicateRemoverApp:
    def __init__(self, root):
        self.root = root
        self.root.title("Duplicate Remover App")

        self.label = tk.Label(root, text="Enter lists of numbers separated by commas:")
        self.label.pack()

        self.text_input = tk.Text(root, height=10, width=50)
        self.text_input.pack()

        self.remove_button = tk.Button(root, text="Remove Duplicates", command=self.remove_duplicates)
        self.remove_button.pack()

        self.result_label = tk.Label(root, text="Result:")
        self.result_label.pack()

        self.result_output = tk.Text(root, height=10, width=50, state='disabled')
        self.result_output.pack()

    def remove_duplicates(self):
        input_text = self.text_input.get("1.0", tk.END).strip()
        try:
            lists = [list(map(int, lst.split(','))) for lst in input_text.split('\n') if lst.strip()]
            result = remove_duplicates(lists)
            self.display_result(result)
        except ValueError:
            messagebox.showerror("Input Error", "Please enter valid lists of numbers.")

    def display_result(self, result):
        self.result_output.config(state='normal')
        self.result_output.delete("1.0", tk.END)
        for lst in result:
            self.result_output.insert(tk.END, ', '.join(map(str, lst)) + '\n')
        self.result_output.config(state='disabled')

if __name__ == "__main__":
    root = tk.Tk()
    app = DuplicateRemoverApp(root)
    root.mainloop()
```

'''

This module provides a function to remove duplicate numbers from lists.

'''

def remove_duplicates(lists):
    """
    Remove duplicate numbers from each list in the given list of lists.

    :param lists: A list of lists containing numbers.
    :return: A list of lists with duplicates removed.
    """
    return [list(dict.fromkeys(lst)) for lst in lists]
```

This module contains the main application class for the Duplicate Remover App.

import tkinter as tk
from tkinter import messagebox
from duplicate_remover import remove_duplicates

class DuplicateRemoverApp:
    def __init__(self, root):
        self.root = root
        self.root.title("Duplicate Remover App")

        self.label = tk.Label(root, text="Enter lists of numbers separated by commas:")
        self.label.pack()

        self.text_input = tk.Text(root, height=10, width=50)
        self.text_input.pack()

        self.remove_button = tk.Button(root, text="Remove Duplicates", command=self.remove_duplicates)
        self.remove_button.pack()

        self.result_label = tk.Label(root, text="Result:")
        self.result_label.pack()

        self.result_output = tk.Text(root, height=10, width=50, state='disabled')
        self.result_output.pack()

    def remove_duplicates(self):
        input_text = self.text_input.get("1.0", tk.END).strip()
        try:
            lists = [list(map(int, lst.split(','))) for lst in input_text.split('\n') if lst.strip()]
            result = remove_duplicates(lists)
            self.display_result(result)
        except ValueError:
            messagebox.showerror("Input Error", "Please enter valid lists of numbers.")

    def display_result(self, result):
        self.result_output.config(state='normal')
        self.result_output.delete("1.0", tk.END)
        for lst in result:
            self.result_output.insert(tk.END, ', '.join(map(str, lst)) + '\n')
        self.result_output.config(state='disabled')

if __name__ == "__main__":
    root = tk.Tk()
    app = DuplicateRemoverApp(root)
    root.mainloop()

    return [list(dict.fromkeys(lst)) for lst in lists]

import tkinter as tk
from tkinter import messagebox
from duplicate_remover import remove_duplicates
class DuplicateRemoverApp:
    def __init__(self, root):
        self.root = root
        self.root.title("Duplicate Remover App")
        self.label = tk.Label(root, text="Enter lists of numbers separated by commas:")
        self.label.pack()
        self.text_input = tk.Text(root, height=10, width=50)
        self.text_input.pack()
        self.remove_button = tk.Button(root, text="Remove Duplicates", command=self.remove_duplicates)
        self.remove_button.pack()
        self.result_label = tk.Label(root, text="Result:")
        self.result_label.pack()
        self.result_output = tk.Text(root, height=10, width=50, state='disabled')
        self.result_output.pack()
    def remove_duplicates(self):
        input_text = self.text_input.get("1.0", tk.END).strip()
        try:
            lists = [list(map(int, lst.split(','))) for lst in input_text.split('\n') if lst.strip()]
            result = remove_duplicates(lists)
            self.display_result(result)
        except ValueError:
            messagebox.showerror("Input Error", "Please enter valid lists of numbers.")
    def display_result(self, result):
        self.result_output.config(state='normal')
        self.result_output.delete("1.0", tk.END)
        for lst in result:
            self.result_output.insert(tk.END, ', '.join(map(str, lst)) + '\n')
        self.result_output.config(state='disabled')
if __name__ == "__main__":
    root = tk.Tk()
    app = DuplicateRemoverApp(root)
    root.mainloop()

'''
This module contains the main application class for the Duplicate Remover App.
It sets up the GUI and handles user interactions.
'''
import tkinter as tk
from tkinter import messagebox
from duplicate_remover import remove_duplicates
class DuplicateRemoverApp:
    def __init__(self, root):
        self.root = root
        self.root.title("Duplicate Remover App")
        self.label = tk.Label(root, text="Enter lists of numbers separated by commas:")
        self.label.pack()
        self.text_input = tk.Text(root, height=10, width=50)
        self.text_input.pack()
        self.remove_button = tk.Button(root, text="Remove Duplicates", command=self.remove_duplicates)
        self.remove_button.pack()
        self.result_label = tk.Label(root, text="Result:")
        self.result_label.pack()
        self.result_output = tk.Text(root, height=10, width=50, state='disabled')
        self.result_output.pack()
    def remove_duplicates(self):
        input_text = self.text_input.get("1.0", tk.END).strip()
        try:
            lists = [list(map(int, lst.split(','))) for lst in input_text.split('\n') if lst.strip()]
            result = remove_duplicates(lists)
            self.display_result(result)
        except ValueError:
            messagebox.showerror("Input Error", "Please enter valid lists of numbers.")
    def display_result(self, result):
        self.result_output.config(state='normal')
        self.result_output.delete("1.0", tk.END)
        for lst in result:
            self.result_output.insert(tk.END, ', '.join(map(str, lst)) + '\n')
        self.result_output.config(state='disabled')
if __name__ == "__main__":
    root = tk.Tk()
    app = DuplicateRemoverApp(root)
    root.mainloop()

'''
This module contains the main application class for the Duplicate Remover App.
It sets up the GUI and handles user interactions.
'''
import tkinter as tk
from tkinter import messagebox
from duplicate_remover import remove_duplicates
class DuplicateRemoverApp:
    def __init__(self, root):
        self.root = root
        self.root.title("Duplicate Remover App")
        self.label = tk.Label(root, text="Enter lists of numbers separated by commas:")
        self.label.pack()
        self.text_input = tk.Text(root, height=10, width=50)
        self.text_input.pack()
        self.remove_button = tk.Button(root, text="Remove Duplicates", command=self.remove_duplicates)
        self.remove_button.pack()
        self.result_label = tk.Label(root, text="Result:")
        self.result_label.pack()
        self.result_output = tk.Text(root, height=10, width=50, state='disabled')
        self.result_output.pack()
    def remove_duplicates(self):
        input_text = self.text_input.get("1.0", tk.END).strip()
        try:
            lists = [list(map(int, lst.split(','))) for lst in input_text.split('\n') if lst.strip()]
            result = remove_duplicates(lists)
            self.display_result(result)
        except ValueError:
            messagebox.showerror("Input Error", "Please enter valid lists of numbers.")
    def display_result(self, result):
        self.result_output.config(state='normal')
        self.result_output.delete("1.0", tk.END)
        for lst in result:
            self.result_output.insert(tk.END, ', '.join(map(str, lst)) + '\n')
        self.result_output.config(state='disabled')
if __name__ == "__main__":
    root = tk.Tk()
    app = DuplicateRemoverApp(root)
    root.mainloop()

def remove_duplicates(lists):
    """
    Remove duplicate numbers from each list in the given list of lists.
    :param lists: A list of lists containing numbers.
    :return: A list of lists with duplicates removed.
    """
    return [list(dict.fromkeys(lst)) for lst in lists]
