'''

Main application file for the GUI that allows users to add a dictionary to a tuple.

'''

import tkinter as tk
from tkinter import messagebox
from utils import add_dict_to_tuple

class MainApp:
    def __init__(self, root):
        self.root = root
        self.root.title("Add Dictionary to Tuple")

        # Labels and Entries for Tuple and Dictionary
        self.tuple_label = tk.Label(root, text="Enter Tuple (e.g., (1, 2, 3)):")
        self.tuple_label.pack()
        self.tuple_entry = tk.Entry(root)
        self.tuple_entry.pack()

        self.dict_label = tk.Label(root, text="Enter Dictionary (e.g., {'key': 'value'}):")
        self.dict_label.pack()
        self.dict_entry = tk.Entry(root)
        self.dict_entry.pack()

        # Button to perform the operation
        self.add_button = tk.Button(root, text="Add Dictionary to Tuple", command=self.add_dictionary_to_tuple)
        self.add_button.pack()

        # Label to display the result
        self.result_label = tk.Label(root, text="")
        self.result_label.pack()

    def add_dictionary_to_tuple(self):
        try:
            # Get user inputs
            original_tuple = eval(self.tuple_entry.get())
            dictionary = eval(self.dict_entry.get())

            # Validate inputs
            if not isinstance(original_tuple, tuple):
                raise ValueError("The first input must be a tuple.")
            if not isinstance(dictionary, dict):
                raise ValueError("The second input must be a dictionary.")

            # Perform the operation
            new_tuple = add_dict_to_tuple(original_tuple, dictionary)

            # Display the result
            self.result_label.config(text=f"Result: {new_tuple}")

        except Exception as e:
            messagebox.showerror("Error", str(e))

if __name__ == "__main__":
    root = tk.Tk()
    app = MainApp(root)
    root.mainloop()

'''

Main application file for the GUI that allows users to add a dictionary to a tuple.

'''

import tkinter as tk
from tkinter import messagebox
from utils import add_dict_to_tuple

class MainApp:
    def __init__(self, root):
        self.root = root
        self.root.title("Add Dictionary to Tuple")

        # Labels and Entries for Tuple and Dictionary
        self.tuple_label = tk.Label(root, text="Enter Tuple (e.g., (1, 2, 3)):")
        self.tuple_label.pack()
        self.tuple_entry = tk.Entry(root)
        self.tuple_entry.pack()

        self.dict_label = tk.Label(root, text="Enter Dictionary (e.g., {'key': 'value'}):")
        self.dict_label.pack()
        self.dict_entry = tk.Entry(root)
        self.dict_entry.pack()

        # Button to perform the operation
        self.add_button = tk.Button(root, text="Add Dictionary to Tuple", command=self.add_dictionary_to_tuple)
        self.add_button.pack()

        # Label to display the result
        self.result_label = tk.Label(root, text="")
        self.result_label.pack()

    def add_dictionary_to_tuple(self):
        try:
            # Get user inputs
            original_tuple = eval(self.tuple_entry.get())
            dictionary = eval(self.dict_entry.get())

            # Validate inputs
            if not isinstance(original_tuple, tuple):
                raise ValueError("The first input must be a tuple.")
            if not isinstance(dictionary, dict):
                raise ValueError("The second input must be a dictionary.")

            # Perform the operation
            new_tuple = add_dict_to_tuple(original_tuple, dictionary)

            # Display the result
            self.result_label.config(text=f"Result: {new_tuple}")

        except Exception as e:
            messagebox.showerror("Error", str(e))

if __name__ == "__main__":
    root = tk.Tk()
    app = MainApp(root)
    root.mainloop()

        self.root = root
        self.root.title("Add Dictionary to Tuple")

        self.tuple_label.pack()

def add_dict_to_tuple(original_tuple, dictionary):

class MainApp:
    def __init__(self, root):
        """
        Initialize the main application window.
        Parameters:
        root (Tk): The root window of the application.
        """
        self.root = root
        self.root.title("Add Dictionary to Tuple")
        # Labels and Entries for Tuple and Dictionary
        self.tuple_label = tk.Label(root, text="Enter Tuple (e.g., (1, 2, 3)):")
        self.tuple_label.pack()
        self.tuple_entry = tk.Entry(root)
        self.tuple_entry.pack()
        self.dict_label = tk.Label(root, text="Enter Dictionary (e.g., {'key': 'value'}):")
        self.dict_label.pack()
        self.dict_entry = tk.Entry(root)
        self.dict_entry.pack()
        # Button to perform the operation
        self.add_button = tk.Button(root, text="Add Dictionary to Tuple", command=self.add_dictionary_to_tuple)
        self.add_button.pack()
        # Label to display the result
        self.result_label = tk.Label(root, text="")
        self.result_label.pack()
    def add_dictionary_to_tuple(self):
        """
        Add a dictionary to a tuple based on user input and display the result.
        """
        try:
            # Get user inputs
            original_tuple = eval(self.tuple_entry.get())
            dictionary = eval(self.dict_entry.get())
            # Validate inputs
            if not isinstance(original_tuple, tuple):
                raise ValueError("The first input must be a tuple.")
            if not isinstance(dictionary, dict):
                raise ValueError("The second input must be a dictionary.")
            # Perform the operation
            new_tuple = add_dict_to_tuple(original_tuple, dictionary)
            # Display the result
            self.result_label.config(text=f"Result: {new_tuple}")
        except SyntaxError:
            messagebox.showerror("Error", "Invalid input format. Please check your inputs.")
        except Exception as e:
            messagebox.showerror("Error", str(e))
if __name__ == "__main__":
    root = tk.Tk()
    app = MainApp(root)
    root.mainloop()
