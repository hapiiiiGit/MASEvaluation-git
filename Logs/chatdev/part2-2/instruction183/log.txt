This is the main application file that sets up the GUI using Tkinter. It allows the user to input a number and calculates the last digit of its factorial.

import tkinter as tk
from factorial_utils import last_digit_of_factorial

class MainApp:
    def __init__(self, root):
        self.root = root
        self.root.title("Factorial Last Digit Finder")

        self.label.pack()

        self.entry = tk.Entry(root)
        self.entry.pack()

        self.calculate_button = tk.Button(root, text="Calculate", command=self.calculate_last_digit)
        self.calculate_button.pack()

        self.result_label = tk.Label(root, text="")
        self.result_label.pack()

    def calculate_last_digit(self):
        try:
            number = int(self.entry.get())
            last_digit = last_digit_of_factorial(number)
            self.result_label.config(text=f"The last digit of {number}! is {last_digit}")

            self.result_label.config(text="Please enter a valid integer.")

if __name__ == "__main__":
    root = tk.Tk()
    app = MainApp(root)
    root.mainloop()

#### factorial_utils.py

```python
'''

This module contains the logic for calculating the last digit of the factorial of a given number.

'''

def last_digit_of_factorial(n):
    """
    Calculate the last digit of the factorial of a given number n.
    For n >= 5, the last digit is always 0 because 5! = 120 and any larger factorial
    will include at least one 10 as a factor (from 2 * 5).
    """
    if n < 0:
        raise ValueError("Negative numbers do not have factorials.")
    elif n == 0 or n == 1:
        return 1
    elif n >= 5:
        return 0
    else:
        factorial = 1
        for i in range(2, n + 1):
            factorial *= i
        return factorial % 10
```

'''

This is the main application file that sets up the GUI using Tkinter. It allows the user to input a number and calculates the last digit of its factorial.

import tkinter as tk
from factorial_utils import last_digit_of_factorial

class MainApp:
    def __init__(self, root):
        self.root = root
        self.root.title("Factorial Last Digit Finder")

        self.label = tk.Label(root, text="Enter a number:")
        self.label.pack()

        self.entry = tk.Entry(root)
        self.entry.pack()

        self.calculate_button = tk.Button(root, text="Calculate", command=self.calculate_last_digit)
        self.calculate_button.pack()

        self.result_label = tk.Label(root, text="")
        self.result_label.pack()

    def calculate_last_digit(self):
        try:
            number = int(self.entry.get())
            last_digit = last_digit_of_factorial(number)
            self.result_label.config(text=f"The last digit of {number}! is {last_digit}")
        except ValueError:
            self.result_label.config(text="Please enter a valid integer.")

if __name__ == "__main__":
    root = tk.Tk()
    app = MainApp(root)
    root.mainloop()

```python
'''

This module contains the logic for calculating the last digit of the factorial of a given number.

'''

def last_digit_of_factorial(n):
    """
    Calculate the last digit of the factorial of a given number n.
    For n >= 5, the last digit is always 0 because 5! = 120 and any larger factorial
    will include at least one 10 as a factor (from 2 * 5).
    """
    if n < 0:
        raise ValueError("Negative numbers do not have factorials.")
    elif n == 0 or n == 1:
        return 1
    elif n >= 5:
        return 0
    else:
        factorial = 1
        for i in range(2, n + 1):
            factorial *= i
        return factorial % 10
```

'''
This module contains the logic for calculating the last digit of the factorial of a given number.
'''
def last_digit_of_factorial(n):
    """
    Calculate the last digit of the factorial of a given number n.
    For n >= 5, the last digit is always 0 because 5! = 120 and any larger factorial
    will include at least one 10 as a factor (from 2 * 5).
    """
    if n < 0:
        raise ValueError("Negative numbers do not have factorials.")
    elif n == 0 or n == 1:
        return 1
    elif n >= 5:
        return 0
    else:
        factorial = 1
        for i in range(2, n + 1):
            factorial *= i
        return factorial % 10

'''
This module contains the logic for calculating the last digit of the factorial of a given number.
'''
def last_digit_of_factorial(n):
    """
    Calculate the last digit of the factorial of a given number n.
    For n >= 5, the last digit is always 0 because 5! = 120 and any larger factorial
    will include at least one 10 as a factor (from 2 * 5).
    """
    if n < 0:
        raise ValueError("Negative numbers do not have factorials.")
    elif n == 0 or n == 1:
        return 1
    elif n >= 5:
        return 0
    else:
        factorial = 1
        for i in range(2, n + 1):
            factorial *= i
        return factorial % 10

'''
This module contains the logic for calculating the last digit of the factorial of a given number.
'''
def last_digit_of_factorial(n):
    """
    Calculate the last digit of the factorial of a given number n.
    For n >= 5, the last digit is always 0 because 5! = 120 and any larger factorial
    will include at least one 10 as a factor (from 2 * 5).
    """
    if n < 0:
        raise ValueError("Negative numbers do not have factorials.")
    elif n == 0 or n == 1:
        return 1
    elif n >= 5:
        return 0
    else:
        factorial = 1
        for i in range(2, n + 1):
            factorial *= i
        return factorial % 10
