```python
'''

FILENAME: main.py

'''

import re
import tkinter as tk
from tkinter import messagebox

def extract_values_between_quotes(text):
    '''
    DOCSTRING: Extract all substrings enclosed in double quotes from the given text.
    Parameters:
        text (str): The input string from which to extract values.
    Returns:
        list: A list of strings found between quotation marks.
    '''
    return re.findall(r'"(.*?)"', text)

class QuotationExtractorApp:
    '''
    DOCSTRING: A class to create a GUI application for extracting values between quotation marks.
    '''
    def __init__(self, root):
        '''
        DOCSTRING: Initialize the GUI components.
        Parameters:
            root (tk.Tk): The root window of the application.
        '''
        self.root = root
        self.root.title("Quotation Extractor")

        self.label = tk.Label(root, text="Enter text:")
        self.label.pack()

        self.text_entry = tk.Entry(root, width=50)
        self.text_entry.pack()

        self.extract_button = tk.Button(root, text="Extract", command=self.extract_quotations)
        self.extract_button.pack()

        self.result_label = tk.Label(root, text="Extracted Values:")
        self.result_label.pack()

        self.result_text = tk.Text(root, height=10, width=50)
        self.result_text.pack()

    def extract_quotations(self):
        '''
        DOCSTRING: Extract values between quotation marks from the input string and display them.
        '''
        input_text = self.text_entry.get()
        extracted_values = extract_values_between_quotes(input_text)
        self.result_text.delete(1.0, tk.END)
        if extracted_values:
            for value in extracted_values:
                self.result_text.insert(tk.END, f"{value}\n")
        else:
            messagebox.showinfo("No Matches", "No values found between quotation marks.")

if __name__ == "__main__":
    root = tk.Tk()
    app = QuotationExtractorApp(root)
    root.mainloop()
```

def extract_values_between_quotes(text):
    '''
    DOCSTRING: Extract all substrings enclosed in double quotes from the given text.

    Returns:
        list: A list of strings found between quotation marks.

'''
FILENAME: main.py
DOCSTRING: This script contains a function to extract values between quotation marks from a given string and a GUI application to facilitate user interaction.
'''
import re
import tkinter as tk
from tkinter import messagebox
def extract_values_between_quotes(text):
    '''
    DOCSTRING: Extract all substrings enclosed in double quotes from the given text.
    Parameters:
        text (str): The input string from which to extract values.
    Returns:
        list: A list of strings found between quotation marks.
    '''
    return re.findall(r'"(.*?)"', text)
class QuotationExtractorApp:
    '''
    DOCSTRING: A class to create a GUI application for extracting values between quotation marks.
    '''
    def __init__(self, root):
        '''
        DOCSTRING: Initialize the GUI components.
        Parameters:
            root (tk.Tk): The root window of the application.
        '''
        self.root = root
        self.root.title("Quotation Extractor")
        self.label = tk.Label(root, text="Enter text:")
        self.label.pack()
        self.text_entry = tk.Entry(root, width=50)
        self.text_entry.pack()
        self.extract_button = tk.Button(root, text="Extract", command=self.extract_quotations)
        self.extract_button.pack()
        self.result_label = tk.Label(root, text="Extracted Values:")
        self.result_label.pack()
        self.result_text = tk.Text(root, height=10, width=50)
        self.result_text.pack()
    def extract_quotations(self):
        '''
        DOCSTRING: Extract values between quotation marks from the input string and display them.
        '''
        input_text = self.text_entry.get()
        extracted_values = extract_values_between_quotes(input_text)
        self.result_text.delete(1.0, tk.END)
        if extracted_values:
            for value in extracted_values:
                self.result_text.insert(tk.END, f"{value}\n")
        else:
            messagebox.showinfo("No Matches", "No values found between quotation marks.")
if __name__ == "__main__":
    root = tk.Tk()
    app = QuotationExtractorApp(root)
    root.mainloop()
