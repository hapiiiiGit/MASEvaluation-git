```python

'''

Function to interleave three lists of the same length into a single flat list.

'''

def interleave_three_lists(list1, list2, list3):
    '''
    Interleave three lists into a single flat list.

    Parameters:
    list1 (list): First list to interleave.
    list2 (list): Second list to interleave.
    list3 (list): Third list to interleave.

    Returns:
    list: Interleaved flat list.
    '''
    interleaved_list = []
    for i in range(len(list1)):
        interleaved_list.append(list1[i])
        interleaved_list.append(list2[i])
        interleaved_list.append(list3[i])

'''
Function to interleave three lists of the same length into a single flat list.
'''
def interleave_three_lists(list1, list2, list3):
    '''
    Interleave three lists into a single flat list.
    Parameters:
    list1 (list): First list to interleave.
    list2 (list): Second list to interleave.
    list3 (list): Third list to interleave.
    Returns:
    list: Interleaved flat list.
    '''
    # Initialize the interleaved list
    interleaved_list = []
    # Iterate through the lists and interleave their elements
    for i in range(len(list1)):
        interleaved_list.append(list1[i])
        interleaved_list.append(list2[i])
        interleaved_list.append(list3[i])
    return interleaved_list

        # Input for List 3
        self.label3 = tk.Label(root, text="Enter List 3 (comma-separated):")

        except Exception as e:
            messagebox.showerror("Error", str(e))

'''
Function to interleave three lists of the same length into a single flat list.
'''
def interleave_three_lists(list1, list2, list3):
    '''
    Interleave three lists into a single flat list.
    Parameters:
    list1 (list): First list to interleave.
    list2 (list): Second list to interleave.
    list3 (list): Third list to interleave.
    Returns:
    list: Interleaved flat list.
    '''
    # Initialize the interleaved list
    interleaved_list = []
    # Iterate through the lists and interleave their elements
    for i in range(len(list1)):
        interleaved_list.append(list1[i])
        interleaved_list.append(list2[i])
        interleaved_list.append(list3[i])
    return interleaved_list

'''
Function to interleave three lists of the same length into a single flat list.
'''
def interleave_three_lists(list1, list2, list3):
    '''
    Interleave three lists into a single flat list.
    Parameters:
    list1 (list): First list to interleave.
    list2 (list): Second list to interleave.
    list3 (list): Third list to interleave.
    Returns:
    list: Interleaved flat list.
    '''
    # Initialize the interleaved list
    interleaved_list = []
    # Iterate through the lists and interleave their elements
    for i in range(len(list1)):
        interleaved_list.append(list1[i])
        interleaved_list.append(list2[i])
        interleaved_list.append(list3[i])
    return interleaved_list
