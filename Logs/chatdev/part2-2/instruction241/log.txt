'''

This module contains the main application for calculating nonagonal numbers using a GUI.

'''

import tkinter as tk
from tkinter import messagebox

class NonagonalNumberCalculator:
    '''
    A class to calculate nonagonal numbers.
    '''
    
    @staticmethod
    def calculate_nth_nonagonal_number(n):
        '''
        Calculate the nth nonagonal number.
        
        Parameters:
        n (int): The position of the nonagonal number to calculate.
        
        Returns:
        int: The nth nonagonal number.
        '''
        return n * (7 * n - 5) // 2

class NonagonalNumberApp:
    '''
    A class to create a GUI application for nonagonal number calculation.
    '''
    
    def __init__(self, root):
        self.root = root
        self.root.title("Nonagonal Number Calculator")
        self.create_widgets()
    
    def create_widgets(self):
        '''
        Create the GUI components.
        '''
        self.label = tk.Label(self.root, text="Enter n:")
        self.label.pack()
        
        self.entry = tk.Entry(self.root)
        self.entry.pack()
        
        self.calculate_button = tk.Button(self.root, text="Calculate", command=self.calculate_and_display)
        self.calculate_button.pack()
        
        self.result_label = tk.Label(self.root, text="")
        self.result_label.pack()
    
    def calculate_and_display(self):
        '''
        Calculate the nonagonal number and display the result.
        '''
        try:
            n = int(self.entry.get())
            if n < 1:
                raise ValueError("n must be a positive integer.")
            result = NonagonalNumberCalculator.calculate_nth_nonagonal_number(n)
            self.result_label.config(text=f"The {n}th nonagonal number is: {result}")
        except ValueError as e:
            messagebox.showerror("Invalid Input", str(e))

import tkinter as tk
from tkinter import messagebox

class NonagonalNumberCalculator:
    '''
    A class to calculate nonagonal numbers.
    '''
    
    @staticmethod
    def calculate_nth_nonagonal_number(n):
        '''
        Calculate the nth nonagonal number.
        
        Parameters:
        n (int): The position of the nonagonal number to calculate.
        
        Returns:
        int: The nth nonagonal number.
        '''
        return n * (7 * n - 5) // 2

class NonagonalNumberApp:
    '''
    A class to create a GUI application for nonagonal number calculation.
    '''
    
    def __init__(self, root):
        self.root = root
        self.root.title("Nonagonal Number Calculator")
        self.create_widgets()
    
    def create_widgets(self):
        '''
        Create the GUI components.
        '''
        self.label = tk.Label(self.root, text="Enter n:")
        self.label.pack()
        
        self.entry = tk.Entry(self.root)
        self.entry.pack()
        
        self.calculate_button = tk.Button(self.root, text="Calculate", command=self.calculate_and_display)
        self.calculate_button.pack()
        
        self.result_label = tk.Label(self.root, text="")
        self.result_label.pack()
    
    def calculate_and_display(self):
        '''
        Calculate the nonagonal number and display the result.
        '''
        try:
            n = int(self.entry.get())
            if n < 1:
                raise ValueError("n must be a positive integer.")
            result = NonagonalNumberCalculator.calculate_nth_nonagonal_number(n)
            self.result_label.config(text=f"The {n}th nonagonal number is: {result}")
        except ValueError as e:
            messagebox.showerror("Invalid Input", str(e))

if __name__ == "__main__":
    root = tk.Tk()
    app = NonagonalNumberApp(root)
    root.mainloop()

# Nonagonal Number Calculator

A simple GUI-based application to calculate the nth nonagonal number.

## 📖 Introduction

The **Nonagonal Number Calculator** is a Python-based application that allows users to calculate the nth nonagonal number. Nonagonal numbers are figurate numbers that represent nonagons (nine-sided polygons). The nth nonagonal number can be calculated using the formula:

\[ N(n) = \frac{n \times (7n - 5)}{2} \]

This application provides a user-friendly graphical interface to input a number \( n \) and instantly calculate the corresponding nonagonal number.
