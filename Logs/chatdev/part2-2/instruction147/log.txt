class MainApp:
    def __init__(self, root):

        '''

        self.root.title("First Element Extractor")

        Processes the input from the user, extracts the first elements of each sublist, and displays the result.

            # Get the first elements
            result = get_first_elements(sublists)
            
            # Display the result
            self.output_field.delete("1.0", tk.END)
            self.output_field.insert(tk.END, str(result))
        except Exception as e:
            messagebox.showerror("Error", f"Invalid input: {e}")

if __name__ == "__main__":
    root = tk.Tk()

from tkinter import messagebox

class MainApp:
    def __init__(self, root):
        '''
        Initializes the GUI application.
        
        Parameters:
        root (tk.Tk): The root window of the tkinter application.
        '''
        self.root = root
        self.root.title("First Element Extractor")
        
        # Input Label
        self.input_label = tk.Label(root, text="Enter a list of sublists:")
        self.input_label.pack(pady=5)
        
        # Input Field
        self.input_field = tk.Entry(root, width=50)
        self.input_field.pack(pady=5)

        self.submit_button = tk.Button(root, text="Get First Elements", command=self.process_input)
        self.submit_button.pack(pady=10)
        
        # Output Label
        self.output_label = tk.Label(root, text="Result:")
        self.output_label.pack(pady=5)
        
        # Output Field
        self.output_field = tk.Text(root, height=5, width=50)
        self.output_field.pack(pady=5)

        Processes the input from the user, extracts the first elements of each sublist, and displays the result.
        '''

            # Display the result

        '''

        '''
        self.input_field.delete(0, tk.END)
        self.output_field.delete("1.0", tk.END)
if __name__ == "__main__":
    root = tk.Tk()
