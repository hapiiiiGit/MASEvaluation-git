This file contains the main application logic and GUI implementation for the task of removing words with a specified length from a given string.

import tkinter as tk
from tkinter import messagebox
from utils import remove_words_with_k_length


class MainApp:
    def __init__(self, root):
        '''
        Initializes the GUI components for the application.
        '''
        self.root = root
        self.root.title("Remove Words with Specified Length")
        self.root.geometry("400x300")

        # Input Label and Entry for String
        self.string_label = tk.Label(root, text="Enter a string:")
        self.string_label.pack(pady=5)
        self.string_entry = tk.Entry(root, width=50)
        self.string_entry.pack(pady=5)

        # Input Label and Entry for k
        self.k_label = tk.Label(root, text="Enter the length of words to remove (k):")
        self.k_label.pack(pady=5)
        self.k_entry = tk.Entry(root, width=10)
        self.k_entry.pack(pady=5)

        # Submit Button
        self.submit_button = tk.Button(root, text="Submit", command=self.on_submit)
        self.submit_button.pack(pady=10)

        # Output Label
        self.output_label = tk.Label(root, text="Result:", font=("Arial", 12))
        self.output_label.pack(pady=5)
        self.output_text = tk.Label(root, text="", wraplength=350, justify="left")
        self.output_text.pack(pady=5)

    def on_submit(self):
        '''
        Handles the user input and displays the result.
        '''
        input_string = self.string_entry.get()
        k_value = self.k_entry.get()

        # Validate inputs
        if not input_string.strip():
            messagebox.showerror("Error", "Please enter a valid string.")
            return

        try:
            k = int(k_value)
            if k <= 0:
                raise ValueError
        except ValueError:
            messagebox.showerror("Error", "Please enter a valid positive integer for k.")
            return

        # Process the string
        result = remove_words_with_k_length(input_string, k)
        self.output_text.config(text=result)


if __name__ == "__main__":
    root = tk.Tk()
    app = MainApp(root)
    root.mainloop()

'''

This file contains the main application logic and GUI implementation for the task of removing words with a specified length from a given string.

'''

import tkinter as tk
from tkinter import messagebox
from utils import remove_words_with_k_length


class MainApp:
    def __init__(self, root):
        '''
        Initializes the GUI components for the application.
        '''
        self.root = root
        self.root.title("Remove Words with Specified Length")
        self.root.geometry("400x300")

        # Input Label and Entry for String
        self.string_label = tk.Label(root, text="Enter a string:")
        self.string_label.pack(pady=5)
        self.string_entry = tk.Entry(root, width=50)
        self.string_entry.pack(pady=5)

        # Input Label and Entry for k
        self.k_label = tk.Label(root, text="Enter the length of words to remove (k):")
        self.k_label.pack(pady=5)
        self.k_entry = tk.Entry(root, width=10)
        self.k_entry.pack(pady=5)

        # Submit Button
        self.submit_button = tk.Button(root, text="Submit", command=self.on_submit)
        self.submit_button.pack(pady=10)

        # Output Label
        self.output_label = tk.Label(root, text="Result:", font=("Arial", 12))
        self.output_label.pack(pady=5)
        self.output_text = tk.Label(root, text="", wraplength=350, justify="left")
        self.output_text.pack(pady=5)

    def on_submit(self):
        '''
        Handles the user input and displays the result.
        '''
        input_string = self.string_entry.get()
        k_value = self.k_entry.get()

        # Validate inputs
        if not input_string.strip():
            messagebox.showerror("Error", "Please enter a valid string.")
            return

        try:
            k = int(k_value)
            if k <= 0:
                raise ValueError
        except ValueError:
            messagebox.showerror("Error", "Please enter a valid positive integer for k.")
            return

        # Process the string
        result = remove_words_with_k_length(input_string, k)
        self.output_text.config(text=result)


if __name__ == "__main__":
    root = tk.Tk()
    app = MainApp(root)
    root.mainloop()

        self.string_entry = tk.Entry(root, width=50)
        self.string_entry.pack(pady=5)
        # Input Label and Entry for k
        self.k_label = tk.Label(root, text="Enter the length of words to remove (k):")

        result = remove_words_with_k_length(input_string, k)
        self.output_text.config(text=result)

    root = tk.Tk()
    app = MainApp(root)
    root.mainloop()

'''
This file contains the utility function to remove words with a specified length from a given string.
'''
def remove_words_with_k_length(string, k):
    '''
    Removes all words with the specified length k from the given string.
    Args:
        string (str): The input string.
        k (int): The length of words to remove.
    Returns:
        str: The processed string with words of length k removed.
    '''
    words = string.split()
    filtered_words = [word for word in words if len(word) != k]
    return ' '.join(filtered_words)

This file contains the utility function to remove words with a specified length from a given string.
'''
def remove_words_with_k_length(string, k):
    '''
    Removes all words with the specified length k from the given string.
    Args:
        string (str): The input string.
        k (int): The length of words to remove.
    Returns:
        str: The processed string with words of length k removed.
    '''
    words = string.split()
    filtered_words = [word for word in words if len(word) != k]
    return ' '.join(filtered_words)
