'''

This file contains the main application class for the Name Length Calculator and the utility function to sum valid name lengths.

'''

import tkinter as tk

def sum_valid_name_lengths(names):
    '''
    Filters out names starting with a lowercase letter and returns the sum of the lengths of the remaining names.

    :param names: List of names (strings)
    :return: Sum of lengths of names not starting with a lowercase letter
    '''
    return sum(len(name) for name in names if not name[0].islower())

class NameLengthCalculatorApp:
    '''
    A tkinter-based GUI application to calculate the sum of the lengths of names after filtering.

    :param root: The root window of the tkinter application
    '''
    def __init__(self, root):
        self.root = root
        self.root.title("Name Length Calculator")

        self.label = tk.Label(root, text="Enter names separated by commas:")
        self.label.pack()

import tkinter as tk

def sum_valid_name_lengths(names):
    '''
    Filters out names starting with a lowercase letter and returns the sum of the lengths of the remaining names.
