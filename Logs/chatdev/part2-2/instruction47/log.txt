    '''
    Calculate the sum of even index binomial coefficients for a given positive integer n.

        n (int): The positive integer input.

        int: The sum of even index binomial coefficients.
    '''

        raise ValueError("Input must be a positive integer.")

    return sum(comb(n, k) for k in range(0, n + 1, 2))

'''
This file contains the main application code for calculating the sum of even index binomial coefficients using a GUI.
'''
import tkinter as tk
from tkinter import messagebox
from math import comb
def calculate_binomial_sum(n):
    '''
    Calculate the sum of even index binomial coefficients for a given positive integer n.
    Parameters:
        n (int): The positive integer input.
    Returns:
        int: The sum of even index binomial coefficients.
    '''
    if not isinstance(n, int) or n <= 0:
        raise ValueError("Input must be a positive integer.")
    # Sum even index binomial coefficients
    return sum(comb(n, k) for k in range(0, n + 1, 2))
class MainApp:
    '''
    Represents the main GUI application for calculating binomial coefficient sums.
    '''
    def __init__(self, root):
        '''
        Initialize the GUI application.
        Parameters:
            root (tk.Tk): The root window of the application.
        '''
        self.root = root
        self.root.title("Binomial Coefficient Sum Calculator")
        # Create GUI components
        self.label = tk.Label(root, text="Enter a positive integer (n):")
        self.label.pack(pady=10)
        self.entry = tk.Entry(root)
        self.entry.pack(pady=10)
        self.calculate_button = tk.Button(root, text="Calculate", command=self.calculate)
        self.calculate_button.pack(pady=10)
        self.result_label = tk.Label(root, text="")
        self.result_label.pack(pady=10)
    def calculate(self):
        '''
        Handle the calculation and display the result.
        '''
        try:
            # Get user input
            n = int(self.entry.get())
            # Validate input
            if n <= 0:
                raise ValueError
            # Calculate result
            result = calculate_binomial_sum(n)
            # Display result
            self.result_label.config(text=f"Sum of even index binomial coefficients: {result}")
        except ValueError:
            messagebox.showerror("Invalid Input", "Please enter a positive integer.")
def run_app():
    '''
    Entry point to run the application.
    '''
    root = tk.Tk()
    app = MainApp(root)
    root.mainloop()
# Run the application
if __name__ == "__main__":
    run_app()
