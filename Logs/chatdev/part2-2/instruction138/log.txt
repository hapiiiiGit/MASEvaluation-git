'''

import tkinter as tk
from tkinter import ttk

def remove_whitespace_from_string(input_string):
    """
    Removes all whitespaces from the given string.
    
    :param input_string: The string from which to remove whitespaces.
    :return: A new string with all whitespaces removed.
    """
    return ''.join(input_string.split())

class MainApp:
    """
    Main application class for the whitespace remover GUI.
    """
    def __init__(self, root):
        """
        Initializes the main application window and layout.
        
        :param root: The root window of the application.
        """
        self.root = root
        self.root.title("Whitespace Remover")

        # Input field
        self.input_label = ttk.Label(root, text="Enter your string:")
        self.input_label.pack(pady=5)

        self.input_entry = ttk.Entry(root, width=50)
        self.input_entry.pack(pady=5)

        # Output field
        self.output_label = ttk.Label(root, text="Result:")
        self.output_label.pack(pady=5)

        self.output_entry = ttk.Entry(root, width=50)
        self.output_entry.pack(pady=5)

        # Remove whitespace button
        self.remove_button = ttk.Button(root, text="Remove Whitespace", command=self.remove_whitespace)
        self.remove_button.pack(pady=20)

    def remove_whitespace(self):
        """
        Removes whitespaces from the input string and updates the output field.
        """
        input_text = self.input_entry.get()
        result = remove_whitespace_from_string(input_text)
        self.output_entry.delete(0, tk.END)
        self.output_entry.insert(0, result)

if __name__ == "__main__":
    root = tk.Tk()
    app = MainApp(root)
    root.mainloop()
```

'''
main.py
This script provides a GUI application to remove all whitespaces from a given string.

import tkinter as tk
from tkinter import ttk
def remove_whitespace_from_string(input_string):
    """
    Removes all whitespaces from the given string.
    :param input_string: The string from which to remove whitespaces.
    :return: A new string with all whitespaces removed.
    """
    return ''.join(input_string.split())
class MainApp:
    """
    Main application class for the whitespace remover GUI.
    """
    def __init__(self, root):
        """
        Initializes the main application window and layout.
        :param root: The root window of the application.
        """
        self.root = root
        self.root.title("Whitespace Remover")
        # Input field
        self.input_label = ttk.Label(root, text="Enter your string:")
        self.input_label.pack(pady=5)
        self.input_entry = ttk.Entry(root, width=50)
        self.input_entry.pack(pady=5)
        # Output field
        self.output_label = ttk.Label(root, text="Result:")
        self.output_label.pack(pady=5)
        self.output_entry = ttk.Entry(root, width=50)
        self.output_entry.pack(pady=5)
        # Remove whitespace button
        self.remove_button = ttk.Button(root, text="Remove Whitespace", command=self.remove_whitespace)
        self.remove_button.pack(pady=20)
    def remove_whitespace(self):
        """
        Removes whitespaces from the input string and updates the output field.
        """
        input_text = self.input_entry.get()
        result = remove_whitespace_from_string(input_text)
        self.output_entry.delete(0, tk.END)
        self.output_entry.insert(0, result)
if __name__ == "__main__":
    root = tk.Tk()
