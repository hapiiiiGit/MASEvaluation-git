'''

This module provides a function to sort a dictionary by its values.

'''

def sort_dict_by_value(dictionary):
    """
    Sorts a dictionary by its values.

    Parameters:
    dictionary (dict): The dictionary to be sorted.

    Returns:
    dict: A new dictionary sorted by values.
    """
    sorted_dict = dict(sorted(dictionary.items(), key=lambda item: item[1]))
    return sorted_dict
```

'''

This module creates a GUI application for sorting a dictionary by its values using Tkinter.

'''

import tkinter as tk
from tkinter import messagebox
from sort_dictionary import sort_dict_by_value

class DictionarySorterApp:
    def __init__(self, master):
        """
        Initializes the DictionarySorterApp with GUI components.

        Parameters:
        master (tk.Tk): The root window.
        """
        self.master = master
        master.title("Dictionary Sorter")

        self.label = tk.Label(master, text="Enter dictionary (e.g., {'a': 2, 'b': 1}):")
        self.label.pack()

        self.entry = tk.Entry(master, width=50)
        self.entry.pack()

        self.sort_button = tk.Button(master, text="Sort", command=self.sort_dictionary)
        self.sort_button.pack()

        self.result_label = tk.Label(master, text="")
        self.result_label.pack()

    def sort_dictionary(self):
        """
        Sorts the dictionary entered by the user and displays the result.
        """
        try:
            user_input = self.entry.get()
            dictionary = eval(user_input)
            if not isinstance(dictionary, dict):
                raise ValueError("Input is not a dictionary.")
            sorted_dict = sort_dict_by_value(dictionary)
            self.result_label.config(text=f"Sorted Dictionary: {sorted_dict}")
        except Exception as e:
            messagebox.showerror("Error", f"Invalid input: {e}")

if __name__ == "__main__":
    root = tk.Tk()
    app = DictionarySorterApp(root)
    root.mainloop()
```

def sort_dict_by_value(dictionary):
    """
    Sorts a dictionary by its values.
    Parameters:
    dictionary (dict): The dictionary to be sorted.
    Returns:
    dict: A new dictionary sorted by values.
    """
    sorted_dict = dict(sorted(dictionary.items(), key=lambda item: item[1]))
    return sorted_dict

def sort_dict_by_value(dictionary):
