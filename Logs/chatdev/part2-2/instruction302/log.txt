'''

This file contains the main application logic and GUI implementation for converting RGB color values to HSV color values.

'''

import tkinter as tk
from tkinter import messagebox

def rgb_to_hsv(r, g, b):
    '''
    Converts RGB color values to HSV color values.
    
    Parameters:
        r (int): Red component (0-255)
        g (int): Green component (0-255)
        b (int): Blue component (0-255)
    
    Returns:
        tuple: HSV values (h, s, v)
    '''
    r, g, b = r / 255.0, g / 255.0, b / 255.0
    max_val = max(r, g, b)
    min_val = min(r, g, b)
    delta = max_val - min_val

    # Calculate Hue
    if delta == 0:
        h = 0
    elif max_val == r:
        h = (60 * ((g - b) / delta) + 360) % 360
    elif max_val == g:
        h = (60 * ((b - r) / delta) + 120) % 360
    elif max_val == b:
        h = (60 * ((r - g) / delta) + 240) % 360

    # Calculate Saturation
    s = 0 if max_val == 0 else (delta / max_val)

    # Calculate Value
    v = max_val

    return round(h, 2), round(s * 100, 2), round(v * 100, 2)

'''

This file contains the main application logic and GUI implementation for converting RGB color values to HSV color values.

'''

import tkinter as tk
from tkinter import messagebox

def rgb_to_hsv(r, g, b):
    '''
    Converts RGB color values to HSV color values.
    
    Parameters:
        r (int): Red component (0-255)
        g (int): Green component (0-255)
        b (int): Blue component (0-255)
    
    Returns:
        tuple: HSV values (h, s, v)
    '''
    r, g, b = r / 255.0, g / 255.0, b / 255.0
    max_val = max(r, g, b)
    min_val = min(r, g, b)
    delta = max_val - min_val

    # Calculate Hue
    if delta == 0:
        h = 0
    elif max_val == r:
        h = (60 * ((g - b) / delta) + 360) % 360
    elif max_val == g:
        h = (60 * ((b - r) / delta) + 120) % 360
    elif max_val == b:
        h = (60 * ((r - g) / delta) + 240) % 360

    # Calculate Saturation
    s = 0 if max_val == 0 else (delta / max_val)

    # Calculate Value
    v = max_val

    return round(h, 2), round(s * 100, 2), round(v * 100, 2)

    A class representing the GUI application for RGB to HSV conversion.

        '''

            # Get RGB values from input fields

'''
This file contains the main application logic and GUI implementation for converting RGB color values to HSV color values.
'''
import tkinter as tk
from tkinter import messagebox
def rgb_to_hsv(r, g, b):
    '''
    Converts RGB color values to HSV color values.
    Parameters:
        r (int): Red component (0-255)
        g (int): Green component (0-255)
        b (int): Blue component (0-255)
    Returns:
        tuple: HSV values (h, s, v)
    '''
    r, g, b = r / 255.0, g / 255.0, b / 255.0
    max_val = max(r, g, b)
    min_val = min(r, g, b)
    delta = max_val - min_val
    # Calculate Hue
    if delta == 0:
        h = 0
    elif max_val == r:
        h = (60 * ((g - b) / delta) + 360) % 360
    elif max_val == g:
        h = (60 * ((b - r) / delta) + 120) % 360
    elif max_val == b:
        h = (60 * ((r - g) / delta) + 240) % 360
    # Calculate Saturation
    s = 0 if max_val == 0 else (delta / max_val)
    # Calculate Value
    v = max_val
    return round(h, 2), round(s * 100, 2), round(v * 100, 2)
class App:
    '''
    A class representing the GUI application for RGB to HSV conversion.
    '''
    def __init__(self, root):
        self.root = root
        self.root.title("RGB to HSV Converter")
        self.root.geometry("400x300")
        # Create input fields for RGB values
        tk.Label(root, text="Red (0-255):").grid(row=0, column=0, padx=10, pady=10)
        self.red_entry = tk.Entry(root)
        self.red_entry.grid(row=0, column=1, padx=10, pady=10)
        tk.Label(root, text="Green (0-255):").grid(row=1, column=0, padx=10, pady=10)
        self.green_entry = tk.Entry(root)
        self.green_entry.grid(row=1, column=1, padx=10, pady=10)
        tk.Label(root, text="Blue (0-255):").grid(row=2, column=0, padx=10, pady=10)
        self.blue_entry = tk.Entry(root)
        self.blue_entry.grid(row=2, column=1, padx=10, pady=10)
        # Create a button to perform the conversion
        self.convert_button = tk.Button(root, text="Convert", command=self.convert_rgb_to_hsv)
        self.convert_button.grid(row=3, column=0, columnspan=2, pady=20)
        # Create a label to display the result
        self.result_label = tk.Label(root, text="", font=("Arial", 12))
        self.result_label.grid(row=4, column=0, columnspan=2, pady=10)
    def convert_rgb_to_hsv(self):
        '''
        Handles the conversion process when the user clicks the "Convert" button.
        '''
        try:
            # Get RGB values from input fields
            r = int(self.red_entry.get())
            g = int(self.green_entry.get())
            b = int(self.blue_entry.get())
            # Validate input values
            if not (0 <= r <= 255 and 0 <= g <= 255 and 0 <= b <= 255):
                raise ValueError("RGB values must be between 0 and 255.")
            # Perform conversion
            h, s, v = rgb_to_hsv(r, g, b)
            # Display result
            self.result_label.config(text=f"HSV: ({h}, {s}%, {v}%)")
        except ValueError as e:
            messagebox.showerror("Invalid Input", str(e))
if __name__ == "__main__":
    root = tk.Tk()
    app = App(root)
    root.mainloop()

'''
This file contains the main application logic and GUI implementation for converting RGB color values to HSV color values.
'''
import tkinter as tk
from tkinter import messagebox
def rgb_to_hsv(r, g, b):
    '''
    Converts RGB color values to HSV color values.
    Parameters:
        r (int): Red component (0-255)
        g (int): Green component (0-255)
        b (int): Blue component (0-255)
    Returns:
        tuple: HSV values (h, s, v)
    '''
    r, g, b = r / 255.0, g / 255.0, b / 255.0
    max_val = max(r, g, b)
    min_val = min(r, g, b)
    delta = max_val - min_val
    # Calculate Hue
    if delta == 0:
        h = 0
    elif max_val == r:
        h = (60 * ((g - b) / delta) + 360) % 360
    elif max_val == g:
        h = (60 * ((b - r) / delta) + 120) % 360
    elif max_val == b:
        h = (60 * ((r - g) / delta) + 240) % 360
    # Calculate Saturation
    s = 0 if max_val == 0 else (delta / max_val)
    # Calculate Value
    v = max_val
    return round(h, 2), round(s * 100, 2), round(v * 100, 2)
class App:
    '''
    A class representing the GUI application for RGB to HSV conversion.
    '''
    def __init__(self, root):
        self.root = root
        self.root.title("RGB to HSV Converter")
        self.root.geometry("400x300")
        # Create input fields for RGB values
        tk.Label(root, text="Red (0-255):").grid(row=0, column=0, padx=10, pady=10)
        self.red_entry = tk.Entry(root)
        self.red_entry.grid(row=0, column=1, padx=10, pady=10)
        tk.Label(root, text="Green (0-255):").grid(row=1, column=0, padx=10, pady=10)
        self.green_entry = tk.Entry(root)
        self.green_entry.grid(row=1, column=1, padx=10, pady=10)
        tk.Label(root, text="Blue (0-255):").grid(row=2, column=0, padx=10, pady=10)
        self.blue_entry = tk.Entry(root)
        self.blue_entry.grid(row=2, column=1, padx=10, pady=10)
        # Create a button to perform the conversion
        self.convert_button = tk.Button(root, text="Convert", command=self.convert_rgb_to_hsv)
        self.convert_button.grid(row=3, column=0, columnspan=2, pady=20)
        # Create a label to display the result
        self.result_label = tk.Label(root, text="", font=("Arial", 12))
        self.result_label.grid(row=4, column=0, columnspan=2, pady=10)
    def convert_rgb_to_hsv(self):
        '''
        Handles the conversion process when the user clicks the "Convert" button.
        '''
        try:
            # Get RGB values from input fields
            r = int(self.red_entry.get())
            g = int(self.green_entry.get())
            b = int(self.blue_entry.get())
            # Validate input values
            if not (0 <= r <= 255 and 0 <= g <= 255 and 0 <= b <= 255):
                raise ValueError("RGB values must be between 0 and 255.")
            # Perform conversion
            h, s, v = rgb_to_hsv(r, g, b)
            # Display result
            self.result_label.config(text=f"HSV: ({h}, {s}%, {v}%)")
        except ValueError as e:
            messagebox.showerror("Invalid Input", str(e))
if __name__ == "__main__":
    root = tk.Tk()
    app = App(root)
    root.mainloop()

import tkinter as tk
from tkinter import messagebox

def rgb_to_hsv(r, g, b):
    '''
    Converts RGB color values to HSV color values.
    Parameters:
        r (int): Red component (0-255)
        g (int): Green component (0-255)
        b (int): Blue component (0-255)
    Returns:
        tuple: HSV values (h, s, v)
    '''
    r, g, b = r / 255.0, g / 255.0, b / 255.0
    max_val = max(r, g, b)
    min_val = min(r, g, b)
    delta = max_val - min_val
    
    # Calculate Hue
    if delta == 0:
        h = 0
    elif max_val == r:
        h = (60 * ((g - b) / delta) + 360) % 360
    elif max_val == g:
        h = (60 * ((b - r) / delta) + 120) % 360
    elif max_val == b:
        h = (60 * ((r - g) / delta) + 240) % 360
    
    # Calculate Saturation
    s = 0 if max_val == 0 else (delta / max_val)
    
    # Calculate Value
    v = max_val
    
    return round(h, 2), round(s * 100, 2), round(v * 100, 2)

'''
This file contains the main application logic and GUI implementation for converting RGB color values to HSV color values.
'''
import tkinter as tk
from tkinter import messagebox
def rgb_to_hsv(r, g, b):
    '''
    Converts RGB color values to HSV color values.
    Parameters:
        r (int): Red component (0-255)
        g (int): Green component (0-255)
        b (int): Blue component (0-255)
    Returns:
        tuple: HSV values (h, s, v)
    '''
    r, g, b = r / 255.0, g / 255.0, b / 255.0
    max_val = max(r, g, b)
    min_val = min(r, g, b)
    delta = max_val - min_val
    # Calculate Hue
    if delta == 0:
        h = 0
    elif max_val == r:
        h = (60 * ((g - b) / delta) + 360) % 360
    elif max_val == g:
        h = (60 * ((b - r) / delta) + 120) % 360
    elif max_val == b:
        h = (60 * ((r - g) / delta) + 240) % 360
    # Calculate Saturation
    s = 0 if max_val == 0 else (delta / max_val)
    # Calculate Value
    v = max_val
    return round(h, 2), round(s * 100, 2), round(v * 100, 2)
class App:
    '''
    A class representing the GUI application for RGB to HSV conversion.
    '''
    def __init__(self, root):
        self.root = root
        self.root.title("RGB to HSV Converter")
        self.root.geometry("400x300")
        # Create input fields for RGB values
        tk.Label(root, text="Red (0-255):").grid(row=0, column=0, padx=10, pady=10)
        self.red_entry = tk.Entry(root)
        self.red_entry.grid(row=0, column=1, padx=10, pady=10)
        tk.Label(root, text="Green (0-255):").grid(row=1, column=0, padx=10, pady=10)
        self.green_entry = tk.Entry(root)
        self.green_entry.grid(row=1, column=1, padx=10, pady=10)
        tk.Label(root, text="Blue (0-255):").grid(row=2, column=0, padx=10, pady=10)
        self.blue_entry = tk.Entry(root)
        self.blue_entry.grid(row=2, column=1, padx=10, pady=10)
        # Create a button to perform the conversion
        self.convert_button = tk.Button(root, text="Convert", command=self.convert_rgb_to_hsv)
        self.convert_button.grid(row=3, column=0, columnspan=2, pady=20)
        # Create a label to display the result
        self.result_label = tk.Label(root, text="", font=("Arial", 12))
        self.result_label.grid(row=4, column=0, columnspan=2, pady=10)
    def convert_rgb_to_hsv(self):
        '''
        Handles the conversion process when the user clicks the "Convert" button.
        '''
        try:
            # Get RGB values from input fields
            r = int(self.red_entry.get())
            g = int(self.green_entry.get())
            b = int(self.blue_entry.get())
            # Validate input values
            if not (0 <= r <= 255 and 0 <= g <= 255 and 0 <= b <= 255):
                raise ValueError("RGB values must be between 0 and 255.")
            # Perform conversion
            h, s, v = rgb_to_hsv(r, g, b)
            # Display result
            self.result_label.config(text=f"HSV: ({h}, {s}%, {v}%)")
        except ValueError as e:
            messagebox.showerror("Invalid Input", str(e))
if __name__ == "__main__":
    root = tk.Tk()
    app = App(root)
    root.mainloop()

'''
This file contains the main application logic and GUI implementation for converting RGB color values to HSV color values.
'''
import tkinter as tk
from tkinter import messagebox
def rgb_to_hsv(r, g, b):
    '''
    Converts RGB color values to HSV color values.
    Parameters:
        r (int): Red component (0-255)
        g (int): Green component (0-255)
        b (int): Blue component (0-255)
    Returns:
        tuple: HSV values (h, s, v)
    '''
    r, g, b = r / 255.0, g / 255.0, b / 255.0
    max_val = max(r, g, b)
    min_val = min(r, g, b)
    delta = max_val - min_val
    # Calculate Hue
    if delta == 0:
        h = 0
    elif max_val == r:
        h = (60 * ((g - b) / delta) + 360) % 360
    elif max_val == g:
        h = (60 * ((b - r) / delta) + 120) % 360
    elif max_val == b:
        h = (60 * ((r - g) / delta) + 240) % 360
    # Calculate Saturation
    s = 0 if max_val == 0 else (delta / max_val)
    # Calculate Value
    v = max_val
    return round(h, 2), round(s * 100, 2), round(v * 100, 2)
class App:
    '''
    A class representing the GUI application for RGB to HSV conversion.
    '''

        self.root = root
        self.root.title("RGB to HSV Converter")

        # Create input fields for RGB values

        self.red_entry = tk.Entry(root)
        self.red_entry.grid(row=0, column=1, padx=10, pady=10)
        tk.Label(root, text="Green (0-255):").grid(row=1, column=0, padx=10, pady=10)
        self.green_entry = tk.Entry(root)
        self.green_entry.grid(row=1, column=1, padx=10, pady=10)
        tk.Label(root, text="Blue (0-255):").grid(row=2, column=0, padx=10, pady=10)
        self.blue_entry = tk.Entry(root)
        self.blue_entry.grid(row=2, column=1, padx=10, pady=10)
        # Create a button to perform the conversion
        self.convert_button = tk.Button(root, text="Convert", command=self.convert_rgb_to_hsv)
        self.convert_button.grid(row=3, column=0, columnspan=2, pady=20)
        # Create a label to display the result

        self.result_label.grid(row=4, column=0, columnspan=2, pady=10)
    def convert_rgb_to_hsv(self):
        '''
        Handles the conversion process when the user clicks the "Convert" button.
        '''
        try:
            # Get RGB values from input fields
            r = int(self.red_entry.get())
            g = int(self.green_entry.get())
            b = int(self.blue_entry.get())
            # Validate input values
            if not (0 <= r <= 255 and 0 <= g <= 255 and 0 <= b <= 255):
                raise ValueError("RGB values must be between 0 and 255.")
            # Perform conversion
            h, s, v = rgb_to_hsv(r, g, b)

        except ValueError as e:
            messagebox.showerror("Invalid Input", str(e))

# RGB to HSV Converter

This application provides a simple graphical user interface (GUI) for converting RGB color values to HSV color values. It is designed to be user-friendly and requires no external dependencies.

## Quick Install

Since this project does not require any external dependencies, you can directly run the application using Python. Ensure you have Python installed on your system.

## 🤔 What is this?

The RGB to HSV Converter is a tool that allows users to input RGB (Red, Green, Blue) values and convert them into HSV (Hue, Saturation, Value) format. This conversion is often used in various fields such as digital imaging, computer graphics, and color science.

### Main Features

- **User-Friendly Interface**: The application provides an easy-to-use GUI for entering RGB values and viewing the converted HSV values.
- **Real-Time Conversion**: Users can instantly convert RGB values to HSV by clicking the "Convert" button.
- **Error Handling**: The application includes error handling to ensure that users input valid RGB values within the range of 0 to 255.

## 📖 How to Use

1. **Launch the Application**: Run the `main.py` file using Python. This will open the GUI window for the RGB to HSV Converter.

   ```bash
   python main.py
   ```

2. **Enter RGB Values**: In the application window, you will see three input fields labeled "Red (0-255)", "Green (0-255)", and "Blue (0-255)". Enter the desired RGB values in these fields.

3. **Convert to HSV**: Click the "Convert" button to perform the conversion. The application will display the corresponding HSV values in the format `(Hue, Saturation%, Value%)`.

4. **Error Messages**: If you enter invalid RGB values (outside the range of 0 to 255), the application will display an error message prompting you to correct the input.

## Example

- **Input**: Red = 255, Green = 0, Blue = 0
- **Output**: HSV: (0.0, 100.0%, 100.0%)

This example demonstrates converting the RGB value for pure red into its HSV equivalent.

## Additional Information

- **Hue (H)**: Represents the color type and is measured in degrees (0-360).
- **Saturation (S)**: Represents the intensity of the color and is expressed as a percentage (0-100%).
- **Value (V)**: Represents the brightness of the color and is also expressed as a percentage (0-100%).

This application is a simple yet effective tool for anyone needing to convert RGB values to HSV, whether for educational purposes, design work, or software development. Enjoy using the RGB to HSV Converter!
