'''

main.py

'''

import tkinter as tk
from tkinter import messagebox

class SubstringCounterApp:
    '''
    A class to create a GUI application for counting non-empty substrings of a given string.
    '''

    def __init__(self, root):
        '''
        Initialize the application with GUI components.
        '''
        self.root = root
        self.root.title("Substring Counter")

        # Create and place the input label and entry
        self.label = tk.Label(root, text="Enter a string:")
        self.label.pack(pady=10)

        self.entry = tk.Entry(root, width=50)
        self.entry.pack(pady=10)

        # Create and place the count button
        self.count_button = tk.Button(root, text="Count Substrings", command=self.count_substrings)
        self.count_button.pack(pady=10)

        # Create and place the result label
        self.result_label = tk.Label(root, text="")
        self.result_label.pack(pady=10)

    def count_substrings(self):
        '''
        Calculate the number of non-empty substrings and update the result label.
        '''
        input_string = self.entry.get()
        if input_string:
            count = self.calculate_substrings(input_string)
            self.result_label.config(text=f"Number of non-empty substrings: {count}")
        else:
            messagebox.showwarning("Input Error", "Please enter a valid string.")

    @staticmethod
    def calculate_substrings(s):
        '''
        Calculate the number of non-empty substrings of a given string.

        :param s: The input string.
        :return: The number of non-empty substrings.
        '''
        n = len(s)
        return n * (n + 1) // 2

if __name__ == "__main__":
    root = tk.Tk()
    app = SubstringCounterApp(root)
    root.mainloop()
```

- **`calculate_substrings` Static Method**: This method computes the number of non-empty substrings using the formula \( n \times (n + 1) / 2 \), where \( n \) is the length of the string.

import tkinter as tk
from tkinter import messagebox

class SubstringCounterApp:
    '''
    A class to create a GUI application for counting non-empty substrings of a given string.
    '''

    def __init__(self, root):
        '''
        Initialize the application with GUI components.
        '''
        self.root = root
        self.root.title("Substring Counter")

        # Create and place the input label and entry
        self.label = tk.Label(root, text="Enter a string:")
        self.label.pack(pady=10)

        self.entry = tk.Entry(root, width=50)
        self.entry.pack(pady=10)

        # Create and place the count button
        self.count_button = tk.Button(root, text="Count Substrings", command=self.count_substrings)
        self.count_button.pack(pady=10)

        # Create and place the result label
        self.result_label = tk.Label(root, text="")
        self.result_label.pack(pady=10)

    def count_substrings(self):
        '''
        Calculate the number of non-empty substrings and update the result label.
        '''
        input_string = self.entry.get()
        if input_string:
            count = self.calculate_substrings(input_string)
            self.result_label.config(text=f"Number of non-empty substrings: {count}")
        else:
            messagebox.showwarning("Input Error", "Please enter a valid string.")

    @staticmethod
    def calculate_substrings(s):
        '''
        Calculate the number of non-empty substrings of a given string.

        :param s: The input string.
        :return: The number of non-empty substrings.
        '''
        n = len(s)
        return n * (n + 1) // 2

if __name__ == "__main__":
    root = tk.Tk()
    app = SubstringCounterApp(root)
    root.mainloop()
```

- **`count_substrings` Method**: This method retrieves the input string, calculates the number of non-empty substrings using the `calculate_substrings` method, and updates the result label.
- **`calculate_substrings` Static Method**: This method computes the number of non-empty substrings using the formula \( n \times (n + 1) / 2 \), where \( n \) is the length of the string.

class SubstringCounterApp:

import tkinter as tk
from tkinter import messagebox
class SubstringCounterApp:
    '''
    A class to create a GUI application for counting non-empty substrings of a given string.
    '''
    def __init__(self, root):
        '''
        Initialize the application with GUI components.
        '''
        self.root = root
        self.root.title("Substring Counter")
        # Create and place the input label and entry
        self.label = tk.Label(root, text="Enter a string:")
        self.label.pack(pady=10)
        self.entry = tk.Entry(root, width=50)
        self.entry.pack(pady=10)
        # Create and place the count button
        self.count_button = tk.Button(root, text="Count Substrings", command=self.count_substrings)
        self.count_button.pack(pady=10)
        # Create and place the result label
        self.result_label = tk.Label(root, text="")
        self.result_label.pack(pady=10)
    def count_substrings(self):
        '''
        Calculate the number of non-empty substrings and update the result label.
        '''
        input_string = self.entry.get()
        if input_string:
            count = self.calculate_substrings(input_string)
            self.result_label.config(text=f"Number of non-empty substrings: {count}")
        else:
            messagebox.showwarning("Input Error", "Please enter a valid string.")
    @staticmethod
    def calculate_substrings(s):
        '''
        Calculate the number of non-empty substrings of a given string.
        :param s: The input string.
        :return: The number of non-empty substrings.
        '''
        n = len(s)
        return n * (n + 1) // 2

import tkinter as tk
from tkinter import messagebox

class SubstringCounterApp:

    A class to create a GUI application for counting non-empty substrings of a given string.
    '''

        Initialize the application with GUI components.

        self.result_label = tk.Label(root, text="")
        self.result_label.pack(pady=10)

import tkinter as tk
from tkinter import messagebox
class SubstringCounterApp:
    '''
    A class to create a GUI application for counting non-empty substrings of a given string.
    '''
    def __init__(self, root):
        '''
        Initialize the application with GUI components.
        '''
        self.root = root
        self.root.title("Substring Counter")
        # Create and place the input label and entry
        self.label = tk.Label(root, text="Enter a string:")
        self.label.pack(pady=10)
        self.entry = tk.Entry(root, width=50)
        self.entry.pack(pady=10)
        # Create and place the count button
        self.count_button = tk.Button(root, text="Count Substrings", command=self.count_substrings)
        self.count_button.pack(pady=10)
        # Create and place the result label
        self.result_label = tk.Label(root, text="")
        self.result_label.pack(pady=10)
    def count_substrings(self):
        '''
        Calculate the number of non-empty substrings and update the result label.
        '''
        input_string = self.entry.get()
        if input_string:
            count = self.calculate_substrings(input_string)
            self.result_label.config(text=f"Number of non-empty substrings: {count}")
        else:
            messagebox.showwarning("Input Error", "Please enter a valid string.")
    @staticmethod
    def calculate_substrings(s):
        '''
        Calculate the number of non-empty substrings of a given string.
        :param s: The input string.
        :return: The number of non-empty substrings.
        '''
        n = len(s)
        return n * (n + 1) // 2
if __name__ == "__main__":
    root = tk.Tk()
    app = SubstringCounterApp(root)
    root.mainloop()

- **Instant Calculation**: With a click of a button, the application calculates and displays the number of non-empty substrings.
