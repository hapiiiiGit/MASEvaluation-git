'''

This file contains the GUI implementation for the square root calculator using the Babylonian method.
It integrates the Babylonian method from the `babylonian.py` file.

'''

import tkinter as tk
from tkinter import messagebox
from babylonian import BabylonianMethod

class SquareRootApp:
    def __init__(self, root):
        self.root = root
        self.root.title("Square Root Calculator")
        self.root.geometry("400x300")
        
        # Input label and entry
        self.label = tk.Label(root, text="Enter a number:", font=("Arial", 12))
        self.label.pack(pady=10)
        
        self.entry = tk.Entry(root, font=("Arial", 12))
        self.entry.pack(pady=10)
        
        # Tolerance label and entry
        self.tolerance_label = tk.Label(root, text="Enter tolerance (e.g., 0.0001):", font=("Arial", 12))
        self.tolerance_label.pack(pady=10)
        
        self.tolerance_entry = tk.Entry(root, font=("Arial", 12))
        self.tolerance_entry.pack(pady=10)
        
        # Compute button
        self.compute_button = tk.Button(root, text="Compute Square Root", command=self.compute_square_root, font=("Arial", 12))
        self.compute_button.pack(pady=20)
        
        # Output label
        self.result_label = tk.Label(root, text="", font=("Arial", 12))
        self.result_label.pack(pady=10)
    
    def compute_square_root(self):
        try:
            number = float(self.entry.get())
            tolerance = float(self.tolerance_entry.get())
            
            if number < 0:
                raise ValueError("Number must be non-negative.")
            
            babylonian = BabylonianMethod()
            result = babylonian.calculate_square_root(number, tolerance)
            self.result_label.config(text=f"Square Root: {result:.6f}")

class BabylonianMethod:
    def calculate_square_root(self, number, tolerance=0.0001):
        '''
        Computes the square root of a given number using the Babylonian method.
        
        Parameters:
            number (float): The number to compute the square root of.
            tolerance (float): The tolerance for the approximation.
        
        Returns:
            float: The approximated square root of the number.

'''
This file contains the implementation of the Babylonian method for calculating square roots.
'''
class BabylonianMethod:
    def calculate_square_root(self, number, tolerance=0.0001):
        '''
        Computes the square root of a given number using the Babylonian method.
        Parameters:
            number (float): The number to compute the square root of.
            tolerance (float): The tolerance for the approximation.
        Returns:
            float: The approximated square root of the number.
        '''
        if number < 0:
            raise ValueError("Cannot compute square root of a negative number.")
        # Initial guess
        guess = number / 2.0
        while True:
            # Compute the next guess
            next_guess = (guess + number / guess) / 2.0
            # Check if the difference is within the tolerance
            if abs(next_guess - guess) < tolerance:
                return next_guess
            guess = next_guess

'''
This file contains the implementation of the Babylonian method for calculating square roots.
'''
class BabylonianMethod:
    def calculate_square_root(self, number, tolerance=0.0001):
        '''
        Computes the square root of a given number using the Babylonian method.
        Parameters:
            number (float): The number to compute the square root of.
            tolerance (float): The tolerance for the approximation.
        Returns:
            float: The approximated square root of the number.
        '''
        if number < 0:
            raise ValueError("Cannot compute square root of a negative number.")
        # Initial guess
        guess = number / 2.0
        while True:
            # Compute the next guess
            next_guess = (guess + number / guess) / 2.0
            # Check if the difference is within the tolerance
            if abs(next_guess - guess) < tolerance:
                return next_guess
            guess = next_guess

'''
This file contains the implementation of the Babylonian method for calculating square roots.
'''
class BabylonianMethod:
    def calculate_square_root(self, number, tolerance=0.0001):
        '''
        Computes the square root of a given number using the Babylonian method.
        Parameters:
            number (float): The number to compute the square root of.
            tolerance (float): The tolerance for the approximation.
        Returns:
            float: The approximated square root of the number.
        '''
        if number < 0:
            raise ValueError("Cannot compute square root of a negative number.")
        # Initial guess
        guess = number / 2.0
        while True:
            # Compute the next guess
            next_guess = (guess + number / guess) / 2.0
            # Check if the difference is within the tolerance
            if abs(next_guess - guess) < tolerance:
                return next_guess
            guess = next_guess

from tkinter import messagebox

    def __init__(self, root):
        '''

        # Tolerance label and entry

'''

class BabylonianMethod:
    def calculate_square_root(self, number, tolerance=0.0001):
        '''
        Computes the square root of a given number using the Babylonian method.
        Parameters:
            number (float): The number to compute the square root of.
            tolerance (float): The tolerance for the approximation.
        Returns:
            float: The approximated square root of the number.
        '''
        if number < 0:
            raise ValueError("Cannot compute square root of a negative number.")
        # Initial guess
        guess = number / 2.0
        while True:
            # Compute the next guess
            next_guess = (guess + number / guess) / 2.0
            # Check if the difference is within the tolerance
            if abs(next_guess - guess) < tolerance:
                return next_guess
            guess = next_guess

'''
This file contains the implementation of the Babylonian method for calculating square roots.
'''
class BabylonianMethod:
    def calculate_square_root(self, number, tolerance=0.0001):
        '''
        Computes the square root of a given number using the Babylonian method.
        Parameters:
            number (float): The number to compute the square root of.
            tolerance (float): The tolerance for the approximation.
        Returns:
            float: The approximated square root of the number.
        '''
        if number < 0:
            raise ValueError("Cannot compute square root of a negative number.")
        # Initial guess
        guess = number / 2.0
        while True:
            # Compute the next guess
            next_guess = (guess + number / guess) / 2.0
            # Check if the difference is within the tolerance
            if abs(next_guess - guess) < tolerance:
                return next_guess
            guess = next_guess

import tkinter as tk
from tkinter import messagebox
from babylonian import BabylonianMethod
class SquareRootApp:
    def __init__(self, root):
        '''
        Initializes the SquareRootApp GUI application.
        Parameters:
            root (tk.Tk): The root window for the application.
        '''
        self.root = root
        self.root.title("Square Root Calculator")
        self.root.geometry("400x300")
        # Input label and entry
        self.label = tk.Label(root, text="Enter a number:", font=("Arial", 12))
        self.label.pack(pady=10)
        self.entry = tk.Entry(root, font=("Arial", 12))
        self.entry.pack(pady=10)
        # Tolerance label and entry
        self.tolerance_label = tk.Label(root, text="Enter tolerance (e.g., 0.0001):", font=("Arial", 12))
        self.tolerance_label.pack(pady=10)
        self.tolerance_entry = tk.Entry(root, font=("Arial", 12))
        self.tolerance_entry.pack(pady=10)
        # Compute button
        self.compute_button = tk.Button(root, text="Compute Square Root", command=self.compute_square_root, font=("Arial", 12))
        self.compute_button.pack(pady=20)
        # Output label
        self.result_label = tk.Label(root, text="", font=("Arial", 12))
        self.result_label.pack(pady=10)
    def compute_square_root(self):
        '''
        Computes the square root of the input number using the Babylonian method.

        try:
            number = float(self.entry.get())

            messagebox.showerror("Error", f"Invalid input: {e}")
        except Exception as e:
            messagebox.showerror("Error", f"An error occurred: {e}")

    app = SquareRootApp(root)
    root.mainloop()

'''
This file contains the implementation of the Babylonian method for calculating square roots.
'''
class BabylonianMethod:
    def calculate_square_root(self, number, tolerance=0.0001):
        '''
        Computes the square root of a given number using the Babylonian method.
        Parameters:
            number (float): The number to compute the square root of.
            tolerance (float): The tolerance for the approximation.
        Returns:
            float: The approximated square root of the number.
        '''
        if number < 0:
            raise ValueError("Cannot compute square root of a negative number.")
        # Initial guess
        guess = number / 2.0
        while True:
            # Compute the next guess
            next_guess = (guess + number / guess) / 2.0
            # Check if the difference is within the tolerance
            if abs(next_guess - guess) < tolerance:
                return next_guess
            guess = next_guess

## About the Babylonian Method

The Babylonian method, also known as Heron's method, is an iterative algorithm for approximating square roots. It starts with an initial guess and refines the guess using the formula:
```
next_guess = (guess + number / guess) / 2
```
The process repeats until the difference between successive guesses is less than the specified tolerance.
