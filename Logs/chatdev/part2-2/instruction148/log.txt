'''

This is the main file for the application. It initializes the GUI and handles user interactions.

'''

import tkinter as tk
from utils import count_uppercase_characters


class MainApp:
    def __init__(self, root):
        '''
        Initializes the GUI components.
        '''
        self.root = root
        self.root.title("Uppercase Character Counter")

        # Input Label
        self.label = tk.Label(root, text="Enter a string:")
        self.label.pack(pady=10)

        # Input Entry
        self.input_entry = tk.Entry(root, width=50)
        self.input_entry.pack(pady=10)

        # Count Button
        self.count_button = tk.Button(root, text="Count Uppercase Characters", command=self.count_uppercase)
        self.count_button.pack(pady=10)

        # Result Label
        self.result_label = tk.Label(root, text="", font=("Arial", 12))
        self.result_label.pack(pady=10)

    def count_uppercase(self):
        '''
        Reads the input string, counts the uppercase characters, and displays the result.
        '''
        input_string = self.input_entry.get()
        count = count_uppercase_characters(input_string)
        self.result_label.config(text=f"Number of uppercase characters: {count}")


if __name__ == "__main__":
    root = tk.Tk()
    app = MainApp(root)
    root.mainloop()
```

#### utils.py

```python
'''

This file contains utility functions for the application.

'''

def count_uppercase_characters(string):
    '''
    Counts the number of uppercase characters in a given string.

    Parameters:
        string (str): The input string.

    Returns:
        int: The count of uppercase characters in the string.
    '''
    return sum(1 for char in string if char.isupper())
```

'''

This is the main file for the application. It initializes the GUI and handles user interactions.

'''

import tkinter as tk
from utils import count_uppercase_characters


class MainApp:
    def __init__(self, root):
        '''
        Initializes the GUI components.
        '''
        self.root = root
        self.root.title("Uppercase Character Counter")

        # Input Label
        self.label = tk.Label(root, text="Enter a string:")
        self.label.pack(pady=10)

        # Input Entry
        self.input_entry = tk.Entry(root, width=50)
        self.input_entry.pack(pady=10)

        # Count Button
        self.count_button = tk.Button(root, text="Count Uppercase Characters", command=self.count_uppercase)
        self.count_button.pack(pady=10)

        # Result Label
        self.result_label = tk.Label(root, text="", font=("Arial", 12))
        self.result_label.pack(pady=10)

    def count_uppercase(self):
        '''
        Reads the input string, counts the uppercase characters, and displays the result.
        '''
        input_string = self.input_entry.get()
        count = count_uppercase_characters(input_string)
        self.result_label.config(text=f"Number of uppercase characters: {count}")


if __name__ == "__main__":
    root = tk.Tk()
    app = MainApp(root)
    root.mainloop()
```

def count_uppercase_characters(string):

    Counts the number of uppercase characters in a given string.

    Parameters:

    Returns:
        int: The count of uppercase characters in the string.
    '''
    return sum(1 for char in string if char.isupper())
