        self.entry.pack()

    def calculate_averages(self):

            tuples = eval(self.entry.get())
            averages = calculate_tuple_averages(tuples)
            self.result_label.config(text=f"Averages: {averages}")
        except Exception as e:
            self.result_label.config(text=f"Error: {str(e)}")

if __name__ == "__main__":
    root = tk.Tk()
    app = MainApplication(root)
    root.mainloop()

#### calculations.py

```python
'''

Function to calculate the average of each tuple in a tuple of tuples.

'''

def calculate_tuple_averages(tuples):
    averages = []
    for t in tuples:
        if len(t) > 0:
            avg = sum(t) / len(t)
            averages.append(avg)
        else:
            averages.append(0)
    return averages
```

def calculate_tuple_averages(tuples):

            averages.append(0)

                averages = calculate_tuple_averages(tuples)
                self.result_label.config(text=f"Averages: {averages}")
            else:
                self.result_label.config(text="Error: Input must be a tuple of tuples.")
        except Exception as e:

if __name__ == "__main__":

def calculate_tuple_averages(tuples):

        if len(t) > 0:
            avg = sum(t) / len(t)
            averages.append(avg)
        else:
            averages.append(0)
    return averages
