```python
'''

This file contains the main application class for the GUI and the utility function to calculate the sum of divisors.

'''

import tkinter as tk
from tkinter import messagebox

def calculate_sum_of_divisors(n):
    '''
    Calculate the sum of all divisors of a given number n.
    
    :param n: Integer, the number to find divisors of.
    :return: Integer, sum of all divisors of n.
    '''
    if n < 1:
        return 0
    total = 0
    for i in range(1, n + 1):
        if n % i == 0:
            total += i
    return total

class DivisorSumApp:
    '''
    A class to create a GUI application for calculating the sum of divisors of a number.
    '''
    
    def __init__(self, root):
        '''
        Initialize the application with the main window and widgets.
        
        :param root: The root window object.
        '''
        self.root = root
        self.root.title("Sum of Divisors Calculator")
        
        # Create and place widgets
        self.label = tk.Label(root, text="Enter a number:")
        self.label.pack(pady=10)
        
        self.entry = tk.Entry(root)
        self.entry.pack(pady=5)
        
        self.calculate_button = tk.Button(root, text="Calculate", command=self.calculate_divisors_sum)
        self.calculate_button.pack(pady=10)
        
        self.result_label = tk.Label(root, text="Sum of Divisors: ")
        self.result_label.pack(pady=10)
    
    def calculate_divisors_sum(self):
        '''
        Calculate the sum of divisors of the number entered by the user and display the result.
        '''
        try:
            number = int(self.entry.get())
            if number < 1:
                raise ValueError("The number must be a positive integer.")
            sum_of_divisors = calculate_sum_of_divisors(number)
            self.result_label.config(text=f"Sum of Divisors: {sum_of_divisors}")
        except ValueError as e:
            messagebox.showerror("Invalid Input", str(e))

if __name__ == "__main__":
    root = tk.Tk()
    app = DivisorSumApp(root)
    root.mainloop()
```

It includes an entry field for the user to input a number, a button to trigger the calculation, and a label to display the result. The method `calculate_divisors_sum` handles the logic for retrieving the input, calculating the sum of divisors, and updating the result label.

```python
'''

This file contains the main application class for the GUI and the utility function to calculate the sum of divisors.

'''

import tkinter as tk
from tkinter import messagebox

def calculate_sum_of_divisors(n):
    '''
    Calculate the sum of all divisors of a given number n.
    
    :param n: Integer, the number to find divisors of.
    :return: Integer, sum of all divisors of n.
    '''
    if n < 1:
        return 0
    total = 0
    for i in range(1, n + 1):
        if n % i == 0:
            total += i
    return total

class DivisorSumApp:
    '''
    A class to create a GUI application for calculating the sum of divisors of a number.
    '''
    
    def __init__(self, root):
        '''
        Initialize the application with the main window and widgets.
        
        :param root: The root window object.
        '''
        self.root = root
        self.root.title("Sum of Divisors Calculator")
        
        # Create and place widgets
        self.label = tk.Label(root, text="Enter a number:")
        self.label.pack(pady=10)
        
        self.entry = tk.Entry(root)
        self.entry.pack(pady=5)
        
        self.calculate_button = tk.Button(root, text="Calculate", command=self.calculate_divisors_sum)
        self.calculate_button.pack(pady=10)
        
        self.result_label = tk.Label(root, text="Sum of Divisors: ")
        self.result_label.pack(pady=10)
    
    def calculate_divisors_sum(self):
        '''
        Calculate the sum of divisors of the number entered by the user and display the result.
        '''
        try:
            number = int(self.entry.get())
            if number < 1:
                raise ValueError("The number must be a positive integer.")
            sum_of_divisors = calculate_sum_of_divisors(number)
            self.result_label.config(text=f"Sum of Divisors: {sum_of_divisors}")
        except ValueError as e:
            messagebox.showerror("Invalid Input", str(e))

if __name__ == "__main__":
    root = tk.Tk()
    app = DivisorSumApp(root)
    root.mainloop()

'''
This file contains the main application class for the GUI and the utility function to calculate the sum of divisors.
'''
import tkinter as tk
from tkinter import messagebox
def calculate_sum_of_divisors(n):
    '''
    Calculate the sum of all divisors of a given number n.
    :param n: Integer, the number to find divisors of.
    :return: Integer, sum of all divisors of n.
    '''
    if n < 1:
        return 0
    total = 0
    for i in range(1, n + 1):
        if n % i == 0:
            total += i
    return total
class DivisorSumApp:
    '''
    A class to create a GUI application for calculating the sum of divisors of a number.
    '''
    def __init__(self, root):
        '''
        Initialize the application with the main window and widgets.
        :param root: The root window object.
        '''
        self.root = root
        self.root.title("Sum of Divisors Calculator")
        # Create and place widgets
        self.label = tk.Label(root, text="Enter a number:")
        self.label.pack(pady=10)
        self.entry = tk.Entry(root)
        self.entry.pack(pady=5)
        self.calculate_button = tk.Button(root, text="Calculate", command=self.calculate_divisors_sum)
        self.calculate_button.pack(pady=10)
        self.result_label = tk.Label(root, text="Sum of Divisors: ")
        self.result_label.pack(pady=10)
    def calculate_divisors_sum(self):
        '''
        Calculate the sum of divisors of the number entered by the user and display the result.
        '''
        try:
            number = int(self.entry.get())
            if number < 1:
                raise ValueError("The number must be a positive integer.")
            sum_of_divisors = calculate_sum_of_divisors(number)
            self.result_label.config(text=f"Sum of Divisors: {sum_of_divisors}")
        except ValueError as e:
            messagebox.showerror("Invalid Input", str(e))
if __name__ == "__main__":
    root = tk.Tk()
    app = DivisorSumApp(root)
    root.mainloop()

'''

import tkinter as tk
from tkinter import messagebox
def calculate_sum_of_divisors(n):
    '''
    Calculate the sum of all divisors of a given number n.
    :param n: Integer, the number to find divisors of.
    :return: Integer, sum of all divisors of n.
    '''
    if n < 1:
        return 0
    total = 0
    for i in range(1, n + 1):
        if n % i == 0:
            total += i
    return total
class DivisorSumApp:
    '''
    A class to create a GUI application for calculating the sum of divisors of a number.
    '''
    def __init__(self, root):
        '''
        Initialize the application with the main window and widgets.
        :param root: The root window object.
        '''
        self.root = root
        self.root.title("Sum of Divisors Calculator")
        # Create and place widgets
        self.label = tk.Label(root, text="Enter a number:")
        self.label.pack(pady=10)
        self.entry = tk.Entry(root)
        self.entry.pack(pady=5)
        self.calculate_button = tk.Button(root, text="Calculate", command=self.calculate_divisors_sum)
        self.calculate_button.pack(pady=10)
        self.result_label = tk.Label(root, text="Sum of Divisors: ")
        self.result_label.pack(pady=10)
    def calculate_divisors_sum(self):
        '''
        Calculate the sum of divisors of the number entered by the user and display the result.
        '''
        try:
            number = int(self.entry.get())
            if number < 1:
                raise ValueError("The number must be a positive integer.")
            sum_of_divisors = calculate_sum_of_divisors(number)
            self.result_label.config(text=f"Sum of Divisors: {sum_of_divisors}")
        except ValueError as e:
            messagebox.showerror("Invalid Input", str(e))
if __name__ == "__main__":
    root = tk.Tk()
    app = DivisorSumApp(root)
    root.mainloop()

import tkinter as tk

def calculate_sum_of_divisors(n):
    '''
