```python
'''

This file contains the main application class for the GUI and the helper function to find the first repeated character in a string.

'''

import tkinter as tk
from tkinter import messagebox

def find_first_repeated_character(s):
    '''
    Finds the first repeated character in a given string.
    
    Parameters:
    s (str): The input string to search for repeated characters.
    
    Returns:
    str: The first repeated character or a message if no repetition is found.
    '''
    seen = set()
    for char in s:
        if char in seen:
            return char
        seen.add(char)
    return "No repeated characters found."

class FirstRepeatedCharApp:
    '''
    A class to create a GUI application for finding the first repeated character in a string.
    '''
    def __init__(self, master):
        '''
        Initializes the main window and widgets.
        
        Parameters:
        master (tk.Tk): The root window of the application.
        '''
        self.master = master
        master.title("First Repeated Character Finder")
        
        self.label = tk.Label(master, text="Enter a string:")
        self.label.pack()
        
        self.entry = tk.Entry(master)
        self.entry.pack()
        
        self.find_button = tk.Button(master, text="Find", command=self.find_first_repeated_char)
        self.find_button.pack()
        
        self.result_label = tk.Label(master, text="")
        self.result_label.pack()

    def find_first_repeated_char(self):
        '''
        Finds the first repeated character in the input string and displays the result.
        '''
        input_string = self.entry.get()
        result = find_first_repeated_character(input_string)
        self.result_label.config(text=f"Result: {result}")
        messagebox.showinfo("Result", f"First Repeated Character: {result}")

if __name__ == "__main__":
    root = tk.Tk()
    app = FirstRepeatedCharApp(root)
    root.mainloop()
```

```python
'''

This file contains the main application class for the GUI and the helper function to find the first repeated character in a string.

'''

import tkinter as tk
from tkinter import messagebox

def find_first_repeated_character(s):
    '''
    Finds the first repeated character in a given string.
    
    Parameters:
    s (str): The input string to search for repeated characters.
    
    Returns:
    str: The first repeated character or a message if no repetition is found.
    '''
    seen = set()
    for char in s:
        if char in seen:
            return char
        seen.add(char)
    return "No repeated characters found."

class FirstRepeatedCharApp:
    '''
    A class to create a GUI application for finding the first repeated character in a string.
    '''
    def __init__(self, master):
        '''
        Initializes the main window and widgets.
        
        Parameters:
        master (tk.Tk): The root window of the application.
        '''
        self.master = master
        master.title("First Repeated Character Finder")
        
        self.label = tk.Label(master, text="Enter a string:")
        self.label.pack()
        
        self.entry = tk.Entry(master)
        self.entry.pack()
        
        self.find_button = tk.Button(master, text="Find", command=self.find_first_repeated_char)
        self.find_button.pack()
        
        self.result_label = tk.Label(master, text="")
        self.result_label.pack()

    def find_first_repeated_char(self):
        '''
        Finds the first repeated character in the input string and displays the result.
        '''
        input_string = self.entry.get()
        result = find_first_repeated_character(input_string)
        self.result_label.config(text=f"Result: {result}")
        messagebox.showinfo("Result", f"First Repeated Character: {result}")

if __name__ == "__main__":
    root = tk.Tk()
    app = FirstRepeatedCharApp(root)
    root.mainloop()
```

'''
This file contains the main application class for the GUI and the helper function to find the first repeated character in a string.
'''
import tkinter as tk
from tkinter import messagebox
def find_first_repeated_character(s):
    '''
    Finds the first repeated character in a given string.
    Parameters:
    s (str): The input string to search for repeated characters.
    Returns:
    str: The first repeated character or a message if no repetition is found.
    '''
    seen = set()
    for char in s:
        if char in seen:
            return char
        seen.add(char)
    return "No repeated characters found."
class FirstRepeatedCharApp:
    '''
    A class to create a GUI application for finding the first repeated character in a string.
    '''
    def __init__(self, master):
        '''
        Initializes the main window and widgets.
        Parameters:
        master (tk.Tk): The root window of the application.
        '''
        self.master = master
        master.title("First Repeated Character Finder")
        self.label = tk.Label(master, text="Enter a string:")
        self.label.pack()
        self.entry = tk.Entry(master)
        self.entry.pack()
        self.find_button = tk.Button(master, text="Find", command=self.find_first_repeated_char)
        self.find_button.pack()
        self.result_label = tk.Label(master, text="")
        self.result_label.pack()
    def find_first_repeated_char(self):
        '''
        Finds the first repeated character in the input string and displays the result.
        '''
        input_string = self.entry.get()
        result = find_first_repeated_character(input_string)
        self.result_label.config(text=f"Result: {result}")
        messagebox.showinfo("Result", f"First Repeated Character: {result}")
if __name__ == "__main__":
    root = tk.Tk()
    app = FirstRepeatedCharApp(root)
    root.mainloop()
