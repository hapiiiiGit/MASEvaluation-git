'''
Main application file for finding the maximum of two numbers using a GUI.
'''

import tkinter as tk
from tkinter import messagebox

def find_maximum(num1, num2):
    '''
    Function to find the maximum of two numbers.
    
    Parameters:
    num1 (float): The first number.
    num2 (float): The second number.
    
    Returns:
    float: The maximum of the two numbers.
    '''
    return max(num1, num2)

class MaxFinderApp:
    '''
    A class to create a GUI application for finding the maximum of two numbers.
    '''
    
    def __init__(self, root):
        '''
        Initialize the application with the main window and widgets.
        
        Parameters:
        root (tk.Tk): The root window of the application.
        '''
        self.root = root
        self.root.title("Max Finder")
        
        self.create_widgets()
    
    def create_widgets(self):
        '''
        Create and place the widgets in the application window.
        '''
        self.label1 = tk.Label(self.root, text="Enter first number:")
        self.label1.grid(row=0, column=0, padx=10, pady=10)
        
        self.entry1 = tk.Entry(self.root)
        self.entry1.grid(row=0, column=1, padx=10, pady=10)
        
        self.label2 = tk.Label(self.root, text="Enter second number:")
        self.label2.grid(row=1, column=0, padx=10, pady=10)
        
        self.entry2 = tk.Entry(self.root)
        self.entry2.grid(row=1, column=1, padx=10, pady=10)
        
        self.find_button = tk.Button(self.root, text="Find Max", command=self.find_max)
        self.find_button.grid(row=2, column=0, columnspan=2, pady=10)
    
    def find_max(self):
        '''
        Retrieve numbers from entries, find the maximum, and display the result.
        '''
        try:
            num1 = float(self.entry1.get())
            num2 = float(self.entry2.get())
            maximum = find_maximum(num1, num2)
            messagebox.showinfo("Result", f"The maximum of {num1} and {num2} is {maximum}.")
        except ValueError:
            messagebox.showerror("Invalid input", "Please enter valid numbers.")

if __name__ == "__main__":
    root = tk.Tk()
    app = MaxFinderApp(root)
    root.mainloop()
```

This code sets up a simple GUI application using `tkinter` where users can input two numbers, click a button, and see the maximum of the two numbers displayed in a message box. The `find_maximum` function is responsible for the core logic of finding the maximum number.

'''
Main application file for finding the maximum of two numbers using a GUI.
'''

import tkinter as tk
from tkinter import messagebox

def find_maximum(num1, num2):
    '''
    Function to find the maximum of two numbers.
    
    Parameters:
    num1 (float): The first number.
    num2 (float): The second number.
    
    Returns:
    float: The maximum of the two numbers.
    '''
    return max(num1, num2)

class MaxFinderApp:
    '''
    A class to create a GUI application for finding the maximum of two numbers.
    '''
    
    def __init__(self, root):
        '''
        Initialize the application with the main window and widgets.
        
        Parameters:
        root (tk.Tk): The root window of the application.
        '''
        self.root = root
        self.root.title("Max Finder")
        
        self.create_widgets()
    
    def create_widgets(self):
        '''
        Create and place the widgets in the application window.
        '''
        self.label1 = tk.Label(self.root, text="Enter first number:")
        self.label1.grid(row=0, column=0, padx=10, pady=10)
        
        self.entry1 = tk.Entry(self.root)
        self.entry1.grid(row=0, column=1, padx=10, pady=10)
        
        self.label2 = tk.Label(self.root, text="Enter second number:")
        self.label2.grid(row=1, column=0, padx=10, pady=10)
        
        self.entry2 = tk.Entry(self.root)
        self.entry2.grid(row=1, column=1, padx=10, pady=10)
        
        self.find_button = tk.Button(self.root, text="Find Max", command=self.find_max)
        self.find_button.grid(row=2, column=0, columnspan=2, pady=10)
    
    def find_max(self):
        '''
        Retrieve numbers from entries, find the maximum, and display the result.
        '''
        try:
            num1 = float(self.entry1.get())
            num2 = float(self.entry2.get())
            maximum = find_maximum(num1, num2)
            messagebox.showinfo("Result", f"The maximum of {num1} and {num2} is {maximum}.")
        except ValueError:
            messagebox.showerror("Invalid input", "Please enter valid numbers.")

if __name__ == "__main__":
    root = tk.Tk()
    app = MaxFinderApp(root)
    root.mainloop()

import tkinter as tk
from tkinter import messagebox
def find_maximum(num1, num2):
    '''
    Function to find the maximum of two numbers.
    Parameters:
    num1 (float): The first number.
    num2 (float): The second number.
    Returns:
    float: The maximum of the two numbers.
    '''
    return max(num1, num2)
class MaxFinderApp:
    '''
    A class to create a GUI application for finding the maximum of two numbers.
    '''
    def __init__(self, root):
        '''
        Initialize the application with the main window and widgets.
        Parameters:
        root (tk.Tk): The root window of the application.
        '''
        self.root = root
        self.root.title("Max Finder")
        self.create_widgets()
    def create_widgets(self):
        '''
        Create and place the widgets in the application window.
        '''
        self.label1 = tk.Label(self.root, text="Enter first number:")
        self.label1.grid(row=0, column=0, padx=10, pady=10)
        self.entry1 = tk.Entry(self.root)
        self.entry1.grid(row=0, column=1, padx=10, pady=10)
        self.label2 = tk.Label(self.root, text="Enter second number:")
        self.label2.grid(row=1, column=0, padx=10, pady=10)

        self.entry2.grid(row=1, column=1, padx=10, pady=10)

    def find_max(self):
        '''
        Retrieve numbers from entries, find the maximum, and display the result.

            num1 = float(self.entry1.get())

    root = tk.Tk()

'''
Main application file for finding the maximum of two numbers using a GUI.
'''
import tkinter as tk
from tkinter import messagebox
def find_maximum(num1, num2):
    '''
    Function to find the maximum of two numbers.
    Parameters:
    num1 (float): The first number.
    num2 (float): The second number.
    Returns:
    float: The maximum of the two numbers.
    '''
    return max(num1, num2)
class MaxFinderApp:
    '''
    A class to create a GUI application for finding the maximum of two numbers.
    '''
    def __init__(self, root):
        '''
        Initialize the application with the main window and widgets.
        Parameters:
        root (tk.Tk): The root window of the application.
        '''
        self.root = root
        self.root.title("Max Finder")
        self.create_widgets()
    def create_widgets(self):
        '''
        Create and place the widgets in the application window.
        '''
        self.label1 = tk.Label(self.root, text="Enter first number:")
        self.label1.grid(row=0, column=0, padx=10, pady=10)
        self.entry1 = tk.Entry(self.root)
        self.entry1.grid(row=0, column=1, padx=10, pady=10)
        self.label2 = tk.Label(self.root, text="Enter second number:")
        self.label2.grid(row=1, column=0, padx=10, pady=10)
        self.entry2 = tk.Entry(self.root)
        self.entry2.grid(row=1, column=1, padx=10, pady=10)
        self.find_button = tk.Button(self.root, text="Find Max", command=self.find_max)
        self.find_button.grid(row=2, column=0, columnspan=2, pady=10)
    def find_max(self):
        '''
        Retrieve numbers from entries, find the maximum, and display the result.
        '''
        try:
            num1 = float(self.entry1.get())
            num2 = float(self.entry2.get())
            maximum = find_maximum(num1, num2)
            messagebox.showinfo("Result", f"The maximum of {num1} and {num2} is {maximum}.")
        except ValueError:
            messagebox.showerror("Invalid input", "Please enter valid numbers.")
if __name__ == "__main__":
    root = tk.Tk()
    app = MaxFinderApp(root)
    root.mainloop()
