'''

This is the main file to run the XOR sum calculator application with a GUI.

'''

import tkinter as tk
from calculator import XorSumCalculator

class XorSumApp:
    def __init__(self):
        self.calculator = XorSumCalculator()
        self.root = tk.Tk()
        self.root.title("XOR Sum Calculator")

        self.label = tk.Label(self.root, text="Enter numbers separated by commas:")
        self.label.pack()

        self.entry = tk.Entry(self.root, width=50)
        self.entry.pack()

        self.calculate_button = tk.Button(self.root, text="Calculate XOR Sum", command=self.calculate_and_display)
        self.calculate_button.pack()

        self.result_label = tk.Label(self.root, text="")
        self.result_label.pack()

    def calculate_and_display(self):
        input_text = self.entry.get()
        try:
            numbers = list(map(int, input_text.split(',')))
            xor_sum = self.calculator.calculate_xor_sum(numbers)
            self.result_label.config(text=f"Sum of XOR of all pairs: {xor_sum}")
        except ValueError:
            self.result_label.config(text="Invalid input. Please enter integers separated by commas.")

    def run(self):
        self.root.mainloop()

if __name__ == "__main__":
    app = XorSumApp()
    app.run()

This file contains the XorSumCalculator class which calculates the sum of XOR of all pairs in a list.

'''

from typing import List

class XorSumCalculator:
    def calculate_xor_sum(self, numbers: List[int]) -> int:
        total_xor_sum = 0
        n = len(numbers)
        for i in range(n):
            for j in range(i + 1, n):
                total_xor_sum += numbers[i] ^ numbers[j]
        return total_xor_sum
```

'''

This is the main file to run the XOR sum calculator application with a GUI.

'''

import tkinter as tk
from calculator import XorSumCalculator

class XorSumApp:
    def __init__(self):
        self.calculator = XorSumCalculator()
        self.root = tk.Tk()
        self.root.title("XOR Sum Calculator")

        self.label = tk.Label(self.root, text="Enter numbers separated by commas:")
        self.label.pack()

        self.entry = tk.Entry(self.root, width=50)
        self.entry.pack()

        self.calculate_button = tk.Button(self.root, text="Calculate XOR Sum", command=self.calculate_and_display)
        self.calculate_button.pack()

        self.result_label = tk.Label(self.root, text="")
        self.result_label.pack()

    def calculate_and_display(self):
        input_text = self.entry.get()
        try:
            numbers = list(map(int, input_text.split(',')))
            xor_sum = self.calculator.calculate_xor_sum(numbers)
            self.result_label.config(text=f"Sum of XOR of all pairs: {xor_sum}")
        except ValueError:
            self.result_label.config(text="Invalid input. Please enter integers separated by commas.")

    def run(self):
        self.root.mainloop()

if __name__ == "__main__":
    app = XorSumApp()
    app.run()

from typing import List
class XorSumCalculator:
    def calculate_xor_sum(self, numbers: List[int]) -> int:
        total_xor_sum = 0
        n = len(numbers)
        for i in range(n):
            for j in range(i + 1, n):
                total_xor_sum += numbers[i] ^ numbers[j]
        return total_xor_sum
