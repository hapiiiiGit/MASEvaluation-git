'''
This is the main file for the application. It sets up the GUI and handles user interactions.
'''

import tkinter as tk
from tkinter import messagebox

def count_matching_indices(list1, list2):
    '''
    Count the number of indices where the two lists have the same values.
    
    Parameters:
    list1 (list): The first list of values.
    list2 (list): The second list of values.
    
    Returns:
    int: The count of indices where the lists have the same values.
    '''
    return sum(1 for x, y in zip(list1, list2) if x == y)

'''

import tkinter as tk
from tkinter import messagebox
def count_matching_indices(list1, list2):
    '''
    Count the number of indices where the two lists have the same values.
    Parameters:
    list1 (list): The first list of values.
    list2 (list): The second list of values.
    Returns:
    int: The count of indices where the lists have the same values.
    '''
    return sum(1 for x, y in zip(list1, list2) if x == y)
class App:
    def __init__(self, root):
        '''
        Initialize the application with the main window and widgets.
        Parameters:
        root (tk.Tk): The root window of the application.
        '''
        self.root = root
        self.root.title("List Comparison Tool")
        # Labels
        self.label1 = tk.Label(root, text="Enter first list (comma-separated):")
        self.label1.pack()
        self.entry1 = tk.Entry(root, width=50)
        self.entry1.pack()
        self.label2 = tk.Label(root, text="Enter second list (comma-separated):")
        self.label2.pack()
        self.entry2 = tk.Entry(root, width=50)
        self.entry2.pack()
        # Submit Button
        self.submit_button = tk.Button(root, text="Submit", command=self.on_submit)
        self.submit_button.pack()
        # Result Label
        self.result_label = tk.Label(root, text="")
        self.result_label.pack()
    def on_submit(self):
        '''
        Handle the submit button click event, process the input lists, and display the result.
        '''
        try:
            list1 = list(map(str.strip, self.entry1.get().split(',')))
            list2 = list(map(str.strip, self.entry2.get().split(',')))
            if len(list1) != len(list2):
                messagebox.showerror("Error", "Both lists must be of the same length.")
                return
            count = count_matching_indices(list1, list2)
            self.result_label.config(text=f"Matching indices count: {count}")
        except Exception as e:
            messagebox.showerror("Error", f"An error occurred: {e}")
if __name__ == "__main__":
    root = tk.Tk()
    app = App(root)
    root.mainloop()

'''
This is the main file for the application. It sets up the GUI and handles user interactions.
'''
import tkinter as tk
from tkinter import messagebox
def count_matching_indices(list1, list2):
    '''
    Count the number of indices where the two lists have the same values.
    Parameters:
    list1 (list): The first list of values.
    list2 (list): The second list of values.
    Returns:
    int: The count of indices where the lists have the same values.
    '''
    return sum(1 for x, y in zip(list1, list2) if x == y)
class App:
    def __init__(self, root):
        '''
        Initialize the application with the main window and widgets.
        Parameters:
        root (tk.Tk): The root window of the application.
        '''
        self.root = root
        self.root.title("List Comparison Tool")
        # Labels
        self.label1 = tk.Label(root, text="Enter first list (comma-separated):")
        self.label1.pack()
        self.entry1 = tk.Entry(root, width=50)
        self.entry1.pack()
        self.label2 = tk.Label(root, text="Enter second list (comma-separated):")
        self.label2.pack()
        self.entry2 = tk.Entry(root, width=50)
        self.entry2.pack()
        # Submit Button
        self.submit_button = tk.Button(root, text="Submit", command=self.on_submit)
        self.submit_button.pack()
        # Result Label
        self.result_label = tk.Label(root, text="")
        self.result_label.pack()
    def on_submit(self):
        '''
        Handle the submit button click event, process the input lists, and display the result.
        '''
        try:
            list1 = list(map(str.strip, self.entry1.get().split(',')))
            list2 = list(map(str.strip, self.entry2.get().split(',')))
            if len(list1) != len(list2):
                messagebox.showerror("Error", "Both lists must be of the same length.")
                return
            count = count_matching_indices(list1, list2)
            self.result_label.config(text=f"Matching indices count: {count}")
        except Exception as e:
            messagebox.showerror("Error", f"An error occurred: {e}")
if __name__ == "__main__":
    root = tk.Tk()
    app = App(root)
    root.mainloop()
