'''

This Python program calculates the volume of a cube given its side length. It provides a graphical user interface (GUI) using the tkinter library, allowing users to input the side length and view the calculated volume.

'''

import tkinter as tk
from tkinter import messagebox


def calculate_volume(side_length):
    '''
    Calculate the volume of a cube given its side length.

    Parameters:
        side_length (float): The length of a side of the cube.

    Returns:
        float: The volume of the cube.
    '''
    return side_length ** 3


def on_calculate():
    '''
    Handle the button click event to calculate the cube's volume.

    Retrieves the side length from the input field, calculates the volume,
    and displays the result in the output label.
    '''
    try:
        # Get the side length from the input field
        side_length = float(entry_side_length.get())
        
        if side_length <= 0:
            raise ValueError("Side length must be positive.")
        
        # Calculate the volume
        volume = calculate_volume(side_length)
        
        # Display the result
        label_result.config(text=f"Volume: {volume:.2f}")
    except ValueError as e:
        # Show an error message if the input is invalid
        messagebox.showerror("Invalid Input", f"Error: {e}")


# Create the main application window
root = tk.Tk()
root.title("Cube Volume Calculator")

# Create and place the input field for side length
label_prompt = tk.Label(root, text="Enter the side length of the cube:")
label_prompt.pack()

entry_side_length = tk.Entry(root)
entry_side_length.pack()

# Create and place the calculate button
button_calculate = tk.Button(root, text="Calculate Volume", command=on_calculate)
button_calculate.pack()

# Create and place the output label for the result
label_result = tk.Label(root, text="Volume: ")
label_result.pack()

# Run the application
root.mainloop()

```

from tkinter import messagebox


def calculate_volume(side_length):
    '''
    Calculate the volume of a cube given its side length.

    Parameters:
        side_length (float): The length of a side of the cube.

    Returns:
        float: The volume of the cube.
    '''
