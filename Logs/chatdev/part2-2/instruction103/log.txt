'''

Main application file for extracting nth element from a list of tuples using a GUI.

'''

import tkinter as tk
from tkinter import messagebox
from utils import extract_nth_element

class MainApp:
    def __init__(self, root):
        self.root = root
        self.root.title("Tuple Element Extractor")

        # Input for list of tuples
        self.tuple_list_label = tk.Label(root, text="Enter list of tuples:")
        self.tuple_list_label.pack()
        self.tuple_list_entry = tk.Entry(root, width=50)
        self.tuple_list_entry.pack()

        # Input for index
        self.index_label = tk.Label(root, text="Enter index to extract:")
        self.index_label.pack()
        self.index_entry = tk.Entry(root, width=10)
        self.index_entry.pack()

        # Button to extract elements
        self.extract_button = tk.Button(root, text="Extract", command=self.extract_elements)
        self.extract_button.pack()

        # Output area
        self.output_label = tk.Label(root, text="Extracted elements:")
        self.output_label.pack()
        self.output_text = tk.Text(root, height=10, width=50)
        self.output_text.pack()

    def extract_elements(self):
        try:
            tuple_list = eval(self.tuple_list_entry.get())
            index = int(self.index_entry.get())
            result = extract_nth_element(tuple_list, index)
            self.output_text.delete(1.0, tk.END)
            self.output_text.insert(tk.END, str(result))
        except Exception as e:
            messagebox.showerror("Error", f"An error occurred: {e}")

if __name__ == "__main__":
    root = tk.Tk()
    app = MainApp(root)
    root.mainloop()
```

#### utils.py

```python
'''

Utility functions for tuple operations.

'''

def extract_nth_element(tuple_list, n):
    """
    Extracts the nth element from each tuple in the list.

    :param tuple_list: List of tuples
    :param n: Index of the element to extract
    :return: List of nth elements
    """
    return [t[n] for t in tuple_list if len(t) > n]
```

'''

Main application file for extracting nth element from a list of tuples using a GUI.

'''

import tkinter as tk
from tkinter import messagebox
from utils import extract_nth_element

class MainApp:
    def __init__(self, root):
        self.root = root
        self.root.title("Tuple Element Extractor")

        # Input for list of tuples
        self.tuple_list_label = tk.Label(root, text="Enter list of tuples:")
        self.tuple_list_label.pack()
        self.tuple_list_entry = tk.Entry(root, width=50)
        self.tuple_list_entry.pack()

        # Input for index
        self.index_label = tk.Label(root, text="Enter index to extract:")
        self.index_label.pack()
        self.index_entry = tk.Entry(root, width=10)
        self.index_entry.pack()

        # Button to extract elements
        self.extract_button = tk.Button(root, text="Extract", command=self.extract_elements)
        self.extract_button.pack()

        # Output area
        self.output_label = tk.Label(root, text="Extracted elements:")
        self.output_label.pack()
        self.output_text = tk.Text(root, height=10, width=50)
        self.output_text.pack()

    def extract_elements(self):
        try:
            tuple_list = eval(self.tuple_list_entry.get())
            index = int(self.index_entry.get())
            result = extract_nth_element(tuple_list, index)
            self.output_text.delete(1.0, tk.END)
            self.output_text.insert(tk.END, str(result))
        except Exception as e:
            messagebox.showerror("Error", f"An error occurred: {e}")

if __name__ == "__main__":
    root = tk.Tk()
    app = MainApp(root)
    root.mainloop()

    """
    Extracts the nth element from each tuple in the list.

    :param tuple_list: List of tuples
    :param n: Index of the element to extract
    :return: List of nth elements

'''
Main application file for extracting nth element from a list of tuples using a GUI.

import tkinter as tk
from tkinter import messagebox
import ast
from utils import extract_nth_element
class MainApp:
    def __init__(self, root):
        self.root = root
        self.root.title("Tuple Element Extractor")
        # Input for list of tuples
        self.tuple_list_label = tk.Label(root, text="Enter list of tuples:")
        self.tuple_list_label.pack()
        self.tuple_list_entry = tk.Entry(root, width=50)
        self.tuple_list_entry.pack()
        # Input for index
        self.index_label = tk.Label(root, text="Enter index to extract:")
        self.index_label.pack()
        self.index_entry = tk.Entry(root, width=10)
        self.index_entry.pack()
        # Button to extract elements
        self.extract_button = tk.Button(root, text="Extract", command=self.extract_elements)
        self.extract_button.pack()
        # Output area
        self.output_label = tk.Label(root, text="Extracted elements:")
        self.output_label.pack()
        self.output_text = tk.Text(root, height=10, width=50)
        self.output_text.pack()
    def extract_elements(self):
        try:
            tuple_list = ast.literal_eval(self.tuple_list_entry.get())
            index = int(self.index_entry.get())
            result = extract_nth_element(tuple_list, index)
            self.output_text.delete(1.0, tk.END)
            self.output_text.insert(tk.END, str(result))
        except Exception as e:
            messagebox.showerror("Error", f"An error occurred: {e}")
if __name__ == "__main__":
    root = tk.Tk()
    app = MainApp(root)
    root.mainloop()

def extract_nth_element(tuple_list, n):
    """
    Extracts the nth element from each tuple in the list.
    :param tuple_list: List of tuples
    :param n: Index of the element to extract
    :return: List of nth elements
    """
    return [t[n] for t in tuple_list if len(t) > n]

    def __init__(self, root):
        self.root = root
        self.root.title("Tuple Element Extractor")
        # Input for list of tuples
        self.tuple_list_label = tk.Label(root, text="Enter list of tuples:")
        self.tuple_list_label.pack()
        self.tuple_list_entry = tk.Entry(root, width=50)
        self.tuple_list_entry.pack()
        # Input for index
        self.index_label = tk.Label(root, text="Enter index to extract:")
        self.index_label.pack()
        self.index_entry = tk.Entry(root, width=10)

            if not isinstance(tuple_list, list) or not all(isinstance(t, tuple) for t in tuple_list):

            index_input = self.index_entry.get()
            if not index_input.isdigit():
                raise ValueError("Index must be a non-negative integer.")
