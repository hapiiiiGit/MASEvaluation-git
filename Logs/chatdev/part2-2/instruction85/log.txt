```python
'''

This file contains the main application logic for the GUI-based program to find the list with the maximum length.

'''

import tkinter as tk
from tkinter import messagebox
from helper import find_max_length


class MainApp:
    def __init__(self, root):
        '''
        Initializes the GUI application.
        '''
        self.root = root
        self.root.title("Find List with Maximum Length")
        self.root.geometry("400x300")

        # List to store user-input lists
        self.lists = []

        # GUI Components
        self.label = tk.Label(root, text="Enter a list (comma-separated):")
        self.label.pack(pady=10)

        self.entry = tk.Entry(root, width=40)
        self.entry.pack(pady=5)

        self.add_button = tk.Button(root, text="Add List", command=self.add_list)
        self.add_button.pack(pady=5)

        self.find_button = tk.Button(root, text="Find Max Length List", command=self.find_max_length_list)
        self.find_button.pack(pady=5)

        self.result_label = tk.Label(root, text="", fg="blue")
        self.result_label.pack(pady=10)

    def add_list(self):
        '''
        Adds a user-input list to the list of lists.
        '''
        input_text = self.entry.get()
        try:
            # Convert the input text into a list
            user_list = [int(x.strip()) for x in input_text.split(",")]
            self.lists.append(user_list)
            self.entry.delete(0, tk.END)
            messagebox.showinfo("Success", "List added successfully!")
        except ValueError:
            messagebox.showerror("Error", "Invalid input! Please enter a comma-separated list of numbers.")

    def find_max_length_list(self):
        '''
        Finds and displays the list with the maximum length.
        '''
        if not self.lists:
            messagebox.showerror("Error", "No lists available! Please add some lists first.")
            return

        max_list = find_max_length(self.lists)
        self.result_label.config(text=f"Max Length List: {max_list}")


if __name__ == "__main__":
    root = tk.Tk()
    app = MainApp(root)
    root.mainloop()
```

### File 2: `helper.py`

This file contains the helper function to find the list with the maximum length.

```python
'''

This file contains helper functions for the application.

'''


def find_max_length(lst_of_lists):
    '''
    Finds the list with the maximum length from a list of lists.

    Args:
        lst_of_lists (list): A list containing multiple lists.

    Returns:
        list: The list with the maximum length.
    '''
    if not lst_of_lists:
        return []

    # Use max() with the key argument to find the list with the maximum length
    return max(lst_of_lists, key=len)
```

'''

This file contains the main application logic for the GUI-based program to find the list with the maximum length.

'''

import tkinter as tk
from tkinter import messagebox
from helper import find_max_length


class MainApp:
    def __init__(self, root):
        '''
        Initializes the GUI application.
        '''
        self.root = root
        self.root.title("Find List with Maximum Length")
        self.root.geometry("400x300")

        # List to store user-input lists
        self.lists = []

        # GUI Components
        self.label = tk.Label(root, text="Enter a list (comma-separated):")
        self.label.pack(pady=10)

        self.entry = tk.Entry(root, width=40)
        self.entry.pack(pady=5)

        self.add_button = tk.Button(root, text="Add List", command=self.add_list)
        self.add_button.pack(pady=5)

        self.find_button = tk.Button(root, text="Find Max Length List", command=self.find_max_length_list)
        self.find_button.pack(pady=5)

        self.result_label = tk.Label(root, text="", fg="blue")
        self.result_label.pack(pady=10)

    def add_list(self):
        '''
        Adds a user-input list to the list of lists.
        '''
        input_text = self.entry.get()
        try:
            # Convert the input text into a list
            user_list = [int(x.strip()) for x in input_text.split(",")]
            self.lists.append(user_list)
            self.entry.delete(0, tk.END)
            messagebox.showinfo("Success", "List added successfully!")
        except ValueError:
            messagebox.showerror("Error", "Invalid input! Please enter a comma-separated list of numbers.")

    def find_max_length_list(self):
        '''
        Finds and displays the list with the maximum length.
        '''
        if not self.lists:
            messagebox.showerror("Error", "No lists available! Please add some lists first.")
            return

        max_list = find_max_length(self.lists)
        self.result_label.config(text=f"Max Length List: {max_list}")


if __name__ == "__main__":
    root = tk.Tk()
    app = MainApp(root)
    root.mainloop()

import tkinter as tk
from tkinter import messagebox
from helper import find_max_length
class MainApp:
    def __init__(self, root):
        '''
        Initializes the GUI application.
        '''
        self.root = root
        self.root.title("Find List with Maximum Length")
        self.root.geometry("400x300")
        # List to store user-input lists
        self.lists = []
        # GUI Components
        self.label = tk.Label(root, text="Enter a list (comma-separated):")
        self.label.pack(pady=10)
        self.entry = tk.Entry(root, width=40)
        self.entry.pack(pady=5)
        self.add_button = tk.Button(root, text="Add List", command=self.add_list)
        self.add_button.pack(pady=5)
        self.find_button = tk.Button(root, text="Find Max Length List", command=self.find_max_length_list)
        self.find_button.pack(pady=5)
        self.result_label = tk.Label(root, text="", fg="blue")
        self.result_label.pack(pady=10)
    def add_list(self):
        '''
        Adds a user-input list to the list of lists.
        '''
        input_text = self.entry.get()
        try:
            # Convert the input text into a list
            user_list = [int(x.strip()) for x in input_text.split(",")]
            self.lists.append(user_list)
            self.entry.delete(0, tk.END)
            messagebox.showinfo("Success", "List added successfully!")
        except ValueError:
            messagebox.showerror("Error", "Invalid input! Please enter a comma-separated list of numbers.")
    def find_max_length_list(self):
        '''
        Finds and displays the list with the maximum length.
        '''
        if not self.lists:
            messagebox.showerror("Error", "No lists available! Please add some lists first.")
            return
        max_list = find_max_length(self.lists)
        self.result_label.config(text=f"Max Length List: {max_list}")
if __name__ == "__main__":
    root = tk.Tk()
    app = MainApp(root)
    root.mainloop()

'''
This file contains the main application logic for the GUI-based program to find the list with the maximum length.

import tkinter as tk
from tkinter import messagebox
from helper import find_max_length
class MainApp:
    def __init__(self, root):
        '''
        Initializes the GUI application.
        '''
        self.root = root
        self.root.title("Find List with Maximum Length")
        self.root.geometry("400x300")
        # List to store user-input lists
        self.lists = []
        # GUI Components
        self.label = tk.Label(root, text="Enter a list (comma-separated):")
        self.label.pack(pady=10)
        self.entry = tk.Entry(root, width=40)
        self.entry.pack(pady=5)
        self.add_button = tk.Button(root, text="Add List", command=self.add_list)
        self.add_button.pack(pady=5)
        self.find_button = tk.Button(root, text="Find Max Length List", command=self.find_max_length_list)
        self.find_button.pack(pady=5)
        self.result_label = tk.Label(root, text="", fg="blue")
        self.result_label.pack(pady=10)
    def add_list(self):
        '''
        Adds a user-input list to the list of lists.
        '''
        input_text = self.entry.get()
        try:
            # Convert the input text into a list
            user_list = [x.strip() for x in input_text.split(",")]
            self.lists.append(user_list)
            self.entry.delete(0, tk.END)
            messagebox.showinfo("Success", "List added successfully!")
        except Exception as e:
            messagebox.showerror("Error", f"Invalid input! Please enter a comma-separated list. Error: {str(e)}")
    def find_max_length_list(self):
        '''
        Finds and displays the list with the maximum length.
        '''
        if not self.lists:
            messagebox.showerror("Error", "No lists available! Please add some lists first.")
            return
        max_list = find_max_length(self.lists)
        self.result_label.config(text=f"Max Length List: {max_list}")
if __name__ == "__main__":
    root = tk.Tk()
    app = MainApp(root)
    root.mainloop()
