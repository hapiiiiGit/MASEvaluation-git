'''

This file contains the MainApplication class which sets up the GUI for the sublist checker application using tkinter.

'''

import tkinter as tk
from sublist_checker import is_sublist

class MainApplication:
    def __init__(self, root):
        self.root = root
        self.root.title("Sublist Checker")

        self.label1 = tk.Label(root, text="Enter Main List (comma separated):")
        self.label1.pack()

        self.main_list_entry = tk.Entry(root, width=50)
        self.main_list_entry.pack()

        self.label2 = tk.Label(root, text="Enter Sub List (comma separated):")
        self.label2.pack()

        self.sub_list_entry = tk.Entry(root, width=50)
        self.sub_list_entry.pack()

        self.check_button = tk.Button(root, text="Check Sublist", command=self.check_sublist)
        self.check_button.pack()

        self.result_label = tk.Label(root, text="")
        self.result_label.pack()

    def check_sublist(self):
        main_list_str = self.main_list_entry.get()
        sub_list_str = self.sub_list_entry.get()

        main_list = [int(item.strip()) for item in main_list_str.split(',')]
        sub_list = [int(item.strip()) for item in sub_list_str.split(',')]

        if is_sublist(main_list, sub_list):
            self.result_label.config(text="The sublist is a part of the main list.")
        else:
            self.result_label.config(text="The sublist is NOT a part of the main list.")

if __name__ == "__main__":
    root = tk.Tk()
    app = MainApplication(root)
    root.mainloop()

'''

This file contains the function is_sublist which checks if one list is a sublist of another.

'''

def is_sublist(main_list, sub_list):
    """Check if sub_list is a sublist of main_list."""
    if not sub_list:
        return True
    if not main_list:
        return False

    sub_len = len(sub_list)
    for i in range(len(main_list) - sub_len + 1):
        if main_list[i:i + sub_len] == sub_list:
            return True
    return False

```

def is_sublist(main_list, sub_list):
    """Check if sub_list is a sublist of main_list."""
    if not sub_list:
        return True
    if not main_list:
        return False
    sub_len = len(sub_list)
    for i in range(len(main_list) - sub_len + 1):
        if main_list[i:i + sub_len] == sub_list:
            return True
    return False

def is_sublist(main_list, sub_list):
    """Check if sub_list is a sublist of main_list."""
    if not sub_list:
        return True  # An empty sublist is always a sublist
    if not main_list:
        return False  # A non-empty sublist cannot be a sublist of an empty list
    
    sub_len = len(sub_list)
    # Iterate through the main list to check for the sublist
    for i in range(len(main_list) - sub_len + 1):
        if main_list[i:i + sub_len] == sub_list:
            return True
    return False

This file contains the function is_sublist which checks if one list is a sublist of another.
'''
def is_sublist(main_list, sub_list):
    """Check if sub_list is a sublist of main_list."""
    if not sub_list:
        return True  # An empty sublist is always a sublist
    if not main_list:
        return False  # A non-empty sublist cannot be a sublist of an empty list
    sub_len = len(sub_list)
    # Iterate through the main list to check for the sublist
    for i in range(len(main_list) - sub_len + 1):
        if main_list[i:i + sub_len] == sub_list:
            return True
    return False

def is_sublist(main_list, sub_list):
    """Check if sub_list is a sublist of main_list."""
    if not sub_list:
        return True  # An empty sublist is always a sublist
    if not main_list:
        return False  # A non-empty sublist cannot be a sublist of an empty list
    sub_len = len(sub_list)
    # Iterate through the main list to check for the sublist
    for i in range(len(main_list) - sub_len + 1):
        if main_list[i:i + sub_len] == sub_list:
            return True
    return False

'''
This file contains the function is_sublist which checks if one list is a sublist of another.
'''
def is_sublist(main_list, sub_list):
    """Check if sub_list is a sublist of main_list."""
    if not sub_list:
        return True  # An empty sublist is always a sublist
    if not main_list:
        return False  # A non-empty sublist cannot be a sublist of an empty list
    sub_len = len(sub_list)
    # Iterate through the main list to check for the sublist
    for i in range(len(main_list) - sub_len + 1):
        if main_list[i:i + sub_len] == sub_list:
            return True
    return False
