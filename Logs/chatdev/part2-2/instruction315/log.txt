#### main.py

```python
'''

Main application file for the array sum calculator with a GUI.

'''

import tkinter as tk
from tkinter import messagebox
from sum_calculator import sum_array

class MainApp:
    def __init__(self, root):
        self.root = root
        self.root.title("Array Sum Calculator")

        self.label = tk.Label(root, text="Enter numbers separated by commas:")
        self.label.pack()

        self.entry = tk.Entry(root, width=50)
        self.entry.pack()

        self.calculate_button = tk.Button(root, text="Calculate Sum", command=self.calculate_sum)
        self.calculate_button.pack()

        self.result_label = tk.Label(root, text="")
        self.result_label.pack()

    def calculate_sum(self):
        try:
            numbers_str = self.entry.get()
            numbers = list(map(float, numbers_str.split(',')))
            result = sum_array(numbers)
            self.result_label.config(text=f"Sum: {result}")
        except ValueError:
            messagebox.showerror("Invalid input", "Please enter a valid list of numbers separated by commas.")

if __name__ == "__main__":
    root = tk.Tk()
    app = MainApp(root)
    root.mainloop()
```

import tkinter as tk
from tkinter import messagebox
from sum_calculator import sum_array
class MainApp:
    def __init__(self, root):
        self.root = root
        self.root.title("Array Sum Calculator")
        self.label = tk.Label(root, text="Enter numbers separated by commas:")
        self.label.pack()
        self.entry = tk.Entry(root, width=50)
        self.entry.pack()
        self.calculate_button = tk.Button(root, text="Calculate Sum", command=self.calculate_sum)
        self.calculate_button.pack()
        self.result_label = tk.Label(root, text="")
        self.result_label.pack()
    def calculate_sum(self):
        try:
            numbers_str = self.entry.get()
            numbers = list(map(float, numbers_str.split(',')))
            result = sum_array(numbers)
            self.result_label.config(text=f"Sum: {result}")
        except ValueError:
            messagebox.showerror("Invalid input", "Please enter a valid list of numbers separated by commas.")
if __name__ == "__main__":
    root = tk.Tk()
    app = MainApp(root)
