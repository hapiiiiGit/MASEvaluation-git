def find_first_adverb(text):
    '''
    DOCSTRING: Finds the first adverb ending with 'ly' in the given text and returns its starting character position.
    '''
    words = text.split()
    current_pos = 0
    for word in words:

            return word, current_pos
        current_pos += len(word) + 1  # Add 1 for the space
    return None, None
class AdverbFinderApp:

import tkinter as tk
from tkinter import messagebox
def find_first_adverb(text):
    '''
    DOCSTRING: Finds the first adverb ending with 'ly' in the given text and returns its starting character position.
    '''
    words = text.split()
    for word in words:
        if word.endswith('ly'):
            position = text.find(word)
            return word, position
    return None, None
class AdverbFinderApp:
    '''
    DOCSTRING: A GUI application to find the first adverb ending with 'ly' in a given string.
    '''

        self.root = root
        self.root.title("Adverb Finder")

        self.result_label = tk.Label(root, text="")
        self.result_label.pack(pady=10)
    def find_adverb(self):
        '''
        DOCSTRING: Finds the first adverb ending with 'ly' and updates the result label.
        '''

        adverb, position = find_first_adverb(text)
        if adverb:
            result_text = f"First adverb: '{adverb}' at character position {position}"

import tkinter as tk
from tkinter import messagebox
def find_first_adverb(text):
    '''
    DOCSTRING: Finds the first adverb ending with 'ly' in the given text and returns its starting character position.
    '''
    words = text.split()
    for word in words:
        if word.endswith('ly'):
            position = text.find(word)
            return word, position
    return None, None
class AdverbFinderApp:
    '''
    DOCSTRING: A GUI application to find the first adverb ending with 'ly' in a given string.
    '''
    def __init__(self, root):
        self.root = root
        self.root.title("Adverb Finder")
        # Create GUI components
        self.label = tk.Label(root, text="Enter a string:")
        self.label.pack(pady=10)
        self.text_entry = tk.Entry(root, width=50)
        self.text_entry.pack(pady=10)
        self.find_button = tk.Button(root, text="Find Adverb", command=self.find_adverb)
        self.find_button.pack(pady=10)
        self.result_label = tk.Label(root, text="")
        self.result_label.pack(pady=10)
    def find_adverb(self):
        '''
        DOCSTRING: Finds the first adverb ending with 'ly' and updates the result label.
        '''
        text = self.text_entry.get()
        adverb, position = find_first_adverb(text)
        if adverb:
            result_text = f"First adverb: '{adverb}' at character position {position}"
        else:

        self.result_label.config(text=result_text)
