'''

This module contains the main application code for the Text Replacer App using tkinter.

'''

import tkinter as tk
from tkinter import ttk

def replace_characters(text):
    '''
    Replaces all occurrences of spaces, commas, and dots with a colon in the given text.
    
    :param text: The input string to be processed.
    :return: A new string with spaces, commas, and dots replaced by colons.
    '''
    return text.replace(' ', ':').replace(',', ':').replace('.', ':')

class TextReplacerApp:
    '''
    A class to create a GUI application for replacing spaces, commas, and dots with colons.
    '''

        '''

        :param master: The root window of the tkinter application.
        '''
        self.master = master
        master.title("Text Replacer App")
        
        # Create and place the input label and entry
        self.label = ttk.Label(master, text="Enter text:")
        self.label.pack(pady=5)
        
        self.entry = ttk.Entry(master, width=50)
        self.entry.pack(pady=5)
        
        # Create and place the replace button
        self.replace_button = ttk.Button(master, text="Replace", command=self.replace_text)
        self.replace_button.pack(pady=5)
        
        # Create and place the output label
        self.output_label = ttk.Label(master, text="Output:")
        self.output_label.pack(pady=5)
        
        self.output_text = tk.Text(master, height=10, width=50)
        self.output_text.pack(pady=5)

'''

import tkinter as tk
from tkinter import ttk
def replace_characters(text):
    '''
    Replaces all occurrences of spaces, commas, and dots with a colon in the given text.
    :param text: The input string to be processed.
    :return: A new string with spaces, commas, and dots replaced by colons.
    '''
    return text.replace(' ', ':').replace(',', ':').replace('.', ':')
class TextReplacerApp:
    '''
    A class to create a GUI application for replacing spaces, commas, and dots with colons.
    '''
    def __init__(self, master):
        '''
        Initializes the TextReplacerApp with the main window and its components.
        :param master: The root window of the tkinter application.
        '''
        self.master = master
        master.title("Text Replacer App")
        # Create and place the input label and entry
        self.label = ttk.Label(master, text="Enter text:")
        self.label.pack(pady=5)
        self.entry = ttk.Entry(master, width=50)
        self.entry.pack(pady=5)
        # Create and place the replace button
        self.replace_button = ttk.Button(master, text="Replace", command=self.replace_text)
        self.replace_button.pack(pady=5)
        # Create and place the output label
        self.output_label = ttk.Label(master, text="Output:")
        self.output_label.pack(pady=5)
        self.output_text = tk.Text(master, height=10, width=50)
        self.output_text.pack(pady=5)
    def replace_text(self):
        '''
        Handles the replace button click event, processes the input text, and displays the result.
        '''
        input_text = self.entry.get()
        replaced_text = replace_characters(input_text)
        self.output_text.delete(1.0, tk.END)
        self.output_text.insert(tk.END, replaced_text)
if __name__ == "__main__":
    root = tk.Tk()
    app = TextReplacerApp(root)
    root.mainloop()

    return text.replace(' ', ':').replace(',', ':').replace('.', ':')
class TextReplacerApp:
    '''
    A class to create a GUI application for replacing spaces, commas, and dots with colons.
    '''
    def __init__(self, master):
        '''
        Initializes the TextReplacerApp with the main window and its components.
        :param master: The root window of the tkinter application.
        '''
        self.master = master
        master.title("Text Replacer App")
        # Create and place the input label and entry
        self.label = ttk.Label(master, text="Enter text:")
        self.label.pack(pady=5)
        self.entry = ttk.Entry(master, width=50)
        self.entry.pack(pady=5)
        # Create and place the replace button
        self.replace_button = ttk.Button(master, text="Replace", command=self.replace_text)
        self.replace_button.pack(pady=5)
        # Create and place the output label
        self.output_label = ttk.Label(master, text="Output:")
        self.output_label.pack(pady=5)
        self.output_text = tk.Text(master, height=10, width=50)
        self.output_text.pack(pady=5)
    def replace_text(self):
        '''
        Handles the replace button click event, processes the input text, and displays the result.
        '''
        input_text = self.entry.get()
        replaced_text = replace_characters(input_text)
        self.output_text.delete(1.0, tk.END)
        self.output_text.insert(tk.END, replaced_text)
if __name__ == "__main__":
    root = tk.Tk()
    app = TextReplacerApp(root)
    root.mainloop()
