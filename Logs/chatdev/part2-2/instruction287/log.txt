```python
'''

This script creates a GUI application using tkinter to find all pairs of consecutive items in a given list.

'''

import tkinter as tk
from tkinter import messagebox

def find_consecutive_pairs(input_list):
    '''
    This function takes a list and returns a list of tuples, each containing a pair of consecutive items.
    
    :param input_list: List of items
    :return: List of tuples with consecutive pairs
    '''
    return [(input_list[i], input_list[i + 1]) for i in range(len(input_list) - 1)]

class ConsecutivePairsApp:
    '''
    This class handles the GUI components and user interactions for finding consecutive pairs in a list.
    '''
    
    def __init__(self, master):
        '''
        Initializes the GUI components.
        
        :param master: The root window
        '''
        self.master = master
        master.title("Consecutive Pairs Finder")
        
        self.label = tk.Label(master, text="Enter a list of items (comma-separated):")
        self.label.pack()
        
        self.entry = tk.Entry(master, width=50)
        self.entry.pack()
        
        self.find_button = tk.Button(master, text="Find Consecutive Pairs", command=self.get_consecutive_pairs)
        self.find_button.pack()
        
        self.result_label = tk.Label(master, text="", fg="blue")
        self.result_label.pack()
    
    def get_consecutive_pairs(self):
        '''
        Retrieves the input list, computes consecutive pairs, and displays the result.
        '''
        input_text = self.entry.get()
        try:
            # Convert input text to a list
            input_list = [item.strip() for item in input_text.split(',')]
            # Find consecutive pairs
            pairs = find_consecutive_pairs(input_list)
            # Display result
            self.result_label.config(text=f"Consecutive Pairs: {pairs}")
        except Exception as e:
            messagebox.showerror("Error", f"An error occurred: {e}")

if __name__ == "__main__":
    root = tk.Tk()
    app = ConsecutivePairsApp(root)
    root.mainloop()
```

'''

This script creates a GUI application using tkinter to find all pairs of consecutive items in a given list.

'''

import tkinter as tk
from tkinter import messagebox

def find_consecutive_pairs(input_list):
    '''
    This function takes a list and returns a list of tuples, each containing a pair of consecutive items.
    
    :param input_list: List of items
    :return: List of tuples with consecutive pairs
    '''
    return [(input_list[i], input_list[i + 1]) for i in range(len(input_list) - 1)]

class ConsecutivePairsApp:
    '''
    This class handles the GUI components and user interactions for finding consecutive pairs in a list.
    '''
    
    def __init__(self, master):
        '''
        Initializes the GUI components.
        
        :param master: The root window
        '''
        self.master = master
        master.title("Consecutive Pairs Finder")
        
        self.label = tk.Label(master, text="Enter a list of items (comma-separated):")
        self.label.pack()
        
        self.entry = tk.Entry(master, width=50)
        self.entry.pack()
        
        self.find_button = tk.Button(master, text="Find Consecutive Pairs", command=self.get_consecutive_pairs)
        self.find_button.pack()
        
        self.result_label = tk.Label(master, text="", fg="blue")
        self.result_label.pack()
    
    def get_consecutive_pairs(self):
        '''
        Retrieves the input list, computes consecutive pairs, and displays the result.
        '''
        input_text = self.entry.get()
        try:
            # Convert input text to a list
            input_list = [item.strip() for item in input_text.split(',')]
            # Find consecutive pairs
            pairs = find_consecutive_pairs(input_list)
            # Display result
            self.result_label.config(text=f"Consecutive Pairs: {pairs}")
        except Exception as e:
            messagebox.showerror("Error", f"An error occurred: {e}")

if __name__ == "__main__":
    root = tk.Tk()
    app = ConsecutivePairsApp(root)
    root.mainloop()

'''

import tkinter as tk
from tkinter import messagebox

def find_consecutive_pairs(input_list):
    '''

    :param input_list: List of items

    '''
    return [(input_list[i], input_list[i + 1]) for i in range(len(input_list) - 1)]

'''
This script creates a GUI application using tkinter to find all pairs of consecutive items in a given list.
'''
import tkinter as tk
from tkinter import messagebox
def find_consecutive_pairs(input_list):
    '''
    This function takes a list and returns a list of tuples, each containing a pair of consecutive items.
    :param input_list: List of items
    :return: List of tuples with consecutive pairs
    '''
    return [(input_list[i], input_list[i + 1]) for i in range(len(input_list) - 1)]
class ConsecutivePairsApp:
    '''
    This class handles the GUI components and user interactions for finding consecutive pairs in a list.
    '''
    def __init__(self, master):
        '''
        Initializes the GUI components.
        :param master: The root window
        '''
        self.master = master
        master.title("Consecutive Pairs Finder")
        # Label for instructions
        self.label = tk.Label(master, text="Enter a list of items (comma-separated):")
        self.label.pack()
        # Entry field for user input
        self.entry = tk.Entry(master, width=50)
        self.entry.pack()
        # Button to trigger the function
        self.find_button = tk.Button(master, text="Find Consecutive Pairs", command=self.get_consecutive_pairs)
        self.find_button.pack()
        # Label to display the result
        self.result_label = tk.Label(master, text="", fg="blue")
        self.result_label.pack()
    def get_consecutive_pairs(self):
        '''
        Retrieves the input list, computes consecutive pairs, and displays the result.
        '''
        input_text = self.entry.get()
        try:
            # Convert input text to a list
            input_list = [item.strip() for item in input_text.split(',') if item.strip()]
            if len(input_list) < 2:
                messagebox.showinfo("Info", "No consecutive pairs can be formed from the input.")
                return
            # Find consecutive pairs
            pairs = find_consecutive_pairs(input_list)
            # Display result
            self.result_label.config(text=f"Consecutive Pairs: {pairs}")
        except Exception as e:
            messagebox.showerror("Error", f"An error occurred: {e}")
if __name__ == "__main__":
    root = tk.Tk()
    app = ConsecutivePairsApp(root)
    root.mainloop()

'''
This script creates a GUI application using tkinter to find all pairs of consecutive items in a given list.
'''
import tkinter as tk
from tkinter import messagebox
def find_consecutive_pairs(input_list):
    '''
    This function takes a list and returns a list of tuples, each containing a pair of consecutive items.
    :param input_list: List of items
    :return: List of tuples with consecutive pairs
    '''
    return [(input_list[i], input_list[i + 1]) for i in range(len(input_list) - 1)]
class ConsecutivePairsApp:
    '''
    This class handles the GUI components and user interactions for finding consecutive pairs in a list.
    '''
    def __init__(self, master):
        '''
        Initializes the GUI components.
        :param master: The root window
        '''
        self.master = master
        master.title("Consecutive Pairs Finder")
        # Label for instructions
        self.label = tk.Label(master, text="Enter a list of items (comma-separated):")
        self.label.pack()
        # Entry field for user input
        self.entry = tk.Entry(master, width=50)
        self.entry.pack()
        # Button to trigger the function
        self.find_button = tk.Button(master, text="Find Consecutive Pairs", command=self.get_consecutive_pairs)
        self.find_button.pack()
        # Label to display the result
        self.result_label = tk.Label(master, text="", fg="blue")
        self.result_label.pack()
    def get_consecutive_pairs(self):
        '''
        Retrieves the input list, computes consecutive pairs, and displays the result.
        '''
        input_text = self.entry.get()
        try:
            # Convert input text to a list
            input_list = [item.strip() for item in input_text.split(',') if item.strip()]
            if len(input_list) < 2:
                messagebox.showinfo("Info", "No consecutive pairs can be formed from the input.")
                return
            # Find consecutive pairs
            pairs = find_consecutive_pairs(input_list)
            # Display result
            self.result_label.config(text=f"Consecutive Pairs: {pairs}")
        except Exception as e:
            messagebox.showerror("Error", f"An error occurred: {e}")
if __name__ == "__main__":
    root = tk.Tk()
    app = ConsecutivePairsApp(root)
    root.mainloop()

# Consecutive Pairs Finder

This application is a simple GUI tool designed to find all pairs of consecutive items in a given list. It is built using Python and the Tkinter library for the graphical user interface.

## Main Functions of the Software

- **Input Handling**: Users can enter a list of items as a comma-separated string.
- **Consecutive Pair Finding**: The application processes the input list and identifies all pairs of consecutive items.
- **Result Display**: The consecutive pairs are displayed in the application window.

## Installation

This project does not require any external dependencies beyond Python's standard library. Ensure you have Python installed on your system.

   ```

   ```
