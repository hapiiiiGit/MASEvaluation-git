'''

This module contains the main application logic and GUI setup for removing consecutive duplicates from a list.

'''

import tkinter as tk
from tkinter import messagebox

class MainApp:
    def __init__(self, root):
        self.root = root
        self.root.title("Remove Consecutive Duplicates")
        
        self.label = tk.Label(root, text="Enter a list of numbers separated by commas:")
        self.label.pack(pady=10)
        
        self.entry = tk.Entry(root, width=50)
        self.entry.pack(pady=10)
        
        self.submit_button = tk.Button(root, text="Submit", command=self.on_submit)
        self.submit_button.pack(pady=10)
        
        self.result_label = tk.Label(root, text="")
        self.result_label.pack(pady=10)
    
    def remove_consecutive_duplicates(self, lst):
        if not lst:
            return []
        result = [lst[0]]
        for item in lst[1:]:
            if item != result[-1]:
                result.append(item)
        return result
    
    def on_submit(self):
        user_input = self.entry.get()
        try:
            lst = [int(x.strip()) for x in user_input.split(',')]
            result = self.remove_consecutive_duplicates(lst)
            self.result_label.config(text=f"Result: {result}")
        except ValueError:
            messagebox.showerror("Invalid Input", "Please enter a valid list of numbers separated by commas.")

def create_gui():
    root = tk.Tk()
    app = MainApp(root)
    root.mainloop()

if __name__ == "__main__":
    create_gui()

```

'''

This module contains the main application logic and GUI setup for removing consecutive duplicates from a list.

'''

import tkinter as tk
from tkinter import messagebox

class MainApp:
    def __init__(self, root):
        self.root = root
        self.root.title("Remove Consecutive Duplicates")
        
        self.label = tk.Label(root, text="Enter a list of numbers separated by commas:")
        self.label.pack(pady=10)
        
        self.entry = tk.Entry(root, width=50)
        self.entry.pack(pady=10)
        
        self.submit_button = tk.Button(root, text="Submit", command=self.on_submit)
        self.submit_button.pack(pady=10)
        
        self.result_label = tk.Label(root, text="")
        self.result_label.pack(pady=10)
    
    def remove_consecutive_duplicates(self, lst):
        if not lst:
            return []
        result = [lst[0]]
        for item in lst[1:]:
            if item != result[-1]:
                result.append(item)
        return result
    
    def on_submit(self):
        user_input = self.entry.get()
        try:
            lst = [int(x.strip()) for x in user_input.split(',')]
            result = self.remove_consecutive_duplicates(lst)
            self.result_label.config(text=f"Result: {result}")
        except ValueError:
            messagebox.showerror("Invalid Input", "Please enter a valid list of numbers separated by commas.")

def create_gui():
    root = tk.Tk()
    app = MainApp(root)
    root.mainloop()

if __name__ == "__main__":
    create_gui()

import tkinter as tk
from tkinter import messagebox

class MainApp:
    def __init__(self, root):
        self.root = root
        self.root.title("Remove Consecutive Duplicates")
        self.label = tk.Label(root, text="Enter a list of numbers separated by commas:")
        self.label.pack(pady=10)

        '''

        self.result_label = tk.Label(root, text="")

        '''
        Removes consecutive duplicates from a list.
        Parameters:

        Returns:
        list: A new list with consecutive duplicates removed.
        '''
        if not lst:

        result = [lst[0]]
        for item in lst[1:]:
            if item != result[-1]:

        return result
