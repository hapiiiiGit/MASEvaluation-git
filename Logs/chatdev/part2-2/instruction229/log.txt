```python
'''

This file contains the main application logic for finding the smallest missing number from a sorted list of natural numbers using a GUI.

'''

import tkinter as tk
from tkinter import messagebox

def find_smallest_missing_number(nums):
    '''
    This function finds the smallest missing number from a sorted list of natural numbers.
    
    :param nums: List of sorted natural numbers
    :return: The smallest missing number
    '''
    if not nums:
        return 0
    for i in range(len(nums)):
        if nums[i] != i:
            return i
    return len(nums)

class SmallestMissingNumberApp:
    '''
    This class creates a GUI application to find the smallest missing number from a sorted list.
    '''
    def __init__(self, master):
        self.master = master
        master.title("Smallest Missing Number Finder")

        self.label = tk.Label(master, text="Enter a sorted list of natural numbers (comma separated):")
        self.label.pack()

        self.entry = tk.Entry(master, width=50)
        self.entry.pack()

        self.find_button = tk.Button(master, text="Find Smallest Missing Number", command=self.find_missing_number)
        self.find_button.pack()

        self.result_label = tk.Label(master, text="")
        self.result_label.pack()

    def find_missing_number(self):
        '''
        This method retrieves the input from the user, processes it, and displays the smallest missing number.
        '''
        try:
            input_data = self.entry.get()
            nums = list(map(int, input_data.split(',')))
            result = find_smallest_missing_number(nums)
            self.display_result(result)
        except ValueError:
            messagebox.showerror("Invalid input", "Please enter a valid list of numbers.")

    def display_result(self, result):
        '''
        This method displays the result in the GUI.
        
        :param result: The smallest missing number
        '''
        self.result_label.config(text=f"The smallest missing number is: {result}")

if __name__ == "__main__":
    root = tk.Tk()
    app = SmallestMissingNumberApp(root)
    root.mainloop()
```

- **`SmallestMissingNumberApp`**: This class sets up the Tkinter GUI, allowing users to input a list of numbers and find the smallest missing number. It handles input validation and displays the result.

```python
'''

This file contains the main application logic for finding the smallest missing number from a sorted list of natural numbers using a GUI.

'''

import tkinter as tk
from tkinter import messagebox

def find_smallest_missing_number(nums):
    '''
    This function finds the smallest missing number from a sorted list of natural numbers.
    
    :param nums: List of sorted natural numbers
    :return: The smallest missing number
    '''
    if not nums:
        return 0
    for i in range(len(nums)):
        if nums[i] != i:
            return i
    return len(nums)

class SmallestMissingNumberApp:
    '''
    This class creates a GUI application to find the smallest missing number from a sorted list.
    '''
    def __init__(self, master):
        self.master = master
        master.title("Smallest Missing Number Finder")

        self.label = tk.Label(master, text="Enter a sorted list of natural numbers (comma separated):")
        self.label.pack()

        self.entry = tk.Entry(master, width=50)
        self.entry.pack()

        self.find_button = tk.Button(master, text="Find Smallest Missing Number", command=self.find_missing_number)
        self.find_button.pack()

        self.result_label = tk.Label(master, text="")
        self.result_label.pack()

    def find_missing_number(self):
        '''
        This method retrieves the input from the user, processes it, and displays the smallest missing number.
        '''
        try:
            input_data = self.entry.get()
            nums = list(map(int, input_data.split(',')))
            result = find_smallest_missing_number(nums)
            self.display_result(result)
        except ValueError:
            messagebox.showerror("Invalid input", "Please enter a valid list of numbers.")

    def display_result(self, result):
        '''
        This method displays the result in the GUI.
        
        :param result: The smallest missing number
        '''
        self.result_label.config(text=f"The smallest missing number is: {result}")

if __name__ == "__main__":
    root = tk.Tk()
    app = SmallestMissingNumberApp(root)
    root.mainloop()

'''
This file contains the main application logic for finding the smallest missing number from a sorted list of natural numbers using a GUI.
'''
import tkinter as tk
from tkinter import messagebox
def find_smallest_missing_number(nums):
    '''
    This function finds the smallest missing number from a sorted list of natural numbers.
    :param nums: List of sorted natural numbers
    :return: The smallest missing number
    '''
    if not nums:
        return 0
    for i in range(len(nums)):
        if nums[i] != i:
            return i
    return len(nums)
class SmallestMissingNumberApp:
    '''
    This class creates a GUI application to find the smallest missing number from a sorted list.
    '''
    def __init__(self, master):
        self.master = master
        master.title("Smallest Missing Number Finder")
        self.label = tk.Label(master, text="Enter a sorted list of natural numbers (comma separated):")
        self.label.pack()
        self.entry = tk.Entry(master, width=50)
        self.entry.pack()
        self.find_button = tk.Button(master, text="Find Smallest Missing Number", command=self.find_missing_number)
        self.find_button.pack()
        self.result_label = tk.Label(master, text="")
        self.result_label.pack()
    def find_missing_number(self):
        '''
        This method retrieves the input from the user, processes it, and displays the smallest missing number.
        '''
        try:
            input_data = self.entry.get()
            nums = list(map(int, input_data.split(',')))
            result = find_smallest_missing_number(nums)
            self.display_result(result)
        except ValueError:
            messagebox.showerror("Invalid input", "Please enter a valid list of numbers.")
    def display_result(self, result):
        '''
        This method displays the result in the GUI.
        :param result: The smallest missing number
        '''
        self.result_label.config(text=f"The smallest missing number is: {result}")
if __name__ == "__main__":
    root = tk.Tk()
    app = SmallestMissingNumberApp(root)
    root.mainloop()

'''
This file contains the main application logic for finding the smallest missing number from a sorted list of natural numbers using a GUI.
'''
import tkinter as tk
from tkinter import messagebox
def find_smallest_missing_number(nums):
    '''
    This function finds the smallest missing number from a sorted list of natural numbers.
    :param nums: List of sorted natural numbers
    :return: The smallest missing number
    '''
    if not nums:
        return 0
    for i in range(len(nums)):
        if nums[i] != i:
            return i
    return len(nums)
class SmallestMissingNumberApp:
    '''
    This class creates a GUI application to find the smallest missing number from a sorted list.
    '''
    def __init__(self, master):
        self.master = master
        master.title("Smallest Missing Number Finder")
        self.label = tk.Label(master, text="Enter a sorted list of natural numbers (comma separated):")
        self.label.pack()
        self.entry = tk.Entry(master, width=50)
        self.entry.pack()
        self.find_button = tk.Button(master, text="Find Smallest Missing Number", command=self.find_missing_number)
        self.find_button.pack()
        self.result_label = tk.Label(master, text="")
        self.result_label.pack()
    def find_missing_number(self):
        '''
        This method retrieves the input from the user, processes it, and displays the smallest missing number.
        '''
        try:
            input_data = self.entry.get()
            nums = list(map(int, input_data.split(',')))
            result = find_smallest_missing_number(nums)
            self.display_result(result)
        except ValueError:
            messagebox.showerror("Invalid input", "Please enter a valid list of numbers.")
    def display_result(self, result):
        '''
        This method displays the result in the GUI.
        :param result: The smallest missing number
        '''
        self.result_label.config(text=f"The smallest missing number is: {result}")
if __name__ == "__main__":
    root = tk.Tk()
    app = SmallestMissingNumberApp(root)
    root.mainloop()

# Smallest Missing Number Finder

This application provides a graphical user interface (GUI) to find the smallest missing number from a sorted list of natural numbers. It is designed to be user-friendly and requires no external dependencies.

## Quick Install

Since this application does not require any external dependencies, you can run it directly with Python. Ensure you have Python installed on your system.

## 🤔 What is this?

The Smallest Missing Number Finder is a simple yet effective tool to determine the smallest missing natural number from a sorted list provided by the user. This can be particularly useful in scenarios where you need to identify gaps in sequences or datasets.

## Main Functions

- **Find Smallest Missing Number**: The core functionality of the application is to compute the smallest missing number from a sorted list of natural numbers. This is achieved through a straightforward algorithm that checks each number in the list against its expected value.

- **Graphical User Interface (GUI)**: The application features a user-friendly GUI built with Tkinter, allowing users to input their data and view results easily.

## How to Use

1. **Run the Application**: Open a terminal or command prompt and navigate to the directory containing `main.py`. Execute the following command to start the application:

   ```bash
   python main.py
   ```

2. **Input Data**: Once the application window opens, you will see a text entry field. Enter a sorted list of natural numbers separated by commas (e.g., `0,1,2,4,5`).

3. **Find the Missing Number**: Click the "Find Smallest Missing Number" button. The application will process your input and display the smallest missing number below the button.

4. **Error Handling**: If you enter invalid data (e.g., non-numeric values or unsorted numbers), the application will display an error message prompting you to enter a valid list.

## Example

- **Input**: `0,1,2,4,5`
- **Output**: The smallest missing number is: `3`

This application is designed to be intuitive and requires minimal setup, making it accessible for users with varying levels of technical expertise. Enjoy finding the smallest missing numbers with ease!
