'''

This module contains the main application logic for converting a list to a tuple using a GUI built with Tkinter.

'''

import tkinter as tk
from tkinter import messagebox

class MainApp:
    def __init__(self, root):
        self.root = root
        self.root.title("List to Tuple Converter")
        
        self.create_gui()
    
    def create_gui(self):
        # Create and place labels, entry, and button
        self.label = tk.Label(self.root, text="Enter a list (comma separated):")
        self.label.pack(pady=10)
        
        self.entry = tk.Entry(self.root, width=50)
        self.entry.pack(pady=10)
        
        self.convert_button = tk.Button(self.root, text="Convert to Tuple", command=self.on_convert_button_click)
        self.convert_button.pack(pady=10)
        
        self.result_label = tk.Label(self.root, text="")
        self.result_label.pack(pady=10)
    
    def convert_list_to_tuple(self, input_list):
        '''
        Converts a list to a tuple.

        Parameters:
        input_list (list): The list to be converted.

        Returns:
        tuple: The converted tuple.
        '''
        return tuple(input_list)
    
    def on_convert_button_click(self):
        '''
        Event handler for the convert button click. Converts the input list to a tuple and displays the result.
        '''
        input_text = self.entry.get()
        try:
            # Convert the input text to a list
            input_list = [x.strip() for x in input_text.split(',')]
            # Convert list to tuple
            result_tuple = self.convert_list_to_tuple(input_list)
            # Display the result
            self.result_label.config(text=f"Converted Tuple: {result_tuple}")
        except Exception as e:
            messagebox.showerror("Error", f"An error occurred: {e}")

if __name__ == "__main__":
    root = tk.Tk()
    app = MainApp(root)
    root.mainloop()
```

class MainApp:

        self.root = root

        self.result_label.pack(pady=10)

        Converts a list to a tuple.

        input_list (list): The list to be converted.

        Returns:
        tuple: The converted tuple.

'''
This module contains the main application logic for converting a list to a tuple using a GUI built with Tkinter.
'''
import tkinter as tk
from tkinter import messagebox

class MainApp:
    def __init__(self, root):
        '''
        Initializes the main application window and its components.
        Parameters:
        root (Tk): The root window of the Tkinter application.
        '''
        self.root = root
        self.root.title("List to Tuple Converter")
        self.create_gui()

    def create_gui(self):
        '''
        Creates the graphical user interface components.
        '''
        # Create and place labels, entry, and button
        self.label = tk.Label(self.root, text="Enter a list (comma separated):")
        self.label.pack(pady=10)
        
        self.entry = tk.Entry(self.root, width=50)
        self.entry.pack(pady=10)
        
        self.convert_button = tk.Button(self.root, text="Convert to Tuple", command=self.on_convert_button_click)
        self.convert_button.pack(pady=10)
        
        self.result_label = tk.Label(self.root, text="")
        self.result_label.pack(pady=10)

    def convert_list_to_tuple(self, input_list):
        '''
        Converts a list to a tuple.
        Parameters:
        input_list (list): The list to be converted.
        Returns:
        tuple: The converted tuple.
        '''
        return tuple(input_list)

    def on_convert_button_click(self):
        '''
        Event handler for the convert button click. Converts the input list to a tuple and displays the result.
        '''
        input_text = self.entry.get()
        try:
            # Convert the input text to a list
            input_list = [x.strip() for x in input_text.split(',')]
            # Convert list to tuple
            result_tuple = self.convert_list_to_tuple(input_list)
            # Display the result
            self.result_label.config(text=f"Converted Tuple: {result_tuple}")
        except Exception as e:
            messagebox.showerror("Error", f"An error occurred: {e}")

if __name__ == "__main__":
    root = tk.Tk()
    app = MainApp(root)
    root.mainloop()

import tkinter as tk
from tkinter import messagebox
class MainApp:
    def __init__(self, root):
        '''
        Initializes the main application window and its components.
        Parameters:
        root (Tk): The root window of the Tkinter application.
        '''
        self.root = root
        self.root.title("List to Tuple Converter")
        self.create_gui()
    def create_gui(self):
        '''
        Creates the graphical user interface components.
        '''
        # Create and place labels, entry, and button
        self.label = tk.Label(self.root, text="Enter a list (comma separated):")
        self.label.pack(pady=10)
        self.entry = tk.Entry(self.root, width=50)
        self.entry.pack(pady=10)
        self.convert_button = tk.Button(self.root, text="Convert to Tuple", command=self.on_convert_button_click)
        self.convert_button.pack(pady=10)
        self.result_label = tk.Label(self.root, text="")
        self.result_label.pack(pady=10)
    def convert_list_to_tuple(self, input_list):
        '''
        Converts a list to a tuple.
        Parameters:
        input_list (list): The list to be converted.
        Returns:
        tuple: The converted tuple.
        '''
        return tuple(input_list)
    def on_convert_button_click(self):
        '''
        Event handler for the convert button click. Converts the input list to a tuple and displays the result.
        '''
        input_text = self.entry.get()
        try:
            # Convert the input text to a list
            input_list = [x.strip() for x in input_text.split(',')]
            # Convert list to tuple
            result_tuple = self.convert_list_to_tuple(input_list)
            # Display the result
            self.result_label.config(text=f"Converted Tuple: {result_tuple}")
        except Exception as e:
            messagebox.showerror("Error", f"An error occurred: {e}")
if __name__ == "__main__":
    root = tk.Tk()
    app = MainApp(root)
    root.mainloop()

        '''
        Creates the graphical user interface components.

        self.result_label = tk.Label(self.root, text="")
        self.result_label.pack(pady=10)

        Converts a list to a tuple.

        Returns:
        tuple: The converted tuple.

            # Display the result

            messagebox.showerror("Error", f"An error occurred: {e}")
