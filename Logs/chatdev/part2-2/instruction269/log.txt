def longest_common_subsequence(X, Y, Z):
    m = len(X)
    n = len(Y)
    o = len(Z)

    # Create a 3D array to store lengths of longest common subsequence
    L = [[[0 for _ in range(o+1)] for _ in range(n+1)] for _ in range(m+1)]

    # Build the LCS table in bottom-up fashion
    for i in range(m+1):
        for j in range(n+1):
            for k in range(o+1):
                if i == 0 or j == 0 or k == 0:
                    L[i][j][k] = 0
                elif X[i-1] == Y[j-1] == Z[k-1]:
                    L[i][j][k] = L[i-1][j-1][k-1] + 1
                else:
                    L[i][j][k] = max(L[i-1][j][k], L[i][j-1][k], L[i][j][k-1])

    # The length of the longest common subsequence
    lcs_length = L[m][n][o]

    # Retrieve the longest common subsequence
    lcs = []
    i, j, k = m, n, o
    while i > 0 and j > 0 and k > 0:
        if X[i-1] == Y[j-1] == Z[k-1]:
            lcs.append(X[i-1])
            i -= 1
            j -= 1
            k -= 1
        elif L[i-1][j][k] >= L[i][j-1][k] and L[i-1][j][k] >= L[i][j][k-1]:
            i -= 1
        elif L[i][j-1][k] >= L[i][j][k-1]:
            j -= 1
        else:
            k -= 1

    return ''.join(reversed(lcs))

'''
Module for calculating the longest common subsequence of three strings.
'''
def longest_common_subsequence(X, Y, Z):
    m = len(X)
    n = len(Y)
    o = len(Z)
    # Create a 3D array to store lengths of longest common subsequence
    L = [[[0 for _ in range(o+1)] for _ in range(n+1)] for _ in range(m+1)]
    # Build the LCS table in bottom-up fashion
    for i in range(m+1):
        for j in range(n+1):
            for k in range(o+1):
                if i == 0 or j == 0 or k == 0:
                    L[i][j][k] = 0
                elif X[i-1] == Y[j-1] == Z[k-1]:
                    L[i][j][k] = L[i-1][j-1][k-1] + 1
                else:
                    L[i][j][k] = max(L[i-1][j][k], L[i][j-1][k], L[i][j][k-1])
    # The length of the longest common subsequence
    lcs_length = L[m][n][o]
    # Retrieve the longest common subsequence
    lcs = []
    i, j, k = m, n, o
    while i > 0 and j > 0 and k > 0:
        if X[i-1] == Y[j-1] == Z[k-1]:
            lcs.append(X[i-1])
            i -= 1
            j -= 1
            k -= 1
        elif L[i-1][j][k] >= L[i][j-1][k] and L[i-1][j][k] >= L[i][j][k-1]:
            i -= 1
        elif L[i][j-1][k] >= L[i][j][k-1]:
            j -= 1
        else:
            k -= 1
    return ''.join(reversed(lcs))

'''
Module for calculating the longest common subsequence of three strings.
'''
def longest_common_subsequence(X, Y, Z):
    m = len(X)
    n = len(Y)
    o = len(Z)

    # Create a 3D array to store lengths of longest common subsequence
    L = [[[0 for _ in range(o+1)] for _ in range(n+1)] for _ in range(m+1)]

    # Build the LCS table in bottom-up fashion
    for i in range(m+1):
        for j in range(n+1):
            for k in range(o+1):
                if i == 0 or j == 0 or k == 0:
                    L[i][j][k] = 0
                elif X[i-1] == Y[j-1] == Z[k-1]:
                    L[i][j][k] = L[i-1][j-1][k-1] + 1
                else:
                    L[i][j][k] = max(L[i-1][j][k], L[i][j-1][k], L[i][j][k-1])

    # The length of the longest common subsequence
    lcs_length = L[m][n][o]

    # Retrieve the longest common subsequence
    lcs = []
    i, j, k = m, n, o
    while i > 0 and j > 0 and k > 0:
        if X[i-1] == Y[j-1] == Z[k-1]:
            lcs.append(X[i-1])
            i -= 1
            j -= 1
            k -= 1
        elif L[i-1][j][k] >= L[i][j-1][k] and L[i-1][j][k] >= L[i][j][k-1]:
            i -= 1
        elif L[i][j-1][k] >= L[i][j][k-1]:
            j -= 1
        else:
            k -= 1

    return ''.join(reversed(lcs))

'''
Module for calculating the longest common subsequence of three strings.
'''
def longest_common_subsequence(X, Y, Z):
    m = len(X)
    n = len(Y)
    o = len(Z)
    # Create a 3D array to store lengths of longest common subsequence
    L = [[[0 for _ in range(o+1)] for _ in range(n+1)] for _ in range(m+1)]
    # Build the LCS table in bottom-up fashion
    for i in range(m+1):
        for j in range(n+1):
            for k in range(o+1):
                if i == 0 or j == 0 or k == 0:
                    L[i][j][k] = 0
                elif X[i-1] == Y[j-1] == Z[k-1]:
                    L[i][j][k] = L[i-1][j-1][k-1] + 1
                else:
                    L[i][j][k] = max(L[i-1][j][k], L[i][j-1][k], L[i][j][k-1])
    # The length of the longest common subsequence
    lcs_length = L[m][n][o]
    # Retrieve the longest common subsequence
    lcs = []
    i, j, k = m, n, o
    while i > 0 and j > 0 and k > 0:
        if X[i-1] == Y[j-1] == Z[k-1]:
            lcs.append(X[i-1])
            i -= 1
            j -= 1
            k -= 1
        elif L[i-1][j][k] >= L[i][j-1][k] and L[i-1][j][k] >= L[i][j][k-1]:
            i -= 1
        elif L[i][j-1][k] >= L[i][j][k-1]:
            j -= 1
        else:
            k -= 1
    return ''.join(reversed(lcs))

'''
Module for calculating the longest common subsequence of three strings.
'''
def longest_common_subsequence(X, Y, Z):
    m = len(X)
    n = len(Y)
    o = len(Z)
    # Create a 3D array to store lengths of longest common subsequence
    L = [[[0 for _ in range(o+1)] for _ in range(n+1)] for _ in range(m+1)]
    # Build the LCS table in bottom-up fashion
    for i in range(m+1):
        for j in range(n+1):
            for k in range(o+1):
                if i == 0 or j == 0 or k == 0:
                    L[i][j][k] = 0
                elif X[i-1] == Y[j-1] == Z[k-1]:
                    L[i][j][k] = L[i-1][j-1][k-1] + 1
                else:
                    L[i][j][k] = max(L[i-1][j][k], L[i][j-1][k], L[i][j][k-1])
    # The length of the longest common subsequence
    lcs_length = L[m][n][o]
    # Retrieve the longest common subsequence
    lcs = []
    i, j, k = m, n, o
    while i > 0 and j > 0 and k > 0:
        if X[i-1] == Y[j-1] == Z[k-1]:
            lcs.append(X[i-1])
            i -= 1
            j -= 1
            k -= 1
        elif L[i-1][j][k] >= L[i][j-1][k] and L[i-1][j][k] >= L[i][j][k-1]:
            i -= 1
        elif L[i][j-1][k] >= L[i][j][k-1]:
            j -= 1
        else:
            k -= 1
    return ''.join(reversed(lcs))

'''
Module for calculating the longest common subsequence of three strings.
'''
def longest_common_subsequence(X, Y, Z):
    m = len(X)
    n = len(Y)
    o = len(Z)
    # Create a 3D array to store lengths of longest common subsequence
    L = [[[0 for _ in range(o+1)] for _ in range(n+1)] for _ in range(m+1)]
    # Build the LCS table in bottom-up fashion
    for i in range(m+1):
        for j in range(n+1):
            for k in range(o+1):
                if i == 0 or j == 0 or k == 0:
                    L[i][j][k] = 0
                elif X[i-1] == Y[j-1] == Z[k-1]:
                    L[i][j][k] = L[i-1][j-1][k-1] + 1
                else:
                    L[i][j][k] = max(L[i-1][j][k], L[i][j-1][k], L[i][j][k-1])
    # The length of the longest common subsequence
    lcs_length = L[m][n][o]
    # Retrieve the longest common subsequence
    lcs = []
    i, j, k = m, n, o
    while i > 0 and j > 0 and k > 0:
        if X[i-1] == Y[j-1] == Z[k-1]:
            lcs.append(X[i-1])
            i -= 1
            j -= 1
            k -= 1
        elif L[i-1][j][k] >= L[i][j-1][k] and L[i-1][j][k] >= L[i][j][k-1]:
            i -= 1
        elif L[i][j-1][k] >= L[i][j][k-1]:
            j -= 1
        else:
            k -= 1
    return ''.join(reversed(lcs))

'''
Module for calculating the longest common subsequence of three strings.
'''
def longest_common_subsequence(X, Y, Z):
    m = len(X)
    n = len(Y)
    o = len(Z)
    # Create a 3D array to store lengths of longest common subsequence
    L = [[[0 for _ in range(o+1)] for _ in range(n+1)] for _ in range(m+1)]
    # Build the LCS table in bottom-up fashion
    for i in range(m+1):
        for j in range(n+1):
            for k in range(o+1):
                if i == 0 or j == 0 or k == 0:
                    L[i][j][k] = 0
                elif X[i-1] == Y[j-1] == Z[k-1]:
                    L[i][j][k] = L[i-1][j-1][k-1] + 1
                else:
                    L[i][j][k] = max(L[i-1][j][k], L[i][j-1][k], L[i][j][k-1])
    # The length of the longest common subsequence
    lcs_length = L[m][n][o]
    # Retrieve the longest common subsequence
    lcs = []
    i, j, k = m, n, o
    while i > 0 and j > 0 and k > 0:
        if X[i-1] == Y[j-1] == Z[k-1]:
            lcs.append(X[i-1])
            i -= 1
            j -= 1
            k -= 1
        elif L[i-1][j][k] >= L[i][j-1][k] and L[i-1][j][k] >= L[i][j][k-1]:
            i -= 1
        elif L[i][j-1][k] >= L[i][j][k-1]:
            j -= 1
        else:
            k -= 1
    return ''.join(reversed(lcs))

        elif L[i-1][j][k] >= L[i][j-1][k] and L[i-1][j][k] >= L[i][j][k-1]:
            i -= 1
        elif L[i][j-1][k] >= L[i][j][k-1]:
            j -= 1
        else:
            k -= 1
    return ''.join(reversed(lcs))
