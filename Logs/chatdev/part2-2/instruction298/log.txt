'''
Main application file for counting lists within a list using a GUI.
'''
import tkinter as tk
from tkinter import messagebox
def count_lists_in_list(input_list):
    '''
    Count the occurrences of each list within a list and return a dictionary with tuples as keys.
    Parameters:
    input_list (list): A list containing other lists.
    Returns:
    dict: A dictionary where keys are tuples converted from lists and values are their counts.
    '''
    count_dict = {}
    for lst in input_list:
        if isinstance(lst, list):
            lst_tuple = tuple(lst)
            if lst_tuple in count_dict:
                count_dict[lst_tuple] += 1
            else:
                count_dict[lst_tuple] = 1
    return count_dict
class ListCounterApp:
    '''
    A GUI application to count lists within a list.
    '''
    def __init__(self, root):
        self.root = root
        self.root.title("List Counter App")
        # Input label and entry
        self.label = tk.Label(root, text="Enter a list of lists:")
        self.label.pack()
        self.entry = tk.Entry(root, width=50)
        self.entry.pack()
        # Count button
        self.count_button = tk.Button(root, text="Count Lists", command=self.count_lists)
        self.count_button.pack()
        # Result label
        self.result_label = tk.Label(root, text="")
        self.result_label.pack()
    def count_lists(self):
        '''
        Process the input list and display the result.
        '''
        try:
            # Evaluate the input to convert it to a list
            input_list = eval(self.entry.get())
            if not isinstance(input_list, list):
                raise ValueError("Input is not a list.")
            # Get the count of lists
            result = count_lists_in_list(input_list)
            # Display the result
            self.result_label.config(text=f"Result: {result}")
        except Exception as e:
            messagebox.showerror("Error", f"Invalid input: {e}")
if __name__ == "__main__":
    root = tk.Tk()
    app = ListCounterApp(root)
    root.mainloop()

'''
Main application file for counting lists within a list using a GUI.
'''
import tkinter as tk
from tkinter import messagebox
def count_lists_in_list(input_list):
    '''
    Count the occurrences of each list within a list and return a dictionary with tuples as keys.
    Parameters:
    input_list (list): A list containing other lists.
    Returns:
    dict: A dictionary where keys are tuples converted from lists and values are their counts.
    '''
    count_dict = {}
    for lst in input_list:
        if isinstance(lst, list):
            lst_tuple = tuple(lst)
            if lst_tuple in count_dict:
                count_dict[lst_tuple] += 1
            else:
                count_dict[lst_tuple] = 1
    return count_dict
class ListCounterApp:
    '''
    A GUI application to count lists within a list.
    '''
    def __init__(self, root):
        self.root = root
        self.root.title("List Counter App")
        # Input label and entry
        self.label = tk.Label(root, text="Enter a list of lists:")
        self.label.pack()
        self.entry = tk.Entry(root, width=50)
        self.entry.pack()
        # Count button
        self.count_button = tk.Button(root, text="Count Lists", command=self.count_lists)
        self.count_button.pack()
        # Result label
        self.result_label = tk.Label(root, text="")
        self.result_label.pack()
    def count_lists(self):
        '''
        Process the input list and display the result.
        '''
        try:
            # Evaluate the input to convert it to a list
            input_list = eval(self.entry.get())
            if not isinstance(input_list, list):
                raise ValueError("Input is not a list.")
            # Check if all elements are lists
            for item in input_list:
                if not isinstance(item, list):
                    raise ValueError("All elements of the input list must be lists.")
            # Get the count of lists
            result = count_lists_in_list(input_list)
            # Display the result
            self.result_label.config(text=f"Result: {result}")
        except Exception as e:
            messagebox.showerror("Error", f"Invalid input: {e}")
if __name__ == "__main__":
    root = tk.Tk()
    app = ListCounterApp(root)
    root.mainloop()

'''
Main application file for counting lists within a list using a GUI.
'''
import tkinter as tk
from tkinter import messagebox
def count_lists_in_list(input_list):
    '''
    Count the occurrences of each list within a list and return a dictionary with tuples as keys.
    Parameters:
    input_list (list): A list containing other lists.
    Returns:
    dict: A dictionary where keys are tuples converted from lists and values are their counts.
    '''
    count_dict = {}
    for lst in input_list:
        if isinstance(lst, list):
            lst_tuple = tuple(lst)
            if lst_tuple in count_dict:
                count_dict[lst_tuple] += 1
            else:
                count_dict[lst_tuple] = 1
    return count_dict
class ListCounterApp:
    '''
    A GUI application to count lists within a list.
    '''
    def __init__(self, root):
        self.root = root
        self.root.title("List Counter App")
        # Input label and entry
        self.label = tk.Label(root, text="Enter a list of lists:")
        self.label.pack()
        self.entry = tk.Entry(root, width=50)
        self.entry.pack()
        # Count button
        self.count_button = tk.Button(root, text="Count Lists", command=self.count_lists)
        self.count_button.pack()
        # Result label
        self.result_label = tk.Label(root, text="")
        self.result_label.pack()
    def count_lists(self):
        '''
        Process the input list and display the result.
        '''
        try:
            # Evaluate the input to convert it to a list
            input_list = eval(self.entry.get())
            if not isinstance(input_list, list):
                raise ValueError("Input is not a list.")
            # Check if all elements are lists
            for item in input_list:
                if not isinstance(item, list):
                    raise ValueError("All elements of the input list must be lists.")
            # Get the count of lists
            result = count_lists_in_list(input_list)
            # Display the result
            self.result_label.config(text=f"Result: {result}")
        except Exception as e:
            messagebox.showerror("Error", f"Invalid input: {e}")
if __name__ == "__main__":
    root = tk.Tk()
    app = ListCounterApp(root)
    root.mainloop()
