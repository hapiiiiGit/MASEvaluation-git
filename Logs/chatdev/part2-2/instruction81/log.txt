'''

This module contains the LucasCalculator class which provides a method to calculate the n'th Lucas number.

'''

class LucasCalculator:
    def calculate_lucas_number(self, n):
        """
        Calculate the n'th Lucas number.
        
        Parameters:
        n (int): The position of the Lucas number to calculate.
        
        Returns:
        int: The n'th Lucas number.
        """
        if n == 0:
            return 2
        elif n == 1:
            return 1
        else:
            a, b = 2, 1
            for _ in range(2, n + 1):
                a, b = b, a + b
            return b
```

class LucasCalculator:

class LucasCalculator:
    def calculate_lucas_number(self, n):
        """
        Calculate the n'th Lucas number.
        Parameters:
        n (int): The position of the Lucas number to calculate.

        int: The n'th Lucas number.

class LucasCalculator:
    def calculate_lucas_number(self, n):
        """
        Calculate the n'th Lucas number.
        Parameters:
        n (int): The position of the Lucas number to calculate.
        Returns:
        int: The n'th Lucas number.
        """
        if n == 0:
            return 2
        elif n == 1:

            a, b = 2, 1
            for _ in range(2, n + 1):
                a, b = b, a + b
