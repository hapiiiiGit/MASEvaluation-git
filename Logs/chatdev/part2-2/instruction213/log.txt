'''

This file contains the main application logic for calculating the nth Bell number. It includes a GUI implemented using tkinter and a function to compute Bell numbers.

'''

import tkinter as tk
from tkinter import messagebox

def bell_number(n):
    '''
    Calculate the nth Bell number using dynamic programming.
    
    Parameters:
        n (int): The position of the Bell number to calculate.
    
    Returns:
        int: The nth Bell number.
    '''
    if n < 0:
        raise ValueError("n must be a non-negative integer.")
    
    # Create a table to store Bell numbers
    bell = [[0 for _ in range(n + 1)] for _ in range(n + 1)]
    bell[0][0] = 1
    
    for i in range(1, n + 1):
        # First value in each row is the last value of the previous row
        bell[i][0] = bell[i - 1][i - 1]
        
        for j in range(1, i + 1):
            bell[i][j] = bell[i - 1][j - 1] + bell[i][j - 1]
    
    return bell[n][0]

class App:
    '''
    A class representing the GUI application for calculating Bell numbers.
    '''
    def __init__(self, root):
        '''
        Initialize the GUI components.
        
        Parameters:
            root (tk.Tk): The root window of the tkinter application.
        '''
        self.root = root
        self.root.title("Bell Number Calculator")
        
        # Input label and entry
        self.label = tk.Label(root, text="Enter n (non-negative integer):")
        self.label.pack(pady=10)
        
        self.entry = tk.Entry(root)
        self.entry.pack(pady=10)
        
        # Calculate button
        self.button = tk.Button(root, text="Calculate", command=self.calculate_bell_number)
        self.button.pack(pady=10)
        
        # Result label
        self.result_label = tk.Label(root, text="", fg="blue")
        self.result_label.pack(pady=10)
    
    def calculate_bell_number(self):
        '''
        Handle the calculation of the Bell number based on user input and update the GUI with the result.
        '''
        try:
            n = int(self.entry.get())
            if n < 0:
                raise ValueError("n must be a non-negative integer.")
            
            result = bell_number(n)
            self.result_label.config(text=f"The {n}th Bell number is: {result}")
        except ValueError as e:
            messagebox.showerror("Invalid Input", str(e))

if __name__ == "__main__":
    root = tk.Tk()
    app = App(root)
    root.mainloop()

```python

'''

This file contains the main application logic for calculating the nth Bell number. It includes a GUI implemented using tkinter and a function to compute Bell numbers.

'''

import tkinter as tk
from tkinter import messagebox

def bell_number(n):
    '''
    Calculate the nth Bell number using dynamic programming.
    
    Parameters:
        n (int): The position of the Bell number to calculate.
    
    Returns:
        int: The nth Bell number.
    '''
    if n < 0:
        raise ValueError("n must be a non-negative integer.")
    
    # Create a table to store Bell numbers
    bell = [[0 for _ in range(n + 1)] for _ in range(n + 1)]
    bell[0][0] = 1
    
    for i in range(1, n + 1):
        # First value in each row is the last value of the previous row
        bell[i][0] = bell[i - 1][i - 1]
        
        for j in range(1, i + 1):
            bell[i][j] = bell[i - 1][j - 1] + bell[i][j - 1]
    
    return bell[n][0]

class App:
    '''
    A class representing the GUI application for calculating Bell numbers.
    '''
    def __init__(self, root):
        '''
        Initialize the GUI components.
        
        Parameters:
            root (tk.Tk): The root window of the tkinter application.
        '''
        self.root = root
        self.root.title("Bell Number Calculator")
        
        # Input label and entry
        self.label = tk.Label(root, text="Enter n (non-negative integer):")
        self.label.pack(pady=10)
        
        self.entry = tk.Entry(root)
        self.entry.pack(pady=10)
        
        # Calculate button
        self.button = tk.Button(root, text="Calculate", command=self.calculate_bell_number)
        self.button.pack(pady=10)
        
        # Result label
        self.result_label = tk.Label(root, text="", fg="blue")
        self.result_label.pack(pady=10)
    
    def calculate_bell_number(self):
        '''
        Handle the calculation of the Bell number based on user input and update the GUI with the result.
        '''
        try:
            n = int(self.entry.get())
            if n < 0:
                raise ValueError("n must be a non-negative integer.")
            
            result = bell_number(n)
            self.result_label.config(text=f"The {n}th Bell number is: {result}")
        except ValueError as e:
            messagebox.showerror("Invalid Input", str(e))

if __name__ == "__main__":
    root = tk.Tk()
    app = App(root)
    root.mainloop()

import tkinter as tk
from tkinter import messagebox
def bell_number(n):
    '''
    Calculate the nth Bell number using dynamic programming.
    Parameters:
        n (int): The position of the Bell number to calculate.
    Returns:
        int: The nth Bell number.
    '''
    if n < 0:
        raise ValueError("n must be a non-negative integer.")
    # Create a table to store Bell numbers
    bell = [[0 for _ in range(n + 1)] for _ in range(n + 1)]
    bell[0][0] = 1
    for i in range(1, n + 1):
        # First value in each row is the last value of the previous row
        bell[i][0] = bell[i - 1][i - 1]
        for j in range(1, i + 1):
            bell[i][j] = bell[i - 1][j - 1] + bell[i][j - 1]
    return bell[n][0]

'''

'''
import tkinter as tk
from tkinter import messagebox
def bell_number(n):
    '''
    Calculate the nth Bell number using dynamic programming.
    Parameters:
        n (int): The position of the Bell number to calculate.
    Returns:
        int: The nth Bell number.
    '''
    if n < 0:
        raise ValueError("n must be a non-negative integer.")
    # Create a table to store Bell numbers
    bell = [[0 for _ in range(n + 1)] for _ in range(n + 1)]
    bell[0][0] = 1
    for i in range(1, n + 1):
        # First value in each row is the last value of the previous row
        bell[i][0] = bell[i - 1][i - 1]
        for j in range(1, i + 1):
            bell[i][j] = bell[i - 1][j - 1] + bell[i][j - 1]
    return bell[n][0]
class App:
    '''
    A class representing the GUI application for calculating Bell numbers.
    '''
    def __init__(self, root):
        '''
        Initialize the GUI components.
        Parameters:
            root (tk.Tk): The root window of the tkinter application.
        '''
        self.root = root
        self.root.title("Bell Number Calculator")
        # Input label and entry
        self.label = tk.Label(root, text="Enter n (non-negative integer):")
        self.label.pack(pady=10)
        self.entry = tk.Entry(root)
        self.entry.pack(pady=10)
        # Calculate button
        self.button = tk.Button(root, text="Calculate", command=self.calculate_bell_number)
        self.button.pack(pady=10)
        # Result label
        self.result_label = tk.Label(root, text="", fg="blue")
        self.result_label.pack(pady=10)
    def calculate_bell_number(self):
        '''
        Handle the calculation of the Bell number based on user input and update the GUI with the result.
        '''
        try:
            n = int(self.entry.get())
            if n < 0:
                raise ValueError("n must be a non-negative integer.")
            result = bell_number(n)
            self.result_label.config(text=f"The {n}th Bell number is: {result}")
        except ValueError as e:
            messagebox.showerror("Invalid Input", str(e))
if __name__ == "__main__":
    root = tk.Tk()
    app = App(root)
    root.mainloop()

'''
This file contains the main application logic for calculating the nth Bell number. It includes a GUI implemented using tkinter and a function to compute Bell numbers.
'''
import tkinter as tk
from tkinter import messagebox
def bell_number(n):
    '''
    Calculate the nth Bell number using dynamic programming.
    Parameters:
        n (int): The position of the Bell number to calculate.
    Returns:
        int: The nth Bell number.
    '''
    if n < 0:
        raise ValueError("n must be a non-negative integer.")
    # Create a table to store Bell numbers
    bell = [[0 for _ in range(n + 1)] for _ in range(n + 1)]
    bell[0][0] = 1
    for i in range(1, n + 1):
        # First value in each row is the last value of the previous row
        bell[i][0] = bell[i - 1][i - 1]
        for j in range(1, i + 1):
            bell[i][j] = bell[i - 1][j - 1] + bell[i][j - 1]
    return bell[n][0]
class App:
    '''
    A class representing the GUI application for calculating Bell numbers.
    '''
    def __init__(self, root):
        '''
        Initialize the GUI components.
        Parameters:
            root (tk.Tk): The root window of the tkinter application.
        '''
        self.root = root
        self.root.title("Bell Number Calculator")
        # Input label and entry
        self.label = tk.Label(root, text="Enter n (non-negative integer):")
        self.label.pack(pady=10)
        self.entry = tk.Entry(root)
        self.entry.pack(pady=10)
        # Calculate button
        self.button = tk.Button(root, text="Calculate", command=self.calculate_bell_number)
        self.button.pack(pady=10)
        # Result label
        self.result_label = tk.Label(root, text="", fg="blue")
        self.result_label.pack(pady=10)
    def calculate_bell_number(self):
        '''
        Handle the calculation of the Bell number based on user input and update the GUI with the result.
        '''
        try:
            n = int(self.entry.get())
            if n < 0:
                raise ValueError("n must be a non-negative integer.")
            result = bell_number(n)
            self.result_label.config(text=f"The {n}th Bell number is: {result}")
        except ValueError as e:
            messagebox.showerror("Invalid Input", str(e))
if __name__ == "__main__":
    root = tk.Tk()
    app = App(root)
    root.mainloop()
