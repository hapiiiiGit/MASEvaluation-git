'''

main.py

This file contains the MainApp class for the GUI application and the max_subarray_sum function to calculate the sum of the largest contiguous sublist.

'''

import tkinter as tk
from tkinter import messagebox

def max_subarray_sum(nums):
    '''
    Calculate the sum of the largest contiguous sublist using Kadane's Algorithm.

    :param nums: List of integers
    :return: Integer sum of the largest contiguous sublist
    '''
    if not nums:
        return 0

    max_current = max_global = nums[0]

    for num in nums[1:]:
        max_current = max(num, max_current + num)
        if max_current > max_global:
            max_global = max_current

    return max_global

class MainApp:
    '''
    Main application class for the GUI.
    '''
    def __init__(self, root):
        self.root = root
        self.root.title("Max Subarray Sum Calculator")

        self.label = tk.Label(root, text="Enter numbers separated by commas:")
        self.label.pack()

        self.entry = tk.Entry(root, width=50)
        self.entry.pack()

        self.calculate_button = tk.Button(root, text="Calculate", command=self.calculate_max_subarray_sum)
        self.calculate_button.pack()

        self.result_label = tk.Label(root, text="")
        self.result_label.pack()

    def calculate_max_subarray_sum(self):
        '''
        Calculate and display the sum of the largest contiguous sublist.
        '''
        try:
            input_text = self.entry.get()
            nums = list(map(int, input_text.split(',')))
            result = max_subarray_sum(nums)
            self.result_label.config(text=f"Max Subarray Sum: {result}")
        except ValueError:
            messagebox.showerror("Invalid input", "Please enter a valid list of integers separated by commas.")

if __name__ == "__main__":
    root = tk.Tk()
    app = MainApp(root)
    root.mainloop()
```

'''

main.py

This file contains the MainApp class for the GUI application and the max_subarray_sum function to calculate the sum of the largest contiguous sublist.

'''

import tkinter as tk
from tkinter import messagebox

def max_subarray_sum(nums):
    '''
    Calculate the sum of the largest contiguous sublist using Kadane's Algorithm.

    :param nums: List of integers
    :return: Integer sum of the largest contiguous sublist
    '''
    if not nums:
        return 0

    max_current = max_global = nums[0]

    for num in nums[1:]:
        max_current = max(num, max_current + num)
        if max_current > max_global:
            max_global = max_current

    return max_global

class MainApp:
    '''
    Main application class for the GUI.
    '''
    def __init__(self, root):
        self.root = root
        self.root.title("Max Subarray Sum Calculator")

        self.label = tk.Label(root, text="Enter numbers separated by commas:")
        self.label.pack()

        self.entry = tk.Entry(root, width=50)
        self.entry.pack()

        self.calculate_button = tk.Button(root, text="Calculate", command=self.calculate_max_subarray_sum)
        self.calculate_button.pack()

        self.result_label = tk.Label(root, text="")
        self.result_label.pack()

    def calculate_max_subarray_sum(self):
        '''
        Calculate and display the sum of the largest contiguous sublist.
        '''
        try:
            input_text = self.entry.get()
            nums = list(map(int, input_text.split(',')))
            result = max_subarray_sum(nums)
            self.result_label.config(text=f"Max Subarray Sum: {result}")
        except ValueError:
            messagebox.showerror("Invalid input", "Please enter a valid list of integers separated by commas.")

if __name__ == "__main__":
    root = tk.Tk()
    app = MainApp(root)
    root.mainloop()
```

main.py
This file contains the MainApp class for the GUI application and the max_subarray_sum function to calculate the sum of the largest contiguous sublist.
'''
import tkinter as tk
from tkinter import messagebox

def max_subarray_sum(nums):
    '''
    Calculate the sum of the largest contiguous sublist using Kadane's Algorithm.
    :param nums: List of integers
    :return: Integer sum of the largest contiguous sublist
    '''
    if not nums:
        return 0
    max_current = max_global = nums[0]
    for num in nums[1:]:
        max_current = max(num, max_current + num)
        if max_current > max_global:
            max_global = max_current
    return max_global

class MainApp:
    '''
    Main application class for the GUI.
    '''
    def __init__(self, root):
        self.root = root
        self.root.title("Max Subarray Sum Calculator")
        self.label = tk.Label(root, text="Enter numbers separated by commas:")
        self.label.pack()
        self.entry = tk.Entry(root, width=50)
        self.entry.pack()
        self.calculate_button = tk.Button(root, text="Calculate", command=self.calculate_max_subarray_sum)
        self.calculate_button.pack()
        self.result_label = tk.Label(root, text="")
        self.result_label.pack()

    def calculate_max_subarray_sum(self):
        '''
        Calculate and display the sum of the largest contiguous sublist.
        '''
        try:
            input_text = self.entry.get()
            nums = list(map(int, input_text.split(',')))
            result = max_subarray_sum(nums)
            self.result_label.config(text=f"Max Subarray Sum: {result}")
        except ValueError:
            messagebox.showerror("Invalid input", "Please enter a valid list of integers separated by commas.")

if __name__ == "__main__":
    root = tk.Tk()
    app = MainApp(root)
    root.mainloop()

import tkinter as tk
from tkinter import messagebox
def max_subarray_sum(nums):
    '''
    Calculate the sum of the largest contiguous sublist using Kadane's Algorithm.
    :param nums: List of integers
    :return: Integer sum of the largest contiguous sublist
    '''
    if not nums:
        return 0
    max_current = max_global = nums[0]
    for num in nums[1:]:
        max_current = max(num, max_current + num)
        if max_current > max_global:
            max_global = max_current
    return max_global

    Main application class for the GUI.

        self.result_label.pack()
    def calculate_max_subarray_sum(self):
        '''
        Calculate and display the sum of the largest contiguous sublist.
        '''
        try:
            input_text = self.entry.get()
            nums = list(map(int, input_text.split(',')))

            self.result_label.config(text=f"Max Subarray Sum: {result}")

import tkinter as tk
from tkinter import messagebox
def max_subarray_sum(nums):
    '''
    Calculate the sum of the largest contiguous sublist using Kadane's Algorithm.
    :param nums: List of integers
    :return: Integer sum of the largest contiguous sublist
    '''
    if not nums:
        return 0
    max_current = max_global = nums[0]
    for num in nums[1:]:
        max_current = max(num, max_current + num)
        if max_current > max_global:
            max_global = max_current
    return max_global

    Main application class for the GUI.
    '''

        '''

import tkinter as tk
from tkinter import messagebox
def max_subarray_sum(nums):
    '''
    Calculate the sum of the largest contiguous sublist using Kadane's Algorithm.
    :param nums: List of integers
    :return: Integer sum of the largest contiguous sublist
    '''
    if not nums:
        return 0
    max_current = max_global = nums[0]
    for num in nums[1:]:
        max_current = max(num, max_current + num)
        if max_current > max_global:
            max_global = max_current
    return max_global
class MainApp:
    '''
    Main application class for the GUI.
    '''
    def __init__(self, root):
        self.root = root
        self.root.title("Max Subarray Sum Calculator")
        self.label = tk.Label(root, text="Enter numbers separated by commas:")
        self.label.pack()
        self.entry = tk.Entry(root, width=50)
        self.entry.pack()
        self.calculate_button = tk.Button(root, text="Calculate", command=self.calculate_max_subarray_sum)
        self.calculate_button.pack()
        self.result_label = tk.Label(root, text="")
        self.result_label.pack()
    def calculate_max_subarray_sum(self):
        '''
        Calculate and display the sum of the largest contiguous sublist.
        '''
        try:
            input_text = self.entry.get()
            nums = list(map(int, input_text.split(',')))
            result = max_subarray_sum(nums)
            self.result_label.config(text=f"Max Subarray Sum: {result}")
        except ValueError:
            messagebox.showerror("Invalid input", "Please enter a valid list of integers separated by commas.")
if __name__ == "__main__":
    root = tk.Tk()
    app = MainApp(root)
    root.mainloop()
