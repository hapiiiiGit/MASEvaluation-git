'''

This file contains the main entry point for the Odd Number Remover application. It initializes the GUI and starts the application.

'''

import tkinter as tk
from tkinter import messagebox


class OddNumberRemoverApp:
    '''
    A class to create a GUI application that removes odd numbers from a given list.
    '''

    def __init__(self, root):
        '''
        Initializes the GUI components.

        Parameters:
        root (tk.Tk): The root window of the application.
        '''
        self.root = root
        self.root.title("Odd Number Remover")

        # Input label and entry
        self.label = tk.Label(root, text="Enter a list of numbers (comma-separated):")
        self.label.pack(pady=10)

        self.input_entry = tk.Entry(root, width=50)
        self.input_entry.pack(pady=10)

        # Remove button
        self.remove_button = tk.Button(root, text="Remove Odd Numbers", command=self.remove_odd_numbers)
        self.remove_button.pack(pady=10)

        # Result label
        self.result_label = tk.Label(root, text="Result: ", wraplength=400, justify="left")
        self.result_label.pack(pady=10)

    def remove_odd_numbers(self):
        '''
        Removes odd numbers from the input list and updates the result label.
        '''
        input_text = self.input_entry.get()

        try:
            # Convert the input string to a list of integers
            number_list = [int(num.strip()) for num in input_text.split(",")]

            # Filter out odd numbers
            even_numbers = [num for num in number_list if num % 2 == 0]

            # Update the result label
            self.result_label.config(text=f"Result: {even_numbers}")

        except ValueError:
            # Handle invalid input
            messagebox.showerror("Invalid Input", "Please enter a valid list of numbers (comma-separated).")


def main():
    '''
    The main entry point of the application.
    '''
    root = tk.Tk()
    app = OddNumberRemoverApp(root)
    root.mainloop()


if __name__ == "__main__":
    main()
```

This file contains the main entry point for the Odd Number Remover application. It initializes the GUI and starts the application.

'''

import tkinter as tk
from tkinter import messagebox


class OddNumberRemoverApp:
    '''
    A class to create a GUI application that removes odd numbers from a given list.
    '''

    def __init__(self, root):
        '''
        Initializes the GUI components.

        Parameters:
        root (tk.Tk): The root window of the application.
        '''
        self.root = root
        self.root.title("Odd Number Remover")

        # Input label and entry
        self.label = tk.Label(root, text="Enter a list of numbers (comma-separated):")
        self.label.pack(pady=10)

        self.input_entry = tk.Entry(root, width=50)
        self.input_entry.pack(pady=10)

        # Remove button
        self.remove_button = tk.Button(root, text="Remove Odd Numbers", command=self.remove_odd_numbers)
        self.remove_button.pack(pady=10)

        # Result label
        self.result_label = tk.Label(root, text="Result: ", wraplength=400, justify="left")
        self.result_label.pack(pady=10)

    def remove_odd_numbers(self):
        '''
        Removes odd numbers from the input list and updates the result label.
        '''
        input_text = self.input_entry.get()

        try:
            # Convert the input string to a list of integers
            number_list = [int(num.strip()) for num in input_text.split(",")]

            # Filter out odd numbers
            even_numbers = [num for num in number_list if num % 2 == 0]

            # Update the result label
            self.result_label.config(text=f"Result: {even_numbers}")

        except ValueError:
            # Handle invalid input
            messagebox.showerror("Invalid Input", "Please enter a valid list of numbers (comma-separated).")


def main():
    '''
    The main entry point of the application.
    '''
    root = tk.Tk()
    app = OddNumberRemoverApp(root)
    root.mainloop()

import tkinter as tk
from tkinter import messagebox
class OddNumberRemoverApp:
    '''
    A class to create a GUI application that removes odd numbers from a given list.
    '''
    def __init__(self, root):
        '''
        Initializes the GUI components.
        Parameters:
        root (tk.Tk): The root window of the application.
        '''
        self.root = root
        self.root.title("Odd Number Remover")
        # Input label and entry
        self.label = tk.Label(root, text="Enter a list of numbers (comma-separated):")
        self.label.pack(pady=10)
        self.input_entry = tk.Entry(root, width=50)
        self.input_entry.pack(pady=10)
        # Remove button
        self.remove_button = tk.Button(root, text="Remove Odd Numbers", command=self.remove_odd_numbers)
        self.remove_button.pack(pady=10)
        # Result label
        self.result_label = tk.Label(root, text="Result: ", wraplength=400, justify="left")
        self.result_label.pack(pady=10)
    def remove_odd_numbers(self):
        '''
        Removes odd numbers from the input list and updates the result label.
        '''
        input_text = self.input_entry.get()
        try:
            # Convert the input string to a list of integers
            number_list = [int(num.strip()) for num in input_text.split(",")]
            # Filter out odd numbers
            even_numbers = [num for num in number_list if num % 2 == 0]
            # Update the result label
            self.result_label.config(text=f"Result: {even_numbers}")
        except ValueError:
            # Handle invalid input
            messagebox.showerror("Invalid Input", "Please enter a valid list of numbers (comma-separated).")
def main():
    '''
    The main entry point of the application.
    '''
    root = tk.Tk()
    app = OddNumberRemoverApp(root)
    root.mainloop()
if __name__ == "__main__":
    main()

'''
This file contains the main entry point for the Odd Number Remover application. It initializes the GUI and starts the application.
'''
import tkinter as tk
from tkinter import messagebox
class OddNumberRemoverApp:
    '''
    A class to create a GUI application that removes odd numbers from a given list.
    '''
    def __init__(self, root):
        '''
        Initializes the GUI components.
        Parameters:
        root (tk.Tk): The root window of the application.
        '''
        self.root = root
        self.root.title("Odd Number Remover")
        # Input label and entry
        self.label = tk.Label(root, text="Enter a list of numbers (comma-separated):")
        self.label.pack(pady=10)
        self.input_entry = tk.Entry(root, width=50)
        self.input_entry.pack(pady=10)
        # Remove button
        self.remove_button = tk.Button(root, text="Remove Odd Numbers", command=self.remove_odd_numbers)
        self.remove_button.pack(pady=10)
        # Result label
        self.result_label = tk.Label(root, text="Result: ", wraplength=400, justify="left")
        self.result_label.pack(pady=10)
    def remove_odd_numbers(self):
        '''
        Removes odd numbers from the input list and updates the result label.
        '''
        input_text = self.input_entry.get()
        try:
            # Convert the input string to a list of integers
            number_list = [int(num.strip()) for num in input_text.split(",")]
            # Filter out odd numbers
            even_numbers = [num for num in number_list if num % 2 == 0]
            # Update the result label
            self.result_label.config(text=f"Result: {even_numbers}")
        except ValueError:
            # Handle invalid input
            messagebox.showerror("Invalid Input", "Please enter a valid list of numbers (comma-separated).")
def main():
    '''
    The main entry point of the application.
    '''
    root = tk.Tk()
    app = OddNumberRemoverApp(root)
    root.mainloop()
if __name__ == "__main__":
    main()

# Odd Number Remover Application

Welcome to the Odd Number Remover Application! This application provides a simple graphical user interface (GUI) to remove odd numbers from a list of numbers you provide. It's designed to be user-friendly and requires no external dependencies.

## Main Functions of the Software

The Odd Number Remover Application allows users to:

- **Input a List of Numbers**: Enter a list of numbers separated by commas.
- **Remove Odd Numbers**: With a click of a button, the application processes the list and removes all odd numbers.
- **Display Results**: The application displays the resulting list of even numbers.

## Installation and Environment Setup

### Requirements

This application does not require any external dependencies beyond Python's standard library. It uses the `tkinter` module, which is included with Python.

### Quick Install

1. **Ensure Python is Installed**: Make sure you have Python installed on your system. You can download it from [python.org](https://www.python.org/downloads/).

2. **Download the Application Code**: Save the `main.py` file to a directory of your choice.

3. **Run the Application**: Open a terminal or command prompt, navigate to the directory where `main.py` is saved, and run the following command:

   ```bash
   python main.py
   ```

This will launch the Odd Number Remover application.

## How to Use the Application

1. **Launch the Application**: Run the `main.py` file as described above.

2. **Enter Numbers**: In the application window, you will see a text entry field. Enter a list of numbers separated by commas (e.g., `1, 2, 3, 4, 5`).

3. **Remove Odd Numbers**: Click the "Remove Odd Numbers" button. The application will process your input and remove all odd numbers from the list.

4. **View Results**: The resulting list of even numbers will be displayed below the button.

5. **Error Handling**: If you enter invalid input (e.g., letters or symbols), the application will show an error message prompting you to enter a valid list of numbers.

Enjoy using the Odd Number Remover Application to quickly and easily filter out odd numbers from your lists!
