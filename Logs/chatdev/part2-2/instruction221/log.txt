```python
'''

Main application file for dividing two lists element-wise using a GUI.

'''

import tkinter as tk
from tkinter import messagebox

class MainApplication:
    def __init__(self, root):
        self.root = root
        self.root.title("Element-wise List Divider")

        # Input fields for lists
        self.label1 = tk.Label(root, text="Enter first list (comma-separated):")
        self.label1.pack()
        self.entry1 = tk.Entry(root, width=50)
        self.entry1.pack()

        self.label2 = tk.Label(root, text="Enter second list (comma-separated):")
        self.label2.pack()
        self.entry2 = tk.Entry(root, width=50)
        self.entry2.pack()

        # Calculate button
        self.calculate_button = tk.Button(root, text="Calculate", command=self.on_calculate)
        self.calculate_button.pack()

        # Result display
        self.result_label = tk.Label(root, text="Result:")
        self.result_label.pack()
        self.result_display = tk.Label(root, text="", fg="blue")
        self.result_display.pack()

    def on_calculate(self):
        list1_str = self.entry1.get()
        list2_str = self.entry2.get()

        try:
            list1 = self.validate_input(list1_str)
            list2 = self.validate_input(list2_str)

            result = self.divide_lists(list1, list2)
            self.result_display.config(text=str(result))
        except ValueError as e:
            messagebox.showerror("Input Error", str(e))

    def validate_input(self, input_str):
        try:
            return [float(x.strip()) for x in input_str.split(',')]
        except ValueError:
            raise ValueError("Invalid input. Please enter numbers separated by commas.")

    def divide_lists(self, list1, list2):
        if len(list1) != len(list2):
            raise ValueError("Lists must be of the same length.")
        
        result = []
        for a, b in zip(list1, list2):
            if b == 0:
                raise ValueError("Division by zero encountered in the second list.")
            result.append(a / b)
        
        return result

if __name__ == "__main__":
    root = tk.Tk()
    app = MainApplication(root)
    root.mainloop()
```

- **validate_input**: Converts a comma-separated string into a list of floats. Raises an error if the input is invalid.
- **divide_lists**: Performs element-wise division of two lists. It checks for division by zero and ensures both lists are of the same length.

import tkinter as tk
from tkinter import messagebox

class MainApplication:
    def __init__(self, root):
        self.root = root
        self.root.title("Element-wise List Divider")

        # Input fields for lists
        self.label1 = tk.Label(root, text="Enter first list (comma-separated):")
        self.label1.pack()
        self.entry1 = tk.Entry(root, width=50)
        self.entry1.pack()

        self.label2 = tk.Label(root, text="Enter second list (comma-separated):")
        self.label2.pack()
        self.entry2 = tk.Entry(root, width=50)
        self.entry2.pack()

        # Calculate button
        self.calculate_button = tk.Button(root, text="Calculate", command=self.on_calculate)
        self.calculate_button.pack()

        # Result display
        self.result_label = tk.Label(root, text="Result:")
        self.result_label.pack()
        self.result_display = tk.Label(root, text="", fg="blue")
        self.result_display.pack()

    def on_calculate(self):
        list1_str = self.entry1.get()
        list2_str = self.entry2.get()

        try:
            list1 = self.validate_input(list1_str)
            list2 = self.validate_input(list2_str)

            result = self.divide_lists(list1, list2)
            self.result_display.config(text=str(result))
        except ValueError as e:
            messagebox.showerror("Input Error", str(e))

    def validate_input(self, input_str):
        try:
            return [float(x.strip()) for x in input_str.split(',')]
        except ValueError:
            raise ValueError("Invalid input. Please enter numbers separated by commas.")

    def divide_lists(self, list1, list2):
        if len(list1) != len(list2):
            raise ValueError("Lists must be of the same length.")
        
        result = []
        for a, b in zip(list1, list2):
            if b == 0:
                raise ValueError("Division by zero encountered in the second list.")
            result.append(a / b)
        
        return result

if __name__ == "__main__":
    root = tk.Tk()

        self.label2 = tk.Label(root, text="Enter second list (comma-separated):")
        self.label2.pack()
        self.entry2 = tk.Entry(root, width=50)
        self.entry2.pack()
        # Calculate button
        self.calculate_button = tk.Button(root, text="Calculate", command=self.on_calculate)
        self.calculate_button.pack()
        # Result display
        self.result_label = tk.Label(root, text="Result:")
        self.result_label.pack()

        for a, b in zip(list1, list2):
            if b == 0:
                raise ValueError("Division by zero encountered in the second list.")

'''
Main application file for dividing two lists element-wise using a GUI.

import tkinter as tk
from tkinter import messagebox
class MainApplication:
    def __init__(self, root):
        self.root = root
        self.root.title("Element-wise List Divider")
        # Input fields for lists
        self.label1 = tk.Label(root, text="Enter first list (comma-separated):")
        self.label1.pack()
        self.entry1 = tk.Entry(root, width=50)
        self.entry1.pack()
        self.label2 = tk.Label(root, text="Enter second list (comma-separated):")
        self.label2.pack()
        self.entry2 = tk.Entry(root, width=50)
        self.entry2.pack()
        # Calculate button
        self.calculate_button = tk.Button(root, text="Calculate", command=self.on_calculate)
        self.calculate_button.pack()
        # Result display
        self.result_label = tk.Label(root, text="Result:")
        self.result_label.pack()
        self.result_display = tk.Label(root, text="", fg="blue")
        self.result_display.pack()
    def on_calculate(self):
        list1_str = self.entry1.get()
        list2_str = self.entry2.get()
        try:
            list1 = self.validate_input(list1_str)
            list2 = self.validate_input(list2_str)
            result = self.divide_lists(list1, list2)
            self.result_display.config(text=str(result))
        except ValueError as e:
            messagebox.showerror("Input Error", str(e))
        except Exception as e:
            messagebox.showerror("Error", f"An unexpected error occurred: {str(e)}")
    def validate_input(self, input_str):
        try:
            return [float(x.strip()) for x in input_str.split(',')]
        except ValueError:
            raise ValueError("Invalid input. Please enter numbers separated by commas.")
    def divide_lists(self, list1, list2):
        if len(list1) != len(list2):
            raise ValueError("Lists must be of the same length.")
        result = []
        for a, b in zip(list1, list2):
            if b == 0:
                raise ValueError("Division by zero encountered in the second list.")
            result.append(a / b)
        return result
if __name__ == "__main__":
    root = tk.Tk()
    app = MainApplication(root)
    root.mainloop()
