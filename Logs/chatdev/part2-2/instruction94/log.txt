'''

This script creates a GUI application using tkinter to find the minimum of two numbers.
The user can input two numbers, and the application will display the minimum of the two.

'''

import tkinter as tk
from tkinter import messagebox


def find_minimum(num1, num2):
    '''
    This function takes two numbers as input and returns the minimum of the two.
    '''
    try:
        # Convert inputs to floats and find the minimum
        num1 = float(num1)
        num2 = float(num2)
        return min(num1, num2)
    except ValueError:
        # Raise an error if inputs are not valid numbers
        raise ValueError("Both inputs must be valid numbers.")


def create_gui():
    '''
    This function creates and displays the GUI for the application.
    '''
    # Initialize the main window
    root = tk.Tk()
    root.title("Find Minimum of Two Numbers")
    root.geometry("400x200")

    # Labels and entry fields for input
    tk.Label(root, text="Enter first number:").grid(row=0, column=0, padx=10, pady=10)
    entry_num1 = tk.Entry(root)
    entry_num1.grid(row=0, column=1, padx=10, pady=10)

    tk.Label(root, text="Enter second number:").grid(row=1, column=0, padx=10, pady=10)
    entry_num2 = tk.Entry(root)
    entry_num2.grid(row=1, column=1, padx=10, pady=10)

    # Label to display the result
    result_label = tk.Label(root, text="Result: ", font=("Arial", 12))
    result_label.grid(row=3, column=0, columnspan=2, pady=10)

    # Function to handle the button click event
    def on_calculate():
        '''
        This function is triggered when the "Calculate Minimum" button is clicked.
        It retrieves the input values, calculates the minimum, and displays the result.
        '''
        num1 = entry_num1.get()
        num2 = entry_num2.get()
        try:
            # Calculate the minimum and display the result
            minimum = find_minimum(num1, num2)
            result_label.config(text=f"Result: Minimum is {minimum}")
        except ValueError as e:
            # Show an error message if inputs are invalid
            messagebox.showerror("Input Error", str(e))

    # Button to calculate the minimum

'''

This script creates a GUI application using tkinter to find the minimum of two numbers.
The user can input two numbers, and the application will display the minimum of the two.

'''

import tkinter as tk
from tkinter import messagebox


def find_minimum(num1, num2):
    '''
    This function takes two numbers as input and returns the minimum of the two.
    '''
    try:
        # Convert inputs to floats and find the minimum
        num1 = float(num1)
        num2 = float(num2)
        return min(num1, num2)
    except ValueError:
        # Raise an error if inputs are not valid numbers
        raise ValueError("Both inputs must be valid numbers.")


def create_gui():
    '''
    This function creates and displays the GUI for the application.
    '''
    # Initialize the main window
    root = tk.Tk()
    root.title("Find Minimum of Two Numbers")
    root.geometry("400x200")

    # Labels and entry fields for input
    tk.Label(root, text="Enter first number:").grid(row=0, column=0, padx=10, pady=10)
    entry_num1 = tk.Entry(root)
    entry_num1.grid(row=0, column=1, padx=10, pady=10)

    tk.Label(root, text="Enter second number:").grid(row=1, column=0, padx=10, pady=10)
    entry_num2 = tk.Entry(root)
    entry_num2.grid(row=1, column=1, padx=10, pady=10)

    # Label to display the result
    result_label = tk.Label(root, text="Result: ", font=("Arial", 12))
    result_label.grid(row=3, column=0, columnspan=2, pady=10)

    # Function to handle the button click event
    def on_calculate():
        '''
        This function is triggered when the "Calculate Minimum" button is clicked.
        It retrieves the input values, calculates the minimum, and displays the result.
        '''
        num1 = entry_num1.get()
        num2 = entry_num2.get()
        try:
            # Calculate the minimum and display the result
            minimum = find_minimum(num1, num2)
            result_label.config(text=f"Result: Minimum is {minimum}")
        except ValueError as e:
            # Show an error message if inputs are invalid
            messagebox.showerror("Input Error", str(e))

    # Button to calculate the minimum
    calculate_button = tk.Button(root, text="Calculate Minimum", command=on_calculate)
    calculate_button.grid(row=2, column=0, columnspan=2, pady=10)

    # Start the main event loop
    root.mainloop()


# Entry point of the application
if __name__ == "__main__":
    create_gui()

import tkinter as tk

    '''
    This function takes two numbers as input and returns the minimum of the two.
    '''
    try:
        # Convert inputs to floats and find the minimum
        num1 = float(num1)
        num2 = float(num2)
        return min(num1, num2)
    except ValueError:
        # Raise an error if inputs are not valid numbers
        raise ValueError("Both inputs must be valid numbers.")

    This function creates and displays the GUI for the application.
    '''

    root = tk.Tk()

    tk.Label(root, text="Enter first number:").grid(row=0, column=0, padx=10, pady=10)
    entry_num1 = tk.Entry(root)
    entry_num1.grid(row=0, column=1, padx=10, pady=10)

        This function is triggered when the "Calculate Minimum" button is clicked.

        '''
        num1 = entry_num1.get()
        num2 = entry_num2.get()

    # Button to calculate the minimum
    calculate_button = tk.Button(root, text="Calculate Minimum", command=on_calculate)

    root.mainloop()
# Entry point of the application
if __name__ == "__main__":

'''
This script creates a GUI application using tkinter to find the minimum of two numbers.
The user can input two numbers, and the application will display the minimum of the two.
'''
import tkinter as tk
from tkinter import messagebox
def find_minimum(num1, num2):
    '''
    This function takes two numbers as input and returns the minimum of the two.
    '''
    try:
        # Convert inputs to floats and find the minimum
        num1 = float(num1)
        num2 = float(num2)
        return min(num1, num2)
    except ValueError:
        # Raise an error if inputs are not valid numbers
        raise ValueError("Both inputs must be valid numbers.")
def create_gui():
    '''
    This function creates and displays the GUI for the application.
    '''
    # Initialize the main window
    root = tk.Tk()
    root.title("Find Minimum of Two Numbers")
    root.geometry("400x200")
    # Labels and entry fields for input
    tk.Label(root, text="Enter first number:").grid(row=0, column=0, padx=10, pady=10)
    entry_num1 = tk.Entry(root)
    entry_num1.grid(row=0, column=1, padx=10, pady=10)
    tk.Label(root, text="Enter second number:").grid(row=1, column=0, padx=10, pady=10)
    entry_num2 = tk.Entry(root)
    entry_num2.grid(row=1, column=1, padx=10, pady=10)
    # Label to display the result
    result_label = tk.Label(root, text="Result: ", font=("Arial", 12))
    result_label.grid(row=3, column=0, columnspan=2, pady=10)
    # Function to handle the button click event
    def on_calculate():
        '''
        This function is triggered when the "Calculate Minimum" button is clicked.
        It retrieves the input values, calculates the minimum, and displays the result.
        '''
        num1 = entry_num1.get()
        num2 = entry_num2.get()
        try:
            # Calculate the minimum and display the result
            minimum = find_minimum(num1, num2)
            result_label.config(text=f"Result: Minimum is {minimum}")
        except ValueError as e:
            # Show an error message if inputs are invalid
            messagebox.showerror("Input Error", str(e))
    # Button to calculate the minimum
    calculate_button = tk.Button(root, text="Calculate Minimum", command=on_calculate)
    calculate_button.grid(row=2, column=0, columnspan=2, pady=10)
    # Start the main event loop
    root.mainloop()
# Entry point of the application
if __name__ == "__main__":
    create_gui()
