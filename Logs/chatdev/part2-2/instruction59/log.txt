'''

This module contains the MainApplication class which sets up the GUI for the application,
and the sum_of_squares_of_even_numbers function which calculates the sum of squares of the first n even natural numbers.

'''

import tkinter as tk
from tkinter import messagebox

def sum_of_squares_of_even_numbers(n):
    '''
    Calculate the sum of squares of the first n even natural numbers.

    Parameters:
    n (int): The number of even natural numbers to consider.

    Returns:
    int: The sum of squares of the first n even natural numbers.
    '''
    sum_squares = sum((2 * i) ** 2 for i in range(1, n + 1))
    return sum_squares

class MainApplication:
    '''
    MainApplication class to create a GUI for calculating the sum of squares of even numbers.
    '''
    def __init__(self, root):
        self.root = root
        self.root.title("Sum of Squares of Even Numbers")

        self.label = tk.Label(root, text="Enter an integer n:")
        self.label.pack()

        self.entry = tk.Entry(root)
        self.entry.pack()

        self.calculate_button = tk.Button(root, text="Calculate", command=self.calculate_sum_of_squares)
        self.calculate_button.pack()

        self.result_label = tk.Label(root, text="")
        self.result_label.pack()

    def calculate_sum_of_squares(self):
        '''
        Calculate the sum of squares of the first n even natural numbers and display the result.
        '''
        try:
            n = int(self.entry.get())
            if n < 0:
                raise ValueError("The number must be non-negative.")
            result = sum_of_squares_of_even_numbers(n)
            self.result_label.config(text=f"Sum of squares: {result}")
        except ValueError as e:
            messagebox.showerror("Invalid Input", str(e))

if __name__ == "__main__":
    root = tk.Tk()
    app = MainApplication(root)
    root.mainloop()
```

    '''

    n (int): The number of even natural numbers to consider.

    Returns:
