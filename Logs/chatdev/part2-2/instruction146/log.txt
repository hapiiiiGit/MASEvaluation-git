        self.input_entry = tk.Entry(root, width=50)
        self.input_entry.pack()

        self.output_label = tk.Label(root, text="Processed string:")

        self.output_text = tk.Text(root, height=10, width=50)

    def remove_uppercase_substrings(self):

        self.output_text.delete(1.0, tk.END)
        self.output_text.insert(tk.END, processed_string)

if __name__ == "__main__":
    root = tk.Tk()

'''

String processing logic to remove uppercase substrings from a given string.

'''

import re

class StringProcessor:
    @staticmethod
    def remove_uppercase_substrings(input_string):
        # Use regular expressions to find and remove uppercase substrings
        return re.sub(r'\b[A-Z]+\b', '', input_string)
```
