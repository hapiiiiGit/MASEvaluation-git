'''
This file contains the main application class for the Even Factor Sum Calculator.
It uses tkinter for the GUI and includes a helper function to calculate the sum of even factors.
'''

import tkinter as tk
from tkinter import messagebox

def sum_of_even_factors(n):
    '''
    Calculate the sum of even factors of a given number n.
    
    Parameters:
    n (int): The number for which to find the sum of even factors.
    
    Returns:
    int: The sum of even factors of n.
    '''
    if n < 1:
        return 0
    sum_even_factors = 0
    for i in range(2, n + 1, 2):
        if n % i == 0:
            sum_even_factors += i
    return sum_even_factors

class EvenFactorSumApp:
    '''
    A class to create a GUI application for calculating the sum of even factors of a number.
    '''
    def __init__(self, master):
        self.master = master
        master.title("Even Factor Sum Calculator")

        self.label = tk.Label(master, text="Enter a number:")
        self.label.pack()

        self.entry = tk.Entry(master)
        self.entry.pack()

        self.calculate_button = tk.Button(master, text="Calculate", command=self.calculate_even_factors)
        self.calculate_button.pack()

        self.result_label = tk.Label(master, text="")
        self.result_label.pack()

    def calculate_even_factors(self):
        '''
        Calculate the sum of even factors and display the result.
        '''
        try:
            number = int(self.entry.get())
            if number < 1:
                raise ValueError("Number must be greater than 0.")
            result = sum_of_even_factors(number)
            self.result_label.config(text=f"Sum of even factors: {result}")
        except ValueError as e:
            messagebox.showerror("Invalid input", str(e))

if __name__ == "__main__":
    root = tk.Tk()
    app = EvenFactorSumApp(root)
    root.mainloop()
```

This code sets up a simple GUI application using `tkinter` that allows the user to input a number and calculates the sum of its even factors. The `sum_of_even_factors` function handles the computation, while the `EvenFactorSumApp` class manages the GUI components and user interactions.

'''
This file contains the main application class for the Even Factor Sum Calculator.
It uses tkinter for the GUI and includes a helper function to calculate the sum of even factors.
'''

import tkinter as tk
from tkinter import messagebox

def sum_of_even_factors(n):
    '''
    Calculate the sum of even factors of a given number n.
    
    Parameters:
    n (int): The number for which to find the sum of even factors.
    
    Returns:
    int: The sum of even factors of n.
    '''
    if n < 1:
        return 0
    sum_even_factors = 0
    for i in range(2, n + 1, 2):
        if n % i == 0:
            sum_even_factors += i
    return sum_even_factors

    '''
    A class to create a GUI application for calculating the sum of even factors of a number.
    '''
    def __init__(self, master):
        self.master = master
        master.title("Even Factor Sum Calculator")

        self.label = tk.Label(master, text="Enter a number:")
        self.label.pack()

        self.entry = tk.Entry(master)
        self.entry.pack()

    root = tk.Tk()

'''

It uses tkinter for the GUI and includes a helper function to calculate the sum of even factors.
'''
import tkinter as tk
from tkinter import messagebox
def sum_of_even_factors(n):
    '''
    Calculate the sum of even factors of a given number n.
    Parameters:
    n (int): The number for which to find the sum of even factors.
    Returns:
    int: The sum of even factors of n.
    '''
    if n < 1:
        return 0

        if n % i == 0:
            sum_even_factors += i

        self.label.pack()
        self.entry = tk.Entry(master)
        self.entry.pack()

        '''
        Calculate the sum of even factors and display the result.

                raise ValueError("Number must be greater than 0.")

'''
This file contains the main application class for the Even Factor Sum Calculator.
It uses tkinter for the GUI and includes a helper function to calculate the sum of even factors.
'''
import tkinter as tk
from tkinter import messagebox
def sum_of_even_factors(n):
    '''
    Calculate the sum of even factors of a given number n.
    Parameters:
    n (int): The number for which to find the sum of even factors.
    Returns:
    int: The sum of even factors of n.
    '''
    if n < 1:
        return 0
    sum_even_factors = 0
    for i in range(2, n + 1, 2):
        if n % i == 0:
            sum_even_factors += i
    return sum_even_factors
class EvenFactorSumApp:
    '''
    A class to create a GUI application for calculating the sum of even factors of a number.
    '''
    def __init__(self, master):
        self.master = master
        master.title("Even Factor Sum Calculator")
        self.label = tk.Label(master, text="Enter a number:")
        self.label.pack()
        self.entry = tk.Entry(master)
        self.entry.pack()
        self.calculate_button = tk.Button(master, text="Calculate", command=self.calculate_even_factors)
        self.calculate_button.pack()
        self.result_label = tk.Label(master, text="")
        self.result_label.pack()
    def calculate_even_factors(self):
        '''
        Calculate the sum of even factors and display the result.
        '''
        try:
            number = int(self.entry.get())
            if number < 1:
                raise ValueError("Number must be greater than 0.")
            result = sum_of_even_factors(number)
            self.result_label.config(text=f"Sum of even factors: {result}")
        except ValueError as e:
            messagebox.showerror("Invalid input", str(e))
if __name__ == "__main__":
    root = tk.Tk()
    app = EvenFactorSumApp(root)
    root.mainloop()
