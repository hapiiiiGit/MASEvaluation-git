import tkinter as tk
from logic import next_bigger_number

class NextBiggerNumberApp:
    def __init__(self, master):
        self.master = master
        master.title("Next Bigger Number Finder")
        
        self.create_widgets()

    def create_widgets(self):
        self.label = tk.Label(self.master, text="Enter a number:")
        self.label.pack()

        self.entry = tk.Entry(self.master)
        self.entry.pack()

        self.find_button = tk.Button(self.master, text="Find Next Bigger Number", command=self.find_next_bigger_number)
        self.find_button.pack()

        self.result_label = tk.Label(self.master, text="")
        self.result_label.pack()

    def find_next_bigger_number(self):
        try:
            number = int(self.entry.get())
            next_number = next_bigger_number(number)
            if next_number:
                self.result_label.config(text=f"Next bigger number: {next_number}")

                self.result_label.config(text="No bigger number can be formed.")
        except ValueError:
            self.result_label.config(text="Please enter a valid integer.")

if __name__ == "__main__":
    root = tk.Tk()

    root.mainloop()

'''

Logic to find the next bigger number by rearranging the digits of a given number.

'''

def next_bigger_number(n: int) -> int:
    digits = list(str(n))
    length = len(digits)

    # Step 1: Find the rightmost digit which is smaller than its next digit
    for i in range(length - 2, -1, -1):
        if digits[i] < digits[i + 1]:
            break
    else:
        return None  # No bigger number can be formed

    # Step 2: Find the smallest digit on the right side of (i) which is greater than digits[i]
    for j in range(length - 1, i, -1):
        if digits[j] > digits[i]:
            # Step 3: Swap the found digits
            digits[i], digits[j] = digits[j], digits[i]
            break

    # Step 4: Reverse the digits after position i
    digits = digits[:i + 1] + sorted(digits[i + 1:])

'''

Logic to find the next bigger number by rearranging the digits of a given number.

'''

def next_bigger_number(n: int) -> int:
    digits = list(str(n))
    length = len(digits)

    # Step 1: Find the rightmost digit which is smaller than its next digit
    for i in range(length - 2, -1, -1):
        if digits[i] < digits[i + 1]:
            break
    else:
        return None  # No bigger number can be formed

    # Step 2: Find the smallest digit on the right side of (i) which is greater than digits[i]
    for j in range(length - 1, i, -1):
        if digits[j] > digits[i]:
            # Step 3: Swap the found digits
            digits[i], digits[j] = digits[j], digits[i]
            break

    # Step 4: Reverse the digits after position i
    digits = digits[:i + 1] + sorted(digits[i + 1:])

    return int(''.join(digits))

'''
Logic to find the next bigger number by rearranging the digits of a given number.
'''
def next_bigger_number(n: int) -> int:
    digits = list(str(n))
    length = len(digits)
    # Step 1: Find the rightmost digit which is smaller than its next digit
    for i in range(length - 2, -1, -1):
        if digits[i] < digits[i + 1]:
            break
    else:
        return None  # No bigger number can be formed
    # Step 2: Find the smallest digit on the right side of (i) which is greater than digits[i]
    for j in range(length - 1, i, -1):
        if digits[j] > digits[i]:
            # Step 3: Swap the found digits
            digits[i], digits[j] = digits[j], digits[i]
            break
    # Step 4: Reverse the digits after position i
    digits = digits[:i + 1] + sorted(digits[i + 1:])
    return int(''.join(digits))

def next_bigger_number(n: int) -> int:
    digits = list(str(n))
    length = len(digits)
    # Step 1: Traverse the digits from right to left to find the first digit that is smaller than the digit next to it.
    for i in range(length - 2, -1, -1):
        if digits[i] < digits[i + 1]:
            break
    else:
        return None  # If no such digit is found, the digits are in descending order, and no bigger number can be formed.
    # Step 2: Find the smallest digit on the right side of the found digit that is larger than the found digit.
    for j in range(length - 1, i, -1):
        if digits[j] > digits[i]:
            # Step 3: Swap the found digit with the smallest larger digit found.
            digits[i], digits[j] = digits[j], digits[i]
            break
    # Step 4: Sort the digits after the position of the swapped digit to get the smallest possible number.
    digits = digits[:i + 1] + sorted(digits[i + 1:])
    return int(''.join(digits))

def next_bigger_number(n: int) -> int:
    digits = list(str(n))
    length = len(digits)
    # Step 1: Traverse the digits from right to left to find the first digit that is smaller than the digit next to it.
    for i in range(length - 2, -1, -1):
        if digits[i] < digits[i + 1]:
            break
    else:
        return -1  # Return -1 instead of None

    # Step 2: Find the smallest digit on the right side of the found digit that is larger than the found digit.
    for j in range(length - 1, i, -1):
        if digits[j] > digits[i]:
            # Step 3: Swap the found digit with the smallest larger digit found.
            digits[i], digits[j] = digits[j], digits[i]
            break

    # Step 4: Sort the digits after the position of the swapped digit to get the smallest possible number.
    digits = digits[:i + 1] + sorted(digits[i + 1:])
    return int(''.join(digits))

'''
Logic to find the next bigger number by rearranging the digits of a given number.
'''
def next_bigger_number(n: int) -> int:
    digits = list(str(n))
    length = len(digits)
    # Step 1: Traverse the digits from right to left to find the first digit that is smaller than the digit next to it.
    for i in range(length - 2, -1, -1):
        if digits[i] < digits[i + 1]:
            break
    else:
        return -1  # Return -1 instead of None
    # Step 2: Find the smallest digit on the right side of the found digit that is larger than the found digit.
    for j in range(length - 1, i, -1):
        if digits[j] > digits[i]:
            # Step 3: Swap the found digit with the smallest larger digit found.
            digits[i], digits[j] = digits[j], digits[i]
            break
    # Step 4: Sort the digits after the position of the swapped digit to get the smallest possible number.
    digits = digits[:i + 1] + sorted(digits[i + 1:])
    return int(''.join(digits))

'''
Logic to find the next bigger number by rearranging the digits of a given number.
'''
def next_bigger_number(n: int) -> int:
    digits = list(str(n))
    length = len(digits)
    # Step 1: Traverse the digits from right to left to find the first digit that is smaller than the digit next to it.
    for i in range(length - 2, -1, -1):
        if digits[i] < digits[i + 1]:
            break
    else:
        return -1  # Return -1 instead of None
    # Step 2: Find the smallest digit on the right side of the found digit that is larger than the found digit.
    for j in range(length - 1, i, -1):
        if digits[j] > digits[i]:
            # Step 3: Swap the found digit with the smallest larger digit found.
            digits[i], digits[j] = digits[j], digits[i]
            break
    # Step 4: Sort the digits after the position of the swapped digit to get the smallest possible number.
    digits = digits[:i + 1] + sorted(digits[i + 1:])
    return int(''.join(digits))

'''
Logic to find the next bigger number by rearranging the digits of a given number.
'''
def next_bigger_number(n: int) -> int:
    digits = list(str(n))
    length = len(digits)
    # Step 1: Traverse the digits from right to left to find the first digit that is smaller than the digit next to it.
    for i in range(length - 2, -1, -1):
        if digits[i] < digits[i + 1]:
            break
    else:
        return -1  # Return -1 instead of None
    # Step 2: Find the smallest digit on the right side of the found digit that is larger than the found digit.
    for j in range(length - 1, i, -1):
        if digits[j] > digits[i]:
            # Step 3: Swap the found digit with the smallest larger digit found.
            digits[i], digits[j] = digits[j], digits[i]
            break
    # Step 4: Sort the digits after the position of the swapped digit to get the smallest possible number.
    digits = digits[:i + 1] + sorted(digits[i + 1:])
    return int(''.join(digits))
