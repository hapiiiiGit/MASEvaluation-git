    def extract_rear_elements(self):

            if isinstance(tuple_of_strings, tuple):
                result = extract_rear_index_elements(tuple_of_strings)

def extract_rear_index_elements(tuple_of_strings):

    :param tuple_of_strings: Tuple containing strings.
    :return: List of rear index elements from each string.
    """
