    '''

        '''
        Initialize the main window and set up the GUI components.
        '''
        self.root = root
        self.root.title("Tuple Multiplier")
        self.create_gui()

    def create_gui(self):
        '''
        Set up the GUI elements like labels, entry fields, and buttons.
        '''
        # Labels
        tk.Label(self.root, text="Tuple 1 (comma-separated):").grid(row=0, column=0, padx=10, pady=10)
        tk.Label(self.root, text="Tuple 2 (comma-separated):").grid(row=1, column=0, padx=10, pady=10)
        tk.Label(self.root, text="Result:").grid(row=2, column=0, padx=10, pady=10)

        # Entry fields
        self.tuple1_entry = tk.Entry(self.root, width=30)
        self.tuple2_entry = tk.Entry(self.root, width=30)
        self.result_label = tk.Label(self.root, text="", width=30, anchor='w')
        self.tuple1_entry.grid(row=0, column=1, padx=10, pady=10)
        self.tuple2_entry.grid(row=1, column=1, padx=10, pady=10)
        self.result_label.grid(row=2, column=1, padx=10, pady=10)

        # Calculate button
        calculate_button = tk.Button(self.root, text="Calculate", command=self.on_calculate)
        calculate_button.grid(row=3, column=0, columnspan=2, pady=20)

    def multiply_tuples(self, tuple1, tuple2):
        '''
        Perform index-wise multiplication of two tuples.
        '''
        if len(tuple1) != len(tuple2):
            raise ValueError("Tuples must be of the same length.")
        return tuple(a * b for a, b in zip(tuple1, tuple2))

        '''
        Event handler for the calculate button to get user input, perform multiplication, and display results.

            tuple1 = tuple(map(int, self.tuple1_entry.get().split(',')))
