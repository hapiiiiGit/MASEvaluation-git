            self.result_label.config(text="Invalid input. Please enter numbers separated by commas.")

if __name__ == "__main__":
    root = tk.Tk()
    app = MainApplication(root)

#### utils.py

```python
'''

Utility functions for the application.

'''

def sum_of_non_repeated_elements(numbers):
    """
    Calculate the sum of non-repeated elements in a list.
    
    :param numbers: List of integers.
    :return: Sum of non-repeated elements.
    """
    element_count = {}
    for num in numbers:
        if num in element_count:
            element_count[num] += 1
        else:
            element_count[num] = 1
    
    non_repeated_sum = sum(num for num, count in element_count.items() if count == 1)
    return non_repeated_sum
```

import tkinter as tk
from utils import sum_of_non_repeated_elements

class MainApplication:
    def __init__(self, root):
        self.root = root
        self.root.title("Sum of Non-Repeated Elements")
        
        self.label = tk.Label(root, text="Enter numbers separated by commas:")
        self.label.pack()
        
        self.entry = tk.Entry(root, width=50)
        self.entry.pack()
        
        self.button = tk.Button(root, text="Calculate Sum", command=self.calculate_sum)
        self.button.pack()
        
        self.result_label = tk.Label(root, text="")
        self.result_label.pack()
    
    def calculate_sum(self):
        input_data = self.entry.get()
        try:
            numbers = list(map(int, input_data.split(',')))
            result = sum_of_non_repeated_elements(numbers)
            self.result_label.config(text=f"Sum of non-repeated elements: {result}")
        except ValueError:
            self.result_label.config(text="Invalid input. Please enter numbers separated by commas.")

if __name__ == "__main__":
    root = tk.Tk()
    app = MainApplication(root)
    root.mainloop()

def sum_of_non_repeated_elements(numbers):
    """
    Calculate the sum of non-repeated elements in a list.
    
    :param numbers: List of integers.
    :return: Sum of non-repeated elements.
    """
    element_count = {}
    for num in numbers:
        if num in element_count:
            element_count[num] += 1
        else:
            element_count[num] = 1
    
    non_repeated_sum = sum(num for num, count in element_count.items() if count == 1)
    return non_repeated_sum

def sum_of_non_repeated_elements(numbers):
    """
    Calculate the sum of non-repeated elements in a list.
    :param numbers: List of integers.
    :return: Sum of non-repeated elements.
    """
    element_count = {}
    for num in numbers:
        if num in element_count:
            element_count[num] += 1
        else:
            element_count[num] = 1
    non_repeated_sum = sum(num for num, count in element_count.items() if count == 1)
    return non_repeated_sum
