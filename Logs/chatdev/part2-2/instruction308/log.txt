```python
'''

Main application file for checking even indices in a list using a GUI.

'''

import tkinter as tk
from tkinter import messagebox

class MainApp:
    def __init__(self, root):
        '''
        Initialize the main application with GUI components.
        
        :param root: The root window of the tkinter application.
        '''
        self.root = root
        self.root.title("Even Index Checker")
        
        # Create and place widgets
        self.label = tk.Label(root, text="Enter a list of numbers separated by commas:")
        self.label.pack(pady=10)
        
        self.entry = tk.Entry(root, width=50)
        self.entry.pack(pady=10)
        
        self.check_button = tk.Button(root, text="Check Even Indices", command=self.on_check_button_click)
        self.check_button.pack(pady=20)
        
    def on_check_button_click(self):
        '''
        Event handler for the check button click. It retrieves the input list,
        checks the even indices, and displays the result.
        '''
        input_text = self.entry.get()
        try:
            # Convert input text to a list of integers
            num_list = list(map(int, input_text.split(',')))
            if self.check_even_indices(num_list):
                messagebox.showinfo("Result", "Every even index contains an even number.")
            else:
                messagebox.showinfo("Result", "Not all even indices contain an even number.")
        except ValueError:
            messagebox.showerror("Error", "Please enter a valid list of numbers separated by commas.")
    
    def check_even_indices(self, num_list):
        '''
        Check if every even index in the list contains an even number.
        
        :param num_list: List of integers to check.
        :return: True if every even index contains an even number, False otherwise.
        '''
        for i in range(0, len(num_list), 2):
            if not self.is_even(num_list[i]):
                return False
        return True
    
    def is_even(self, number):
        '''
        Determine if a number is even.
        
        :param number: The number to check.
        :return: True if the number is even, False otherwise.
        '''
        return number % 2 == 0

if __name__ == "__main__":
    root = tk.Tk()
    app = MainApp(root)
    root.mainloop()
```

- **is_even Method**: A simple utility function to check if a number is even.

Main application file for checking even indices in a list using a GUI.

import tkinter as tk
from tkinter import messagebox

class MainApp:
    def __init__(self, root):
        '''
        Initialize the main application with GUI components.
        
        :param root: The root window of the tkinter application.
        '''
        self.root = root
        self.root.title("Even Index Checker")
        
        # Create and place widgets
        self.label = tk.Label(root, text="Enter a list of numbers separated by commas:")
        self.label.pack(pady=10)
        
        self.entry = tk.Entry(root, width=50)
        self.entry.pack(pady=10)
        
        self.check_button = tk.Button(root, text="Check Even Indices", command=self.on_check_button_click)
        self.check_button.pack(pady=20)
        
    def on_check_button_click(self):
        '''
        Event handler for the check button click. It retrieves the input list,
        checks the even indices, and displays the result.
        '''
        input_text = self.entry.get()
        try:
            # Convert input text to a list of integers
            num_list = list(map(int, input_text.split(',')))
            if self.check_even_indices(num_list):
                messagebox.showinfo("Result", "Every even index contains an even number.")
            else:
                messagebox.showinfo("Result", "Not all even indices contain an even number.")
        except ValueError:
            messagebox.showerror("Error", "Please enter a valid list of numbers separated by commas.")
    
    def check_even_indices(self, num_list):
        '''
        Check if every even index in the list contains an even number.
        
        :param num_list: List of integers to check.
        :return: True if every even index contains an even number, False otherwise.
        '''
        for i in range(0, len(num_list), 2):
            if not self.is_even(num_list[i]):
                return False
        return True
    
    def is_even(self, number):
        '''
        Determine if a number is even.
        
        :param number: The number to check.
        :return: True if the number is even, False otherwise.
        '''
        return number % 2 == 0

if __name__ == "__main__":
    root = tk.Tk()
    app = MainApp(root)
    root.mainloop()

'''
Main application file for checking even indices in a list using a GUI.
'''
import tkinter as tk
from tkinter import messagebox
class MainApp:
    def __init__(self, root):
        '''
        Initialize the main application with GUI components.
        :param root: The root window of the tkinter application.
        '''
        self.root = root
        self.root.title("Even Index Checker")
        # Create and place widgets
        self.label = tk.Label(root, text="Enter a list of numbers separated by commas:")
        self.label.pack(pady=10)
        self.entry = tk.Entry(root, width=50)
        self.entry.pack(pady=10)
        self.check_button = tk.Button(root, text="Check Even Indices", command=self.on_check_button_click)
        self.check_button.pack(pady=20)
    def on_check_button_click(self):
        '''
        Event handler for the check button click. It retrieves the input list,
        checks the even indices, and displays the result.
        '''
        input_text = self.entry.get()
        try:
            # Convert input text to a list of integers
            num_list = list(map(int, input_text.split(',')))
            if self.check_even_indices(num_list):
                messagebox.showinfo("Result", "Every even index contains an even number.")
            else:
                messagebox.showinfo("Result", "Not all even indices contain an even number.")
        except ValueError:
            messagebox.showerror("Error", "Please enter a valid list of numbers separated by commas.")
    def check_even_indices(self, num_list):
        '''
        Check if every even index in the list contains an even number.
        :param num_list: List of integers to check.
        :return: True if every even index contains an even number, False otherwise.
        '''
        for i in range(0, len(num_list), 2):
            if not self.is_even(num_list[i]):
                return False
        return True
    def is_even(self, number):
        '''
        Determine if a number is even.
        :param number: The number to check.
        :return: True if the number is even, False otherwise.
        '''
        return number % 2 == 0
if __name__ == "__main__":
    root = tk.Tk()
    app = MainApp(root)
    root.mainloop()

Main application file for checking even indices in a list using a GUI.

import tkinter as tk
from tkinter import messagebox
class MainApp:
    def __init__(self, root):
        '''
        Initialize the main application with GUI components.
        :param root: The root window of the tkinter application.
        '''
        self.root = root
        self.root.title("Even Index Checker")
        # Create and place widgets
        self.label = tk.Label(root, text="Enter a list of numbers separated by commas:")
        self.label.pack(pady=10)
        self.entry = tk.Entry(root, width=50)
        self.entry.pack(pady=10)
        self.check_button = tk.Button(root, text="Check Even Indices", command=self.on_check_button_click)
        self.check_button.pack(pady=20)
    def on_check_button_click(self):
        '''
        Event handler for the check button click. It retrieves the input list,
        checks the even indices, and displays the result.
        '''
        input_text = self.entry.get()
        try:
            # Convert input text to a list of integers
            num_list = list(map(int, input_text.split(',')))
            if self.check_even_indices(num_list):
                messagebox.showinfo("Result", "Every even index contains an even number.")
            else:
                messagebox.showinfo("Result", "Not all even indices contain an even number.")
        except ValueError:
            messagebox.showerror("Error", "Please enter a valid list of numbers separated by commas.")
    def check_even_indices(self, num_list):
        '''
        Check if every even index in the list contains an even number.
        :param num_list: List of integers to check.
        :return: True if every even index contains an even number, False otherwise.
        '''
        for i in range(0, len(num_list), 2):
            if not self.is_even(num_list[i]):
                return False
        return True
    def is_even(self, number):
        '''
        Determine if a number is even.
        :param number: The number to check.
        :return: True if the number is even, False otherwise.
        '''
        return number % 2 == 0
if __name__ == "__main__":
