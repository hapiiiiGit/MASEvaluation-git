import heapq

class ProductCalculator:
    def find_largest_products(self, list1, list2, n):
        # Use a min-heap to keep track of the largest products
        min_heap = []
        for num1 in list1:
            for num2 in list2:
                product = num1 * num2
                if len(min_heap) < n:
                    heapq.heappush(min_heap, product)
                else:
                    heapq.heappushpop(min_heap, product)
        # Return the n largest products sorted in descending order
        return sorted(min_heap, reverse=True)

'''

import heapq
class ProductCalculator:
    def find_largest_products(self, list1, list2, n):
        # Use a min-heap to keep track of the largest products
        min_heap = []
        for num1 in list1:
            for num2 in list2:
                product = num1 * num2
                if len(min_heap) < n:
                    heapq.heappush(min_heap, product)
                else:
                    heapq.heappushpop(min_heap, product)
        # Return the n largest products sorted in descending order
        return sorted(min_heap, reverse=True)

'''
ProductCalculator class to find the largest products from two lists.
'''
import heapq
class ProductCalculator:
    def find_largest_products(self, list1, list2, n):
        # Use a min-heap to keep track of the largest products
        min_heap = []
        for num1 in list1:
            for num2 in list2:
                product = num1 * num2
                if len(min_heap) < n:
                    heapq.heappush(min_heap, product)
                else:
                    heapq.heappushpop(min_heap, product)
        # Return the n largest products sorted in descending order
        return sorted(min_heap, reverse=True)

'''
ProductCalculator class to find the largest products from two lists.
'''
import heapq
class ProductCalculator:
    def find_largest_products(self, list1, list2, n):
        # Use a min-heap to keep track of the largest products
        min_heap = []
        for num1 in list1:
            for num2 in list2:
                product = num1 * num2
                if len(min_heap) < n:
                    heapq.heappush(min_heap, product)
                else:
                    heapq.heappushpop(min_heap, product)
        # Return the n largest products sorted in descending order
        return sorted(min_heap, reverse=True)
