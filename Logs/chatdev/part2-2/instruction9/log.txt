import tkinter as tk
from tkinter import messagebox
class ArrayRearranger:
    '''
    This class contains the logic to rearrange the array.
    '''
    @staticmethod
    def rearrange(array, n):
        '''
        Rearranges the first n elements of the array so that all negative elements
        appear before positive ones, preserving their relative order.
        :param array: List of integers
        :param n: Integer, number of elements to rearrange
        :return: List of integers with rearranged elements
        '''
        if n > len(array):
            n = len(array)
        negatives = [x for x in array[:n] if x < 0]
        positives = [x for x in array[:n] if x >= 0]
        return negatives + positives + array[n:]
class App:
    '''
    This class manages the GUI components and user interactions.
    '''
    def __init__(self, root):
        self.root = root
        self.root.title("Array Rearranger")
        self.label = tk.Label(root, text="Enter array elements separated by commas:")
        self.label.pack()
        self.array_entry = tk.Entry(root, width=50)
        self.array_entry.pack()
        self.label_n = tk.Label(root, text="Enter the number of elements to rearrange:")

        self.n_entry.pack()
        self.rearrange_button = tk.Button(root, text="Rearrange", command=self.rearrange_array)
        self.rearrange_button.pack()
        self.result_label = tk.Label(root, text="")
        self.result_label.pack()
    def rearrange_array(self):
        '''
        Handles the rearrangement of the array based on user input.
        '''
        try:
            array = list(map(int, self.array_entry.get().split(',')))
            n = int(self.n_entry.get())
            rearranged_array = ArrayRearranger.rearrange(array, n)
            self.result_label.config(text=f"Rearranged Array: {rearranged_array}")
        except ValueError:
            messagebox.showerror("Invalid input", "Please enter valid integers for the array and n.")
if __name__ == "__main__":
    root = tk.Tk()
    app = App(root)
    root.mainloop()

'''
This file contains the main application logic and GUI setup for rearranging an array.
'''
import tkinter as tk
from tkinter import messagebox
class ArrayRearranger:
    '''
    This class contains the logic to rearrange the array.
    '''
    @staticmethod
    def rearrange(array, n):
        '''
        Rearranges the first n elements of the array so that all negative elements
        appear before positive ones, preserving their relative order.
        :param array: List of integers
        :param n: Integer, number of elements to rearrange
        :return: List of integers with rearranged elements
        '''
        if n > len(array):
            n = len(array)
        negatives = [x for x in array[:n] if x < 0]
        positives = [x for x in array[:n] if x >= 0]
        return negatives + positives + array[n:]
class App:
    '''
    This class manages the GUI components and user interactions.
    '''
    def __init__(self, root):
        self.root = root
        self.root.title("Array Rearranger")
        self.label = tk.Label(root, text="Enter array elements separated by commas:")
        self.label.pack()
        self.array_entry = tk.Entry(root, width=50)
        self.array_entry.pack()
        self.label_n = tk.Label(root, text="Enter the number of elements to rearrange:")
        self.label_n.pack()
        self.n_entry = tk.Entry(root, width=10)
        self.n_entry.pack()
        self.rearrange_button = tk.Button(root, text="Rearrange", command=self.rearrange_array)
        self.rearrange_button.pack()
        self.result_label = tk.Label(root, text="")
        self.result_label.pack()
    def rearrange_array(self):
        '''
        Handles the rearrangement of the array based on user input.
        '''
        try:
            array = list(map(int, self.array_entry.get().split(',')))
            n = int(self.n_entry.get())
            rearranged_array = ArrayRearranger.rearrange(array, n)
            self.result_label.config(text=f"Rearranged Array: {rearranged_array}")
        except ValueError:
            messagebox.showerror("Invalid input", "Please enter valid integers for the array and n.")
if __name__ == "__main__":
    root = tk.Tk()
    app = App(root)
    root.mainloop()
