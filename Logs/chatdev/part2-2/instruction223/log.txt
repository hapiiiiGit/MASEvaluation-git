'''

subset_logic.py

This file contains the function to find the size of the largest subset of a list of numbers where every pair is divisible.

'''

def find_largest_divisible_subset(nums):
    if not nums:
        return 0
    
    nums.sort()
    n = len(nums)
    dp = [1] * n
    max_size = 1
    
    for i in range(n):
        for j in range(i):
            if nums[i] % nums[j] == 0:
                dp[i] = max(dp[i], dp[j] + 1)
        max_size = max(max_size, dp[i])
    
    return max_size

def find_largest_divisible_subset(nums):
    if not nums:
        return 0
    
    nums.sort()
    n = len(nums)
    dp = [1] * n
    max_size = 1
    
    for i in range(n):
        for j in range(i):
            if nums[i] % nums[j] == 0:
                dp[i] = max(dp[i], dp[j] + 1)
        max_size = max(max_size, dp[i])
    
    return max_size

def find_largest_divisible_subset(nums):
    if not nums:
        return 0
    nums.sort()
    n = len(nums)
    dp = [1] * n
    max_size = 1
    for i in range(n):
        for j in range(i):
            if nums[i] % nums[j] == 0:
                dp[i] = max(dp[i], dp[j] + 1)
        max_size = max(max_size, dp[i])
    return max_size

def find_largest_divisible_subset(nums):
    if not nums:
        return 0
    nums.sort()
    n = len(nums)
    dp = [1] * n
    max_size = 1
    for i in range(n):
        for j in range(i):
            if nums[i] % nums[j] == 0:
                dp[i] = max(dp[i], dp[j] + 1)
        max_size = max(max_size, dp[i])
    return max_size

'''
subset_logic.py
This file contains the function to find the size of the largest subset of a list of numbers where every pair is divisible.
'''
def find_largest_divisible_subset(nums):
    if not nums:
        return 0
    nums.sort()
    n = len(nums)
    dp = [1] * n
    max_size = 1
    for i in range(n):
        for j in range(i):
            if nums[i] % nums[j] == 0:
                dp[i] = max(dp[i], dp[j] + 1)
        max_size = max(max_size, dp[i])
    return max_size

import tkinter as tk
from tkinter import messagebox
from subset_logic import find_largest_divisible_subset
class DivisibleSubsetApp:
    def __init__(self, root):
        self.root = root
        self.root.title("Largest Divisible Subset Finder")
        self.label = tk.Label(root, text="Enter numbers separated by commas:")
        self.label.pack()

        self.entry.pack()

            messagebox.showerror("Invalid input", "Please enter a valid list of numbers separated by commas.")
        except Exception as e:
            messagebox.showerror("Error", str(e))

'''
subset_logic.py
This file contains the function to find the size of the largest subset of a list of numbers where every pair is divisible.
'''
def find_largest_divisible_subset(nums):
    if not nums:
        return 0
    nums.sort()
    n = len(nums)
    dp = [1] * n
    max_size = 1
    for i in range(n):
        for j in range(i):
            if nums[i] % nums[j] == 0:
                dp[i] = max(dp[i], dp[j] + 1)
        max_size = max(max_size, dp[i])
    return max_size
