import tkinter as tk
from tkinter import messagebox

def find_rotation_count(arr):
    '''
    DOCSTRING: Calculate the number of rotations required to sort the array.

    return 0

        self.result_label.pack()

    def count_rotations(self):
        try:
            array_str = self.entry.get()
            array = list(map(int, array_str.split(',')))
            rotations = find_rotation_count(array)
            self.display_result(rotations)
        except ValueError:
            messagebox.showerror("Invalid input", "Please enter a valid array of integers.")

import tkinter as tk
from tkinter import messagebox

def find_rotation_count(arr):
    '''
    DOCSTRING: Calculate the number of rotations required to sort the array.
    '''
    n = len(arr)
    ascending_rotations = 0
    descending_rotations = 0
    
    # Check for ascending order rotations
    for i in range(n):
        if arr[i] > arr[(i + 1) % n]:
            ascending_rotations = (i + 1) % n
            break
    
    # Check for descending order rotations
    for i in range(n):
        if arr[i] < arr[(i + 1) % n]:
            descending_rotations = (i + 1) % n
            break
    
    # Return the minimum rotations needed for either order
    return min(ascending_rotations, descending_rotations)

import tkinter as tk
from tkinter import messagebox
def find_rotation_count(arr):
    '''
    Calculate the number of rotations required to sort the array in ascending order.
    '''
    n = len(arr)
    for i in range(n):
        if arr[i] > arr[(i + 1) % n]:
            return (i + 1) % n
    return 0  # If no rotation is needed, return 0
class RotationCounterApp:
    '''
    GUI application for counting rotations to sort an array.
    '''
    def __init__(self, master):
        self.master = master
        master.title("Rotation Counter")
        self.label = tk.Label(master, text="Enter array elements separated by commas:")
        self.label.pack()
        self.entry = tk.Entry(master, width=50)
        self.entry.pack()
        self.count_button = tk.Button(master, text="Count Rotations", command=self.count_rotations)
        self.count_button.pack()
        self.result_label = tk.Label(master, text="")
        self.result_label.pack()
    def count_rotations(self):
        try:
            array_str = self.entry.get()
            array = list(map(int, array_str.split(',')))
            rotations = find_rotation_count(array)
            self.display_result(rotations)
        except ValueError:

'''
Calculate the number of rotations required to sort the array in ascending order.
'''
import tkinter as tk
from tkinter import messagebox
def find_rotation_count(arr):
    '''
    Calculate the number of rotations required to sort the array in ascending order.
    '''
    n = len(arr)
    for i in range(n):
        if arr[i] > arr[(i + 1) % n]:
            return (i + 1) % n
    return 0  # If no rotation is needed, return 0
class RotationCounterApp:
    '''
    GUI application for counting rotations to sort an array.
    '''
    def __init__(self, master):
        self.master = master
        master.title("Rotation Counter")
        self.label = tk.Label(master, text="Enter array elements separated by commas:")
        self.label.pack()
        self.entry = tk.Entry(master, width=50)
        self.entry.pack()
        self.count_button = tk.Button(master, text="Count Rotations", command=self.count_rotations)
        self.count_button.pack()
        self.result_label = tk.Label(master, text="")
        self.result_label.pack()
    def count_rotations(self):
        try:
            array_str = self.entry.get()
            array = list(map(int, array_str.split(',')))
            rotations = find_rotation_count(array)
            self.display_result(rotations)
        except ValueError:
            messagebox.showerror("Invalid input", "Please enter a valid array of integers.")
    def display_result(self, result):
        self.result_label.config(text=f"Number of rotations required: {result}")
if __name__ == "__main__":
    root = tk.Tk()
    app = RotationCounterApp(root)
    root.mainloop()
