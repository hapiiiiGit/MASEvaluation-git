class InsertionPointFinder:
    def find_insertion_point(self, sorted_list, value):
        """
        Find the left insertion point for a specified value in sorted order.
        :param sorted_list: List of integers sorted in ascending order.
        :param value: Integer value to find the insertion point for.

        """
        for i, num in enumerate(sorted_list):
            if num >= value:

        return len(sorted_list)

from bisect import bisect_left
class InsertionPointFinder:
    def find_insertion_point(self, sorted_list, value):
        """
        Find the left insertion point for a specified value in sorted order using binary search.
        :param sorted_list: List of integers sorted in ascending order.
        :param value: Integer value to find the insertion point for.
        :return: Index of the left insertion point.
        """
        return bisect_left(sorted_list, value)

'''
This file contains the logic for finding the left insertion point in a sorted list using binary search.
'''
from bisect import bisect_left
class InsertionPointFinder:
    def find_insertion_point(self, sorted_list, value):
        """
        Find the left insertion point for a specified value in sorted order using binary search.
        :param sorted_list: List of integers sorted in ascending order.
        :param value: Integer value to find the insertion point for.
        :return: Index of the left insertion point.
        """
        return bisect_left(sorted_list, value)
