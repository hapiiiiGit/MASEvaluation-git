'''

This module contains the main application class for the TupleApp, which provides a GUI for extracting an element from a tuple based on user input.

'''

import tkinter as tk
from tkinter import messagebox

def get_tuple_element(input_tuple, index):
    '''
    Returns the element at the specified index from the input tuple.
    
    Parameters:
    input_tuple (tuple): The tuple from which to extract the element.
    index (int): The index of the element to extract.
    
    Returns:
    element: The element at the specified index.
    '''
    try:
        return input_tuple[index]
    except IndexError:
        return None

class TupleApp:
    '''
    A GUI application for extracting an element from a tuple.
    '''
    def __init__(self, root):
        self.root = root
        self.root.title("Tuple Element Extractor")
        
        # Input for tuple
        self.tuple_label = tk.Label(root, text="Enter a tuple (comma-separated):")
        self.tuple_label.pack()
        
        self.tuple_entry = tk.Entry(root, width=50)
        self.tuple_entry.pack()
        
        # Input for index
        self.index_label = tk.Label(root, text="Enter index:")
        self.index_label.pack()
        
        self.index_entry = tk.Entry(root, width=10)
        self.index_entry.pack()
        
        # Button to get element
        self.get_button = tk.Button(root, text="Get Element", command=self.get_element)
        self.get_button.pack()
        
        # Result display
        self.result_label = tk.Label(root, text="Element at index:")
        self.result_label.pack()
        
        self.result_display = tk.Label(root, text="", fg="blue")
        self.result_display.pack()
    
    def get_element(self):
        '''
        Retrieves the element from the tuple at the specified index and displays it.
        '''
        try:
            # Parse the tuple from the entry
            tuple_input = self.tuple_entry.get()
            input_tuple = tuple(map(str.strip, tuple_input.split(',')))
            
            # Parse the index
            index = int(self.index_entry.get())
            
            # Get the element
            element = get_tuple_element(input_tuple, index)
            
            if element is not None:
                self.result_display.config(text=str(element))
            else:
                messagebox.showerror("Error", "Index out of range.")
        
        except ValueError:
            messagebox.showerror("Error", "Invalid input. Please enter a valid tuple and index.")

if __name__ == "__main__":
    root = tk.Tk()
    app = TupleApp(root)
    root.mainloop()
```

'''

This module contains the main application class for the TupleApp, which provides a GUI for extracting an element from a tuple based on user input.

'''

import tkinter as tk
from tkinter import messagebox

def get_tuple_element(input_tuple, index):
    '''
    Returns the element at the specified index from the input tuple.
    
    Parameters:
    input_tuple (tuple): The tuple from which to extract the element.
    index (int): The index of the element to extract.
    
    Returns:
    element: The element at the specified index.
    '''
    try:
        return input_tuple[index]
    except IndexError:
        return None

class TupleApp:
    '''
    A GUI application for extracting an element from a tuple.
    '''
    def __init__(self, root):
        self.root = root
        self.root.title("Tuple Element Extractor")
        
        # Input for tuple
        self.tuple_label = tk.Label(root, text="Enter a tuple (comma-separated):")
        self.tuple_label.pack()
        
        self.tuple_entry = tk.Entry(root, width=50)
        self.tuple_entry.pack()
        
        # Input for index
        self.index_label = tk.Label(root, text="Enter index:")
        self.index_label.pack()
        
        self.index_entry = tk.Entry(root, width=10)
        self.index_entry.pack()
        
        # Button to get element
        self.get_button = tk.Button(root, text="Get Element", command=self.get_element)
        self.get_button.pack()
        
        # Result display
        self.result_label = tk.Label(root, text="Element at index:")
        self.result_label.pack()
        
        self.result_display = tk.Label(root, text="", fg="blue")
        self.result_display.pack()
    
    def get_element(self):
        '''
        Retrieves the element from the tuple at the specified index and displays it.
        '''
        try:
            # Parse the tuple from the entry
            tuple_input = self.tuple_entry.get()
            input_tuple = tuple(map(str.strip, tuple_input.split(',')))
            
            # Parse the index
            index = int(self.index_entry.get())
            
            # Get the element
            element = get_tuple_element(input_tuple, index)
            
            if element is not None:
                self.result_display.config(text=str(element))
            else:
                messagebox.showerror("Error", "Index out of range.")
        
        except ValueError:
            messagebox.showerror("Error", "Invalid input. Please enter a valid tuple and index.")

if __name__ == "__main__":
    root = tk.Tk()
    app = TupleApp(root)
    root.mainloop()

    '''

'''
This module contains the main application class for the TupleApp, which provides a GUI for extracting an element from a tuple based on user input.
'''
import tkinter as tk
from tkinter import messagebox
import ast
def get_tuple_element(input_tuple, index):
    '''
    Returns the element at the specified index from the input tuple.
    Parameters:
    input_tuple (tuple): The tuple from which to extract the element.
    index (int): The index of the element to extract.
    Returns:
    element: The element at the specified index or None if the index is out of range.
    '''
    try:
        return input_tuple[index]
    except IndexError:
        return None
class TupleApp:
    '''
    A GUI application for extracting an element from a tuple.
    '''
    def __init__(self, root):
        self.root = root
        self.root.title("Tuple Element Extractor")
        # Input for tuple
        self.tuple_label = tk.Label(root, text="Enter a tuple (e.g., (1, 'a', 3.5)):")
        self.tuple_label.pack()
        self.tuple_entry = tk.Entry(root, width=50)
        self.tuple_entry.pack()
        # Input for index
        self.index_label = tk.Label(root, text="Enter index:")
        self.index_label.pack()
        self.index_entry = tk.Entry(root, width=10)
        self.index_entry.pack()
        # Button to get element
        self.get_button = tk.Button(root, text="Get Element", command=self.get_element)
        self.get_button.pack()
        # Result display
        self.result_label = tk.Label(root, text="Element at index:")
        self.result_label.pack()
        self.result_display = tk.Label(root, text="", fg="blue")
        self.result_display.pack()
    def get_element(self):
        '''
        Retrieves the element from the tuple at the specified index and displays it.
        '''
        try:
            # Parse the tuple from the entry
            tuple_input = self.tuple_entry.get()
            input_tuple = ast.literal_eval(tuple_input)
            if not isinstance(input_tuple, tuple):
                raise ValueError("Input must be a tuple.")
            # Validate and parse the index
            index_input = self.index_entry.get()
            if not index_input.isdigit():
                raise ValueError("Index must be an integer.")
            index = int(index_input)
            # Get the element
            element = get_tuple_element(input_tuple, index)
            if element is not None:
                self.result_display.config(text=str(element))
            else:
                messagebox.showerror("Error", "Index out of range.")
        except (ValueError, SyntaxError) as ve:
            messagebox.showerror("Error", str(ve))
if __name__ == "__main__":
    root = tk.Tk()
    app = TupleApp(root)
    root.mainloop()

'''
This module contains the main application class for the TupleApp, which provides a GUI for extracting an element from a tuple based on user input.
'''
import tkinter as tk
from tkinter import messagebox
import ast
def get_tuple_element(input_tuple, index):
    '''
    Returns the element at the specified index from the input tuple.
    Parameters:
    input_tuple (tuple): The tuple from which to extract the element.
    index (int): The index of the element to extract.
    Returns:
    element: The element at the specified index or None if the index is out of range.
    '''
    try:
        return input_tuple[index]
    except IndexError:
        return None
class TupleApp:
    '''
    A GUI application for extracting an element from a tuple.
    '''
    def __init__(self, root):
        self.root = root
        self.root.title("Tuple Element Extractor")
        # Input for tuple
        self.tuple_label = tk.Label(root, text="Enter a tuple (e.g., (1, 'a', 3.5)):")
        self.tuple_label.pack()
        self.tuple_entry = tk.Entry(root, width=50)
        self.tuple_entry.pack()
        # Input for index
        self.index_label = tk.Label(root, text="Enter index:")
        self.index_label.pack()
        self.index_entry = tk.Entry(root, width=10)
        self.index_entry.pack()
        # Button to get element
        self.get_button = tk.Button(root, text="Get Element", command=self.get_element)
        self.get_button.pack()
        # Result display
        self.result_label = tk.Label(root, text="Element at index:")
        self.result_label.pack()
        self.result_display = tk.Label(root, text="", fg="blue")
        self.result_display.pack()
    def get_element(self):
        '''
        Retrieves the element from the tuple at the specified index and displays it.
        '''
        try:
            # Parse the tuple from the entry
            tuple_input = self.tuple_entry.get()
            input_tuple = ast.literal_eval(tuple_input)
            if not isinstance(input_tuple, tuple):
                raise ValueError("Input must be a tuple.")
            # Validate and parse the index
            index_input = self.index_entry.get()
            if not index_input.isdigit():
                raise ValueError("Index must be an integer.")
            index = int(index_input)
            # Get the element
            element = get_tuple_element(input_tuple, index)
            if element is not None:
                self.result_display.config(text=str(element))
            else:
                messagebox.showerror("Error", "Index out of range.")
        except (ValueError, SyntaxError) as ve:
            messagebox.showerror("Error", str(ve))
if __name__ == "__main__":
    root = tk.Tk()
    app = TupleApp(root)
    root.mainloop()
