'''

Main application file for the String Extractor GUI.

'''

import tkinter as tk
from tkinter import messagebox
from string_extractor import extract_strings_of_size

class StringExtractorApp:
    def __init__(self, root):
        self.root = root
        self.root.title("String Extractor")

        # Input for list of strings
        self.label_strings = tk.Label(root, text="Enter strings (comma-separated):")
        self.label_strings.pack()
        self.entry_strings = tk.Entry(root, width=50)
        self.entry_strings.pack()

        # Input for size of strings to extract
        self.label_size = tk.Label(root, text="Enter size of strings to extract:")
        self.label_size.pack()
        self.entry_size = tk.Entry(root, width=10)
        self.entry_size.pack()

        # Button to trigger extraction
        self.button_extract = tk.Button(root, text="Extract", command=self.extract_strings)
        self.button_extract.pack()

        # Output area
        self.label_result = tk.Label(root, text="Result:")
        self.label_result.pack()
        self.text_result = tk.Text(root, height=10, width=50)
        self.text_result.pack()

    def extract_strings(self):
        try:
            # Get input values
            strings = self.entry_strings.get().split(',')
            size = int(self.entry_size.get())

            # Extract strings of specified size
            result = extract_strings_of_size(strings, size)

            # Display result
            self.text_result.delete(1.0, tk.END)
            self.text_result.insert(tk.END, ', '.join(result))
        except ValueError:
            messagebox.showerror("Input Error", "Please enter a valid size.")

    def run(self):
        self.root.mainloop()

if __name__ == "__main__":
    root = tk.Tk()
    app = StringExtractorApp(root)
    app.run()
```

#### Utility Module: `string_extractor.py`

```python
'''

Utility module for extracting strings of a specified size.

'''

def extract_strings_of_size(strings, size):
    '''
    Extract strings of a specified size from a list.

    :param strings: List of strings to filter.
    :param size: The size of strings to extract.
    :return: List of strings that match the specified size.
    '''
    return [s.strip() for s in strings if len(s.strip()) == size]
```

Main application file for the String Extractor GUI.

import tkinter as tk
from tkinter import messagebox
from string_extractor import extract_strings_of_size

    def __init__(self, root):

        self.root.title("String Extractor")

import tkinter as tk
from tkinter import messagebox
from string_extractor import extract_strings_of_size
class StringExtractorApp:
    def __init__(self, root):
        self.root = root
        self.root.title("String Extractor")
        # Input for list of strings
        self.label_strings = tk.Label(root, text="Enter strings (comma-separated):")
        self.label_strings.pack()
        self.entry_strings = tk.Entry(root, width=50)
        self.entry_strings.pack()
        # Input for size of strings to extract
        self.label_size = tk.Label(root, text="Enter size of strings to extract:")
        self.label_size.pack()
        self.entry_size = tk.Entry(root, width=10)
        self.entry_size.pack()
        # Button to trigger extraction
        self.button_extract = tk.Button(root, text="Extract", command=self.extract_strings)
        self.button_extract.pack()
        # Output area
        self.label_result = tk.Label(root, text="Result:")
        self.label_result.pack()
        self.text_result = tk.Text(root, height=10, width=50)
        self.text_result.pack()
    def extract_strings(self):
        try:
            # Get input values
            strings = self.entry_strings.get().split(',')
            size = int(self.entry_size.get())

            self.text_result.delete(1.0, tk.END)

        except ValueError:

'''
Utility module for extracting strings of a specified size.
'''
def extract_strings_of_size(strings, size):
    '''
    Extract strings of a specified size from a list.
    :param strings: List of strings to filter.
    :param size: The size of strings to extract.
    :return: List of strings that match the specified size.
    '''
    return [s.strip() for s in strings if len(s.strip()) == size]

'''
Main application file for the String Extractor GUI.

import tkinter as tk
from tkinter import messagebox
from string_extractor import extract_strings_of_size
class StringExtractorApp:
    def __init__(self, root):
        self.root = root
        self.root.title("String Extractor")
        # Input for list of strings
        self.label_strings = tk.Label(root, text="Enter strings (comma-separated):")
        self.label_strings.pack()
        self.entry_strings = tk.Entry(root, width=50)
        self.entry_strings.pack()
        # Input for size of strings to extract
        self.label_size = tk.Label(root, text="Enter size of strings to extract:")
        self.label_size.pack()
        self.entry_size = tk.Entry(root, width=10)
        self.entry_size.pack()
        # Button to trigger extraction
        self.button_extract = tk.Button(root, text="Extract", command=self.extract_strings)
        self.button_extract.pack()
        # Output area
        self.label_result = tk.Label(root, text="Result:")
        self.label_result.pack()
        self.text_result = tk.Text(root, height=10, width=50)
        self.text_result.pack()
    def extract_strings(self):
        try:
            # Get input values
            strings = self.entry_strings.get().split(',')
            size = int(self.entry_size.get())
            # Extract strings of specified size
            result = extract_strings_of_size(strings, size)
            # Display result
            self.text_result.delete(1.0, tk.END)
            self.text_result.insert(tk.END, ', '.join(result))
        except ValueError:
            messagebox.showerror("Input Error", "Please enter a valid size.")
    def run(self):
        self.root.mainloop()
if __name__ == "__main__":
    root = tk.Tk()
    app = StringExtractorApp(root)
    app.run()

'''
Utility module for extracting strings of a specified size.
'''
def extract_strings_of_size(strings, size):
    '''
    Extract strings of a specified size from a list.
    :param strings: List of strings to filter.
    :param size: The size of strings to extract.
    :return: List of strings that match the specified size.
    '''
    return [s.strip() for s in strings if len(s.strip()) == size]
