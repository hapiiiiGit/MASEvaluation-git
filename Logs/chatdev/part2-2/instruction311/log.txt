```python
'''

This module contains the MainApp class for the GUI and the function to find the last position of an element in a sorted array.

'''

import tkinter as tk
from tkinter import messagebox

def find_last_position_in_sorted_array(array, element):
    '''
    Finds the last position of an element in a sorted array.

    Parameters:
    array (list of int): The sorted array.
    element (int): The element to find.

    Returns:
    int: The last position of the element in the array, or -1 if not found.
    '''
    last_position = -1
    for i in range(len(array)):
        if array[i] == element:
            last_position = i
    return last_position

class MainApp:
    '''
    Main application class for the GUI.
    '''
    def __init__(self, root):
        self.root = root
        self.root.title("Find Last Position in Sorted Array")

        self.label_array = tk.Label(root, text="Enter sorted array (comma-separated):")
        self.label_array.pack()

        self.entry_array = tk.Entry(root, width=50)
        self.entry_array.pack()

        self.label_element = tk.Label(root, text="Enter element to find:")
        self.label_element.pack()

        self.entry_element = tk.Entry(root, width=50)
        self.entry_element.pack()

        self.button_find = tk.Button(root, text="Find Last Position", command=self.find_last_position)
        self.button_find.pack()

        self.label_result = tk.Label(root, text="")
        self.label_result.pack()

    def find_last_position(self):
        try:
            array_str = self.entry_array.get()
            element_str = self.entry_element.get()

            array = list(map(int, array_str.split(',')))
            element = int(element_str)

            last_position = find_last_position_in_sorted_array(array, element)

            if last_position != -1:
                self.label_result.config(text=f"Last position of {element}: {last_position}")
            else:
                self.label_result.config(text=f"Element {element} not found in array.")
        except ValueError:
            messagebox.showerror("Invalid input", "Please enter valid numbers.")

if __name__ == "__main__":
    root = tk.Tk()
    app = MainApp(root)
    root.mainloop()
```

'''

This module contains the MainApp class for the GUI and the function to find the last position of an element in a sorted array.

'''

import tkinter as tk
from tkinter import messagebox

def find_last_position_in_sorted_array(array, element):
    '''
    Finds the last position of an element in a sorted array.

    Parameters:
    array (list of int): The sorted array.
    element (int): The element to find.

    Returns:
    int: The last position of the element in the array, or -1 if not found.
    '''
    last_position = -1
    for i in range(len(array)):
        if array[i] == element:
            last_position = i
    return last_position

class MainApp:
    '''
    Main application class for the GUI.
    '''
    def __init__(self, root):
        self.root = root
        self.root.title("Find Last Position in Sorted Array")

        self.label_array = tk.Label(root, text="Enter sorted array (comma-separated):")
        self.label_array.pack()

        self.entry_array = tk.Entry(root, width=50)
        self.entry_array.pack()

        self.label_element = tk.Label(root, text="Enter element to find:")
        self.label_element.pack()

        self.entry_element = tk.Entry(root, width=50)
        self.entry_element.pack()

        self.button_find = tk.Button(root, text="Find Last Position", command=self.find_last_position)
        self.button_find.pack()

        self.label_result = tk.Label(root, text="")
        self.label_result.pack()

    def find_last_position(self):
        try:
            array_str = self.entry_array.get()
            element_str = self.entry_element.get()

            array = list(map(int, array_str.split(',')))
            element = int(element_str)

            last_position = find_last_position_in_sorted_array(array, element)

            if last_position != -1:
                self.label_result.config(text=f"Last position of {element}: {last_position}")
            else:
                self.label_result.config(text=f"Element {element} not found in array.")
        except ValueError:
            messagebox.showerror("Invalid input", "Please enter valid numbers.")

if __name__ == "__main__":
    root = tk.Tk()
    app = MainApp(root)
    root.mainloop()

    low, high = 0, len(array) - 1
    last_position = -1
    while low <= high:
        mid = (low + high) // 2

            low = mid + 1
        else:

import tkinter as tk
from tkinter import messagebox

def find_last_position_in_sorted_array(array, element):
    '''
    Finds the last position of an element in a sorted array using binary search.
    Parameters:
    array (list of int): The sorted array.
    element (int): The element to find.
    Returns:
    int: The last position of the element in the array, or -1 if not found.

    low, high = 0, len(array) - 1
    last_position = -1

            low = mid + 1  # Continue searching in the right half
        elif array[mid] < element:
            low = mid + 1
        else:
            high = mid - 1

import tkinter as tk

def find_last_position_in_sorted_array(array, element):
    '''
    Finds the last position of an element in a sorted array using binary search.
    Parameters:
    array (list of int): The sorted array.
    element (int): The element to find.
    Returns:
    int: The last position of the element in the array, or -1 if not found.
    '''
    low, high = 0, len(array) - 1
    last_position = -1
    while low <= high:
        mid = (low + high) // 2
        if array[mid] == element:
            last_position = mid
            low = mid + 1  # Continue searching in the right half
        elif array[mid] < element:
            low = mid + 1
        else:
            high = mid - 1
    return last_position
