```python
'''

This file contains the main application class and the date conversion function for converting dates from yyyy-mm-dd format to dd-mm-yyyy format using a tkinter GUI.

'''

import tkinter as tk
from tkinter import messagebox

def convert_date_format(date_str):
    '''
    Convert a date from yyyy-mm-dd format to dd-mm-yyyy format.

    Parameters:
    date_str (str): Date in yyyy-mm-dd format.

    Returns:
    str: Date in dd-mm-yyyy format.
    '''
    try:
        year, month, day = date_str.split('-')
        return f"{day}-{month}-{year}"
    except ValueError:
        return None

class DateConverterApp:
    '''
    Main application class for the date converter GUI.
    '''
    def __init__(self, root):
        self.root = root
        self.root.title("Date Converter")

        # Create and place widgets
        self.label = tk.Label(root, text="Enter date (yyyy-mm-dd):")
        self.label.pack(pady=10)

        self.entry = tk.Entry(root)
        self.entry.pack(pady=5)

        self.convert_button = tk.Button(root, text="Convert", command=self.convert_date)
        self.convert_button.pack(pady=10)

        self.result_label = tk.Label(root, text="")
        self.result_label.pack(pady=10)

    def convert_date(self):
        '''
        Handle the conversion of the date input by the user.
        '''
        date_str = self.entry.get()
        converted_date = convert_date_format(date_str)
        if converted_date:
            self.result_label.config(text=f"Converted Date: {converted_date}")
        else:
            messagebox.showerror("Error", "Invalid date format. Please enter date in yyyy-mm-dd format.")

if __name__ == "__main__":
    root = tk.Tk()
    app = DateConverterApp(root)
    root.mainloop()
```

'''

This file contains the main application class and the date conversion function for converting dates from yyyy-mm-dd format to dd-mm-yyyy format using a tkinter GUI.

'''

import tkinter as tk
from tkinter import messagebox

def convert_date_format(date_str):
    '''
    Convert a date from yyyy-mm-dd format to dd-mm-yyyy format.

    Parameters:
    date_str (str): Date in yyyy-mm-dd format.

    Returns:
    str: Date in dd-mm-yyyy format.
    '''
    try:
        year, month, day = date_str.split('-')
        return f"{day}-{month}-{year}"
    except ValueError:
        return None

class DateConverterApp:
    '''
    Main application class for the date converter GUI.
    '''
    def __init__(self, root):
        self.root = root
        self.root.title("Date Converter")

        # Create and place widgets
        self.label = tk.Label(root, text="Enter date (yyyy-mm-dd):")
        self.label.pack(pady=10)

        self.entry = tk.Entry(root)
        self.entry.pack(pady=5)

        self.convert_button = tk.Button(root, text="Convert", command=self.convert_date)
        self.convert_button.pack(pady=10)

        self.result_label = tk.Label(root, text="")
        self.result_label.pack(pady=10)

    def convert_date(self):
        '''
        Handle the conversion of the date input by the user.
        '''
        date_str = self.entry.get()
        converted_date = convert_date_format(date_str)
        if converted_date:
            self.result_label.config(text=f"Converted Date: {converted_date}")
        else:
            messagebox.showerror("Error", "Invalid date format. Please enter date in yyyy-mm-dd format.")

if __name__ == "__main__":
    root = tk.Tk()
    app = DateConverterApp(root)
    root.mainloop()

import tkinter as tk
from tkinter import messagebox

def convert_date_format(date_str):
    '''
    Convert a date from yyyy-mm-dd format to dd-mm-yyyy format.

'''

import tkinter as tk
from tkinter import messagebox
def convert_date_format(date_str):
    '''
    Convert a date from yyyy-mm-dd format to dd-mm-yyyy format.
    Parameters:
    date_str (str): Date in yyyy-mm-dd format.
    Returns:
    str: Date in dd-mm-yyyy format.
    '''
    try:
        year, month, day = date_str.split('-')
        # Validate that year, month, and day are numeric and within valid ranges
        if len(year) == 4 and len(month) == 2 and len(day) == 2 and \
           1 <= int(month) <= 12 and 1 <= int(day) <= 31:
            return f"{day}-{month}-{year}"
        else:
            return None
    except ValueError:
        return None

This file contains the main application class and the date conversion function for converting dates from yyyy-mm-dd format to dd-mm-yyyy format using a tkinter GUI.
'''
import tkinter as tk
from tkinter import messagebox
def convert_date_format(date_str):
    '''
    Convert a date from yyyy-mm-dd format to dd-mm-yyyy format.
    Parameters:
    date_str (str): Date in yyyy-mm-dd format.
    Returns:
    str: Date in dd-mm-yyyy format.
    '''
    try:
        year, month, day = date_str.split('-')
        # Validate that year, month, and day are numeric and within valid ranges
        if len(year) == 4 and len(month) == 2 and len(day) == 2 and \
           1 <= int(month) <= 12 and 1 <= int(day) <= 31:
            return f"{day}-{month}-{year}"
        else:
            return None
    except ValueError:
        return None
class DateConverterApp:
    '''
    Main application class for the date converter GUI.
    '''
    def __init__(self, root):
        self.root = root
        self.root.title("Date Converter")
        self.root.geometry("300x200")  # Set a fixed window size
        self.root.resizable(False, False)  # Disable resizing of the window
        # Create and place widgets
        self.label = tk.Label(root, text="Enter date (yyyy-mm-dd):")
        self.label.pack(pady=10)
        self.entry = tk.Entry(root)
        self.entry.pack(pady=5)
        self.convert_button = tk.Button(root, text="Convert", command=self.convert_date)
        self.convert_button.pack(pady=10)
        self.result_label = tk.Label(root, text="", fg="blue")
        self.result_label.pack(pady=10)
    def convert_date(self):
        '''
        Handle the conversion of the date input by the user.
        '''
        date_str = self.entry.get().strip()  # Remove any leading/trailing spaces
        converted_date = convert_date_format(date_str)
        if converted_date:
            self.result_label.config(text=f"Converted Date: {converted_date}")
        else:
            messagebox.showerror("Error", "Invalid date format. Please enter date in yyyy-mm-dd format.")
if __name__ == "__main__":
    root = tk.Tk()
    app = DateConverterApp(root)
    root.mainloop()

'''
This file contains the main application class and the date conversion function for converting dates from yyyy-mm-dd format to dd-mm-yyyy format using a tkinter GUI.
'''
import tkinter as tk
from tkinter import messagebox
def convert_date_format(date_str):
    '''
    Convert a date from yyyy-mm-dd format to dd-mm-yyyy format.
    Parameters:
    date_str (str): Date in yyyy-mm-dd format.
    Returns:
    str: Date in dd-mm-yyyy format.
    '''
    try:
        year, month, day = date_str.split('-')
        # Validate that year, month, and day are numeric and within valid ranges
        if len(year) == 4 and len(month) == 2 and len(day) == 2 and \
           1 <= int(month) <= 12 and 1 <= int(day) <= 31:
            return f"{day}-{month}-{year}"
        else:
            return None
    except ValueError:
        return None
class DateConverterApp:
    '''
    Main application class for the date converter GUI.
    '''
    def __init__(self, root):
        self.root = root
        self.root.title("Date Converter")
        self.root.geometry("300x200")  # Set a fixed window size
        self.root.resizable(False, False)  # Disable resizing of the window
        # Create and place widgets
        self.label = tk.Label(root, text="Enter date (yyyy-mm-dd):")
        self.label.pack(pady=10)
        self.entry = tk.Entry(root)
        self.entry.pack(pady=5)
        self.convert_button = tk.Button(root, text="Convert", command=self.convert_date)
        self.convert_button.pack(pady=10)
        self.result_label = tk.Label(root, text="", fg="blue")
        self.result_label.pack(pady=10)
    def convert_date(self):
        '''
        Handle the conversion of the date input by the user.
        '''
        date_str = self.entry.get().strip()  # Remove any leading/trailing spaces
        converted_date = convert_date_format(date_str)
        if converted_date:
            self.result_label.config(text=f"Converted Date: {converted_date}")
        else:
            messagebox.showerror("Error", "Invalid date format. Please enter date in yyyy-mm-dd format.")
if __name__ == "__main__":
    root = tk.Tk()
    app = DateConverterApp(root)
    root.mainloop()
