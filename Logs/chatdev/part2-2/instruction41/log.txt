'''

This file initializes and runs the GUI application for calculating the n'th star number.

'''

import tkinter as tk
from tkinter import messagebox
from star_number import calculate_star_number


class StarNumberApp:
    def __init__(self, root):
        self.root = root
        self.root.title("Star Number Calculator")
        self.root.geometry("400x200")
        
        # Create GUI components
        self.label = tk.Label(root, text="Enter n (positive integer):", font=("Arial", 12))
        self.label.pack(pady=10)
        
        self.entry = tk.Entry(root, font=("Arial", 12))
        self.entry.pack(pady=10)
        
        self.calculate_button = tk.Button(root, text="Calculate Star Number", font=("Arial", 12), command=self.calculate_star_number)
        self.calculate_button.pack(pady=10)
        
        self.result_label = tk.Label(root, text="", font=("Arial", 12), fg="blue")
        self.result_label.pack(pady=10)
    
    def calculate_star_number(self):
        try:
            n = int(self.entry.get())
            if n <= 0:
                raise ValueError("n must be a positive integer.")
            star_number = calculate_star_number(n)
            self.result_label.config(text=f"The {n}'th star number is: {star_number}")

import tkinter as tk
from tkinter import messagebox
from star_number import calculate_star_number
class StarNumberApp:
    def __init__(self, root):
        """
        Initialize the Star Number Calculator application.
        Parameters:
        root (tk.Tk): The root window of the application.
        """
        self.root = root
        self.root.title("Star Number Calculator")
        self.root.geometry("400x200")
        # Create GUI components
        self.label = tk.Label(root, text="Enter n (positive integer):", font=("Arial", 12))
        self.label.pack(pady=10)
        self.entry = tk.Entry(root, font=("Arial", 12))
        self.entry.pack(pady=10)
        self.calculate_button = tk.Button(root, text="Calculate Star Number", font=("Arial", 12), command=self.calculate_star_number)
        self.calculate_button.pack(pady=10)
        self.result_label = tk.Label(root, text="", font=("Arial", 12), fg="blue")
        self.result_label.pack(pady=10)
    def calculate_star_number(self):
        """
        Handle the calculation of the star number based on user input.
        """
        try:
            n = int(self.entry.get())
            if n <= 0:
                raise ValueError("n must be a positive integer.")
            star_number = calculate_star_number(n)
            self.result_label.config(text=f"The {n}'th star number is: {star_number}")
        except ValueError as e:
            messagebox.showerror("Invalid Input", str(e))
if __name__ == "__main__":
    root = tk.Tk()
    app = StarNumberApp(root)
    root.mainloop()

'''

import tkinter as tk
from tkinter import messagebox
from star_number import calculate_star_number
class StarNumberApp:
    def __init__(self, root):
        """
        Initialize the Star Number Calculator application.
        Parameters:
        root (tk.Tk): The root window of the application.
        """
        self.root = root
        self.root.title("Star Number Calculator")
        self.root.geometry("400x250")
        # Create GUI components
        self.label = tk.Label(root, text="Enter n (positive integer):", font=("Arial", 12))
        self.label.pack(pady=10)
        self.entry = tk.Entry(root, font=("Arial", 12))
        self.entry.pack(pady=10)
        self.calculate_button = tk.Button(root, text="Calculate Star Number", font=("Arial", 12), command=self.calculate_star_number)
        self.calculate_button.pack(pady=10)
        self.result_label = tk.Label(root, text="", font=("Arial", 12), fg="blue")
        self.result_label.pack(pady=10)
        self.clear_button = tk.Button(root, text="Clear", font=("Arial", 12), command=self.clear_input)
        self.clear_button.pack(pady=10)
    def calculate_star_number(self):
        """
        Handle the calculation of the star number based on user input.
        """
        try:
            n = self.entry.get()
            if not n.isdigit():
                raise ValueError("Input must be a positive integer.")
            n = int(n)
            if n <= 0:
                raise ValueError("n must be a positive integer.")
            star_number = calculate_star_number(n)
            self.result_label.config(text=f"The {n}'th star number is: {star_number}")
        except ValueError as e:
            messagebox.showerror("Invalid Input", str(e))
    def clear_input(self):
        """
        Clear the input field and result label.
        """
        self.entry.delete(0, tk.END)
        self.result_label.config(text="")
if __name__ == "__main__":
    root = tk.Tk()
    app = StarNumberApp(root)
    root.mainloop()

'''
This file initializes and runs the GUI application for calculating the n'th star number.

import tkinter as tk
from tkinter import messagebox
from star_number import calculate_star_number
class StarNumberApp:
    def __init__(self, root):
        """
        Initialize the Star Number Calculator application.
        Parameters:
        root (tk.Tk): The root window of the application.
        """
        self.root = root
        self.root.title("Star Number Calculator")
        self.root.geometry("400x250")
        # Create GUI components
        self.label = tk.Label(root, text="Enter n (positive integer):", font=("Arial", 12))
        self.label.pack(pady=10)
        self.entry = tk.Entry(root, font=("Arial", 12))
        self.entry.pack(pady=10)
        self.calculate_button = tk.Button(root, text="Calculate Star Number", font=("Arial", 12), command=self.calculate_star_number)
        self.calculate_button.pack(pady=10)
        self.result_label = tk.Label(root, text="", font=("Arial", 12), fg="blue")
        self.result_label.pack(pady=10)
        self.clear_button = tk.Button(root, text="Clear", font=("Arial", 12), command=self.clear_input)
        self.clear_button.pack(pady=10)
    def calculate_star_number(self):
        """
        Handle the calculation of the star number based on user input.
        """
        try:
            n = self.entry.get()
            if not n.isdigit():
                raise ValueError("Input must be a positive integer.")
            n = int(n)
            if n <= 0:
                raise ValueError("n must be a positive integer.")
            star_number = calculate_star_number(n)
            self.result_label.config(text=f"The {n}'th star number is: {star_number}")
        except ValueError as e:
            messagebox.showerror("Invalid Input", str(e))
    def clear_input(self):
        """
        Clear the input field and result label.
        """
        self.entry.delete(0, tk.END)
        self.result_label.config(text="")
if __name__ == "__main__":
    root = tk.Tk()
    app = StarNumberApp(root)
    root.mainloop()

'''
This module contains the logic for calculating the n'th star number.
'''
def calculate_star_number(n):
    """
    Calculate the n'th star number using the formula:
    S_n = 6n(n-1) + 1
    Parameters:
    n (int): The position in the star number sequence (must be a positive integer).
    Returns:
    int: The n'th star number.
    """
    if n <= 0:
        raise ValueError("n must be a positive integer.")
    return 6 * n * (n - 1) + 1
