'''

Main application file for checking if the count of divisors is even using a GUI.

'''

import tkinter as tk
from utils import count_divisors, is_even

class MainApplication:
    def __init__(self, root):
        self.root = root
        self.root.title("Divisor Count Checker")

        self.label = tk.Label(root, text="Enter a number:")
        self.label.pack()

        self.entry = tk.Entry(root)
        self.entry.pack()

        self.button = tk.Button(root, text="Check", command=self.check_divisors)
        self.button.pack()

        self.result_label = tk.Label(root, text="")
        self.result_label.pack()

    def check_divisors(self):
        try:
            number = int(self.entry.get())
            divisor_count = count_divisors(number)
            if is_even(divisor_count):

def count_divisors(n):
    '''
    Calculate the number of divisors of a given integer n.
    Args:
        n (int): The number to calculate divisors for.
    Returns:
        int: The count of divisors of n.
    '''
    count = 0
    for i in range(1, n + 1):
        if n % i == 0:
            count += 1
    return count

    '''
    Check if a given number n is even.
    Args:
        n (int): The number to check.
    Returns:
        bool: True if n is even, False otherwise.
    '''
    return n % 2 == 0

'''
Main application file for checking if the count of divisors is even using a GUI.

import tkinter as tk
from utils import count_divisors, is_even
class MainApplication:
    def __init__(self, root):
        '''
        Initialize the main application with GUI components.
        '''
        self.root = root
        self.root.title("Divisor Count Checker")
        self.label = tk.Label(root, text="Enter a number:")
        self.label.pack()
        self.entry = tk.Entry(root)
        self.entry.pack()
        self.button = tk.Button(root, text="Check", command=self.check_divisors)
        self.button.pack()
        self.result_label = tk.Label(root, text="")
        self.result_label.pack()
    def check_divisors(self):
        '''
        Handle the logic for checking divisors and updating the result label.
        '''
        try:
            number = int(self.entry.get())
            divisor_count = count_divisors(number)
            if is_even(divisor_count):
                result_text = f"The count of divisors of {number} is even."
            else:
                result_text = f"The count of divisors of {number} is odd."
            self.result_label.config(text=result_text)
        except ValueError:
            self.result_label.config(text="Please enter a valid integer.")
if __name__ == "__main__":
    root = tk.Tk()
    app = MainApplication(root)
    root.mainloop()

'''
Utility functions for calculating divisors and checking evenness.

def count_divisors(n):
    '''
    Calculate the number of divisors of a given integer n.
    Args:
        n (int): The number to calculate divisors for.
    Returns:
        int: The count of divisors of n.
    '''
    count = 0
    for i in range(1, n + 1):
        if n % i == 0:
            count += 1
    return count
def is_even(n):
    '''
    Check if a given number n is even.
    Args:
        n (int): The number to check.
    Returns:
        bool: True if n is even, False otherwise.
    '''
    return n % 2 == 0
