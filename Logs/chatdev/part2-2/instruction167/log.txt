'''
Main application file for the Toggle Case application using tkinter.
'''
import tkinter as tk
from toggle_case import toggle_case_function
class ToggleCaseApp:
    def __init__(self, root):
        self.root = root
        self.root.title("Toggle Case Application")
        self.label = tk.Label(root, text="Enter a string:")
        self.label.pack()
        self.entry = tk.Entry(root, width=50)
        self.entry.pack()
        self.button = tk.Button(root, text="Toggle Case", command=self.toggle_case)
        self.button.pack()
        self.result_label = tk.Label(root, text="")
        self.result_label.pack()
    def toggle_case(self):
        input_string = self.entry.get()
        toggled_string = toggle_case_function(input_string)
        self.result_label.config(text=f"Toggled String: {toggled_string}")
    def run(self):
        self.root.mainloop()
if __name__ == "__main__":
    root = tk.Tk()
    app = ToggleCaseApp(root)
    app.run()
