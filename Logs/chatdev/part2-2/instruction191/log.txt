'''

This is the main file for the application. It contains the MainApp class, which implements the graphical user interface (GUI) for the program. The application allows users to load an Excel file, specify the number of most expensive items to retrieve, and display the results.

'''

import tkinter as tk
from tkinter import filedialog, messagebox
from tkinter import ttk
import pandas as pd
from helper import read_excel, get_top_n_expensive_items


class MainApp:
    def __init__(self, root):
        self.root = root
        self.root.title("Find Most Expensive Items")
        self.root.geometry("600x400")
        
        # GUI Components
        self.file_label = tk.Label(root, text="No file selected", wraplength=400)
        self.file_label.pack(pady=10)
        
        self.load_button = tk.Button(root, text="Load Excel File", command=self.load_file)
        self.load_button.pack(pady=10)
        
        self.n_label = tk.Label(root, text="Enter the number of most expensive items (n):")
        self.n_label.pack(pady=5)
        
        self.n_entry = tk.Entry(root)
        self.n_entry.pack(pady=5)
        
        self.find_button = tk.Button(root, text="Find Most Expensive Items", command=self.find_expensive_items)
        self.find_button.pack(pady=10)
        
        self.results_label = tk.Label(root, text="Results:")
        self.results_label.pack(pady=10)
        
        self.results_text = tk.Text(root, height=10, width=70)
        self.results_text.pack(pady=10)
        
        self.data = None  # Placeholder for the loaded data

    def load_file(self):
        """Opens a file dialog to load an Excel file."""
        file_path = filedialog.askopenfilename(filetypes=[("Excel files", "*.xlsx *.xls")])
        if file_path:
            self.file_label.config(text=f"Loaded file: {file_path}")
            try:
                self.data = read_excel(file_path)
                messagebox.showinfo("Success", "File loaded successfully!")
            except Exception as e:
                messagebox.showerror("Error", f"Failed to load file: {e}")
        else:
            self.file_label.config(text="No file selected")

    def find_expensive_items(self):
        """Finds the n most expensive items in the dataset."""
        if self.data is None:
            messagebox.showerror("Error", "No file loaded. Please load an Excel file first.")

import pandas as pd
def read_excel(file_path):
    """
    Reads an Excel file and returns the data as a pandas DataFrame.
    Args:
        file_path (str): The path to the Excel file.
    Returns:
        pd.DataFrame: The data from the Excel file.
    """
    try:
        return pd.read_excel(file_path)
    except Exception as e:
        raise ValueError(f"Error reading Excel file: {e}")
def get_top_n_expensive_items(data, n):
    """
    Finds the n most expensive items in the dataset.
    Args:
        data (pd.DataFrame): The dataset containing items and their prices.
        n (int): The number of most expensive items to find.
    Returns:
        pd.DataFrame: A DataFrame containing the n most expensive items.
    """
    # Ensure the dataset has the required columns
    if 'Item' not in data.columns or 'Price' not in data.columns:
        raise ValueError("The dataset must contain 'Item' and 'Price' columns.")
    try:
        # Sort the data by price in descending order and get the top n items
        sorted_data = data.sort_values(by='Price', ascending=False)
        return sorted_data.head(n)
    except Exception as e:
        raise ValueError(f"Error processing data: {e}")

import pandas as pd

def read_excel(file_path):
    """
    Reads an Excel file and returns the data as a pandas DataFrame.
    Args:
        file_path (str): The path to the Excel file.
    Returns:
        pd.DataFrame: The data from the Excel file.
    """
    try:
        return pd.read_excel(file_path)
    except Exception as e:
        raise ValueError(f"Error reading Excel file: {e}")

def get_top_n_expensive_items(data, n):
    """
    Finds the n most expensive items in the dataset.
    Args:
        data (pd.DataFrame): The dataset containing items and their prices.
        n (int): The number of most expensive items to find.
    Returns:
        pd.DataFrame: A DataFrame containing the n most expensive items.
    """
    # Ensure the dataset has the required columns
    if 'Item' not in data.columns or 'Price' not in data.columns:
        raise ValueError("The dataset must contain 'Item' and 'Price' columns.")
    try:
        # Sort the data by price in descending order and get the top n items
        sorted_data = data.sort_values(by='Price', ascending=False)
        return sorted_data.head(n)
    except Exception as e:
        raise ValueError(f"Error processing data: {e}")

if __name__ == "__main__":
    root = tk.Tk()

import pandas as pd
def read_excel(file_path):
    """
    Reads an Excel file and returns the data as a pandas DataFrame.
    Args:
        file_path (str): The path to the Excel file.
    Returns:
        pd.DataFrame: The data from the Excel file.
    """
    try:
        return pd.read_excel(file_path)
    except Exception as e:
        raise ValueError(f"Error reading Excel file: {e}")
def get_top_n_expensive_items(data, n):
    """
    Finds the n most expensive items in the dataset.
    Args:
        data (pd.DataFrame): The dataset containing items and their prices.
        n (int): The number of most expensive items to find.
    Returns:
        pd.DataFrame: A DataFrame containing the n most expensive items.
    """
    # Ensure the dataset has the required columns
    if 'Item' not in data.columns or 'Price' not in data.columns:
        raise ValueError("The dataset must contain 'Item' and 'Price' columns.")
    try:
        # Sort the data by price in descending order and get the top n items
        sorted_data = data.sort_values(by='Price', ascending=False)
        return sorted_data.head(n)
    except Exception as e:
        raise ValueError(f"Error processing data: {e}")

import pandas as pd
def read_excel(file_path):
    """
    Reads an Excel file and returns the data as a pandas DataFrame.
    Args:
        file_path (str): The path to the Excel file.
    Returns:
        pd.DataFrame: The data from the Excel file.
    """
    try:
        return pd.read_excel(file_path)
    except Exception as e:
        raise ValueError(f"Error reading Excel file: {e}")
def get_top_n_expensive_items(data, n):
    """
    Finds the n most expensive items in the dataset.
    Args:
        data (pd.DataFrame): The dataset containing items and their prices.
        n (int): The number of most expensive items to find.
    Returns:
        pd.DataFrame: A DataFrame containing the n most expensive items.
    """
    # Ensure the dataset has the required columns
    if 'Item' not in data.columns or 'Price' not in data.columns:
        raise ValueError("The dataset must contain 'Item' and 'Price' columns.")
    try:
        # Sort the data by price in descending order and get the top n items
        sorted_data = data.sort_values(by='Price', ascending=False)
        return sorted_data.head(n)
    except Exception as e:
        raise ValueError(f"Error processing data: {e}")

## 🤔 What is this?

Large language models (LLMs) are emerging as a transformative technology, enabling developers to build applications that they previously could not. However, using these LLMs in isolation is often insufficient for creating a truly powerful app - the real power comes when you can combine them with other sources of computation or knowledge.

This library aims to assist in the development of those types of applications. Common examples of these applications include:

# Find Most Expensive Items Application

Welcome to the user manual for the "Find Most Expensive Items" application. This software allows users to load an Excel file, specify the number of most expensive items to retrieve, and display the results in a user-friendly graphical interface.

## Main Functions of the Software

1. **Load Excel File**: Users can load an Excel file containing data about items and their prices. The application reads the file and prepares the data for processing.

2. **Specify Number of Items**: Users can input the number of most expensive items they wish to retrieve from the dataset.

3. **Display Results**: The application processes the data and displays the n most expensive items in a text widget within the application interface.

## Installation Guide

### Environment Dependencies

To run the "Find Most Expensive Items" application, you need to have Python installed on your system along with the required dependencies. Follow the steps below to set up your environment:

1. **Install Python**: Ensure that Python is installed on your system. You can download it from the [official Python website](https://www.python.org/downloads/).

2. **Install Required Packages**: The application requires the `pandas` library. You can install it using pip:

   ```bash
   pip install pandas>=1.1.4
   ```

   Alternatively, you can use the `requirements.txt` file to install the dependencies:

   ```bash
   pip install -r requirements.txt
   ```

## How to Use the Application

1. **Launch the Application**: Run the `main.py` file to start the application. You can do this by navigating to the directory containing the file and executing the following command in your terminal or command prompt:

   ```bash
   python main.py
   ```

2. **Load an Excel File**: Click on the "Load Excel File" button to open a file dialog. Select the Excel file containing your dataset. The file should have at least two columns: 'Item' and 'Price'.

3. **Enter the Number of Items**: In the text entry field, input the number of most expensive items you wish to retrieve. Ensure that this is a positive integer.

4. **Find Most Expensive Items**: Click on the "Find Most Expensive Items" button. The application will process the data and display the results in the text widget below.

5. **View Results**: The results will be displayed in the text area, showing the n most expensive items from your dataset.

## Troubleshooting

- **Error Loading File**: Ensure that the file you are trying to load is a valid Excel file with the correct format.

- **Invalid Input for n**: Make sure to enter a positive integer in the input field for the number of items.

- **Missing Columns**: The dataset must contain 'Item' and 'Price' columns. If these are missing, the application will not be able to process the data.

## Support

For further assistance, please contact our support team or refer to the documentation provided with the application.
