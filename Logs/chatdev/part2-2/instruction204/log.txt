'''

This module contains the main application logic for checking Armstrong numbers using a GUI.

'''

import tkinter as tk
from tkinter import messagebox

def is_armstrong_number(number):
    '''
    Check if a number is an Armstrong number.

    An Armstrong number is an n-digit number that is equal to the sum of its own digits each raised to the power of n.
    '''
    digits = [int(d) for d in str(number)]
    num_digits = len(digits)
    sum_of_powers = sum(d ** num_digits for d in digits)
    return sum_of_powers == number

class ArmstrongCheckerApp:
    '''
    GUI application for checking Armstrong numbers.
    '''
    def __init__(self, root):
        self.root = root
        self.root.title("Armstrong Number Checker")

        # Create and place widgets
        self.label = tk.Label(root, text="Enter a number:")
        self.label.pack(pady=10)

        self.entry = tk.Entry(root)
        self.entry.pack(pady=5)

        self.check_button = tk.Button(root, text="Check", command=self.on_check_button_click)
        self.check_button.pack(pady=10)

    def on_check_button_click(self):
        '''
        Handle the button click event to check if the entered number is an Armstrong number.
        '''
        try:
            number = int(self.entry.get())
            if is_armstrong_number(number):
                messagebox.showinfo("Result", f"{number} is an Armstrong number.")
            else:
                messagebox.showinfo("Result", f"{number} is not an Armstrong number.")
        except ValueError:
            messagebox.showerror("Invalid input", "Please enter a valid integer.")

if __name__ == "__main__":
    root = tk.Tk()
    app = ArmstrongCheckerApp(root)
    root.mainloop()
```

- **is_armstrong_number**: This function calculates whether a given number is an Armstrong number by summing the powers of its digits and comparing the result to the original number.
- **ArmstrongCheckerApp**: This class sets up the GUI using `tkinter`. It includes a label, an entry field for user input, and a button to trigger the check. The result is displayed in a message box.
- **on_check_button_click**: This method retrieves the user input, checks if it's a valid integer, and then uses `is_armstrong_number` to determine if it's an Armstrong number, displaying the result in a message box.

This setup provides a simple and interactive way for users to check Armstrong numbers using a GUI application.

'''

This module contains the main application logic for checking Armstrong numbers using a GUI.

'''

import tkinter as tk
from tkinter import messagebox

def is_armstrong_number(number):
    '''
    Check if a number is an Armstrong number.

    An Armstrong number is an n-digit number that is equal to the sum of its own digits each raised to the power of n.
    '''
    digits = [int(d) for d in str(number)]
    num_digits = len(digits)
    sum_of_powers = sum(d ** num_digits for d in digits)
    return sum_of_powers == number

class ArmstrongCheckerApp:
    '''
    GUI application for checking Armstrong numbers.
    '''
    def __init__(self, root):
        self.root = root
        self.root.title("Armstrong Number Checker")

        # Create and place widgets
        self.label = tk.Label(root, text="Enter a number:")
        self.label.pack(pady=10)

        self.entry = tk.Entry(root)
        self.entry.pack(pady=5)

        self.check_button = tk.Button(root, text="Check", command=self.on_check_button_click)
        self.check_button.pack(pady=10)

    def on_check_button_click(self):
        '''
        Handle the button click event to check if the entered number is an Armstrong number.
        '''

            number = int(self.entry.get())
            if is_armstrong_number(number):
                messagebox.showinfo("Result", f"{number} is an Armstrong number.")

            messagebox.showerror("Invalid input", "Please enter a valid integer.")

if __name__ == "__main__":
    root = tk.Tk()
    app = ArmstrongCheckerApp(root)
    root.mainloop()

import tkinter as tk
from tkinter import messagebox

def is_armstrong_number(number):
    '''
    Check if a number is an Armstrong number.
    An Armstrong number is an n-digit number that is equal to the sum of its own digits each raised to the power of n.
    '''
    digits = [int(d) for d in str(number)]
    num_digits = len(digits)
    sum_of_powers = sum(d ** num_digits for d in digits)
    return sum_of_powers == number

class ArmstrongCheckerApp:
    '''
    GUI application for checking Armstrong numbers.
    '''
    def __init__(self, root):
        self.root = root
        self.root.title("Armstrong Number Checker")
        # Create and place widgets
        self.label = tk.Label(root, text="Enter a number:")
        self.label.pack(pady=10)
        self.entry = tk.Entry(root)
        self.entry.pack(pady=5)
        self.check_button = tk.Button(root, text="Check", command=self.on_check_button_click)
        self.check_button.pack(pady=10)

    def on_check_button_click(self):
        '''
        Handle the button click event to check if the entered number is an Armstrong number.
        '''
        try:
            number = int(self.entry.get())
            if is_armstrong_number(number):
                messagebox.showinfo("Result", f"{number} is an Armstrong number.")
            else:
                messagebox.showinfo("Result", f"{number} is not an Armstrong number.")
        except ValueError:
            messagebox.showerror("Invalid input", "Please enter a valid integer.")

if __name__ == "__main__":
    root = tk.Tk()
    app = ArmstrongCheckerApp(root)
    root.mainloop()

'''
This module contains the main application logic for checking Armstrong numbers using a GUI.
'''
import tkinter as tk
from tkinter import messagebox
def is_armstrong_number(number):
    '''
    Check if a number is an Armstrong number.
    An Armstrong number is an n-digit number that is equal to the sum of its own digits each raised to the power of n.
    '''
    digits = [int(d) for d in str(number)]
    num_digits = len(digits)
    sum_of_powers = sum(d ** num_digits for d in digits)
    return sum_of_powers == number
class ArmstrongCheckerApp:
    '''
    GUI application for checking Armstrong numbers.
    '''
    def __init__(self, root):
        '''
        Initialize the GUI application and create widgets.
        '''
        self.root = root
        self.root.title("Armstrong Number Checker")
        # Create and place widgets
        self.label = tk.Label(root, text="Enter a number:")
        self.label.pack(pady=10)
        self.entry = tk.Entry(root)
        self.entry.pack(pady=5)
        self.check_button = tk.Button(root, text="Check", command=self.on_check_button_click)
        self.check_button.pack(pady=10)
    def on_check_button_click(self):
        '''
        Handle the button click event to check if the entered number is an Armstrong number.
        '''
        try:
            number = int(self.entry.get())
            if is_armstrong_number(number):
                messagebox.showinfo("Result", f"{number} is an Armstrong number.")
            else:
                messagebox.showinfo("Result", f"{number} is not an Armstrong number.")
        except ValueError:
            messagebox.showerror("Invalid input", "Please enter a valid integer.")
if __name__ == "__main__":
    '''
    Entry point for the application.
    '''
    root = tk.Tk()
    app = ArmstrongCheckerApp(root)
    root.mainloop()
