'''

This module contains the logic for checking the frequency of digits in an integer.

'''

def check_digit_frequency(integer):
    """
    Check whether the frequency of each digit in the integer is less than or equal to the digit itself.

    Parameters:
    integer (int): The integer to be checked.

    Returns:
    bool: True if the frequency of each digit is less than or equal to the digit itself, False otherwise.
    """
    str_integer = str(integer)
    digit_count = {}
    
    # Count the frequency of each digit
    for digit in str_integer:
        if digit in digit_count:
            digit_count[digit] += 1
        else:
            digit_count[digit] = 1
    
    # Check if frequency of each digit is less than or equal to the digit itself
    for digit, count in digit_count.items():
        if count > int(digit):
            return False
    
    return True
```

'''

This module contains the logic for checking the frequency of digits in an integer.

'''

def check_digit_frequency(integer):
    """
    Check whether the frequency of each digit in the integer is less than or equal to the digit itself.

    Parameters:
    integer (int): The integer to be checked.

    Returns:
    bool: True if the frequency of each digit is less than or equal to the digit itself, False otherwise.
    """
    str_integer = str(integer)
    digit_count = {}
    
    # Count the frequency of each digit
    for digit in str_integer:
        if digit in digit_count:
            digit_count[digit] += 1
        else:
            digit_count[digit] = 1
    
    # Check if frequency of each digit is less than or equal to the digit itself
    for digit, count in digit_count.items():
        if count > int(digit):
            return False
    
    return True

'''
This module contains the logic for checking the frequency of digits in an integer.
'''
def check_digit_frequency(integer):
    """
    Check whether the frequency of each digit in the integer is less than or equal to the digit itself.
    Parameters:
    integer (int): The integer to be checked.
    Returns:
    bool: True if the frequency of each digit is less than or equal to the digit itself, False otherwise.
    """
    str_integer = str(integer)
    digit_count = {}
    # Count the frequency of each digit
    for digit in str_integer:
        if digit in digit_count:
            digit_count[digit] += 1
        else:
            digit_count[digit] = 1
    # Check if frequency of each digit is less than or equal to the digit itself
    for digit, count in digit_count.items():
        if count > int(digit):
            return False
    return True

def check_digit_frequency(integer):
    """
    Check whether the frequency of each digit in the integer is less than or equal to the digit itself.
    Parameters:
    integer (int): The integer to be checked.
    Returns:
    bool: True if the frequency of each digit is less than or equal to the digit itself, False otherwise.
    """
    str_integer = str(integer)
    digit_count = {}
    # Count the frequency of each digit
    for digit in str_integer:
        if digit in digit_count:
            digit_count[digit] += 1
        else:
            digit_count[digit] = 1
    # Check if frequency of each digit is less than or equal to the digit itself
    for digit, count in digit_count.items():
        if count > int(digit):
            return False
    return True

'''
This module contains the logic for checking the frequency of digits in an integer.
'''
def check_digit_frequency(integer):
    """
    Check whether the frequency of each digit in the integer is less than or equal to the digit itself.
    Parameters:
    integer (int): The integer to be checked.
    Returns:
    bool: True if the frequency of each digit is less than or equal to the digit itself, False otherwise.
    """
    str_integer = str(integer)
    digit_count = {}
    # Count the frequency of each digit
    for digit in str_integer:
        if digit in digit_count:
            digit_count[digit] += 1
        else:
            digit_count[digit] = 1
    # Check if frequency of each digit is less than or equal to the digit itself
    for digit, count in digit_count.items():
        if count > int(digit):
            return False
    return True

'''
This module contains the logic for checking the frequency of digits in an integer.
'''
def check_digit_frequency(integer):
    """
    Check whether the frequency of each digit in the integer is less than or equal to the digit itself.
    Parameters:
    integer (int): The integer to be checked.
    Returns:
    bool: True if the frequency of each digit is less than or equal to the digit itself, False otherwise.
    """
    str_integer = str(integer)
    digit_count = {}
    # Count the frequency of each digit
    for digit in str_integer:
        if digit in digit_count:
            digit_count[digit] += 1
        else:
            digit_count[digit] = 1
    # Check if frequency of each digit is less than or equal to the digit itself
    for digit, count in digit_count.items():
        if count > int(digit):
            return False
    return True
