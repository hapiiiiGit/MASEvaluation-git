'''

This file contains the main application class for the Parity Checker App and the function to check if a number's parity is odd.

'''

import tkinter as tk
from tkinter import messagebox

def is_odd(number):
    '''
    Determine if the given number is odd.

    Parameters:
    number (int): The number to check.

    Returns:
    bool: True if the number is odd, False otherwise.
    '''
    return number % 2 != 0

class ParityCheckerApp:
    '''
    A simple GUI application to check the parity of a number.
    '''

    def __init__(self, root):
        '''
        Initialize the application with the main window and widgets.

        Parameters:
        root (tk.Tk): The root window of the application.
        '''
        self.root = root
        self.root.title("Parity Checker")

        # Create and place widgets
        self.label = tk.Label(root, text="Enter a number:")
        self.label.pack()

        self.entry = tk.Entry(root)
        self.entry.pack()

        self.check_button = tk.Button(root, text="Check Parity", command=self.check_parity)
        self.check_button.pack()

    def check_parity(self):
        '''
        Check the parity of the number entered in the entry widget.
        '''
        try:
            number = int(self.entry.get())
            if is_odd(number):
                messagebox.showinfo("Result", f"The number {number} is odd.")
            else:
                messagebox.showinfo("Result", f"The number {number} is even.")
        except ValueError:
            messagebox.showerror("Invalid Input", "Please enter a valid integer.")

if __name__ == "__main__":
    root = tk.Tk()
    app = ParityCheckerApp(root)
    root.mainloop()

```

'''
This file contains the main application class for the Parity Checker App and the function to check if a number's parity is odd.
'''
import tkinter as tk
from tkinter import messagebox

def is_odd(number):
    '''
    Determine if the given number is odd.
    
    Parameters:
    number (int): The number to check.
    
    Returns:
    bool: True if the number is odd, False otherwise.
    '''
    return number % 2 != 0

class ParityCheckerApp:
    '''
    A simple GUI application to check the parity of a number.
    '''
    def __init__(self, root):
        '''
        Initialize the application with the main window and widgets.
        
        Parameters:
        root (tk.Tk): The root window of the application.
        '''
        self.root = root
        self.root.title("Parity Checker")
        
        # Create and place widgets
        self.label = tk.Label(root, text="Enter a number:")
        self.label.pack()
        
        self.entry = tk.Entry(root)
        self.entry.pack()
        
        self.check_button = tk.Button(root, text="Check Parity", command=self.check_parity)
        self.check_button.pack()

    def check_parity(self):
        '''
        Check the parity of the number entered in the entry widget.
        '''
        try:
            number = int(self.entry.get())
            if is_odd(number):
                messagebox.showinfo("Result", f"The number {number} is odd.")
            else:
                messagebox.showinfo("Result", f"The number {number} is even.")
        except ValueError:
            messagebox.showerror("Invalid Input", "Please enter a valid integer.")

if __name__ == "__main__":
    root = tk.Tk()
    app = ParityCheckerApp(root)
    root.mainloop()

'''
This file contains the main application class for the Parity Checker App and the function to check if a number's parity is odd.

import tkinter as tk
from tkinter import messagebox
def is_odd(number):
    '''
    Determine if the given number is odd.
    Parameters:
    number (int): The number to check.
    Returns:
    bool: True if the number is odd, False otherwise.
    '''
    return number % 2 != 0
class ParityCheckerApp:
    '''
    A simple GUI application to check the parity of a number.
    '''
    def __init__(self, root):
        '''
        Initialize the application with the main window and widgets.
        Parameters:
        root (tk.Tk): The root window of the application.
        '''
        self.root = root
        self.root.title("Parity Checker")
        # Create and place widgets
        self.label = tk.Label(root, text="Enter a number:")
        self.label.pack()
        self.entry = tk.Entry(root)
        self.entry.pack()
        self.check_button = tk.Button(root, text="Check Parity", command=self.check_parity)
        self.check_button.pack()
    def check_parity(self):
        '''
        Check the parity of the number entered in the entry widget.
        '''
        try:
            number = int(self.entry.get())
            if is_odd(number):
                messagebox.showinfo("Result", f"The number {number} is odd.")
            else:
                messagebox.showinfo("Result", f"The number {number} is even.")
        except ValueError:
            messagebox.showerror("Invalid Input", "Please enter a valid integer.")
if __name__ == "__main__":
    root = tk.Tk()
    app = ParityCheckerApp(root)
    root.mainloop()
