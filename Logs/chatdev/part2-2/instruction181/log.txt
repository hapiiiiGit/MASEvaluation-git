'''

This file contains the main implementation of the Cylinder Surface Area Calculator application.
It includes a GUI built using tkinter and a function to calculate the surface area of a cylinder.

'''

import tkinter as tk
from tkinter import messagebox


class CylinderCalculator:
    """
    A class to calculate the surface area of a cylinder.
    """

    @staticmethod
    def calculate_surface_area(radius, height):
        """
        Calculate the surface area of a cylinder given its radius and height.

        Surface Area = 2 * π * r * (r + h)

        :param radius: The radius of the cylinder.
        :param height: The height of the cylinder.
        :return: The surface area of the cylinder.
        """
        import math
        return 2 * math.pi * radius * (radius + height)


class CylinderApp:
    """
    A class to implement the GUI for the Cylinder Surface Area Calculator.
    """

    def __init__(self):
        """
        Initialize the GUI components.
        """
        self.root = tk.Tk()
        self.root.title("Cylinder Surface Area Calculator")

        # Radius input
        self.label_radius = tk.Label(self.root, text="Radius:")
        self.label_radius.grid(row=0, column=0, padx=10, pady=10)
        self.entry_radius = tk.Entry(self.root)

'''

This file contains the main implementation of the Cylinder Surface Area Calculator application.
It includes a GUI built using tkinter and a function to calculate the surface area of a cylinder.

'''

import tkinter as tk
from tkinter import messagebox


class CylinderCalculator:
    """
    A class to calculate the surface area of a cylinder.
    """

    @staticmethod
    def calculate_surface_area(radius, height):
        """
        Calculate the surface area of a cylinder given its radius and height.

        Surface Area = 2 * π * r * (r + h)

        :param radius: The radius of the cylinder.
        :param height: The height of the cylinder.
        :return: The surface area of the cylinder.
        """
        import math
        return 2 * math.pi * radius * (radius + height)


class CylinderApp:
    """
    A class to implement the GUI for the Cylinder Surface Area Calculator.
    """

    def __init__(self):
        """
        Initialize the GUI components.
        """
        self.root = tk.Tk()
        self.root.title("Cylinder Surface Area Calculator")

        # Radius input
        self.label_radius = tk.Label(self.root, text="Radius:")
        self.label_radius.grid(row=0, column=0, padx=10, pady=10)
        self.entry_radius = tk.Entry(self.root)
        self.entry_radius.grid(row=0, column=1, padx=10, pady=10)

        # Height input
        self.label_height = tk.Label(self.root, text="Height:")
        self.label_height.grid(row=1, column=0, padx=10, pady=10)
        self.entry_height = tk.Entry(self.root)
        self.entry_height.grid(row=1, column=1, padx=10, pady=10)

        # Calculate button
        self.button_calculate = tk.Button(self.root, text="Calculate", command=self.calculate_and_display)
        self.button_calculate.grid(row=2, column=0, columnspan=2, pady=10)

        # Result label
        self.label_result = tk.Label(self.root, text="Surface Area: ")
        self.label_result.grid(row=3, column=0, columnspan=2, pady=10)

    def calculate_and_display(self):
        """
        Fetch user input, calculate the surface area, and display the result.
        """
        try:
            # Get user input
            radius = float(self.entry_radius.get())
            height = float(self.entry_height.get())

            # Validate input
            if radius <= 0 or height <= 0:
                raise ValueError("Radius and height must be positive numbers.")

            # Calculate surface area
            surface_area = CylinderCalculator.calculate_surface_area(radius, height)

            # Display result
            self.label_result.config(text=f"Surface Area: {surface_area:.2f}")
        except ValueError as e:
            # Show error message
            messagebox.showerror("Input Error", str(e))

    def run(self):
        """
        Run the application.
        """
        self.root.mainloop()


if __name__ == "__main__":
    app = CylinderApp()
    app.run()

import tkinter as tk
from tkinter import messagebox
class CylinderCalculator:
    """
    A class to calculate the surface area of a cylinder.
    """
    @staticmethod
    def calculate_surface_area(radius, height):
        """
        Calculate the surface area of a cylinder given its radius and height.
        Surface Area = 2 * π * r * (r + h)
        :param radius: The radius of the cylinder.
        :param height: The height of the cylinder.
        :return: The surface area of the cylinder.
        """
        import math
        return 2 * math.pi * radius * (radius + height)
class CylinderApp:
    """
    A class to implement the GUI for the Cylinder Surface Area Calculator.
    """

        self.entry_radius = tk.Entry(self.root)

        # Height input
        self.label_height = tk.Label(self.root, text="Height:")
        self.label_height.grid(row=1, column=0, padx=10, pady=10)
        self.entry_height = tk.Entry(self.root)
        self.entry_height.grid(row=1, column=1, padx=10, pady=10)
        # Calculate button
        self.button_calculate = tk.Button(self.root, text="Calculate", command=self.calculate_and_display)
        self.button_calculate.grid(row=2, column=0, columnspan=2, pady=10)
        # Result label
        self.label_result = tk.Label(self.root, text="Surface Area: ")
        self.label_result.grid(row=3, column=0, columnspan=2, pady=10)
    def calculate_and_display(self):
        """
        Fetch user input, calculate the surface area, and display the result.
        """
        try:
            # Get user input
            radius = float(self.entry_radius.get())

            # Validate input
            if radius <= 0 or height <= 0:

        except ValueError as e:
            # Show error message

        """

'''
This file contains the main implementation of the Cylinder Surface Area Calculator application.
It includes a GUI built using tkinter and a function to calculate the surface area of a cylinder.
'''
import tkinter as tk
from tkinter import messagebox
class CylinderCalculator:
    """
    A class to calculate the surface area of a cylinder.
    """
    @staticmethod
    def calculate_surface_area(radius, height):
        """
        Calculate the surface area of a cylinder given its radius and height.
        Surface Area = 2 * π * r * (r + h)
        :param radius: The radius of the cylinder.
        :param height: The height of the cylinder.
        :return: The surface area of the cylinder.
        """
        import math
        return 2 * math.pi * radius * (radius + height)
class CylinderApp:
    """
    A class to implement the GUI for the Cylinder Surface Area Calculator.
    """
    def __init__(self):
        """
        Initialize the GUI components.
        """
        self.root = tk.Tk()
        self.root.title("Cylinder Surface Area Calculator")
        # Radius input
        self.label_radius = tk.Label(self.root, text="Radius:")
        self.label_radius.grid(row=0, column=0, padx=10, pady=10)
        self.entry_radius = tk.Entry(self.root)
        self.entry_radius.grid(row=0, column=1, padx=10, pady=10)
        # Height input
        self.label_height = tk.Label(self.root, text="Height:")
        self.label_height.grid(row=1, column=0, padx=10, pady=10)
        self.entry_height = tk.Entry(self.root)
        self.entry_height.grid(row=1, column=1, padx=10, pady=10)
        # Calculate button
        self.button_calculate = tk.Button(self.root, text="Calculate", command=self.calculate_and_display)
        self.button_calculate.grid(row=2, column=0, columnspan=2, pady=10)
        # Result label
        self.label_result = tk.Label(self.root, text="Surface Area: ")
        self.label_result.grid(row=3, column=0, columnspan=2, pady=10)
    def calculate_and_display(self):
        """
        Fetch user input, calculate the surface area, and display the result.
        """
        try:
            # Get user input
            radius = float(self.entry_radius.get())
            height = float(self.entry_height.get())
            # Validate input
            if radius <= 0 or height <= 0:
                raise ValueError("Radius and height must be positive numbers.")
            # Calculate surface area
            surface_area = CylinderCalculator.calculate_surface_area(radius, height)
            # Display result
            self.label_result.config(text=f"Surface Area: {surface_area:.2f}")
        except ValueError as e:
            # Show error message
            messagebox.showerror("Input Error", str(e))
    def run(self):
        """
        Run the application.
        """
        self.root.mainloop()
if __name__ == "__main__":
    app = CylinderApp()
    app.run()

    """
    @staticmethod
    def calculate_surface_area(radius, height):
        """
        Calculate the surface area of a cylinder given its radius and height.
        Surface Area = 2 * π * r * (r + h)
        :param radius: The radius of the cylinder.
        :param height: The height of the cylinder.
        :return: The surface area of the cylinder.
        """
        import math
        return 2 * math.pi * radius * (radius + height)
