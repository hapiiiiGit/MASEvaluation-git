'''

This file contains the main application logic for the GUI that allows users to input a list of lists and find the list with the highest sum of elements.

'''

import tkinter as tk
from tkinter import messagebox
from utils import calculate_highest_sum_list

class MainApplication:
    def __init__(self, root):
        self.root = root
        self.root.title("List Sum Finder")

        self.input_label = tk.Label(root, text="Enter list of lists (e.g., [[1,2],[3,4],[5,6]]):")
        self.input_label.pack()

        self.input_entry = tk.Entry(root, width=50)
        self.input_entry.pack()

        self.submit_button = tk.Button(root, text="Find List with Highest Sum", command=self.find_highest_sum_list)
        self.submit_button.pack()

        self.result_label = tk.Label(root, text="")
        self.result_label.pack()

    def find_highest_sum_list(self):
        try:
            lists = eval(self.input_entry.get())
            if not all(isinstance(lst, list) for lst in lists):
                raise ValueError("Input must be a list of lists.")
            highest_sum_list = calculate_highest_sum_list(lists)
            self.result_label.config(text=f"List with highest sum: {highest_sum_list}")
        except Exception as e:
            messagebox.showerror("Error", f"Invalid input: {e}")

def main():
    root = tk.Tk()
    app = MainApplication(root)
    root.mainloop()

if __name__ == "__main__":
    main()
```

This file contains utility functions for the application, including the function to calculate the list with the highest sum of elements.

def calculate_highest_sum_list(lists):
    '''
    Calculate the list with the highest sum of elements from a list of lists.

    Parameters:
    lists (list of lists): The input list of lists.

    Returns:
    list: The list with the highest sum of elements.
    '''
    return max(lists, key=sum)

import tkinter as tk
from tkinter import messagebox
from utils import calculate_highest_sum_list

class MainApplication:
    def __init__(self, root):
        self.root = root
        self.root.title("List Sum Finder")

        self.input_label = tk.Label(root, text="Enter list of lists (e.g., [[1,2],[3,4],[5,6]]):")
        self.input_label.pack()

        self.input_entry = tk.Entry(root, width=50)
        self.input_entry.pack()

        self.submit_button = tk.Button(root, text="Find List with Highest Sum", command=self.find_highest_sum_list)
        self.submit_button.pack()

        self.result_label = tk.Label(root, text="")
        self.result_label.pack()

    def find_highest_sum_list(self):
        try:
            lists = eval(self.input_entry.get())
            if not all(isinstance(lst, list) for lst in lists):

import tkinter as tk
from tkinter import messagebox
from utils import calculate_highest_sum_list

class MainApplication:
    def __init__(self, root):
        self.root = root
        self.root.title("List Sum Finder")
        self.input_label = tk.Label(root, text="Enter list of lists (e.g., [[1,2],[3,4],[5,6]]):")
        self.input_label.pack()
        self.input_entry = tk.Entry(root, width=50)
        self.input_entry.pack()
        self.submit_button = tk.Button(root, text="Find List with Highest Sum", command=self.find_highest_sum_list)
        self.submit_button.pack()
        self.result_label = tk.Label(root, text="")
        self.result_label.pack()

    def find_highest_sum_list(self):
        try:
            lists = eval(self.input_entry.get())
            if not all(isinstance(lst, list) for lst in lists):
                raise ValueError("Input must be a list of lists.")
            highest_sum_list = calculate_highest_sum_list(lists)
            self.result_label.config(text=f"List with highest sum: {highest_sum_list}")
        except Exception as e:
            messagebox.showerror("Error", f"Invalid input: {e}")

def main():
    root = tk.Tk()
    app = MainApplication(root)
    root.mainloop()

if __name__ == "__main__":
    main()

import tkinter as tk
from tkinter import messagebox
from utils import calculate_highest_sum_list
class MainApplication:
    def __init__(self, root):
        '''
        Initialize the main application window and its components.
        Parameters:
        root (Tk): The root Tkinter window.
        '''
        self.root = root
        self.root.title("List Sum Finder")
        self.input_label = tk.Label(root, text="Enter list of lists (e.g., [[1,2],[3,4],[5,6]]):")
        self.input_label.pack()
        self.input_entry = tk.Entry(root, width=50)
        self.input_entry.pack()
        self.submit_button = tk.Button(root, text="Find List with Highest Sum", command=self.find_highest_sum_list)
        self.submit_button.pack()
        self.result_label = tk.Label(root, text="")
        self.result_label.pack()
    def find_highest_sum_list(self):
        '''
        Handle the button click event to find the list with the highest sum of elements.
        '''
        try:
            lists = eval(self.input_entry.get())
            if not isinstance(lists, list) or not all(isinstance(lst, list) for lst in lists):
                raise ValueError("Input must be a list of lists.")
            highest_sum_list = calculate_highest_sum_list(lists)
            self.result_label.config(text=f"List with highest sum: {highest_sum_list}")
        except Exception as e:
            messagebox.showerror("Error", f"Invalid input: {e}")
def main():
    '''
    Entry point for the application.
    '''
    root = tk.Tk()
    app = MainApplication(root)
    root.mainloop()
if __name__ == "__main__":
    main()

'''

import tkinter as tk
from tkinter import messagebox
from utils import calculate_highest_sum_list
class MainApplication:
    def __init__(self, root):
        '''
        Initialize the main application window and its components.
        Parameters:
        root (Tk): The root Tkinter window.
        '''
        self.root = root
        self.root.title("List Sum Finder")
        self.input_label = tk.Label(root, text="Enter list of lists (e.g., [[1,2],[3,4],[5,6]]):")
        self.input_label.pack()
        self.input_entry = tk.Entry(root, width=50)
        self.input_entry.pack()
        self.submit_button = tk.Button(root, text="Find List with Highest Sum", command=self.find_highest_sum_list)
        self.submit_button.pack()
        self.result_label = tk.Label(root, text="")
        self.result_label.pack()
    def find_highest_sum_list(self):
        '''
        Handle the button click event to find the list with the highest sum of elements.
        '''
        try:
            lists = eval(self.input_entry.get())
            if not isinstance(lists, list) or not all(isinstance(lst, list) for lst in lists):
                raise ValueError("Input must be a list of lists.")
            highest_sum_list = calculate_highest_sum_list(lists)
            self.result_label.config(text=f"List with highest sum: {highest_sum_list}")
        except Exception as e:
            messagebox.showerror("Error", f"Invalid input: {e}")
def main():
    '''
    Entry point for the application.
    '''
    root = tk.Tk()
    app = MainApplication(root)
    root.mainloop()
if __name__ == "__main__":
    main()

def calculate_highest_sum_list(lists):
    '''
    Calculate the list with the highest sum of elements from a list of lists.
    Parameters:
    lists (list of lists): The input list of lists.
    Returns:
    list: The list with the highest sum of elements.
    '''
    if not lists:
        raise ValueError("The input list of lists cannot be empty.")
    return max(lists, key=sum)
