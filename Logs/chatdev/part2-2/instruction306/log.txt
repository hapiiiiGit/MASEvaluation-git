'''

This file contains the MainApplication class for the GUI and the match_pattern function to check if a string contains the pattern "a" followed by three "b"s.

'''

import tkinter as tk
import re

def match_pattern(input_string):
    '''
    Checks if the input string contains the pattern "a" followed by three "b"s.

    Parameters:
    input_string (str): The string to be checked.

    Returns:
    bool: True if the pattern is found, otherwise False.
    '''
    pattern = r'a{1}b{3}'
    return bool(re.search(pattern, input_string))

class MainApplication:
    '''
    Main application class for the GUI.
    '''
    def __init__(self, root):
        '''
        Initializes the GUI components.

        Parameters:
        root (tk.Tk): The root window of the application.
        '''
        self.root = root
        self.root.title("Pattern Matcher")

        self.label = tk.Label(root, text="Enter a string:")
        self.label.pack()

        self.entry = tk.Entry(root)
        self.entry.pack()

        self.button = tk.Button(root, text="Check Pattern", command=self.check_pattern)
        self.button.pack()

        self.result_label = tk.Label(root, text="")
        self.result_label.pack()

    def check_pattern(self):
        '''
        Checks if the input string matches the pattern "a" followed by three "b"s and updates the result label.
        '''
        input_string = self.entry.get()
        if match_pattern(input_string):
            self.result_label.config(text="Match found!")
        else:
            self.result_label.config(text="No match found.")

if __name__ == "__main__":
    root = tk.Tk()
    app = MainApplication(root)
    root.mainloop()

```

import tkinter as tk

    Checks if the input string contains the pattern "a" followed by three "b"s.

import tkinter as tk

    input_string (str): The string to be checked.

This file contains the MainApplication class for the GUI and the match_pattern function to check if a string contains the pattern "a" followed by three "b"s.
'''
import tkinter as tk
import re
def match_pattern(input_string):
    '''
    Checks if the input string contains the pattern "a" followed by three "b"s.
    Parameters:
    input_string (str): The string to be checked.
    Returns:
    bool: True if the pattern is found, otherwise False.
    '''
    pattern = r'a{1}b{3}'
    return bool(re.search(pattern, input_string))
class MainApplication:
    '''
    Main application class for the GUI.
    '''
    def __init__(self, root):
        '''
        Initializes the GUI components.
        Parameters:
        root (tk.Tk): The root window of the application.
        '''
        self.root = root
        self.root.title("Pattern Matcher")
        self.label = tk.Label(root, text="Enter a string:")
        self.label.pack()
        self.entry = tk.Entry(root)
        self.entry.pack()
        self.button = tk.Button(root, text="Check Pattern", command=self.check_pattern)
        self.button.pack()
        self.result_label = tk.Label(root, text="")
        self.result_label.pack()
    def check_pattern(self):
        '''
        Checks if the input string matches the pattern "a" followed by three "b"s and updates the result label.
        '''
        input_string = self.entry.get()
        if match_pattern(input_string):
            self.result_label.config(text="Match found!")
        else:
            self.result_label.config(text="No match found.")
if __name__ == "__main__":
    root = tk.Tk()
    app = MainApplication(root)
    root.mainloop()
