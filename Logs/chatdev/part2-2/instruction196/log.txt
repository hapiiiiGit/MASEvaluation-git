```python
'''

This module contains the PolarToRectConverter class, which provides the logic for converting
polar coordinates to rectangular coordinates.

'''

import math

class PolarToRectConverter:
    @staticmethod
    def convert_to_rectangular(r, theta):
        '''
        Converts polar coordinates (r, θ) to rectangular coordinates (x, y).

        Parameters:
        r (float): The radius (distance from the origin).
        theta (float): The angle in degrees.

        Returns:
        tuple: A tuple containing the rectangular coordinates (x, y).
        '''
        # Convert theta from degrees to radians
        theta_rad = math.radians(theta)
        
        # Calculate rectangular coordinates
        x = r * math.cos(theta_rad)
        y = r * math.sin(theta_rad)
        
        return x, y
```

```python
'''

This module contains the PolarToRectConverter class, which provides the logic for converting
polar coordinates to rectangular coordinates.

'''

import math

class PolarToRectConverter:
    @staticmethod
    def convert_to_rectangular(r, theta):
        '''
        Converts polar coordinates (r, θ) to rectangular coordinates (x, y).

        Parameters:
        r (float): The radius (distance from the origin).
        theta (float): The angle in degrees.

        Returns:
        tuple: A tuple containing the rectangular coordinates (x, y).
        '''
        # Convert theta from degrees to radians
        theta_rad = math.radians(theta)
        
        # Calculate rectangular coordinates
        x = r * math.cos(theta_rad)
        y = r * math.sin(theta_rad)
        
        return x, y

'''
This module contains the PolarToRectConverter class, which provides the logic for converting
polar coordinates to rectangular coordinates.
'''
import math
class PolarToRectConverter:
    @staticmethod
    def convert_to_rectangular(r, theta):
        '''
        Converts polar coordinates (r, θ) to rectangular coordinates (x, y).
        Parameters:
        r (float): The radius (distance from the origin).
        theta (float): The angle in degrees.
        Returns:
        tuple: A tuple containing the rectangular coordinates (x, y).
        '''
        # Convert theta from degrees to radians
        theta_rad = math.radians(theta)
        # Calculate rectangular coordinates
        x = r * math.cos(theta_rad)
        y = r * math.sin(theta_rad)
        return x, y

'''  
This module contains the PolarToRectConverter class, which provides the logic for converting  
polar coordinates to rectangular coordinates.  
'''  
import math  
class PolarToRectConverter:  
    @staticmethod  
    def convert_to_rectangular(r, theta):  
        '''  
        Converts polar coordinates (r, θ) to rectangular coordinates (x, y).  
        Parameters:  
        r (float): The radius (distance from the origin).  
        theta (float): The angle in degrees.  
        Returns:  
        tuple: A tuple containing the rectangular coordinates (x, y).  
        '''  
        try:  
            # Convert theta from degrees to radians  
            theta_rad = math.radians(theta)  
            # Calculate rectangular coordinates  
            x = r * math.cos(theta_rad)  
            y = r * math.sin(theta_rad)  
            return x, y  
        except Exception as e:  
            raise ValueError(f"Error in conversion: {e}")

'''  
This module contains the PolarToRectConverter class, which provides the logic for converting  
polar coordinates to rectangular coordinates.  
'''  
import math  
class PolarToRectConverter:  
    @staticmethod  
    def convert_to_rectangular(r, theta):  
        '''  
        Converts polar coordinates (r, θ) to rectangular coordinates (x, y).  
        Parameters:  
        r (float): The radius (distance from the origin).  
        theta (float): The angle in degrees.  
        Returns:  
        tuple: A tuple containing the rectangular coordinates (x, y).  
        '''  
        try:  
            # Convert theta from degrees to radians  
            theta_rad = math.radians(theta)  
            # Calculate rectangular coordinates  
            x = r * math.cos(theta_rad)  
            y = r * math.sin(theta_rad)  
            return x, y  
        except Exception as e:  
            raise ValueError(f"Error in conversion: {e}")

import tkinter as tk
from tkinter import messagebox
from converter import PolarToRectConverter

class GUI:

        Initializes the GUI components.

        self.root = tk.Tk()
        self.root.title("Polar to Rectangular Converter")

            # Display result
            self.result_var.set(f"({x:.2f}, {y:.2f})")
        except ValueError as ve:
            # Handle invalid input
            messagebox.showerror("Invalid Input", f"Error: {ve}")
        except Exception as e:
            # Handle unexpected errors
            messagebox.showerror("Error", f"An unexpected error occurred: {e}")

    def run(self):
        '''
        Starts the main event loop for the GUI.

'''
This module contains the PolarToRectConverter class, which provides the logic for converting
polar coordinates to rectangular coordinates.
'''
import math
class PolarToRectConverter:
    @staticmethod
    def convert_to_rectangular(r, theta):
        '''
        Converts polar coordinates (r, θ) to rectangular coordinates (x, y).
        Parameters:
        r (float): The radius (distance from the origin).
        theta (float): The angle in degrees.
        Returns:
        tuple: A tuple containing the rectangular coordinates (x, y).
        '''
        try:
            # Convert theta from degrees to radians
            theta_rad = math.radians(theta)
            # Calculate rectangular coordinates
            x = r * math.cos(theta_rad)
            y = r * math.sin(theta_rad)
            return x, y
        except Exception as e:
            raise ValueError(f"Error in conversion: {e}")

'''
This module contains the PolarToRectConverter class, which provides the logic for converting
polar coordinates to rectangular coordinates.
'''
import math
class PolarToRectConverter:
    @staticmethod
    def convert_to_rectangular(r, theta):
        '''
        Converts polar coordinates (r, θ) to rectangular coordinates (x, y).
        Parameters:
        r (float): The radius (distance from the origin).
        theta (float): The angle in degrees.
        Returns:
        tuple: A tuple containing the rectangular coordinates (x, y).
        '''
        try:
            # Convert theta from degrees to radians
            theta_rad = math.radians(theta)
            # Calculate rectangular coordinates
            x = r * math.cos(theta_rad)
            y = r * math.sin(theta_rad)
            return x, y
        except Exception as e:
            raise ValueError(f"Error in conversion: {e}")

'''
This module contains the PolarToRectConverter class, which provides the logic for converting
polar coordinates to rectangular coordinates.
'''
import math
class PolarToRectConverter:
    @staticmethod
    def convert_to_rectangular(r, theta):
        '''
        Converts polar coordinates (r, θ) to rectangular coordinates (x, y).
        Parameters:
        r (float): The radius (distance from the origin).
        theta (float): The angle in degrees.
        Returns:
        tuple: A tuple containing the rectangular coordinates (x, y).
        '''
        try:
            # Convert theta from degrees to radians
            theta_rad = math.radians(theta)
            # Calculate rectangular coordinates
            x = r * math.cos(theta_rad)
            y = r * math.sin(theta_rad)
            return x, y
        except Exception as e:
            raise ValueError(f"Error in conversion: {e}")

'''
This module contains the PolarToRectConverter class, which provides the logic for converting
polar coordinates to rectangular coordinates.
'''
import math
class PolarToRectConverter:
    @staticmethod
    def convert_to_rectangular(r, theta):
        '''
        Converts polar coordinates (r, θ) to rectangular coordinates (x, y).
        Parameters:
        r (float): The radius (distance from the origin).
        theta (float): The angle in degrees.
        Returns:
        tuple: A tuple containing the rectangular coordinates (x, y).
        '''
        try:
            # Convert theta from degrees to radians
            theta_rad = math.radians(theta)
            # Calculate rectangular coordinates
            x = r * math.cos(theta_rad)
            y = r * math.sin(theta_rad)
            return x, y
        except Exception as e:
            raise ValueError(f"Error in conversion: {e}")

# Polar to Rectangular Coordinate Converter

Convert polar coordinates to rectangular coordinates with ease using this Python-based application.

## Introduction

The Polar to Rectangular Coordinate Converter is a user-friendly application designed to help users convert polar coordinates (radius `r` and angle `θ`) into rectangular coordinates (`x` and `y`). This application features a graphical user interface (GUI) for seamless interaction and provides accurate results based on mathematical calculations.

### Main Functions

1. **Polar to Rectangular Conversion**: 
   - Converts polar coordinates `(r, θ)` to rectangular coordinates `(x, y)` using the formula:
     - `x = r * cos(θ)`
