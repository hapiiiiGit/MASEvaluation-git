### utils.py

```python
'''

This utility file contains a function to find the minimum value in a heterogeneous list.

'''

def find_minimum_in_list(lst):
    '''
    Finds the minimum value in a heterogeneous list.

    Parameters:
    lst (list): The list containing elements of different types.

    Returns:
    The minimum value among comparable elements.
    '''
    # Filter out non-comparable elements
    comparable_elements = [x for x in lst if isinstance(x, (int, float, str))]
    
    if not comparable_elements:
        raise ValueError("No comparable elements in the list.")

    # Return the minimum value
    return min(comparable_elements)
```

'''

This utility file contains a function to find the minimum value in a heterogeneous list.

'''

def find_minimum_in_list(lst):
    '''
    Finds the minimum value in a heterogeneous list.

    Parameters:
    lst (list): The list containing elements of different types.

    Returns:
    The minimum value among comparable elements.
    '''
    # Filter out non-comparable elements
    comparable_elements = [x for x in lst if isinstance(x, (int, float, str))]
    
    if not comparable_elements:
        raise ValueError("No comparable elements in the list.")

    # Return the minimum value
    return min(comparable_elements)

def find_minimum_in_list(lst):
    '''
    Finds the minimum value in a heterogeneous list.

import ast
from utils import find_minimum_in_list
class MinValueApp:
    def __init__(self, root):
        self.root = root
        self.root.title("Minimum Value Finder")
        # Create input label and entry
        self.label = tk.Label(root, text="Enter a list (e.g., [1, 'a', 3.5, 2]):")

        self.result_label.pack()
    def find_minimum(self):
        try:
            # Safely evaluate the input string to a list
            input_list = ast.literal_eval(self.entry.get())
            if not isinstance(input_list, list):
                raise ValueError("Input is not a list.")
            # Find the minimum value using the utility function
            min_value = find_minimum_in_list(input_list)
            self.display_result(min_value)
        except Exception as e:

        self.result_label.config(text=f"Minimum value: {min_value}")
if __name__ == "__main__":
    root = tk.Tk()

    root.mainloop()

def find_minimum_in_list(lst):
    '''
    Finds the minimum value in a heterogeneous list.

    '''

'''
This is the main application file for the MinValueApp. It creates a GUI using tkinter to allow users to input a heterogeneous list and find the minimum value.
'''
import tkinter as tk
from tkinter import messagebox
import ast
from utils import find_minimum_in_list
class MinValueApp:
    def __init__(self, root):
        self.root = root
        self.root.title("Minimum Value Finder")
        # Create input label and entry
        self.label = tk.Label(root, text="Enter a list (e.g., [1, 'a', 3.5, 2]):")
        self.label.pack()
        self.entry = tk.Entry(root, width=50)
        self.entry.pack()
        # Create a button to find the minimum value
        self.find_button = tk.Button(root, text="Find Minimum", command=self.find_minimum)
        self.find_button.pack()
        # Create a label to display the result
        self.result_label = tk.Label(root, text="")
        self.result_label.pack()
    def find_minimum(self):
        try:
            # Safely evaluate the input string to a list
            input_list = ast.literal_eval(self.entry.get())
            if not isinstance(input_list, list):
                raise ValueError("Input is not a list.")
            # Find the minimum value using the utility function
            min_value = find_minimum_in_list(input_list)
            if min_value is None:
                self.display_result("No numeric elements found in the list.")
            else:
                self.display_result(min_value)
        except Exception as e:
            messagebox.showerror("Error", f"Invalid input: {e}")
    def display_result(self, min_value):
        self.result_label.config(text=f"Minimum value: {min_value}")
if __name__ == "__main__":
    root = tk.Tk()
    app = MinValueApp(root)
    root.mainloop()

'''
This utility file contains a function to find the minimum numeric value in a heterogeneous list.
'''
def find_minimum_in_list(lst):
    '''
    Finds the minimum numeric value in a heterogeneous list.
    Parameters:
    lst (list): The list containing elements of different types.
    Returns:
    The minimum numeric value or None if no numeric elements are found.
    '''
    # Filter out non-numeric elements
    numeric_elements = [x for x in lst if isinstance(x, (int, float))]
    if not numeric_elements:
        return None  # Return None if no numeric elements are found
    # Return the minimum numeric value
    return min(numeric_elements)
