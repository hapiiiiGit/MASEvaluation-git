def is_majority_element(array, n, element):
    '''
    Determine if the given element is the majority element in the sorted array.

    Parameters:
    array (list): The sorted array of integers.
    n (int): The length of the array.
    element (int): The element to check.

    Returns:
    bool: True if the element is the majority element, False otherwise.
    '''
    # Find the first occurrence of the element
    first_index = binary_search(array, 0, n - 1, element)

    # If the element is not present, return False
    if first_index == -1:
        return False

    # Check if the element appears more than n/2 times
    if (first_index + n // 2) < n and array[first_index + n // 2] == element:
        return True

    return False

def binary_search(arr, low, high, x):
    '''
    Perform binary search to find the first occurrence of x in arr.

    Parameters:
    arr (list): The array to search.
    low (int): The starting index.
    high (int): The ending index.
    x (int): The element to find.

    Returns:
    int: The index of the first occurrence of x, or -1 if not found.

'''
This module contains the logic to determine if an element is the majority element in a sorted array.
'''
def is_majority_element(array, n, element):
    '''
    Determine if the given element is the majority element in the sorted array.
    Parameters:
    array (list): The sorted array of integers.
    n (int): The length of the array.
    element (int): The element to check.
    Returns:
    bool: True if the element is the majority element, False otherwise.
    '''
    # Find the first occurrence of the element
    first_index = binary_search(array, 0, n - 1, element)
    # If the element is not present, return False
    if first_index == -1:
        return False
    # Check if the element appears more than n/2 times
    if (first_index + n // 2) < n and array[first_index + n // 2] == element:
        return True
    return False
def binary_search(arr, low, high, x):
    '''
    Perform binary search to find the first occurrence of x in arr.
    Parameters:
    arr (list): The array to search.
    low (int): The starting index.
    high (int): The ending index.
    x (int): The element to find.
    Returns:
    int: The index of the first occurrence of x, or -1 if not found.
    '''
    if high >= low:
        mid = (high + low) // 2
        # Check if mid is the first occurrence of x
        if (mid == 0 or x > arr[mid - 1]) and arr[mid] == x:
            return mid
        elif x > arr[mid]:
            return binary_search(arr, mid + 1, high, x)
        else:
            return binary_search(arr, low, mid - 1, x)
    return -1

This module contains the logic to determine if an element is the majority element in a sorted array.

def is_majority_element(array, n, element):
    '''
    Determine if the given element is the majority element in the sorted array.
    Parameters:
    array (list): The sorted array of integers.
    n (int): The length of the array.
    element (int): The element to check.
    Returns:
    bool: True if the element is the majority element, False otherwise.
    '''
    # Find the first occurrence of the element
    first_index = binary_search(array, 0, n - 1, element)
    # If the element is not present, return False
    if first_index == -1:
        return False
    # Check if the element appears more than n/2 times
    if (first_index + n // 2) < n and array[first_index + n // 2] == element:
        return True
    return False
def binary_search(arr, low, high, x):
    '''
    Perform binary search to find the first occurrence of x in arr.
    Parameters:
    arr (list): The array to search.
    low (int): The starting index.
    high (int): The ending index.
    x (int): The element to find.
    Returns:
    int: The index of the first occurrence of x, or -1 if not found.
    '''
    if high >= low:
        mid = (high + low) // 2
        # Check if mid is the first occurrence of x
        if (mid == 0 or x > arr[mid - 1]) and arr[mid] == x:
            return mid
        elif x > arr[mid]:
            return binary_search(arr, mid + 1, high, x)
        else:
            return binary_search(arr, low, mid - 1, x)
    return -1

def is_majority_element(array, n, element):
    '''
    Determine if the given element is the majority element in the sorted array.
    Parameters:
    array (list): The sorted array of integers.
    n (int): The length of the array.
    element (int): The element to check.
    Returns:
    bool: True if the element is the majority element, False otherwise.
    '''
    # Find the first occurrence of the element
    first_index = binary_search(array, 0, n - 1, element)
    # If the element is not present, return False
    if first_index == -1:
        return False
    # Check if the element appears more than n/2 times
    if (first_index + n // 2) < n and array[first_index + n // 2] == element:
        return True
    return False
def binary_search(arr, low, high, x):
    '''
    Perform binary search to find the first occurrence of x in arr.
    Parameters:
    arr (list): The array to search.
    low (int): The starting index.
    high (int): The ending index.
    x (int): The element to find.
    Returns:
    int: The index of the first occurrence of x, or -1 if not found.
    '''
    if high >= low:
        mid = (high + low) // 2
        # Check if mid is the first occurrence of x
        if (mid == 0 or x > arr[mid - 1]) and arr[mid] == x:
            return mid
        elif x > arr[mid]:
            return binary_search(arr, mid + 1, high, x)
        else:
            return binary_search(arr, low, mid - 1, x)
    return -1

def is_majority_element(array, n, element):
    '''
    Determine if the given element is the majority element in the sorted array.
    Parameters:
    array (list): The sorted array of integers.
    n (int): The length of the array.
    element (int): The element to check.
    Returns:
    bool: True if the element is the majority element, False otherwise.
    '''
    # Find the first occurrence of the element
    first_index = binary_search(array, 0, n - 1, element)
    # If the element is not present, return False
    if first_index == -1:
        return False
    # Check if the element appears more than n/2 times
    if (first_index + n // 2) < n and array[first_index + n // 2] == element:
        return True
    return False
def binary_search(arr, low, high, x):
    '''
    Perform binary search to find the first occurrence of x in arr.
    Parameters:
    arr (list): The array to search.
    low (int): The starting index.
    high (int): The ending index.
    x (int): The element to find.
    Returns:
    int: The index of the first occurrence of x, or -1 if not found.
    '''
    if low <= high:
        mid = (high + low) // 2
        # Check if mid is the first occurrence of x
        if (mid == 0 or x > arr[mid - 1]) and arr[mid] == x:
            return mid
        elif x > arr[mid]:
            return binary_search(arr, mid + 1, high, x)
        else:
            return binary_search(arr, low, mid - 1, x)
    return -1

def is_majority_element(array, n, element):
    '''

def is_majority_element(array, n, element):
    '''
    Determine if the given element is the majority element in the sorted array.
    Parameters:
    array (list): The sorted array of integers.
    n (int): The length of the array.
    element (int): The element to check.
    Returns:
    bool: True if the element is the majority element, False otherwise.
    '''
    # Find the first occurrence of the element
    first_index = binary_search(array, 0, n - 1, element)
    # If the element is not present, return False
    if first_index == -1:
        return False
    # Check if the element appears more than n/2 times
    if (first_index + n // 2) < n and array[first_index + n // 2] == element:
        return True
    return False
def binary_search(arr, low, high, x):
    '''
    Perform binary search to find the first occurrence of x in arr.
    Parameters:
    arr (list): The array to search.
    low (int): The starting index.
    high (int): The ending index.
    x (int): The element to find.
    Returns:
    int: The index of the first occurrence of x, or -1 if not found.
    '''
    if low <= high:
        mid = (high + low) // 2
        # Check if mid is the first occurrence of x
        if (mid == 0 or x > arr[mid - 1]) and arr[mid] == x:
            return mid
        elif x > arr[mid]:
            return binary_search(arr, mid + 1, high, x)
        else:
            return binary_search(arr, low, mid - 1, x)
    return -1

'''
This module contains the logic to determine if an element is the majority element in a sorted array.
'''
def is_majority_element(array, n, element):
    '''
    Determine if the given element is the majority element in the sorted array.
    Parameters:
    array (list): The sorted array of integers.
    n (int): The length of the array.
    element (int): The element to check.
    Returns:
    bool: True if the element is the majority element, False otherwise.
    '''
    # Find the first occurrence of the element
    first_index = binary_search(array, 0, n - 1, element)
    # If the element is not present, return False
    if first_index == -1:
        return False
    # Count occurrences of the element
    count = 0
    for i in range(first_index, n):
        if array[i] == element:
            count += 1
        else:
            break
    # Check if the element appears more than n/2 times
    return count > n // 2
def binary_search(arr, low, high, x):
    '''
    Perform binary search to find the first occurrence of x in arr.
    Parameters:
    arr (list): The array to search.
    low (int): The starting index.
    high (int): The ending index.
    x (int): The element to find.
    Returns:
    int: The index of the first occurrence of x, or -1 if not found.
    '''
    if low <= high:
        mid = (high + low) // 2
        # Check if mid is the first occurrence of x
        if (mid == 0 or x > arr[mid - 1]) and arr[mid] == x:
            return mid
        elif x > arr[mid]:
            return binary_search(arr, mid + 1, high, x)
        else:
            return binary_search(arr, low, mid - 1, x)
    return -1

'''
This module contains the logic to determine if an element is the majority element in a sorted array.
'''
def is_majority_element(array, n, element):
    '''
    Determine if the given element is the majority element in the sorted array.
    Parameters:
    array (list): The sorted array of integers.
    n (int): The length of the array.
    element (int): The element to check.
    Returns:
    bool: True if the element is the majority element, False otherwise.
    '''
    # Find the first occurrence of the element
    first_index = binary_search(array, 0, n - 1, element)
    # If the element is not present, return False
    if first_index == -1:
        return False
    # Count occurrences of the element
    count = 0
    for i in range(first_index, n):
        if array[i] == element:
            count += 1
        else:
            break
    # Check if the element appears more than n/2 times
    return count > n // 2
def binary_search(arr, low, high, x):
    '''
    Perform binary search to find the first occurrence of x in arr.
    Parameters:
    arr (list): The array to search.
    low (int): The starting index.
    high (int): The ending index.
    x (int): The element to find.
    Returns:
    int: The index of the first occurrence of x, or -1 if not found.
    '''
    if low <= high:
        mid = (high + low) // 2
        # Check if mid is the first occurrence of x
        if (mid == 0 or x > arr[mid - 1]) and arr[mid] == x:
            return mid
        elif x > arr[mid]:
            return binary_search(arr, mid + 1, high, x)
        else:
            return binary_search(arr, low, mid - 1, x)
    return -1

# Majority Element Checker

The Majority Element Checker is a Python-based application that allows users to determine if a given element is the majority element in a sorted array. The majority element is defined as an element that appears more than n/2 times in the array, where n is the length of the array.

## Main Functions

The application provides the following main functions:

- **Input Handling**: Users can input a sorted array and an element to check through a graphical user interface (GUI).
- **Majority Element Check**: The application checks if the specified element is the majority element in the array.
- **User Feedback**: The application provides feedback through message boxes indicating whether the element is the majority element or not.
