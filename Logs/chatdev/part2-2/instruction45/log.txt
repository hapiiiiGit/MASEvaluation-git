'''

This file contains the main application logic for the GUI that allows users to input a list of tuples and displays the rear elements of each tuple.

'''

import tkinter as tk
from tkinter import messagebox
from utils import extract_rear_elements_from_tuples

class MainApplication:
    def __init__(self, root):
        self.root = root
        self.root.title("Tuple Rear Element Extractor")

        self.label = tk.Label(root, text="Enter a list of tuples (e.g., [(1, 2), (3, 4)]):")
        self.label.pack()

        self.entry = tk.Entry(root, width=50)
        self.entry.pack()

        self.button = tk.Button(root, text="Extract Rear Elements", command=self.extract_rear_elements)
        self.button.pack()

        self.result_label = tk.Label(root, text="")
        self.result_label.pack()

    def extract_rear_elements(self):
        try:
            tuples_list = eval(self.entry.get())
            if not isinstance(tuples_list, list) or not all(isinstance(t, tuple) for t in tuples_list):
                raise ValueError("Input must be a list of tuples.")
            rear_elements = extract_rear_elements_from_tuples(tuples_list)
            self.result_label.config(text=f"Rear elements: {rear_elements}")
        except Exception as e:
            messagebox.showerror("Error", f"Invalid input: {e}")

if __name__ == "__main__":
    root = tk.Tk()
    app = MainApplication(root)
    root.mainloop()
```

import tkinter as tk
from tkinter import messagebox
from utils import extract_rear_elements_from_tuples
class MainApplication:
    def __init__(self, root):
        self.root = root
        self.root.title("Tuple Rear Element Extractor")
        self.label = tk.Label(root, text="Enter a list of tuples (e.g., [(1, 2), (3, 4)]):")
        self.label.pack()
        self.entry = tk.Entry(root, width=50)
        self.entry.pack()
        self.button = tk.Button(root, text="Extract Rear Elements", command=self.extract_rear_elements)
        self.button.pack()
        self.result_label = tk.Label(root, text="")
        self.result_label.pack()
    def extract_rear_elements(self):
        try:
            tuples_list = eval(self.entry.get())
            if not isinstance(tuples_list, list) or not all(isinstance(t, tuple) for t in tuples_list):
                raise ValueError("Input must be a list of tuples.")
            rear_elements = extract_rear_elements_from_tuples(tuples_list)
            self.result_label.config(text=f"Rear elements: {rear_elements}")
        except Exception as e:
            messagebox.showerror("Error", f"Invalid input: {e}")
if __name__ == "__main__":
    root = tk.Tk()
