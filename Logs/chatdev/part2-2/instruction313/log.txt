'''
This file contains the main application class for the SumDictionaryApp, which provides a GUI to calculate the sum of all items in a given dictionary.
'''

import tkinter as tk
from tkinter import messagebox

def calculate_sum(dictionary):
    '''
    Calculate the sum of all values in the given dictionary.

    :param dictionary: dict, the dictionary with numeric values
    :return: int/float, the sum of all values
    '''
    return sum(dictionary.values())

class SumDictionaryApp:
    '''
    A GUI application to calculate the sum of all items in a given dictionary.
    '''

    def __init__(self, root):
        '''
        Initialize the application with the main window and widgets.

        :param root: Tk, the main window of the application
        '''
        self.root = root
        self.root.title("Sum of Dictionary Values")

        # Create and place widgets
        self.label = tk.Label(root, text="Enter dictionary (e.g., {'a': 1, 'b': 2}):")
        self.label.pack()

        self.entry = tk.Entry(root, width=50)
        self.entry.pack()

        self.calculate_button = tk.Button(root, text="Calculate Sum", command=self.calculate)
        self.calculate_button.pack()

        self.result_label = tk.Label(root, text="Result: ")
        self.result_label.pack()

    def calculate(self):
        '''
        Handle the calculation of the sum when the button is pressed.
        '''
        try:
            # Get the dictionary from the entry
            user_input = self.entry.get()
            dictionary = eval(user_input)  # Convert string to dictionary

            if isinstance(dictionary, dict):
                # Calculate the sum
                result = calculate_sum(dictionary)
                self.result_label.config(text=f"Result: {result}")
            else:
                raise ValueError("Input is not a dictionary.")
        except Exception as e:
            messagebox.showerror("Error", f"Invalid input: {e}")

if __name__ == "__main__":
    root = tk.Tk()
    app = SumDictionaryApp(root)
    root.mainloop()
```

'''
This file contains the main application class for the SumDictionaryApp, which provides a GUI to calculate the sum of all items in a given dictionary.
'''

import tkinter as tk
from tkinter import messagebox

def calculate_sum(dictionary):
    '''
    Calculate the sum of all values in the given dictionary.

    :param dictionary: dict, the dictionary with numeric values
    :return: int/float, the sum of all values
    '''
    return sum(dictionary.values())

class SumDictionaryApp:
    '''
    A GUI application to calculate the sum of all items in a given dictionary.
    '''

    def __init__(self, root):
        '''
        Initialize the application with the main window and widgets.

        :param root: Tk, the main window of the application
        '''
        self.root = root
        self.root.title("Sum of Dictionary Values")

        # Create and place widgets
        self.label = tk.Label(root, text="Enter dictionary (e.g., {'a': 1, 'b': 2}):")
        self.label.pack()

        self.entry = tk.Entry(root, width=50)
        self.entry.pack()

        self.calculate_button = tk.Button(root, text="Calculate Sum", command=self.calculate)
        self.calculate_button.pack()

        self.result_label = tk.Label(root, text="Result: ")
        self.result_label.pack()

    def calculate(self):
        '''
        Handle the calculation of the sum when the button is pressed.
        '''
        try:
            # Get the dictionary from the entry
            user_input = self.entry.get()
            dictionary = eval(user_input)  # Convert string to dictionary

            if isinstance(dictionary, dict):
                # Calculate the sum
                result = calculate_sum(dictionary)
                self.result_label.config(text=f"Result: {result}")
            else:
                raise ValueError("Input is not a dictionary.")
        except Exception as e:
            messagebox.showerror("Error", f"Invalid input: {e}")

if __name__ == "__main__":
    root = tk.Tk()
    app = SumDictionaryApp(root)
    root.mainloop()
```

'''
This file contains the main application class for the SumDictionaryApp, which provides a GUI to calculate the sum of all items in a given dictionary.
'''
import tkinter as tk
from tkinter import messagebox
import ast
def calculate_sum(dictionary):
    '''
    Calculate the sum of all values in the given dictionary.
    :param dictionary: dict, the dictionary with numeric values
    :return: int/float, the sum of all values
    '''
    return sum(dictionary.values())
class SumDictionaryApp:
    '''
    A GUI application to calculate the sum of all items in a given dictionary.
    '''
    def __init__(self, root):
        '''
        Initialize the application with the main window and widgets.
        :param root: Tk, the main window of the application
        '''
        self.root = root
        self.root.title("Sum of Dictionary Values")
        # Create and place widgets
        self.label = tk.Label(root, text="Enter dictionary (e.g., {'a': 1, 'b': 2}):")
        self.label.pack()
        self.entry = tk.Entry(root, width=50)
        self.entry.pack()
        self.calculate_button = tk.Button(root, text="Calculate Sum", command=self.calculate)
        self.calculate_button.pack()
        self.result_label = tk.Label(root, text="Result: ")
        self.result_label.pack()
    def calculate(self):
        '''
        Handle the calculation of the sum when the button is pressed.
        '''
        try:
            # Get the dictionary from the entry
            user_input = self.entry.get()
            dictionary = ast.literal_eval(user_input)  # Safely convert string to dictionary
            if isinstance(dictionary, dict):
                # Calculate the sum
                result = calculate_sum(dictionary)
                self.result_label.config(text=f"Result: {result}")
            else:
                raise ValueError("Input is not a dictionary.")
        except Exception as e:
            messagebox.showerror("Error", f"Invalid input: {e}")
if __name__ == "__main__":
    root = tk.Tk()
    app = SumDictionaryApp(root)
    root.mainloop()
