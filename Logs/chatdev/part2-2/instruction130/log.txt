'''

This module contains the main application class and utility function to calculate the ratio of positive numbers in an array of integers.

'''

import tkinter as tk
from tkinter import messagebox

def calculate_positive_ratio(numbers):
    '''
    Calculate the ratio of positive numbers in the given list of integers.

    Parameters:
    numbers (list): A list of integers.

    Returns:
    float: The ratio of positive numbers in the list.
    '''
    if not numbers:
        return 0.0
    positive_count = sum(1 for num in numbers if num > 0)
    return positive_count / len(numbers)

class PositiveRatioApp:
    '''
    A class to create a GUI application for calculating the ratio of positive numbers in an array.
    '''
    def __init__(self, root):
        self.root = root
        self.root.title("Positive Number Ratio Calculator")

        self.label = tk.Label(root, text="Enter numbers separated by commas:")
        self.label.pack()

        self.entry = tk.Entry(root, width=50)
        self.entry.pack()

        self.calculate_button = tk.Button(root, text="Calculate Ratio", command=self.calculate_ratio)
        self.calculate_button.pack()

        self.result_label = tk.Label(root, text="Ratio of positive numbers: ")
        self.result_label.pack()

'''
This module contains the main application class and utility function to calculate the ratio of positive numbers in an array of integers.
'''
import tkinter as tk
from tkinter import messagebox

def calculate_positive_ratio(numbers):
    '''
    Calculate the ratio of positive numbers in the given list of integers.
    
    Parameters:
    numbers (list): A list of integers.
    
    Returns:
    float: The ratio of positive numbers in the list.
    '''
    if not numbers:
        return 0.0
    positive_count = sum(1 for num in numbers if num > 0)
    return positive_count / len(numbers)

class PositiveRatioApp:
    '''
    A class to create a GUI application for calculating the ratio of positive numbers in an array.
    '''
    def __init__(self, root):
        self.root = root
        self.root.title("Positive Number Ratio Calculator")
        
        self.label = tk.Label(root, text="Enter numbers separated by commas:")
        self.label.pack()
        
        self.entry = tk.Entry(root, width=50)
        self.entry.pack()
        
        self.calculate_button = tk.Button(root, text="Calculate Ratio", command=self.calculate_ratio)
        self.calculate_button.pack()
        
        self.result_label = tk.Label(root, text="Ratio of positive numbers: ")
        self.result_label.pack()

    def calculate_ratio(self):
        '''
        Calculate the ratio of positive numbers and update the result label.
        '''
        try:
            numbers = list(map(int, self.entry.get().split(',')))
            ratio = calculate_positive_ratio(numbers)
            self.result_label.config(text=f"Ratio of positive numbers: {ratio:.2f}")
        except ValueError:
            messagebox.showerror("Invalid input", "Please enter a valid list of integers separated by commas.")

if __name__ == "__main__":
    root = tk.Tk()
    app = PositiveRatioApp(root)
    root.mainloop()

'''
This module contains the main application class and utility function to calculate the ratio of positive numbers in an array of integers.
'''
import tkinter as tk
from tkinter import messagebox
def calculate_positive_ratio(numbers):
    '''
    Calculate the ratio of positive numbers in the given list of integers.
    Parameters:
    numbers (list): A list of integers.
    Returns:
    float: The ratio of positive numbers in the list.
    '''
    if not numbers:
        return 0.0
    positive_count = sum(1 for num in numbers if num > 0)
    return positive_count / len(numbers)
class PositiveRatioApp:
    '''
    A class to create a GUI application for calculating the ratio of positive numbers in an array.
    '''
    def __init__(self, root):
        self.root = root
        self.root.title("Positive Number Ratio Calculator")
        self.label = tk.Label(root, text="Enter numbers separated by commas:")
        self.label.pack()
        self.entry = tk.Entry(root, width=50)
        self.entry.pack()
        self.calculate_button = tk.Button(root, text="Calculate Ratio", command=self.calculate_ratio)
        self.calculate_button.pack()
        self.result_label = tk.Label(root, text="Ratio of positive numbers: ")
        self.result_label.pack()
    def calculate_ratio(self):
        '''
        Calculate the ratio of positive numbers and update the result label.
        '''
        try:
            numbers = list(map(int, self.entry.get().split(',')))
            ratio = calculate_positive_ratio(numbers)
            self.result_label.config(text=f"Ratio of positive numbers: {ratio:.2f}")
        except ValueError:
            messagebox.showerror("Invalid input", "Please enter a valid list of integers separated by commas.")
if __name__ == "__main__":
    root = tk.Tk()
    app = PositiveRatioApp(root)
    root.mainloop()
