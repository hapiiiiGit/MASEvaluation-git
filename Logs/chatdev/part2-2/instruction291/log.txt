'''

This file contains the function to check if the parentheses in a given expression are balanced.

def is_balanced(expression):
    '''
    Checks if the parentheses in the given expression are balanced.
    :param expression: The input expression as a string.
    :return: True if the expression is balanced, False otherwise.
    '''
    stack = []
    # Dictionary to match opening and closing parentheses
    matching_parentheses = {')': '(', '}': '{', ']': '['}
    
    for char in expression:
        if char in "({[":
            # Push opening parentheses onto the stack
            stack.append(char)
        elif char in ")}]":
            # Check if stack is empty or top of stack does not match
            if not stack or stack[-1] != matching_parentheses[char]:
                return False
            # Pop the matching opening parenthesis
            stack.pop()
    
    # If stack is empty, all parentheses are balanced
    return not stack

def is_balanced(expression):
    '''
    Checks if the parentheses in the given expression are balanced.
    :param expression: The input expression as a string.
    :return: True if the expression is balanced, False otherwise.
    '''
    stack = []
    # Dictionary to match opening and closing parentheses
    matching_parentheses = {')': '(', '}': '{', ']': '['}
    for char in expression:
        if char in "({[":
            # Push opening parentheses onto the stack
            stack.append(char)
        elif char in ")}]":
            # Check if stack is empty or top of stack does not match
            if not stack or stack[-1] != matching_parentheses[char]:
                return False
            # Pop the matching opening parenthesis
            stack.pop()
    # If stack is empty, all parentheses are balanced
    return not stack

def is_balanced(expression):
    '''
    Checks if the parentheses in the given expression are balanced.
    :param expression: The input expression as a string.
    :return: True if the expression is balanced, False otherwise.
    '''
    stack = []

    matching_parentheses = {')': '(', '}': '{', ']': '['}
    for char in expression:
        if char in "({[":
            # Push opening parentheses onto the stack
            stack.append(char)
        elif char in ")}]":
            # Check if stack is empty or top of stack does not match

                return False
            # Pop the matching opening parenthesis
            stack.pop()

'''
This file contains the function to check if the parentheses in a given expression are balanced.
'''
def is_balanced(expression):
    '''
    Checks if the parentheses in the given expression are balanced.
    :param expression: The input expression as a string.
    :return: True if the expression is balanced, False otherwise.
    '''
    stack = []
    # Dictionary to match opening and closing parentheses
    matching_parentheses = {')': '(', '}': '{', ']': '['}
    for char in expression:
        if char in "({[":
            # Push opening parentheses onto the stack
            stack.append(char)
        elif char in ")}]":
            # Check if stack is empty or top of stack does not match
            if not stack or stack[-1] != matching_parentheses[char]:
                return False
            # Pop the matching opening parenthesis
            stack.pop()
    # If stack is empty, all parentheses are balanced
    return not stack

def is_balanced(expression):
    '''
    Checks if the parentheses in the given expression are balanced.
    :param expression: The input expression as a string.
    :return: True if the expression is balanced, False otherwise.
    '''
    stack = []
    # Dictionary to match opening and closing parentheses
    matching_parentheses = {')': '(', '}': '{', ']': '['}
    for char in expression:
        if char in "({[":
            # Push opening parentheses onto the stack
            stack.append(char)
        elif char in ")}]":
            # Check if stack is empty or top of stack does not match
            if not stack or stack[-1] != matching_parentheses[char]:
                return False
            # Pop the matching opening parenthesis
            stack.pop()
    # If stack is empty, all parentheses are balanced
    return not stack

'''
This file contains the function to check if the parentheses in a given expression are balanced.
'''
def is_balanced(expression):
    '''
    Checks if the parentheses in the given expression are balanced.
    :param expression: The input expression as a string.
    :return: True if the expression is balanced, False otherwise.
    '''
    stack = []
    # Dictionary to match opening and closing parentheses
    matching_parentheses = {')': '(', '}': '{', ']': '['}
    for char in expression:
        if char in "({[":
            # Push opening parentheses onto the stack
            stack.append(char)
        elif char in ")}]":
            # Check if stack is empty or top of stack does not match
            if not stack or stack[-1] != matching_parentheses[char]:
                return False
            # Pop the matching opening parenthesis
            stack.pop()
    # If stack is empty, all parentheses are balanced
    return not stack
