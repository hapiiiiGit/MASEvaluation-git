'''

Main application file for counting occurrences of 'std' in a given string using a GUI.

'''

import tkinter as tk
from tkinter import messagebox
from count import count_std_occurrences

class StdCounterApp:
    '''
    Class to create and manage the GUI for counting occurrences of 'std'.
    '''

    def __init__(self, root):
        '''
        Initialize the GUI components.
        '''
        self.root = root
        self.root.title("STD Occurrence Counter")

        self.label = tk.Label(root, text="Enter a string:")
        self.label.pack()

        self.entry = tk.Entry(root, width=50)
        self.entry.pack()

        self.button = tk.Button(root, text="Count 'std'", command=self.count_occurrences)
        self.button.pack()

        self.result_label = tk.Label(root, text="")
        self.result_label.pack()

    def count_occurrences(self):
        '''
        Retrieve input from the user, call the counting function, and display the result.
        '''
        input_string = self.entry.get()
        count = count_std_occurrences(input_string)
        self.result_label.config(text=f"Occurrences of 'std': {count}")

if __name__ == "__main__":

'''
Main application file for counting occurrences of 'std' in a given string using a GUI.
'''
import tkinter as tk
from tkinter import messagebox
from count import count_std_occurrences

class StdCounterApp:
    '''
    Class to create and manage the GUI for counting occurrences of 'std'.
    '''
    def __init__(self, root):
        '''
        Initialize the GUI components.
        '''
        self.root = root
        self.root.title("STD Occurrence Counter")
        self.root.geometry("400x200")  # Set a fixed window size
        self.label = tk.Label(root, text="Enter a string:")
        self.label.pack(pady=10)
        self.entry = tk.Entry(root, width=50)
        self.entry.pack(pady=5)
        self.button = tk.Button(root, text="Count 'std'", command=self.count_occurrences)
        self.button.pack(pady=10)
        self.result_label = tk.Label(root, text="", fg="blue")
        self.result_label.pack(pady=10)

    def count_occurrences(self):
        '''
        Retrieve input from the user, call the counting function, and display the result.
        '''
        input_string = self.entry.get()
        if not input_string.strip():
            messagebox.showwarning("Input Error", "Please enter a valid string.")
            return
        count = count_std_occurrences(input_string)
        self.result_label.config(text=f"Occurrences of 'std': {count}")

if __name__ == "__main__":
    root = tk.Tk()
    app = StdCounterApp(root)
    root.mainloop()

'''
Main application file for counting occurrences of 'std' in a given string using a GUI.
'''
import tkinter as tk
from tkinter import messagebox
from count import count_std_occurrences
class StdCounterApp:
    '''
    Class to create and manage the GUI for counting occurrences of 'std'.
    '''
    def __init__(self, root):
        '''
        Initialize the GUI components.
        '''
        self.root = root
        self.root.title("STD Occurrence Counter")
        self.root.geometry("400x200")  # Set a fixed window size
        self.label = tk.Label(root, text="Enter a string:")
        self.label.pack(pady=10)
        self.entry = tk.Entry(root, width=50)
        self.entry.pack(pady=5)
        self.button = tk.Button(root, text="Count 'std'", command=self.count_occurrences)
        self.button.pack(pady=10)
        self.result_label = tk.Label(root, text="", fg="blue")
        self.result_label.pack(pady=10)
    def count_occurrences(self):
        '''
        Retrieve input from the user, call the counting function, and display the result.
        '''
        input_string = self.entry.get()
        if not input_string.strip():
            messagebox.showwarning("Input Error", "Please enter a valid string.")
            return
        count = count_std_occurrences(input_string)
        self.result_label.config(text=f"Occurrences of 'std': {count}")
if __name__ == "__main__":
    root = tk.Tk()
    app = StdCounterApp(root)
    root.mainloop()

import tkinter as tk
from tkinter import messagebox
from count import count_std_occurrences

class StdCounterApp:
    '''
    Class to create and manage the GUI for counting occurrences of 'std'.
    '''
    def __init__(self, root):
        '''
        Initialize the GUI components.
        '''
        self.root = root
        self.root.title("STD Occurrence Counter")
        self.root.geometry("400x200")  # Set a fixed window size

        self.label = tk.Label(root, text="Enter a string:")
        self.label.pack(pady=10)

        self.entry = tk.Entry(root, width=50)
        self.entry.pack(pady=5)

        self.button = tk.Button(root, text="Count 'std'", command=self.count_occurrences)

        self.result_label = tk.Label(root, text="", fg="blue")
        self.result_label.pack(pady=10)

    def count_occurrences(self):
        '''
        Retrieve input from the user, call the counting function, and display the result.

        input_string = self.entry.get()
        if not input_string.strip():

        self.result_label.config(text=f"Occurrences of 'std': {count}")

if __name__ == "__main__":
