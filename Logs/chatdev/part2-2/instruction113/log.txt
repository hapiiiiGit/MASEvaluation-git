'''

This file contains the main application code for a GUI-based Python program that counts the number of sublists containing a particular element.

'''

import tkinter as tk
from tkinter import messagebox

def count_sublists_containing_element(lst, element):
    '''
    Counts the number of sublists in the given list that contain the specified element.

    Parameters:
    lst (list): The list of sublists to search through.
    element (any): The element to search for within the sublists.

    Returns:
    int: The number of sublists containing the element.
    '''
    count = 0
    for sublist in lst:
        if element in sublist:
            count += 1
    return count

class SublistsCounterApp:
    '''
    A class to create a GUI application for counting sublists containing a specific element.
    '''

    def __init__(self, master):
        '''
        Initializes the SublistsCounterApp with the given master widget.

        Parameters:
        master (tk.Tk): The root window of the application.
        '''
        self.master = master
        master.title("Sublists Counter")

        self.label = tk.Label(master, text="Enter list of sublists (e.g., [[1, 2], [3, 4], [1, 5]]):")
        self.label.pack()

        self.entry = tk.Entry(master, width=50)
        self.entry.pack()

        self.element_label = tk.Label(master, text="Enter element to search for:")
        self.element_label.pack()

        self.element_entry = tk.Entry(master, width=20)
        self.element_entry.pack()

        self.count_button = tk.Button(master, text="Count Sublists", command=self.count_sublists)
        self.count_button.pack()

        self.result_label = tk.Label(master, text="")
        self.result_label.pack()

    def count_sublists(self):
        '''
        Counts the sublists containing the specified element and updates the result label.
        '''
        try:
            lst = eval(self.entry.get())
            element = eval(self.element_entry.get())
            result = count_sublists_containing_element(lst, element)
            self.display_result(result)
        except Exception as e:
            messagebox.showerror("Error", f"Invalid input: {e}")

    def display_result(self, result):
        '''
        Displays the result of the sublist count in the GUI.

        Parameters:
        result (int): The number of sublists containing the specified element.
        '''
        self.result_label.config(text=f"Number of sublists containing the element: {result}")

if __name__ == "__main__":
    root = tk.Tk()
    app = SublistsCounterApp(root)
    root.mainloop()
```

'''

This file contains the main application code for a GUI-based Python program that counts the number of sublists containing a particular element.

'''

import tkinter as tk
from tkinter import messagebox

def count_sublists_containing_element(lst, element):
    '''
    Counts the number of sublists in the given list that contain the specified element.

    Parameters:
    lst (list): The list of sublists to search through.
    element (any): The element to search for within the sublists.

    Returns:
    int: The number of sublists containing the element.
    '''
    count = 0
    for sublist in lst:
        if element in sublist:
            count += 1
    return count

class SublistsCounterApp:
    '''
    A class to create a GUI application for counting sublists containing a specific element.
    '''

    def __init__(self, master):
        '''
        Initializes the SublistsCounterApp with the given master widget.

        Parameters:
        master (tk.Tk): The root window of the application.
        '''
        self.master = master
        master.title("Sublists Counter")

        self.label = tk.Label(master, text="Enter list of sublists (e.g., [[1, 2], [3, 4], [1, 5]]):")
        self.label.pack()

        self.entry = tk.Entry(master, width=50)
        self.entry.pack()

        self.element_label = tk.Label(master, text="Enter element to search for:")
        self.element_label.pack()

        self.element_entry = tk.Entry(master, width=20)
        self.element_entry.pack()

        self.count_button = tk.Button(master, text="Count Sublists", command=self.count_sublists)
        self.count_button.pack()

        self.result_label = tk.Label(master, text="")
        self.result_label.pack()

    def count_sublists(self):
        '''
        Counts the sublists containing the specified element and updates the result label.
        '''
        try:
            lst = eval(self.entry.get())
            element = eval(self.element_entry.get())
            result = count_sublists_containing_element(lst, element)
            self.display_result(result)
        except Exception as e:
            messagebox.showerror("Error", f"Invalid input: {e}")

    def display_result(self, result):
        '''
        Displays the result of the sublist count in the GUI.

        Parameters:
        result (int): The number of sublists containing the specified element.
        '''
        self.result_label.config(text=f"Number of sublists containing the element: {result}")

if __name__ == "__main__":
    root = tk.Tk()
    app = SublistsCounterApp(root)
    root.mainloop()

def count_sublists_containing_element(lst, element):
    '''

    '''
