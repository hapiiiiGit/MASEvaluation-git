'''

This module contains the main application logic for calculating decagonal numbers and displaying the result in a GUI.

'''

import tkinter as tk
from tkinter import messagebox

class DecagonalNumberCalculator:
    '''
    A class to calculate the nth decagonal number.
    '''
    @staticmethod
    def calculate_decagonal_number(n):
        '''
        Calculate the nth decagonal number using the formula: D_n = 4n(n-1) + 1.
        
        :param n: The position of the decagonal number to calculate.
        :return: The nth decagonal number.
        '''
        return 4 * n * (n - 1) + 1

class DecagonalNumberApp:
    '''
    A class to create a GUI application for calculating decagonal numbers.
    '''
    def __init__(self, master):
        '''
        Initialize the GUI components.
        
        :param master: The root window of the application.
        '''
        self.master = master
        master.title("Decagonal Number Calculator")

        self.label = tk.Label(master, text="Enter n:")
        self.label.pack()

        self.entry = tk.Entry(master)
        self.entry.pack()

        self.calculate_button = tk.Button(master, text="Calculate", command=self.calculate_and_display)
        self.calculate_button.pack()

        self.result_label = tk.Label(master, text="")
        self.result_label.pack()

    def calculate_and_display(self):
        '''
        Retrieve user input, calculate the decagonal number, and display the result.
        '''
        try:
            n = int(self.entry.get())
            if n < 1:
                raise ValueError("n must be a positive integer.")
            result = DecagonalNumberCalculator.calculate_decagonal_number(n)
            self.result_label.config(text=f"The {n}th decagonal number is: {result}")
        except ValueError as e:
            messagebox.showerror("Invalid input", str(e))

if __name__ == "__main__":
    root = tk.Tk()
    app = DecagonalNumberApp(root)
    root.mainloop()

'''

This module contains the main application logic for calculating decagonal numbers and displaying the result in a GUI.

'''

import tkinter as tk
from tkinter import messagebox

class DecagonalNumberCalculator:
    '''
    A class to calculate the nth decagonal number.
    '''
    @staticmethod
    def calculate_decagonal_number(n):
        '''
        Calculate the nth decagonal number using the formula: D_n = 4n(n-1) + 1.
        
        :param n: The position of the decagonal number to calculate.
        :return: The nth decagonal number.
        '''
        return 4 * n * (n - 1) + 1

class DecagonalNumberApp:
    '''
    A class to create a GUI application for calculating decagonal numbers.
    '''
    def __init__(self, master):
        '''
        Initialize the GUI components.
        
        :param master: The root window of the application.
        '''
        self.master = master
        master.title("Decagonal Number Calculator")

        self.label = tk.Label(master, text="Enter n:")
        self.label.pack()

        self.entry = tk.Entry(master)
        self.entry.pack()

        self.calculate_button = tk.Button(master, text="Calculate", command=self.calculate_and_display)
        self.calculate_button.pack()

        self.result_label = tk.Label(master, text="")
        self.result_label.pack()

    def calculate_and_display(self):
        '''
        Retrieve user input, calculate the decagonal number, and display the result.
        '''
        try:
            n = int(self.entry.get())
            if n < 1:
                raise ValueError("n must be a positive integer.")
            result = DecagonalNumberCalculator.calculate_decagonal_number(n)
            self.result_label.config(text=f"The {n}th decagonal number is: {result}")
        except ValueError as e:
            messagebox.showerror("Invalid input", str(e))

if __name__ == "__main__":
    root = tk.Tk()
    app = DecagonalNumberApp(root)
    root.mainloop()

'''
This module contains the main application logic for calculating decagonal numbers and displaying the result in a GUI.
'''
import tkinter as tk
from tkinter import messagebox

class DecagonalNumberCalculator:
    '''
    A class to calculate the nth decagonal number.
    '''
    @staticmethod
    def calculate_decagonal_number(n):
        '''
        Calculate the nth decagonal number using the formula: D_n = 4n(n-1) + 1.
        :param n: The position of the decagonal number to calculate.
        :return: The nth decagonal number.
        '''
        return 4 * n * (n - 1) + 1

class DecagonalNumberApp:
    '''
    A class to create a GUI application for calculating decagonal numbers.
    '''
    def __init__(self, master):
        '''
        Initialize the GUI components.
        :param master: The root window of the application.
        '''
        self.master = master
        master.title("Decagonal Number Calculator")
        
        self.label = tk.Label(master, text="Enter n:")
        self.label.pack()
        
        self.entry = tk.Entry(master)
        self.entry.pack()
        
        self.calculate_button = tk.Button(master, text="Calculate", command=self.calculate_and_display)
        self.calculate_button.pack()
        
        self.result_label = tk.Label(master, text="")
        self.result_label.pack()

    def calculate_and_display(self):
        '''
        Retrieve user input, calculate the decagonal number, and display the result.
        '''
        try:
            n = int(self.entry.get())
            if n < 1:
                raise ValueError("n must be a positive integer.")
            result = DecagonalNumberCalculator.calculate_decagonal_number(n)
            self.result_label.config(text=f"The {n}th decagonal number is: {result}")
        except ValueError as e:
            messagebox.showerror("Invalid input", str(e))

if __name__ == "__main__":
    root = tk.Tk()
    app = DecagonalNumberApp(root)
    root.mainloop()

'''

import tkinter as tk
from tkinter import messagebox
class DecagonalNumberCalculator:
    '''
    A class to calculate the nth decagonal number.
    '''
    @staticmethod
    def calculate_decagonal_number(n):
        '''
        Calculate the nth decagonal number using the formula: D_n = 4n(n-1) + 1.
        :param n: The position of the decagonal number to calculate.
        :return: The nth decagonal number.
        '''
        return 4 * n * (n - 1) + 1
class DecagonalNumberApp:
    '''
    A class to create a GUI application for calculating decagonal numbers.
    '''
    def __init__(self, master):
        '''
        Initialize the GUI components.
        :param master: The root window of the application.
        '''
        self.master = master
        master.title("Decagonal Number Calculator")
        self.label = tk.Label(master, text="Enter n:")
        self.label.pack()
        self.entry = tk.Entry(master)
        self.entry.pack()
        self.calculate_button = tk.Button(master, text="Calculate", command=self.calculate_and_display)
        self.calculate_button.pack()
        self.result_label = tk.Label(master, text="")
        self.result_label.pack()
    def calculate_and_display(self):
        '''
        Retrieve user input, calculate the decagonal number, and display the result.
        '''
        try:
            n = int(self.entry.get())
            if n < 1:
                raise ValueError("n must be a positive integer.")
            result = DecagonalNumberCalculator.calculate_decagonal_number(n)
            self.result_label.config(text=f"The {n}th decagonal number is: {result}")
        except ValueError as e:
            messagebox.showerror("Invalid input", str(e))
if __name__ == "__main__":
    root = tk.Tk()
    app = DecagonalNumberApp(root)
