#### utils.py

```python
'''

Utility functions for string processing.

'''

def find_max_run(s):
    '''
    Find the maximum run of uppercase characters in the given string.

    Parameters:
    s (str): The input string.

    Returns:
    int: The length of the maximum run of uppercase characters.
    '''
    max_run = 0
    current_run = 0

    for char in s:
        if char.isupper():
            current_run += 1
            max_run = max(max_run, current_run)
        else:
            current_run = 0

    return max_run
```

'''

Utility functions for string processing.

def find_max_run(s):
    '''
    Find the maximum run of uppercase characters in the given string.

    Parameters:
    s (str): The input string.

    Returns:
    int: The length of the maximum run of uppercase characters.
    '''
    max_run = 0
    current_run = 0

    for char in s:
        if char.isupper():
            current_run += 1
            max_run = max(max_run, current_run)
        else:
            current_run = 0

    return max_run

'''
Utility functions for string processing.
'''
def find_max_run(s):
    '''
    Find the maximum run of uppercase characters in the given string.
    Parameters:
    s (str): The input string.
    Returns:
    int: The length of the maximum run of uppercase characters.
    '''
    max_run = 0
    current_run = 0
    for char in s:
        if char.isupper():
            current_run += 1
            max_run = max(max_run, current_run)
        else:
            current_run = 0
    return max_run
