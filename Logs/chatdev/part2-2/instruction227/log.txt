```python
'''

Main file to run the ListSwapperApp application using tkinter.

'''

import tkinter as tk
from tkinter import messagebox

def swap_first_last(lst):
    '''
    Swap the first and last elements of the list.
    
    Parameters:
    lst (list): The list whose elements are to be swapped.
    
    Returns:
    list: The list with the first and last elements swapped.
    '''
    if len(lst) < 2:
        return lst
    lst[0], lst[-1] = lst[-1], lst[0]
    return lst

class ListSwapperApp:
    '''
    A class to create a GUI application for swapping the first and last elements of a list.
    '''
    
    def __init__(self, master):
        '''
        Initialize the GUI components.
        
        Parameters:
        master (tk.Tk): The root window.
        '''
        self.master = master
        master.title("List Swapper")
        
        self.label = tk.Label(master, text="Enter a list of numbers separated by commas:")
        self.label.pack()
        
        self.entry = tk.Entry(master, width=50)
        self.entry.pack()
        
        self.swap_button = tk.Button(master, text="Swap", command=self.swap_elements)
        self.swap_button.pack()
        
        self.result_label = tk.Label(master, text="")
        self.result_label.pack()
    
    def swap_elements(self):
        '''
        Logic to interchange the first and last elements of the list.
        '''
        input_text = self.entry.get()
        try:
            # Convert input text to a list of integers
            input_list = list(map(int, input_text.split(',')))
            # Swap the first and last elements
            swapped_list = swap_first_last(input_list)
            # Display the result
            self.display_result(swapped_list)
        except ValueError:
            messagebox.showerror("Invalid input", "Please enter a valid list of numbers separated by commas.")
    
    def display_result(self, result):
        '''
        Display the result in the GUI.
        
        Parameters:
        result (list): The list with swapped elements.
        '''
        self.result_label.config(text=f"Swapped List: {result}")

if __name__ == "__main__":
    root = tk.Tk()
    app = ListSwapperApp(root)
    root.mainloop()
```

Main file to run the ListSwapperApp application using tkinter.

import tkinter as tk
from tkinter import messagebox

def swap_first_last(lst):
    '''
    Swap the first and last elements of the list.
    
    Parameters:
    lst (list): The list whose elements are to be swapped.
    
    Returns:
    list: The list with the first and last elements swapped.
    '''
    if len(lst) < 2:
        return lst
    lst[0], lst[-1] = lst[-1], lst[0]
    return lst

class ListSwapperApp:
    '''
    A class to create a GUI application for swapping the first and last elements of a list.
    '''
    
    def __init__(self, master):
        '''
        Initialize the GUI components.
        
        Parameters:
        master (tk.Tk): The root window.
        '''
        self.master = master
        master.title("List Swapper")
        
        self.label = tk.Label(master, text="Enter a list of numbers separated by commas:")
        self.label.pack()
        
        self.entry = tk.Entry(master, width=50)
        self.entry.pack()
        
        self.swap_button = tk.Button(master, text="Swap", command=self.swap_elements)
        self.swap_button.pack()
        
        self.result_label = tk.Label(master, text="")
        self.result_label.pack()
    
    def swap_elements(self):
        '''
        Logic to interchange the first and last elements of the list.
        '''
        input_text = self.entry.get()
        try:
            # Convert input text to a list of integers
            input_list = list(map(int, input_text.split(',')))
            # Swap the first and last elements
            swapped_list = swap_first_last(input_list)
            # Display the result
            self.display_result(swapped_list)
        except ValueError:
            messagebox.showerror("Invalid input", "Please enter a valid list of numbers separated by commas.")
    
    def display_result(self, result):
        '''
        Display the result in the GUI.
        
        Parameters:
        result (list): The list with swapped elements.
        '''
        self.result_label.config(text=f"Swapped List: {result}")

if __name__ == "__main__":
    root = tk.Tk()
    app = ListSwapperApp(root)
    root.mainloop()

'''

import tkinter as tk
from tkinter import messagebox
def swap_first_last(lst):
    '''
    Swap the first and last elements of the list.
    Parameters:
    lst (list): The list whose elements are to be swapped.
    Returns:
    list: The list with the first and last elements swapped.
    '''
    if len(lst) < 2:
        return lst
    lst[0], lst[-1] = lst[-1], lst[0]
    return lst
class ListSwapperApp:
    '''
    A class to create a GUI application for swapping the first and last elements of a list.
    '''
    def __init__(self, master):
        '''
        Initialize the GUI components.
        Parameters:
        master (tk.Tk): The root window.
        '''
        self.master = master
        master.title("List Swapper")
        self.label = tk.Label(master, text="Enter a list of numbers separated by commas:")
        self.label.pack()
        self.entry = tk.Entry(master, width=50)
        self.entry.pack()
        self.swap_button = tk.Button(master, text="Swap", command=self.swap_elements)
        self.swap_button.pack()
        self.result_label = tk.Label(master, text="")
        self.result_label.pack()
    def swap_elements(self):
        '''
        Logic to interchange the first and last elements of the list.
        '''
        input_text = self.entry.get()
        try:
            # Convert input text to a list of integers
            input_list = list(map(int, input_text.split(',')))
            # Swap the first and last elements
            swapped_list = swap_first_last(input_list)
            # Display the result
            self.display_result(swapped_list)
        except ValueError:
            messagebox.showerror("Invalid input", "Please enter a valid list of numbers separated by commas.")
    def display_result(self, result):
        '''
        Display the result in the GUI.
        Parameters:
        result (list): The list with swapped elements.
        '''
        self.result_label.config(text=f"Swapped List: {result}")
if __name__ == "__main__":
    root = tk.Tk()
    app = ListSwapperApp(root)
    root.mainloop()

'''
Main file to run the ListSwapperApp application using tkinter.
'''
import tkinter as tk
from tkinter import messagebox
def swap_first_last(lst):
    '''
    Swap the first and last elements of the list.
    Parameters:
    lst (list): The list whose elements are to be swapped.
    Returns:
    list: The list with the first and last elements swapped.
    '''
    if len(lst) < 2:
        return lst
    lst[0], lst[-1] = lst[-1], lst[0]
    return lst
class ListSwapperApp:
    '''
    A class to create a GUI application for swapping the first and last elements of a list.
    '''
    def __init__(self, master):
        '''
        Initialize the GUI components.
        Parameters:
        master (tk.Tk): The root window.
        '''
        self.master = master
        master.title("List Swapper")
        self.label = tk.Label(master, text="Enter a list of numbers separated by commas:")
        self.label.pack()
        self.entry = tk.Entry(master, width=50)
        self.entry.pack()
        self.swap_button = tk.Button(master, text="Swap", command=self.swap_elements)
        self.swap_button.pack()
        self.result_label = tk.Label(master, text="")
        self.result_label.pack()
    def swap_elements(self):
        '''
        Logic to interchange the first and last elements of the list.
        '''
        input_text = self.entry.get()
        try:
            # Convert input text to a list of integers
            input_list = list(map(int, input_text.split(',')))
            # Swap the first and last elements
            swapped_list = swap_first_last(input_list)
            # Display the result
            self.display_result(swapped_list)
        except ValueError:
            messagebox.showerror("Invalid input", "Please enter a valid list of numbers separated by commas.")
    def display_result(self, result):
        '''
        Display the result in the GUI.
        Parameters:
        result (list): The list with swapped elements.
        '''
        self.result_label.config(text=f"Swapped List: {result}")
if __name__ == "__main__":
    root = tk.Tk()
    app = ListSwapperApp(root)
    root.mainloop()

'''
Main file to run the ListSwapperApp application using tkinter.
'''
import tkinter as tk
from tkinter import messagebox
def swap_first_last(lst):
    '''
    Swap the first and last elements of the list.
    Parameters:
    lst (list): The list whose elements are to be swapped.
    Returns:
    list: The list with the first and last elements swapped.
    '''
    if len(lst) < 2:
        return lst
    lst[0], lst[-1] = lst[-1], lst[0]
    return lst
class ListSwapperApp:
    '''
    A class to create a GUI application for swapping the first and last elements of a list.
    '''
    def __init__(self, master):
        '''
        Initialize the GUI components.
        Parameters:
        master (tk.Tk): The root window.
        '''
        self.master = master
        master.title("List Swapper")
        self.label = tk.Label(master, text="Enter a list of numbers separated by commas:")
        self.label.pack()
        self.entry = tk.Entry(master, width=50)
        self.entry.pack()
        self.swap_button = tk.Button(master, text="Swap", command=self.swap_elements)
        self.swap_button.pack()
        self.result_label = tk.Label(master, text="")
        self.result_label.pack()
    def swap_elements(self):
        '''
        Logic to interchange the first and last elements of the list.
        '''
        input_text = self.entry.get()
        try:
            # Convert input text to a list of integers
            input_list = list(map(int, input_text.split(',')))
            # Swap the first and last elements
            swapped_list = swap_first_last(input_list)
            # Display the result
            self.display_result(swapped_list)
        except ValueError:
            messagebox.showerror("Invalid input", "Please enter a valid list of numbers separated by commas.")
    def display_result(self, result):
        '''
        Display the result in the GUI.
        Parameters:
        result (list): The list with swapped elements.
        '''
        self.result_label.config(text=f"Swapped List: {result}")
if __name__ == "__main__":
    root = tk.Tk()
    app = ListSwapperApp(root)
    root.mainloop()
