'''

This file contains the utility function to check if a string is a decimal number with a precision of 2.

'''

def is_decimal_with_precision(s: str) -> bool:
    '''
    Check if the given string is a decimal number with a precision of 2.
    
    Parameters:
    s (str): The string to check.
    
    Returns:
    bool: True if the string is a decimal number with two decimal places, otherwise False.
    '''
    try:
        parts = s.split('.')
        if len(parts) == 2 and len(parts[1]) == 2:
            float(s)  # Check if it's a valid float
            return True
        return False
    except ValueError:
        return False
```

def is_decimal_with_precision(s: str) -> bool:
    '''
    Check if the given string is a decimal number with a precision of 2.
    
    Parameters:
    s (str): The string to check.

def is_decimal_with_precision(s: str) -> bool:
    '''
    Check if the given string is a decimal number with a precision of 2.
    Parameters:
    s (str): The string to check.
    Returns:
    bool: True if the string is a decimal number with two decimal places, otherwise False.
    '''
    try:
        s = s.strip()  # Remove leading and trailing spaces
        if s.startswith('-'):
            s = s[1:]  # Remove the negative sign for checking
        parts = s.split('.')
        if len(parts) == 2 and len(parts[1]) == 2 and (parts[0].isdigit() or parts[0] == ''):
            float(s)  # Check if it's a valid float
            return True
        return False
    except ValueError:
        return False

'''
This file contains the utility function to check if a string is a decimal number with a precision of 2.
'''
def is_decimal_with_precision(s: str) -> bool:
    '''
    Check if the given string is a decimal number with a precision of 2.
    Parameters:
    s (str): The string to check.
    Returns:
    bool: True if the string is a decimal number with two decimal places, otherwise False.
    '''
    try:
        s = s.strip()  # Remove leading and trailing spaces
        if s.startswith('-'):
            s = s[1:]  # Remove the negative sign for checking
        parts = s.split('.')
        if len(parts) == 2 and len(parts[1]) == 2 and (parts[0].isdigit() or parts[0] == ''):
            float(s)  # Check if it's a valid float
            return True
        return False
    except ValueError:
        return False

def is_decimal_with_precision(s: str) -> bool:
    '''
    Check if the given string is a decimal number with a precision of 2.
    Parameters:
    s (str): The string to check.
    Returns:
    bool: True if the string is a decimal number with two decimal places, otherwise False.
