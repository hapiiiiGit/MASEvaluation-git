```python
'''

This file contains the implementation of a simple GUI application using tkinter to reverse words in a given string.

'''

import tkinter as tk

def reverse_words_in_string(input_string):
    '''
    This function takes a string as input and returns a new string with the words reversed.
    
    :param input_string: str - The input string containing words separated by spaces.
    :return: str - The string with words reversed.
    '''
    words = input_string.split()
    reversed_words = ' '.join(reversed(words))
    return reversed_words

class MainApp:
    '''
    This class represents the main application window for the GUI.
    '''
    def __init__(self, root):
        '''
        Initializes the GUI components.
        
        :param root: tk.Tk - The root window of the tkinter application.
        '''
        self.root = root
        self.root.title("Reverse Words Application")
        
        # Create and place the input label and entry
        self.input_label = tk.Label(root, text="Enter a string:")
        self.input_label.pack()
        
        self.input_entry = tk.Entry(root, width=50)
        self.input_entry.pack()
        
        # Create and place the reverse button
        self.reverse_button = tk.Button(root, text="Reverse Words", command=self.reverse_words)
        self.reverse_button.pack()
        
        # Create and place the output label
        self.output_label = tk.Label(root, text="Reversed string will appear here.")
        self.output_label.pack()
    
    def reverse_words(self):
        '''
        This method is called when the reverse button is clicked. It reverses the words in the input string.
        '''
        input_string = self.input_entry.get()
        reversed_string = reverse_words_in_string(input_string)
        self.output_label.config(text=reversed_string)

if __name__ == "__main__":
    root = tk.Tk()
    app = MainApp(root)
    root.mainloop()
```

- **MainApp**: This class sets up the GUI with an entry field for input, a button to trigger the reversal, and a label to display the result.
- **__init__**: Initializes the GUI components.
- **reverse_words**: Retrieves the input string, processes it using `reverse_words_in_string`, and updates the output label with the reversed string.

This application provides a simple interface for users to input a string, click a button, and see the words reversed in the output label.

This file contains the implementation of a simple GUI application using tkinter to reverse words in a given string.

import tkinter as tk

def reverse_words_in_string(input_string):
    '''
    This function takes a string as input and returns a new string with the words reversed.
    
    :param input_string: str - The input string containing words separated by spaces.
    :return: str - The string with words reversed.

'''
This file contains the implementation of a simple GUI application using tkinter to reverse words in a given string.

import tkinter as tk
def reverse_words_in_string(input_string):
    '''
    This function takes a string as input and returns a new string with the words reversed.
    :param input_string: str - The input string containing words separated by spaces.
    :return: str - The string with words reversed.

    words = input_string.split()

    '''
    This class represents the main application window for the GUI.
    '''

        '''
        Initializes the GUI components.
        :param root: tk.Tk - The root window of the tkinter application.
        '''
        self.root = root
        self.root.title("Reverse Words Application")
        # Create and place the input label and entry
        self.input_label = tk.Label(root, text="Enter a string:")
        self.input_label.pack()
        self.input_entry = tk.Entry(root, width=50)
        self.input_entry.pack()
        # Create and place the reverse button
        self.reverse_button = tk.Button(root, text="Reverse Words", command=self.reverse_words)
