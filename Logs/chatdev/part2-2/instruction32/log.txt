'''

import tkinter as tk
from tkinter import messagebox


def swap_numbers(a, b):
    '''
    Takes two numbers and returns a tuple with the second number first and the first number second.

    Args:
        a (float): The first number.
        b (float): The second number.

    Returns:
        tuple: A tuple containing the second number first and the first number second.
    '''
    return b, a


class NumberSwapperApp:
    '''
    A GUI application for swapping two numbers and displaying the result.
    '''

    def __init__(self, root):
        '''
        Initializes the GUI components.

        Args:
            root (tk.Tk): The root window of the tkinter application.
        '''
        self.root = root
        self.root.title("Number Swapper")

        # Input fields
        self.label1 = tk.Label(root, text="Enter the first number:")

'''

This file contains the main application logic for a Python program that swaps two numbers and displays the result in a graphical user interface (GUI) using tkinter.

'''

import tkinter as tk
from tkinter import messagebox


def swap_numbers(a, b):
    '''
    Takes two numbers and returns a tuple with the second number first and the first number second.

    Args:
        a (float): The first number.
        b (float): The second number.

    Returns:
        tuple: A tuple containing the second number first and the first number second.
    '''
    return b, a


class NumberSwapperApp:
    '''
    A GUI application for swapping two numbers and displaying the result.
    '''

    def __init__(self, root):
        '''
        Initializes the GUI components.

        Args:
            root (tk.Tk): The root window of the tkinter application.
        '''
        self.root = root
        self.root.title("Number Swapper")

        # Input fields
        self.label1 = tk.Label(root, text="Enter the first number:")
        self.label1.grid(row=0, column=0, padx=10, pady=10)
        self.entry1 = tk.Entry(root)
        self.entry1.grid(row=0, column=1, padx=10, pady=10)

        self.label2 = tk.Label(root, text="Enter the second number:")
        self.label2.grid(row=1, column=0, padx=10, pady=10)
        self.entry2 = tk.Entry(root)
        self.entry2.grid(row=1, column=1, padx=10, pady=10)

        # Swap button
        self.swap_button = tk.Button(root, text="Swap Numbers", command=self.swap_numbers)
        self.swap_button.grid(row=2, column=0, columnspan=2, pady=10)

        # Result label
        self.result_label = tk.Label(root, text="Result: ")
        self.result_label.grid(row=3, column=0, columnspan=2, pady=10)

    def swap_numbers(self):
        '''
        Takes two numbers from the input fields, swaps them, and displays the result.
        '''
        try:
            # Get input values
            num1 = float(self.entry1.get())
            num2 = float(self.entry2.get())

            # Swap numbers
            swapped = swap_numbers(num1, num2)

            # Display result
            self.result_label.config(text=f"Result: {swapped}")
        except ValueError:
            # Handle invalid input
            messagebox.showerror("Invalid Input", "Please enter valid numbers.")


if __name__ == "__main__":
    # Create the main window
    root = tk.Tk()

    # Initialize the application
    app = NumberSwapperApp(root)

    # Run the application
    root.mainloop()

def swap_numbers(a, b):
    '''
    Takes two numbers and returns a tuple with the second number first and the first number second.
    Args:
        a (float): The first number.
        b (float): The second number.
    Returns:
        tuple: A tuple containing the second number first and the first number second.

        self.entry2.grid(row=1, column=1, padx=10, pady=10)

import tkinter as tk
from tkinter import messagebox

def swap_numbers(a, b):
    '''

    Args:
        a (float): The first number.

            num2 = float(self.entry2.get())

'''
This file contains the main application logic for a Python program that swaps two numbers and displays the result in a graphical user interface (GUI) using tkinter.
'''
import tkinter as tk
from tkinter import messagebox
def swap_numbers(a, b):
    '''
    Takes two numbers and returns a tuple with the second number first and the first number second.
    Args:
        a (float): The first number.
        b (float): The second number.
    Returns:
        tuple: A tuple containing the second number first and the first number second.
    '''
    return b, a
class NumberSwapperApp:
    '''
    A GUI application for swapping two numbers and displaying the result.
    '''
    def __init__(self, root):
        '''
        Initializes the GUI components.
        Args:
            root (tk.Tk): The root window of the tkinter application.
        '''
        self.root = root
        self.root.title("Number Swapper")
        # Input fields
        self.label1 = tk.Label(root, text="Enter the first number:")
        self.label1.grid(row=0, column=0, padx=10, pady=10)
        self.entry1 = tk.Entry(root)
        self.entry1.grid(row=0, column=1, padx=10, pady=10)
        self.label2 = tk.Label(root, text="Enter the second number:")
        self.label2.grid(row=1, column=0, padx=10, pady=10)
        self.entry2 = tk.Entry(root)
        self.entry2.grid(row=1, column=1, padx=10, pady=10)
        # Swap button
        self.swap_button = tk.Button(root, text="Swap Numbers", command=self.swap_numbers)
        self.swap_button.grid(row=2, column=0, columnspan=2, pady=10)
        # Result label
        self.result_label = tk.Label(root, text="Result: ")
        self.result_label.grid(row=3, column=0, columnspan=2, pady=10)
    def swap_numbers(self):
        '''
        Takes two numbers from the input fields, swaps them, and displays the result.
        '''
        try:
            # Get input values
            num1 = float(self.entry1.get())
            num2 = float(self.entry2.get())
            # Swap numbers
            swapped = swap_numbers(num1, num2)
            # Display result
            self.result_label.config(text=f"Result: {swapped}")
        except ValueError:
            # Handle invalid input
            messagebox.showerror("Invalid Input", "Please enter valid numbers.")
if __name__ == "__main__":
    '''
    Entry point for the application. Creates the main window and initializes the GUI.
    '''
    # Create the main window
    root = tk.Tk()
    # Initialize the application
    app = NumberSwapperApp(root)
    # Run the application
    root.mainloop()
