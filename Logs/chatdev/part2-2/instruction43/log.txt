#### `utils.py`

'''

utils.py

This file contains utility functions for the application. Specifically, it includes the function to calculate the sum of even numbers at even positions in a list.

'''


def find_sum_even_positions(numbers):
    '''
    Calculate the sum of even numbers at even positions in a list.

    Args:
        numbers (list): A list of integers.

    Returns:
        int: The sum of even numbers at even positions.
    '''
    # Even positions in Python are 0-based indices: 0, 2, 4, ...
    return sum(num for idx, num in enumerate(numbers) if idx % 2 == 0 and num % 2 == 0)

'''

main.py

This file contains the main application class for the GUI-based Python application. It allows users to input a list of numbers, calculates the sum of even numbers at even positions, and displays the result.

'''

import tkinter as tk
from tkinter import messagebox
from utils import find_sum_even_positions


class SumEvenNumbersApp:
    def __init__(self, root):
        '''
        Initialize the GUI components for the application.
        '''
        self.root = root
        self.root.title("Sum of Even Numbers at Even Positions")

        # Input Label
        self.label = tk.Label(root, text="Enter a list of numbers (comma-separated):")

        self.entry = tk.Entry(root, width=50)
        self.entry.pack(pady=10)

        # Result Label
        self.result_label = tk.Label(root, text="", fg="blue")

        Process the input list, calculate the sum of even numbers at even positions, and display the result.

def find_sum_even_positions(numbers):
    '''
    Calculate the sum of even numbers at even positions in a list.

    Args:
        numbers (list): A list of integers.

    Returns:
        int: The sum of even numbers at even positions.

    # Even positions in Python are 0-based indices: 0, 2, 4, ...

import tkinter as tk
from tkinter import messagebox
from utils import find_sum_even_positions
class SumEvenNumbersApp:
    def __init__(self, root):
        '''
        Initialize the GUI components for the application.
        '''
        self.root = root
        self.root.title("Sum of Even Numbers at Even Positions")
        # Input Label
        self.label = tk.Label(root, text="Enter a list of numbers (comma-separated):")
        self.label.pack(pady=10)
        # Input Entry
        self.entry = tk.Entry(root, width=50)
        self.entry.pack(pady=10)
        # Calculate Button
        self.calculate_button = tk.Button(root, text="Calculate Sum", command=self.calculate_sum)
        self.calculate_button.pack(pady=10)
        # Result Label
        self.result_label = tk.Label(root, text="", fg="blue")
        self.result_label.pack(pady=10)
    def calculate_sum(self):
        '''
        Process the input list, calculate the sum of even numbers at even positions, and display the result.
        '''
        input_text = self.entry.get()
        try:
            # Convert input text to a list of integers
            numbers = list(map(int, input_text.split(',')))
            # Calculate the sum using the helper function
            result = find_sum_even_positions(numbers)
            # Display the result
            self.result_label.config(text=f"Sum of even numbers at even positions: {result}")
        except ValueError:
            # Handle invalid input
            messagebox.showerror("Invalid Input", "Please enter a valid list of numbers (comma-separated).")
if __name__ == "__main__":
    root = tk.Tk()
    app = SumEvenNumbersApp(root)
    root.mainloop()

def find_sum_even_positions(numbers):
    '''

'''
main.py
This file contains the main application class for the GUI-based Python application. It allows users to input a list of numbers, calculates the sum of even numbers at even positions, and displays the result.
'''
import tkinter as tk
from tkinter import messagebox
from utils import find_sum_even_positions

class SumEvenNumbersApp:
    def __init__(self, root):
        '''
        Initialize the GUI components for the application.
        '''
        self.root = root
        self.root.title("Sum of Even Numbers at Even Positions")
        # Input Label
        self.label = tk.Label(root, text="Enter a list of numbers (comma-separated):")
        self.label.pack(pady=10)
        # Input Entry
        self.entry = tk.Entry(root, width=50)
        self.entry.pack(pady=10)
        # Calculate Button
        self.calculate_button = tk.Button(root, text="Calculate Sum", command=self.calculate_sum)
        self.calculate_button.pack(pady=10)
        # Result Label
        self.result_label = tk.Label(root, text="", fg="blue")
        self.result_label.pack(pady=10)

    def calculate_sum(self):
        '''
        Process the input list, calculate the sum of even numbers at even positions, and display the result.
        '''
        input_text = self.entry.get()
        try:
            # Convert input text to a list of integers
            numbers = list(map(int, input_text.split(',')))
            # Calculate the sum using the helper function
            result = find_sum_even_positions(numbers)
            # Display the result
            self.result_label.config(text=f"Sum of even numbers at even positions: {result}")
        except ValueError:
            # Handle invalid input
            messagebox.showerror("Invalid Input", "Please enter a valid list of numbers (comma-separated).")

if __name__ == "__main__":
    root = tk.Tk()
    app = SumEvenNumbersApp(root)
    root.mainloop()

def find_sum_even_positions(numbers):
    '''
    Calculate the sum of even numbers at even positions in a list.
    Args:
        numbers (list): A list of integers.
    Returns:
        int: The sum of even numbers at even positions.
    '''
    # Even positions in Python are 0-based indices: 0, 2, 4, ...
    return sum(num for idx, num in enumerate(numbers) if idx % 2 == 0 and num % 2 == 0)

import tkinter as tk
from tkinter import messagebox
from utils import find_sum_even_positions
class SumEvenNumbersApp:
    def __init__(self, root):
        '''
        Initialize the GUI components for the application.
        '''
        self.root = root
        self.root.title("Sum of Even Numbers at Even Positions")
        # Input Label
        self.label = tk.Label(root, text="Enter a list of numbers (comma-separated):")
        self.label.pack(pady=10)
        # Input Entry
        self.entry = tk.Entry(root, width=50)
        self.entry.pack(pady=10)
        # Calculate Button
        self.calculate_button = tk.Button(root, text="Calculate Sum", command=self.calculate_sum)
        self.calculate_button.pack(pady=10)
        # Result Label
        self.result_label = tk.Label(root, text="", fg="blue")
        self.result_label.pack(pady=10)
    def calculate_sum(self):
        '''
        Process the input list, calculate the sum of even numbers at even positions, and display the result.
        '''
        input_text = self.entry.get()
        try:
            # Convert input text to a list of integers
            numbers = list(map(int, input_text.split(',')))
            # Calculate the sum using the helper function
            result = find_sum_even_positions(numbers)
            # Display the result
            self.result_label.config(text=f"Sum of even numbers at even positions: {result}")
        except ValueError:
            # Handle invalid input
            messagebox.showerror("Invalid Input", "Please enter a valid list of numbers (comma-separated).")
if __name__ == "__main__":
    root = tk.Tk()
    app = SumEvenNumbersApp(root)
    root.mainloop()

def find_sum_even_positions(numbers):
    '''
    Calculate the sum of even numbers at even positions in a list.
    Args:
        numbers (list): A list of integers.
    Returns:
        int: The sum of even numbers at even positions.
    '''
    # Even positions in Python are 0-based indices: 0, 2, 4, ...
    return sum(num for idx, num in enumerate(numbers) if idx % 2 == 0 and num % 2 == 0)
