def find_common_elements_in_lists(l1, l2, l3):

'''
Main application file for finding common elements in three lists using a GUI.
'''
import tkinter as tk
from tkinter import messagebox
import ast
def find_common_elements_in_lists(l1, l2, l3):
    """
    Finds elements that are common in the same index across three lists.
    Parameters:
    l1 (list): First list.
    l2 (list): Second list.
    l3 (list): Third list.
    Returns:
    list: A list of common elements at the same index.
    """
    common_elements = []
    min_length = min(len(l1), len(l2), len(l3))
    for i in range(min_length):
        if l1[i] == l2[i] == l3[i]:
            common_elements.append(l1[i])
    return common_elements
class MainApp:
    def __init__(self, root):
        """
        Initializes the main application window and its components.
        Parameters:
        root (tk.Tk): The root window object.
        """
        self.root = root
        self.root.title("Common Elements Finder")
        # Labels and Entries for list inputs
        self.label1 = tk.Label(root, text="List 1:")
        self.label1.pack()
        self.entry1 = tk.Entry(root, width=50)
        self.entry1.pack()
        self.label2 = tk.Label(root, text="List 2:")
        self.label2.pack()
        self.entry2 = tk.Entry(root, width=50)
        self.entry2.pack()
        self.label3 = tk.Label(root, text="List 3:")
        self.label3.pack()
        self.entry3 = tk.Entry(root, width=50)
        self.entry3.pack()
        # Button to find common elements
        self.find_button = tk.Button(root, text="Find Common Elements", command=self.find_common_elements)
        self.find_button.pack()
    def find_common_elements(self):
        """
        Retrieves input lists, processes them, and displays the result.
        """
        try:
            # Convert input strings to lists using ast.literal_eval for safety
            list1 = ast.literal_eval(self.entry1.get())
            list2 = ast.literal_eval(self.entry2.get())
            list3 = ast.literal_eval(self.entry3.get())
            # Find common elements
            common_elements = find_common_elements_in_lists(list1, list2, list3)
            # Display the result
            messagebox.showinfo("Result", f"Common Elements: {common_elements}")
        except Exception as e:
            messagebox.showerror("Error", f"Invalid input: {e}")
if __name__ == "__main__":
    root = tk.Tk()
    app = MainApp(root)
    root.mainloop()

'''
Main application file for finding common elements in three lists using a GUI.
'''
import tkinter as tk
from tkinter import messagebox
import ast
def find_common_elements_in_lists(l1, l2, l3):
    """
    Finds elements that are common in the same index across three lists.
    Parameters:
    l1 (list): First list.
    l2 (list): Second list.
    l3 (list): Third list.
    Returns:
    list: A list of common elements at the same index.
    """
    common_elements = []
    min_length = min(len(l1), len(l2), len(l3))
    for i in range(min_length):
        if l1[i] == l2[i] == l3[i]:
            common_elements.append(l1[i])
    return common_elements
class MainApp:
    def __init__(self, root):
        """
        Initializes the main application window and its components.
        Parameters:
        root (tk.Tk): The root window object.
        """
        self.root = root
        self.root.title("Common Elements Finder")
        # Labels and Entries for list inputs
        self.label1 = tk.Label(root, text="List 1:")
        self.label1.pack()
        self.entry1 = tk.Entry(root, width=50)
        self.entry1.pack()
        self.label2 = tk.Label(root, text="List 2:")
        self.label2.pack()
        self.entry2 = tk.Entry(root, width=50)
        self.entry2.pack()
        self.label3 = tk.Label(root, text="List 3:")
        self.label3.pack()
        self.entry3 = tk.Entry(root, width=50)
        self.entry3.pack()
        # Button to find common elements
        self.find_button = tk.Button(root, text="Find Common Elements", command=self.find_common_elements)
        self.find_button.pack()
    def find_common_elements(self):
        """
        Retrieves input lists, processes them, and displays the result.
        """
        try:
            # Convert input strings to lists using ast.literal_eval for safety
            list1 = ast.literal_eval(self.entry1.get())
            list2 = ast.literal_eval(self.entry2.get())
            list3 = ast.literal_eval(self.entry3.get())
            # Find common elements
            common_elements = find_common_elements_in_lists(list1, list2, list3)
            # Display the result
            messagebox.showinfo("Result", f"Common Elements: {common_elements}")
        except Exception as e:
            messagebox.showerror("Error", f"Invalid input: {e}")
if __name__ == "__main__":
    root = tk.Tk()
    app = MainApp(root)
    root.mainloop()
