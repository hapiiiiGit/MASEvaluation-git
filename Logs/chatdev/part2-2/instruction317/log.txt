'''

This is the main file for the Whitespace Remover application. It initializes and runs the GUI application.

'''

import tkinter as tk
from utils import remove_whitespace_from_string

class WhitespaceRemoverApp:
    def __init__(self, root):
        '''
        Initializes the GUI components.
        
        :param root: The root window of the tkinter application.
        '''
        self.root = root
        self.root.title("Whitespace Remover")

        # Input label and entry
        self.input_label = tk.Label(root, text="Enter String:")
        self.input_label.pack()

        self.input_entry = tk.Entry(root, width=50)
        self.input_entry.pack()

        # Button to trigger whitespace removal
        self.remove_button = tk.Button(root, text="Remove Whitespace", command=self.remove_whitespace)
        self.remove_button.pack()

        # Output label
        self.output_label = tk.Label(root, text="Output:")
        self.output_label.pack()

        self.output_text = tk.Label(root, text="", wraplength=400)
        self.output_text.pack()

    def remove_whitespace(self):
        '''
        Removes whitespaces from the input string and updates the output label.
        '''
        input_string = self.input_entry.get()
        result = remove_whitespace_from_string(input_string)
        self.output_text.config(text=result)

if __name__ == "__main__":
    root = tk.Tk()
    app = WhitespaceRemoverApp(root)
    root.mainloop()
```

#### utils.py

```python
'''

This file contains utility functions for the Whitespace Remover application.

'''

def remove_whitespace_from_string(input_string):
    '''
    Removes all whitespaces from the given input string.
    
    :param input_string: The string from which to remove whitespaces.
    :return: A new string with all whitespaces removed.
    '''
    return ''.join(input_string.split())
```

        self.root.title("Whitespace Remover")

        # Input label and entry
        self.input_label = tk.Label(root, text="Enter String:")
        self.input_label.pack()

        self.input_entry = tk.Entry(root, width=50)
        self.input_entry.pack()

        # Button to trigger whitespace removal
        self.remove_button = tk.Button(root, text="Remove Whitespace", command=self.remove_whitespace)
        self.remove_button.pack()

        # Output label
        self.output_label = tk.Label(root, text="Output:")
        self.output_label.pack()

        self.output_text = tk.Label(root, text="", wraplength=400)
        self.output_text.pack()

    def remove_whitespace(self):
        '''
        Removes whitespaces from the input string and updates the output label.
        '''
        input_string = self.input_entry.get()
        result = remove_whitespace_from_string(input_string)
        self.output_text.config(text=result)

if __name__ == "__main__":
    root = tk.Tk()
    app = WhitespaceRemoverApp(root)
    root.mainloop()

import tkinter as tk

class WhitespaceRemoverApp:
    def __init__(self, root):
        '''
        Initializes the GUI components.
        :param root: The root window of the tkinter application.
        '''
        self.root = root
        self.root.title("Whitespace Remover")
        self.root.geometry("500x300")  # Set a fixed window size for better layout
        self.root.resizable(False, False)  # Disable resizing for consistent appearance
        # Input label and entry
        self.input_label = tk.Label(root, text="Enter String:", font=("Arial", 12))
        self.input_label.pack(pady=10)
        self.input_entry = tk.Entry(root, width=50, font=("Arial", 12))
        self.input_entry.pack(pady=5)
        # Button to trigger whitespace removal
        self.remove_button = tk.Button(root, text="Remove Whitespace", command=self.remove_whitespace, font=("Arial", 12), bg="#4CAF50", fg="white")
        self.remove_button.pack(pady=10)
        # Output label
        self.output_label = tk.Label(root, text="Output:", font=("Arial", 12))
        self.output_label.pack(pady=10)
        self.output_text = tk.Label(root, text="", wraplength=400, font=("Arial", 12), fg="blue")
        self.output_text.pack(pady=5)
    def remove_whitespace(self):
        '''
        Removes whitespaces from the input string and updates the output label.
        '''
        input_string = self.input_entry.get()
        if not input_string.strip():
            # Handle empty or whitespace-only input
            self.output_text.config(text="Error: Input string is empty or contains only whitespaces.", fg="red")
        else:
            result = remove_whitespace_from_string(input_string)
            self.output_text.config(text=result, fg="blue")
if __name__ == "__main__":
    root = tk.Tk()
    app = WhitespaceRemoverApp(root)
    root.mainloop()

        self.root.title("Whitespace Remover")

        # Input label and entry

        # Output label
        self.output_label = tk.Label(root, text="Output:", font=("Arial", 12))

        '''
        Removes whitespaces from the input string and updates the output label.

        if not input_string.strip():
            # Handle empty or whitespace-only input
            self.output_text.config(text="Error: Input string is empty or contains only whitespaces.", fg="red")
        else:
            result = remove_whitespace_from_string(input_string)
            self.output_text.config(text=result, fg="blue")
