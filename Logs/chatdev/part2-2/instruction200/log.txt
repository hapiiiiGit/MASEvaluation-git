def find_difference(numbers):
    """

'''
This file contains the core logic for calculating the difference between the first even and first odd number in a list.
'''
def find_difference(numbers):
    """
    Finds the difference between the first even and first odd number in the given list.
    Args:
        numbers (list): A list of integers.
    Returns:
        int: The difference between the first even and first odd number.
    Raises:
        ValueError: If no even or odd number is found in the list.
    """
    first_even = None
    first_odd = None
    for num in numbers:
        if num % 2 == 0 and first_even is None:
            first_even = num
        elif num % 2 != 0 and first_odd is None:
            first_odd = num
        # Break the loop if both numbers are found
        if first_even is not None and first_odd is not None:
            break
    if first_even is None or first_odd is None:
        raise ValueError("The list must contain at least one even and one odd number.")
    return first_even - first_odd

def find_difference(numbers):  
    '''  
    Finds the difference between the first even and first odd number in the given list.  
    Args:  
        numbers (list): A list of integers.  
    Returns:  
        int: The difference between the first even and first odd number.  
    Raises:  
        ValueError: If no even or odd number is found in the list.  
    '''  
    first_even = None  
    first_odd = None  
    for num in numbers:  
        if num % 2 == 0 and first_even is None:  
            first_even = num  
        elif num % 2 != 0 and first_odd is None:  
            first_odd = num  
        # Break the loop if both numbers are found  
        if first_even is not None and first_odd is not None:  
            break  
    if first_even is None or first_odd is None:  
        raise ValueError("The list must contain at least one even and one odd number.")  
    return first_even - first_odd

'''  
This file contains the core logic for calculating the difference between the first even and first odd number in a list.  
'''  
def find_difference(numbers):  
    '''  
    Finds the difference between the first even and first odd number in the given list.  
    Args:  
        numbers (list): A list of integers.  
    Returns:  
        int: The difference between the first even and first odd number.  
    Raises:  
        ValueError: If no even or odd number is found in the list.  
    '''  
    first_even = None  
    first_odd = None  
    for num in numbers:  
        if num % 2 == 0 and first_even is None:  
            first_even = num  
        elif num % 2 != 0 and first_odd is None:  
            first_odd = num  
        # Break the loop if both numbers are found  
        if first_even is not None and first_odd is not None:  
            break  
    if first_even is None or first_odd is None:  
        raise ValueError("The list must contain at least one even and one odd number.")  
    return first_even - first_odd

'''  
This file contains the core logic for calculating the difference between the first even and first odd number in a list.  
'''  
def find_difference(numbers):  
    '''  
    Finds the difference between the first even and first odd number in the given list.  
    Args:  
        numbers (list): A list of integers.  
    Returns:  
        int: The difference between the first even and first odd number.  
    Raises:  
        ValueError: If no even or odd number is found in the list.  
    '''  
    first_even = None  
    first_odd = None  
    for num in numbers:  
        if num % 2 == 0 and first_even is None:  
            first_even = num  
        elif num % 2 != 0 and first_odd is None:  
            first_odd = num  
        # Break the loop if both numbers are found  
        if first_even is not None and first_odd is not None:  
            break  
    if first_even is None or first_odd is None:  
        raise ValueError("The list must contain at least one even and one odd number.")  
    return first_even - first_odd
