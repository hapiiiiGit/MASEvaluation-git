'''

This file contains the function to calculate the lateral surface area of a cone given its radius and height.

'''

import math

def calculate_lateral_surface_area(radius, height):
    '''
    Calculate the lateral surface area of a cone.

    Parameters:
    radius (float): The radius of the cone's base.
    height (float): The height of the cone.

    Returns:
    float: The lateral surface area of the cone.
    '''
    slant_height = math.sqrt(radius**2 + height**2)
    lateral_surface_area = math.pi * radius * slant_height
    return lateral_surface_area

import math
def calculate_lateral_surface_area(radius, height):
    '''
    Calculate the lateral surface area of a cone.
    Parameters:
    radius (float): The radius of the cone's base.
    height (float): The height of the cone.
    Returns:
    float: The lateral surface area of the cone.
    '''
    slant_height = math.sqrt(radius**2 + height**2)
    lateral_surface_area = math.pi * radius * slant_height
    return lateral_surface_area

'''
This file contains the function to calculate the lateral surface area of a cone given its radius and height.
'''
import math
def calculate_lateral_surface_area(radius, height):
    '''
    Calculate the lateral surface area of a cone.
    Parameters:
    radius (float): The radius of the cone's base.
    height (float): The height of the cone.
    Returns:
    float: The lateral surface area of the cone.
    '''
    slant_height = math.sqrt(radius**2 + height**2)
    lateral_surface_area = math.pi * radius * slant_height
    return lateral_surface_area

'''
This file contains the function to calculate the lateral surface area of a cone given its radius and height.
'''
import math
def calculate_lateral_surface_area(radius, height):
    '''
    Calculate the lateral surface area of a cone.
    Parameters:
    radius (float): The radius of the cone's base.
    height (float): The height of the cone.
    Returns:
    float: The lateral surface area of the cone.
    '''
    slant_height = math.sqrt(radius**2 + height**2)
    lateral_surface_area = math.pi * radius * slant_height
    return lateral_surface_area

'''
This file contains the function to calculate the lateral surface area of a cone given its radius and height.
'''
import math
def calculate_lateral_surface_area(radius, height):
    '''
    Calculate the lateral surface area of a cone.
    Parameters:
    radius (float): The radius of the cone's base.
    height (float): The height of the cone.
    Returns:
    float: The lateral surface area of the cone.
    '''
    slant_height = math.sqrt(radius**2 + height**2)
    lateral_surface_area = math.pi * radius * slant_height
    return lateral_surface_area

## Main Functions

The main function of this software is to calculate the lateral surface area of a cone using the formula:

\[ \text{Lateral Surface Area} = \pi \times \text{radius} \times \text{slant height} \]

Where the slant height is calculated as:

\[ \text{slant height} = \sqrt{\text{radius}^2 + \text{height}^2} \]

The application allows users to input the radius and height of the cone and then calculates and displays the lateral surface area.
