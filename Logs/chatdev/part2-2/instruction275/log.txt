class TupleProcessor:

'''
Module to process tuples and find minimum k records.
'''
class TupleProcessor:
    @staticmethod
    def find_min_k_records(tuples, k):
        '''
        Finds the minimum k records from a list of tuples based on the first element of each tuple.
        :param tuples: List of tuples
        :param k: Number of minimum records to find
        :return: List of k tuples with the smallest first elements
        '''
        if k > len(tuples):
            raise ValueError("k cannot be greater than the number of tuples.")
        # Sort tuples based on the first element
        sorted_tuples = sorted(tuples, key=lambda x: x[0])
        # Return the first k tuples
        return sorted_tuples[:k]

'''
Module to process tuples and find minimum k records.
'''
class TupleProcessor:
    @staticmethod
    def find_min_k_records(tuples, k):
        '''
        Finds the minimum k records from a list of tuples based on the first element of each tuple.
        :param tuples: List of tuples
        :param k: Number of minimum records to find
        :return: List of k tuples with the smallest first elements
        '''
        if k > len(tuples):
            raise ValueError("k cannot be greater than the number of tuples.")
        # Sort tuples based on the first element
        sorted_tuples = sorted(tuples, key=lambda x: x[0])
        # Return the first k tuples
        return sorted_tuples[:k]

class TupleProcessor:
    @staticmethod
    def find_min_k_records(tuples, k):
        '''
        Finds the minimum k records from a list of tuples based on the first element of each tuple.
        :param tuples: List of tuples
