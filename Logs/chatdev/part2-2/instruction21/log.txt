'''

Main file for the GUI application. Integrates the core functionality with a user-friendly interface.

'''

import tkinter as tk
from tkinter import messagebox
from bitonic_subsequence import find_max_sum_bitonic_subsequence

class App:
    def __init__(self, root):
        self.root = root
        self.root.title("Bitonic Subsequence Calculator")
        self.root.geometry("400x300")
        self.create_gui()

    def create_gui(self):
        # Input label and entry
        self.label = tk.Label(self.root, text="Enter an array (comma-separated):")
        self.label.pack(pady=10)

        self.entry = tk.Entry(self.root, width=40)
        self.entry.pack(pady=10)

        # Calculate button
        self.calculate_button = tk.Button(self.root, text="Calculate", command=self.on_calculate)
        self.calculate_button.pack(pady=10)

        # Result label
        self.result_label = tk.Label(self.root, text="", fg="blue")
        self.result_label.pack(pady=10)

    def on_calculate(self):
        try:
            # Get input and convert to list of integers
            input_text = self.entry.get()
            array = list(map(int, input_text.split(',')))

            # Calculate maximum sum of bitonic subsequence

'''
Contains the core function to calculate the maximum sum of a bitonic subsequence for a given array.
'''
def find_max_sum_bitonic_subsequence(arr):
    '''
    Function to find the maximum sum of a bitonic subsequence in the given array.
    A sequence is bitonic if it is first increasing and then decreasing.
    Parameters:
        arr (list): List of integers.
    Returns:
        int: Maximum sum of a bitonic subsequence.
    '''
    n = len(arr)
    # Step 1: Calculate increasing subsequence sums
    inc_sum = [0] * n
    for i in range(n):
        inc_sum[i] = arr[i]
        for j in range(i):
            if arr[j] < arr[i]:
                inc_sum[i] = max(inc_sum[i], inc_sum[j] + arr[i])
    # Step 2: Calculate decreasing subsequence sums
    dec_sum = [0] * n
    for i in range(n - 1, -1, -1):
        dec_sum[i] = arr[i]
        for j in range(i + 1, n):
            if arr[j] < arr[i]:
                dec_sum[i] = max(dec_sum[i], dec_sum[j] + arr[i])
    # Step 3: Calculate maximum sum of bitonic subsequence
    max_sum = 0
    for i in range(n):
        max_sum = max(max_sum, inc_sum[i] + dec_sum[i] - arr[i])
    return max_sum

'''
Contains the core function to calculate the maximum sum of a bitonic subsequence for a given array.
'''
def find_max_sum_bitonic_subsequence(arr):
    '''
    Function to find the maximum sum of a bitonic subsequence in the given array.
    A sequence is bitonic if it is first increasing and then decreasing.
    Parameters:
        arr (list): List of integers.
    Returns:
        int: Maximum sum of a bitonic subsequence.
    '''
    n = len(arr)
    # Step 1: Calculate increasing subsequence sums
    inc_sum = [0] * n
    for i in range(n):
        inc_sum[i] = arr[i]
        for j in range(i):
            if arr[j] < arr[i]:
                inc_sum[i] = max(inc_sum[i], inc_sum[j] + arr[i])
    # Step 2: Calculate decreasing subsequence sums
    dec_sum = [0] * n
    for i in range(n - 1, -1, -1):
        dec_sum[i] = arr[i]
        for j in range(i + 1, n):
            if arr[j] < arr[i]:
                dec_sum[i] = max(dec_sum[i], dec_sum[j] + arr[i])
    # Step 3: Calculate maximum sum of bitonic subsequence
    max_sum = 0
    for i in range(n):
        max_sum = max(max_sum, inc_sum[i] + dec_sum[i] - arr[i])
    return max_sum

'''  
Contains the core function to calculate the maximum sum of a bitonic subsequence for a given array.  
'''  
def find_max_sum_bitonic_subsequence(arr):  
    '''  
    Function to find the maximum sum of a bitonic subsequence in the given array.  
    A sequence is bitonic if it is first increasing and then decreasing.  
    Parameters:  
        arr (list): List of integers.  
    Returns:  
        int: Maximum sum of a bitonic subsequence.  
    '''  
    n = len(arr)  

    if n == 0:  
        return 0  

    # Step 1: Calculate increasing subsequence sums  
    inc_sum = [0] * n  
    for i in range(n):  
        inc_sum[i] = arr[i]  
        for j in range(i):  
            if arr[j] < arr[i]:  
                inc_sum[i] = max(inc_sum[i], inc_sum[j] + arr[i])  

    # Step 2: Calculate decreasing subsequence sums  
    dec_sum = [0] * n  
    for i in range(n - 1, -1, -1):  
        dec_sum[i] = arr[i]  
        for j in range(i + 1, n):  
            if arr[j] < arr[i]:  
                dec_sum[i] = max(dec_sum[i], dec_sum[j] + arr[i])  

    # Step 3: Calculate maximum sum of bitonic subsequence  
    max_sum = 0  
    for i in range(n):  
        max_sum = max(max_sum, inc_sum[i] + dec_sum[i] - arr[i])  

    return max_sum

'''  
Contains the core function to calculate the maximum sum of a bitonic subsequence for a given array.  
'''  
def find_max_sum_bitonic_subsequence(arr):  
    '''  
    Function to find the maximum sum of a bitonic subsequence in the given array.  
    A sequence is bitonic if it is first increasing and then decreasing.  
    Parameters:  
        arr (list): List of integers.  
    Returns:  
        int: Maximum sum of a bitonic subsequence.  
    '''  
    n = len(arr)  
    if n == 0:  
        return 0  
    # Step 1: Calculate increasing subsequence sums  
    inc_sum = [0] * n  
    for i in range(n):  
        inc_sum[i] = arr[i]  
        for j in range(i):  
            if arr[j] < arr[i]:  
                inc_sum[i] = max(inc_sum[i], inc_sum[j] + arr[i])  
    # Step 2: Calculate decreasing subsequence sums  
    dec_sum = [0] * n  
    for i in range(n - 1, -1, -1):  
        dec_sum[i] = arr[i]  
        for j in range(i + 1, n):  
            if arr[j] < arr[i]:  
                dec_sum[i] = max(dec_sum[i], dec_sum[j] + arr[i])  
    # Step 3: Calculate maximum sum of bitonic subsequence  
    max_sum = 0  
    for i in range(n):  
        max_sum = max(max_sum, inc_sum[i] + dec_sum[i] - arr[i])  
    return max_sum

'''
Contains the core function to calculate the maximum sum of a bitonic subsequence for a given array.
'''
def find_max_sum_bitonic_subsequence(arr):
    '''
    Function to find the maximum sum of a bitonic subsequence in the given array.
    A sequence is bitonic if it is first increasing and then decreasing.
    Parameters:
        arr (list): List of integers.
    Returns:
        int: Maximum sum of a bitonic subsequence.
    '''
    n = len(arr)
    if n == 0:
        return 0
    # Step 1: Calculate increasing subsequence sums
    inc_sum = [0] * n
    for i in range(n):
        inc_sum[i] = arr[i]
        for j in range(i):
            if arr[j] < arr[i]:
                inc_sum[i] = max(inc_sum[i], inc_sum[j] + arr[i])
    # Step 2: Calculate decreasing subsequence sums
    dec_sum = [0] * n
    for i in range(n - 1, -1, -1):
        dec_sum[i] = arr[i]
        for j in range(i + 1, n):
            if arr[j] < arr[i]:
                dec_sum[i] = max(dec_sum[i], dec_sum[j] + arr[i])
    # Step 3: Calculate maximum sum of bitonic subsequence
    max_sum = 0
    for i in range(n):
        max_sum = max(max_sum, inc_sum[i] + dec_sum[i] - arr[i])
    return max_sum

'''
Contains the core function to calculate the maximum sum of a bitonic subsequence for a given array.
'''
def find_max_sum_bitonic_subsequence(arr):
    '''
    Function to find the maximum sum of a bitonic subsequence in the given array.
    A sequence is bitonic if it is first increasing and then decreasing.
    Parameters:
        arr (list): List of integers.
    Returns:
        int: Maximum sum of a bitonic subsequence.
    '''
    n = len(arr)
    if n == 0:
        return 0
    # Step 1: Calculate increasing subsequence sums
    inc_sum = [0] * n
    for i in range(n):
        inc_sum[i] = arr[i]
        for j in range(i):
            if arr[j] < arr[i]:
                inc_sum[i] = max(inc_sum[i], inc_sum[j] + arr[i])
    # Step 2: Calculate decreasing subsequence sums
    dec_sum = [0] * n
    for i in range(n - 1, -1, -1):
        dec_sum[i] = arr[i]
        for j in range(i + 1, n):
            if arr[j] < arr[i]:
                dec_sum[i] = max(dec_sum[i], dec_sum[j] + arr[i])
    # Step 3: Calculate maximum sum of bitonic subsequence
    max_sum = 0
    for i in range(n):
        max_sum = max(max_sum, inc_sum[i] + dec_sum[i] - arr[i])
    return max_sum

'''
Contains the core function to calculate the maximum sum of a bitonic subsequence for a given array.
'''
def find_max_sum_bitonic_subsequence(arr):
    '''
    Function to find the maximum sum of a bitonic subsequence in the given array.
    A sequence is bitonic if it is first increasing and then decreasing.
    Parameters:
        arr (list): List of integers.
    Returns:
        int: Maximum sum of a bitonic subsequence.
    '''
    n = len(arr)
    if n == 0:
        return 0
    # Step 1: Calculate increasing subsequence sums
    inc_sum = [0] * n
    for i in range(n):
        inc_sum[i] = arr[i]
        for j in range(i):
            if arr[j] < arr[i]:
                inc_sum[i] = max(inc_sum[i], inc_sum[j] + arr[i])
    # Step 2: Calculate decreasing subsequence sums
    dec_sum = [0] * n
    for i in range(n - 1, -1, -1):
        dec_sum[i] = arr[i]
        for j in range(i + 1, n):
            if arr[j] < arr[i]:
                dec_sum[i] = max(dec_sum[i], dec_sum[j] + arr[i])
    # Step 3: Calculate maximum sum of bitonic subsequence
    max_sum = 0
    for i in range(n):
        max_sum = max(max_sum, inc_sum[i] + dec_sum[i] - arr[i])
    return max_sum
