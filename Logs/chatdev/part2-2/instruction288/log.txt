'''

This module contains the PairCounter class with a method to count pairs in a list whose sum equals a given target.

'''

class PairCounter:
    @staticmethod
    def count_pairs(numbers, target_sum):
        """
        Count the number of pairs in the list 'numbers' whose sum is equal to 'target_sum'.
        
        :param numbers: List of integers.
        :param target_sum: Integer representing the target sum.
        :return: Integer count of pairs.
        """
        count = 0
        seen = {}
        
        for number in numbers:
            complement = target_sum - number
            if complement in seen:
                count += seen[complement]
            
            if number in seen:
                seen[number] += 1
            else:
                seen[number] = 1
        
        return count
```

'''

This module contains the PairCounter class with a method to count pairs in a list whose sum equals a given target.

'''

class PairCounter:
    @staticmethod
    def count_pairs(numbers, target_sum):
        """
        Count the number of pairs in the list 'numbers' whose sum is equal to 'target_sum'.
        
        :param numbers: List of integers.
        :param target_sum: Integer representing the target sum.
        :return: Integer count of pairs.
        """
        count = 0
        seen = {}
        
        for number in numbers:
            complement = target_sum - number
            if complement in seen:
                count += seen[complement]
            
            if number in seen:
                seen[number] += 1
            else:
                seen[number] = 1
        
        return count
```

'''
This module contains the PairCounter class with a method to count pairs in a list whose sum equals a given target.
'''
class PairCounter:
    @staticmethod
    def count_pairs(numbers, target_sum):
        """
        Count the number of pairs in the list 'numbers' whose sum is equal to 'target_sum'.
        :param numbers: List of integers.
        :param target_sum: Integer representing the target sum.
        :return: Integer count of pairs.
        """
        count = 0
        seen = {}
        for number in numbers:
            complement = target_sum - number
            if complement in seen:
                count += seen[complement]
            if number in seen:
                seen[number] += 1
            else:
                seen[number] = 1
        return count

'''

'''
class PairCounter:
    @staticmethod
    def count_pairs(numbers, target_sum):
        """
        Count the number of pairs in the list 'numbers' whose sum is equal to 'target_sum'.
        :param numbers: List of integers.
        :param target_sum: Integer representing the target sum.
        :return: Integer count of pairs.
        """
        count = 0
        seen = {}
        for number in numbers:
            complement = target_sum - number
            if complement in seen:
                count += seen[complement]
            if number in seen:
                seen[number] += 1
            else:
                seen[number] = 1
        return count

'''
This module contains the PairCounter class with a method to count pairs in a list whose sum equals a given target.
'''
class PairCounter:
    @staticmethod
    def count_pairs(numbers, target_sum):
        """
        Count the number of pairs in the list 'numbers' whose sum is equal to 'target_sum'.
        :param numbers: List of integers.
        :param target_sum: Integer representing the target sum.
        :return: Integer count of pairs.
        """
        count = 0
        seen = {}
        for number in numbers:
            complement = target_sum - number
            if complement in seen:
                count += seen[complement]
            if number in seen:
                seen[number] += 1
            else:
                seen[number] = 1
        return count

class PairCounterApp:
    def __init__(self, master):
        """

        master.title("Pair Counter")

        """
        Create and place the widgets in the GUI.
        """
        self.instructions_label = tk.Label(self.master, text="Enter numbers separated by commas:")
        self.instructions_label.pack()
        self.numbers_entry = tk.Entry(self.master)
        self.numbers_entry.pack()
        self.sum_label = tk.Label(self.master, text="Enter target sum:")
        self.sum_label.pack()

        self.sum_entry.pack()
        self.calculate_button = tk.Button(self.master, text="Calculate Pairs", command=self.calculate_pairs)
        self.calculate_button.pack()

        self.result_label.pack()

'''
This module contains the PairCounter class with a method to count pairs in a list whose sum equals a given target.
'''
class PairCounter:
    @staticmethod
    def count_pairs(numbers, target_sum):
        """
        Count the number of pairs in the list 'numbers' whose sum is equal to 'target_sum'.
        :param numbers: List of integers.
        :param target_sum: Integer representing the target sum.
        :return: Integer count of pairs.
        """
        count = 0
        seen = {}
        for number in numbers:
            complement = target_sum - number
            if complement in seen:
                count += seen[complement]
            if number in seen:
                seen[number] += 1
            else:
                seen[number] = 1
        return count

'''

import tkinter as tk
from pair_counter import PairCounter
class PairCounterApp:
    def __init__(self, master):
        """
        Initialize the PairCounterApp with the main window.
        :param master: The root window.
        """
        self.master = master
        master.title("Pair Counter")
        self.create_widgets()
    def create_widgets(self):
        """
        Create and place the widgets in the GUI.
        """
        self.instructions_label = tk.Label(self.master, text="Enter numbers separated by commas:")
        self.instructions_label.pack()
        self.numbers_entry = tk.Entry(self.master)
        self.numbers_entry.pack()
        self.sum_label = tk.Label(self.master, text="Enter target sum:")
        self.sum_label.pack()
        self.sum_entry = tk.Entry(self.master)
        self.sum_entry.pack()
        self.calculate_button = tk.Button(self.master, text="Calculate Pairs", command=self.calculate_pairs)
        self.calculate_button.pack()
        self.result_label = tk.Label(self.master, text="")
        self.result_label.pack()
    def calculate_pairs(self):
        """
        Retrieve user input, calculate the number of pairs, and display the result.
        """
        numbers_str = self.numbers_entry.get()
        target_sum_str = self.sum_entry.get()
        try:
            numbers = list(map(int, numbers_str.split(',')))
            target_sum = int(target_sum_str)
            pair_count = PairCounter.count_pairs(numbers, target_sum)
            self.result_label.config(text=f"Number of pairs: {pair_count}")
        except ValueError:
            self.result_label.config(text="Invalid input. Please enter integers only.")

    root = tk.Tk()

The Pair Counter Application is a simple tool designed to help users count the number of pairs in a list of numbers whose sum equals a specified target. This application provides a graphical user interface (GUI) for ease of use and does not require any external dependencies.

## Main Functions

- **Count Pairs**: The core functionality of the application is to count the number of pairs in a list of numbers that add up to a given target sum. This is achieved through the `PairCounter` class, which implements an efficient algorithm to perform this task.

- **Graphical User Interface**: The application features a user-friendly GUI built using the Tkinter library. This interface allows users to input their list of numbers and target sum, and displays the result directly on the screen.

## Installation

### Prerequisites

- **Python**: Ensure that Python is installed on your system. You can download it from [python.org](https://www.python.org/).

### Quick Install

Since the application does not require any external dependencies, you can directly run the application without additional installations. Simply ensure that Python is installed and proceed to the usage instructions.

## Usage

1. **Download the Code**: Obtain the source code files `pair_counter.py` and `main.py`.

2. **Run the Application**: Open a terminal or command prompt, navigate to the directory containing the downloaded files, and execute the following command:

   ```bash
   python main.py
   ```

3. **Using the GUI**:
   - Enter the list of numbers separated by commas in the input field labeled "Enter numbers separated by commas".
   - Enter the target sum in the input field labeled "Enter target sum".
   - Click the "Calculate Pairs" button to compute the number of pairs.
   - The result will be displayed below the button, indicating the number of pairs whose sum equals the target.

## Example

Suppose you have the following list of numbers: `1, 2, 3, 4, 5` and you want to find the number of pairs that sum up to `5`. Enter `1,2,3,4,5` in the numbers input field and `5` in the target sum field, then click "Calculate Pairs". The application will display the result: "Number of pairs: 2" (i.e., pairs (1, 4) and (2, 3)).

## Troubleshooting

- **Invalid Input**: If you enter non-integer values or improperly formatted input, the application will display an error message: "Invalid input. Please enter integers only."

## Conclusion

The Pair Counter Application is a straightforward tool for quickly determining the number of pairs in a list that sum to a specified target. Its intuitive GUI makes it accessible for users with varying levels of technical expertise. Enjoy using the application to simplify your pair counting tasks!
