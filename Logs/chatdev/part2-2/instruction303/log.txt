def find_product_of_first_even_and_odd(numbers):
    '''
    Finds the product of the first even and odd numbers in the given list.

    Parameters:
    numbers (list): A list of integers.

    Returns:
    int: The product of the first even and odd numbers, or None if not possible.
    '''
    first_even = None
    first_odd = None
    
    for num in numbers:
        if first_even is None and num % 2 == 0:
            first_even = num
        if first_odd is None and num % 2 != 0:
            first_odd = num
        if first_even is not None and first_odd is not None:
            break
    
    if first_even is not None and first_odd is not None:
        return first_even * first_odd
    else:
        return None

'''
This module contains the implementation of a simple GUI application using tkinter.
The application allows users to input a list of numbers and calculates the product
of the first even and odd numbers in the list.
'''
import tkinter as tk
from tkinter import messagebox
def find_product_of_first_even_and_odd(numbers):
    '''
    Finds the product of the first even and odd numbers in the given list.
    Parameters:
    numbers (list): A list of integers.
    Returns:
    int: The product of the first even and odd numbers, or None if not possible.
    '''
    first_even = None
    first_odd = None
    for num in numbers:
        if first_even is None and num % 2 == 0:
            first_even = num
        if first_odd is None and num % 2 != 0:
            first_odd = num
        if first_even is not None and first_odd is not None:
            break
    if first_even is not None and first_odd is not None:
        return first_even * first_odd
    else:
        return None
class MainApp:
    '''
    Main application class for the GUI.
    '''
    def __init__(self, root):
        self.root = root
        self.root.title("Product of First Even and Odd")
        self.label = tk.Label(root, text="Enter numbers separated by commas:")
        self.label.pack()
        self.entry = tk.Entry(root, width=50)
        self.entry.pack()
        self.calculate_button = tk.Button(root, text="Calculate Product", command=self.calculate_product)
        self.calculate_button.pack()
        self.result_label = tk.Label(root, text="")
        self.result_label.pack()
    def calculate_product(self):
        '''
        Retrieves the input list, processes it, and displays the result.
        '''
        input_text = self.entry.get()
        try:
            numbers = list(map(int, input_text.split(',')))
            product = find_product_of_first_even_and_odd(numbers)
            if product is not None:
                self.result_label.config(text=f"Product of first even and odd: {product}")
            else:
                self.result_label.config(text="Could not find both even and odd numbers.")
        except ValueError:
            messagebox.showerror("Invalid input", "Please enter a valid list of integers separated by commas.")
if __name__ == "__main__":
    root = tk.Tk()
    app = MainApp(root)
    root.mainloop()

'''
This module contains the implementation of a simple GUI application using tkinter.
The application allows users to input a list of numbers and calculates the product
of the first even and odd numbers in the list.
'''
import tkinter as tk
from tkinter import messagebox
def find_product_of_first_even_and_odd(numbers):
    '''
    Finds the product of the first even and odd numbers in the given list.
    Parameters:
    numbers (list): A list of integers.
    Returns:
    int: The product of the first even and odd numbers, or None if not possible.
    '''
    first_even = None
    first_odd = None
    for num in numbers:
        if first_even is None and num % 2 == 0:
            first_even = num
        if first_odd is None and num % 2 != 0:
            first_odd = num
        if first_even is not None and first_odd is not None:
            break
    if first_even is not None and first_odd is not None:
        return first_even * first_odd
    else:
        return None
class MainApp:
    '''
    Main application class for the GUI.
    '''
    def __init__(self, root):
        self.root = root
        self.root.title("Product of First Even and Odd")
        self.label = tk.Label(root, text="Enter numbers separated by commas:")
        self.label.pack()
        self.entry = tk.Entry(root, width=50)
        self.entry.pack()
        self.calculate_button = tk.Button(root, text="Calculate Product", command=self.calculate_product)
        self.calculate_button.pack()
        self.result_label = tk.Label(root, text="")
        self.result_label.pack()
    def calculate_product(self):
        '''
        Retrieves the input list, processes it, and displays the result.
        '''
        input_text = self.entry.get()
        try:
            numbers = list(map(int, input_text.split(',')))
            product = find_product_of_first_even_and_odd(numbers)
            if product is not None:
                self.result_label.config(text=f"Product of first even and odd: {product}")
            else:
                self.result_label.config(text="Could not find both even and odd numbers.")
        except ValueError:
            messagebox.showerror("Invalid input", "Please enter a valid list of integers separated by commas.")
if __name__ == "__main__":
    root = tk.Tk()
    app = MainApp(root)
    root.mainloop()

'''
This module contains the implementation of a simple GUI application using tkinter.
The application allows users to input a list of numbers and calculates the product
of the first even and odd numbers in the list.
'''
import tkinter as tk
from tkinter import messagebox
def find_product_of_first_even_and_odd(numbers):
    '''
    Finds the product of the first even and odd numbers in the given list.
    Parameters:
    numbers (list): A list of integers.
    Returns:
    int: The product of the first even and odd numbers, or None if not possible.
    '''
    first_even = None
    first_odd = None
    for num in numbers:
        if first_even is None and num % 2 == 0:
            first_even = num
        if first_odd is None and num % 2 != 0:
            first_odd = num
        if first_even is not None and first_odd is not None:
            break
    if first_even is not None and first_odd is not None:
        return first_even * first_odd
    else:
        return None
class MainApp:
    '''
    Main application class for the GUI.
    '''
    def __init__(self, root):
        self.root = root
        self.root.title("Product of First Even and Odd")
        self.label = tk.Label(root, text="Enter numbers separated by commas:")
        self.label.pack()
        self.entry = tk.Entry(root, width=50)
        self.entry.pack()
        self.calculate_button = tk.Button(root, text="Calculate Product", command=self.calculate_product)
        self.calculate_button.pack()
        self.result_label = tk.Label(root, text="")
        self.result_label.pack()
    def calculate_product(self):
        '''
        Retrieves the input list, processes it, and displays the result.
        '''
        input_text = self.entry.get()
        try:
            numbers = list(map(int, input_text.split(',')))
            product = find_product_of_first_even_and_odd(numbers)
            if product is not None:
                self.result_label.config(text=f"Product of first even and odd: {product}")
            else:
                self.result_label.config(text="Could not find both even and odd numbers.")
        except ValueError:
            messagebox.showerror("Invalid input", "Please enter a valid list of integers separated by commas.")
if __name__ == "__main__":
    root = tk.Tk()
    app = MainApp(root)
    root.mainloop()
