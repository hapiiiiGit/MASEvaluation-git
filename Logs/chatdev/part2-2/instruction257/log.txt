'''

Main application file for toggling bits of a number except the first and last bit using tkinter for GUI.

'''

import tkinter as tk

def toggle_bits_function(n):
    '''
    Toggle bits of the number except the first and last bit.
    
    Parameters:
    n (int): The input number whose bits are to be toggled.
    
    Returns:
    int: The number with toggled bits except the first and last bit.
    '''
    # Convert number to binary string
    binary_str = bin(n)[2:]  # Remove '0b' prefix
    if len(binary_str) <= 2:
        return n  # No bits to toggle if length is 2 or less
    
    # Toggle bits except the first and last
    toggled_bits = binary_str[0] + ''.join('1' if bit == '0' else '0' for bit in binary_str[1:-1]) + binary_str[-1]
    
    # Convert back to integer
    return int(toggled_bits, 2)

import tkinter as tk

def toggle_bits_function(n):
    '''
    Toggle bits of the number except the first and last bit.
    
    Parameters:
    n (int): The input number whose bits are to be toggled.
    
    Returns:
    int: The number with toggled bits except the first and last bit.

    # Convert back to integer
    return int(toggled_bits, 2)

'''
Main application file for toggling bits of a number except the first and last bit using tkinter for GUI.

import tkinter as tk
def toggle_bits_function(n):
    '''
    Toggle bits of the number except the first and last bit.
    Parameters:
    n (int): The input number whose bits are to be toggled.
    Returns:
    int: The number with toggled bits except the first and last bit.
    '''
    # Convert number to binary string
    binary_str = bin(n)[2:]  # Remove '0b' prefix
    if len(binary_str) <= 2:
        return n  # No bits to toggle if length is 2 or less
    # Toggle bits except the first and last
    toggled_bits = binary_str[0] + ''.join('1' if bit == '0' else '0' for bit in binary_str[1:-1]) + binary_str[-1]
    # Convert back to integer
    return int(toggled_bits, 2)
class MainApplication:
    def __init__(self, root):
        '''
        Initialize the main application with GUI components.
        Parameters:
        root (tk.Tk): The root window of the tkinter application.
        '''
        self.root = root
        self.root.title("Bit Toggle Application")
        # Create and place widgets
        self.label = tk.Label(root, text="Enter a number:")
        self.label.pack()
        self.entry = tk.Entry(root)
        self.entry.pack()
        self.toggle_button = tk.Button(root, text="Toggle Bits", command=self.on_toggle_button_click)
        self.toggle_button.pack()
        self.result_label = tk.Label(root, text="")
        self.result_label.pack()
    def on_toggle_button_click(self):
        '''
        Handle the toggle button click event to perform bit toggling.
        '''
        try:
            number = int(self.entry.get())
            toggled_number = toggle_bits_function(number)
            self.result_label.config(text=f"Toggled Number: {toggled_number}")
        except ValueError:
            self.result_label.config(text="Invalid input. Please enter a valid integer.")
if __name__ == "__main__":
    root = tk.Tk()
    app = MainApplication(root)
    root.mainloop()

'''
Main application file for toggling bits of a number except the first and last bit using tkinter for GUI.
'''
import tkinter as tk
def toggle_bits_function(n):
    '''
    Toggle bits of the number except the first and last bit.
    Parameters:
    n (int): The input number whose bits are to be toggled.
    Returns:
    int: The number with toggled bits except the first and last bit.
    '''
    # Convert number to binary string
    binary_str = bin(n)[2:]  # Remove '0b' prefix
    if len(binary_str) <= 2:
        return n  # No bits to toggle if length is 2 or less
    # Toggle bits except the first and last
    toggled_bits = binary_str[0] + ''.join('1' if bit == '0' else '0' for bit in binary_str[1:-1]) + binary_str[-1]
    # Convert back to integer
    return int(toggled_bits, 2)
class MainApplication:
    def __init__(self, root):
        '''
        Initialize the main application with GUI components.
        Parameters:
        root (tk.Tk): The root window of the tkinter application.
        '''
        self.root = root
        self.root.title("Bit Toggle Application")
        # Create and place widgets
        self.label = tk.Label(root, text="Enter a number:")
        self.label.pack()
        self.entry = tk.Entry(root)
        self.entry.pack()
        self.toggle_button = tk.Button(root, text="Toggle Bits", command=self.on_toggle_button_click)
        self.toggle_button.pack()
        self.result_label = tk.Label(root, text="")
        self.result_label.pack()
    def on_toggle_button_click(self):
        '''
        Handle the toggle button click event to perform bit toggling.
        '''
        try:
            number = int(self.entry.get())
            toggled_number = toggle_bits_function(number)
            self.result_label.config(text=f"Toggled Number: {toggled_number}")
        except ValueError:
            self.result_label.config(text="Invalid input. Please enter a valid integer.")
if __name__ == "__main__":
    root = tk.Tk()
    app = MainApplication(root)
    root.mainloop()

'''
Main application file for toggling bits of a number except the first and last bit using tkinter for GUI.
'''
import tkinter as tk
def toggle_bits_function(n):
    '''
    Toggle bits of the number except the first and last bit.
    Parameters:
    n (int): The input number whose bits are to be toggled.
    Returns:
    int: The number with toggled bits except the first and last bit.
    '''
    # Convert number to binary string
    binary_str = bin(n)[2:]  # Remove '0b' prefix
    if len(binary_str) <= 2:
        return n  # No bits to toggle if length is 2 or less
    # Toggle bits except the first and last
    toggled_bits = binary_str[0] + ''.join('1' if bit == '0' else '0' for bit in binary_str[1:-1]) + binary_str[-1]
    # Convert back to integer
    return int(toggled_bits, 2)
class MainApplication:
    def __init__(self, root):
        '''
        Initialize the main application with GUI components.
        Parameters:
        root (tk.Tk): The root window of the tkinter application.
        '''
        self.root = root
        self.root.title("Bit Toggle Application")
        # Create and place widgets
        self.label = tk.Label(root, text="Enter a number:")
        self.label.pack()
        self.entry = tk.Entry(root)
        self.entry.pack()
        self.toggle_button = tk.Button(root, text="Toggle Bits", command=self.on_toggle_button_click)
        self.toggle_button.pack()
        self.result_label = tk.Label(root, text="")
        self.result_label.pack()
    def on_toggle_button_click(self):
        '''
        Handle the toggle button click event to perform bit toggling.
        '''
        try:
            number = int(self.entry.get())
            toggled_number = toggle_bits_function(number)
            self.result_label.config(text=f"Toggled Number: {toggled_number}")
        except ValueError:
            self.result_label.config(text="Invalid input. Please enter a valid integer.")
if __name__ == "__main__":
    root = tk.Tk()
    app = MainApplication(root)
    root.mainloop()
