'''

This file contains the MainApplication class and the flatten function to create a GUI application that flattens nested lists.

'''

import tkinter as tk
from tkinter import messagebox

def flatten(nested_list):
    '''
    Recursively flattens a nested list.
    
    Parameters:
    nested_list (list): The list to be flattened.
    
    Returns:
    list: A flattened list.
    '''
    flat_list = []
    for item in nested_list:
        if isinstance(item, list):
            flat_list.extend(flatten(item))
        else:
            flat_list.append(item)
    return flat_list

class MainApplication:
    '''
    Manages the GUI components and user interactions for the list flattening application.
    '''
    def __init__(self, root):
        self.root = root
        self.root.title("Nested List Flattener")

        self.label = tk.Label(root, text="Enter a nested list:")
        self.label.pack()

        self.entry = tk.Entry(root, width=50)
        self.entry.pack()

        self.button = tk.Button(root, text="Flatten List", command=self.flatten_list)
        self.button.pack()

        self.result_label = tk.Label(root, text="Flattened List:")
        self.result_label.pack()

        self.result = tk.Label(root, text="", wraplength=400)
        self.result.pack()

    def flatten_list(self):
        '''
        Handles the flattening of the nested list and updates the GUI with the result.
        '''
        try:
            nested_list = eval(self.entry.get())
            if not isinstance(nested_list, list):
                raise ValueError("Input is not a list.")
            flattened = flatten(nested_list)
            self.result.config(text=str(flattened))
        except Exception as e:
            messagebox.showerror("Error", f"Invalid input: {e}")

if __name__ == "__main__":
    root = tk.Tk()
    app = MainApplication(root)
    root.mainloop()

```

This file contains the MainApplication class and the flatten function to create a GUI application that flattens nested lists.

import tkinter as tk
from tkinter import messagebox

def flatten(nested_list):
    '''
    Recursively flattens a nested list.
    
    Parameters:
    nested_list (list): The list to be flattened.
    
    Returns:
    list: A flattened list.
    '''
    flat_list = []
    for item in nested_list:
        if isinstance(item, list):
            flat_list.extend(flatten(item))
        else:
            flat_list.append(item)
    return flat_list

        self.root.title("Nested List Flattener")

'''
This file contains the MainApplication class and the flatten function to create a GUI application that flattens nested lists.
'''
import tkinter as tk
from tkinter import messagebox
import ast
def flatten(nested_list):
    '''
    Recursively flattens a nested list.
    Parameters:
    nested_list (list): The list to be flattened.
    Returns:
    list: A flattened list.
    '''
    flat_list = []
    for item in nested_list:
        if isinstance(item, list):
            flat_list.extend(flatten(item))
        else:
            flat_list.append(item)
    return flat_list
class MainApplication:
    '''
    Manages the GUI components and user interactions for the list flattening application.
    '''
    def __init__(self, root):
        self.root = root
        self.root.title("Nested List Flattener")
        self.label = tk.Label(root, text="Enter a nested list:")
        self.label.pack()
        self.entry = tk.Entry(root, width=50)
        self.entry.pack()
        self.button = tk.Button(root, text="Flatten List", command=self.flatten_list)
        self.button.pack()
        self.result_label = tk.Label(root, text="Flattened List:")
        self.result_label.pack()
        self.result = tk.Label(root, text="", wraplength=400)
        self.result.pack()
    def flatten_list(self):
        '''
        Handles the flattening of the nested list and updates the GUI with the result.
        '''
        try:
            nested_list = ast.literal_eval(self.entry.get())
            if not isinstance(nested_list, list):
                raise ValueError("Input is not a list.")
            flattened = flatten(nested_list)
            self.result.config(text=str(flattened))
        except Exception as e:
            messagebox.showerror("Error", f"Invalid input: {e}")
if __name__ == "__main__":
    root = tk.Tk()
    app = MainApplication(root)
    root.mainloop()

'''
This file contains the MainApplication class and the flatten function to create a GUI application that flattens nested lists.
'''
import tkinter as tk
from tkinter import messagebox
import ast
def flatten(nested_list):
    '''
    Recursively flattens a nested list.
    Parameters:
    nested_list (list): The list to be flattened.
    Returns:
    list: A flattened list.
    '''
    flat_list = []
    for item in nested_list:
        if isinstance(item, list):
            flat_list.extend(flatten(item))
        else:
            flat_list.append(item)
    return flat_list
class MainApplication:
    '''
    Manages the GUI components and user interactions for the list flattening application.
    '''
    def __init__(self, root):
        self.root = root
        self.root.title("Nested List Flattener")
        self.label = tk.Label(root, text="Enter a nested list:")
        self.label.pack()
        self.entry = tk.Entry(root, width=50)
        self.entry.pack()
        self.button = tk.Button(root, text="Flatten List", command=self.flatten_list)
        self.button.pack()
        self.result_label = tk.Label(root, text="Flattened List:")
        self.result_label.pack()
        self.result = tk.Label(root, text="", wraplength=400)
        self.result.pack()
    def flatten_list(self):
        '''
        Handles the flattening of the nested list and updates the GUI with the result.
        '''
        try:
            nested_list = ast.literal_eval(self.entry.get())
            if not isinstance(nested_list, list):
                raise ValueError("Input is not a list.")
            flattened = flatten(nested_list)
            self.result.config(text=str(flattened))
        except Exception as e:
            messagebox.showerror("Error", f"Invalid input: {e}")
if __name__ == "__main__":
    root = tk.Tk()
    app = MainApplication(root)
    root.mainloop()

'''
This file contains the MainApplication class and the flatten function to create a GUI application that flattens nested lists.
'''
import tkinter as tk
from tkinter import messagebox
import ast

def flatten(nested_list):
    '''
    Recursively flattens a nested list.
    
    Parameters:
    nested_list (list): The list to be flattened.
    
    Returns:
    list: A flattened list.
    '''
    flat_list = []
    for item in nested_list:
        if isinstance(item, list):
            flat_list.extend(flatten(item))
        else:
            flat_list.append(item)
    return flat_list

class MainApplication:
    '''
    Manages the GUI components and user interactions for the list flattening application.
    '''
    def __init__(self, root):
        self.root = root
        self.root.title("Nested List Flattener")
        
        self.label = tk.Label(root, text="Enter a nested list:")
        self.label.pack()
        
        self.entry = tk.Entry(root, width=50)
        self.entry.pack()
        
        self.button = tk.Button(root, text="Flatten List", command=self.flatten_list)
        self.button.pack()
        
        self.result_label = tk.Label(root, text="Flattened List:")
        self.result_label.pack()
        
        self.result = tk.Label(root, text="", wraplength=400)
        self.result.pack()

    def flatten_list(self):
        '''
        Handles the flattening of the nested list and updates the GUI with the result.
        '''
        try:
            nested_list = ast.literal_eval(self.entry.get())
            if not isinstance(nested_list, list):
                raise ValueError("Input is not a list.")
            flattened = flatten(nested_list)
            self.result.config(text=str(flattened))
        except Exception as e:
            messagebox.showerror("Error", f"Invalid input: {e}")

if __name__ == "__main__":
    root = tk.Tk()
    app = MainApplication(root)
    root.mainloop()

'''
This file contains the MainApplication class and the flatten function to create a GUI application that flattens nested lists.
'''
import tkinter as tk
from tkinter import messagebox
import ast
def flatten(nested_list):
    '''
    Recursively flattens a nested list.
    Parameters:
    nested_list (list): The list to be flattened.
    Returns:
    list: A flattened list.
    '''
    flat_list = []
    for item in nested_list:
        if isinstance(item, list):
            flat_list.extend(flatten(item))
        else:
            flat_list.append(item)
    return flat_list
class MainApplication:
    '''
    Manages the GUI components and user interactions for the list flattening application.
    '''
    def __init__(self, root):
        self.root = root
        self.root.title("Nested List Flattener")
        self.label = tk.Label(root, text="Enter a nested list:")
        self.label.pack()
        self.entry = tk.Entry(root, width=50)
        self.entry.pack()
        self.button = tk.Button(root, text="Flatten List", command=self.flatten_list)
        self.button.pack()
        self.result_label = tk.Label(root, text="Flattened List:")
        self.result_label.pack()
        self.result = tk.Label(root, text="", wraplength=400)
        self.result.pack()
    def flatten_list(self):
        '''
        Handles the flattening of the nested list and updates the GUI with the result.
        '''
        try:
            nested_list = ast.literal_eval(self.entry.get())
            if not isinstance(nested_list, list):
                raise ValueError("Input is not a list.")
            flattened = flatten(nested_list)
            self.result.config(text=str(flattened))
        except Exception as e:
            messagebox.showerror("Error", f"Invalid input: {e}")
if __name__ == "__main__":
    root = tk.Tk()
    app = MainApplication(root)
    root.mainloop()
