'''

This module contains the main application class for calculating the area of a rectangle using a GUI.

'''

import tkinter as tk
from tkinter import messagebox

def calculate_rectangle_area(width, height):
    '''
    Calculate the area of a rectangle given its width and height.

    :param width: The width of the rectangle.
    :param height: The height of the rectangle.
    :return: The area of the rectangle.
    '''
    return width * height

class RectangleAreaApp:
    '''
    A class to create a GUI application for calculating the area of a rectangle.
    '''

    def __init__(self, master):
        '''
        Initialize the application with the main window and its components.

        :param master: The root window of the application.
        '''
        self.master = master
        master.title("Rectangle Area Calculator")

        # Create and place labels, entries, and button
        self.label_width = tk.Label(master, text="Width:")
        self.label_width.grid(row=0, column=0, padx=10, pady=10)

        self.entry_width = tk.Entry(master)
        self.entry_width.grid(row=0, column=1, padx=10, pady=10)

        self.label_height = tk.Label(master, text="Height:")
        self.label_height.grid(row=1, column=0, padx=10, pady=10)

        self.entry_height = tk.Entry(master)
        self.entry_height.grid(row=1, column=1, padx=10, pady=10)

        self.calculate_button = tk.Button(master, text="Calculate Area", command=self.calculate_area)
        self.calculate_button.grid(row=2, column=0, columnspan=2, pady=10)

        self.result_label = tk.Label(master, text="")
        self.result_label.grid(row=3, column=0, columnspan=2, pady=10)

    def calculate_area(self):
        '''
        Calculate the area of the rectangle and display the result.
        '''
        try:
            width = float(self.entry_width.get())
            height = float(self.entry_height.get())

'''
This module contains the main application class for calculating the area of a rectangle using a GUI.
'''
import tkinter as tk
from tkinter import messagebox
def calculate_rectangle_area(width, height):
    '''
    Calculate the area of a rectangle given its width and height.
    :param width: The width of the rectangle.
    :param height: The height of the rectangle.
    :return: The area of the rectangle.
    '''
    return width * height
class RectangleAreaApp:
    '''
    A class to create a GUI application for calculating the area of a rectangle.
    '''
    def __init__(self, master):
        '''
        Initialize the application with the main window and its components.
        :param master: The root window of the application.
        '''
        self.master = master
        master.title("Rectangle Area Calculator")
        # Create and place labels, entries, and button
        self.label_width = tk.Label(master, text="Width:")
        self.label_width.grid(row=0, column=0, padx=10, pady=10)
        self.entry_width = tk.Entry(master)
        self.entry_width.grid(row=0, column=1, padx=10, pady=10)
        self.label_height = tk.Label(master, text="Height:")
        self.label_height.grid(row=1, column=0, padx=10, pady=10)
        self.entry_height = tk.Entry(master)
        self.entry_height.grid(row=1, column=1, padx=10, pady=10)
        self.calculate_button = tk.Button(master, text="Calculate Area", command=self.calculate_area)
        self.calculate_button.grid(row=2, column=0, columnspan=2, pady=10)
        self.result_label = tk.Label(master, text="")
        self.result_label.grid(row=3, column=0, columnspan=2, pady=10)
    def calculate_area(self):
        '''
        Calculate the area of the rectangle and display the result.
        '''
        try:
            width = float(self.entry_width.get())
            height = float(self.entry_height.get())
            area = calculate_rectangle_area(width, height)
            self.result_label.config(text=f"Area: {area}")
        except ValueError:
            messagebox.showerror("Invalid input", "Please enter valid numbers for width and height.")
if __name__ == "__main__":
    root = tk.Tk()
    app = RectangleAreaApp(root)
    root.mainloop()

'''
This module contains the main application class for calculating the area of a rectangle using a GUI.
'''
import tkinter as tk
from tkinter import messagebox
def calculate_rectangle_area(width, height):
    '''
    Calculate the area of a rectangle given its width and height.
    :param width: The width of the rectangle.
    :param height: The height of the rectangle.
    :return: The area of the rectangle.
    '''
    return width * height
class RectangleAreaApp:
    '''
    A class to create a GUI application for calculating the area of a rectangle.
    '''
    def __init__(self, master):
        '''
        Initialize the application with the main window and its components.
        :param master: The root window of the application.
        '''
        self.master = master
        master.title("Rectangle Area Calculator")
        # Create and place labels, entries, and button
        self.label_width = tk.Label(master, text="Width:")
        self.label_width.grid(row=0, column=0, padx=10, pady=10)
        self.entry_width = tk.Entry(master)
        self.entry_width.grid(row=0, column=1, padx=10, pady=10)
        self.label_height = tk.Label(master, text="Height:")
        self.label_height.grid(row=1, column=0, padx=10, pady=10)
        self.entry_height = tk.Entry(master)
        self.entry_height.grid(row=1, column=1, padx=10, pady=10)
        self.calculate_button = tk.Button(master, text="Calculate Area", command=self.calculate_area)
        self.calculate_button.grid(row=2, column=0, columnspan=2, pady=10)
        self.result_label = tk.Label(master, text="")
        self.result_label.grid(row=3, column=0, columnspan=2, pady=10)
    def calculate_area(self):
        '''
        Calculate the area of the rectangle and display the result.
        '''
        try:
            width = float(self.entry_width.get())
            height = float(self.entry_height.get())
            area = calculate_rectangle_area(width, height)
            self.result_label.config(text=f"Area: {area}")
        except ValueError:
            messagebox.showerror("Invalid input", "Please enter valid numbers for width and height.")
if __name__ == "__main__":
    root = tk.Tk()
    app = RectangleAreaApp(root)
    root.mainloop()
