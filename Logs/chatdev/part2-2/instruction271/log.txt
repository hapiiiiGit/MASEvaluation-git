import tkinter as tk
from tkinter import messagebox
def sort_numeric_strings(strings):
    '''
    Sorts a list of strings representing numbers numerically.
    Parameters:
    strings (list): A list of strings representing numbers.
    Returns:
    list: A sorted list of strings based on their numeric value.
    '''
    return sorted(strings, key=lambda x: int(x))
class MainApplication:
    '''
    Main application class for the GUI.
    '''
    def __init__(self, root):
        '''
        Initializes the main application window and its components.
        Parameters:
        root (tk.Tk): The root window of the application.

        self.root = root
        self.root.title("Numeric String Sorter")
        self.label = tk.Label(root, text="Enter numbers separated by commas:")

        self.entry = tk.Entry(root, width=50)
        self.entry.pack(pady=10)
        self.sort_button = tk.Button(root, text="Sort", command=self.sort_numbers)
        self.sort_button.pack(pady=10)
        self.result_label = tk.Label(root, text="")
        self.result_label.pack(pady=10)
    def sort_numbers(self):
        '''
        Sorts the numbers entered by the user and displays the result.
        '''
        input_text = self.entry.get()

            # Split the input text by commas and strip any extra spaces
            number_strings = [s.strip() for s in input_text.split(',')]
            # Sort the numeric strings
            sorted_numbers = sort_numeric_strings(number_strings)
            # Display the sorted result
            self.result_label.config(text="Sorted: " + ", ".join(sorted_numbers))
        except ValueError:
            messagebox.showerror("Invalid input", "Please enter valid numbers separated by commas.")

    root = tk.Tk()

def sort_numeric_strings(strings):
    '''
    Sorts a list of strings representing numbers numerically.
    Parameters:
    strings (list): A list of strings representing numbers.
    Returns:
    list: A sorted list of strings based on their numeric value.
    '''

    '''
    Main application class for the GUI.

        '''
