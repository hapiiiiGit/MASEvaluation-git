'''

Main application file for checking odd indices in a list using a GUI.

'''

import tkinter as tk
from tkinter import messagebox

def check_odd_indices(numbers):
    '''
    Check if every odd index in the list contains an odd number.

    Parameters:
    numbers (list): List of integers.

    Returns:
    bool: True if every odd index contains an odd number, False otherwise.
    '''
    for i in range(1, len(numbers), 2):
        if numbers[i] % 2 == 0:
            return False
    return True

class MainApplication(tk.Tk):
    '''
    Main application class for the GUI.
    '''
    def __init__(self):
        super().__init__()
        self.title("Odd Index Checker")
        self.geometry("300x200")

        self.label = tk.Label(self, text="Enter numbers separated by commas:")
        self.label.pack(pady=10)

        self.entry = tk.Entry(self, width=30)
        self.entry.pack(pady=10)

        self.check_button = tk.Button(self, text="Check", command=self.on_check_button_click)
        self.check_button.pack(pady=10)

    def on_check_button_click(self):
        '''
        Event handler for the check button click.
        '''
        input_text = self.entry.get()
        try:
            numbers = list(map(int, input_text.split(',')))
            result = check_odd_indices(numbers)
            message = "Every odd index contains an odd number." if result else "Not every odd index contains an odd number."
            messagebox.showinfo("Result", message)
        except ValueError:
            messagebox.showerror("Error", "Please enter a valid list of numbers.")

if __name__ == "__main__":
    app = MainApplication()
    app.mainloop()

```

import tkinter as tk
from tkinter import messagebox

def check_odd_indices(numbers):
    '''
    Check if every odd index in the list contains an odd number.
    
    Parameters:
    numbers (list): List of integers.
    
    Returns:
    bool: True if every odd index contains an odd number, False otherwise.
    '''
    for i in range(1, len(numbers), 2):
        if numbers[i] % 2 == 0:
            return False
    return True

    Main application class for the GUI.
    '''

        self.label = tk.Label(self, text="Enter numbers separated by commas:")
        self.label.pack(pady=10)
        self.entry = tk.Entry(self, width=30)
        self.entry.pack(pady=10)
        self.check_button = tk.Button(self, text="Check", command=self.on_check_button_click)
        self.check_button.pack(pady=10)

        '''
        Event handler for the check button click.

            numbers = list(map(int, input_text.split(',')))

            message = "Every odd index contains an odd number." if result else "Not every odd index contains an odd number."

Main application file for checking odd indices in a list using a GUI.

import tkinter as tk
from tkinter import messagebox
def check_odd_indices(numbers):
    '''
    Check if every odd index in the list contains an odd number.
    Parameters:
    numbers (list): List of integers.
    Returns:
    bool: True if every odd index contains an odd number, False otherwise.
    '''
    for i in range(1, len(numbers), 2):
        if numbers[i] % 2 == 0:
            return False
    return True
class MainApplication(tk.Tk):
    '''
    Main application class for the GUI.
    '''
    def __init__(self):
        super().__init__()
        self.title("Odd Index Checker")
        self.geometry("300x200")
        self.label = tk.Label(self, text="Enter numbers separated by commas:")
        self.label.pack(pady=10)
        self.entry = tk.Entry(self, width=30)
        self.entry.pack(pady=10)
        self.check_button = tk.Button(self, text="Check", command=self.on_check_button_click)
        self.check_button.pack(pady=10)
    def on_check_button_click(self):
        '''
        Event handler for the check button click.
        '''
        input_text = self.entry.get()
        try:
            numbers = list(map(int, input_text.split(',')))
            result = check_odd_indices(numbers)

        except ValueError:
            messagebox.showerror("Error", "Please enter a valid list of numbers.")
