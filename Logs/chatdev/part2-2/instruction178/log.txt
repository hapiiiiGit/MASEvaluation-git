from utils import sort_each_sublist
class SortApp:

        '''
        Initialize the GUI components.
        '''
        self.root = root
        self.root.title("Sort Sublists Application")
        self.input_label = tk.Label(root, text="Enter list of lists:")
        self.input_label.pack()
        self.input_text = tk.Text(root, height=10, width=50)
        self.input_text.pack()
        self.sort_button = tk.Button(root, text="Sort Sublists", command=self.on_sort_button_click)
        self.sort_button.pack()
        self.output_label = tk.Label(root, text="Sorted list of lists:")
        self.output_label.pack()
        self.output_text = tk.Text(root, height=10, width=50, state='disabled')
        self.output_text.pack()
    def on_sort_button_click(self):
        '''
        Handle the sort button click event.
        '''
        try:
            input_data = self.input_text.get("1.0", tk.END).strip()
            list_of_lists = eval(input_data)  # Convert string input to list of lists
            if not isinstance(list_of_lists, list) or not all(isinstance(sublist, list) for sublist in list_of_lists):
                raise ValueError("Input must be a list of lists.")
            sorted_list_of_lists = sort_each_sublist(list_of_lists)
            self.output_text.config(state='normal')
            self.output_text.delete("1.0", tk.END)
            self.output_text.insert(tk.END, str(sorted_list_of_lists))
            self.output_text.config(state='disabled')
        except Exception as e:
            messagebox.showerror("Error", f"Invalid input: {e}")
if __name__ == "__main__":
    root = tk.Tk()
    app = SortApp(root)
    root.mainloop()

import tkinter as tk
from tkinter import messagebox
from utils import sort_each_sublist
class SortApp:
    def __init__(self, root):
        '''
        Initialize the GUI components.
        '''
        self.root = root
        self.root.title("Sort Sublists Application")
        self.input_label = tk.Label(root, text="Enter list of lists:")
        self.input_label.pack()
        self.input_text = tk.Text(root, height=10, width=50)
        self.input_text.pack()
        self.sort_button = tk.Button(root, text="Sort Sublists", command=self.on_sort_button_click)
        self.sort_button.pack()
        self.output_label = tk.Label(root, text="Sorted list of lists:")
        self.output_label.pack()
        self.output_text = tk.Text(root, height=10, width=50, state='disabled')
        self.output_text.pack()
    def on_sort_button_click(self):
        '''
        Handle the sort button click event.
        '''
        try:
            input_data = self.input_text.get("1.0", tk.END).strip()
            list_of_lists = eval(input_data)  # Convert string input to list of lists
            if not isinstance(list_of_lists, list) or not all(isinstance(sublist, list) for sublist in list_of_lists):
                raise ValueError("Input must be a list of lists.")
            sorted_list_of_lists = sort_each_sublist(list_of_lists)
            self.output_text.config(state='normal')
            self.output_text.delete("1.0", tk.END)
            self.output_text.insert(tk.END, str(sorted_list_of_lists))
            self.output_text.config(state='disabled')
        except Exception as e:
            messagebox.showerror("Error", f"Invalid input: {e}")
if __name__ == "__main__":
    root = tk.Tk()
    app = SortApp(root)
    root.mainloop()

'''
Utility functions for sorting sublists.

def sort_each_sublist(list_of_lists):
    '''
    Sort each sublist of strings in the given list of lists.
    :param list_of_lists: List of lists containing strings.
    :return: New list of lists with each sublist sorted.
    '''
    return [sorted(sublist) for sublist in list_of_lists]

import tkinter as tk
from tkinter import messagebox
from utils import sort_each_sublist
class SortApp:
    def __init__(self, root):
        '''
        Initialize the GUI components.
        '''
        self.root = root
        self.root.title("Sort Sublists Application")
        self.input_label = tk.Label(root, text="Enter list of lists:")
        self.input_label.pack()
        self.input_text = tk.Text(root, height=10, width=50)
        self.input_text.pack()
        self.sort_button = tk.Button(root, text="Sort Sublists", command=self.on_sort_button_click)
        self.sort_button.pack()
        self.output_label = tk.Label(root, text="Sorted list of lists:")
        self.output_label.pack()
        self.output_text = tk.Text(root, height=10, width=50, state='disabled')
        self.output_text.pack()
    def on_sort_button_click(self):
        '''
        Handle the sort button click event.
        '''
        try:
            input_data = self.input_text.get("1.0", tk.END).strip()
            list_of_lists = eval(input_data)  # Convert string input to list of lists
            if not isinstance(list_of_lists, list) or not all(isinstance(sublist, list) for sublist in list_of_lists):
                raise ValueError("Input must be a list of lists.")
            sorted_list_of_lists = sort_each_sublist(list_of_lists)
            self.output_text.config(state='normal')
            self.output_text.delete("1.0", tk.END)
            self.output_text.insert(tk.END, str(sorted_list_of_lists))
            self.output_text.config(state='disabled')
        except Exception as e:
            messagebox.showerror("Error", f"Invalid input: {e}")
if __name__ == "__main__":
    root = tk.Tk()
    app = SortApp(root)
    root.mainloop()

    :param list_of_lists: List of lists containing strings.
