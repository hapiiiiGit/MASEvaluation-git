   - **Methods**:
     - `__init__`: Initialize the GUI components.
     - `sort_list`: Method to sort the list entered by the user.
     - `display_sorted_list`: Method to display the sorted list in the GUI.

```python
'''

main.py

'''

import tkinter as tk
from tkinter import messagebox

def sort_function(input_list):
    """
    Sorts a list of numbers in ascending order.
    
    Parameters:
    input_list (list): A list of numbers to be sorted.
    
    Returns:
    list: A sorted list of numbers.
    """
    return sorted(input_list)

class SortApp:
    """
    A GUI application for sorting a list of numbers.
    """
    
    def __init__(self, root):
        """
        Initializes the application with GUI components.
        
        Parameters:
        root (tk.Tk): The root window of the application.
        """
        self.root = root
        self.root.title("List Sorter")
        
        self.label = tk.Label(root, text="Enter numbers separated by commas:")
        self.label.pack()
        
        self.entry = tk.Entry(root, width=50)
        self.entry.pack()
        
        self.sort_button = tk.Button(root, text="Sort", command=self.sort_list)
        self.sort_button.pack()
        
        self.result_label = tk.Label(root, text="Sorted List:")
        self.result_label.pack()
        
        self.result_display = tk.Label(root, text="", fg="blue")
        self.result_display.pack()
    
    def sort_list(self):
        """
        Retrieves the list from the entry, sorts it, and displays the result.
        """
        try:
            # Get input from entry
            input_text = self.entry.get()
            # Convert input text to a list of numbers
            input_list = list(map(float, input_text.split(',')))
            # Sort the list
            sorted_list = sort_function(input_list)
            # Display the sorted list
            self.display_sorted_list(sorted_list)
        except ValueError:
            messagebox.showerror("Invalid input", "Please enter a valid list of numbers separated by commas.")
    
    def display_sorted_list(self, sorted_list):
        """
        Displays the sorted list in the GUI.
        
        Parameters:
        sorted_list (list): The sorted list of numbers.
        """
        self.result_display.config(text=str(sorted_list))

if __name__ == "__main__":
    root = tk.Tk()
    app = SortApp(root)
    root.mainloop()
```

### Explanation

- **sort_function**: This function takes a list of numbers and returns it sorted in ascending order using Python's built-in `sorted` function.
  
- **SortApp Class**: This class sets up the GUI using `tkinter`. It includes an entry widget for user input, a button to trigger the sorting, and labels to display the input prompt and the sorted result.

- **sort_list Method**: This method retrieves the input from the entry widget, converts it to a list of numbers, sorts it using `sort_function`, and then displays the sorted list. It also handles invalid input by showing an error message.

- **display_sorted_list Method**: This method updates the label to show the sorted list.

'''

main.py

import tkinter as tk
from tkinter import messagebox

def sort_function(input_list):
    """
    Sorts a list of numbers in ascending order.
    
    Parameters:
    input_list (list): A list of numbers to be sorted.
    
    Returns:
    list: A sorted list of numbers.
    """
    return sorted(input_list)

class SortApp:
    """
    A GUI application for sorting a list of numbers.
    """
    
    def __init__(self, root):
        """
        Initializes the application with GUI components.
        
        Parameters:
        root (tk.Tk): The root window of the application.
        """
        self.root = root
        self.root.title("List Sorter")
        
        self.label = tk.Label(root, text="Enter numbers separated by commas:")
        self.label.pack()
        
        self.entry = tk.Entry(root, width=50)
        self.entry.pack()
        
        self.sort_button = tk.Button(root, text="Sort", command=self.sort_list)
        self.sort_button.pack()
        
        self.result_label = tk.Label(root, text="Sorted List:")
        self.result_label.pack()
        
        self.result_display = tk.Label(root, text="", fg="blue")
        self.result_display.pack()
    
    def sort_list(self):
        """
        Retrieves the list from the entry, sorts it, and displays the result.
        """
        try:
            # Get input from entry
            input_text = self.entry.get()
            # Convert input text to a list of numbers
            input_list = list(map(float, input_text.split(',')))
            # Sort the list
            sorted_list = sort_function(input_list)
            # Display the sorted list
            self.display_sorted_list(sorted_list)
        except ValueError:
            messagebox.showerror("Invalid input", "Please enter a valid list of numbers separated by commas.")
    
    def display_sorted_list(self, sorted_list):
        """
        Displays the sorted list in the GUI.
        
        Parameters:
        sorted_list (list): The sorted list of numbers.
        """
        self.result_display.config(text=str(sorted_list))

if __name__ == "__main__":
    root = tk.Tk()
    app = SortApp(root)
    root.mainloop()

- **sort_list Method**: This method retrieves the input from the entry widget, converts it to a list of numbers, sorts it using `sort_function`, and then displays the sorted list. It also handles invalid input by showing an error message.

from tkinter import messagebox

def sort_function(input_list):
    """
    Sorts a list of numbers in ascending order.
    Parameters:
    input_list (list): A list of numbers to be sorted.
    Returns:
    list: A sorted list of numbers.
    """

class SortApp:
    """
    A GUI application for sorting a list of numbers.
    """

        Initializes the application with GUI components.
        Parameters:
        root (tk.Tk): The root window of the application.

        """
        Retrieves the list from the entry, sorts it, and displays the result.

            # Get input from entry

            # Convert input text to a list of numbers

import tkinter as tk

def sort_function(input_list):
    """

class SortApp:
    """

    """
    def __init__(self, root):

        Initializes the application with GUI components.

        self.sort_button = tk.Button(root, text="Sort", command=self.sort_list, font=("Arial", 10), bg="lightblue")
