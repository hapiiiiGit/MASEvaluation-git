'''

This file contains the main application class for the Snake to Camel Case Converter.
It sets up the GUI and handles user interactions.

'''

import tkinter as tk
from tkinter import messagebox
from utils import convert_snake_to_camel_case


class SnakeToCamelApp:
    def __init__(self, root):
        '''
        Initialize the application and set up the GUI components.
        '''
        self.root = root
        self.root.title("Snake to Camel Case Converter")
        self.root.geometry("400x200")

        # Input label and entry
        self.input_label = tk.Label(root, text="Enter Snake Case String:")
        self.input_label.pack(pady=10)

        self.input_entry = tk.Entry(root, width=40)
        self.input_entry.pack(pady=10)

        # Convert button
        self.convert_button = tk.Button(root, text="Convert", command=self.on_convert_button_click)
        self.convert_button.pack(pady=10)

        # Output label
        self.output_label = tk.Label(root, text="Camel Case String:")
        self.output_label.pack(pady=10)

        self.output_entry = tk.Entry(root, width=40, state="readonly")
        self.output_entry.pack(pady=10)

    def on_convert_button_click(self):
        '''
        Handle the button click event to perform the conversion.
        '''
        snake_case_string = self.input_entry.get()
        if not snake_case_string:
            messagebox.showerror("Error", "Please enter a snake case string.")
            return

        try:
            camel_case_string = convert_snake_to_camel_case(snake_case_string)
            self.output_entry.config(state="normal")
            self.output_entry.delete(0, tk.END)
            self.output_entry.insert(0, camel_case_string)

def convert_snake_to_camel_case(snake_case_string):
    '''
    Convert a snake case string to camel case string.
    Args:
        snake_case_string (str): The snake case string to convert.
    Returns:
        str: The converted camel case string.
    '''
    words = snake_case_string.split('_')
    camel_case_string = words[0] + ''.join(word.capitalize() for word in words[1:])
    return camel_case_string

def convert_snake_to_camel_case(snake_case_string):
    '''
    Convert a snake case string to camel case string.
    Args:
        snake_case_string (str): The snake case string to convert.
    Returns:
        str: The converted camel case string.
    '''
    if not isinstance(snake_case_string, str):
        raise ValueError("Input must be a string.")
    if '_' not in snake_case_string:
        return snake_case_string  # Return as is if no underscores are present
    words = snake_case_string.split('_')
    camel_case_string = words[0] + ''.join(word.capitalize() for word in words[1:])
    return camel_case_string

        Initialize the application and set up the GUI components.

        self.root.title("Snake to Camel Case Converter")

        # Input label and entry

```python
'''
This file contains utility functions for the Snake to Camel Case Converter application.
'''
def convert_snake_to_camel_case(snake_case_string):
    '''
    Convert a snake case string to camel case string.
    Args:
        snake_case_string (str): The snake case string to convert.
    Returns:
        str: The converted camel case string.
    '''
    if not isinstance(snake_case_string, str):
        raise ValueError("Input must be a string.")
    if '_' not in snake_case_string:
        return snake_case_string  # Return as is if no underscores are present
    words = snake_case_string.split('_')
    camel_case_string = words[0] + ''.join(word.capitalize() for word in words[1:])
    return camel_case_string

import tkinter as tk
from tkinter import messagebox
from utils import convert_snake_to_camel_case
class SnakeToCamelApp:
    '''
    A GUI application for converting snake case strings to camel case strings.
    '''
    def __init__(self, root):
        '''
        Initialize the application and set up the GUI components.
        '''
        self.root = root
        self.root.title("Snake to Camel Case Converter")
        self.root.geometry("400x250")
        # Input label and entry
        self.input_label = tk.Label(root, text="Enter Snake Case String:")
        self.input_label.pack(pady=10)
        self.input_entry = tk.Entry(root, width=40)
        self.input_entry.pack(pady=10)
        # Convert button
        self.convert_button = tk.Button(root, text="Convert", command=self.on_convert_button_click)
        self.convert_button.pack(pady=10)
        # Output label
        self.output_label = tk.Label(root, text="Camel Case String:")
        self.output_label.pack(pady=10)
        self.output_entry = tk.Entry(root, width=40, state="readonly")
        self.output_entry.pack(pady=10)
        # Clear button
        self.clear_button = tk.Button(root, text="Clear", command=self.on_clear_button_click)
        self.clear_button.pack(pady=10)
    def on_convert_button_click(self):
        '''
        Handle the button click event to perform the conversion.
        '''
        snake_case_string = self.input_entry.get()
        if not snake_case_string:
            messagebox.showerror("Error", "Please enter a snake case string.")
            return
        try:
            camel_case_string = convert_snake_to_camel_case(snake_case_string)
            self.output_entry.config(state="normal")
            self.output_entry.delete(0, tk.END)
            self.output_entry.insert(0, camel_case_string)
            self.output_entry.config(state="readonly")
        except Exception as e:
            messagebox.showerror("Error", f"An error occurred: {e}")
    def on_clear_button_click(self):
        '''
        Handle the button click event to clear the input and output fields.
        '''
        self.input_entry.delete(0, tk.END)
        self.output_entry.config(state="normal")
        self.output_entry.delete(0, tk.END)
        self.output_entry.config(state="readonly")
if __name__ == "__main__":
    root = tk.Tk()
    app = SnakeToCamelApp(root)
    root.mainloop()

'''
This file contains utility functions for the Snake to Camel Case Converter application.
'''
def convert_snake_to_camel_case(snake_case_string):
    '''
    Convert a snake case string to camel case string.
    Args:
        snake_case_string (str): The snake case string to convert.
    Returns:
        str: The converted camel case string.
    '''
    if not isinstance(snake_case_string, str):
        raise ValueError("Input must be a string.")
    if '_' not in snake_case_string:
        return snake_case_string  # Return as is if no underscores are present
    words = snake_case_string.split('_')
    camel_case_string = words[0] + ''.join(word.capitalize() for word in words[1:])
    return camel_case_string
