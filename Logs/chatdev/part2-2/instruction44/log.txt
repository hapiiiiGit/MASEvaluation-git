import tkinter as tk

def sum_of_even_powers(n):
    """
    Calculate the sum of the first n even natural numbers raised to the fifth power.
    
    Parameters:
    n (int): The number of even natural numbers to consider.
    
    Returns:
    int: The calculated sum.
    """
    sum_even_powers = sum((2 * i) ** 5 for i in range(1, n + 1))

import tkinter as tk

def sum_of_even_powers(n):
    """
    Calculate the sum of the first n even natural numbers raised to the fifth power.
    Parameters:
    n (int): The number of even natural numbers to consider.
    Returns:
    int: The calculated sum.
    """
    sum_even_powers = sum((2 * i) ** 5 for i in range(1, n + 1))
    return sum_even_powers
class EvenPowerSumApp:
    """
    A class to create a GUI application for calculating the sum of the first n even natural numbers raised to the fifth power.
    """

        """
        Initialize the GUI components.

        self.label = tk.Label(master, text="Enter the number of even natural numbers:")

        self.calculate_button = tk.Button(master, text="Calculate", command=self.calculate_sum)

        """
        Calculate the sum and display the result.

        try:

        Display the result in the GUI.

import tkinter as tk
from tkinter import messagebox
def sum_of_even_powers(n):
    """
    Calculate the sum of the first n even natural numbers raised to the fifth power.
    Parameters:
    n (int): The number of even natural numbers to consider.
    Returns:
    int: The calculated sum.
    """
    sum_even_powers = sum((2 * i) ** 5 for i in range(1, n + 1))
    return sum_even_powers
class EvenPowerSumApp:
    """
    A class to create a GUI application for calculating the sum of the first n even natural numbers raised to the fifth power.
    """
    def __init__(self, master):
        """
        Initialize the GUI components.
        Parameters:
        master (tk.Tk): The root window.
        """
        self.master = master
        master.title("Even Power Sum Calculator")
        self.label = tk.Label(master, text="Enter the number of even natural numbers:")
        self.label.pack()
        self.entry = tk.Entry(master)
        self.entry.pack()
        self.calculate_button = tk.Button(master, text="Calculate", command=self.calculate_sum)
        self.calculate_button.pack()
        self.result_label = tk.Label(master, text="")
        self.result_label.pack()
    def calculate_sum(self):
        """
        Calculate the sum and display the result.
        """
        try:
            n = int(self.entry.get())
            if n < 1:
                raise ValueError("The number must be a positive integer.")
            result = sum_of_even_powers(n)
            self.display_result(n, result)
        except ValueError as e:
            messagebox.showerror("Invalid input", str(e))
    def display_result(self, n, result):
        """
        Display the result in the GUI.
        Parameters:
        n (int): The number of even natural numbers considered.
        result (int): The calculated sum to display.
        """
        self.result_label.config(text=f"Sum of the first {n} even natural numbers raised to the fifth power is: {result}")
if __name__ == "__main__":
    root = tk.Tk()
    app = EvenPowerSumApp(root)
    root.mainloop()
