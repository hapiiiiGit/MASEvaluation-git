'''

Main application file for the Longest Sublist Length Finder with GUI.

'''

import tkinter as tk
from tkinter import messagebox
from utils import find_longest_sublists_length

class LongestSublistApp:
    def __init__(self, root):
        self.root = root
        self.root.title("Longest Sublist Length Finder")

        self.label = tk.Label(root, text="Enter list of lists (e.g., [[1,2,3],[4,5],[6,7,8,9]]):")
        self.label.pack()

        self.entry = tk.Entry(root, width=50)
        self.entry.pack()

        self.calculate_button = tk.Button(root, text="Calculate", command=self.on_calculate)
        self.calculate_button.pack()

        self.result_label = tk.Label(root, text="")
        self.result_label.pack()

    def on_calculate(self):
        try:
            lists = eval(self.entry.get())
            if not isinstance(lists, list) or not all(isinstance(sublist, list) for sublist in lists):
                raise ValueError("Input must be a list of lists.")
            length = find_longest_sublists_length(lists)
            self.result_label.config(text=f"Length of the longest sublist: {length}")
        except Exception as e:
            messagebox.showerror("Error", f"Invalid input: {e}")

if __name__ == "__main__":
    root = tk.Tk()
    app = LongestSublistApp(root)
    root.mainloop()
```

'''

Utility functions for the Longest Sublist Length Finder.

def find_longest_sublists_length(lists):
    '''
    Finds the length of the longest sublist in a list of lists.

    Parameters:
    lists (list of lists): The input list of lists.

    Returns:
    int: The length of the longest sublist.
    '''
    if not lists:
        return 0
    return max(len(sublist) for sublist in lists)

- **Error Handling**: The application includes error handling to ensure users input valid data formats.
- **Simple and Efficient**: The underlying logic efficiently computes the length of the longest sublist using Python's built-in functions.

## Quick Install

No external dependencies are required for this project. Ensure you have Python installed on your system.

## How to Use

### Step 1: Clone the Repository

Clone the repository to your local machine using the following command:

```bash
git clone <repository-url>
```

### Step 2: Navigate to the Project Directory

Navigate to the directory where the project files are located:

```bash
cd <project-directory>
```

### Step 3: Run the Application

Execute the main application file to start the GUI:

```bash
python main.py
```

### Step 4: Input Data

- Enter your list of lists in the provided text entry field. Ensure the format is correct, e.g., `[[1,2,3],[4,5],[6,7,8,9]]`.
- Click the "Calculate" button to compute the length of the longest sublist.

### Step 5: View Results

- The application will display the length of the longest sublist below the input field.
- If the input format is incorrect, an error message will be shown.
