'''

This module contains the main application code for checking if a given tuple contains no duplicates using a GUI.

'''

import tkinter as tk
from tkinter import messagebox

def check_no_duplicates(tuple_data):
    '''
    Check if the given tuple contains no duplicates.

    Parameters:
    tuple_data (tuple): The tuple to check for duplicates.

    Returns:
    bool: True if no duplicates are found, False otherwise.
    '''
    return len(tuple_data) == len(set(tuple_data))

class App:
    '''
    A class to create a GUI application for checking duplicates in a tuple.
    '''

    def __init__(self, root):
        '''
        Initialize the application with the main window.

        Parameters:
        root (Tk): The root window of the application.
        '''
        self.root = root
        self.root.title("Tuple Duplicate Checker")

        self.label = tk.Label(root, text="Enter a tuple (e.g., 1, 2, 3):")
        self.label.pack()

        self.entry = tk.Entry(root)
        self.entry.pack()

        self.check_button = tk.Button(root, text="Check", command=self.check_duplicates)
        self.check_button.pack()

        self.result_label = tk.Label(root, text="")
        self.result_label.pack()

    def check_duplicates(self):
        '''
        Check for duplicates in the entered tuple and display the result.
        '''
        try:
            tuple_data = tuple(map(int, self.entry.get().split(',')))
            if check_no_duplicates(tuple_data):
                self.result_label.config(text="No duplicates found.")
            else:
                self.result_label.config(text="Duplicates found.")
        except ValueError:
            messagebox.showerror("Invalid Input", "Please enter a valid tuple.")

if __name__ == "__main__":
    root = tk.Tk()
    app = App(root)
    root.mainloop()
```

This code provides a complete solution with a GUI for checking if a given tuple contains no duplicates. The user can input a tuple in the specified format, and the application will display whether duplicates are present or not.

import tkinter as tk
from tkinter import messagebox

def check_no_duplicates(tuple_data):
    '''
    Check if the given tuple contains no duplicates.

    Parameters:
    tuple_data (tuple): The tuple to check for duplicates.

    Returns:
    bool: True if no duplicates are found, False otherwise.
    '''
    return len(tuple_data) == len(set(tuple_data))

class App:
    '''
    A class to create a GUI application for checking duplicates in a tuple.
    '''

    def __init__(self, root):
        '''
        Initialize the application with the main window.

        Parameters:
        root (Tk): The root window of the application.
        '''
        self.root = root
        self.root.title("Tuple Duplicate Checker")

        self.label = tk.Label(root, text="Enter a tuple (e.g., 1, 2, 3):")
        self.label.pack()

        self.entry = tk.Entry(root)
        self.entry.pack()

        self.check_button = tk.Button(root, text="Check", command=self.check_duplicates)
        self.check_button.pack()

        self.result_label = tk.Label(root, text="")
        self.result_label.pack()

    def check_duplicates(self):
        '''
        Check for duplicates in the entered tuple and display the result.
        '''
        try:
            tuple_data = tuple(map(int, self.entry.get().split(',')))
            if check_no_duplicates(tuple_data):
                self.result_label.config(text="No duplicates found.")
            else:
                self.result_label.config(text="Duplicates found.")

            messagebox.showerror("Invalid Input", "Please enter a valid tuple.")

if __name__ == "__main__":
    root = tk.Tk()
    app = App(root)

'''  
This module contains the main application code for checking if a given tuple contains no duplicates using a GUI.  
'''  
import tkinter as tk  
from tkinter import messagebox  
def check_no_duplicates(tuple_data):  
    '''  
    Check if the given tuple contains no duplicates.  
    Parameters:  
    tuple_data (tuple): The tuple to check for duplicates.  
    Returns:  
    bool: True if no duplicates are found, False otherwise.  
    '''  
    return len(tuple_data) == len(set(tuple_data))  
class App:  
    '''  
    A class to create a GUI application for checking duplicates in a tuple.  
    '''  
    def __init__(self, root):  
        '''  
        Initialize the application with the main window.  
        Parameters:  
        root (Tk): The root window of the application.  
        '''  
        self.root = root  
        self.root.title("Tuple Duplicate Checker")  
        self.label = tk.Label(root, text="Enter a tuple (e.g., 1, 2, 3):")  
        self.label.pack()  
        self.entry = tk.Entry(root)  
        self.entry.pack()  
        self.check_button = tk.Button(root, text="Check", command=self.check_duplicates)  
        self.check_button.pack()  
        self.result_label = tk.Label(root, text="")  
        self.result_label.pack()  
    def check_duplicates(self):  
        '''  
        Check for duplicates in the entered tuple and display the result.  
        '''  
        try:  
            # Convert the input string into a tuple  
            tuple_data = tuple(map(int, self.entry.get().split(',')))  
            # Check for duplicates and display the result  
            if check_no_duplicates(tuple_data):  
                self.result_label.config(text="No duplicates found.")  
            else:  
                self.result_label.config(text="Duplicates found.")  
        except ValueError:  
            # Handle invalid input and show an error message  
            messagebox.showerror("Invalid Input", "Please enter a valid tuple.")  
if __name__ == "__main__":  
    root = tk.Tk()  
    app = App(root)  
    root.mainloop()

# Tuple Duplicate Checker

A Python-based GUI application to check if a given tuple contains duplicates.

## 🤔 What is this?

The Tuple Duplicate Checker is a simple and intuitive application that allows users to input a tuple and check whether it contains duplicate values. It provides a graphical user interface (GUI) built using Python's `tkinter` library, making it easy for users to interact with the application.

### Main Features:
- **Duplicate Detection**: Quickly checks if the input tuple contains any duplicate values.
- **User-Friendly Interface**: A clean and simple GUI for entering tuples and viewing results.
- **Error Handling**: Displays error messages for invalid inputs.

---

   ```

     1, 2, 3
