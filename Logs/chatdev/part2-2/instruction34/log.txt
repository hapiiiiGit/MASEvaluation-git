'''

This file initializes the GUI application for finding the nth Newman–Shanks–Williams prime number.
It connects user input to the computation logic and displays the result.

'''

import tkinter as tk
from tkinter import messagebox
from nsw_primes import generate_nsw_primes

class App:
    def __init__(self, root):
        self.root = root
        self.root.title("NSW Prime Finder")
        
        # Create GUI components
        self.label = tk.Label(root, text="Enter n (to find nth NSW prime):")
        self.label.pack(pady=10)
        
        self.entry = tk.Entry(root)
        self.entry.pack(pady=10)
        
        self.button = tk.Button(root, text="Find NSW Prime", command=self.find_nsw_prime)
        self.button.pack(pady=10)
        
        self.result_label = tk.Label(root, text="", fg="blue")
        self.result_label.pack(pady=10)
    
    def find_nsw_prime(self):
        try:
            n = int(self.entry.get())
            if n <= 0:
                raise ValueError("n must be a positive integer.")
            
            # Compute the nth NSW prime
            nsw_prime = generate_nsw_primes(n)
            self.result_label.config(text=f"The {n}th NSW prime is: {nsw_prime}")
        except ValueError as e:
            messagebox.showerror("Error", str(e))

if __name__ == "__main__":
    root = tk.Tk()
    app = App(root)
    root.mainloop()
```

This file contains the logic for computing Newman–Shanks–Williams prime numbers.
It includes helper functions for prime checking and generating NSW primes.

'''

def is_prime(num):
    """
    Check if a number is prime.
    """
    if num < 2:
        return False
    for i in range(2, int(num**0.5) + 1):
        if num % i == 0:
            return False
    return True

def generate_nsw_primes(n):
    """
    Generate the nth Newman–Shanks–Williams prime number.
    """
    # Initialize the NSW sequence
    nsw_sequence = [1, 1]
    
    # Generate NSW sequence until we find n primes
    primes = []
    index = 2
    while len(primes) < n:
        next_nsw = 2 * nsw_sequence[index - 1] + nsw_sequence[index - 2]
        nsw_sequence.append(next_nsw)
        
        if is_prime(next_nsw):
            primes.append(next_nsw)
        
        index += 1
    
    return primes[-1]
```

'''

It connects user input to the computation logic and displays the result.

import tkinter as tk
from tkinter import messagebox
from nsw_primes import generate_nsw_primes

class App:
    def __init__(self, root):
        self.root = root
        self.root.title("NSW Prime Finder")
        
        # Create GUI components
        self.label = tk.Label(root, text="Enter n (to find nth NSW prime):")
        self.label.pack(pady=10)
        
        self.entry = tk.Entry(root)
        self.entry.pack(pady=10)
        
        self.button = tk.Button(root, text="Find NSW Prime", command=self.find_nsw_prime)
        self.button.pack(pady=10)
        
        self.result_label = tk.Label(root, text="", fg="blue")
        self.result_label.pack(pady=10)
    
    def find_nsw_prime(self):
        try:
            n = int(self.entry.get())
            if n <= 0:
                raise ValueError("n must be a positive integer.")
            
            # Compute the nth NSW prime
            nsw_prime = generate_nsw_primes(n)
            self.result_label.config(text=f"The {n}th NSW prime is: {nsw_prime}")
        except ValueError as e:
            messagebox.showerror("Error", str(e))

if __name__ == "__main__":
    root = tk.Tk()
    app = App(root)
    root.mainloop()

def is_prime(num):
    """
    Check if a number is prime.
    """
    if num < 2:
        return False
    for i in range(2, int(num**0.5) + 1):
        if num % i == 0:
            return False
    return True
def generate_nsw_primes(n):
    """
    Generate the nth Newman–Shanks–Williams prime number.
    """
    # Initialize the NSW sequence
    nsw_sequence = [1, 1]
    # Generate NSW sequence until we find n primes
    primes = []
    index = 2
    while len(primes) < n:
        next_nsw = 2 * nsw_sequence[index - 1] + nsw_sequence[index - 2]
        nsw_sequence.append(next_nsw)
        if is_prime(next_nsw):
            primes.append(next_nsw)
        index += 1
    return primes[-1]

def is_prime(num):
    """
    Check if a number is prime.
    :param num: Integer to check for primality.
    :return: True if num is prime, False otherwise.
    """
    if num < 2:
        return False
    if num == 2:
        return True
    if num % 2 == 0:
        return False
    for i in range(3, int(num**0.5) + 1, 2):
        if num % i == 0:
            return False
    return True

def generate_nsw_primes(n):
    """
    Generate the nth Newman–Shanks–Williams prime number.
    :param n: The position of the NSW prime to find.
    :return: The nth NSW prime number.
    """
    # Initialize the NSW sequence
    nsw_sequence = [1, 1]
    # Generate NSW sequence until we find n primes
    primes = []
    index = 2
    while len(primes) < n:
        next_nsw = 2 * nsw_sequence[index - 1] + nsw_sequence[index - 2]
        nsw_sequence.append(next_nsw)
        if is_prime(next_nsw):
            primes.append(next_nsw)
        index += 1
    return primes[-1]

def is_prime(num):
    """
    Check if a number is prime.
    :param num: Integer to check for primality.
    :return: True if num is prime, False otherwise.
    """
    if num < 2:
        return False
    if num == 2:
        return True
    if num % 2 == 0:
        return False
    for i in range(3, int(num**0.5) + 1, 2):
        if num % i == 0:
            return False
    return True
def generate_nsw_primes(n):
    """
    Generate the nth Newman–Shanks–Williams prime number.
    :param n: The position of the NSW prime to find.
    :return: The nth NSW prime number.
    """
    # Initialize the NSW sequence
    nsw_sequence = [1, 1]
    # Generate NSW sequence until we find n primes
    primes = []
    index = 2
    while len(primes) < n:
        next_nsw = 2 * nsw_sequence[index - 1] + nsw_sequence[index - 2]
        nsw_sequence.append(next_nsw)
        if is_prime(next_nsw):
            primes.append(next_nsw)
        index += 1
    return primes[-1]

def is_prime(num):  
    """  
    Check if a number is prime.  
    :param num: Integer to check for primality.  
    :return: True if num is prime, False otherwise.  
    """  
    if num < 2:  
        return False  
    if num == 2:  
        return True  
    if num % 2 == 0:  
        return False  
    for i in range(3, int(num**0.5) + 1, 2):  
        if num % i == 0:  
            return False  
    return True  

def generate_nsw_primes(n):  
    """  
    Generate the nth Newman–Shanks–Williams prime number.  
    :param n: The position of the NSW prime to find.  
    :return: The nth NSW prime number.  
    """  
    # Initialize the NSW sequence  
    nsw_sequence = [1, 1]  
    # Generate NSW sequence until we find n primes  
    primes = []  
    index = 2  
    while len(primes) < n:  
        next_nsw = 2 * nsw_sequence[index - 1] + nsw_sequence[index - 2]  
        nsw_sequence.append(next_nsw)  
        if is_prime(next_nsw):  
            primes.append(next_nsw)  
        index += 1  
    return primes[-1]

'''  
This file contains the logic for computing Newman–Shanks–Williams prime numbers.  
It includes helper functions for prime checking and generating NSW primes.

def is_prime(num):  
    """  
    Check if a number is prime.  
    :param num: Integer to check for primality.  
    :return: True if num is prime, False otherwise.  
    """  
    if num < 2:  
        return False  
    if num == 2:  
        return True  
    if num % 2 == 0:  
        return False  
    for i in range(3, int(num**0.5) + 1, 2):  
        if num % i == 0:  
            return False  
    return True  
def generate_nsw_primes(n):  
    """  
    Generate the nth Newman–Shanks–Williams prime number.  
    :param n: The position of the NSW prime to find.  
    :return: The nth NSW prime number.  
    """  
    # Initialize the NSW sequence  
    nsw_sequence = [1, 1]  
    # Generate NSW sequence until we find n primes  
    primes = []  
    index = 2  
    while len(primes) < n:  
        next_nsw = 2 * nsw_sequence[index - 1] + nsw_sequence[index - 2]  
        nsw_sequence.append(next_nsw)  
        if is_prime(next_nsw):  
            primes.append(next_nsw)  
        index += 1  
    return primes[-1]

def is_prime(num):
    """
    Check if a number is prime.
    :param num: Integer to check for primality.
    :return: True if num is prime, False otherwise.
    """
    if num < 2:
        return False
    if num == 2:
        return True
    if num % 2 == 0:
        return False
    for i in range(3, int(num**0.5) + 1, 2):
        if num % i == 0:
            return False
    return True

def generate_nsw_primes(n):
    """
    Generate the nth Newman–Shanks–Williams prime number.
    :param n: The position of the NSW prime to find.
    :return: The nth NSW prime number.
    """
    # Initialize the NSW sequence
    nsw_sequence = [1, 1]
    # Generate NSW sequence until we find n primes
    primes = []
    index = 2
    while len(primes) < n:
        next_nsw = 2 * nsw_sequence[index - 1] + nsw_sequence[index - 2]
        nsw_sequence.append(next_nsw)
        if is_prime(next_nsw):
            primes.append(next_nsw)
        index += 1
    return primes[-1]

This file contains the logic for computing Newman–Shanks–Williams prime numbers.
It includes helper functions for prime checking and generating NSW primes.

def is_prime(num):
    """
    Check if a number is prime.
    :param num: Integer to check for primality.
    :return: True if num is prime, False otherwise.
    """
    if num < 2:
        return False
    if num == 2:
        return True
    if num % 2 == 0:
        return False
    for i in range(3, int(num**0.5) + 1, 2):
        if num % i == 0:
            return False
    return True
def generate_nsw_primes(n):
    """
    Generate the nth Newman–Shanks–Williams prime number.
    :param n: The position of the NSW prime to find.
    :return: The nth NSW prime number.
    """
    # Initialize the NSW sequence
    nsw_sequence = [1, 1]
    # Generate NSW sequence until we find n primes
    primes = []
    index = 2
    while len(primes) < n:
        next_nsw = 2 * nsw_sequence[index - 1] + nsw_sequence[index - 2]
        nsw_sequence.append(next_nsw)
        if is_prime(next_nsw):
            primes.append(next_nsw)
        index += 1
    return primes[-1]

# NSW Prime Finder Application

Welcome to the NSW Prime Finder Application! This software allows users to find the nth Newman–Shanks–Williams (NSW) prime number using a user-friendly graphical interface. Below is a detailed guide on how to install, use, and understand the application.

---

## 🛠 Main Functions of the Software

### 1. **Find nth NSW Prime**
The application computes the nth Newman–Shanks–Williams prime number based on user input. NSW primes are a special sequence of prime numbers derived from the Newman–Shanks–Williams sequence.

### 2. **User-Friendly GUI**
The software provides a graphical user interface (GUI) for easy interaction. Users can input the desired position `n` and receive the corresponding NSW prime number.

---

## 📦 Installation Guide

### 1. **Environment Setup**
Ensure you have Python installed on your system. The application is compatible with Python 3.6 or later.

### 2. **Install Required Dependencies**
The application uses the `tkinter` library for GUI development, which is included in the standard Python library. No additional dependencies are required.

### 3. **Download the Application**
Download the following files:
- `main.py`: Contains the GUI logic.
- `nsw_primes.py`: Contains the computation logic for NSW primes.

Place both files in the same directory.

---

## 🚀 How to Use the Application

### Step 1: Run the Application
Open a terminal or command prompt and navigate to the directory containing the files. Run the following command:
```
python main.py
```

### Step 2: Enter the Desired Position
Once the application window opens:
1. Enter a positive integer `n` in the input field.
2. Click the "Find NSW Prime" button.

### Step 3: View the Result
The application will compute and display the nth NSW prime number below the input field. If an error occurs (e.g., invalid input), an error message will be displayed.

---

## 🧠 How It Works

### 1. **NSW Sequence Generation**
The Newman–Shanks–Williams sequence is generated using the formula:
```
S(n) = 2 * S(n-1) + S(n-2)
```
The sequence starts with `S(0) = 1` and `S(1) = 1`.

### 2. **Prime Checking**
Each number in the NSW sequence is checked for primality using the `is_prime` function. Only prime numbers are added to the list of NSW primes.

### 3. **GUI Interaction**
The GUI connects user input to the computation logic. When the user clicks the "Find NSW Prime" button, the application computes the result and displays it.

---

## 🛡 Error Handling

The application includes robust error handling:
- **Invalid Input:** If the user enters a non-integer or a negative number, an error message is displayed.
- **Unexpected Errors:** Any unexpected errors are caught and displayed to the user.

---

## 📖 Example Usage

### Example 1: Finding the 5th NSW Prime
1. Run the application.
2. Enter `5` in the input field.
3. Click "Find NSW Prime."
4. The result, `13`, will be displayed.

### Example 2: Handling Invalid Input
1. Run the application.
2. Enter `-3` in the input field.
3. Click "Find NSW Prime."
4. An error message will appear: "Please enter a positive integer greater than 0."

---

## 💡 Additional Notes

- The application is designed for educational and computational purposes.
- NSW primes are a subset of prime numbers with unique properties. For more information, refer to mathematical resources on the Newman–Shanks–Williams sequence.

---

Thank you for using the NSW Prime Finder Application! If you encounter any issues or have suggestions for improvement, feel free to reach out to our support team.
