'''

This script provides a graphical user interface (GUI) to calculate the volume of a cone.
The user inputs the radius and height of the cone, and the program calculates and displays the volume.

'''

import tkinter as tk
from tkinter import messagebox
import math


def calculate_volume(radius, height):
    '''
    Calculate the volume of a cone given its radius and height.

    Parameters:
        radius (float): The radius of the cone's base.
        height (float): The height of the cone.

    Returns:
        float: The volume of the cone.
    '''
    return (1 / 3) * math.pi * radius**2 * height


def on_calculate():
    '''
    Event handler for the "Calculate" button. Reads user input, validates it, calculates the volume,
    and displays the result in the GUI.

    except ValueError as e:

'''

This script provides a graphical user interface (GUI) to calculate the volume of a cone.
The user inputs the radius and height of the cone, and the program calculates and displays the volume.

'''

import tkinter as tk
from tkinter import messagebox
import math


def calculate_volume(radius, height):
    '''
    Calculate the volume of a cone given its radius and height.

    Parameters:
        radius (float): The radius of the cone's base.
        height (float): The height of the cone.

    Returns:
        float: The volume of the cone.
    '''
    return (1 / 3) * math.pi * radius**2 * height


def on_calculate():
    '''
    Event handler for the "Calculate" button. Reads user input, validates it, calculates the volume,
    and displays the result in the GUI.
    '''

        # Get user input

        # Calculate volume
        volume = calculate_volume(radius, height)

        # Display result
        label_result.config(text=f"Volume: {volume:.2f} cubic units")

    except ValueError as e:
        # Show error message
        messagebox.showerror("Invalid Input", str(e))

'''

import tkinter as tk
from tkinter import messagebox
import math
def calculate_volume(radius, height):
    '''
    Calculate the volume of a cone given its radius and height.
    Parameters:
        radius (float): The radius of the cone's base.
        height (float): The height of the cone.
    Returns:
        float: The volume of the cone.
    '''
    return (1 / 3) * math.pi * radius**2 * height
def on_calculate():
    '''
    Event handler for the "Calculate" button. Reads user input, validates it, calculates the volume,
    and displays the result in the GUI.

        # Validate input

        # Calculate volume
        volume = calculate_volume(radius, height)

        # Show error message

# Create the main application window
