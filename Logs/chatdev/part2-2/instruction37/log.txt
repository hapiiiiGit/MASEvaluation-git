```python
'''

Main application file for calculating a dog's age in dog years using a GUI.

'''

import tkinter as tk

def calculate_dog_years(human_years):
    '''
    Convert human years to dog years.
    
    Parameters:
    human_years (int): The age of the dog in human years.

    Returns:
    float: The age of the dog in dog years.
    '''
    if human_years <= 2:
        return human_years * 10.5
    else:
        return 21 + (human_years - 2) * 4

class DogAgeApp:
    '''
    A class to create a GUI application for calculating dog age in dog years.
    '''
    def __init__(self, master):
        '''
        Initialize the application with the main window and widgets.
        
        Parameters:
        master (tk.Tk): The main window of the application.
        '''
        self.master = master
        master.title("Dog Age Calculator")

        self.label = tk.Label(master, text="Enter dog's age in human years:")
        self.label.pack()

        self.entry = tk.Entry(master)
        self.entry.pack()

        self.calculate_button = tk.Button(master, text="Calculate", command=self.calculate_dog_years)
        self.calculate_button.pack()

        self.result_label = tk.Label(master, text="")
        self.result_label.pack()

    def calculate_dog_years(self):
        '''
        Calculate the dog's age in dog years and update the result label.
        '''
        try:
            human_years = int(self.entry.get())
            dog_years = calculate_dog_years(human_years)
            self.result_label.config(text=f"Dog's age in dog years: {dog_years}")
        except ValueError:
            self.result_label.config(text="Please enter a valid number.")

if __name__ == "__main__":
    root = tk.Tk()
    app = DogAgeApp(root)
    root.mainloop()
```

Main application file for calculating a dog's age in dog years using a GUI.

'''

import tkinter as tk

def calculate_dog_years(human_years):
    '''
    Convert human years to dog years.
    
    Parameters:
    human_years (int): The age of the dog in human years.

    Returns:
    float: The age of the dog in dog years.
    '''
    if human_years <= 2:
        return human_years * 10.5
    else:
        return 21 + (human_years - 2) * 4

class DogAgeApp:
    '''
    A class to create a GUI application for calculating dog age in dog years.
    '''
    def __init__(self, master):
        '''
        Initialize the application with the main window and widgets.
        
        Parameters:
        master (tk.Tk): The main window of the application.
        '''
        self.master = master
        master.title("Dog Age Calculator")

        self.label = tk.Label(master, text="Enter dog's age in human years:")
        self.label.pack()

        self.entry = tk.Entry(master)
        self.entry.pack()

        self.calculate_button = tk.Button(master, text="Calculate", command=self.calculate_dog_years)
        self.calculate_button.pack()

        self.result_label = tk.Label(master, text="")
        self.result_label.pack()

    def calculate_dog_years(self):
        '''
        Calculate the dog's age in dog years and update the result label.
        '''
        try:
            human_years = int(self.entry.get())
            dog_years = calculate_dog_years(human_years)
            self.result_label.config(text=f"Dog's age in dog years: {dog_years}")
        except ValueError:
            self.result_label.config(text="Please enter a valid number.")

if __name__ == "__main__":
    root = tk.Tk()
    app = DogAgeApp(root)
    root.mainloop()

'''
Main application file for calculating a dog's age in dog years using a GUI.
'''
import tkinter as tk

def calculate_dog_years(human_years):
    '''
    Convert human years to dog years.
    Parameters:
    human_years (int): The age of the dog in human years.
    Returns:
    float: The age of the dog in dog years.
    '''
    if human_years <= 2:
        return human_years * 10.5
    else:
        return 21 + (human_years - 2) * 4

class DogAgeApp:
    '''
    A class to create a GUI application for calculating dog age in dog years.
    '''
    def __init__(self, master):
        '''
        Initialize the application with the main window and widgets.
        Parameters:
        master (tk.Tk): The main window of the application.
        '''
        self.master = master
        master.title("Dog Age Calculator")
        
        # Label for user input
        self.label = tk.Label(master, text="Enter dog's age in human years:")
        self.label.pack()
        
        # Entry widget for user input
        self.entry = tk.Entry(master)
        self.entry.pack()
        
        # Button to trigger calculation
        self.calculate_button = tk.Button(master, text="Calculate", command=self.calculate_dog_years)
        self.calculate_button.pack()
        
        # Label to display the result
        self.result_label = tk.Label(master, text="")
        self.result_label.pack()

    def calculate_dog_years(self):
        '''
        Calculate the dog's age in dog years and update the result label.
        '''
        try:
            human_years = int(self.entry.get())
            if human_years < 0:
                self.result_label.config(text="Age cannot be negative. Please enter a valid number.")
            else:
                dog_years = calculate_dog_years(human_years)
                self.result_label.config(text=f"Dog's age in dog years: {dog_years}")
        except ValueError:
            self.result_label.config(text="Please enter a valid number.")

if __name__ == "__main__":
    root = tk.Tk()
    app = DogAgeApp(root)
    root.mainloop()
```

import tkinter as tk

def calculate_dog_years(human_years):
    '''
    Convert human years to dog years.
    Parameters:
    human_years (int): The age of the dog in human years.
    Returns:
    float: The age of the dog in dog years.
    '''
    if human_years <= 2:
        return human_years * 10.5
    else:
        return 21 + (human_years - 2) * 4

class DogAgeApp:
    '''
    A class to create a GUI application for calculating dog age in dog years.
    '''
    def __init__(self, master):
        '''
        Initialize the application with the main window and widgets.
        Parameters:
        master (tk.Tk): The main window of the application.
        '''
        self.master = master
        master.title("Dog Age Calculator")
        
        # Label for user input
        self.label = tk.Label(master, text="Enter dog's age in human years:")
        self.label.pack(pady=5)
        
        # Entry widget for user input
        self.entry = tk.Entry(master)
        self.entry.pack(pady=5)
        
        # Button to trigger calculation
        self.calculate_button = tk.Button(master, text="Calculate", command=self.calculate_dog_years)
        self.calculate_button.pack(pady=5)
        
        # Label to display the result
        self.result_label = tk.Label(master, text="", fg="blue")
        self.result_label.pack(pady=5)
        
        # Button to clear input and result
        self.clear_button = tk.Button(master, text="Clear", command=self.clear_fields)
        self.clear_button.pack(pady=5)

    def calculate_dog_years(self):
        '''
        Calculate the dog's age in dog years and update the result label.
        '''
        try:
            human_years = int(self.entry.get())
            if human_years < 0:
                self.result_label.config(text="Age cannot be negative. Please enter a valid number.", fg="red")
            else:
                dog_years = calculate_dog_years(human_years)
                self.result_label.config(text=f"Dog's age in dog years: {dog_years}", fg="blue")
        except ValueError:
            self.result_label.config(text="Please enter a valid number.", fg="red")
    
    def clear_fields(self):
        '''
        Clear the input field and result label.
        '''
        self.entry.delete(0, tk.END)
        self.result_label.config(text="")

if __name__ == "__main__":
    root = tk.Tk()
    app = DogAgeApp(root)
    root.mainloop()

'''  
Main application file for calculating a dog's age in dog years using a GUI.  
'''  
import tkinter as tk  
def calculate_dog_years(human_years):  
    '''  
    Convert human years to dog years.  
    Parameters:  
    human_years (int): The age of the dog in human years.  
    Returns:  
    float: The age of the dog in dog years.  
    '''  
    if human_years <= 2:  
        return human_years * 10.5  
    else:  
        return 21 + (human_years - 2) * 4  
class DogAgeApp:  
    '''  
    A class to create a GUI application for calculating dog age in dog years.  
    '''  
    def __init__(self, master):  
        '''  
        Initialize the application with the main window and widgets.  
        Parameters:  
        master (tk.Tk): The main window of the application.  
        '''  
        self.master = master  
        master.title("Dog Age Calculator")  
        # Label for user input  
        self.label = tk.Label(master, text="Enter dog's age in human years:")  
        self.label.pack(pady=5)  
        # Entry widget for user input  
        self.entry = tk.Entry(master)  
        self.entry.pack(pady=5)  
        # Button to trigger calculation  
        self.calculate_button = tk.Button(master, text="Calculate", command=self.calculate_dog_years)  
        self.calculate_button.pack(pady=5)  
        # Label to display the result  
        self.result_label = tk.Label(master, text="", fg="blue")  
        self.result_label.pack(pady=5)  
        # Button to clear input and result  
        self.clear_button = tk.Button(master, text="Clear", command=self.clear_fields)  
        self.clear_button.pack(pady=5)  
    def calculate_dog_years(self):  
        '''  
        Calculate the dog's age in dog years and update the result label.  
        '''  
        try:  
            human_years = int(self.entry.get())  
            if human_years < 0:  
                self.result_label.config(text="Age cannot be negative. Please enter a valid number.", fg="red")  
            else:  
                dog_years = calculate_dog_years(human_years)  
                self.result_label.config(text=f"Dog's age in dog years: {dog_years}", fg="blue")  
        except ValueError:  
            self.result_label.config(text="Please enter a valid number.", fg="red")  
    def clear_fields(self):  
        '''  
        Clear the input field and result label.  
        '''  
        self.entry.delete(0, tk.END)  
        self.result_label.config(text="")  
if __name__ == "__main__":  
    root = tk.Tk()  
    app = DogAgeApp(root)  
    root.mainloop()

'''  
Main application file for calculating a dog's age in dog years using a GUI.  
'''  
import tkinter as tk  
def calculate_dog_years(human_years):  
    '''  
    Convert human years to dog years.  
    Parameters:  
    human_years (int): The age of the dog in human years.  
    Returns:  
    float: The age of the dog in dog years.  
    '''  
    if human_years <= 2:  
        return human_years * 10.5  
    else:  
        return 21 + (human_years - 2) * 4  
class DogAgeApp:  
    '''  
    A class to create a GUI application for calculating dog age in dog years.  
    '''  
    def __init__(self, master):  
        '''  
        Initialize the application with the main window and widgets.  
        Parameters:  
        master (tk.Tk): The main window of the application.  
        '''  
        self.master = master  
        master.title("Dog Age Calculator")  
        # Label for user input  
        self.label = tk.Label(master, text="Enter dog's age in human years:")  
        self.label.pack(pady=5)  
        # Entry widget for user input  
        self.entry = tk.Entry(master)  
        self.entry.pack(pady=5)  
        # Button to trigger calculation  
        self.calculate_button = tk.Button(master, text="Calculate", command=self.calculate_dog_years)  
        self.calculate_button.pack(pady=5)  
        # Label to display the result  
        self.result_label = tk.Label(master, text="", fg="blue")  
        self.result_label.pack(pady=5)  
        # Button to clear input and result  
        self.clear_button = tk.Button(master, text="Clear", command=self.clear_fields)  
        self.clear_button.pack(pady=5)  
    def calculate_dog_years(self):  
        '''  
        Calculate the dog's age in dog years and update the result label.  
        '''  
        try:  
            human_years = int(self.entry.get())  
            if human_years < 0:  
                self.result_label.config(text="Age cannot be negative. Please enter a valid number.", fg="red")  
            else:  
                dog_years = calculate_dog_years(human_years)  
                self.result_label.config(text=f"Dog's age in dog years: {dog_years}", fg="blue")  
        except ValueError:  
            self.result_label.config(text="Please enter a valid number.", fg="red")  
    def clear_fields(self):  
        '''  
        Clear the input field and result label.  
        '''  
        self.entry.delete(0, tk.END)  
        self.result_label.config(text="")
