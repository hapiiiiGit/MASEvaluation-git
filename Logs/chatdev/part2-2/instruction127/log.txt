'''

Main application file for the Integer Joiner application using Tkinter.

'''

import tkinter as tk
from tkinter import messagebox
from utils import join_integer_list

class IntegerJoinerApp:
    def __init__(self, master):
        '''
        Initialize the main application window and its components.
        '''
        self.master = master
        master.title("Integer Joiner")

        self.label = tk.Label(master, text="Enter integers separated by commas:")
        self.label.pack()

        self.entry = tk.Entry(master)
        self.entry.pack()

        self.join_button = tk.Button(master, text="Join Integers", command=self.join_integers)
        self.join_button.pack()

        self.result_label = tk.Label(master, text="")
        self.result_label.pack()

    def join_integers(self):
        '''
        Join the list of integers entered by the user and display the result.
        '''
        input_text = self.entry.get()
        try:
            int_list = list(map(int, input_text.split(',')))
            result = join_integer_list(int_list)
            self.result_label.config(text=f"Result: {result}")
        except ValueError:
            messagebox.showerror("Invalid input", "Please enter a valid list of integers separated by commas.")

if __name__ == "__main__":
    root = tk.Tk()
    app = IntegerJoinerApp(root)
    root.mainloop()
```

```python
'''

Utility functions for the Integer Joiner application.

def join_integer_list(int_list):
    '''
    Convert a list of integers into a single concatenated integer.
    
    Parameters:
    int_list (list of int): The list of integers to join.

    Returns:
    int: The concatenated integer.
    '''
    return int(''.join(map(str, int_list)))

        Join the list of integers entered by the user and display the result.

import tkinter as tk
from tkinter import messagebox
from utils import join_integer_list
class IntegerJoinerApp:
    def __init__(self, master):
        '''
        Initialize the main application window and its components.
        '''
        self.master = master
        master.title("Integer Joiner")
        self.label = tk.Label(master, text="Enter integers separated by commas:")
        self.label.pack()
        self.entry = tk.Entry(master)
        self.entry.pack()
        self.join_button = tk.Button(master, text="Join Integers", command=self.join_integers)
        self.join_button.pack()

        self.result_label.pack()
    def join_integers(self):
        '''
        Join the list of integers entered by the user and display the result.
        '''
        input_text = self.entry.get()
        try:
            # Split the input text by commas and convert to integers
            int_list = list(map(int, input_text.split(',')))
            # Call the utility function to join the integers
            result = join_integer_list(int_list)
            # Display the result in the result label
            self.result_label.config(text=f"Result: {result}")
        except ValueError:
            # Show an error message if the input is invalid
            messagebox.showerror("Invalid input", "Please enter a valid list of integers separated by commas.")
if __name__ == "__main__":
    # Create the main Tkinter window and run the application
    root = tk.Tk()
    app = IntegerJoinerApp(root)
    root.mainloop()

'''
Main application file for the Integer Joiner application using Tkinter.

import tkinter as tk
from tkinter import messagebox
from utils import join_integer_list
class IntegerJoinerApp:
    def __init__(self, master):
        '''
        Initialize the main application window and its components.
        '''
        self.master = master
        master.title("Integer Joiner")
        self.label = tk.Label(master, text="Enter integers separated by commas:")
        self.label.pack()
        self.entry = tk.Entry(master)
        self.entry.pack()
        self.join_button = tk.Button(master, text="Join Integers", command=self.join_integers)
        self.join_button.pack()
        self.result_label = tk.Label(master, text="")
        self.result_label.pack()
    def join_integers(self):
        '''
        Join the list of integers entered by the user and display the result.
        '''
        input_text = self.entry.get()
        try:
            # Split the input text by commas and convert to integers
            int_list = list(map(int, input_text.split(',')))
            # Call the utility function to join the integers
            result = join_integer_list(int_list)
            # Display the result in the result label
            self.result_label.config(text=f"Result: {result}")
        except ValueError:
            # Show an error message if the input is invalid
            messagebox.showerror("Invalid input", "Please enter a valid list of integers separated by commas.")
if __name__ == "__main__":
    # Create the main Tkinter window and run the application
    root = tk.Tk()
    app = IntegerJoinerApp(root)
    root.mainloop()

# Integer Joiner Application

Welcome to the Integer Joiner Application! This software allows users to join a list of integers into a single concatenated integer using a simple graphical user interface (GUI) built with Tkinter.

## Main Functions

The Integer Joiner Application provides the following main functions:

- **User Input:** Enter a list of integers separated by commas in the provided input field.
- **Join Integers:** Click the "Join Integers" button to concatenate the list of integers into a single integer.
- **Display Result:** View the concatenated integer result displayed on the application interface.
- **Error Handling:** Receive error messages if the input is invalid (e.g., non-integer values or incorrect formatting).

## Quick Install

To run the Integer Joiner Application, you need to have Python installed on your system. Additionally, you need to install the Tkinter library, which is typically included with Python installations. If not, you can install it using the following command:

```bash
sudo apt-get install python3-tk
```

## How to Use

1. **Clone the Repository:**
   First, clone the repository containing the application code to your local machine.

   ```bash
   git clone <repository-url>
   ```

2. **Navigate to the Application Directory:**
   Change your directory to the location where the application files are stored.

   ```bash
   cd <repository-directory>
   ```

3. **Run the Application:**
   Execute the main application file to start the Integer Joiner Application.

   ```bash
   python main.py
   ```

4. **Enter Integers:**
   In the application window, enter a list of integers separated by commas in the input field.

5. **Join Integers:**
   Click the "Join Integers" button to process the input and display the concatenated integer result.

6. **View Result:**
   The result will be displayed below the input field. If the input is invalid, an error message will be shown.

## Example

- **Input:** `1, 23, 456`
- **Output:** `123456`

## Troubleshooting

- **Invalid Input Error:** Ensure that you enter only integers separated by commas. Avoid spaces or non-numeric characters.
- **Tkinter Installation Issues:** If you encounter issues with Tkinter, ensure it is installed correctly on your system.

## Support
