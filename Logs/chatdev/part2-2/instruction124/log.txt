'''

This script creates a GUI application using tkinter to extract and display negative numbers from a user-provided list.

'''

import tkinter as tk
from tkinter import messagebox

class MainApp:
    def __init__(self, root):
        self.root = root
        self.root.title("Negative Number Extractor")

        # Create and place widgets
        self.label = tk.Label(root, text="Enter a list of numbers separated by commas:")
        self.label.pack(pady=10)

        self.entry = tk.Entry(root, width=50)
        self.entry.pack(pady=10)

        self.submit_button = tk.Button(root, text="Submit", command=self.on_submit)
        self.submit_button.pack(pady=10)

        self.result_label = tk.Label(root, text="")
        self.result_label.pack(pady=10)

    def get_negative_numbers(self, numbers):
        '''
        Extracts negative numbers from a list of integers.

        :param numbers: List of integers
        :return: List of negative integers
        '''
        return [num for num in numbers if num < 0]

    def on_submit(self):
        '''
        Handles the submit button click event, processes the input, and displays the result.
        '''
        input_text = self.entry.get()
        try:
            # Convert input string to a list of integers
            numbers = list(map(int, input_text.split(',')))
            # Get negative numbers
            negative_numbers = self.get_negative_numbers(numbers)
            # Display result
            self.result_label.config(text=f"Negative numbers: {negative_numbers}")
        except ValueError:
            messagebox.showerror("Invalid input", "Please enter a valid list of numbers separated by commas.")

def main():
    root = tk.Tk()
    app = MainApp(root)
    root.mainloop()

if __name__ == "__main__":
    main()
```

- **on_submit Method**: This method handles the logic when the submit button is clicked. It reads the input, processes it to extract negative numbers, and updates the result label. It also handles invalid input using a message box.
- **main Function**: This function initializes the `tkinter` main loop to run the application.

This script creates a GUI application using tkinter to extract and display negative numbers from a user-provided list.

'''

import tkinter as tk
from tkinter import messagebox

class MainApp:
    def __init__(self, root):
        self.root = root
        self.root.title("Negative Number Extractor")

        # Create and place widgets
        self.label = tk.Label(root, text="Enter a list of numbers separated by commas:")
        self.label.pack(pady=10)

        self.entry = tk.Entry(root, width=50)
        self.entry.pack(pady=10)

        self.submit_button = tk.Button(root, text="Submit", command=self.on_submit)
        self.submit_button.pack(pady=10)

        self.result_label = tk.Label(root, text="")
        self.result_label.pack(pady=10)

    def get_negative_numbers(self, numbers):
        '''
        Extracts negative numbers from a list of integers.

        :param numbers: List of integers
        :return: List of negative integers
        '''
        return [num for num in numbers if num < 0]

    def on_submit(self):
        '''
        Handles the submit button click event, processes the input, and displays the result.
        '''
        input_text = self.entry.get()
        try:
            # Convert input string to a list of integers
            numbers = list(map(int, input_text.split(',')))
            # Get negative numbers
            negative_numbers = self.get_negative_numbers(numbers)
            # Display result
            self.result_label.config(text=f"Negative numbers: {negative_numbers}")
        except ValueError:
            messagebox.showerror("Invalid input", "Please enter a valid list of numbers separated by commas.")

def main():
    root = tk.Tk()
    app = MainApp(root)
    root.mainloop()

if __name__ == "__main__":
    main()
```

- **MainApp Class**: This class initializes the GUI components using `tkinter`. It has an entry widget for user input, a submit button, and a label to display the result.
- **get_negative_numbers Method**: This method takes a list of integers and returns a list of negative numbers.
- **on_submit Method**: This method handles the logic when the submit button is clicked. It reads the input, processes it to extract negative numbers, and updates the result label. It also handles invalid input using a message box.
- **main Function**: This function initializes the `tkinter` main loop to run the application.

        self.label.pack(pady=10)

        self.entry.pack(pady=10)

        self.submit_button.pack(pady=10)
        self.result_label = tk.Label(root, text="")

        :return: List of negative integers

def main():
