'''
main.py
A GUI application to find numbers within a given range where each number is divisible by every digit it contains.

import tkinter as tk
from tkinter import messagebox
def is_divisible_by_digits(number):
    '''
    Check if a number is divisible by each of its digits.
    '''
    original_number = number
    while number > 0:
        digit = number % 10
        if digit == 0 or original_number % digit != 0:
            return False
        number //= 10
    return True
class DivisibleDigitsApp:
    '''
    A GUI application to find numbers divisible by their digits within a given range.
    '''
    def __init__(self, master):
        self.master = master
        master.title("Divisible by Digits Finder")
        self.label = tk.Label(master, text="Enter range (start and end):")
        self.label.pack()
        self.start_entry = tk.Entry(master)
        self.start_entry.pack()
        self.end_entry = tk.Entry(master)
        self.end_entry.pack()
        self.find_button = tk.Button(master, text="Find Numbers", command=self.find_divisible_numbers)
        self.find_button.pack()
        self.result_label = tk.Label(master, text="", wraplength=300)
        self.result_label.pack()
    def find_divisible_numbers(self):
        '''
        Find and display numbers divisible by their digits within the specified range.
        '''
        try:
            start_num = int(self.start_entry.get())
            end_num = int(self.end_entry.get())
            if start_num > end_num:
                raise ValueError("Start number must be less than or equal to end number.")
            results = [num for num in range(start_num, end_num + 1) if is_divisible_by_digits(num)]
            self.display_results(results)
        except ValueError as e:
            messagebox.showerror("Invalid input", str(e))
    def display_results(self, results):
        '''
        Display the results in the GUI.
        '''
        if results:
            result_text = "Numbers divisible by their digits:\n" + ", ".join(map(str, results))
        else:
            result_text = "No numbers found in the given range."
        self.result_label.config(text=result_text)
if __name__ == "__main__":
    root = tk.Tk()
    app = DivisibleDigitsApp(root)
    root.mainloop()

'''
main.py
A GUI application to find numbers within a given range where each number is divisible by every digit it contains.
'''
import tkinter as tk
from tkinter import messagebox
def is_divisible_by_digits(number):
    '''
    Check if a number is divisible by each of its digits.
    '''
    original_number = number
    while number > 0:
        digit = number % 10
        if digit == 0 or original_number % digit != 0:
            return False
        number //= 10
    return True
class DivisibleDigitsApp:
    '''
    A GUI application to find numbers divisible by their digits within a given range.
    '''
    def __init__(self, master):
        self.master = master
        master.title("Divisible by Digits Finder")
        self.label = tk.Label(master, text="Enter range (start and end):")
        self.label.pack()
        self.start_entry = tk.Entry(master)
        self.start_entry.pack()
        self.end_entry = tk.Entry(master)
        self.end_entry.pack()
        self.find_button = tk.Button(master, text="Find Numbers", command=self.find_divisible_numbers)
        self.find_button.pack()
        self.result_label = tk.Label(master, text="", wraplength=300)
        self.result_label.pack()
    def find_divisible_numbers(self):
        '''
        Find and display numbers divisible by their digits within the specified range.
        '''
        try:
            start_num = int(self.start_entry.get())
            end_num = int(self.end_entry.get())
            if start_num > end_num:
                raise ValueError("Start number must be less than or equal to end number.")
            results = [num for num in range(start_num, end_num + 1) if is_divisible_by_digits(num)]
            self.display_results(results)
        except ValueError as e:
            messagebox.showerror("Invalid input", str(e))
    def display_results(self, results):
        '''
        Display the results in the GUI.
        '''
        if results:
            result_text = "Numbers divisible by their digits:\n" + ", ".join(map(str, results))
        else:
            result_text = "No numbers found in the given range."
        self.result_label.config(text=result_text)
if __name__ == "__main__":
    root = tk.Tk()
    app = DivisibleDigitsApp(root)
    root.mainloop()

- **Input Range:** Users can input a starting and ending number to define the range within which they want to find numbers divisible by their digits.
- **Find Numbers:** The application processes the input range and displays all numbers that meet the criteria.
- **Display Results:** Results are shown in the application interface, providing a clear list of numbers that are divisible by their digits.

## Installation

### Environment Setup

1. **Python Version:** Ensure you have Python version 3.6 or higher installed on your system. You can download Python from the [official website](https://www.python.org/downloads/).

2. **Tkinter:** Tkinter is part of the standard Python library, so no additional installation is required for this dependency.

### Quick Install

Clone the repository or download the source code files to your local machine.

git clone <repository-url>
```

## How to Use

1. **Launch the Application:**
   - Run the `main.py` file using Python to start the application.
   ```bash
   python main.py
   ```

2. **Enter Range:**
   - In the application window, enter the starting and ending numbers in the provided input fields.

3. **Find Numbers:**
   - Click the "Find Numbers" button to execute the search within the specified range.

4. **View Results:**
   - The application will display the numbers that are divisible by their digits in the results section of the GUI.

5. **Error Handling:**
   - If invalid input is provided (e.g., non-numeric values or a start number greater than the end number), an error message will be displayed.
