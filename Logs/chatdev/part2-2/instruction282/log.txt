'''

Main application file for the Distinct Element Checker.

'''

import tkinter as tk
from tkinter import messagebox

def has_one_distinct_element(numbers):
    '''
    Check if the list contains only one distinct element.
    
    :param numbers: List of numbers to check.
    :return: True if only one distinct element exists, False otherwise.
    '''
    return len(set(numbers)) == 1

class DistinctElementCheckerApp:
    '''
    GUI application for checking if a list contains only one distinct element.
    '''
    def __init__(self, root):
        self.root = root
        self.root.title("Distinct Element Checker")
        
        self.label = tk.Label(root, text="Enter numbers separated by commas:")
        self.label.pack(pady=10)
        
        self.entry = tk.Entry(root, width=50)
        self.entry.pack(pady=10)
        
        self.check_button = tk.Button(root, text="Check", command=self.check_distinct_element)
        self.check_button.pack(pady=10)
        
    def check_distinct_element(self):
        '''
        Handle the button click event to check for distinct elements.
        '''
        input_text = self.entry.get()
        try:
            numbers = list(map(int, input_text.split(',')))
            if has_one_distinct_element(numbers):
                messagebox.showinfo("Result", "The list contains only one distinct element.")
            else:
                messagebox.showinfo("Result", "The list contains more than one distinct element.")
        except ValueError:
            messagebox.showerror("Error", "Please enter a valid list of numbers separated by commas.")

if __name__ == "__main__":
    root = tk.Tk()
    app = DistinctElementCheckerApp(root)
    root.mainloop()
```

- **Main Application**: The `main.py` file initializes the Tkinter root window, creates an instance of `DistinctElementCheckerApp`, and starts the main event loop.

Main application file for the Distinct Element Checker.

import tkinter as tk
from tkinter import messagebox

def has_one_distinct_element(numbers):
    '''
    Check if the list contains only one distinct element.
    
    :param numbers: List of numbers to check.
    :return: True if only one distinct element exists, False otherwise.
    '''
    return len(set(numbers)) == 1

class DistinctElementCheckerApp:
    '''
    GUI application for checking if a list contains only one distinct element.
    '''
    def __init__(self, root):
        self.root = root
        self.root.title("Distinct Element Checker")
        
        self.label = tk.Label(root, text="Enter numbers separated by commas:")
        self.label.pack(pady=10)
        
        self.entry = tk.Entry(root, width=50)
        self.entry.pack(pady=10)
        
        self.check_button = tk.Button(root, text="Check", command=self.check_distinct_element)
        self.check_button.pack(pady=10)
        
    def check_distinct_element(self):
        '''
        Handle the button click event to check for distinct elements.
        '''
        input_text = self.entry.get()
        try:
            numbers = list(map(int, input_text.split(',')))
            if has_one_distinct_element(numbers):
                messagebox.showinfo("Result", "The list contains only one distinct element.")
            else:
                messagebox.showinfo("Result", "The list contains more than one distinct element.")
        except ValueError:
            messagebox.showerror("Error", "Please enter a valid list of numbers separated by commas.")

if __name__ == "__main__":
    root = tk.Tk()
    app = DistinctElementCheckerApp(root)
    root.mainloop()

import tkinter as tk

def has_one_distinct_element(numbers):

'''
Main application file for the Distinct Element Checker.
'''
import tkinter as tk
from tkinter import messagebox
def has_one_distinct_element(numbers):
    '''
    Check if the list contains only one distinct element.
    :param numbers: List of numbers to check.
    :return: True if only one distinct element exists, False otherwise.
    '''
    return len(set(numbers)) == 1
class DistinctElementCheckerApp:
    '''
    GUI application for checking if a list contains only one distinct element.
    '''
    def __init__(self, root):
        '''
        Initialize the GUI components.
        :param root: The root window of the application.
        '''
        self.root = root
        self.root.title("Distinct Element Checker")
        self.label = tk.Label(root, text="Enter numbers separated by commas:")
        self.label.pack(pady=10)
        self.entry = tk.Entry(root, width=50)
        self.entry.pack(pady=10)
        self.check_button = tk.Button(root, text="Check", command=self.check_distinct_element)
        self.check_button.pack(pady=10)
    def check_distinct_element(self):
        '''
        Handle the button click event to check for distinct elements.
        '''
        input_text = self.entry.get()
        try:
            numbers = list(map(int, input_text.split(',')))
            if has_one_distinct_element(numbers):
                messagebox.showinfo("Result", "The list contains only one distinct element.")
            else:
                messagebox.showinfo("Result", "The list contains more than one distinct element.")
        except ValueError:
            messagebox.showerror("Error", "Please enter a valid list of numbers separated by commas.")
if __name__ == "__main__":
    root = tk.Tk()
    app = DistinctElementCheckerApp(root)
    root.mainloop()
