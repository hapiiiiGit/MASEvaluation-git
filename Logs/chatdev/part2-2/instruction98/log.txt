def find_k_smallest_pairs(nums1, nums2, k):
    """
    Find k smallest pairs consisting of one element from nums1 and one element from nums2.
    """
    if not nums1 or not nums2 or k <= 0:
        return []

    import heapq
    min_heap = []
    for i in range(min(k, len(nums1))):
        heapq.heappush(min_heap, (nums1[i] + nums2[0], i, 0))

    result = []
    while min_heap and len(result) < k:
        sum_val, i, j = heapq.heappop(min_heap)
        result.append((nums1[i], nums2[j]))
        if j + 1 < len(nums2):
            heapq.heappush(min_heap, (nums1[i] + nums2[j + 1], i, j + 1))

    return result

import tkinter as tk
from tkinter import messagebox

def find_k_smallest_pairs(nums1, nums2, k):
    """
    Find k smallest pairs consisting of one element from nums1 and one element from nums2.
    """
    if not nums1 or not nums2 or k <= 0:
        return []

    import heapq
    min_heap = []
    for i in range(min(k, len(nums1))):
        heapq.heappush(min_heap, (nums1[i] + nums2[0], i, 0))

    result = []
    while min_heap and len(result) < k:
        sum_val, i, j = heapq.heappop(min_heap)
        result.append((nums1[i], nums2[j]))
        if j + 1 < len(nums2):
            heapq.heappush(min_heap, (nums1[i] + nums2[j + 1], i, j + 1))

    return result

def find_k_smallest_pairs(nums1, nums2, k):
    """
    Find k smallest pairs consisting of one element from nums1 and one element from nums2.
    """
    if not nums1 or not nums2 or k <= 0:
        return []
    
    import heapq
    min_heap = []
    
    # Initialize the heap with the first element of nums2 paired with each element of nums1
    for i in range(min(k, len(nums1))):
        heapq.heappush(min_heap, (nums1[i] + nums2[0], i, 0))

def find_k_smallest_pairs(nums1, nums2, k):
    """
    Find k smallest pairs consisting of one element from nums1 and one element from nums2.
    """
    if not nums1 or not nums2 or k <= 0:
        return []
    import heapq
    min_heap = []
    # Initialize the heap with the first element of nums2 paired with each element of nums1
    for i in range(min(k, len(nums1))):
        heapq.heappush(min_heap, (nums1[i] + nums2[0], i, 0))
    result = []
    # Extract the smallest pairs from the heap
    while min_heap and len(result) < k:
        sum_val, i, j = heapq.heappop(min_heap)
        result.append((nums1[i], nums2[j]))
        # If there are more elements in nums2, add the next pair to the heap
        if j + 1 < len(nums2):
            heapq.heappush(min_heap, (nums1[i] + nums2[j + 1], i, j + 1))
    return result

def find_k_smallest_pairs(nums1, nums2, k):
    """
    Find k smallest pairs consisting of one element from nums1 and one element from nums2.
    """
    if not nums1 or not nums2 or k <= 0:
        return []
    import heapq
    min_heap = []
    # Initialize the heap with the first element of nums2 paired with each element of nums1
    for i in range(min(k, len(nums1))):
        heapq.heappush(min_heap, (nums1[i] + nums2[0], i, 0))

import tkinter as tk

def find_k_smallest_pairs(nums1, nums2, k):
    """
    Find k smallest pairs consisting of one element from nums1 and one element from nums2.
    """
    if not nums1 or not nums2 or k <= 0:
        return []
    import heapq
    min_heap = []
    # Initialize the heap with the first element of nums2 paired with each element of nums1
    for i in range(min(k, len(nums1))):
        heapq.heappush(min_heap, (nums1[i] + nums2[0], i, 0))
    result = []
    # Extract the smallest pairs from the heap
    while min_heap and len(result) < k:
        sum_val, i, j = heapq.heappop(min_heap)
        result.append((nums1[i], nums2[j]))
        # If there are more elements in nums2, add the next pair to the heap
        if j + 1 < len(nums2):
            heapq.heappush(min_heap, (nums1[i] + nums2[j + 1], i, j + 1))
    return result

import tkinter as tk
from tkinter import messagebox
def find_k_smallest_pairs(nums1, nums2, k):
    """
    Find k smallest pairs consisting of one element from nums1 and one element from nums2.
    """
    if not nums1 or not nums2 or k <= 0:
        return []
    import heapq
    min_heap = []
    # Initialize the heap with the first element of nums2 paired with each element of nums1
    for i in range(min(k, len(nums1))):
        heapq.heappush(min_heap, (nums1[i] + nums2[0], i, 0))
    result = []
    # Extract the smallest pairs from the heap
    while min_heap and len(result) < k:
        sum_val, i, j = heapq.heappop(min_heap)
        result.append((nums1[i], nums2[j]))
        # If there are more elements in nums2, add the next pair to the heap
        if j + 1 < len(nums2):
            heapq.heappush(min_heap, (nums1[i] + nums2[j + 1], i, j + 1))
    return result
class MainApp:
    """
    Main application class for the GUI.

        self.root = root

        self.label_nums1 = tk.Label(root, text="Array 1 (comma-separated):")

        self.entry_nums1.pack()
        self.label_nums2 = tk.Label(root, text="Array 2 (comma-separated):")
        self.label_nums2.pack()
        self.entry_nums2 = tk.Entry(root, width=50)

        self.label_k = tk.Label(root, text="Number of pairs (k):")
        self.label_k.pack()
        self.entry_k = tk.Entry(root, width=10)
        self.entry_k.pack()

        self.submit_button = tk.Button(root, text="Find Pairs", command=self.on_submit)
        self.submit_button.pack()
        # Result display
        self.result_label = tk.Label(root, text="", wraplength=400)
        self.result_label.pack()

        """

        try:
            nums1 = list(map(int, self.entry_nums1.get().split(',')))
            nums2 = list(map(int, self.entry_nums2.get().split(',')))
            k = int(self.entry_k.get())
            pairs = find_k_smallest_pairs(nums1, nums2, k)
            self.result_label.config(text=f"Smallest {k} pairs: {pairs}")
        except ValueError:
            messagebox.showerror("Input Error", "Please enter valid integers for arrays and k.")
if __name__ == "__main__":
