'''

This module contains the logic for finding perfect squares between two given numbers.

'''

import math

class PerfectSquareFinder:
    @staticmethod
    def find_perfect_squares(start, end):
        """
        Finds all perfect squares between two given numbers (inclusive).

        Args:
            start (int): The starting number.
            end (int): The ending number.

        Returns:
            list: A list of perfect squares between start and end.
        """
        perfect_squares = []
        for num in range(start, end + 1):
            if math.isqrt(num) ** 2 == num:  # Check if the number is a perfect square
                perfect_squares.append(num)
        return perfect_squares
```

```python
'''

This module contains the logic for finding perfect squares between two given numbers.

'''

import math

class PerfectSquareFinder:
    @staticmethod
    def find_perfect_squares(start, end):
        """
        Finds all perfect squares between two given numbers (inclusive).

        Args:
            start (int): The starting number.
            end (int): The ending number.

        Returns:
            list: A list of perfect squares between start and end.
        """
        perfect_squares = []
        for num in range(start, end + 1):
            if math.isqrt(num) ** 2 == num:  # Check if the number is a perfect square
                perfect_squares.append(num)
        return perfect_squares

'''
This module contains the logic for finding perfect squares between two given numbers.
'''
import math
class PerfectSquareFinder:
    @staticmethod
    def find_perfect_squares(start, end):
        """
        Finds all perfect squares between two given numbers (inclusive).
        Args:
            start (int): The starting number.
            end (int): The ending number.
        Returns:
            list: A list of perfect squares between start and end.
        """
        perfect_squares = []
        for num in range(start, end + 1):
            if math.isqrt(num) ** 2 == num:  # Check if the number is a perfect square
                perfect_squares.append(num)
        return perfect_squares

'''
This module contains the logic for finding perfect squares between two given numbers.
'''
import math
class PerfectSquareFinder:
    @staticmethod
    def find_perfect_squares(start, end):
        """
        Finds all perfect squares between two given numbers (inclusive).
        Args:
            start (int): The starting number.
            end (int): The ending number.
        Returns:
            list: A list of perfect squares between start and end.
        """
        perfect_squares = []
        for num in range(start, end + 1):
            if math.isqrt(num) ** 2 == num:  # Check if the number is a perfect square
                perfect_squares.append(num)
        return perfect_squares

'''

'''
import math
class PerfectSquareFinder:
    @staticmethod
    def find_perfect_squares(start, end):
        """
        Finds all perfect squares between two given numbers (inclusive).
        Args:
            start (int): The starting number.

            list: A list of perfect squares between start and end.

            if math.isqrt(num) ** 2 == num:  # Check if the number is a perfect square
                perfect_squares.append(num)
        return perfect_squares

'''
This module contains the logic for finding perfect squares between two given numbers.
'''
import math

class PerfectSquareFinder:
    @staticmethod
    def find_perfect_squares(start, end):
        """
        Finds all perfect squares between two given numbers (inclusive).
        Args:
            start (int): The starting number.
            end (int): The ending number.
        Returns:
            list: A list of perfect squares between start and end.
        """
        perfect_squares = []
        for num in range(start, end + 1):
            if math.isqrt(num) ** 2 == num:  # Check if the number is a perfect square
                perfect_squares.append(num)
        return perfect_squares

'''
This module contains the logic for finding perfect squares between two given numbers.
'''
import math
class PerfectSquareFinder:
    @staticmethod
    def find_perfect_squares(start, end):
        """
        Finds all perfect squares between two given numbers (inclusive).
        Args:
            start (int): The starting number.
            end (int): The ending number.
        Returns:
            list: A list of perfect squares between start and end.
        """
        perfect_squares = []
        for num in range(start, end + 1):
            if math.isqrt(num) ** 2 == num:  # Check if the number is a perfect square
                perfect_squares.append(num)
        return perfect_squares

'''
This module contains the logic for finding perfect squares between two given numbers.
'''
import math
class PerfectSquareFinder:
    @staticmethod
    def find_perfect_squares(start, end):
        """
        Finds all perfect squares between two given numbers (inclusive).
        Args:
            start (int): The starting number.
            end (int): The ending number.
        Returns:
            list: A list of perfect squares between start and end.
        """
        perfect_squares = []
        for num in range(start, end + 1):
            if math.isqrt(num) ** 2 == num:  # Check if the number is a perfect square
                perfect_squares.append(num)
        return perfect_squares

'''
This module contains the logic for finding perfect squares between two given numbers.
'''
import math
class PerfectSquareFinder:
    @staticmethod
    def find_perfect_squares(start, end):
        """
        Finds all perfect squares between two given numbers (inclusive).
        Args:
            start (int): The starting number.
            end (int): The ending number.
        Returns:
            list: A list of perfect squares between start and end.
        """
        perfect_squares = []
        for num in range(start, end + 1):
            if math.isqrt(num) ** 2 == num:  # Check if the number is a perfect square
                perfect_squares.append(num)
        return perfect_squares

'''
This is the main file for the application. It creates a graphical user interface (GUI) for finding perfect squares between two numbers.
'''
import tkinter as tk
from tkinter import messagebox
from perfect_square_finder import PerfectSquareFinder
class GUI:
    def __init__(self):
        self.window = tk.Tk()
        self.window.title("Perfect Square Finder")
        self.window.geometry("400x300")
        # Create GUI elements
        self.create_main_window()
    def create_main_window(self):
        """
        Creates the main application window with input fields and buttons.
        """
        # Labels
        tk.Label(self.window, text="Enter Start Number:", font=("Arial", 12)).pack(pady=10)
        self.start_entry = tk.Entry(self.window, font=("Arial", 12))
        self.start_entry.pack(pady=5)
        tk.Label(self.window, text="Enter End Number:", font=("Arial", 12)).pack(pady=10)
        self.end_entry = tk.Entry(self.window, font=("Arial", 12))
        self.end_entry.pack(pady=5)
        # Submit Button
        submit_button = tk.Button(self.window, text="Find Perfect Squares", font=("Arial", 12), command=self.on_submit)
        submit_button.pack(pady=20)
        # Output Label
        self.output_label = tk.Label(self.window, text="", font=("Arial", 12), fg="blue")
        self.output_label.pack(pady=10)
    def on_submit(self):
        """
        Handles the submission of user input and displays the perfect squares.
        """
        try:
            # Get user input
            start = int(self.start_entry.get())
            end = int(self.end_entry.get())
            # Validate input
            if start > end:
                messagebox.showerror("Error", "Start number must be less than or equal to end number.")
                return
            # Find perfect squares
            perfect_squares = PerfectSquareFinder.find_perfect_squares(start, end)
            # Display results
            if perfect_squares:
                self.output_label.config(text=f"Perfect Squares: {', '.join(map(str, perfect_squares))}")
            else:
                self.output_label.config(text="No perfect squares found.")
        except ValueError:
            messagebox.showerror("Error", "Please enter valid integers.")
    def run(self):
        """
        Runs the main application loop.
        """
        self.window.mainloop()
# Entry point
if __name__ == "__main__":
    app = GUI()
    app.run()
