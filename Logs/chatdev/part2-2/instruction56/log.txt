'''

Main application file for checking if all elements in a list are equal to a given element using a GUI.

'''

import tkinter as tk
from tkinter import messagebox

def check_all_elements_equal(lst, element):
    '''
    Check if all elements in the list are equal to the given element.

    Parameters:
    lst (list): The list of elements to check.
    element: The element to compare against.

    Returns:
    bool: True if all elements are equal to the given element, False otherwise.
    '''
    return all(item == element for item in lst)

class App:
    '''
    Main application class to create the GUI and handle user interactions.
    '''
    def __init__(self, root):
        self.root = root
        self.root.title("Element Equality Checker")

        # Input for list
        self.list_label = tk.Label(root, text="Enter list (comma-separated):")
        self.list_label.pack()
        self.list_entry = tk.Entry(root, width=50)
        self.list_entry.pack()

        # Input for element
        self.element_label = tk.Label(root, text="Enter element:")

'''
Main application file for checking if all elements in a list are equal to a given element using a GUI.
'''
import tkinter as tk
from tkinter import messagebox
def check_all_elements_equal(lst, element):
    '''
    Check if all elements in the list are equal to the given element.
    Parameters:
    lst (list): The list of elements to check.
    element: The element to compare against.
    Returns:
    bool: True if all elements are equal to the given element, False otherwise.
    '''
    return all(item == element for item in lst)
class App:
    '''
    Main application class to create the GUI and handle user interactions.
    '''
    def __init__(self, root):
        self.root = root
        self.root.title("Element Equality Checker")
        # Input for list
        self.list_label = tk.Label(root, text="Enter list (comma-separated):")
        self.list_label.pack()
        self.list_entry = tk.Entry(root, width=50)
        self.list_entry.pack()
        # Input for element
        self.element_label = tk.Label(root, text="Enter element:")
        self.element_label.pack()
        self.element_entry = tk.Entry(root, width=50)
        self.element_entry.pack()
        # Check button
        self.check_button = tk.Button(root, text="Check", command=self.check_elements)
        self.check_button.pack()
    def check_elements(self):
        '''
        Check if all elements in the list are equal to the given element and display the result.
        '''
        try:
            # Parse the list input
            lst = [item.strip() for item in self.list_entry.get().split(',')]
            element = self.element_entry.get().strip()
            # Check if all elements are equal
            result = check_all_elements_equal(lst, element)
            # Display result
            if result:
                messagebox.showinfo("Result", "All elements are equal to the given element.")
            else:
                messagebox.showinfo("Result", "Not all elements are equal to the given element.")
        except Exception as e:
            messagebox.showerror("Error", f"An error occurred: {e}")
if __name__ == "__main__":
    root = tk.Tk()
    app = App(root)
    root.mainloop()

'''
Main application file for checking if all elements in a list are equal to a given element using a GUI.
'''
import tkinter as tk
from tkinter import messagebox
def check_all_elements_equal(lst, element):
    '''
    Check if all elements in the list are equal to the given element.
    Parameters:
    lst (list): The list of elements to check.
    element: The element to compare against.
    Returns:
    bool: True if all elements are equal to the given element, False otherwise.
    '''
    return all(item == element for item in lst)
class App:
    '''
    Main application class to create the GUI and handle user interactions.
    '''
    def __init__(self, root):
        self.root = root
        self.root.title("Element Equality Checker")
        # Input for list
        self.list_label = tk.Label(root, text="Enter list (comma-separated):")
        self.list_label.pack()
        self.list_entry = tk.Entry(root, width=50)
        self.list_entry.pack()
        # Input for element
        self.element_label = tk.Label(root, text="Enter element:")
        self.element_label.pack()
        self.element_entry = tk.Entry(root, width=50)
        self.element_entry.pack()
        # Check button
        self.check_button = tk.Button(root, text="Check", command=self.check_elements)
        self.check_button.pack()
    def check_elements(self):
        '''
        Check if all elements in the list are equal to the given element and display the result.
        '''
        try:
            # Parse the list input
            lst = [item.strip() for item in self.list_entry.get().split(',')]
            element = self.element_entry.get().strip()
            # Check if all elements are equal
            result = check_all_elements_equal(lst, element)
            # Display result
            if result:
                messagebox.showinfo("Result", "All elements are equal to the given element.")
            else:
                messagebox.showinfo("Result", "Not all elements are equal to the given element.")
        except Exception as e:
            messagebox.showerror("Error", f"An error occurred: {e}")
if __name__ == "__main__":
    root = tk.Tk()
    app = App(root)
    root.mainloop()
