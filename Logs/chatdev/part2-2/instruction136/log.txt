- P(2) = 2
- For n > 2, P(n) = P(n-2) + P(n-3)

'''

This is the main file for the Perrin number sum calculator application.
It initializes the GUI and handles user interactions.

'''

import tkinter as tk
from tkinter import messagebox

class PerrinCalculator:
    def calculate_perrin(self, n):
        if n == 0:
            return 3
        elif n == 1:
            return 0
        elif n == 2:
            return 2
        else:
            a, b, c = 3, 0, 2
            for _ in range(3, n + 1):
                a, b, c = b, c, b + a
            return c

    def sum_perrin(self, n):
        total = 0
        for i in range(n):
            total += self.calculate_perrin(i)
        return total

class PerrinApp:
    def __init__(self, root):
        self.root = root
        self.root.title("Perrin Number Sum Calculator")

        self.label = tk.Label(root, text="Enter the number of terms:")
        self.label.pack()

        self.entry = tk.Entry(root)
        self.entry.pack()

        self.calculate_button = tk.Button(root, text="Calculate Sum", command=self.calculate_sum)
        self.calculate_button.pack()

        self.result_label = tk.Label(root, text="")
        self.result_label.pack()

        self.calculator = PerrinCalculator()

    def calculate_sum(self):
        try:
            n = int(self.entry.get())
            if n < 0:
                raise ValueError("The number of terms must be non-negative.")
            result = self.calculator.sum_perrin(n)
            self.result_label.config(text=f"Sum of first {n} Perrin numbers: {result}")
        except ValueError as e:
            messagebox.showerror("Invalid Input", str(e))

if __name__ == "__main__":
    root = tk.Tk()
    app = PerrinApp(root)
    root.mainloop()

import tkinter as tk
from tkinter import messagebox

class PerrinCalculator:
    def calculate_perrin(self, n):
        if n == 0:
            return 3
        elif n == 1:
            return 0
        elif n == 2:
            return 2
        else:
            a, b, c = 3, 0, 2
            for _ in range(3, n + 1):
                a, b, c = b, c, b + a
            return c

    def sum_perrin(self, n):
        total = 0
        for i in range(n):

        return total

if __name__ == "__main__":

import tkinter as tk
from tkinter import messagebox
class PerrinCalculator:
    def calculate_perrin(self, n):
        '''
        Calculate the nth Perrin number.
        '''
        if n == 0:
            return 3
        elif n == 1:
            return 0
        elif n == 2:
            return 2
        else:
            a, b, c = 3, 0, 2
            for _ in range(3, n + 1):
                a, b, c = b, c, b + a
            return c
    def sum_perrin(self, n):
        '''
        Calculate the sum of the first n Perrin numbers.
        The loop iterates from 0 to n-1 to include the nth term in the sum.
        '''
        total = 0
        for i in range(n):
            total += self.calculate_perrin(i)
        return total

        Initialize the Perrin number sum calculator application.

'''
This is the main file for the Perrin number sum calculator application.
It initializes the GUI and handles user interactions.

import tkinter as tk
from tkinter import messagebox
class PerrinCalculator:
    def calculate_perrin(self, n):
        '''
        Calculate the nth Perrin number.
        '''
        if n == 0:
            return 3
        elif n == 1:
            return 0
        elif n == 2:
            return 2
        else:
            a, b, c = 3, 0, 2
            for _ in range(3, n + 1):
                a, b, c = b, c, b + a
            return c
    def sum_perrin(self, n):
        '''
        Calculate the sum of the first n Perrin numbers.
        The loop iterates from 0 to n-1 to include the nth term in the sum.
        '''
        total = 0
        for i in range(n):
            total += self.calculate_perrin(i)
        return total
class PerrinApp:
    def __init__(self, root):
        '''
        Initialize the Perrin number sum calculator application.
        Set up the GUI components.
        '''
        self.root = root
        self.root.title("Perrin Number Sum Calculator")
        self.label = tk.Label(root, text="Enter the number of terms:")
        self.label.pack()
        self.entry = tk.Entry(root)
        self.entry.pack()
        self.calculate_button = tk.Button(root, text="Calculate Sum", command=self.calculate_sum)
        self.calculate_button.pack()
        self.result_label = tk.Label(root, text="")
        self.result_label.pack()
        self.calculator = PerrinCalculator()
    def calculate_sum(self):
        '''
        Handle the calculation of the sum of Perrin numbers based on user input.
        Display the result or an error message if input is invalid.
        '''
        try:
            n = int(self.entry.get())
            if n < 0:
                raise ValueError("The number of terms must be non-negative.")
            result = self.calculator.sum_perrin(n)
            self.result_label.config(text=f"Sum of first {n} Perrin numbers: {result}")
        except ValueError as e:
            messagebox.showerror("Invalid Input", str(e))
if __name__ == "__main__":
    root = tk.Tk()
    app = PerrinApp(root)
    root.mainloop()

# Perrin Number Sum Calculator

A Python-based application to calculate the sum of Perrin numbers using a graphical user interface (GUI).

## 📖 Introduction

The Perrin Number Sum Calculator is a simple and intuitive application designed to calculate the sum of the first `n` Perrin numbers. Perrin numbers are a sequence of integers defined by the recurrence relation:

```
P(n) = P(n-2) + P(n-3)
```

with the initial values:
```
P(0) = 3, P(1) = 0, P(2) = 2
```

This application provides a user-friendly interface to input the number of terms (`n`) and calculates the sum of the first `n` Perrin numbers.

## 🚀 Features

- **Calculate Perrin Numbers**: Computes the nth Perrin number using the recurrence relation.
- **Sum of Perrin Numbers**: Calculates the sum of the first `n` Perrin numbers.
- **Graphical User Interface**: A simple and interactive GUI built with `tkinter` for ease of use.

## 🛠️ Installation

### Prerequisites

- Python 3.6 or higher must be installed on your system.

### Steps to Install

1. Clone or download the project files to your local machine.
2. Navigate to the project directory.
3. Install the required dependencies. This project does not require any external dependencies, so you can skip this step.

Alternatively, you can create a virtual environment for the project:

```bash
# Create a virtual environment
python -m venv perrin_env

# Activate the virtual environment
# On Windows:
perrin_env\Scripts\activate
# On macOS/Linux:
source perrin_env/bin/activate
```

4. Run the application by executing the `main.py` file:

```bash
python main.py
```

## 🖥️ How to Use

1. Launch the application by running `main.py`.
2. A GUI window will appear with the title "Perrin Number Sum Calculator".
3. Enter the number of terms (`n`) in the input field.
   - Example: Enter `5` to calculate the sum of the first 5 Perrin numbers.
4. Click the "Calculate Sum" button.
5. The result will be displayed below the button, showing the sum of the first `n` Perrin numbers.
6. If the input is invalid (e.g., negative numbers or non-integer values), an error message will be displayed.

## Example Usage

### Input:
- Number of terms: `5`

### Output:
- Sum of first 5 Perrin numbers: `10`

### Error Handling:
- If you enter a negative number, the application will display an error message: "The number of terms must be non-negative."
- If you enter a non-integer value, the application will display an error message: "Invalid Input."

## 🧑‍💻 Code Overview

The application consists of two main classes:

1. **PerrinCalculator**:
   - `calculate_perrin(n)`: Calculates the nth Perrin number.
   - `sum_perrin(n)`: Calculates the sum of the first `n` Perrin numbers.

2. **PerrinApp**:
   - Initializes the GUI components using `tkinter`.
   - Handles user input and displays the calculated results.

## 📂 File Structure

├── main.py             # Main application file
├── requirements.txt    # Dependencies file (empty as no external libraries are required)
```
