'''

This module contains the implementation of a GUI application to check if a list of numbers is consecutive. It uses tkinter for the GUI and includes a utility function to perform the check.

'''

import tkinter as tk
from tkinter import messagebox

def are_consecutive(lst):
    '''
    Check if the given list contains consecutive numbers.

    Parameters:
    lst (list): A list of integers.

    Returns:
    bool: True if the list contains consecutive numbers, False otherwise.
    '''
    if not lst:
        return False
    sorted_lst = sorted(lst)
    return all(x + 1 == y for x, y in zip(sorted_lst, sorted_lst[1:]))

class ConsecutiveCheckerApp:
    '''
    A tkinter-based GUI application to check if a list of numbers is consecutive.
    '''

    def __init__(self, master):
        self.master = master
        master.title("Consecutive Numbers Checker")

        self.label = tk.Label(master, text="Enter numbers separated by commas:")
        self.label.pack()

        self.entry = tk.Entry(master)
        self.entry.pack()

        self.check_button = tk.Button(master, text="Check", command=self.check_consecutive)
        self.check_button.pack()

        self.result_label = tk.Label(master, text="")
        self.result_label.pack()

    def check_consecutive(self):
        '''
        Check if the input list contains consecutive numbers and update the result label.
        '''
        try:
            input_text = self.entry.get()
            num_list = list(map(int, input_text.split(',')))
            if are_consecutive(num_list):
                self.result_label.config(text="The numbers are consecutive.")
            else:
                self.result_label.config(text="The numbers are not consecutive.")
        except ValueError:
            messagebox.showerror("Invalid input", "Please enter a valid list of integers separated by commas.")

'''

This module contains the implementation of a GUI application to check if a list of numbers is consecutive. It uses tkinter for the GUI and includes a utility function to perform the check.

'''

import tkinter as tk
from tkinter import messagebox

def are_consecutive(lst):
    '''
    Check if the given list contains consecutive numbers.

    Parameters:
    lst (list): A list of integers.

    Returns:
    bool: True if the list contains consecutive numbers, False otherwise.
    '''
    if not lst:
        return False
    sorted_lst = sorted(lst)
    return all(x + 1 == y for x, y in zip(sorted_lst, sorted_lst[1:]))

class ConsecutiveCheckerApp:
    '''
    A tkinter-based GUI application to check if a list of numbers is consecutive.
    '''

    def __init__(self, master):
        self.master = master
        master.title("Consecutive Numbers Checker")

        self.label = tk.Label(master, text="Enter numbers separated by commas:")
        self.label.pack()

        self.entry = tk.Entry(master)
        self.entry.pack()

        self.check_button = tk.Button(master, text="Check", command=self.check_consecutive)
        self.check_button.pack()

        self.result_label = tk.Label(master, text="")
        self.result_label.pack()

    def check_consecutive(self):
        '''
        Check if the input list contains consecutive numbers and update the result label.
        '''
        try:
            input_text = self.entry.get()
            num_list = list(map(int, input_text.split(',')))
            if are_consecutive(num_list):
                self.result_label.config(text="The numbers are consecutive.")
            else:
                self.result_label.config(text="The numbers are not consecutive.")
        except ValueError:
            messagebox.showerror("Invalid input", "Please enter a valid list of integers separated by commas.")

if __name__ == "__main__":
    root = tk.Tk()
    app = ConsecutiveCheckerApp(root)
    root.mainloop()

'''
This module contains the implementation of a GUI application to check if a list of numbers is consecutive. It uses tkinter for the GUI and includes a utility function to perform the check.
'''
import tkinter as tk
from tkinter import messagebox

def are_consecutive(lst):
    '''
    Check if the given list contains consecutive numbers.
    Parameters:
    lst (list): A list of integers.
    Returns:
    bool: True if the list contains consecutive numbers, False otherwise.
    '''
    if not lst:
        return False
    sorted_lst = sorted(lst)
    return all(x + 1 == y for x, y in zip(sorted_lst, sorted_lst[1:]))

class ConsecutiveCheckerApp:
    '''
    A tkinter-based GUI application to check if a list of numbers is consecutive.
    '''
    def __init__(self, master):
        '''
        Initialize the GUI application with necessary widgets.
        Parameters:
        master (tk.Tk): The root tkinter window.
        '''
        self.master = master
        master.title("Consecutive Numbers Checker")
        self.label = tk.Label(master, text="Enter numbers separated by commas:")
        self.label.pack()
        self.entry = tk.Entry(master)
        self.entry.pack()
        self.check_button = tk.Button(master, text="Check", command=self.check_consecutive)
        self.check_button.pack()
        self.result_label = tk.Label(master, text="")
        self.result_label.pack()

    def check_consecutive(self):
        '''
        Check if the input list contains consecutive numbers and update the result label.
        '''
        try:
            input_text = self.entry.get()
            num_list = list(map(int, input_text.split(',')))
            if are_consecutive(num_list):
                self.result_label.config(text="The numbers are consecutive.")
            else:
                self.result_label.config(text="The numbers are not consecutive.")
        except ValueError:
            messagebox.showerror("Invalid input", "Please enter a valid list of integers separated by commas.")

if __name__ == "__main__":
    root = tk.Tk()
    app = ConsecutiveCheckerApp(root)
    root.mainloop()

'''
import tkinter as tk
from tkinter import messagebox
def are_consecutive(lst):
    '''
    Check if the given list contains consecutive numbers.
    Parameters:
    lst (list): A list of integers.
    Returns:
    bool: True if the list contains consecutive numbers, False otherwise.
    '''
    if not lst:
        return False
    sorted_lst = sorted(lst)
    return all(x + 1 == y for x, y in zip(sorted_lst, sorted_lst[1:]))

import tkinter as tk
from tkinter import messagebox

def are_consecutive(lst):
    '''
    Check if the given list contains consecutive numbers.
    Parameters:
    lst (list): A list of integers.
    Returns:

    if not lst:

'''
This module contains the implementation of a GUI application to check if a list of numbers is consecutive. It uses tkinter for the GUI and includes a utility function to perform the check.
'''
import tkinter as tk
from tkinter import messagebox
def are_consecutive(lst):
    '''
    Check if the given list contains consecutive numbers.
    Parameters:
    lst (list): A list of integers.
    Returns:
    bool: True if the list contains consecutive numbers, False otherwise.
    '''
    if not lst:
        return False
    sorted_lst = sorted(lst)
    return all(x + 1 == y for x, y in zip(sorted_lst, sorted_lst[1:]))
class ConsecutiveCheckerApp:
    '''
    A tkinter-based GUI application to check if a list of numbers is consecutive.
    '''
    def __init__(self, master):
        '''
        Initialize the GUI application with necessary widgets.
        Parameters:
        master (tk.Tk): The root tkinter window.
        '''
        self.master = master
        master.title("Consecutive Numbers Checker")
        self.label = tk.Label(master, text="Enter numbers separated by commas:")
        self.label.pack()
        self.entry = tk.Entry(master)
        self.entry.pack()
        self.check_button = tk.Button(master, text="Check", command=self.check_consecutive)
        self.check_button.pack()
        self.result_label = tk.Label(master, text="")
        self.result_label.pack()
    def check_consecutive(self):
        '''
        Check if the input list contains consecutive numbers and update the result label.
        '''
        try:
            input_text = self.entry.get()
            num_list = list(map(int, input_text.split(',')))
            if are_consecutive(num_list):
                self.result_label.config(text="The numbers are consecutive.", fg="green")
            else:
                self.result_label.config(text="The numbers are not consecutive.", fg="red")
        except ValueError:
            messagebox.showerror("Invalid input", "Please enter a valid list of integers separated by commas.")
if __name__ == "__main__":
    root = tk.Tk()
    app = ConsecutiveCheckerApp(root)
    root.mainloop()
