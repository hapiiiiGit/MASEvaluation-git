'''

This file contains the logic to check whether the entered number is greater than all elements of the given array.

'''

def is_number_greater(number, array):
    """
    Checks if the given number is greater than all elements in the array.
    
    Args:
        number (float): The number to compare.
        array (list of float): The array of numbers to compare against.
    
    Returns:
        bool: True if the number is greater than all elements, False otherwise.
    """
    for element in array:
        if number <= element:
            return False
    return True
```

def is_number_greater(number, array):

def is_number_greater(number, array):
    """
    Checks if the given number is greater than all elements in the array.
    Args:
        number (float): The number to compare.
        array (list of float): The array of numbers to compare against.
    Returns:
        bool: True if the number is greater than all elements, False otherwise.
    """
    for element in array:
        if number <= element:
            return False
    return True

def is_number_greater(number, array):
    """
    Checks if the given number is greater than all elements in the array.
    Args:
        number (float): The number to compare.
        array (list of float): The array of numbers to compare against.

    """
    for element in array:

'''
This file contains the logic to check whether the entered number is greater than all elements of the given array.
'''
def is_number_greater(number, array):
    """
    Checks if the given number is greater than all elements in the array.
    Args:
        number (float): The number to compare.
        array (list of float): The array of numbers to compare against.
    Returns:
        bool: True if the number is greater than all elements, False otherwise.
    """
    for element in array:
        if number <= element:
            return False
    return True
