'''

This file contains the main application logic and GUI implementation for the Tuple Counter application. It uses tkinter for the graphical interface.

'''

import tkinter as tk
from tkinter import messagebox
from utils import count_occurrences

class TupleCounterApp:
    def __init__(self, root):
        self.root = root
        self.root.title("Tuple Element Counter")
        
        # GUI Components
        self.label_tuple = tk.Label(root, text="Enter Tuple (comma-separated):")
        self.label_tuple.pack()
        
        self.entry_tuple = tk.Entry(root, width=50)
        self.entry_tuple.pack()
        
        self.label_element = tk.Label(root, text="Enter Element to Count:")
        self.label_element.pack()
        
        self.entry_element = tk.Entry(root, width=50)
        self.entry_element.pack()
        
        self.button_count = tk.Button(root, text="Count Occurrences", command=self.count_occurrences)
        self.button_count.pack()
        
        self.result_label = tk.Label(root, text="")
        self.result_label.pack()
    
    def count_occurrences(self):
        try:
            # Get user inputs
            tuple_input = self.entry_tuple.get()
            element_input = self.entry_element.get()
            
            # Convert tuple input to actual tuple
            tuple_data = tuple(map(str.strip, tuple_input.split(',')))
            
            # Count occurrences using utility function
            count = count_occurrences(tuple_data, element_input)
            
            # Display result
            self.result_label.config(text=f"Occurrences of '{element_input}': {count}")
        except Exception as e:
            messagebox.showerror("Error", f"An error occurred: {e}")

if __name__ == "__main__":
    root = tk.Tk()
    app = TupleCounterApp(root)
    root.mainloop()

```

def count_occurrences(input_tuple, element):
    """
    Counts the occurrences of an element in a tuple.

    Args:
        input_tuple (tuple): The tuple to search.
        element (str): The element to count.

    Returns:
        int: The number of occurrences of the element in the tuple.
    """
    return input_tuple.count(element)

'''

This file contains the main application logic and GUI implementation for the Tuple Counter application. It uses tkinter for the graphical interface.

'''

import tkinter as tk
from tkinter import messagebox
from utils import count_occurrences

class TupleCounterApp:
    def __init__(self, root):
        self.root = root
        self.root.title("Tuple Element Counter")
        
        # GUI Components
        self.label_tuple = tk.Label(root, text="Enter Tuple (comma-separated):")
        self.label_tuple.pack()
        
        self.entry_tuple = tk.Entry(root, width=50)
        self.entry_tuple.pack()
        
        self.label_element = tk.Label(root, text="Enter Element to Count:")
        self.label_element.pack()
        
        self.entry_element = tk.Entry(root, width=50)
        self.entry_element.pack()
        
        self.button_count = tk.Button(root, text="Count Occurrences", command=self.count_occurrences)
        self.button_count.pack()
        
        self.result_label = tk.Label(root, text="")
        self.result_label.pack()
    
    def count_occurrences(self):
        try:
            # Get user inputs
            tuple_input = self.entry_tuple.get()
            element_input = self.entry_element.get()
            
            # Convert tuple input to actual tuple
            tuple_data = tuple(map(str.strip, tuple_input.split(',')))
            
            # Count occurrences using utility function
            count = count_occurrences(tuple_data, element_input)
            
            # Display result
            self.result_label.config(text=f"Occurrences of '{element_input}': {count}")
        except Exception as e:
            messagebox.showerror("Error", f"An error occurred: {e}")

if __name__ == "__main__":
    root = tk.Tk()
    app = TupleCounterApp(root)
    root.mainloop()

'''  
This file contains the main application logic and GUI implementation for the Tuple Counter application. It uses tkinter for the graphical interface.  
'''  
import tkinter as tk  
from tkinter import messagebox  
from utils import count_occurrences  
class TupleCounterApp:  
    def __init__(self, root):  
        self.root = root  
        self.root.title("Tuple Element Counter")  
        # GUI Components  
        self.label_tuple = tk.Label(root, text="Enter Tuple (comma-separated):")  
        self.label_tuple.pack()  
        self.entry_tuple = tk.Entry(root, width=50)  
        self.entry_tuple.pack()  
        self.label_element = tk.Label(root, text="Enter Element to Count:")  
        self.label_element.pack()  
        self.entry_element = tk.Entry(root, width=50)  
        self.entry_element.pack()  
        self.button_count = tk.Button(root, text="Count Occurrences", command=self.count_occurrences)  
        self.button_count.pack()  
        self.result_label = tk.Label(root, text="")  
        self.result_label.pack()  
    def count_occurrences(self):  
        try:  
            # Get user inputs  
            tuple_input = self.entry_tuple.get()  
            element_input = self.entry_element.get()  
            # Validate tuple input  
            if not tuple_input:  
                messagebox.showerror("Error", "Tuple input cannot be empty.")  
                return  
            # Convert tuple input to actual tuple  
            tuple_data = tuple(map(str.strip, tuple_input.split(',')))  
            # Count occurrences using utility function  
            count = count_occurrences(tuple_data, element_input)  
            # Display result  
            self.result_label.config(text=f"Occurrences of '{element_input}': {count}")  
        except Exception as e:  
            messagebox.showerror("Error", f"An error occurred: {e}")  
if __name__ == "__main__":  
    root = tk.Tk()  
    app = TupleCounterApp(root)  
    root.mainloop()

'''

import tkinter as tk
from tkinter import messagebox
from utils import count_occurrences
class TupleCounterApp:
    def __init__(self, root):
        self.root = root
        self.root.title("Tuple Element Counter")
        # GUI Components
        self.label_tuple = tk.Label(root, text="Enter Tuple (comma-separated):")
        self.label_tuple.pack()
        self.entry_tuple = tk.Entry(root, width=50)
        self.entry_tuple.pack()
        self.label_element = tk.Label(root, text="Enter Element to Count:")
        self.label_element.pack()
        self.entry_element = tk.Entry(root, width=50)
        self.entry_element.pack()
        self.button_count = tk.Button(root, text="Count Occurrences", command=self.count_occurrences)
        self.button_count.pack()
        self.result_label = tk.Label(root, text="")
        self.result_label.pack()
    def count_occurrences(self):
        try:
            # Get user inputs
            tuple_input = self.entry_tuple.get()
            element_input = self.entry_element.get()
            # Validate tuple input
            if not tuple_input:
                messagebox.showerror("Error", "Tuple input cannot be empty.")
                return
            # Convert tuple input to actual tuple
            tuple_data = tuple(map(str.strip, tuple_input.split(',')))
            # Count occurrences using utility function
            count = count_occurrences(tuple_data, element_input)
            # Display result
            self.result_label.config(text=f"Occurrences of '{element_input}': {count}")
        except Exception as e:
            messagebox.showerror("Error", f"An error occurred: {e}")
if __name__ == "__main__":
    root = tk.Tk()
    app = TupleCounterApp(root)
    root.mainloop()

# Tuple Element Counter Application

A Python-based application to count occurrences of a specific element in a tuple, featuring a user-friendly graphical interface built with `tkinter`.

---

## 📖 Introduction

The Tuple Element Counter application allows users to input a tuple and an element, and it calculates the number of times the element appears in the tuple. This tool is especially useful for data analysis tasks or educational purposes where tuple manipulation is required.

---

## 🛠️ Features

1. **User-Friendly Interface**: The application provides an intuitive graphical interface for easy input and interaction.
2. **Tuple Validation**: Ensures that the input is a valid tuple format.
3. **Accurate Counting**: Uses a robust utility function to count occurrences of the specified element.
4. **Error Handling**: Provides clear error messages for invalid inputs or unexpected issues.

---

## 🚀 Installation Guide

### Prerequisites

Ensure you have Python installed on your system. The application is compatible with Python 3.6 and above.

### Step 1: Clone the Repository

Clone the project repository to your local machine:

```bash
git clone https://github.com/your-repo/tuple-counter-app.git
cd tuple-counter-app
```

### Step 2: Install Dependencies

Install the required dependencies using `pip`:

```bash
pip install tkinter
```

Note: `tkinter` is included by default in most Python installations. If you encounter issues, ensure your Python installation includes `tkinter`.

---

## 📋 How to Use

### Step 1: Run the Application

Navigate to the project directory and execute the `main.py` file:

```bash
python main.py
```

### Step 2: Input Tuple and Element

1. **Enter Tuple**: Type a tuple in comma-separated format (e.g., `apple, banana, apple, cherry`).
2. **Enter Element**: Type the element you want to count (e.g., `apple`).

### Step 3: Count Occurrences

Click the **Count Occurrences** button. The application will display the number of times the specified element appears in the tuple.

---

## 🐞 Troubleshooting

### Common Issues

1. **Empty Tuple Input**: Ensure the tuple field is not empty. The application will show an error message if left blank.
2. **Invalid Tuple Format**: Ensure the tuple is entered in a comma-separated format. For example, `apple, banana, cherry`.
3. **Element Not Found**: If the element does not exist in the tuple, the result will display `Occurrences of 'element': 0`.

### Error Handling

If an unexpected error occurs, the application will display an error message with details. Ensure all inputs are correct and retry.

---

## 📂 Project Structure

```
tuple-counter-app/
├── main.py       # Main application logic and GUI implementation
├── utils.py      # Utility functions for tuple processing
└── README.md     # Project documentation
```
