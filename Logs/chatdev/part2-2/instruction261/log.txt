'''

This module contains the logic to find the index of the smallest triangular number with n digits.

'''

def find_index_of_smallest_triangular_number(n):
    """
    Finds the index of the smallest triangular number with n digits.
    
    Parameters:
    n (int): The number of digits.
    
    Returns:
    int: The index of the smallest triangular number with n digits.
    """
    index = 1
    triangular_number = 1
    
    while len(str(triangular_number)) < n:
        index += 1
        triangular_number = index * (index + 1) // 2
    
    return index

def find_index_of_smallest_triangular_number(n):
    """
    Finds the index of the smallest triangular number with n digits.
    
    Parameters:
    n (int): The number of digits.
    
    Returns:
    int: The index of the smallest triangular number with n digits.

    index = 1

import math
def find_index_of_smallest_triangular_number(n):
    """
    Finds the index of the smallest triangular number with at least n digits.
    Parameters:
    n (int): The number of digits.
    Returns:
    int: The index of the smallest triangular number with at least n digits.
    """
    # Calculate the smallest number with n digits
    smallest_number = 10**(n-1)
    # Use the formula for the nth triangular number: T_k = k * (k + 1) / 2
    # We need to solve for k in the inequality: k * (k + 1) / 2 >= smallest_number
    # This simplifies to: k^2 + k - 2 * smallest_number >= 0
    # Using the quadratic formula: k = (-b + sqrt(b^2 - 4ac)) / 2a
    # Here a = 1, b = 1, c = -2 * smallest_number
    a = 1
    b = 1
    c = -2 * smallest_number
    # Calculate the discriminant
    discriminant = b**2 - 4*a*c
    # Calculate the positive root of the quadratic equation
    k = (-b + math.sqrt(discriminant)) / (2*a)
    # Return the ceiling of k, as k must be an integer
    return math.ceil(k)

'''
This module contains the function to find the index of the smallest triangular number with at least n digits.
'''
import math
def find_index_of_smallest_triangular_number(n):
    """
    Finds the index of the smallest triangular number with at least n digits.
    Parameters:
    n (int): The number of digits.
    Returns:
    int: The index of the smallest triangular number with at least n digits.
    """
    # Calculate the smallest number with n digits
    smallest_number = 10**(n-1)
    # Use the formula for the nth triangular number: T_k = k * (k + 1) / 2
    # We need to solve for k in the inequality: k * (k + 1) / 2 >= smallest_number
    # This simplifies to: k^2 + k - 2 * smallest_number >= 0
    # Using the quadratic formula: k = (-b + sqrt(b^2 - 4ac)) / 2a
    # Here a = 1, b = 1, c = -2 * smallest_number
    a = 1
    b = 1
    c = -2 * smallest_number
    # Calculate the discriminant
    discriminant = b**2 - 4*a*c
    # Calculate the positive root of the quadratic equation
    k = (-b + math.sqrt(discriminant)) / (2*a)
    # Return the ceiling of k, as k must be an integer
    return math.ceil(k)

import math

    n (int): The number of digits.
    Returns:
    int: The index of the smallest triangular number with at least n digits.
    """
    # Calculate the smallest number with n digits
    smallest_number = 10**(n-1)
    # Use the formula for the nth triangular number: T_k = k * (k + 1) / 2
    # We need to solve for k in the inequality: k * (k + 1) / 2 >= smallest_number

'''
This module contains the function to find the index of the smallest triangular number with at least n digits.
'''
import math
def find_index_of_smallest_triangular_number(n):
    """
    Finds the index of the smallest triangular number with at least n digits.
    Parameters:
    n (int): The number of digits.
    Returns:
    int: The index of the smallest triangular number with at least n digits.
    """
    # Calculate the smallest number with n digits
    smallest_number = 10**(n-1)
    # Use the formula for the nth triangular number: T_k = k * (k + 1) / 2
    # We need to solve for k in the inequality: k * (k + 1) / 2 >= smallest_number
    # This simplifies to: k^2 + k - 2 * smallest_number >= 0
    # Using the quadratic formula: k = (-b + sqrt(b^2 - 4ac)) / 2a
    # Here a = 1, b = 1, c = -2 * smallest_number
    a = 1
    b = 1
    c = -2 * smallest_number
    # Calculate the discriminant
    discriminant = b**2 - 4*a*c
    # Calculate the positive root of the quadratic equation
    k = (-b + math.sqrt(discriminant)) / (2*a)
    # Return the ceiling of k, as k must be an integer
    return math.ceil(k)
