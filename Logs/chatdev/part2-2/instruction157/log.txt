'''

This module contains the main application logic for finding the intersection of elements in a list of tuples using a GUI.

'''

import tkinter as tk
from tkinter import messagebox

def tuple_intersection(tuple_list):
    '''
    Computes the intersection of elements in a list of tuples.

    Parameters:
    tuple_list (list of tuples): The list of tuples to find the intersection.

    Returns:
    set: A set containing the intersection of elements.
    '''
    if not tuple_list:
        return set()
    
    # Convert each tuple to a set and find the intersection
    intersection_set = set(tuple_list[0])
    for tup in tuple_list[1:]:
        intersection_set.intersection_update(tup)
    
    return intersection_set

class TupleIntersectionApp:
    '''
    A GUI application for finding the intersection of elements in a list of tuples.
    '''
    def __init__(self, master):
        self.master = master
        master.title("Tuple Intersection Finder")

        self.label = tk.Label(master, text="Enter tuples (e.g., (1, 2), (2, 3), (2, 4)):")
        self.label.pack()

        self.entry = tk.Entry(master, width=50)
        self.entry.pack()

        self.find_button = tk.Button(master, text="Find Intersection", command=self.find_intersection)
        self.find_button.pack()

        self.result_label = tk.Label(master, text="Intersection: ")
        self.result_label.pack()

    def find_intersection(self):
        try:
            # Evaluate the input to convert it into a list of tuples
            tuple_list = eval(self.entry.get())
            if not isinstance(tuple_list, list) or not all(isinstance(t, tuple) for t in tuple_list):
                raise ValueError("Input must be a list of tuples.")
            
            # Find the intersection
            intersection = tuple_intersection(tuple_list)
            self.result_label.config(text=f"Intersection: {intersection}")
        except Exception as e:
            messagebox.showerror("Error", f"Invalid input: {e}")

if __name__ == "__main__":

This module contains the main application logic for finding the intersection of elements in a list of tuples using a GUI.

'''

import tkinter as tk
from tkinter import messagebox

def tuple_intersection(tuple_list):
    '''
    Computes the intersection of elements in a list of tuples.

    Parameters:
    tuple_list (list of tuples): The list of tuples to find the intersection.

    Returns:
    set: A set containing the intersection of elements.
    '''
    if not tuple_list:
        return set()
    
    # Convert each tuple to a set and find the intersection
    intersection_set = set(tuple_list[0])
    for tup in tuple_list[1:]:
        intersection_set.intersection_update(tup)
    
    return intersection_set

class TupleIntersectionApp:
    '''
    A GUI application for finding the intersection of elements in a list of tuples.
    '''
    def __init__(self, master):
        self.master = master
        master.title("Tuple Intersection Finder")

        self.label = tk.Label(master, text="Enter tuples (e.g., (1, 2), (2, 3), (2, 4)):")

        self.entry = tk.Entry(master, width=50)
        self.entry.pack()

        self.find_button = tk.Button(master, text="Find Intersection", command=self.find_intersection)
        self.find_button.pack()

        self.result_label = tk.Label(master, text="Intersection: ")

    def find_intersection(self):
        try:
            # Evaluate the input to convert it into a list of tuples
            tuple_list = eval(self.entry.get())
            if not isinstance(tuple_list, list) or not all(isinstance(t, tuple) for t in tuple_list):

            # Find the intersection
            intersection = tuple_intersection(tuple_list)
            self.result_label.config(text=f"Intersection: {intersection}")
        except Exception as e:
            messagebox.showerror("Error", f"Invalid input: {e}")

if __name__ == "__main__":
    root = tk.Tk()

'''
This module contains the main application logic for finding the intersection of elements in a list of tuples using a GUI.
'''
import tkinter as tk
from tkinter import messagebox
import ast

def tuple_intersection(tuple_list):
    '''
    Computes the intersection of elements in a list of tuples.
    Parameters:
    tuple_list (list of tuples): The list of tuples to find the intersection.
    Returns:
    set: A set containing the intersection of elements.
    '''
    if not tuple_list:
        return set()
    # Convert each tuple to a set and find the intersection
    intersection_set = set(tuple_list[0])
    for tup in tuple_list[1:]:
        intersection_set.intersection_update(tup)
    return intersection_set

class TupleIntersectionApp:
    '''
    A GUI application for finding the intersection of elements in a list of tuples.
    '''
    def __init__(self, master):
        self.master = master
        master.title("Tuple Intersection Finder")
        self.label = tk.Label(master, text="Enter tuples (e.g., (1, 2), (2, 3), (2, 4)):")
        self.label.pack()
        self.entry = tk.Entry(master, width=50)
        self.entry.pack()
        self.find_button = tk.Button(master, text="Find Intersection", command=self.find_intersection)
        self.find_button.pack()
        self.result_label = tk.Label(master, text="Intersection: ")
        self.result_label.pack()

    def find_intersection(self):
        try:
            # Safely evaluate the input to convert it into a list of tuples
            tuple_list = ast.literal_eval(self.entry.get())
            if not isinstance(tuple_list, list) or not all(isinstance(t, tuple) for t in tuple_list):
                raise ValueError("Input must be a list of tuples.")
            # Find the intersection
            intersection = tuple_intersection(tuple_list)
            self.result_label.config(text=f"Intersection: {intersection}")
        except Exception as e:
            messagebox.showerror("Error", f"Invalid input: {e}")

if __name__ == "__main__":
    root = tk.Tk()
    app = TupleIntersectionApp(root)
    root.mainloop()

'''
This module contains the main application logic for finding the intersection of elements in a list of tuples using a GUI.
'''
import tkinter as tk
from tkinter import messagebox
import ast
def tuple_intersection(tuple_list):
    '''
    Computes the intersection of elements in a list of tuples.
    Parameters:
    tuple_list (list of tuples): The list of tuples to find the intersection.
    Returns:
    set: A set containing the intersection of elements.
    '''
    if not tuple_list:
        return set()
    # Convert each tuple to a set and find the intersection
    intersection_set = set(tuple_list[0])
    for tup in tuple_list[1:]:
        intersection_set.intersection_update(tup)
    return intersection_set
class TupleIntersectionApp:
    '''
    A GUI application for finding the intersection of elements in a list of tuples.
    '''
    def __init__(self, master):
        '''
        Initializes the GUI application.
        Parameters:
        master (Tk): The root window of the application.
        '''
        self.master = master
        master.title("Tuple Intersection Finder")
        self.label = tk.Label(master, text="Enter tuples (e.g., (1, 2), (2, 3), (2, 4)):")
        self.label.pack()
        self.entry = tk.Entry(master, width=50)
        self.entry.pack()
        self.find_button = tk.Button(master, text="Find Intersection", command=self.find_intersection)
        self.find_button.pack()
        self.result_label = tk.Label(master, text="Intersection: ")
        self.result_label.pack()
    def find_intersection(self):
        '''
        Handles the button click event to find the intersection of tuples.
        '''
        try:
            # Safely evaluate the input to convert it into a list of tuples
            tuple_list = ast.literal_eval(self.entry.get())
            if not isinstance(tuple_list, list) or not all(isinstance(t, tuple) for t in tuple_list):
                raise ValueError("Input must be a list of tuples.")
            # Find the intersection
            intersection = tuple_intersection(tuple_list)
            self.result_label.config(text=f"Intersection: {intersection}")
        except Exception as e:
            messagebox.showerror("Error", f"Invalid input: {e}")
if __name__ == "__main__":
    '''
    Entry point for the application.
    '''
    root = tk.Tk()
    app = TupleIntersectionApp(root)
    root.mainloop()

'''
This module contains the main application logic for finding the intersection of elements in a list of tuples using a GUI.
'''
import tkinter as tk
from tkinter import messagebox
import ast
def tuple_intersection(tuple_list):
    '''
    Computes the intersection of elements in a list of tuples.
    Parameters:
    tuple_list (list of tuples): The list of tuples to find the intersection.
    Returns:
    set: A set containing the intersection of elements.
    '''
    if not tuple_list:
        return set()
    # Convert each tuple to a set and find the intersection
    intersection_set = set(tuple_list[0])
    for tup in tuple_list[1:]:
        intersection_set.intersection_update(tup)
    return intersection_set
class TupleIntersectionApp:
    '''
    A GUI application for finding the intersection of elements in a list of tuples.
    '''
    def __init__(self, master):
        '''
        Initializes the GUI application.
        Parameters:
        master (Tk): The root window of the application.
        '''
        self.master = master
        master.title("Tuple Intersection Finder")
        self.label = tk.Label(master, text="Enter tuples (e.g., (1, 2), (2, 3), (2, 4)):")
        self.label.pack()
        self.entry = tk.Entry(master, width=50)
        self.entry.pack()
        self.find_button = tk.Button(master, text="Find Intersection", command=self.find_intersection)
        self.find_button.pack()
        self.result_label = tk.Label(master, text="Intersection: ")
        self.result_label.pack()
    def find_intersection(self):
        '''
        Handles the button click event to find the intersection of tuples.
        '''
        try:
            # Safely evaluate the input to convert it into a list of tuples
            tuple_list = ast.literal_eval(self.entry.get())
            if not isinstance(tuple_list, list) or not all(isinstance(t, tuple) for t in tuple_list):
                raise ValueError("Input must be a list of tuples.")
            # Find the intersection
            intersection = tuple_intersection(tuple_list)
            self.result_label.config(text=f"Intersection: {intersection}")
        except Exception as e:
            messagebox.showerror("Error", f"Invalid input: {e}")
if __name__ == "__main__":
    '''
    Entry point for the application.
    '''
    root = tk.Tk()
    app = TupleIntersectionApp(root)
    root.mainloop()
