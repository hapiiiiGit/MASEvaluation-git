'''

This file contains the main application code for a Python program that finds the pair with the highest product from a given array of integers. It includes a graphical user interface (GUI) built using tkinter.

'''

import tkinter as tk
from tkinter import messagebox


def find_pair_with_highest_product(arr):
    '''
    This function takes an array of integers as input and returns the pair of integers
    that produce the highest product, along with the product value.
    '''
    if len(arr) < 2:
        raise ValueError("Array must contain at least two integers.")

    # Sort the array to find the largest and smallest numbers
    arr.sort()

    # Calculate the product of the two largest numbers
    max1, max2 = arr[-1], arr[-2]
    max_product = max1 * max2

    # Calculate the product of the two smallest numbers (in case of negative numbers)
    min1, min2 = arr[0], arr[1]
    min_product = min1 * min2

    # Return the pair with the highest product
    if max_product > min_product:
        return (max1, max2, max_product)
    else:
        return (min1, min2, min_product)

```python
'''

This file contains the main application code for a Python program that finds the pair with the highest product from a given array of integers. It includes a graphical user interface (GUI) built using tkinter.

'''

import tkinter as tk
from tkinter import messagebox


def find_pair_with_highest_product(arr):
    '''
    This function takes an array of integers as input and returns the pair of integers
    that produce the highest product, along with the product value.
    '''
    if len(arr) < 2:
        raise ValueError("Array must contain at least two integers.")

    # Sort the array to find the largest and smallest numbers
    arr.sort()

    # Calculate the product of the two largest numbers
    max1, max2 = arr[-1], arr[-2]
    max_product = max1 * max2

    # Calculate the product of the two smallest numbers (in case of negative numbers)
    min1, min2 = arr[0], arr[1]
    min_product = min1 * min2

    # Return the pair with the highest product
    if max_product > min_product:
        return (max1, max2, max_product)
    else:
        return (min1, min2, min_product)


class MainApp:
    '''
    This class represents the main GUI application for finding the pair with the highest product.
    '''
    def __init__(self, root):
        self.root = root
        self.root.title("Highest Product Finder")

        # Input label and entry
        self.label = tk.Label(root, text="Enter integers separated by commas:")
        self.label.pack(pady=10)

        self.entry = tk.Entry(root, width=50)
        self.entry.pack(pady=10)

        # Submit button
        self.submit_button = tk.Button(root, text="Find Pair", command=self.find_highest_product)
        self.submit_button.pack(pady=10)

        # Result label
        self.result_label = tk.Label(root, text="", fg="blue")
        self.result_label.pack(pady=10)

    def find_highest_product(self):
        '''
        This method retrieves the input array, processes it, and displays the result.
        '''
        try:
            # Get input from the entry box
            input_text = self.entry.get()
            # Convert the input string to a list of integers
            arr = list(map(int, input_text.split(',')))

            # Find the pair with the highest product
            pair = find_pair_with_highest_product(arr)

            # Display the result
            self.display_result(pair)

        except ValueError as e:
            # Show an error message if the input is invalid
            messagebox.showerror("Error", str(e))

    def display_result(self, pair):
        '''
        This method displays the result in the GUI.
        '''
        self.result_label.config(
            text=f"The pair with the highest product is {pair[0]} and {pair[1]} with a product of {pair[2]}."
        )


if __name__ == "__main__":
    # Create the main tkinter window
    root = tk.Tk()
    app = MainApp(root)
    root.mainloop()

'''

import tkinter as tk
from tkinter import messagebox
def find_pair_with_highest_product(arr):
    '''
    This function takes an array of integers as input and returns the pair of integers
    that produce the highest product, along with the product value.
    '''
    if len(arr) < 2:
        raise ValueError("Array must contain at least two integers.")

    arr.sort()

    max1, max2 = arr[-1], arr[-2]
    max_product = max1 * max2

    min1, min2 = arr[0], arr[1]
    min_product = min1 * min2
    # Return the pair with the highest product
    if max_product > min_product:
        return (max1, max2, max_product)
    else:
        return (min1, min2, min_product)

'''
import tkinter as tk
from tkinter import messagebox
def find_pair_with_highest_product(arr):
    '''
    This function takes an array of integers as input and returns the pair of integers
    that produce the highest product, along with the product value.
    '''
    if len(arr) < 2:
        raise ValueError("Array must contain at least two integers.")
    # Sort the array to find the largest and smallest numbers
    arr.sort()
    # Calculate the product of the two largest numbers
    max1, max2 = arr[-1], arr[-2]
    max_product = max1 * max2
    # Calculate the product of the two smallest numbers (in case of negative numbers)
    min1, min2 = arr[0], arr[1]
    min_product = min1 * min2
    # Return the pair with the highest product
    if max_product > min_product:
        return (max1, max2, max_product)
    else:
        return (min1, min2, min_product)
class MainApp:
    '''
    This class represents the main GUI application for finding the pair with the highest product.
    '''
    def __init__(self, root):
        self.root = root
        self.root.title("Highest Product Finder")
        # Input label and entry
        self.label = tk.Label(root, text="Enter integers separated by commas:")
        self.label.pack(pady=10)
        self.entry = tk.Entry(root, width=50)
        self.entry.pack(pady=10)
        # Submit button
        self.submit_button = tk.Button(root, text="Find Pair", command=self.find_highest_product)
        self.submit_button.pack(pady=10)
        # Result label
        self.result_label = tk.Label(root, text="", fg="blue")
        self.result_label.pack(pady=10)
    def find_highest_product(self):
        '''
        This method retrieves the input array, processes it, and displays the result.
        '''
        try:
            # Get input from the entry box
            input_text = self.entry.get()
            # Convert the input string to a list of integers
            arr = list(map(int, input_text.split(',')))
            # Find the pair with the highest product
            pair = find_pair_with_highest_product(arr)
            # Display the result
            self.display_result(pair)
        except ValueError as e:
            # Show an error message if the input is invalid
            messagebox.showerror("Error", str(e))
    def display_result(self, pair):
        '''
        This method displays the result in the GUI.
        '''
        self.result_label.config(
            text=f"The pair with the highest product is {pair[0]} and {pair[1]} with a product of {pair[2]}."
        )
if __name__ == "__main__":
    # Create the main tkinter window

'''
This file contains the main application code for a Python program that finds the pair with the highest product from a given array of integers. It includes a graphical user interface (GUI) built using tkinter.
'''
import tkinter as tk
from tkinter import messagebox
def find_pair_with_highest_product(arr):
    '''
    This function takes an array of integers as input and returns the pair of integers
    that produce the highest product, along with the product value.
    '''
    if len(arr) < 2:
        raise ValueError("Array must contain at least two integers.")
    # Sort the array to find the largest and smallest numbers
    arr.sort()
    # Calculate the product of the two largest numbers
    max1, max2 = arr[-1], arr[-2]
    max_product = max1 * max2
    # Calculate the product of the two smallest numbers (in case of negative numbers)
    min1, min2 = arr[0], arr[1]
    min_product = min1 * min2
    # Return the pair with the highest product
    if max_product > min_product:
        return (max1, max2, max_product)
    else:
        return (min1, min2, min_product)
class MainApp:
    '''
    This class represents the main GUI application for finding the pair with the highest product.
    '''
    def __init__(self, root):
        self.root = root
        self.root.title("Highest Product Finder")
        # Input label and entry
        self.label = tk.Label(root, text="Enter integers separated by commas:")
        self.label.pack(pady=10)
        self.entry = tk.Entry(root, width=50)
        self.entry.pack(pady=10)
        # Submit button
        self.submit_button = tk.Button(root, text="Find Pair", command=self.find_highest_product)
        self.submit_button.pack(pady=10)
        # Result label
        self.result_label = tk.Label(root, text="", fg="blue")
        self.result_label.pack(pady=10)
    def find_highest_product(self):
        '''
        This method retrieves the input array, processes it, and displays the result.
        '''
        try:
            # Get input from the entry box
            input_text = self.entry.get()
            # Convert the input string to a list of integers
            arr = list(map(int, input_text.split(',')))
            # Find the pair with the highest product
            pair = find_pair_with_highest_product(arr)
            # Display the result
            self.display_result(pair)
        except ValueError as e:
            # Show an error message if the input is invalid
            messagebox.showerror("Error", str(e))
    def display_result(self, pair):
        '''
        This method displays the result in the GUI.
        '''
        self.result_label.config(
            text=f"The pair with the highest product is {pair[0]} and {pair[1]} with a product of {pair[2]}."
        )
if __name__ == "__main__":
    # Create the main tkinter window
    root = tk.Tk()
    app = MainApp(root)
    root.mainloop()

'''
This file contains the main application code for a Python program that finds the pair with the highest product from a given array of integers. It includes a graphical user interface (GUI) built using tkinter.
'''
import tkinter as tk
from tkinter import messagebox
def find_pair_with_highest_product(arr):
    '''
    This function takes an array of integers as input and returns the pair of integers
    that produce the highest product, along with the product value.
    '''
    if len(arr) < 2:
        raise ValueError("Array must contain at least two integers.")
    # Sort the array to find the largest and smallest numbers
    arr.sort()
    # Calculate the product of the two largest numbers
    max1, max2 = arr[-1], arr[-2]
    max_product = max1 * max2
    # Calculate the product of the two smallest numbers (in case of negative numbers)
    min1, min2 = arr[0], arr[1]
    min_product = min1 * min2
    # Return the pair with the highest product
    if max_product > min_product:
        return (max1, max2, max_product)
    else:
        return (min1, min2, min_product)
class MainApp:
    '''
    This class represents the main GUI application for finding the pair with the highest product.
    '''
    def __init__(self, root):
        self.root = root
        self.root.title("Highest Product Finder")
        # Input label and entry
        self.label = tk.Label(root, text="Enter integers separated by commas:")
        self.label.pack(pady=10)
        self.entry = tk.Entry(root, width=50)
        self.entry.pack(pady=10)
        # Submit button
        self.submit_button = tk.Button(root, text="Find Pair", command=self.find_highest_product)
        self.submit_button.pack(pady=10)
        # Result label
        self.result_label = tk.Label(root, text="", fg="blue")
        self.result_label.pack(pady=10)
    def find_highest_product(self):
        '''
        This method retrieves the input array, processes it, and displays the result.
        '''
        try:
            # Get input from the entry box
            input_text = self.entry.get()
            # Convert the input string to a list of integers
            arr = list(map(int, input_text.split(',')))
            # Find the pair with the highest product
            pair = find_pair_with_highest_product(arr)
            # Display the result
            self.display_result(pair)
        except ValueError as e:
            # Show an error message if the input is invalid
            messagebox.showerror("Error", str(e))
    def display_result(self, pair):
        '''
        This method displays the result in the GUI.
        '''
        self.result_label.config(
            text=f"The pair with the highest product is {pair[0]} and {pair[1]} with a product of {pair[2]}."
        )
if __name__ == "__main__":
    # Create the main tkinter window
    root = tk.Tk()
    app = MainApp(root)
    root.mainloop()
