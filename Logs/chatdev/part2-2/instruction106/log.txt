'''

This module contains the GUI application for finding the common first element in a list of tuples using Tkinter.

'''

class CommonElementFinder:
    '''
    A class to find the common first element in a list of tuples.
    '''
    
    def find_common_first_element(self, tuples_list):
        '''
        Finds the common first element in a list of tuples.
        
        :param tuples_list: List of tuples
        :return: The common first element or None if no common element exists
        '''
        if not tuples_list:
            return None
        
        first_elements = [t[0] for t in tuples_list]
        common_element = first_elements[0]
        
        for element in first_elements:
            if element != common_element:
                return None
        
        return common_element

import ast
class CommonElementFinder:
    '''

    def find_common_first_element(self, tuples_list):
        '''

'''
This module contains the GUI application for finding the most common first element in a list of tuples using Tkinter.
'''
import tkinter as tk
from tkinter import messagebox
import ast

class CommonElementFinder:
    '''
    A class to find the most common first element in a list of tuples.
    '''
    def find_common_first_element(self, tuples_list):
        '''
        Finds the most common first element in a list of tuples.
        :param tuples_list: List of tuples
        :return: The most common first element or None if no common element exists
        '''
        if not tuples_list:
            return None
        first_elements = [t[0] for t in tuples_list]
        element_counts = {element: first_elements.count(element) for element in set(first_elements)}
        most_common_element = max(element_counts, key=element_counts.get)
        return most_common_element if element_counts[most_common_element] > 1 else None

class CommonElementApp:
    '''
    A class to create a GUI application for finding the most common first element in a list of tuples.
    '''
    def __init__(self, root):
        '''
        Initializes the GUI components.
        :param root: The root window of the Tkinter application
        '''
        self.root = root
        self.root.title("Common First Element Finder")
        self.label = tk.Label(root, text="Enter list of tuples (e.g., [(1,2), (1,3), (1,4)]):")
        self.label.pack()
        self.entry = tk.Entry(root, width=50)
        self.entry.pack()
        self.find_button = tk.Button(root, text="Find", command=self.on_find_button_click)
        self.find_button.pack()
        self.result_label = tk.Label(root, text="")
        self.result_label.pack()
        self.finder = CommonElementFinder()

    def on_find_button_click(self):
        '''
        Handles the event when the user clicks the "Find" button.
        '''
        input_text = self.entry.get()
        try:
            tuples_list = ast.literal_eval(input_text)
            if not isinstance(tuples_list, list) or not all(isinstance(t, tuple) for t in tuples_list):
                raise ValueError
            common_element = self.finder.find_common_first_element(tuples_list)
            if common_element is not None:
                self.result_label.config(text=f"Common first element: {common_element}")
            else:
                self.result_label.config(text="No common first element found.")
        except (SyntaxError, ValueError):
            messagebox.showerror("Invalid Input", "Please enter a valid list of tuples.")

    def run(self):
        '''
        Starts the Tkinter main loop.
        '''
        self.root.mainloop()

if __name__ == "__main__":
    root = tk.Tk()
    app = CommonElementApp(root)
    app.run()

'''
This module contains the GUI application for finding the most common first element in a list of tuples using Tkinter.
'''
import tkinter as tk
from tkinter import messagebox
import ast
class CommonElementFinder:
    '''
    A class to find the most common first element in a list of tuples.
    '''
    def find_common_first_element(self, tuples_list):
        '''
        Finds the most common first element in a list of tuples.
        :param tuples_list: List of tuples
        :return: The most common first element or None if no common element exists
        '''
        if not tuples_list:
            return None
        first_elements = [t[0] for t in tuples_list]
        element_counts = {element: first_elements.count(element) for element in set(first_elements)}
        most_common_element = max(element_counts, key=element_counts.get)
        return most_common_element if element_counts[most_common_element] > 1 else None
class CommonElementApp:
    '''
    A class to create a GUI application for finding the most common first element in a list of tuples.
    '''
    def __init__(self, root):
        '''
        Initializes the GUI components.
        :param root: The root window of the Tkinter application
        '''
        self.root = root
        self.root.title("Common First Element Finder")
        self.label = tk.Label(root, text="Enter list of tuples (e.g., [(1,2), (1,3), (1,4)]):")

        self.result_label.pack()
        self.finder = CommonElementFinder()
    def on_find_button_click(self):
        '''
        Handles the event when the user clicks the "Find" button.
        '''
        input_text = self.entry.get()

            tuples_list = ast.literal_eval(input_text)
            if not isinstance(tuples_list, list) or not all(isinstance(t, tuple) for t in tuples_list):
                raise ValueError
            common_element = self.finder.find_common_first_element(tuples_list)
            if common_element is not None:
                self.result_label.config(text=f"Common first element: {common_element}")
            else:
                self.result_label.config(text="No common first element found.")
        except (SyntaxError, ValueError):

    def run(self):
        '''
        Starts the Tkinter main loop.
        '''
        self.root.mainloop()
if __name__ == "__main__":

'''
This module contains the GUI application for finding the most common first element in a list of tuples using Tkinter.
'''
import tkinter as tk
from tkinter import messagebox
import ast
class CommonElementFinder:
    '''
    A class to find the most common first element in a list of tuples.
    '''
    def find_common_first_element(self, tuples_list):
        '''
        Finds the most common first element in a list of tuples.
        :param tuples_list: List of tuples
        :return: The most common first element or None if no common element exists
        '''
        if not tuples_list:
            return None
        first_elements = [t[0] for t in tuples_list]
        element_counts = {element: first_elements.count(element) for element in set(first_elements)}
        most_common_element = max(element_counts, key=element_counts.get)
        return most_common_element if element_counts[most_common_element] > 1 else None
class CommonElementApp:
    '''
    A class to create a GUI application for finding the most common first element in a list of tuples.
    '''
    def __init__(self, root):
        '''
        Initializes the GUI components.
        :param root: The root window of the Tkinter application
        '''
        self.root = root
        self.root.title("Common First Element Finder")
        self.label = tk.Label(root, text="Enter list of tuples (e.g., [(1,2), (1,3), (1,4)]):")
        self.label.pack()
        self.entry = tk.Entry(root, width=50)
        self.entry.pack()
        self.find_button = tk.Button(root, text="Find", command=self.on_find_button_click)
        self.find_button.pack()
        self.result_label = tk.Label(root, text="")
        self.result_label.pack()
        self.finder = CommonElementFinder()
    def on_find_button_click(self):
        '''
        Handles the event when the user clicks the "Find" button.
        '''
        input_text = self.entry.get()
        try:
            tuples_list = ast.literal_eval(input_text)
            if not isinstance(tuples_list, list) or not all(isinstance(t, tuple) for t in tuples_list):
                raise ValueError
            common_element = self.finder.find_common_first_element(tuples_list)
            if common_element is not None:
                self.result_label.config(text=f"Common first element: {common_element}")
            else:
                self.result_label.config(text="No common first element found.")
        except (SyntaxError, ValueError):
            messagebox.showerror("Invalid Input", "Please enter a valid list of tuples.")
    def run(self):
        '''
        Starts the Tkinter main loop.
        '''
        self.root.mainloop()
if __name__ == "__main__":
    root = tk.Tk()
    app = CommonElementApp(root)
    app.run()
