import tkinter as tk
from tkinter import messagebox

def rotate_right(lst, n):
    '''
    Rotate a given list by n positions to the right.
    
    Parameters:
    lst (list): The list to be rotated.
    n (int): The number of positions to rotate the list.
    
    Returns:
    list: The rotated list.
    '''
    if not lst:
        return lst
    n = n % len(lst)  # Handle cases where n > len(lst)
    return lst[-n:] + lst[:-n]

class ListRotatorApp:
    '''
    A GUI application to rotate a list by a specified number of items to the right.
    '''
    def __init__(self, master):
        self.master = master
        master.title("List Rotator")

        self.label = tk.Label(master, text="Enter a list of numbers (comma-separated):")
        self.label.pack()

        self.entry = tk.Entry(master)
        self.entry.pack()

        self.label_n = tk.Label(master, text="Enter number of positions to rotate:")
        self.label_n.pack()

        self.entry_n = tk.Entry(master)
        self.entry_n.pack()

        self.rotate_button = tk.Button(master, text="Rotate", command=self.rotate_list)
        self.rotate_button.pack()

        self.result_label = tk.Label(master, text="")
        self.result_label.pack()

    def rotate_list(self):
        try:
            lst = list(map(int, self.entry.get().split(',')))
            n = int(self.entry_n.get())
            rotated_list = rotate_right(lst, n)
            self.display_result(rotated_list)
        except ValueError:
            messagebox.showerror("Invalid input", "Please enter valid numbers.")

    def display_result(self, result):
        self.result_label.config(text=f"Rotated List: {result}")

if __name__ == "__main__":
    root = tk.Tk()
    app = ListRotatorApp(root)
    root.mainloop()
```

'''
main.py
A Python application using Tkinter to rotate a list by a specified number of items to the right.
'''
import tkinter as tk
from tkinter import messagebox
def rotate_right(lst, n):
    '''
    Rotate a given list by n positions to the right.
    Parameters:
    lst (list): The list to be rotated.
    n (int): The number of positions to rotate the list.
    Returns:
    list: The rotated list.
    '''
    if not lst:
        return lst
    n = n % len(lst)  # Handle cases where n > len(lst)
    return lst[-n:] + lst[:-n]
class ListRotatorApp:
    '''
    A GUI application to rotate a list by a specified number of items to the right.
    '''
    def __init__(self, master):
        self.master = master
        master.title("List Rotator")
        self.label = tk.Label(master, text="Enter a list of numbers (comma-separated):")
        self.label.pack()
        self.entry = tk.Entry(master)
        self.entry.pack()
        self.label_n = tk.Label(master, text="Enter number of positions to rotate:")
        self.label_n.pack()
        self.entry_n = tk.Entry(master)
        self.entry_n.pack()
        self.rotate_button = tk.Button(master, text="Rotate", command=self.rotate_list)
        self.rotate_button.pack()
        self.result_label = tk.Label(master, text="")
        self.result_label.pack()
    def rotate_list(self):
        try:
            lst = list(map(int, self.entry.get().split(',')))
            n = int(self.entry_n.get())
            rotated_list = rotate_right(lst, n)
            self.display_result(rotated_list)
        except ValueError:
            messagebox.showerror("Invalid input", "Please enter valid numbers.")
    def display_result(self, result):
        self.result_label.config(text=f"Rotated List: {result}")
if __name__ == "__main__":
    root = tk.Tk()
    app = ListRotatorApp(root)
    root.mainloop()
