'''

This file contains the function to calculate the next perfect square greater than a given number.

'''

import math


def find_next_perfect_square(number):
    '''
    A function to calculate the next perfect square greater than the given number.

    Args:
        number (int): The input number.

    Returns:
        int: The next perfect square greater than the given number.
    '''
    # Find the square root of the number
    sqrt = math.sqrt(number)

    # Round up to the next integer
    next_int = math.ceil(sqrt)

    # Calculate the next perfect square
    next_square = next_int ** 2

    return next_square
```

'''

This is the main file for the application. It initializes the tkinter GUI, handles user input, and displays the next perfect square greater than a given number.

'''

import tkinter as tk
from tkinter import messagebox
from calculations import find_next_perfect_square


class MainApp:
    '''
    The main GUI application class that initializes and manages the tkinter interface.
    '''
    def __init__(self, root):
        self.root = root
        self.root.title("Next Perfect Square Finder")
        self.root.geometry("400x200")

        # Input Label
        self.label = tk.Label(root, text="Enter a number:", font=("Arial", 12))
        self.label.pack(pady=10)

        # Input Entry
        self.entry = tk.Entry(root, font=("Arial", 12))
        self.entry.pack(pady=5)

        # Calculate Button
        self.calculate_button = tk.Button(
            root, text="Find Next Perfect Square", font=("Arial", 12), command=self.calculate_next_square
        )
        self.calculate_button.pack(pady=10)

        # Result Label
        self.result_label = tk.Label(root, text="", font=("Arial", 12), fg="blue")
        self.result_label.pack(pady=10)

    def calculate_next_square(self):
        '''
        Retrieves user input, calculates the next perfect square using find_next_perfect_square, and displays the result.
        '''
        try:
            # Get the input number
            number = int(self.entry.get())

            # Calculate the next perfect square
            next_square = find_next_perfect_square(number)

            # Display the result
            self.result_label.config(text=f"The next perfect square is: {next_square}")
        except ValueError:
            # Handle invalid input
            messagebox.showerror("Invalid Input", "Please enter a valid integer.")


def main():
    '''
    The entry point of the application. It initializes the tkinter root and starts the application.
    '''
    root = tk.Tk()
    app = MainApp(root)
    root.mainloop()


if __name__ == "__main__":
    main()

'''

This file contains the function to calculate the next perfect square greater than a given number.

'''

import math


def find_next_perfect_square(number):
    '''
    A function to calculate the next perfect square greater than the given number.

    Args:
        number (int): The input number.

    Returns:
        int: The next perfect square greater than the given number.
    '''
    # Find the square root of the number
    sqrt = math.sqrt(number)

    # Round up to the next integer
    next_int = math.ceil(sqrt)

    # Calculate the next perfect square
    next_square = next_int ** 2

    return next_square

import math  
def find_next_perfect_square(number):  
    '''  
    A function to calculate the next perfect square greater than the given number.  
    Args:  
        number (int): The input number.  
    Returns:  
        int: The next perfect square greater than the given number.  
    '''  
    # Find the square root of the number  
    sqrt = math.sqrt(number)  
    # Round up to the next integer  
    next_int = math.ceil(sqrt)  
    # Calculate the next perfect square  
    next_square = (next_int + 1) ** 2  
    return next_square

import math  
def find_next_perfect_square(number):  
    '''  
    A function to calculate the next perfect square greater than the given number.  
    Args:  
        number (int): The input number.  
    Returns:  
        int: The next perfect square greater than the given number.  
    '''  
    # Find the square root of the number  
    sqrt = math.sqrt(number)  
    # Round up to the next integer  
    next_int = math.ceil(sqrt)  
    # Calculate the next perfect square  
    next_square = (next_int + 1) ** 2  
    return next_square

import math
def find_next_perfect_square(number):
    '''
    A function to calculate the next perfect square greater than the given number.
    Args:
        number (int): The input number.
    Returns:
        int: The next perfect square greater than the given number.
    '''
    # Find the square root of the number
    sqrt = math.sqrt(number)
    # Round up to the next integer
    next_int = math.ceil(sqrt)
    # Calculate the next perfect square
    next_square = (next_int + 1) ** 2
    return next_square

import tkinter as tk
from tkinter import messagebox
from calculations import find_next_perfect_square
class MainApp:
    '''
    The main GUI application class that initializes and manages the tkinter interface.
    '''
    def __init__(self, root):
        self.root = root
        self.root.title("Next Perfect Square Finder")
        self.root.geometry("400x200")
        # Input Label
        self.label = tk.Label(root, text="Enter a number:", font=("Arial", 12))
        self.label.pack(pady=10)
        # Input Entry
        self.entry = tk.Entry(root, font=("Arial", 12))
        self.entry.pack(pady=5)
        # Calculate Button
        self.calculate_button = tk.Button(
            root, text="Find Next Perfect Square", font=("Arial", 12), command=self.calculate_next_square
        )
        self.calculate_button.pack(pady=10)
        # Result Label
        self.result_label = tk.Label(root, text="", font=("Arial", 12), fg="blue")
        self.result_label.pack(pady=10)

            # Get the input number

            self.result_label.config(text=f"The next perfect square is: {next_square}")

            # Handle invalid input
            messagebox.showerror("Invalid Input", "Please enter a valid non-negative integer.")

    '''

    main()

'''
This file contains the function to calculate the next perfect square greater than a given number.
'''
import math
def find_next_perfect_square(number):
    '''
    A function to calculate the next perfect square greater than the given number.
    Args:
        number (int): The input number.
    Returns:
        int: The next perfect square greater than the given number.
    '''
    # Find the square root of the number
    sqrt = math.sqrt(number)
    # Round up to the next integer
    next_int = math.ceil(sqrt)
    # Calculate the next perfect square
    next_square = (next_int + 1) ** 2
    return next_square
