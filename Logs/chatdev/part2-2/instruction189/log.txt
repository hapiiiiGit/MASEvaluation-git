'''

This module provides a function to calculate the nth Catalan number.

'''

class CatalanCalculator:
    @staticmethod
    def calculate_nth_catalan(n):
        '''
        Calculate the nth Catalan number using a dynamic programming approach.

        :param n: The index of the Catalan number to calculate.
        :return: The nth Catalan number.
        '''
        if n == 0 or n == 1:
            return 1

        catalan = [0] * (n + 1)
        catalan[0], catalan[1] = 1, 1

        for i in range(2, n + 1):
            catalan[i] = 0
            for j in range(i):
                catalan[i] += catalan[j] * catalan[i - j - 1]

        return catalan[n]
```

'''

This module provides a function to calculate the nth Catalan number.

'''

class CatalanCalculator:
    @staticmethod
    def calculate_nth_catalan(n):
        '''
        Calculate the nth Catalan number using a dynamic programming approach.

        :param n: The index of the Catalan number to calculate.
        :return: The nth Catalan number.
        '''
        if n == 0 or n == 1:
            return 1

        catalan = [0] * (n + 1)
        catalan[0], catalan[1] = 1, 1

        for i in range(2, n + 1):
            catalan[i] = 0
            for j in range(i):
                catalan[i] += catalan[j] * catalan[i - j - 1]

        return catalan[n]
```

'''
This module provides a function to calculate the nth Catalan number.
'''
class CatalanCalculator:
    @staticmethod
    def calculate_nth_catalan(n):
        '''
        Calculate the nth Catalan number using a dynamic programming approach.
        :param n: The index of the Catalan number to calculate.
        :return: The nth Catalan number.
        '''
        if n == 0 or n == 1:
            return 1
        catalan = [0] * (n + 1)
        catalan[0], catalan[1] = 1, 1
        for i in range(2, n + 1):
            catalan[i] = 0
            for j in range(i):
                catalan[i] += catalan[j] * catalan[i - j - 1]
        return catalan[n]

'''
This module provides a function to calculate the nth Catalan number.
'''
class CatalanCalculator:
    @staticmethod
    def calculate_nth_catalan(n):
        '''
        Calculate the nth Catalan number using a dynamic programming approach.
        :param n: The index of the Catalan number to calculate.
        :return: The nth Catalan number.
        '''
        if n == 0 or n == 1:
            return 1
        catalan = [0] * (n + 1)
        catalan[0], catalan[1] = 1, 1
        for i in range(2, n + 1):
            catalan[i] = 0
            for j in range(i):
                catalan[i] += catalan[j] * catalan[i - j - 1]
        return catalan[n]

'''
This module provides a function to calculate the nth Catalan number.
'''
class CatalanCalculator:
    @staticmethod
    def calculate_nth_catalan(n):
        '''
        Calculate the nth Catalan number using a dynamic programming approach.
        :param n: The index of the Catalan number to calculate.
        :return: The nth Catalan number.
        '''
        if n == 0 or n == 1:
            return 1
        catalan = [0] * (n + 1)
        catalan[0], catalan[1] = 1, 1
        for i in range(2, n + 1):
            catalan[i] = 0
            for j in range(i):
                catalan[i] += catalan[j] * catalan[i - j - 1]
        return catalan[n]

'''
This module provides a function to calculate the nth Catalan number.
'''
class CatalanCalculator:
    @staticmethod
    def calculate_nth_catalan(n):
        '''
        Calculate the nth Catalan number using a dynamic programming approach.
        :param n: The index of the Catalan number to calculate.
        :return: The nth Catalan number.
        '''
        if n == 0 or n == 1:
            return 1
        catalan = [0] * (n + 1)
        catalan[0], catalan[1] = 1, 1
        for i in range(2, n + 1):
            catalan[i] = 0
            for j in range(i):
                catalan[i] += catalan[j] * catalan[i - j - 1]
        return catalan[n]

'''
This script sets up a GUI application to calculate the nth Catalan number using tkinter.

import tkinter as tk
from tkinter import messagebox
from catalan_calculator import CatalanCalculator
class CatalanApp:
    def __init__(self, root):
        '''
        Initialize the GUI application.
        :param root: The root window of the tkinter application.
        '''
        self.root = root
        self.root.title("Catalan Number Calculator")
        self.label = tk.Label(root, text="Enter n:")
        self.label.pack(pady=10)
        self.entry = tk.Entry(root)
        self.entry.pack(pady=10)
        self.calculate_button = tk.Button(root, text="Calculate", command=self.calculate_catalan)
        self.calculate_button.pack(pady=10)
        self.result_label = tk.Label(root, text="")
        self.result_label.pack(pady=10)
    def calculate_catalan(self):
        '''
        Calculate the nth Catalan number and display the result.
        '''
        try:
            n = int(self.entry.get())
            if n < 0:
                raise ValueError("n must be a non-negative integer.")
            result = CatalanCalculator.calculate_nth_catalan(n)
            self.result_label.config(text=f"Catalan number C({n}) = {result}")
        except ValueError as e:
            messagebox.showerror("Invalid Input", str(e))
def main():
    '''
    Main function to run the Catalan number calculator application.
    '''
    root = tk.Tk()
    app = CatalanApp(root)
    root.mainloop()
