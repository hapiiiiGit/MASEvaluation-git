```python
'''

main.py

'''

import tkinter as tk
from tkinter import messagebox

def subtract_lists(list1, list2):
    '''
    Subtracts two lists element-wise.
    
    Parameters:
    list1 (list): The first list of numbers.
    list2 (list): The second list of numbers.
    
    Returns:
    list: A new list containing the element-wise subtraction of list1 and list2.
    '''
    if len(list1) != len(list2):
        raise ValueError("Both lists must have the same length.")
    return [a - b for a, b in zip(list1, list2)]

class MainApp:
    '''
    Main application class for the GUI.
    '''
    def __init__(self, root):
        self.root = root
        self.root.title("List Subtraction")

        self.label1 = tk.Label(root, text="Enter first list (comma-separated):")
        self.label1.pack()

        self.entry1 = tk.Entry(root, width=50)
        self.entry1.pack()

        self.label2 = tk.Label(root, text="Enter second list (comma-separated):")
        self.label2.pack()

        self.entry2 = tk.Entry(root, width=50)
        self.entry2.pack()

        self.submit_button = tk.Button(root, text="Subtract", command=self.on_submit)
        self.submit_button.pack()

        self.result_label = tk.Label(root, text="Result:")
        self.result_label.pack()

    def on_submit(self):
        '''
        Handles the submit button click event.
        '''
        try:
            list1 = list(map(int, self.entry1.get().split(',')))
            list2 = list(map(int, self.entry2.get().split(',')))
            result = subtract_lists(list1, list2)
            self.result_label.config(text=f"Result: {result}")
        except ValueError as e:
            messagebox.showerror("Error", str(e))

if __name__ == "__main__":
    root = tk.Tk()
    app = MainApp(root)
    root.mainloop()
```

### Explanation

- **subtract_lists**: This function takes two lists as input and returns a new list with the element-wise subtraction. It raises a `ValueError` if the lists are not of the same length.

- **MainApp**: This class sets up the GUI using `tkinter`. It includes input fields for the user to enter two lists, a button to perform the subtraction, and a label to display the result.

- **on_submit**: This method is triggered when the user clicks the "Subtract" button. It reads the input lists, performs the subtraction using `subtract_lists`, and displays the result. If an error occurs (e.g., lists of different lengths), it shows an error message.

This code provides a complete, functional application that allows users to input two lists and see their element-wise subtraction.

import tkinter as tk
from tkinter import messagebox

def subtract_lists(list1, list2):
    '''
    Subtracts two lists element-wise.
    
    Parameters:
    list1 (list): The first list of numbers.
    list2 (list): The second list of numbers.
    
    Returns:
    list: A new list containing the element-wise subtraction of list1 and list2.
    '''
    if len(list1) != len(list2):
        raise ValueError("Both lists must have the same length.")
    return [a - b for a, b in zip(list1, list2)]

class MainApp:
    '''
    Main application class for the GUI.
    '''
    def __init__(self, root):
        self.root = root
        self.root.title("List Subtraction")

        self.label1 = tk.Label(root, text="Enter first list (comma-separated):")
        self.label1.pack()

        self.entry1 = tk.Entry(root, width=50)
        self.entry1.pack()

        self.label2 = tk.Label(root, text="Enter second list (comma-separated):")
        self.label2.pack()

        self.entry2 = tk.Entry(root, width=50)
        self.entry2.pack()

        self.submit_button = tk.Button(root, text="Subtract", command=self.on_submit)
        self.submit_button.pack()

        self.result_label = tk.Label(root, text="Result:")
        self.result_label.pack()

    def on_submit(self):
        '''
        Handles the submit button click event.
        '''
        try:
            list1 = list(map(int, self.entry1.get().split(',')))
            list2 = list(map(int, self.entry2.get().split(',')))
            result = subtract_lists(list1, list2)
            self.result_label.config(text=f"Result: {result}")

if __name__ == "__main__":
    root = tk.Tk()

import tkinter as tk
from tkinter import messagebox

def subtract_lists(list1, list2):
    '''
    Subtracts two lists element-wise.
    Parameters:
    list1 (list): The first list of numbers.
    list2 (list): The second list of numbers.
    Returns:
    list: A new list containing the element-wise subtraction of list1 and list2.
    '''
    if len(list1) != len(list2):
        raise ValueError("Both lists must have the same length.")
    return [a - b for a, b in zip(list1, list2)]

class MainApp:
    '''
    Main application class for the GUI.
    '''
    def __init__(self, root):
        self.root = root
        self.root.title("List Subtraction")
        
        # First list input
        self.label1 = tk.Label(root, text="Enter first list (comma-separated):")
        self.label1.pack()
        self.entry1 = tk.Entry(root, width=50)
        self.entry1.pack()
        
        # Second list input
        self.label2 = tk.Label(root, text="Enter second list (comma-separated):")
        self.label2.pack()
        self.entry2 = tk.Entry(root, width=50)
        self.entry2.pack()
        
        # Submit button
        self.submit_button = tk.Button(root, text="Subtract", command=self.on_submit)
        self.submit_button.pack()
        
        # Result display
        self.result_label = tk.Label(root, text="Result:")
        self.result_label.pack()

    def on_submit(self):
        '''
        Handles the submit button click event.
        '''
        try:
            # Parse input lists
            list1 = list(map(int, self.entry1.get().split(',')))
            list2 = list(map(int, self.entry2.get().split(',')))
            
            # Perform subtraction
            result = subtract_lists(list1, list2)
            
            # Display result
            self.result_label.config(text=f"Result: {result}")
        except ValueError as e:
            # Display error message
            messagebox.showerror("Error", str(e))

if __name__ == "__main__":
    root = tk.Tk()
    app = MainApp(root)
    root.mainloop()

import tkinter as tk
from tkinter import messagebox

def subtract_lists(list1, list2):
    '''
    Subtracts two lists element-wise.
    
    Parameters:
    list1 (list): The first list of numbers.
    list2 (list): The second list of numbers.
    
    Returns:
    list: A new list containing the element-wise subtraction of list1 and list2.

    ValueError: If the lists are not of the same length.
    '''
    if len(list1) != len(list2):

'''
main.py
This script provides a GUI application for subtracting two lists element-wise.
Users can input two lists of numbers, and the application will display the result
of their element-wise subtraction.

import tkinter as tk
from tkinter import messagebox
def subtract_lists(list1, list2):
    '''
    Subtracts two lists element-wise.
    Parameters:
    list1 (list): The first list of numbers.
    list2 (list): The second list of numbers.
    Returns:
    list: A new list containing the element-wise subtraction of list1 and list2.
    Raises:
    ValueError: If the lists are not of the same length.
    '''
    if len(list1) != len(list2):
        raise ValueError("Both lists must have the same length.")
    return [a - b for a, b in zip(list1, list2)]
class MainApp:
    '''
    Main application class for the GUI.
    '''
    def __init__(self, root):
        '''
        Initializes the main application window.
        Parameters:
        root (Tk): The root window of the Tkinter application.
        '''
        self.root = root
        self.root.title("List Subtraction")
        # First list input
        self.label1 = tk.Label(root, text="Enter first list (comma-separated):")
        self.label1.pack()
        self.entry1 = tk.Entry(root, width=50)
        self.entry1.pack()
        # Second list input
        self.label2 = tk.Label(root, text="Enter second list (comma-separated):")
        self.label2.pack()
        self.entry2 = tk.Entry(root, width=50)
        self.entry2.pack()
        # Submit button
        self.submit_button = tk.Button(root, text="Subtract", command=self.on_submit)
        self.submit_button.pack()
        # Result display
        self.result_label = tk.Label(root, text="Result:")
        self.result_label.pack()
    def on_submit(self):
        '''
        Handles the submit button click event.
        Parses the input lists, performs subtraction, and displays the result.
        '''
        try:
            # Parse input lists
            list1 = list(map(int, self.entry1.get().split(',')))
            list2 = list(map(int, self.entry2.get().split(',')))
            # Perform subtraction
            result = subtract_lists(list1, list2)
            # Display result
            self.result_label.config(text=f"Result: {result}")
        except ValueError as e:
            # Display error message
            messagebox.showerror("Error", str(e))
if __name__ == "__main__":
    root = tk.Tk()

'''
main.py
This script provides a GUI application for subtracting two lists element-wise.
Users can input two lists of numbers, and the application will display the result
of their element-wise subtraction.
'''
import tkinter as tk
from tkinter import messagebox
def subtract_lists(list1, list2):
    '''
    Subtracts two lists element-wise.
    Parameters:
    list1 (list): The first list of numbers.
    list2 (list): The second list of numbers.
    Returns:
    list: A new list containing the element-wise subtraction of list1 and list2.
    Raises:
    ValueError: If the lists are not of the same length.
    '''
    if len(list1) != len(list2):
        raise ValueError("Both lists must have the same length.")
    return [a - b for a, b in zip(list1, list2)]
class MainApp:
    '''
    Main application class for the GUI.
    '''
    def __init__(self, root):
        '''
        Initializes the main application window.
        Parameters:
        root (Tk): The root window of the Tkinter application.
        '''
        self.root = root
        self.root.title("List Subtraction")
        # First list input
        self.label1 = tk.Label(root, text="Enter first list (comma-separated):")
        self.label1.pack()
        self.entry1 = tk.Entry(root, width=50)
        self.entry1.pack()
        # Second list input
        self.label2 = tk.Label(root, text="Enter second list (comma-separated):")
        self.label2.pack()
        self.entry2 = tk.Entry(root, width=50)
        self.entry2.pack()
        # Submit button
        self.submit_button = tk.Button(root, text="Subtract", command=self.on_submit)
        self.submit_button.pack()
        # Result display
        self.result_label = tk.Label(root, text="Result:")
        self.result_label.pack()
    def on_submit(self):
        '''
        Handles the submit button click event.
        Parses the input lists, performs subtraction, and displays the result.
        '''
        try:
            # Parse input lists
            list1 = list(map(int, self.entry1.get().split(',')))
            list2 = list(map(int, self.entry2.get().split(',')))
            # Perform subtraction
            result = subtract_lists(list1, list2)
            # Display result
            self.result_label.config(text=f"Result: {result}")
        except ValueError as e:
            # Display error message
            messagebox.showerror("Error", str(e))
if __name__ == "__main__":
    root = tk.Tk()
    app = MainApp(root)
    root.mainloop()
