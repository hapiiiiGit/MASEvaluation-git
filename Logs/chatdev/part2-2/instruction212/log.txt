Main application file for regex search with GUI using tkinter.

'''

import re
import tkinter as tk
from tkinter import messagebox

class RegexSearchApp:
    '''
    A class to create a GUI application for searching a regex pattern in a string.
    '''
    def __init__(self, master):
        '''
        Initialize the GUI components.
        '''
        self.master = master
        master.title("Regex Search Application")

        # Input fields
        self.label_text = tk.Label(master, text="Enter Text:")
        self.label_text.grid(row=0, column=0, padx=5, pady=5)

        self.entry_text = tk.Entry(master, width=50)
        self.entry_text.grid(row=0, column=1, padx=5, pady=5)

        self.label_pattern = tk.Label(master, text="Enter Regex Pattern:")
        self.label_pattern.grid(row=1, column=0, padx=5, pady=5)

        self.entry_pattern = tk.Entry(master, width=50)
        self.entry_pattern.grid(row=1, column=1, padx=5, pady=5)

        # Search button
        self.search_button = tk.Button(master, text="Search", command=self.search_pattern)
        self.search_button.grid(row=2, column=0, columnspan=2, pady=10)

        # Result display
        self.result_label = tk.Label(master, text="Result:")
        self.result_label.grid(row=3, column=0, padx=5, pady=5)

        self.result_text = tk.Text(master, height=4, width=50, state='disabled')
        self.result_text.grid(row=3, column=1, padx=5, pady=5)

    def search_pattern(self):
        '''
        Perform the regex search and update the GUI with results.
        '''
        text = self.entry_text.get()
        pattern = self.entry_pattern.get()

        try:
            match = search_regex(text, pattern)
            if match:
                result = f"Match: {match[0]}\nStart Index: {match[1]}\nEnd Index: {match[2]}"
            else:
                result = "No match found."
        except re.error as e:
            result = f"Invalid regex pattern: {e}"

        self.result_text.config(state='normal')
        self.result_text.delete(1.0, tk.END)
        self.result_text.insert(tk.END, result)
        self.result_text.config(state='disabled')

def search_regex(text, pattern):
    '''
    Search for a regex pattern in a given text.

    Parameters:
    text (str): The text to search within.
    pattern (str): The regex pattern to search for.

    Returns:
    tuple: A tuple containing the matching substring, start index, and end index, or None if no match is found.
    '''
    match = re.search(pattern, text)
    if match:
        return (match.group(), match.start(), match.end())
    return None

if __name__ == "__main__":
    root = tk.Tk()
    app = RegexSearchApp(root)
    root.mainloop()

'''
Main application file for regex search with GUI using tkinter.
'''
import re
import tkinter as tk
from tkinter import messagebox
class RegexSearchApp:
    '''
    A class to create a GUI application for searching a regex pattern in a string.
    '''
    def __init__(self, master):
        '''
        Initialize the GUI components.
        '''
        self.master = master
        master.title("Regex Search Application")
        # Input fields
        self.label_text = tk.Label(master, text="Enter Text:")
        self.label_text.grid(row=0, column=0, padx=5, pady=5)
        self.entry_text = tk.Entry(master, width=50)
        self.entry_text.grid(row=0, column=1, padx=5, pady=5)
        self.label_pattern = tk.Label(master, text="Enter Regex Pattern:")
        self.label_pattern.grid(row=1, column=0, padx=5, pady=5)
        self.entry_pattern = tk.Entry(master, width=50)
        self.entry_pattern.grid(row=1, column=1, padx=5, pady=5)
        # Search button
        self.search_button = tk.Button(master, text="Search", command=self.search_pattern)
        self.search_button.grid(row=2, column=0, columnspan=2, pady=10)
        # Result display
        self.result_label = tk.Label(master, text="Result:")
        self.result_label.grid(row=3, column=0, padx=5, pady=5)
        self.result_text = tk.Text(master, height=4, width=50, state='disabled')
        self.result_text.grid(row=3, column=1, padx=5, pady=5)
    def search_pattern(self):
        '''
        Perform the regex search and update the GUI with results.
        '''
        text = self.entry_text.get()
        pattern = self.entry_pattern.get()
        try:
            match = search_regex(text, pattern)
            if match:
                result = f"Match: {match[0]}\nStart Index: {match[1]}\nEnd Index: {match[2]}"
            else:
                result = "No match found."
        except re.error as e:
            result = f"Invalid regex pattern: {e}"
        self.result_text.config(state='normal')
        self.result_text.delete(1.0, tk.END)
        self.result_text.insert(tk.END, result)
        self.result_text.config(state='disabled')
def search_regex(text, pattern):
    '''
    Search for a regex pattern in a given text.
    Parameters:
    text (str): The text to search within.
    pattern (str): The regex pattern to search for.
    Returns:
    tuple: A tuple containing the matching substring, start index, and end index, or None if no match is found.
    '''
    match = re.search(pattern, text)
    if match:
        return (match.group(), match.start(), match.end())
    return None
if __name__ == "__main__":
    root = tk.Tk()
    app = RegexSearchApp(root)
    root.mainloop()

'''
Main application file for regex search with GUI using tkinter.

import re
import tkinter as tk
from tkinter import messagebox
class RegexSearchApp:
    '''
    A class to create a GUI application for searching a regex pattern in a string.
    '''
    def __init__(self, master):
        '''
        Initialize the GUI components.
        '''
        self.master = master
        master.title("Regex Search Application")
        # Input fields
        self.label_text = tk.Label(master, text="Enter Text:")
        self.label_text.grid(row=0, column=0, padx=5, pady=5)
        self.entry_text = tk.Entry(master, width=50)
        self.entry_text.grid(row=0, column=1, padx=5, pady=5)
        self.label_pattern = tk.Label(master, text="Enter Regex Pattern:")
        self.label_pattern.grid(row=1, column=0, padx=5, pady=5)
        self.entry_pattern = tk.Entry(master, width=50)
        self.entry_pattern.grid(row=1, column=1, padx=5, pady=5)
        # Search button
        self.search_button = tk.Button(master, text="Search", command=self.search_pattern)
        self.search_button.grid(row=2, column=0, columnspan=2, pady=10)
        # Result display
        self.result_label = tk.Label(master, text="Result:")
        self.result_label.grid(row=3, column=0, padx=5, pady=5)
        self.result_text = tk.Text(master, height=4, width=50, state='disabled')
        self.result_text.grid(row=3, column=1, padx=5, pady=5)
    def search_pattern(self):
        '''
        Perform the regex search and update the GUI with results.
        '''
        text = self.entry_text.get()
        pattern = self.entry_pattern.get()
        try:
            match = search_regex(text, pattern)
            if match:
                result = f"Match: {match[0]}\nStart Index: {match[1]}\nEnd Index: {match[2]}"
            else:
                result = "No match found."
        except re.error as e:
            result = f"Invalid regex pattern: {e}"
        self.result_text.config(state='normal')
        self.result_text.delete(1.0, tk.END)
        self.result_text.insert(tk.END, result)
        self.result_text.config(state='disabled')
def search_regex(text, pattern):
    '''
    Search for a regex pattern in a given text.
    Parameters:
    text (str): The text to search within.
    pattern (str): The regex pattern to search for.
    Returns:
    tuple: A tuple containing the matching substring, start index, and end index, or None if no match is found.
    '''
    match = re.search(pattern, text)
    if match:
        return (match.group(), match.start(), match.end())
    return None
if __name__ == "__main__":
    root = tk.Tk()
    app = RegexSearchApp(root)
    root.mainloop()

'''
Main application file for regex search with GUI using tkinter.
'''
import re
import tkinter as tk
from tkinter import messagebox
class RegexSearchApp:
    '''
    A class to create a GUI application for searching a regex pattern in a string.
    '''
    def __init__(self, master):
        '''
        Initialize the GUI components.
        '''
        self.master = master
        master.title("Regex Search Application")
        # Input fields
        self.label_text = tk.Label(master, text="Enter Text:")
        self.label_text.grid(row=0, column=0, padx=5, pady=5)
        self.entry_text = tk.Entry(master, width=50)
        self.entry_text.grid(row=0, column=1, padx=5, pady=5)
        self.label_pattern = tk.Label(master, text="Enter Regex Pattern:")
        self.label_pattern.grid(row=1, column=0, padx=5, pady=5)
        self.entry_pattern = tk.Entry(master, width=50)
        self.entry_pattern.grid(row=1, column=1, padx=5, pady=5)
        # Search button
        self.search_button = tk.Button(master, text="Search", command=self.search_pattern)
        self.search_button.grid(row=2, column=0, columnspan=2, pady=10)
        # Result display
        self.result_label = tk.Label(master, text="Result:")
        self.result_label.grid(row=3, column=0, padx=5, pady=5)
        self.result_text = tk.Text(master, height=4, width=50, state='disabled')
        self.result_text.grid(row=3, column=1, padx=5, pady=5)
    def search_pattern(self):
        '''
        Perform the regex search and update the GUI with results.
        '''
        text = self.entry_text.get()
        pattern = self.entry_pattern.get()
        try:
            match = search_regex(text, pattern)
            if match:
                result = f"Match: {match[0]}\nStart Index: {match[1]}\nEnd Index: {match[2]}"
            else:
                result = "No match found."
        except re.error as e:
            result = f"Invalid regex pattern: {e}"
        self.result_text.config(state='normal')
        self.result_text.delete(1.0, tk.END)
        self.result_text.insert(tk.END, result)
        self.result_text.config(state='disabled')
def search_regex(text, pattern):
    '''
    Search for a regex pattern in a given text.
    Parameters:
    text (str): The text to search within.
    pattern (str): The regex pattern to search for.
    Returns:
    tuple: A tuple containing the matching substring, start index, and end index, or None if no match is found.
    '''
    match = re.search(pattern, text)
    if match:
        return (match.group(), match.start(), match.end())
    return None
if __name__ == "__main__":
    root = tk.Tk()
    app = RegexSearchApp(root)
    root.mainloop()

'''
Main application file for regex search with GUI using tkinter.
'''
import re
import tkinter as tk
from tkinter import messagebox
class RegexSearchApp:
    '''
    A class to create a GUI application for searching a regex pattern in a string.
    '''
    def __init__(self, master):
        '''
        Initialize the GUI components.
        '''
        self.master = master
        master.title("Regex Search Application")
        # Input fields
        self.label_text = tk.Label(master, text="Enter Text:")
        self.label_text.grid(row=0, column=0, padx=5, pady=5)
        self.entry_text = tk.Entry(master, width=50)
        self.entry_text.grid(row=0, column=1, padx=5, pady=5)
        self.label_pattern = tk.Label(master, text="Enter Regex Pattern:")
        self.label_pattern.grid(row=1, column=0, padx=5, pady=5)
        self.entry_pattern = tk.Entry(master, width=50)
        self.entry_pattern.grid(row=1, column=1, padx=5, pady=5)
        # Search button
        self.search_button = tk.Button(master, text="Search", command=self.search_pattern)
        self.search_button.grid(row=2, column=0, columnspan=2, pady=10)
        # Result display
        self.result_label = tk.Label(master, text="Result:")
        self.result_label.grid(row=3, column=0, padx=5, pady=5)
        self.result_text = tk.Text(master, height=4, width=50, state='disabled')
        self.result_text.grid(row=3, column=1, padx=5, pady=5)
    def search_pattern(self):
        '''
        Perform the regex search and update the GUI with results.
        '''
        text = self.entry_text.get()
        pattern = self.entry_pattern.get()
        try:
            match = search_regex(text, pattern)
            if match:
                result = f"Match: {match[0]}\nStart Index: {match[1]}\nEnd Index: {match[2]}"
            else:
                result = "No match found."
        except re.error as e:
            result = f"Invalid regex pattern: {e}"
        self.result_text.config(state='normal')
        self.result_text.delete(1.0, tk.END)
        self.result_text.insert(tk.END, result)
        self.result_text.config(state='disabled')
def search_regex(text, pattern):
    '''
    Search for a regex pattern in a given text.
    Parameters:
    text (str): The text to search within.
    pattern (str): The regex pattern to search for.
    Returns:
    tuple: A tuple containing the matching substring, start index, and end index, or None if no match is found.
    '''
    match = re.search(pattern, text)
    if match:
        return (match.group(), match.start(), match.end())
    return None
if __name__ == "__main__":
    root = tk.Tk()
    app = RegexSearchApp(root)
    root.mainloop()
