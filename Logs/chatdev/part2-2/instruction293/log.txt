```python
'''

This module contains the SubstringFinder class, which provides functionality to find occurrences and positions of substrings within a string.

'''

class SubstringFinder:
    @staticmethod
    def find_substring_occurrences(string, substring):
        '''
        Finds all occurrences and positions of the substring in the string.
        
        Parameters:
            string (str): The main string to search within.
            substring (str): The substring to search for.
        
        Returns:
            list: A list of positions where the substring occurs, or None if no match is found.
        '''
        if not substring:
            return None
        
        positions = []
        index = string.find(substring)
        while index != -1:
            positions.append(index)
            index = string.find(substring, index + 1)
        
        return positions if positions else None

'''
This module contains the SubstringFinder class, which provides functionality to find occurrences and positions of substrings within a string.
'''
class SubstringFinder:
    @staticmethod
    def find_substring_occurrences(string, substring):
        '''
        Finds all occurrences and positions of the substring in the string.
        Parameters:
            string (str): The main string to search within.
            substring (str): The substring to search for.
        Returns:
            list: A list of positions where the substring occurs, or None if no match is found.
        '''
        if not substring:
            return None
        positions = []
        index = string.find(substring)
        while index != -1:
            positions.append(index)
            index = string.find(substring, index + 1)
        return positions if positions else None

'''
This module contains the SubstringFinder class, which provides functionality to find occurrences and positions of substrings within a string.
'''
class SubstringFinder:
    @staticmethod
    def find_substring_occurrences(string, substring):
        '''
        Finds all occurrences and positions of the substring in the string.
        Parameters:
            string (str): The main string to search within.
            substring (str): The substring to search for.
        Returns:
            list: A list of positions where the substring occurs, or None if no match is found.
        '''
        if not substring:
            return None
        positions = []
        index = string.find(substring)
        while index != -1:
            positions.append(index)
            index = string.find(substring, index + 1)
        return positions if positions else None

'''
This module contains the SubstringFinder class, which provides functionality to find occurrences and positions of substrings within a string.
'''
class SubstringFinder:
    @staticmethod
    def find_substring_occurrences(string, substring):
        '''
        Finds all occurrences and positions of the substring in the string.
        Parameters:
            string (str): The main string to search within.
            substring (str): The substring to search for.
        Returns:
            list: A list of positions where the substring occurs, or None if no match is found.
        '''
        if not substring:
            return None
        positions = []
        index = string.find(substring)
        while index != -1:
            positions.append(index)
            index = string.find(substring, index + 1)
        return positions if positions else None

'''
This module contains the SubstringFinder class, which provides functionality to find occurrences and positions of substrings within a string.
'''
class SubstringFinder:
    @staticmethod
    def find_substring_occurrences(string, substring):
        '''
        Finds all occurrences and positions of the substring in the string.
        Parameters:
            string (str): The main string to search within.
            substring (str): The substring to search for.
        Returns:
            list: A list of positions where the substring occurs, or None if no match is found.
        '''
        if not substring:
            return None  # Return None if the substring is empty
        positions = []
        index = string.find(substring)
        # Loop to find all occurrences of the substring
        while index != -1:
            positions.append(index)
            index = string.find(substring, index + 1)
        return positions if positions else None  # Return None if no positions are found

```python
'''
This module contains the SubstringFinder class, which provides functionality to find occurrences and positions of substrings within a string.
'''
class SubstringFinder:
    @staticmethod
    def find_substring_occurrences(string, substring):
        '''
        Finds all occurrences and positions of the substring in the string.
        Parameters:
            string (str): The main string to search within.
            substring (str): The substring to search for.
        Returns:
            list: A list of positions where the substring occurs, or None if no match is found.
        '''
        if not substring:
            return None  # Return None if the substring is empty
        positions = []
        index = string.find(substring)
        # Loop to find all occurrences of the substring
        while index != -1:
            positions.append(index)
            index = string.find(substring, index + 1)
        return positions if positions else None  # Return None if no positions are found

'''
This module contains the SubstringFinder class, which provides functionality to find occurrences and positions of substrings within a string.
'''
class SubstringFinder:
    @staticmethod
    def find_substring_occurrences(string, substring):
        '''
        Finds all occurrences and positions of the substring in the string.
        Parameters:
            string (str): The main string to search within.
            substring (str): The substring to search for.
        Returns:
            list: A list of positions where the substring occurs, or None if no match is found.
        '''
        if not substring:
            return None  # Return None if the substring is empty
        positions = []
        index = string.find(substring)
        # Loop to find all occurrences of the substring
        while index != -1:
            positions.append(index)
            index = string.find(substring, index + 1)
        return positions if positions else None  # Return None if no positions are found

# Substring Finder Application

Welcome to the **Substring Finder Application**! This software is designed to help users find the occurrences and positions of substrings within a given string. If no match is found, the application will return `None`. The application features a user-friendly graphical interface for easy interaction.

---

## Main Functions of the Software

### 1. **Substring Finder**
The core functionality of the application is to locate all occurrences of a specified substring within a main string. It provides:
- **Positions of occurrences**: Displays the indices where the substring appears in the main string.
- **No match handling**: Returns `None` if the substring does not exist in the main string.

### 2. **Graphical User Interface (GUI)**
The application includes a simple and intuitive GUI for users to:
- Input the main string and substring.
- Click a button to find occurrences.
- View results in a dedicated output field.

---

## Installation Guide

### Step 1: Install Python
Ensure that Python version 3.6 or higher is installed on your system. You can download Python from [python.org](https://www.python.org/).

```bash
pip install -r requirements.txt
```

This will ensure all necessary dependencies are installed.

---

## How to Use the Application

### Step 1: Clone or Download the Repository
Download the project files to your local machine.

### Step 2: Run the Application
Navigate to the directory containing the project files and execute the following command:

```bash
python main.py
```

This will launch the graphical user interface.

### Step 3: Use the GUI
1. **Enter the Main String**: Type the string you want to search within.
2. **Enter the Substring**: Type the substring you want to find.
3. **Click "Find Occurrences"**: Press the button to search for the substring.
4. **View Results**: The positions of the substring occurrences will be displayed in the output field. If no match is found, the output will display `None`.
