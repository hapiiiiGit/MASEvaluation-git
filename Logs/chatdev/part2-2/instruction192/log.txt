'''

This file contains the main application logic for the GUI using tkinter.
It allows users to input a list and an index, and displays the list split at the nth element.

import tkinter as tk
from utils import split_at_nth_element

class MainApplication:
    def __init__(self, master):
        self.master = master
        master.title("List Splitter")

        self.label = tk.Label(master, text="Enter a list (comma-separated):")
        self.label.pack()

        self.list_entry = tk.Entry(master)
        self.list_entry.pack()

        self.label_n = tk.Label(master, text="Enter the index to split at:")
        self.label_n.pack()

        self.index_entry = tk.Entry(master)
        self.index_entry.pack()

        self.submit_button = tk.Button(master, text="Submit", command=self.split_list)
        self.submit_button.pack()

        self.result_label = tk.Label(master, text="")
        self.result_label.pack()

    def split_list(self):

            lst = [int(item.strip()) for item in self.list_entry.get().split(',')]
            n = int(self.index_entry.get())

        except Exception as e:
            self.result_label.config(text=f"Error: {e}")

if __name__ == "__main__":
    root = tk.Tk()
    app = MainApplication(root)
    root.mainloop()

'''

This file contains utility functions for list operations.
Specifically, it includes a function to split a list at the nth element and add the first part to the end.

'''

def split_at_nth_element(lst, n):
    """
    Splits the list at the nth element and adds the first part to the end.
    
    Parameters:
    lst (list): The list to be split.
    n (int): The index at which to split the list.
    
    Returns:
    list: The rearranged list.
    """
    if n < 0 or n >= len(lst):
        raise ValueError("Index out of range")
    return lst[n:] + lst[:n]

def split_at_nth_element(lst, n):
    """
    Splits the list at the nth element and adds the first part to the end.

def split_at_nth_element(lst, n):
    """
    Splits the list at the nth element and adds the first part to the end.

    lst (list): The list to be split.

    Returns:

    if n < 0 or n >= len(lst):
        raise ValueError("Index out of range")

def split_at_nth_element(lst, n):
    """
    Splits the list at the nth element and adds the first part to the end.
    Parameters:
    lst (list): The list to be split.
    n (int): The index at which to split the list.
    Returns:
    list: The rearranged list.
    """

        raise ValueError("Input must be a list")

        raise ValueError("Index must be an integer")
    if n < 0 or n >= len(lst):
        raise ValueError("Index out of range")
