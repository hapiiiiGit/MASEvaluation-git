import tkinter as tk
from tkinter import messagebox
from utils import remove_parenthesis

def create_gui():
    '''
    Sets up the GUI for the application.
    '''
    # Create the main window
    root = tk.Tk()
    root.title("Remove Parenthesis Tool")
    root.geometry("400x300")
    root.resizable(False, False)  # Disable resizing for better user experience

    # Create and place widgets
    tk.Label(root, text="Enter a string:", font=("Arial", 12)).pack(pady=10)
    input_entry = tk.Entry(root, width=40, font=("Arial", 12))
    input_entry.pack(pady=5)

    tk.Label(root, text="Result:", font=("Arial", 12)).pack(pady=10)
    result_label = tk.Label(root, text="", font=("Arial", 12), fg="blue")
    result_label.pack(pady=5)

    def process_string():
        '''
        Event handler for processing the input string.
        '''
        input_string = input_entry.get()
        if not input_string.strip():
            messagebox.showerror("Error", "Please enter a valid string.")
            return
        try:
            result = remove_parenthesis(input_string)
            result_label.config(text=result)
        except Exception as e:
            messagebox.showerror("Error", f"An error occurred: {e}")

    tk.Button(root, text="Remove Parenthesis", command=process_string, font=("Arial", 12)).pack(pady=20)

    # Start the main loop
    root.mainloop()

if __name__ == "__main__":
    create_gui()

'''
This file contains the main GUI application logic for the program. It uses tkinter to create a graphical interface where users can input a string, process it to remove parenthesis and their content, and display the result.

import tkinter as tk
from tkinter import messagebox
from utils import remove_parenthesis
def create_gui():
    '''
    Sets up the GUI for the application.
    '''
    # Create the main window
    root = tk.Tk()
    root.title("Remove Parenthesis Tool")
    root.geometry("400x300")
    root.resizable(False, False)  # Disable resizing for better user experience
    # Create and place widgets
    tk.Label(root, text="Enter a string:", font=("Arial", 12)).pack(pady=10)
    input_entry = tk.Entry(root, width=40, font=("Arial", 12))
    input_entry.pack(pady=5)
    tk.Label(root, text="Result:", font=("Arial", 12)).pack(pady=10)
    result_label = tk.Label(root, text="", font=("Arial", 12), fg="blue")
    result_label.pack(pady=5)
    def process_string():
        '''
        Event handler for processing the input string.
        '''
        input_string = input_entry.get()
        if not input_string.strip():
            messagebox.showerror("Error", "Please enter a valid string.")
            return
        try:
            result = remove_parenthesis(input_string)
            result_label.config(text=result)
        except Exception as e:
            messagebox.showerror("Error", f"An error occurred: {e}")
    tk.Button(root, text="Remove Parenthesis", command=process_string, font=("Arial", 12)).pack(pady=20)
    # Start the main loop
    root.mainloop()
if __name__ == "__main__":
    create_gui()

import re
def remove_parenthesis(input_string):
    '''
    Removes parenthesis and their content from the input string.
    Args:
        input_string (str): The string to process.
    Returns:
        str: The processed string with parenthesis and their content removed.
    '''
    # Use regex to remove parenthesis and their content
    try:
        return re.sub(r'\(.*?\)', '', input_string).strip()
    except Exception as e:
        raise ValueError(f"Invalid input or processing error: {e}")
