import tkinter as tk

def split_list(list_input, length):
    '''
    Splits the given list into two parts where the length of the first part is the specified integer L.
    Parameters:
        list_input (list): The list to be split.
        length (int): The length of the first part of the list.
    Returns:
        tuple: A tuple containing two lists (first part, second part).
    '''
    if not isinstance(list_input, list):
        raise ValueError("Input must be a list.")
    if not isinstance(length, int):
        raise ValueError("Length must be an integer.")
    if length < 0 or length > len(list_input):
        raise ValueError("Length must be between 0 and the length of the list.")
    first_part = list_input[:length]
    second_part = list_input[length:]
    return (first_part, second_part)

    A class to define the main GUI application for splitting a list.

        self.list_entry.pack()
        # Input for the integer L
        self.length_label = tk.Label(root, text="Enter the length of the first part (L):")

        self.result_text = tk.Text(root, height=5, width=50)
        self.result_text.pack()
    def split_and_display(self):
        '''
        Handles the user input, calls the split_list function, and displays the results.
        '''
        try:
            # Get user input
            list_input = self.list_entry.get()
            length_input = self.length_entry.get()
            # Convert list input to a Python list
            list_input = [item.strip() for item in list_input.split(",") if item.strip()]
            # Convert length input to an integer
            length = int(length_input)
            # Split the list
            first_part, second_part = split_list(list_input, length)
            # Display the result
            self.result_text.delete(1.0, tk.END)

'''
This file contains the main application logic for a Python program that splits a list into two parts based on a given integer length L. It includes a GUI implemented using tkinter.
'''
import tkinter as tk
from tkinter import messagebox
def split_list(list_input, length):
    '''
    Splits the given list into two parts where the length of the first part is the specified integer L.
    Parameters:
        list_input (list): The list to be split.
        length (int): The length of the first part of the list.
    Returns:
        tuple: A tuple containing two lists (first part, second part).
    '''
    if not isinstance(list_input, list):
        raise ValueError("Input must be a list.")
    if not isinstance(length, int):
        raise ValueError("Length must be an integer.")
    if length < 0 or length > len(list_input):
        raise ValueError("Length must be between 0 and the length of the list.")
    first_part = list_input[:length]
    second_part = list_input[length:]
    return (first_part, second_part)
class MainApp:
    '''
    A class to define the main GUI application for splitting a list.
    '''
    def __init__(self, root):
        '''
        Initializes the GUI components.
        Parameters:
            root (tk.Tk): The root window of the tkinter application.
        '''
        self.root = root
        self.root.title("List Splitter")
        # Input for the list
        self.list_label = tk.Label(root, text="Enter a list (comma-separated):")
        self.list_label.pack()
        self.list_entry = tk.Entry(root, width=50)
        self.list_entry.pack()
        # Input for the integer L
        self.length_label = tk.Label(root, text="Enter the length of the first part (L):")
        self.length_label.pack()
        self.length_entry = tk.Entry(root, width=10)
        self.length_entry.pack()
        # Button to split the list
        self.split_button = tk.Button(root, text="Split List", command=self.split_and_display)
        self.split_button.pack()
        # Output area
        self.result_label = tk.Label(root, text="Result:")
        self.result_label.pack()
        self.result_text = tk.Text(root, height=5, width=50)
        self.result_text.pack()
    def split_and_display(self):
        '''
        Handles the user input, calls the split_list function, and displays the results.
        '''
        try:
            # Get user input
            list_input = self.list_entry.get()
            length_input = self.length_entry.get()
            # Convert list input to a Python list
            list_input = [item.strip() for item in list_input.split(",") if item.strip()]
            # Convert length input to an integer
            length = int(length_input)
            # Split the list
            first_part, second_part = split_list(list_input, length)
            # Display the result
            self.result_text.delete(1.0, tk.END)
            self.result_text.insert(tk.END, f"First Part: {first_part}\nSecond Part: {second_part}")
        except ValueError as ve:
            messagebox.showerror("Input Error", str(ve))
        except Exception as e:
            messagebox.showerror("Error", f"An unexpected error occurred: {str(e)}")
# Main execution
if __name__ == "__main__":
    root = tk.Tk()
    app = MainApp(root)
    root.mainloop()

import tkinter as tk
from tkinter import messagebox
def split_list(list_input, length):
    '''
    Splits the given list into two parts where the length of the first part is the specified integer L.
    Parameters:
        list_input (list): The list to be split.
        length (int): The length of the first part of the list.
    Returns:
        tuple: A tuple containing two lists (first part, second part).
    '''
    if not isinstance(list_input, list):
        raise ValueError("Input must be a list.")
    if not isinstance(length, int):
        raise ValueError("Length must be an integer.")
    if length < 0 or length > len(list_input):
        raise ValueError("Length must be between 0 and the length of the list.")
    first_part = list_input[:length]
    second_part = list_input[length:]
    return (first_part, second_part)

        self.list_entry.pack()
        # Input for the integer L
        self.length_label = tk.Label(root, text="Enter the length of the first part (L):")

        # Button to split the list

        self.result_text.pack()
    def split_and_display(self):
        '''

        '''
        try:
            # Get user input
