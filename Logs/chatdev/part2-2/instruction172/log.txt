```python
'''

This module contains the main application class for the Quote Extractor App and the function to extract values between quotation marks from a string.

'''

import tkinter as tk
from tkinter import messagebox

def extract_quoted_values(input_string):
    '''
    Extracts values between quotation marks from the input string.

    Parameters:
    input_string (str): The string from which to extract quoted values.

    Returns:
    list: A list of values found between quotation marks.
    '''
    import re
    # Regular expression to find text between quotes
    pattern = r'"(.*?)"'
    return re.findall(pattern, input_string)

class QuoteExtractorApp:
    '''
    A class to create a GUI application for extracting quoted values from a string.
    '''
    def __init__(self, root):
        self.root = root
        self.root.title("Quote Extractor")

        # Create and place widgets
        self.label = tk.Label(root, text="Enter text:")
        self.label.pack(pady=5)

        self.text_entry = tk.Entry(root, width=50)
        self.text_entry.pack(pady=5)

        self.extract_button = tk.Button(root, text="Extract Quotes", command=self.extract_quotes)
        self.extract_button.pack(pady=5)

        self.result_label = tk.Label(root, text="Extracted Values:")
        self.result_label.pack(pady=5)

        self.result_text = tk.Text(root, height=10, width=50)
        self.result_text.pack(pady=5)

    def extract_quotes(self):
        '''
        Extracts quoted values from the input text and displays them in the result text area.
        '''
        input_text = self.text_entry.get()
        if not input_text:
            messagebox.showwarning("Input Error", "Please enter some text.")
            return

        # Extract quoted values
        quoted_values = extract_quoted_values(input_text)

        # Display results
        self.result_text.delete(1.0, tk.END)  # Clear previous results
        if quoted_values:
            for value in quoted_values:
                self.result_text.insert(tk.END, f"{value}\n")
        else:
            self.result_text.insert(tk.END, "No quoted values found.")

if __name__ == "__main__":
    root = tk.Tk()
    app = QuoteExtractorApp(root)
    root.mainloop()
```

- **GUI Elements**: The GUI consists of labels, an entry field for input, a button to execute the extraction, and a text area to display the extracted values.

'''

This module contains the main application class for the Quote Extractor App and the function to extract values between quotation marks from a string.

'''

import tkinter as tk
from tkinter import messagebox

def extract_quoted_values(input_string):
    '''
    Extracts values between quotation marks from the input string.

    Parameters:
    input_string (str): The string from which to extract quoted values.

    Returns:
    list: A list of values found between quotation marks.
    '''
    import re
    # Regular expression to find text between quotes
    pattern = r'"(.*?)"'
    return re.findall(pattern, input_string)

class QuoteExtractorApp:
    '''
    A class to create a GUI application for extracting quoted values from a string.
    '''
    def __init__(self, root):
        self.root = root
        self.root.title("Quote Extractor")

        # Create and place widgets
        self.label = tk.Label(root, text="Enter text:")
        self.label.pack(pady=5)

        self.text_entry = tk.Entry(root, width=50)
        self.text_entry.pack(pady=5)

        self.extract_button = tk.Button(root, text="Extract Quotes", command=self.extract_quotes)
        self.extract_button.pack(pady=5)

        self.result_label = tk.Label(root, text="Extracted Values:")
        self.result_label.pack(pady=5)

        self.result_text = tk.Text(root, height=10, width=50)
        self.result_text.pack(pady=5)

    def extract_quotes(self):
        '''
        Extracts quoted values from the input text and displays them in the result text area.
        '''
        input_text = self.text_entry.get()
        if not input_text:
            messagebox.showwarning("Input Error", "Please enter some text.")
            return

        # Extract quoted values
        quoted_values = extract_quoted_values(input_text)

        # Display results
        self.result_text.delete(1.0, tk.END)  # Clear previous results
        if quoted_values:
            for value in quoted_values:
                self.result_text.insert(tk.END, f"{value}\n")
        else:
            self.result_text.insert(tk.END, "No quoted values found.")

if __name__ == "__main__":
    root = tk.Tk()
    app = QuoteExtractorApp(root)
    root.mainloop()

'''

import tkinter as tk
from tkinter import messagebox
def extract_quoted_values(input_string):
    '''
    Extracts values between quotation marks from the input string.
    Parameters:
    input_string (str): The string from which to extract quoted values.
    Returns:
    list: A list of values found between quotation marks.
    '''
    import re
    # Regular expression to find text between quotes
    pattern = r'"(.*?)"'
    return re.findall(pattern, input_string)
class QuoteExtractorApp:
    '''
    A class to create a GUI application for extracting quoted values from a string.
    '''
    def __init__(self, root):
        self.root = root
        self.root.title("Quote Extractor")
        # Create and place widgets
        self.label = tk.Label(root, text="Enter text:")
        self.label.pack(pady=5)
        self.text_entry = tk.Entry(root, width=50)
        self.text_entry.pack(pady=5)
        self.extract_button = tk.Button(root, text="Extract Quotes", command=self.extract_quotes)

        self.result_label = tk.Label(root, text="Extracted Values:")
        self.result_label.pack(pady=5)
        self.result_text = tk.Text(root, height=10, width=50)
        self.result_text.pack(pady=5)
    def extract_quotes(self):
        '''
        Extracts quoted values from the input text and displays them in the result text area.
        '''
        input_text = self.text_entry.get()
        if not input_text:
            messagebox.showwarning("Input Error", "Please enter some text.")
            return
        # Extract quoted values
        quoted_values = extract_quoted_values(input_text)
        # Display results
        self.result_text.delete(1.0, tk.END)  # Clear previous results
        if quoted_values:
            for value in quoted_values:
                self.result_text.insert(tk.END, f"{value}\n")
        else:
            self.result_text.insert(tk.END, "No quoted values found.")
if __name__ == "__main__":
    root = tk.Tk()
    app = QuoteExtractorApp(root)
    root.mainloop()

'''
This module contains the main application class for the Quote Extractor App and the function to extract values between quotation marks from a string.
'''
import tkinter as tk
from tkinter import messagebox
def extract_quoted_values(input_string):
    '''
    Extracts values between quotation marks from the input string.
    Parameters:
    input_string (str): The string from which to extract quoted values.
    Returns:
    list: A list of values found between quotation marks.
    '''
    import re
    # Regular expression to find text between quotes
    pattern = r'"(.*?)"'
    return re.findall(pattern, input_string)
class QuoteExtractorApp:
    '''
    A class to create a GUI application for extracting quoted values from a string.
    '''
    def __init__(self, root):
        self.root = root
        self.root.title("Quote Extractor")
        # Create and place widgets
        self.label = tk.Label(root, text="Enter text:")
        self.label.pack(pady=5)
        self.text_entry = tk.Entry(root, width=50)
        self.text_entry.pack(pady=5)
        self.extract_button = tk.Button(root, text="Extract Quotes", command=self.extract_quotes)
        self.extract_button.pack(pady=5)
        self.result_label = tk.Label(root, text="Extracted Values:")
        self.result_label.pack(pady=5)
        self.result_text = tk.Text(root, height=10, width=50)
        self.result_text.pack(pady=5)
    def extract_quotes(self):
        '''
        Extracts quoted values from the input text and displays them in the result text area.
        '''
        input_text = self.text_entry.get()
        if not input_text:
            messagebox.showwarning("Input Error", "Please enter some text.")
            return
        # Extract quoted values
        quoted_values = extract_quoted_values(input_text)
        # Display results
        self.result_text.delete(1.0, tk.END)  # Clear previous results
        if quoted_values:
            for value in quoted_values:
                self.result_text.insert(tk.END, f"{value}\n")
        else:
            self.result_text.insert(tk.END, "No quoted values found.")
if __name__ == "__main__":
    root = tk.Tk()
    app = QuoteExtractorApp(root)
    root.mainloop()

'''
This module contains the main application class for the Quote Extractor App and the function to extract values between quotation marks from a string.
'''
import tkinter as tk
from tkinter import messagebox
def extract_quoted_values(input_string):
    '''
    Extracts values between quotation marks from the input string.
    Parameters:
    input_string (str): The string from which to extract quoted values.
    Returns:
    list: A list of values found between quotation marks.
    '''
    import re
    # Regular expression to find text between quotes
    pattern = r'"(.*?)"'
    return re.findall(pattern, input_string)
class QuoteExtractorApp:
    '''
    A class to create a GUI application for extracting quoted values from a string.
    '''
    def __init__(self, root):
        self.root = root
        self.root.title("Quote Extractor")
        # Create and place widgets
        self.label = tk.Label(root, text="Enter text:")
        self.label.pack(pady=5)
        self.text_entry = tk.Entry(root, width=50)
        self.text_entry.pack(pady=5)
        self.extract_button = tk.Button(root, text="Extract Quotes", command=self.extract_quotes)
        self.extract_button.pack(pady=5)
        self.result_label = tk.Label(root, text="Extracted Values:")
        self.result_label.pack(pady=5)
        self.result_text = tk.Text(root, height=10, width=50)
        self.result_text.pack(pady=5)
    def extract_quotes(self):
        '''
        Extracts quoted values from the input text and displays them in the result text area.
        '''
        input_text = self.text_entry.get()
        if not input_text:
            messagebox.showwarning("Input Error", "Please enter some text.")
            return
        # Extract quoted values
        quoted_values = extract_quoted_values(input_text)
        # Display results
        self.result_text.delete(1.0, tk.END)  # Clear previous results
        if quoted_values:
            for value in quoted_values:
                self.result_text.insert(tk.END, f"{value}\n")
        else:
            self.result_text.insert(tk.END, "No quoted values found.")
if __name__ == "__main__":
    root = tk.Tk()
    app = QuoteExtractorApp(root)
    root.mainloop()

# Quote Extractor App

Extract values between quotation marks from a string using a simple GUI application.

## Overview

The Quote Extractor App is a straightforward application designed to help users extract text enclosed in quotation marks from a given string. This application is built using Python and Tkinter, providing a user-friendly graphical interface for ease of use.

## Features

- **Extract Quoted Values**: Automatically identifies and extracts all text enclosed in quotation marks from the input string.
- **User-Friendly Interface**: Simple and intuitive GUI for easy interaction.
- **Real-Time Results**: Displays extracted values instantly in the application window.

## Installation

### Prerequisites

- **Python**: Ensure you have Python installed on your system. The application is compatible with Python version 3.6 and above.

### Installation Steps

1. **Clone the Repository**: Download the source code from the repository or copy the provided code into a file named `main.py`.

2. **Install Dependencies**: This application does not require any external dependencies beyond Python and Tkinter, which is included with standard Python installations.

3. **Run the Application**: Execute the application by running the following command in your terminal or command prompt:

   ```bash
   python main.py
   ```

## Usage

1. **Launch the Application**: Run the `main.py` script to open the Quote Extractor App.

2. **Enter Text**: In the application window, enter the text from which you want to extract quoted values into the input field labeled "Enter text:".

3. **Extract Quotes**: Click the "Extract Quotes" button to process the input text.

4. **View Results**: The extracted values will be displayed in the "Extracted Values:" section of the application. If no quoted values are found, a message will indicate this.

5. **Clear and Repeat**: You can clear the input and results by simply entering new text and clicking the "Extract Quotes" button again.

## Troubleshooting

- **No Output**: Ensure that the input text contains values enclosed in quotation marks. The application only extracts text between double quotes (`"`).

- **Input Error**: If no text is entered, a warning message will prompt you to provide input.
