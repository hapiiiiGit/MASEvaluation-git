```python

'''

This module contains the main application class for the string replacement GUI and the function to replace characters in a string.

'''

import tkinter as tk
from tkinter import messagebox

def replace_characters(input_string, old_chars, new_chars):
    '''
    Replace characters in a string.

    Parameters:
    input_string (str): The original string.
    old_chars (str): Characters to be replaced.
    new_chars (str): Characters to replace with.

    Returns:
    str: The modified string with characters replaced.
    '''
    translation_table = str.maketrans(old_chars, new_chars)
    return input_string.translate(translation_table)

class StringReplacerApp:
    '''
    A class to create a GUI application for replacing characters in a string.
    '''
    def __init__(self, root):
        self.root = root
        self.root.title("String Replacer")

        # Input string
        self.input_label = tk.Label(root, text="Input String:")
        self.input_label.pack()
        self.input_entry = tk.Entry(root, width=50)
        self.input_entry.pack()

        # Characters to replace
        self.old_label = tk.Label(root, text="Characters to Replace:")
        self.old_label.pack()
        self.old_entry = tk.Entry(root, width=50)
        self.old_entry.pack()

        # Replacement characters
        self.new_label = tk.Label(root, text="Replacement Characters:")
        self.new_label.pack()
        self.new_entry = tk.Entry(root, width=50)
        self.new_entry.pack()

        # Replace button
        self.replace_button = tk.Button(root, text="Replace", command=self.replace_action)
        self.replace_button.pack()

        # Result display
        self.result_label = tk.Label(root, text="Result:")
        self.result_label.pack()
        self.result_display = tk.Label(root, text="", wraplength=400)
        self.result_display.pack()

    def replace_action(self):
        '''
        Handle the replace button click event.
        '''
        input_string = self.input_entry.get()
        old_chars = self.old_entry.get()
        new_chars = self.new_entry.get()

        if len(old_chars) != len(new_chars):
            messagebox.showerror("Error", "Old and new characters must have the same length.")
            return

        result = replace_characters(input_string, old_chars, new_chars)
        self.result_display.config(text=result)

if __name__ == "__main__":
    root = tk.Tk()
    app = StringReplacerApp(root)
    root.mainloop()

```

This code provides a complete implementation of the string replacement application with a GUI. The `StringReplacerApp` class handles the user interface, and the `replace_characters` function performs the actual string manipulation. The application checks for equal lengths of old and new characters to ensure proper replacement.

'''

This module contains the main application class for the string replacement GUI and the function to replace characters in a string.

'''

import tkinter as tk
from tkinter import messagebox

def replace_characters(input_string, old_chars, new_chars):
    '''
    Replace characters in a string.

    Parameters:
    input_string (str): The original string.
    old_chars (str): Characters to be replaced.
    new_chars (str): Characters to replace with.

    Returns:
    str: The modified string with characters replaced.
    '''
    translation_table = str.maketrans(old_chars, new_chars)
    return input_string.translate(translation_table)

class StringReplacerApp:
    '''
    A class to create a GUI application for replacing characters in a string.
    '''
    def __init__(self, root):
        self.root = root
        self.root.title("String Replacer")

        # Input string
        self.input_label = tk.Label(root, text="Input String:")
        self.input_label.pack()
        self.input_entry = tk.Entry(root, width=50)
        self.input_entry.pack()

        # Characters to replace
        self.old_label = tk.Label(root, text="Characters to Replace:")
        self.old_label.pack()
        self.old_entry = tk.Entry(root, width=50)
        self.old_entry.pack()

        # Replacement characters
        self.new_label = tk.Label(root, text="Replacement Characters:")
        self.new_label.pack()
        self.new_entry = tk.Entry(root, width=50)
        self.new_entry.pack()

        # Replace button
        self.replace_button = tk.Button(root, text="Replace", command=self.replace_action)
        self.replace_button.pack()

        # Result display
        self.result_label = tk.Label(root, text="Result:")
        self.result_label.pack()
        self.result_display = tk.Label(root, text="", wraplength=400)
        self.result_display.pack()

    def replace_action(self):
        '''
        Handle the replace button click event.
        '''
        input_string = self.input_entry.get()
        old_chars = self.old_entry.get()
        new_chars = self.new_entry.get()

        if len(old_chars) != len(new_chars):
            messagebox.showerror("Error", "Old and new characters must have the same length.")
            return

        result = replace_characters(input_string, old_chars, new_chars)
        self.result_display.config(text=result)

if __name__ == "__main__":
    root = tk.Tk()
    app = StringReplacerApp(root)
    root.mainloop()

```

'''
This module contains the main application class for the string replacement GUI and the function to replace characters in a string.
'''
import tkinter as tk
from tkinter import messagebox
def replace_characters(input_string, old_chars, new_chars):
    '''
    Replace characters in a string.
    Parameters:
    input_string (str): The original string.
    old_chars (str): Characters to be replaced.
    new_chars (str): Characters to replace with.
    Returns:
    str: The modified string with characters replaced.
    '''
    translation_table = str.maketrans(old_chars, new_chars)
    return input_string.translate(translation_table)
class StringReplacerApp:
    '''
    A class to create a GUI application for replacing characters in a string.
    '''
    def __init__(self, root):
        self.root = root
        self.root.title("String Replacer")
        # Input string
        self.input_label = tk.Label(root, text="Input String:")
        self.input_label.pack()
        self.input_entry = tk.Entry(root, width=50)
        self.input_entry.pack()
        # Characters to replace
        self.old_label = tk.Label(root, text="Characters to Replace:")
        self.old_label.pack()
        self.old_entry = tk.Entry(root, width=50)
        self.old_entry.pack()
        # Replacement characters
        self.new_label = tk.Label(root, text="Replacement Characters:")
        self.new_label.pack()
        self.new_entry = tk.Entry(root, width=50)
        self.new_entry.pack()
        # Replace button
        self.replace_button = tk.Button(root, text="Replace", command=self.replace_action)
        self.replace_button.pack()
        # Result display
        self.result_label = tk.Label(root, text="Result:")
        self.result_label.pack()
        self.result_display = tk.Label(root, text="", wraplength=400)
        self.result_display.pack()
    def replace_action(self):
        '''
        Handle the replace button click event.
        '''
        input_string = self.input_entry.get()
        old_chars = self.old_entry.get()
        new_chars = self.new_entry.get()
        if len(old_chars) != len(new_chars):
            messagebox.showerror("Error", "Old and new characters must have the same length.")
            return
        result = replace_characters(input_string, old_chars, new_chars)
        self.result_display.config(text=result)
if __name__ == "__main__":
    root = tk.Tk()
    app = StringReplacerApp(root)
    root.mainloop()

'''
This module contains the main application class for the string replacement GUI and the function to replace characters in a string.
'''
import tkinter as tk
from tkinter import messagebox
def replace_characters(input_string, old_chars, new_chars):
    '''
    Replace characters in a string.
    Parameters:
    input_string (str): The original string.
    old_chars (str): Characters to be replaced.
    new_chars (str): Characters to replace with.
    Returns:
    str: The modified string with characters replaced.
    '''
    translation_table = str.maketrans(old_chars, new_chars)
    return input_string.translate(translation_table)
class StringReplacerApp:
    '''
    A class to create a GUI application for replacing characters in a string.
    '''
    def __init__(self, root):
        self.root = root
        self.root.title("String Replacer")
        # Input string
        self.input_label = tk.Label(root, text="Input String:")
        self.input_label.pack()
        self.input_entry = tk.Entry(root, width=50)
        self.input_entry.pack()
        # Characters to replace
        self.old_label = tk.Label(root, text="Characters to Replace:")
        self.old_label.pack()
        self.old_entry = tk.Entry(root, width=50)
        self.old_entry.pack()
        # Replacement characters
        self.new_label = tk.Label(root, text="Replacement Characters:")
        self.new_label.pack()
        self.new_entry = tk.Entry(root, width=50)
        self.new_entry.pack()
        # Replace button
        self.replace_button = tk.Button(root, text="Replace", command=self.replace_action)
        self.replace_button.pack()
        # Result display
        self.result_label = tk.Label(root, text="Result:")
        self.result_label.pack()
        self.result_display = tk.Label(root, text="", wraplength=400)
        self.result_display.pack()
    def replace_action(self):
        '''
        Handle the replace button click event.
        '''
        input_string = self.input_entry.get()
        old_chars = self.old_entry.get()
        new_chars = self.new_entry.get()
        if len(old_chars) != len(new_chars):
            messagebox.showerror("Error", "Old and new characters must have the same length.")
            return
        result = replace_characters(input_string, old_chars, new_chars)
        self.result_display.config(text=result)
if __name__ == "__main__":
    root = tk.Tk()
    app = StringReplacerApp(root)
    root.mainloop()

It provides an intuitive interface for users to input a string, specify characters to replace, and define the replacement characters. The application then processes the input and displays the modified string.

- **Character Replacement**: The core functionality of the application is to replace specified characters in a given string with new characters. This is achieved through a user-friendly graphical interface.
