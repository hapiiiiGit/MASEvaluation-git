def element_wise_division(tuple1, tuple2):
    '''
    Performs element-wise division between two tuples.
    Raises an error if division by zero occurs or if tuples have different lengths.
    '''
    if len(tuple1) != len(tuple2):

    result = []

def element_wise_division(tuple1, tuple2):
    '''
    Performs element-wise division between two tuples.
    Raises an error if division by zero occurs or if tuples have different lengths.
    '''
    if len(tuple1) != len(tuple2):
        raise ValueError("Tuples must have the same length.")
    result = []
    for a, b in zip(tuple1, tuple2):
        if b == 0:
            raise ZeroDivisionError("Division by zero encountered.")
        result.append(a / b)
    return tuple(result)

def element_wise_division(tuple1, tuple2):  
    '''  
    Performs element-wise division between two tuples.  
    Raises an error if division by zero occurs or if tuples have different lengths.

    if len(tuple1) != len(tuple2):

    for a, b in zip(tuple1, tuple2):

        result.append(a / b)

import tkinter as tk  
from tkinter import messagebox  
from division_utils import element_wise_division  
class TupleDividerApp:  
    '''  
    Represents the main GUI application for tuple division.  
    '''  
    def __init__(self, root):  
        self.root = root  
        self.root.title("Tuple Divider")  
        # Input fields for tuples  
        self.label_tuple1 = tk.Label(root, text="Enter Tuple 1:")  
        self.label_tuple1.pack()  
        self.entry_tuple1 = tk.Entry(root, width=50)  
        self.entry_tuple1.pack()  
        self.label_tuple2 = tk.Label(root, text="Enter Tuple 2:")  
        self.label_tuple2.pack()  
        self.entry_tuple2 = tk.Entry(root, width=50)  
        self.entry_tuple2.pack()  
        # Button to perform division  
        self.divide_button = tk.Button(root, text="Divide Tuples", command=self.perform_division)  
        self.divide_button.pack()  
        # Output field for results  
        self.result_label = tk.Label(root, text="Result:")  
        self.result_label.pack()  
        self.result_text = tk.Text(root, height=5, width=50)  
        self.result_text.pack()  
    def validate_input(self, tuple_str):  
        '''  
        Validates and converts string input into a tuple.  
        '''  
        try:  
            # Convert string to tuple  
            tuple_data = eval(tuple_str)  
            if not isinstance(tuple_data, tuple):  
                raise ValueError("Input is not a tuple.")  
            return tuple_data  
        except Exception as e:  
            messagebox.showerror("Invalid Input", f"Error: {e}")  
            return None  
    def perform_division(self):  
        '''  
        Performs element-wise division of the tuples and displays the result.  
        '''  
        tuple1_str = self.entry_tuple1.get()  
        tuple2_str = self.entry_tuple2.get()  
        tuple1 = self.validate_input(tuple1_str)  
        tuple2 = self.validate_input(tuple2_str)  
        if tuple1 is not None and tuple2 is not None:  
            try:  
                result = element_wise_division(tuple1, tuple2)  
                self.result_text.delete(1.0, tk.END)  
                self.result_text.insert(tk.END, str(result))  
            except Exception as e:  
                messagebox.showerror("Error", f"Division Error: {e}")  
if __name__ == "__main__":  
    root = tk.Tk()  
    app = TupleDividerApp(root)  
    root.mainloop()

'''  
This file contains the utility function for performing element-wise division of tuples.  
'''  
def element_wise_division(tuple1, tuple2):  
    '''  
    Performs element-wise division between two tuples.  
    Raises an error if division by zero occurs or if tuples have different lengths.  
    '''  
    if len(tuple1) != len(tuple2):  
        raise ValueError("Tuples must have the same length.")  
    result = []  
    for a, b in zip(tuple1, tuple2):  
        if b == 0:  
            raise ZeroDivisionError("Division by zero encountered.")  
        result.append(a / b)  
    return tuple(result)

'''  
This file contains the utility function for performing element-wise division of tuples.  
'''  
def element_wise_division(tuple1, tuple2):  
    '''  
    Performs element-wise division between two tuples.  
    Raises an error if division by zero occurs or if tuples have different lengths.  
    '''  
    if len(tuple1) != len(tuple2):  
        raise ValueError("Tuples must have the same length.")  
    result = []  
    for a, b in zip(tuple1, tuple2):  
        if b == 0:  
            raise ZeroDivisionError("Division by zero encountered.")  
        result.append(a / b)  
    return tuple(result)

'''
This file contains the utility function for performing element-wise division of tuples.
'''
def element_wise_division(tuple1, tuple2):
    '''
    Performs element-wise division between two tuples.
    Raises an error if division by zero occurs or if tuples have different lengths.
    '''
    if len(tuple1) != len(tuple2):
        raise ValueError("Tuples must have the same length.")
    result = []
    for a, b in zip(tuple1, tuple2):
        if b == 0:
            raise ZeroDivisionError("Division by zero encountered.")
        result.append(a / b)
    return tuple(result)

import tkinter as tk

class TupleDividerApp:
    '''
    Represents the main GUI application for tuple division.
    '''
    def __init__(self, root):
        '''
        Initializes the Tuple Divider application with GUI components.
        '''
        self.root = root
        self.root.title("Tuple Divider")
        # Input fields for tuples
        self.label_tuple1 = tk.Label(root, text="Enter Tuple 1:")
        self.label_tuple1.pack()
        self.entry_tuple1 = tk.Entry(root, width=50)
        self.entry_tuple1.pack()
        self.label_tuple2 = tk.Label(root, text="Enter Tuple 2:")
        self.label_tuple2.pack()
        self.entry_tuple2 = tk.Entry(root, width=50)
        self.entry_tuple2.pack()
        # Button to perform division
        self.divide_button = tk.Button(root, text="Divide Tuples", command=self.perform_division)
        self.divide_button.pack()
        # Output field for results
        self.result_label = tk.Label(root, text="Result:")
        self.result_label.pack()
        self.result_text = tk.Text(root, height=5, width=50)
        self.result_text.pack()
    def validate_input(self, tuple_str):
        '''
        Validates and converts string input into a tuple.
        '''
        try:
            # Convert string to tuple
            tuple_data = eval(tuple_str)
            if not isinstance(tuple_data, tuple):
                raise ValueError("Input is not a tuple.")
            return tuple_data
        except Exception as e:
            messagebox.showerror("Invalid Input", f"Error: {e}")
            return None
    def perform_division(self):
        '''
        Performs element-wise division of the tuples and displays the result.
        '''
        tuple1_str = self.entry_tuple1.get()
        tuple2_str = self.entry_tuple2.get()
        tuple1 = self.validate_input(tuple1_str)
        tuple2 = self.validate_input(tuple2_str)
        if tuple1 is not None and tuple2 is not None:
            try:
                result = element_wise_division(tuple1, tuple2)
                self.result_text.delete(1.0, tk.END)
                self.result_text.insert(tk.END, str(result))
            except Exception as e:
                messagebox.showerror("Error", f"Division Error: {e}")
if __name__ == "__main__":
    root = tk.Tk()

'''
This file contains the utility function for performing element-wise division of tuples.
'''
def element_wise_division(tuple1, tuple2):
    '''
    Performs element-wise division between two tuples.
    Raises an error if division by zero occurs or if tuples have different lengths.
    '''
    if len(tuple1) != len(tuple2):
        raise ValueError("Tuples must have the same length.")
    result = []
    for a, b in zip(tuple1, tuple2):
        if b == 0:
            raise ZeroDivisionError("Division by zero encountered.")
        result.append(a / b)
    return tuple(result)

'''
This file contains the utility function for performing element-wise division of tuples.
'''
def element_wise_division(tuple1, tuple2):
    '''
    Performs element-wise division between two tuples.
    Raises an error if division by zero occurs or if tuples have different lengths.
    '''
    if len(tuple1) != len(tuple2):
        raise ValueError("Tuples must have the same length.")
    result = []
    for a, b in zip(tuple1, tuple2):
        if b == 0:
            raise ZeroDivisionError("Division by zero encountered.")
        result.append(a / b)
    return tuple(result)
