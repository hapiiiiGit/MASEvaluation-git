'''

This module contains the OddSumCalculator class which provides a method to calculate the sum of all odd natural numbers within a specified range.

'''

class OddSumCalculator:
    @staticmethod
    def calculate_sum(l, r):
        '''
        Calculate the sum of all odd natural numbers between l and r (inclusive).
        
        Parameters:
        l (int): The lower bound of the range.
        r (int): The upper bound of the range.
        
        Returns:
        int: The sum of all odd natural numbers within the range.
        '''
        return sum(i for i in range(l, r + 1) if i % 2 != 0)
```

'''

class OddSumCalculator:
    @staticmethod
    def calculate_sum(l, r):
        '''
        Calculate the sum of all odd natural numbers between l and r (inclusive).

        l (int): The lower bound of the range.

        return sum(i for i in range(l, r + 1) if i % 2 != 0)

'''
This module contains the OddSumCalculator class which provides a method to calculate the sum of all odd natural numbers within a specified range.
'''
class OddSumCalculator:
    @staticmethod
    def calculate_sum(l, r):
        '''
        Calculate the sum of all odd natural numbers between l and r (inclusive).
        Parameters:
        l (int): The lower bound of the range.
        r (int): The upper bound of the range.
        Returns:
        int: The sum of all odd natural numbers within the range.
        '''
        return sum(i for i in range(l, r + 1) if i % 2 != 0)

```python
'''
This module contains the OddSumCalculator class which provides a method to calculate the sum of all odd natural numbers within a specified range.
'''
class OddSumCalculator:
    @staticmethod
    def calculate_sum(l, r):
        '''
        Calculate the sum of all odd natural numbers between l and r (inclusive).
        Parameters:
        l (int): The lower bound of the range.
        r (int): The upper bound of the range.
        Returns:
        int: The sum of all odd natural numbers within the range.
        '''
        return sum(i for i in range(l, r + 1) if i % 2 != 0)
```

```python
'''
This is the main module for the application. It sets up the GUI using tkinter and allows the user to input a range and calculate the sum of odd numbers within that range.
'''
import tkinter as tk
from tkinter import messagebox
from calculator import OddSumCalculator
class MainApplication:
    def __init__(self, root):
        '''
        Initialize the main application with GUI components.
        Parameters:
        root (tk.Tk): The root window of the tkinter application.
        '''
        self.root = root
        self.root.title("Odd Sum Calculator")
        # Create and place widgets
        self.label1 = tk.Label(root, text="Enter the lower bound (l):")
        self.label1.pack()
        self.entry1 = tk.Entry(root)
        self.entry1.pack()
        self.label2 = tk.Label(root, text="Enter the upper bound (r):")
        self.label2.pack()
        self.entry2 = tk.Entry(root)
        self.entry2.pack()
        self.calculate_button = tk.Button(root, text="Calculate Sum", command=self.calculate_and_display_sum)
        self.calculate_button.pack()
        self.result_label = tk.Label(root, text="")
        self.result_label.pack()
    def calculate_and_display_sum(self):
        '''
        Calculate the sum of odd numbers in the specified range and display the result.
        '''
        try:
            l = int(self.entry1.get())
            r = int(self.entry2.get())
            if l > r:
                raise ValueError("Lower bound must be less than or equal to upper bound.")
            result = OddSumCalculator.calculate_sum(l, r)
            self.result_label.config(text=f"Sum of odd numbers: {result}")
        except ValueError as e:
            messagebox.showerror("Invalid input", f"Please enter valid integers.\nError: {e}")
if __name__ == "__main__":
    root = tk.Tk()
    app = MainApplication(root)
    root.mainloop()

'''
This module contains the OddSumCalculator class which provides a method to calculate the sum of all odd natural numbers within a specified range.
'''
class OddSumCalculator:
    @staticmethod
    def calculate_sum(l, r):
        '''
        Calculate the sum of all odd natural numbers between l and r (inclusive).
        Parameters:
        l (int): The lower bound of the range.
        r (int): The upper bound of the range.
        Returns:
        int: The sum of all odd natural numbers within the range.
        '''
        return sum(i for i in range(l, r + 1) if i % 2 != 0)

```python
'''
This is the main module for the application. It sets up the GUI using tkinter and allows the user to input a range and calculate the sum of odd numbers within that range.
'''
import tkinter as tk
from tkinter import messagebox
from calculator import OddSumCalculator
class MainApplication:
    def __init__(self, root):
        '''
        Initialize the main application with GUI components.
        Parameters:
        root (tk.Tk): The root window of the tkinter application.
        '''
        self.root = root
        self.root.title("Odd Sum Calculator")
        # Create and place widgets
        self.label1 = tk.Label(root, text="Enter the lower bound (l):")
        self.label1.pack()
        self.entry1 = tk.Entry(root)
        self.entry1.pack()
        self.label2 = tk.Label(root, text="Enter the upper bound (r):")
        self.label2.pack()
        self.entry2 = tk.Entry(root)
        self.entry2.pack()
        self.calculate_button = tk.Button(root, text="Calculate Sum", command=self.calculate_and_display_sum)
        self.calculate_button.pack()
        self.result_label = tk.Label(root, text="")
        self.result_label.pack()
    def calculate_and_display_sum(self):
        '''
        Calculate the sum of odd numbers in the specified range and display the result.
        '''
        try:
            l = int(self.entry1.get())
            r = int(self.entry2.get())
            if l > r:
                raise ValueError("Lower bound must be less than or equal to upper bound.")
            result = OddSumCalculator.calculate_sum(l, r)
            self.result_label.config(text=f"Sum of odd numbers: {result}")
        except ValueError as e:
            messagebox.showerror("Invalid input", f"Please enter valid integers.\nError: {e}")
if __name__ == "__main__":
    root = tk.Tk()
    app = MainApplication(root)
    root.mainloop()
