'''

This module contains the TrapeziumCalculator class which provides a method to calculate the median length of a trapezium.

'''

class TrapeziumCalculator:
    @staticmethod
    def calculate_median(base1, base2):
        """
        Calculate the median length of a trapezium given the lengths of the two bases.
        
        :param base1: Length of the first base.
        :param base2: Length of the second base.
        :return: Median length of the trapezium.
        """
        return (base1 + base2) / 2
```

class TrapeziumCalculator:
    @staticmethod

        """

class TrapeziumCalculator:
    @staticmethod
    def calculate_median(base1, base2):
        """
        Calculate the median length of a trapezium given the lengths of the two bases.
        :param base1: Length of the first base.

        :return: Median length of the trapezium.

'''
class TrapeziumCalculator:
    @staticmethod
    def calculate_median(base1, base2):
        """
        Calculate the median length of a trapezium given the lengths of the two bases.
        :param base1: Length of the first base.
        :param base2: Length of the second base.
        :return: Median length of the trapezium.
        """
        return (base1 + base2) / 2
