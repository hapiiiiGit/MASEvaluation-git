```python
'''

This module contains the MainApplication class for the GUI and the find_union function to compute the union of two tuples.

'''

import tkinter as tk
from tkinter import messagebox

def find_union(tuple1, tuple2):
    '''
    Computes the union of two tuples and returns the sorted result.

    Parameters:
    tuple1 (tuple): The first tuple.
    tuple2 (tuple): The second tuple.

    Returns:
    list: A sorted list containing the union of the elements from both tuples.
    '''
    union_set = set(tuple1).union(set(tuple2))
    sorted_union = sorted(union_set)
    return sorted_union

class MainApplication(tk.Tk):
    '''
    Main application class for the GUI.
    '''
    def __init__(self):
        super().__init__()
        self.title("Tuple Union Finder")
        self.geometry("400x300")

        # Labels and Entry widgets for tuple inputs
        self.label1 = tk.Label(self, text="Enter first tuple (comma-separated):")
        self.label1.pack(pady=5)
        self.entry1 = tk.Entry(self)
        self.entry1.pack(pady=5)

        self.label2 = tk.Label(self, text="Enter second tuple (comma-separated):")
        self.label2.pack(pady=5)
        self.entry2 = tk.Entry(self)
        self.entry2.pack(pady=5)

        # Button to calculate union
        self.calculate_button = tk.Button(self, text="Calculate Union", command=self.calculate_union)
        self.calculate_button.pack(pady=20)

        # Label to display result
        self.result_label = tk.Label(self, text="Union Result:")
        self.result_label.pack(pady=5)
        self.result_display = tk.Label(self, text="", fg="blue")
        self.result_display.pack(pady=5)

    def calculate_union(self):
        '''
        Retrieves input from the user, calculates the union, and updates the GUI with the result.
        '''
        try:
            tuple1 = tuple(map(int, self.entry1.get().split(',')))
            tuple2 = tuple(map(int, self.entry2.get().split(',')))
            result = find_union(tuple1, tuple2)
            self.result_display.config(text=str(result))
        except ValueError:
            messagebox.showerror("Invalid Input", "Please enter valid tuples with integer elements.")

if __name__ == "__main__":
    app = MainApplication()
    app.mainloop()

import tkinter as tk
from tkinter import messagebox

def find_union(tuple1, tuple2):
    '''
    Computes the union of two tuples and returns the sorted result.

    Parameters:
    tuple1 (tuple): The first tuple.
    tuple2 (tuple): The second tuple.

    Returns:
    list: A sorted list containing the union of the elements from both tuples.
    '''
    union_set = set(tuple1).union(set(tuple2))
    sorted_union = sorted(union_set)
    return sorted_union

import tkinter as tk

def find_union(tuple1, tuple2):
    '''
    Computes the union of two tuples and returns the sorted result.
    Parameters:
    tuple1 (tuple): The first tuple.
    tuple2 (tuple): The second tuple.
    Returns:
    list: A sorted list containing the union of the elements from both tuples.
    '''

    sorted_union = sorted(union_set)
    return sorted_union

'''
This module contains the MainApplication class for the GUI and the find_union function to compute the union of two tuples.
'''
import tkinter as tk
from tkinter import messagebox
def find_union(tuple1, tuple2):
    '''
    Computes the union of two tuples and returns the sorted result.
    Parameters:
    tuple1 (tuple): The first tuple.
    tuple2 (tuple): The second tuple.
    Returns:
    list: A sorted list containing the union of the elements from both tuples.
    '''
    union_set = set(tuple1).union(set(tuple2))
    sorted_union = sorted(union_set)
    return sorted_union
class MainApplication(tk.Tk):
    '''
    Main application class for the GUI.
    '''
    def __init__(self):
        super().__init__()
        self.title("Tuple Union Finder")
        self.geometry("400x300")
        # Labels and Entry widgets for tuple inputs
        self.label1 = tk.Label(self, text="Enter first tuple (comma-separated):")
        self.label1.pack(pady=5)
        self.entry1 = tk.Entry(self)
        self.entry1.pack(pady=5)
        self.label2 = tk.Label(self, text="Enter second tuple (comma-separated):")
        self.label2.pack(pady=5)
        self.entry2 = tk.Entry(self)
        self.entry2.pack(pady=5)
        # Button to calculate union
        self.calculate_button = tk.Button(self, text="Calculate Union", command=self.calculate_union)
        self.calculate_button.pack(pady=20)
        # Label to display result
        self.result_label = tk.Label(self, text="Union Result:")
        self.result_label.pack(pady=5)
        self.result_display = tk.Label(self, text="", fg="blue")
        self.result_display.pack(pady=5)
    def calculate_union(self):
        '''
        Retrieves input from the user, calculates the union, and updates the GUI with the result.
        '''
        try:
            tuple1 = tuple(map(int, self.entry1.get().split(',')))
            tuple2 = tuple(map(int, self.entry2.get().split(',')))
            result = find_union(tuple1, tuple2)
            self.result_display.config(text=str(result))
        except ValueError:
            messagebox.showerror("Invalid Input", "Please enter valid tuples with integer elements.")
if __name__ == "__main__":
    app = MainApplication()
    app.mainloop()

'''
This module contains the MainApplication class for the GUI and the find_union function to compute the union of two tuples.
'''
import tkinter as tk
from tkinter import messagebox
def find_union(tuple1, tuple2):
    '''
    Computes the union of two tuples and returns the sorted result.
    Parameters:
    tuple1 (tuple): The first tuple.
    tuple2 (tuple): The second tuple.
    Returns:
    list: A sorted list containing the union of the elements from both tuples.
    '''
    union_set = set(tuple1).union(set(tuple2))
    sorted_union = sorted(union_set)
    return sorted_union
class MainApplication(tk.Tk):
    '''
    Main application class for the GUI.
    '''
    def __init__(self):
        super().__init__()
        self.title("Tuple Union Finder")
        self.geometry("400x300")
        # Labels and Entry widgets for tuple inputs
        self.label1 = tk.Label(self, text="Enter first tuple (comma-separated):")
        self.label1.pack(pady=5)
        self.entry1 = tk.Entry(self)
        self.entry1.pack(pady=5)
        self.label2 = tk.Label(self, text="Enter second tuple (comma-separated):")
        self.label2.pack(pady=5)
        self.entry2 = tk.Entry(self)
        self.entry2.pack(pady=5)
        # Button to calculate union
        self.calculate_button = tk.Button(self, text="Calculate Union", command=self.calculate_union)
        self.calculate_button.pack(pady=20)
        # Label to display result
        self.result_label = tk.Label(self, text="Union Result:")
        self.result_label.pack(pady=5)
        self.result_display = tk.Label(self, text="", fg="blue")
        self.result_display.pack(pady=5)
    def calculate_union(self):
        '''
        Retrieves input from the user, calculates the union, and updates the GUI with the result.
        '''
        try:
            tuple1 = tuple(map(int, self.entry1.get().split(',')))
            tuple2 = tuple(map(int, self.entry2.get().split(',')))
            result = find_union(tuple1, tuple2)
            self.result_display.config(text=str(result))
        except ValueError:
            messagebox.showerror("Invalid Input", "Please enter valid tuples with integer elements.")
if __name__ == "__main__":
    app = MainApplication()
    app.mainloop()
