import tkinter as tk
from tkinter import messagebox
def filter_students_by_criteria(students, min_height, min_weight):
    '''
    Filters students based on minimum height and weight criteria.
    Parameters:
    students (dict): A dictionary with student names as keys and tuples of (height, weight) as values.
    min_height (float): The minimum height to filter students.
    min_weight (float): The minimum weight to filter students.
    Returns:
    dict: A dictionary of students who meet the criteria.
    '''
    filtered_students = {name: (height, weight) for name, (height, weight) in students.items()
                         if height > min_height and weight > min_weight}
    return filtered_students
class StudentFilterApp:
    '''
    A GUI application to filter students based on height and weight criteria.
    '''
    def __init__(self, root):
        self.root = root
        self.root.title("Student Filter App")
        self.create_widgets()
    def create_widgets(self):
        # Labels and Entries for student data
        self.student_data_label = tk.Label(self.root, text="Enter Student Data (name:height,weight):")
        self.student_data_label.pack()
        self.student_data_entry = tk.Entry(self.root, width=50)
        self.student_data_entry.pack()
        # Labels and Entries for minimum height and weight
        self.min_height_label = tk.Label(self.root, text="Enter Minimum Height:")
        self.min_height_label.pack()
        self.min_height_entry = tk.Entry(self.root)
        self.min_height_entry.pack()
        self.min_weight_label = tk.Label(self.root, text="Enter Minimum Weight:")
        self.min_weight_label.pack()
        self.min_weight_entry = tk.Entry(self.root)
        self.min_weight_entry.pack()
        # Filter Button
        self.filter_button = tk.Button(self.root, text="Filter Students", command=self.filter_students)
        self.filter_button.pack()
        # Result Display
        self.result_label = tk.Label(self.root, text="")
        self.result_label.pack()
    def filter_students(self):
        try:
            # Parse student data
            student_data = self.student_data_entry.get()
            students = {}
            for entry in student_data.split(','):
                name, hw = entry.split(':')
                height, weight = map(float, hw.split(','))
                students[name.strip()] = (height, weight)
            # Get minimum height and weight
            min_height = float(self.min_height_entry.get())
            min_weight = float(self.min_weight_entry.get())
            # Filter students
            filtered_students = filter_students_by_criteria(students, min_height, min_weight)
            # Display results
            result_text = "Filtered Students:\n" + "\n".join([f"{name}: {hw}" for name, hw in filtered_students.items()])
            self.result_label.config(text=result_text)
        except Exception as e:
            messagebox.showerror("Error", f"An error occurred: {e}")
if __name__ == "__main__":
    root = tk.Tk()
    app = StudentFilterApp(root)
    root.mainloop()

import tkinter as tk
from tkinter import messagebox
def filter_students_by_criteria(students, min_height, min_weight):
    '''
    Filters students based on minimum height and weight criteria.
    Parameters:
    students (dict): A dictionary with student names as keys and tuples of (height, weight) as values.
    min_height (float): The minimum height to filter students.
    min_weight (float): The minimum weight to filter students.
    Returns:
    dict: A dictionary of students who meet the criteria.
    '''
    filtered_students = {name: (height, weight) for name, (height, weight) in students.items()
                         if height > min_height and weight > min_weight}
    return filtered_students
class StudentFilterApp:
    '''
    A GUI application to filter students based on height and weight criteria.
    '''
    def __init__(self, root):
        self.root = root
        self.root.title("Student Filter App")
        self.create_widgets()
    def create_widgets(self):
        # Labels and Entries for student data
        self.student_data_label = tk.Label(self.root, text="Enter Student Data (name:height,weight):")
        self.student_data_label.pack()
        self.student_data_entry = tk.Entry(self.root, width=50)
        self.student_data_entry.pack()
        # Labels and Entries for minimum height and weight
        self.min_height_label = tk.Label(self.root, text="Enter Minimum Height:")
        self.min_height_label.pack()
        self.min_height_entry = tk.Entry(self.root)
        self.min_height_entry.pack()
        self.min_weight_label = tk.Label(self.root, text="Enter Minimum Weight:")
        self.min_weight_label.pack()
        self.min_weight_entry = tk.Entry(self.root)
        self.min_weight_entry.pack()
        # Filter Button
        self.filter_button = tk.Button(self.root, text="Filter Students", command=self.filter_students)
        self.filter_button.pack()
        # Result Display
        self.result_label = tk.Label(self.root, text="")
        self.result_label.pack()
    def filter_students(self):
        try:
            # Parse student data
            student_data = self.student_data_entry.get()
            students = {}
            for entry in student_data.split(','):
                name, hw = entry.split(':')
                height, weight = map(float, hw.split(','))
                students[name.strip()] = (height, weight)
            # Get minimum height and weight
            min_height = float(self.min_height_entry.get())
            min_weight = float(self.min_weight_entry.get())
            # Filter students
            filtered_students = filter_students_by_criteria(students, min_height, min_weight)
            # Display results
            result_text = "Filtered Students:\n" + "\n".join([f"{name}: {hw}" for name, hw in filtered_students.items()])
            self.result_label.config(text=result_text)
        except Exception as e:
            messagebox.showerror("Error", f"An error occurred: {e}")
if __name__ == "__main__":
    root = tk.Tk()
    app = StudentFilterApp(root)
    root.mainloop()

import tkinter as tk
from tkinter import messagebox
def filter_students_by_criteria(students, min_height, min_weight):
    '''
    Filters students based on minimum height and weight criteria.
    Parameters:
    students (dict): A dictionary with student names as keys and tuples of (height, weight) as values.
    min_height (float): The minimum height to filter students.
    min_weight (float): The minimum weight to filter students.
    Returns:
    dict: A dictionary of students who meet the criteria.
    '''
    filtered_students = {name: (height, weight) for name, (height, weight) in students.items()
                         if height > min_height and weight > min_weight}
    return filtered_students
class StudentFilterApp:
    '''
    A GUI application to filter students based on height and weight criteria.
    '''
    def __init__(self, root):
        self.root = root
        self.root.title("Student Filter App")
        self.create_widgets()
    def create_widgets(self):
        # Labels and Entries for student data
        self.student_data_label = tk.Label(self.root, text="Enter Student Data (name:height,weight):")
        self.student_data_label.pack()
        self.student_data_entry = tk.Entry(self.root, width=50)
        self.student_data_entry.pack()
        # Labels and Entries for minimum height and weight
        self.min_height_label = tk.Label(self.root, text="Enter Minimum Height:")
        self.min_height_label.pack()
        self.min_height_entry = tk.Entry(self.root)
        self.min_height_entry.pack()
        self.min_weight_label = tk.Label(self.root, text="Enter Minimum Weight:")
        self.min_weight_label.pack()
        self.min_weight_entry = tk.Entry(self.root)
        self.min_weight_entry.pack()
        # Filter Button
        self.filter_button = tk.Button(self.root, text="Filter Students", command=self.filter_students)
        self.filter_button.pack()
        # Result Display
        self.result_label = tk.Label(self.root, text="")
        self.result_label.pack()
    def filter_students(self):
        try:
            # Parse student data
            student_data = self.student_data_entry.get()
            students = {}
            for entry in student_data.split(','):
                name, hw = entry.split(':')
                height, weight = map(float, hw.split(','))
                students[name.strip()] = (height, weight)
            # Get minimum height and weight
            min_height = float(self.min_height_entry.get())
            min_weight = float(self.min_weight_entry.get())
            # Filter students
            filtered_students = filter_students_by_criteria(students, min_height, min_weight)
            # Display results
            result_text = "Filtered Students:\n" + "\n".join([f"{name}: {hw}" for name, hw in filtered_students.items()])
            self.result_label.config(text=result_text)
        except Exception as e:
            messagebox.showerror("Error", f"An error occurred: {e}")
if __name__ == "__main__":
    root = tk.Tk()
    app = StudentFilterApp(root)
    root.mainloop()
