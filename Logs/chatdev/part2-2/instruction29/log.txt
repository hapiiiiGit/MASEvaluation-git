'''

This file contains the main application logic and GUI implementation for the Integer Counter application.

'''

import tkinter as tk
from tkinter import messagebox
from utils import count_integers_in_list


class IntegerCounterApp:
    '''
    A GUI application that allows users to input a list and calculates the number of integer elements in the list.
    '''
    def __init__(self, root):
        self.root = root
        self.root.title("Integer Counter")
        self.root.geometry("400x200")
        self.create_widgets()

    def create_widgets(self):
        '''
        Creates and organizes GUI components.
        '''
        self.label = tk.Label(self.root, text="Enter a list (comma-separated):", font=("Arial", 12))
        self.label.pack(pady=10)

        self.entry = tk.Entry(self.root, width=50)
        self.entry.pack(pady=10)

        self.button = tk.Button(self.root, text="Count Integers", command=self.process_input)
        self.button.pack(pady=10)

        self.result_label = tk.Label(self.root, text="", font=("Arial", 12))
        self.result_label.pack(pady=10)

    def process_input(self):
        '''
        Handles user input, calls the counting function, and displays the result.
        '''
        user_input = self.entry.get()
        try:
            # Convert the input string to a list
            input_list = eval(user_input)
            if not isinstance(input_list, list):
                raise ValueError("Input must be a list.")
            
            # Count integers in the list
            integer_count = count_integers_in_list(input_list)
            self.result_label.config(text=f"Number of integers: {integer_count}")
        except Exception as e:
            messagebox.showerror("Error", f"Invalid input: {e}")


if __name__ == "__main__":
    root = tk.Tk()
    app = IntegerCounterApp(root)
    root.mainloop()

'''

This file contains utility functions for the Integer Counter application.

def count_integers_in_list(input_list):
    '''
    Counts the number of integer elements in the given list.

    Args:
        input_list (list): The list to process.

    Returns:
        int: The count of integer elements in the list.
    '''
    return sum(1 for item in input_list if isinstance(item, int))

This file contains the main application logic and GUI implementation for the Integer Counter application.

import tkinter as tk
from tkinter import messagebox
from utils import count_integers_in_list


class IntegerCounterApp:
    '''
    A GUI application that allows users to input a list and calculates the number of integer elements in the list.
    '''
    def __init__(self, root):
        self.root = root
        self.root.title("Integer Counter")
        self.root.geometry("400x200")
        self.create_widgets()

    def create_widgets(self):
        '''
        Creates and organizes GUI components.
        '''
        self.label = tk.Label(self.root, text="Enter a list (comma-separated):", font=("Arial", 12))
        self.label.pack(pady=10)

        self.entry = tk.Entry(self.root, width=50)
        self.entry.pack(pady=10)

        self.button = tk.Button(self.root, text="Count Integers", command=self.process_input)
        self.button.pack(pady=10)

        self.result_label = tk.Label(self.root, text="", font=("Arial", 12))
        self.result_label.pack(pady=10)

    def process_input(self):
        '''
        Handles user input, calls the counting function, and displays the result.
        '''
        user_input = self.entry.get()
        try:
            # Convert the input string to a list
            input_list = eval(user_input)
            if not isinstance(input_list, list):
                raise ValueError("Input must be a list.")
            
            # Count integers in the list
            integer_count = count_integers_in_list(input_list)
            self.result_label.config(text=f"Number of integers: {integer_count}")
        except Exception as e:
            messagebox.showerror("Error", f"Invalid input: {e}")


if __name__ == "__main__":
    root = tk.Tk()
    app = IntegerCounterApp(root)
    root.mainloop()

import tkinter as tk
from tkinter import messagebox
from utils import count_integers_in_list
class IntegerCounterApp:
    '''
    A GUI application that allows users to input a list and calculates the number of integer elements in the list.
    '''
    def __init__(self, root):
        self.root = root
        self.root.title("Integer Counter")
        self.root.geometry("400x200")
        self.create_widgets()
    def create_widgets(self):
        '''
        Creates and organizes GUI components.
        '''
        self.label = tk.Label(self.root, text="Enter a list (comma-separated):", font=("Arial", 12))
        self.label.pack(pady=10)
        self.entry = tk.Entry(self.root, width=50)
        self.entry.pack(pady=10)
        self.button = tk.Button(self.root, text="Count Integers", command=self.process_input)
        self.button.pack(pady=10)
        self.result_label = tk.Label(self.root, text="", font=("Arial", 12))
        self.result_label.pack(pady=10)
    def process_input(self):
        '''
        Handles user input, calls the counting function, and displays the result.
        '''
        user_input = self.entry.get()
        try:
            # Convert the input string to a list
            input_list = eval(user_input)
            if not isinstance(input_list, list):
                raise ValueError("Input must be a list.")
            # Count integers in the list
            integer_count = count_integers_in_list(input_list)
            self.result_label.config(text=f"Number of integers: {integer_count}")
        except Exception as e:
            messagebox.showerror("Error", f"Invalid input: {e}")
if __name__ == "__main__":
    root = tk.Tk()
    app = IntegerCounterApp(root)
    root.mainloop()

import tkinter as tk
from tkinter import messagebox
from utils import count_integers_in_list
class IntegerCounterApp:
    '''
    A GUI application that allows users to input a list and calculates the number of integer elements in the list.
    '''
    def __init__(self, root):
        self.root = root
        self.root.title("Integer Counter")
        self.root.geometry("400x200")
        self.create_widgets()
    def create_widgets(self):
        '''
        Creates and organizes GUI components.
        '''
        self.label = tk.Label(self.root, text="Enter a list (comma-separated):", font=("Arial", 12))
        self.label.pack(pady=10)
        self.entry = tk.Entry(self.root, width=50)
        self.entry.pack(pady=10)
        self.button = tk.Button(self.root, text="Count Integers", command=self.process_input)
        self.button.pack(pady=10)
        self.result_label = tk.Label(self.root, text="", font=("Arial", 12))
        self.result_label.pack(pady=10)
    def process_input(self):
        '''
        Handles user input, calls the counting function, and displays the result.
        '''
        user_input = self.entry.get()
        try:
            # Convert the input string to a list
            input_list = eval(user_input)
            if not isinstance(input_list, list):
                raise ValueError("Input must be a list.")
            # Count integers in the list
            integer_count = count_integers_in_list(input_list)
            self.result_label.config(text=f"Number of integers: {integer_count}")
        except Exception as e:
            messagebox.showerror("Error", f"Invalid input: {e}")
if __name__ == "__main__":
    root = tk.Tk()
    app = IntegerCounterApp(root)
    root.mainloop()

import tkinter as tk
from tkinter import messagebox
from utils import count_integers_in_list
class IntegerCounterApp:
    '''
    A GUI application that allows users to input a list and calculates the number of integer elements in the list.
    '''
    def __init__(self, root):
        self.root = root
        self.root.title("Integer Counter")
        self.root.geometry("400x200")
        self.create_widgets()
    def create_widgets(self):
        '''
        Creates and organizes GUI components.
        '''
        self.label = tk.Label(self.root, text="Enter a list (comma-separated):", font=("Arial", 12))
        self.label.pack(pady=10)
        self.entry = tk.Entry(self.root, width=50)
        self.entry.pack(pady=10)
        self.button = tk.Button(self.root, text="Count Integers", command=self.process_input)
        self.button.pack(pady=10)
        self.result_label = tk.Label(self.root, text="", font=("Arial", 12))
        self.result_label.pack(pady=10)
    def process_input(self):
        '''
        Handles user input, calls the counting function, and displays the result.
        '''
        user_input = self.entry.get()
        try:
            # Convert the input string to a list
            input_list = eval(user_input)
            if not isinstance(input_list, list):
                raise ValueError("Input must be a list.")
            # Count integers in the list
            integer_count = count_integers_in_list(input_list)
            self.result_label.config(text=f"Number of integers: {integer_count}")
        except Exception as e:
            messagebox.showerror("Error", f"Invalid input: {e}")
if __name__ == "__main__":
    root = tk.Tk()
    app = IntegerCounterApp(root)
    root.mainloop()
