'''

main.py

'''

import tkinter as tk
from tkinter import messagebox

def find_second_smallest_number(numbers):
    '''
    Find the second smallest number in a list of numbers.
    
    Parameters:
    numbers (list): A list of integers.
    
    Returns:
    int: The second smallest number in the list.
    '''
    if len(numbers) < 2:
        raise ValueError("List must contain at least two numbers.")
    
    first, second = float('inf'), float('inf')
    for number in numbers:
        if number < first:
            first, second = number, first
        elif first < number < second:
            second = number
    
    if second == float('inf'):
        raise ValueError("There is no second smallest number.")
    
    return second

class SecondSmallestFinderApp:
    '''
    A GUI application to find the second smallest number in a list.
    '''
    def __init__(self, master):
        self.master = master
        master.title("Second Smallest Finder")
        
        self.label = tk.Label(master, text="Enter numbers separated by commas:")
        self.label.pack()
        
        self.entry = tk.Entry(master, width=50)
        self.entry.pack()
        
        self.find_button = tk.Button(master, text="Find Second Smallest", command=self.find_second_smallest)
        self.find_button.pack()
        
        self.result_label = tk.Label(master, text="", fg="blue")
        self.result_label.pack()
    
    def find_second_smallest(self):
        try:
            numbers = list(map(int, self.entry.get().split(',')))
            second_smallest = find_second_smallest_number(numbers)
            self.display_result(f"The second smallest number is: {second_smallest}")
        except ValueError as e:
            messagebox.showerror("Error", str(e))
    
    def display_result(self, result):
        self.result_label.config(text=result)

if __name__ == "__main__":
    root = tk.Tk()
    app = SecondSmallestFinderApp(root)
    root.mainloop()

import tkinter as tk
from tkinter import messagebox

def find_second_smallest_number(numbers):
    '''
    Find the second smallest number in a list of numbers.
    
    Parameters:
    numbers (list): A list of integers.
    
    Returns:
    int: The second smallest number in the list.
    '''
    if len(numbers) < 2:
        raise ValueError("List must contain at least two numbers.")
    
    first, second = float('inf'), float('inf')
    for number in numbers:
        if number < first:
            first, second = number, first
        elif first < number < second:
            second = number
    
    if second == float('inf'):
        raise ValueError("There is no second smallest number.")
    
    return second

class SecondSmallestFinderApp:
    '''
    A GUI application to find the second smallest number in a list.
    '''
    def __init__(self, master):
        self.master = master
        master.title("Second Smallest Finder")
        
        self.label = tk.Label(master, text="Enter numbers separated by commas:")
        self.label.pack()
        
        self.entry = tk.Entry(master, width=50)
        self.entry.pack()
        
        self.find_button = tk.Button(master, text="Find Second Smallest", command=self.find_second_smallest)
        self.find_button.pack()
        
        self.result_label = tk.Label(master, text="", fg="blue")
        self.result_label.pack()
    
    def find_second_smallest(self):
        try:
            numbers = list(map(int, self.entry.get().split(',')))
            second_smallest = find_second_smallest_number(numbers)
            self.display_result(f"The second smallest number is: {second_smallest}")
        except ValueError as e:
            messagebox.showerror("Error", str(e))
    
    def display_result(self, result):
        self.result_label.config(text=result)

if __name__ == "__main__":
    root = tk.Tk()
    app = SecondSmallestFinderApp(root)
    root.mainloop()

import tkinter as tk
from tkinter import messagebox
def find_second_smallest_number(numbers):
    '''
    Find the second smallest distinct number in a list of numbers.
    Parameters:
    numbers (list): A list of integers.
    Returns:
    int: The second smallest distinct number in the list.
    '''
    if len(numbers) < 2:
        raise ValueError("List must contain at least two numbers.")
    unique_numbers = list(set(numbers))
    if len(unique_numbers) < 2:
        raise ValueError("There is no second smallest distinct number.")
    unique_numbers.sort()
    return unique_numbers[1]
class SecondSmallestFinderApp:
    '''
    A GUI application to find the second smallest number in a list.
    '''
    def __init__(self, master):
        self.master = master
        master.title("Second Smallest Finder")
        self.label = tk.Label(master, text="Enter numbers separated by commas:")
        self.label.pack()
        self.entry = tk.Entry(master, width=50)
        self.entry.pack()
        self.find_button = tk.Button(master, text="Find Second Smallest", command=self.find_second_smallest)
        self.find_button.pack()
        self.result_label = tk.Label(master, text="", fg="blue")
        self.result_label.pack()
    def find_second_smallest(self):
        '''
        Retrieve user input, process it, and display the second smallest number or an error message.
        '''
        try:
            numbers = list(map(int, self.entry.get().split(',')))
            second_smallest = find_second_smallest_number(numbers)
            self.display_result(f"The second smallest number is: {second_smallest}")
        except ValueError as e:
            messagebox.showerror("Error", str(e))
        except Exception as e:
            messagebox.showerror("Error", "Invalid input. Please enter integers separated by commas.")
    def display_result(self, result):
        '''
        Display the result in the GUI.
        Parameters:
        result (str): The result message to display.
        '''
        self.result_label.config(text=result)
if __name__ == "__main__":
    root = tk.Tk()

import tkinter as tk
from tkinter import messagebox
def find_second_smallest_number(numbers):
    '''
    Find the second smallest distinct number in a list of numbers.
    Parameters:
    numbers (list): A list of integers.
    Returns:
    int: The second smallest distinct number in the list.
    '''
    if len(numbers) < 2:
        raise ValueError("List must contain at least two numbers.")
    unique_numbers = list(set(numbers))
    if len(unique_numbers) < 2:
        raise ValueError("There is no second smallest distinct number.")
    unique_numbers.sort()
    return unique_numbers[1]
class SecondSmallestFinderApp:
    '''
    A GUI application to find the second smallest number in a list.
    '''
    def __init__(self, master):
        self.master = master
        master.title("Second Smallest Finder")
        self.label = tk.Label(master, text="Enter numbers separated by commas:")
        self.label.pack()
        self.entry = tk.Entry(master, width=50)
        self.entry.pack()
        self.find_button = tk.Button(master, text="Find Second Smallest", command=self.find_second_smallest)
        self.find_button.pack()
        self.result_label = tk.Label(master, text="", fg="blue")
        self.result_label.pack()
    def find_second_smallest(self):
        '''
        Retrieve user input, process it, and display the second smallest number or an error message.
        '''
        try:
            numbers = list(map(int, self.entry.get().split(',')))
            second_smallest = find_second_smallest_number(numbers)
            self.display_result(f"The second smallest number is: {second_smallest}")
        except ValueError as e:
            messagebox.showerror("Error", str(e))
        except Exception as e:
            messagebox.showerror("Error", "Invalid input. Please enter integers separated by commas.")
    def display_result(self, result):
        '''
        Display the result in the GUI.
        Parameters:
        result (str): The result message to display.
        '''
        self.result_label.config(text=result)
if __name__ == "__main__":
    root = tk.Tk()
    app = SecondSmallestFinderApp(root)
    root.mainloop()
