def multiply_adjacent_elements(input_tuple):
    '''
    Takes a tuple of numbers and returns a tuple where each element is the product of adjacent elements.
    Parameters:
    input_tuple (tuple): A tuple of numbers (t_1, ..., t_{N+1}).
    Returns:
    tuple: A tuple of length N where each element is t_i * t_{i+1}.
    '''
    if not isinstance(input_tuple, tuple):
        raise TypeError("Input must be a tuple.")
    if not all(isinstance(x, (int, float)) for x in input_tuple):
        raise ValueError("All elements in the tuple must be numbers.")
    return tuple(input_tuple[i] * input_tuple[i + 1] for i in range(len(input_tuple) - 1))

'''
Module for tuple operations.
'''
def multiply_adjacent_elements(input_tuple):
    '''
    Takes a tuple of numbers and returns a tuple where each element is the product of adjacent elements.
    Parameters:
    input_tuple (tuple): A tuple of numbers (t_1, ..., t_{N+1}).
    Returns:
    tuple: A tuple of length N where each element is t_i * t_{i+1}.
    '''
    if not isinstance(input_tuple, tuple):
        raise TypeError("Input must be a tuple.")
    if not all(isinstance(x, (int, float)) for x in input_tuple):
        raise ValueError("All elements in the tuple must be numbers.")
    return tuple(input_tuple[i] * input_tuple[i + 1] for i in range(len(input_tuple) - 1))

'''
Module for tuple operations.
'''
def multiply_adjacent_elements(input_tuple):
    '''
    Takes a tuple of numbers and returns a tuple where each element is the product of adjacent elements.
    Parameters:
    input_tuple (tuple): A tuple of numbers (t_1, ..., t_{N+1}).
    Returns:
    tuple: A tuple of length N where each element is t_i * t_{i+1}.
    '''
    if not isinstance(input_tuple, tuple):
        raise TypeError("Input must be a tuple.")
    if not all(isinstance(x, (int, float)) for x in input_tuple):
        raise ValueError("All elements in the tuple must be numbers.")
    return tuple(input_tuple[i] * input_tuple[i + 1] for i in range(len(input_tuple) - 1))
