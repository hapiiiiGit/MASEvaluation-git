def count_most_common_words(word_dict):
    from collections import Counter
    
    # Flatten the dictionary values into a single list of words
    words = [word for sublist in word_dict.values() for word in sublist]
    
    # Count the occurrences of each word
    word_counts = Counter(words)
    
    # Return the most common words and their counts
    return word_counts.most_common()
