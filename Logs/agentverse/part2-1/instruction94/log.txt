def min_length_sublist(lst, target_sum):
    n = len(lst)
    min_length = float('inf')
    current_sum = 0
    start = 0

    for end in range(n):
        current_sum += lst[end]

        while current_sum >= target_sum:
            min_length = min(min_length, end - start + 1)
            current_sum -= lst[start]
