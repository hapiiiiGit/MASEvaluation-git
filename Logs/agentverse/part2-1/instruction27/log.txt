        return 1
    # Calculate the binomial coefficient using the formula C(n, k) = n! / (k! * (n-k)!)
    numerator = 1
    denominator = 1
    for i in range(1, k + 1):
        numerator *= n - (i - 1)
        denominator *= i
    return numerator // denominator
