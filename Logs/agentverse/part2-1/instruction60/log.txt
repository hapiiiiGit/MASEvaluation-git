def count_substrings(s):
    count = 0
    n = len(s)
    
    for i in range(n):
        digit_sum = 0
        for j in range(i, n):
            digit_sum += int(s[j])  # Incrementally add the digit to the sum
            length = j - i + 1  # Calculate the length of the substring
            if digit_sum == length:  # Check if the sum of digits equals the length
                count += 1
    
    return count
