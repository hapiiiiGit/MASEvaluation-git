    and false otherwise.
    Knowing that (a) is less then 100. 
    Example:
    is_multiply_prime(30) == True
    30 = 2 * 3 * 5
    """
    if a < 2:
        return False

    def is_prime(n):

            return False

                if primes[i] * primes[j] * primes[k] == a:
                    return True
    return False
