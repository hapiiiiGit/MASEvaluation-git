def largest_divisor(n: int) -> int:
    """ For a given number n, find the largest number that divides n evenly, smaller than n
    >>> largest_divisor(15)
    5
    """
    if n <= 1:
        return 0  # No divisors for 0 or 1
    for i in range(int(n**0.5), 0, -1):
        if n % i == 0:
            if i < n:
                return n // i  # Return the larger divisor found
            return i  # Return the divisor if it's less than n
    return 1  # In case n is prime
