    """
    Given a positive integer n, return the count of the numbers of n-digit
    positive integers that start or end with 1.
    """
    if n == 1:
        return 1  # Only the number 1 itself

    # Count of n-digit numbers that start with 1
    start_with_1 = 10**(n-1)  # 1 followed by (n-1) digits (0-9)

    # Count of n-digit numbers that end with 1
    end_with_1 = 9 * 10**(n-2)  # (1-9) for the first digit and (0-9) for (n-2) digits

    # Total count, subtracting the overlap (numbers that both start and end with 1)
    overlap = 1  # Only the number 1...1 (n times)

    return start_with_1 + end_with_1 - overlap
