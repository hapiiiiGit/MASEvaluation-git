from rest_framework.views import APIView
from rest_framework.response import Response
from rest_framework.permissions import IsAuthenticated
import openai

openai.api_key = 'your_openai_api_key_here'  # Replace with your actual key

class ChatbotAPIView(APIView):
    permission_classes = [IsAuthenticated]

    def post(self, request):
        user_message = request.data.get('message')
        if not user_message:
            return Response({"error": "Message is required."}, status=400)

        response = openai.ChatCompletion.create(
            model="gpt-4",
            messages=[
                {"role": "system", "content": "You are a helpful assistant for global summits and conferences."},
                {"role": "user", "content": user_message}
            ]
        )
        answer = response['choices'][0]['message']['content']
        return Response({"response": answer})

from rest_framework import viewsets, permissions, status
from rest_framework.decorators import action
from rest_framework.response import Response
from .models import Payment
from .serializers import PaymentSerializer
from tickets.models import Ticket

class PaymentViewSet(viewsets.ModelViewSet):
    serializer_class = PaymentSerializer
    permission_classes = [permissions.IsAuthenticated]

    def get_queryset(self):
        return Payment.objects.filter(ticket__user=self.request.user)

    @action(detail=False, methods=['post'])
    def process_payment(self, request):
        ticket_id = request.data.get('ticket_id')
        amount = request.data.get('amount')
        # Here you would integrate with a payment gateway like Stripe or PayPal
        # For demonstration, we simulate a successful payment
        try:
            ticket = Ticket.objects.get(id=ticket_id, user=request.user)
        except Ticket.DoesNotExist:
            return Response({"error": "Ticket not found."}, status=status.HTTP_404_NOT_FOUND)

        payment = Payment.objects.create(
            ticket=ticket,
            amount=amount,
            payment_status='completed',
            transaction_id='SIMULATED_TXN_123456'
        )
        ticket.is_paid = True
        ticket.save()
        serializer = PaymentSerializer(payment)
        return Response(serializer.data, status=status.HTTP_201_CREATED)
