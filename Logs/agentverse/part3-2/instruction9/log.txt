# Please help me implement a Python program to set up Google Document AI for extracting specified fields from supplier invoices in image or PDF format, ensuring the model handles various scan conditions, and outputs the results in JSON or CSV format for database storage.

from google.cloud import documentai_v1 as documentai
from google.cloud.documentai_v1 import DocumentProcessorServiceClient
import json
import csv
import os

# Set your Google Cloud project details
PROJECT_ID = "your-project-id"
LOCATION = "us"  # Format is 'us' or 'eu'
PROCESSOR_ID = "your-processor-id"  # The processor ID for your invoice processor

# Path to the local file (image or PDF)
FILE_PATH = "path/to/your/invoice.pdf"

# Output paths
OUTPUT_JSON = "output_invoice.json"
OUTPUT_CSV = "output_invoice.csv"

def process_document(file_path):
    client = DocumentProcessorServiceClient()

    # The full resource name of the processor
    name = f"projects/{PROJECT_ID}/locations/{LOCATION}/processors/{PROCESSOR_ID}"

    # Read the file into memory
    with open(file_path, "rb") as image:
        image_content = image.read()

    # Configure the process request
    document = {"content": image_content, "mime_type": "application/pdf" if file_path.lower().endswith(".pdf") else "image/jpeg"}

    request = {
        "name": name,
        "raw_document": document,
    }

    # Use the client to process the document
    result = client.process_document(request=request)
    document = result.document

    return document

def extract_fields(document):
    # Extract entities from the document
    fields = {}
    for entity in document.entities:
        # The entity type corresponds to the field name, e.g. "invoice_date", "supplier_name"
        field_name = entity.type_
        field_value = entity.mention_text
        fields[field_name] = field_value
    return fields

def save_json(data, output_path):
    with open(output_path, "w") as f:
        json.dump(data, f, indent=2)

def save_csv(data, output_path):
    # Save the extracted fields as a single-row CSV
    with open(output_path, "w", newline="") as csvfile:
        writer = csv.DictWriter(csvfile, fieldnames=data.keys())
        writer.writeheader()
        writer.writerow(data)

def main():
    document = process_document(FILE_PATH)
    fields = extract_fields(document)
    save_json(fields, OUTPUT_JSON)
    save_csv(fields, OUTPUT_CSV)
    print(f"Extraction complete. Results saved to {OUTPUT_JSON} and {OUTPUT_CSV}")

if __name__ == "__main__":
    # Make sure to set the environment variable for authentication before running:
    # export GOOGLE_APPLICATION_CREDENTIALS="path/to/your/service-account-file.json"
    main()
