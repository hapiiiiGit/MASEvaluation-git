    "https://yourmachform3.com/api/get_entries"

def fetch_machform_data():
    all_entries = []
    headers = {"Authorization": f"Bearer {MACHFORM_API_KEY}"}
    for url in MACHFORM_API_URLS:
        response = requests.get(url, headers=headers)
        response.raise_for_status()
        data = response.json()
        # Assuming data['entries'] contains the list of form entries
        all_entries.extend(data.get('entries', []))
    return all_entries

def append_to_master_sheet(entries):
    # Prepare rows for appending
    rows = []
    for entry in entries:
        # Extract relevant fields from entry, example:

            entry.get('student_id', ''),

        spreadsheetId=GOOGLE_SHEETS_ID,
        range=f"{MASTER_SHEET_NAME}!A1",
        valueInputOption="USER_ENTERED",
        insertDataOption="INSERT_ROWS",
        body=body
    ).execute()

def generate_academic_pathway(student_data):
    prompt = f"""
    Given the following student data, generate a personalized academic pathway:
    {json.dumps(student_data, indent=2)}
    Provide a clear step-by-step plan.
    """
    response = openai_client.chat.completions.create(
        model="gpt-4o-mini",
        messages=[{"role": "user", "content": prompt}],
        max_tokens=500
    )
    pathway = response.choices[0].message.content.strip()
    return pathway

def copy_google_doc_template():
    body = {
        'name': 'Student Academic Pathway',
        'parents': [GOOGLE_DRIVE_FOLDER_ID]
    }
    copied_file = drive_service.files().copy(
        fileId=GOOGLE_DOCS_TEMPLATE_ID,
        body=body
    ).execute()
    return copied_file['id']

def fill_google_doc(doc_id, student_data, pathway_text):
    requests = []
    # Replace placeholders in the template with actual data
    replacements = {
        "{{StudentName}}": student_data.get('name', ''),
        "{{StudentID}}": student_data.get('student_id', ''),
        "{{Program}}": student_data.get('program', ''),
        "{{AcademicPathway}}": pathway_text
    }
    for placeholder, replacement in replacements.items():
        requests.append({
            'replaceAllText': {
                'containsText': {
                    'text': placeholder,
                    'matchCase': True
                },
                'replaceText': replacement
            }
        })
    docs_service.documents().batchUpdate(
        documentId=doc_id,
        body={'requests': requests}
    ).execute()

def export_doc_as_pdf(doc_id, output_path):
    request = drive_service.files().export_media(fileId=doc_id,
                                                 mimeType='application/pdf')
    fh = io.FileIO(output_path, 'wb')
    downloader = MediaIoBaseDownload(fh, request)
    done = False
    while done is False:
        status, done = downloader.next_chunk()
    fh.close()

def send_email_with_attachment(to_email, subject, body_text, attachment_path):
    msg = MIMEMultipart()
    msg['From'] = EMAIL_SENDER
    msg['To'] = to_email
    msg['Subject'] = subject

    msg.attach(MIMEText(body_text, 'plain'))

    with open(attachment_path, "rb") as f:
        part = MIMEApplication(f.read(), Name=os.path.basename(attachment_path))
    part['Content-Disposition'] = f'attachment; filename="{os.path.basename(attachment_path)}"'
    msg.attach(part)

    with smtplib.SMTP(EMAIL_SMTP_SERVER, EMAIL_SMTP_PORT) as server:
        server.starttls()
        server.login(EMAIL_SMTP_USERNAME, EMAIL_SMTP_PASSWORD)
        server.send_message(msg)

def store_in_salesforce(student_data, pathway_text, pdf_path):
    # Create or update a Salesforce record for the student
    sf_contact = None
    try:
        sf_contact = sf.Contact.get_by_custom_id('Student_ID__c', student_data.get('student_id'))
    except Exception:
        pass

    contact_data = {
        'FirstName': student_data.get('name', '').split()[0],
        'LastName': student_data.get('name', '').split()[-1] if len(student_data.get('name', '').split()) > 1 else '',
        'Email': student_data.get('email', ''),
        'Student_ID__c': student_data.get('student_id', ''),
        'Program__c': student_data.get('program', ''),
        'Academic_Pathway__c': pathway_text
    }

    if sf_contact:
        sf.Contact.update(sf_contact['Id'], contact_data)
        contact_id = sf_contact['Id']
    else:
        contact_id = sf.Contact.create(contact_data)['id']

    # Upload PDF as attachment
    with open(pdf_path, "rb") as f:
        pdf_content = f.read()
    attachment = {
        'ParentId': contact_id,
        'Name': os.path.basename(pdf_path),
        'Body': base64.b64encode(pdf_content).decode('utf-8'),
        'ContentType': 'application/pdf'
    }
    sf.Attachment.create(attachment)

def main():
    # Step 1: Fetch data from MachForms
    entries = fetch_machform_data()

            'name': entry.get('name', ''),

        pdf_path = f"/tmp/{student_data['student_id']}_academic_pathway.pdf"
        export_doc_as_pdf(doc_id, pdf_path)
        email_subject = "Your Personalized Academic Pathway"
        email_body = f"Dear {student_data['name']},\n\nPlease find attached your personalized academic pathway.\n\nBest regards,\nAdmissions Team"
        send_email_with_attachment(student_data['email'], email_subject, email_body, pdf_path)
        store_in_salesforce(student_data, pathway_text, pdf_path)

    msg.attach(part)

    with smtplib.SMTP(EMAIL_SMTP_SERVER, EMAIL_SMTP_PORT) as server:

    # Try to find existing contact by Student_ID__c custom field

            'Email': student_data.get('email', ''),

        })
    else:
        contact = sf.Contact.create({
            'FirstName': student_data.get('name', '').split()[0] if student_data.get('name') else '',
            'LastName': student_data.get('name', '').split()[-1] if student_data.get('name') else '',
            'Email': student_data.get('email', ''),
            'Student_ID__c': student_data.get('student_id', ''),
            'Program__c': student_data.get('program', ''),
            'Academic_Pathway__c': pathway_text
        })
        contact_id = contact['id']

    # Upload PDF as attachment
    with open(pdf_path, "rb") as f:
        pdf_content = f.read()
    attachment = {
        'ParentId': contact_id,
        'Name': os.path.basename(pdf_path),
        'Body': base64.b64encode(pdf_content).decode('utf-8'),
        'ContentType': 'application/pdf'
    }
    sf.Attachment.create(attachment)

def append_to_master_sheet(entries):

    rows = []
    for entry in entries:

            entry.get('student_id', ''),

def generate_academic_pathway(student_data):
    prompt = f"""
Given the following student data, generate a personalized academic pathway:
{json.dumps(student_data, indent=2)}
Provide a clear step-by-step plan.
"""
    response = openai.ChatCompletion.create(
        model="gpt-4o-mini",
        messages=[{"role": "user", "content": prompt}],
        max_tokens=500
    )
    pathway = response.choices[0].message.content.strip()
    return pathway

def fill_google_doc(doc_id, student_data, pathway_text):
    requests = []
    replacements = {
        "{{StudentName}}": student_data.get('name', ''),
        "{{StudentID}}": student_data.get('student_id', ''),
        "{{Program}}": student_data.get('program', ''),
        "{{AcademicPathway}}": pathway_text
    }
    for placeholder, replacement in replacements.items():
        requests.append({
            'replaceAllText': {
                'containsText': {
                    'text': placeholder,
                    'matchCase': True
                },
                'replaceText': replacement
            }
        })
    docs_service.documents().batchUpdate(
        documentId=doc_id,
        body={'requests': requests}
    ).execute()

def export_doc_as_pdf(doc_id, output_path):
    request = drive_service.files().export_media(fileId=doc_id,
                                                 mimeType='application/pdf')
    with io.FileIO(output_path, 'wb') as fh:
        downloader = MediaIoBaseDownload(fh, request)
        done = False
        while not done:
            status, done = downloader.next_chunk()

        f"Given the student data: {json.dumps(student_data)}, "

    )

        temperature=0.7,
    )
    pathway_text = response.choices[0].message.content.strip()
    return pathway_text

# Function to send email with PDF attachment
def send_email_with_pdf(smtp_server, smtp_port, smtp_user, smtp_password,
                        sender_email, recipient_email, subject, body, pdf_path):
    msg = EmailMessage()
    msg['Subject'] = subject
    msg['From'] = sender_email
    msg['To'] = recipient_email
    msg.set_content(body)

    with open(pdf_path, 'rb') as f:
        pdf_data = f.read()
    msg.add_attachment(pdf_data, maintype='application', subtype='pdf', filename=os.path.basename(pdf_path))

    with smtplib.SMTP_SSL(smtp_server, smtp_port) as server:
        server.login(smtp_user, smtp_password)
        server.send_message(msg)

    with open(pdf_path, 'rb') as f:
        pdf_data = f.read()

    with smtplib.SMTP_SSL(smtp_server, smtp_port) as server:
        server.login(smtp_user, smtp_password)

    prompt = (
        f"Given the student data: {json.dumps(student_data)}, "

# Function to send email with PDF attachment
def send_email_with_pdf(smtp_server, smtp_port, smtp_user, smtp_password,
                        sender_email, recipient_email, subject, body, pdf_path):
    msg = EmailMessage()
    msg['Subject'] = subject
    msg['From'] = sender_email
    msg['To'] = recipient_email
    msg.set_content(body)

    with open(pdf_path, 'rb') as f:
        pdf_data = f.read()
    msg.add_attachment(pdf_data, maintype='application', subtype='pdf', filename=os.path.basename(pdf_path))

    with smtplib.SMTP_SSL(smtp_server, smtp_port) as server:
        server.login(smtp_user, smtp_password)
        server.send_message(msg)

# Function to store the PDF in Salesforce as an Attachment or ContentVersion
def store_pdf_in_salesforce(sf_instance_url, sf_access_token, student_id, pdf_path, filename):
    headers = {
        'Authorization': f'Bearer {sf_access_token}',
        'Content-Type': 'application/json'
    }
    with open(pdf_path, 'rb') as f:
        pdf_data = f.read()
    encoded_pdf = pdf_data.encode('base64') if hasattr(pdf_data, 'encode') else pdf_data.encode('base64')

    # Using ContentVersion to upload file
    content_version_url = f"{sf_instance_url}/services/data/v57.0/sobjects/ContentVersion"
    body = {
        "Title": filename,
        "PathOnClient": filename,
        "VersionData": encoded_pdf,
        "FirstPublishLocationId": student_id  # Assuming student_id is Salesforce Contact/Lead Id
    }
    response = requests.post(content_version_url, headers=headers, json=body)
    if response.status_code not in (200, 201):
        raise Exception(f"Failed to upload PDF to Salesforce: {response.text}")
    return response.json()

# Function to store the PDF in Salesforce as a ContentVersion (file)
def store_pdf_in_salesforce(sf_instance_url, sf_access_token, parent_record_id, pdf_path, filename):
    # Read and base64 encode the PDF file content
    with open(pdf_path, 'rb') as f:
        pdf_data = f.read()
    encoded_pdf = base64.b64encode(pdf_data).decode('utf-8')

    # Prepare the ContentVersion record body
    body = {
        "Title": filename,
        "PathOnClient": filename,
        "VersionData": encoded_pdf,
        "FirstPublishLocationId": parent_record_id  # Salesforce record ID (e.g., Contact or Lead)
    }

    headers = {
        'Authorization': f'Bearer {sf_access_token}',
        'Content-Type': 'application/json'
    }

    url = f"{sf_instance_url}/services/data/v57.0/sobjects/ContentVersion"

    response = requests.post(url, headers=headers, json=body)
    if response.status_code not in (200, 201):
        raise Exception(f"Failed to upload PDF to Salesforce: {response.status_code} {response.text}")
    return response.json()

# Function to append entries from MachForm data to a master Google Sheet,
# avoiding duplicates based on a unique student ID.
def append_to_master_sheet(sheet_service, spreadsheet_id, sheet_name, entries, unique_id_col=0):
    try:
        # Get existing data to check for duplicates
        result = sheet_service.spreadsheets().values().get(
            spreadsheetId=spreadsheet_id,
            range=sheet_name
        ).execute()
        existing_values = result.get('values', [])
        existing_ids = set(row[unique_id_col] for row in existing_values if len(row) > unique_id_col)

        # Filter out entries with duplicate IDs
        new_rows = []
        for entry in entries:
            student_id = entry.get('student_id', '')
            if student_id and student_id not in existing_ids:
                # Convert entry dict to row list matching sheet columns order
                # Assuming columns: student_id, name, email, program, etc.
                row = [
                    student_id,
                    entry.get('name', ''),
                    entry.get('email', ''),
                    entry.get('program', ''),
                    # Add other fields as needed
                ]
                new_rows.append(row)
                existing_ids.add(student_id)

        if new_rows:
            append_body = {
                'values': new_rows
            }
            sheet_service.spreadsheets().values().append(
                spreadsheetId=spreadsheet_id,
                range=sheet_name,
                valueInputOption='USER_ENTERED',
                insertDataOption='INSERT_ROWS',
                body=append_body
            ).execute()
    except HttpError as e:
        print(f"An error occurred while appending to sheet: {e}")
        raise

# Function to create a personalized Google Doc from a template by replacing placeholders
def create_personalized_doc(docs_service, drive_service, template_doc_id, student_data, output_title):
    try:
        # Copy the template
        copied_file = drive_service.files().copy(
            fileId=template_doc_id,
            body={'name': output_title}
        ).execute()
        document_id = copied_file.get('id')

        # Prepare requests to replace placeholders
        requests = []
        placeholders = {
            '{{StudentName}}': student_data.get('name', ''),
            '{{StudentID}}': student_data.get('student_id', ''),
            '{{Program}}': student_data.get('program', ''),
            '{{AcademicPathway}}': student_data.get('academic_pathway', ''),
            # Add other placeholders as needed
        }
        for placeholder, replacement in placeholders.items():
            requests.append({
                'replaceAllText': {
                    'containsText': {
                        'text': placeholder,
                        'matchCase': True
                    },
                    'replaceText': replacement
                }
            })

        docs_service.documents().batchUpdate(
            documentId=document_id,
            body={'requests': requests}
        ).execute()

        return document_id
    except HttpError as e:
        print(f"An error occurred while creating personalized doc: {e}")
        raise

# Function to export a Google Doc as PDF to a local temporary file
def export_doc_as_pdf(drive_service, document_id):
    try:
        request = drive_service.files().export_media(
            fileId=document_id,
            mimeType='application/pdf'
        )
        fh = io.BytesIO()
        downloader = MediaIoBaseDownload(fh, request)
        done = False
        while not done:
            status, done = downloader.next_chunk()
        fh.seek(0)

        # Save to a temporary file
        temp_pdf = tempfile.NamedTemporaryFile(delete=False, suffix='.pdf')
        temp_pdf.write(fh.read())
        temp_pdf.close()
        return temp_pdf.name
    except HttpError as e:
        print(f"An error occurred while exporting doc as PDF: {e}")
        raise
