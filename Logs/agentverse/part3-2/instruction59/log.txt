# Async function to fetch product data from API
async def fetch_from_api(session, api_info, query):
    try:
        url = f"{api_info['api_url']}?q={query}"
        async with async_timeout.timeout(10):
            async with session.get(url) as response:
                if response.status == 200:
                    data = await response.json()
                    # Normalize data to a common format
                    products = []
                    for item in data.get('products', data.get('items', [])):
                        products.append({
                            'name': item.get('name'),
                            'price': float(item.get('price', 0)),
                            'url': AFFILIATE_PREFIX + item.get('url', ''),
                            'source': api_info['name']
                        })
                    return products
    except Exception:
        return None
    return None

# Async function to scrape product data if API fails
async def scrape_products(session, api_info, query):
    try:
        url = api_info['scrape_url'].format(query=query)
        async with async_timeout.timeout(10):
            async with session.get(url) as response:
                if response.status == 200:
                    html = await response.text()
                    soup = BeautifulSoup(html, 'html.parser')
                    products = []
                    # Example scraping logic (to be customized per site)
                    for product_div in soup.select('.product-item'):
                        name_tag = product_div.select_one('.product-title')
                        price_tag = product_div.select_one('.product-price')
                        link_tag = product_div.select_one('a')
                        if name_tag and price_tag and link_tag:
                            try:
                                price_text = price_tag.text.strip().replace('$', '').replace(',', '')
                                price = float(price_text)
                                url = AFFILIATE_PREFIX + link_tag['href']
                                products.append({
                                    'name': name_tag.text.strip(),
                                    'price': price,
                                    'url': url,
                                    'source': api_info['name']
                                })
                            except Exception:
                                continue
                    return products
    except Exception:
        return None
    return None

# Aggregate product data from all sources
async def aggregate_products(query):
    async with aiohttp.ClientSession() as session:
        tasks = []
        for api_info in ECOMMERCE_APIS:
            tasks.append(fetch_from_api(session, api_info, query))
        results = await asyncio.gather(*tasks)

        # For APIs that failed, fallback to scraping
        for i, res in enumerate(results):
            if res is None:
                scraped = await scrape_products(session, ECOMMERCE_APIS[i], query)
                results[i] = scraped if scraped else []

        # Flatten and sort by price
        all_products = [item for sublist in results if sublist for item in sublist]
        all_products.sort(key=lambda x: x['price'])
        return all_products

# Stripe payment intent creation
async def create_payment_intent(request):
    data = await request.json()
    email = data.get('email')
    amount = data.get('amount')  # in cents
    if not email or not amount:
        return web.json_response({'error': 'Missing email or amount'}, status=400)
    try:
        intent = stripe.PaymentIntent.create(
            amount=amount,
            currency='usd',
            receipt_email=email,
            metadata={'integration_check': 'accept_a_payment'}
        )
        return web.json_response({'client_secret': intent.client_secret})
    except Exception as e:
        return web.json_response({'error': str(e)}, status=500)

# Web server handlers
async def handle_search(request):
    params = request.rel_url.query
    query = params.get('q')
    email = params.get('email')
    if not query:
        return web.json_response({'error': 'Missing query parameter q'}, status=400)
    # Check subscription for premium features (optional)
    if email and not check_subscription(email):
        return web.json_response({'error': 'Subscription required'}, status=403)
    products = await aggregate_products(query)
    return web.json_response({'products': products})

async def handle_subscribe(request):
    data = await request.json()
    email = data.get('email')
    if not email:
        return web.json_response({'error': 'Missing email'}, status=400)
    create_subscription(email)
    return web.json_response({'message': 'Subscription activated'})

# Main app setup
app = web.Application()
app.router.add_get('/search', handle_search)
app.router.add_post('/subscribe', handle_subscribe)
app.router.add_post('/create-payment-intent', create_payment_intent)

if __name__ == '__main__':
    web.run_app(app, port=8080)

    except Exception as e:
        logger.error(f"Error fetching from API {api_info['name']}: {e}")
    return None

# Async function to scrape product data if API fails
async def scrape_products(session, api_info, query):
    try:
        url = api_info['scrape_url'].format(query=urllib.parse.quote(query))
        async with async_timeout.timeout(10):
            async with session.get(url) as response:
                if response.status == 200:
                    html = await response.text()
                    soup = BeautifulSoup(html, 'html.parser')
                    products = []
                    for product_div in soup.select('.product-item'):
                        name_tag = product_div.select_one('.product-title')
                        price_tag = product_div.select_one('.product-price')
                        link_tag = product_div.select_one('a')
                        if name_tag and price_tag and link_tag:
                            try:
                                price_text = price_tag.text.strip().replace('$', '').replace(',', '')
                                price = float(price_text)
                                product_url = link_tag['href']
                                encoded_url = urllib.parse.quote(product_url, safe='')
                                products.append({
                                    'name': name_tag.text.strip(),
                                    'price': price,
                                    'url': AFFILIATE_PREFIX + encoded_url,
                                    'source': api_info['name']
                                })
                            except Exception:
                                continue
                    return products
                else:
                    logger.warning(f"Scraping {api_info['name']} returned status {response.status}")
    except Exception as e:
        logger.error(f"Error scraping {api_info['name']}: {e}")
    return None

# Aggregate product data from all sources concurrently
async def aggregate_products(query):
    async with aiohttp.ClientSession() as session:
        # Fetch APIs concurrently
        api_tasks = [fetch_from_api(session, api_info, query) for api_info in ECOMMERCE_APIS]
        api_results = await asyncio.gather(*api_tasks)

        # For APIs that failed, fallback to scraping concurrently
        scrape_tasks = []
        for i, res in enumerate(api_results):
            if res is None:
                scrape_tasks.append(scrape_products(session, ECOMMERCE_APIS[i], query))
            else:
                scrape_tasks.append(asyncio.sleep(0, result=res))  # dummy task returning existing result

        scrape_results = await asyncio.gather(*scrape_tasks)

        # Flatten and sort by price
        all_products = [item for sublist in scrape_results if sublist for item in sublist]
        all_products.sort(key=lambda x: x['price'])
        return all_products

# Subscription management with async DB access
async def create_subscription(email):
    async with aiosqlite.connect(DATABASE) as db:
        await db.execute(
            "INSERT OR IGNORE INTO users (email, subscription_active) VALUES (?, ?)",
            (email, True)
        )
        await db.commit()

async def check_subscription(email):
    async with aiosqlite.connect(DATABASE) as db:
        async with db.execute("SELECT subscription_active FROM users WHERE email=?", (email,)) as cursor:
            row = await cursor.fetchone()
            return bool(row and row[0])

# Stripe payment intent creation with server-side validation
async def create_payment_intent(request):
    try:
        data = await request.json()
        email = data.get('email')
        amount = data.get('amount')  # in cents

        if not email or not amount:
            return web.json_response({'error': 'Missing email or amount'}, status=400)

        # Validate amount server-side (example: only allow predefined subscription amounts)
        allowed_amounts = {999, 1999, 2999}  # e.g., $9.99, $19.99, $29.99
        if amount not in allowed_amounts:
            return web.json_response({'error': 'Invalid amount'}, status=400)

        intent = stripe.PaymentIntent.create(
            amount=amount,
            currency='usd',
            receipt_email=email,
            metadata={'integration_check': 'accept_a_payment'}
        )
        return web.json_response({'client_secret': intent.client_secret})
    except Exception as e:
        logger.error(f"Stripe payment intent error: {e}")
        return web.json_response({'error': str(e)}, status=500)

# Web server handlers
async def handle_search(request):
    params = request.rel_url.query
    query = params.get('q')
    email = params.get('email')
    if not query:
        return web.json_response({'error': 'Missing query parameter q'}, status=400)
    if email:
        subscribed = await check_subscription(email)
        if not subscribed:
            return web.json_response({'error': 'Subscription required'}, status=403)
    products = await aggregate_products(query)
    return web.json_response({'products': products})

async def handle_subscribe(request):
    data = await request.json()
    email = data.get('email')
    if not email:
        return web.json_response({'error': 'Missing email'}, status=400)
    await create_subscription(email)
    return web.json_response({'message': 'Subscription activated'})

app.router.add_post('/create-payment-intent', create_payment_intent)
app.on_startup.append(on_startup)

    web.run_app(app, port=8080)
