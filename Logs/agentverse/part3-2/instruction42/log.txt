import json
import logging
import time

import requests
import queue
import threading

logging.basicConfig(
    level=logging.INFO,
    format='%(asctime)s %(levelname)s %(threadName)s %(message)s',
    handlers=[
        logging.StreamHandler()

class QuestDBClient:
    def __init__(self, host: str = 'http://localhost:9000'):
        self.host = host
        self.session = requests.Session()

    def insert(self, table: str, data: Dict[str, Any]) -> bool:
        """
        Insert data into QuestDB using line protocol.
        """
        try:
            line = self._to_line_protocol(table, data)
            url = f"{self.host}/imp"
            headers = {'Content-Type': 'text/plain'}
            response = self.session.post(url, data=line, headers=headers, timeout=5)
            response.raise_for_status()
            logging.debug(f"Inserted data into QuestDB: {line}")
            return True
        except requests.RequestException as e:
            logging.error(f"Failed to insert data into QuestDB: {e}")
            return False

    def _to_line_protocol(self, table: str, data: Dict[str, Any]) -> str:
        """
        Convert dictionary to QuestDB line protocol.
        Assumes data keys are columns and values are fields.
        """
        # Escape measurement and tag keys/values if needed
        # For simplicity, assume no tags, only fields
        fields = []
        for k, v in data.items():
            if isinstance(v, str):
                v = v.replace('"', '\\"')
                fields.append(f'{k}="{v}"')
            elif isinstance(v, bool):
                fields.append(f'{k}={str(v).lower()}')
            elif isinstance(v, (int, float)):
                fields.append(f'{k}={v}')
            else:
                # Skip unsupported types
                continue
        line = f"{table} {','.join(fields)} {int(time.time() * 1e9)}"
        return line

class ETLPipeline:
    def __init__(self, questdb_client: QuestDBClient, max_retries: int = 3):
        self.questdb_client = questdb_client
        self.queue = queue.Queue(maxsize=10000)
        self.max_retries = max_retries
        self.stop_event = threading.Event()

    def extract(self, raw_message: str) -> Dict[str, Any]:
        """
        Extract JSON data from raw message string.
        """
        try:
            data = json.loads(raw_message)
            logging.debug(f"Extracted data: {data}")
            return data
        except json.JSONDecodeError as e:
            logging.error(f"JSON decode error: {e} - Raw message: {raw_message}")
            return {}

    def transform(self, data: Dict[str, Any]) -> Dict[str, Any]:
        """
        Transform data to fit QuestDB schema.
        For Claude Code conversations, assume fields like conversation_id, user_id, message, timestamp.
        """
        transformed = {}
        try:
            transformed['conversation_id'] = str(data.get('conversation_id', 'unknown'))
            transformed['user_id'] = str(data.get('user_id', 'unknown'))
            transformed['message'] = data.get('message', '')
            # Convert timestamp to int nanoseconds if present, else current time
            ts = data.get('timestamp')
            if ts:
                # Assume timestamp is in ISO format or epoch seconds
                if isinstance(ts, str):
                    # Try to parse ISO format
                    from dateutil import parser
                    dt = parser.isoparse(ts)
                    transformed['timestamp'] = int(dt.timestamp() * 1e9)
                elif isinstance(ts, (int, float)):
                    transformed['timestamp'] = int(float(ts) * 1e9)
                else:
                    transformed['timestamp'] = int(time.time() * 1e9)
            else:
                transformed['timestamp'] = int(time.time() * 1e9)
            logging.debug(f"Transformed data: {transformed}")
        except Exception as e:
            logging.error(f"Error in transform: {e} - Data: {data}")
        return transformed

    def load(self, data: Dict[str, Any]) -> bool:
        """
        Load data into QuestDB with retry logic.
        """
        for attempt in range(1, self.max_retries + 1):
            success = self.questdb_client.insert('claude_conversations', data)
            if success:
                logging.info(f"Data loaded successfully on attempt {attempt}")
                return True
            else:
                logging.warning(f"Retrying load attempt {attempt} for data: {data}")
                time.sleep(2 ** attempt)  # Exponential backoff
        logging.error(f"Failed to load data after {self.max_retries} attempts: {data}")
        return False

    def worker(self):
        """
        Worker thread to process queue items.
        """
        while not self.stop_event.is_set():
            try:
                raw_message = self.queue.get(timeout=1)
                extracted = self.extract(raw_message)
                if not extracted:
                    self.queue.task_done()
                    continue
                transformed = self.transform(extracted)
                if transformed:
                    self.load(transformed)
                self.queue.task_done()
            except queue.Empty:
                continue
            except Exception as e:
                logging.error(f"Unexpected error in worker: {e}")

    def start(self, num_workers: int = 4):
        """
        Start worker threads.
        """
        self.threads = []
        for i in range(num_workers):
            t = threading.Thread(target=self.worker, name=f"Worker-{i+1}", daemon=True)
            t.start()
            self.threads.append(t)
        logging.info(f"Started {num_workers} worker threads.")

    def stop(self):
        """
        Stop worker threads gracefully.
        """
        self.stop_event.set()
        for t in self.threads:
            t.join()
        logging.info("All worker threads stopped.")

    def enqueue(self, raw_message: str):
        """
        Add raw message to processing queue.
        """
        try:
            self.queue.put(raw_message, block=False)
            logging.debug("Enqueued raw message.")
        except queue.Full:
            logging.error("Queue is full. Dropping message.")

        For Claude Code conversations, assume fields like conversation_id, user_id, message, timestamp.
        """
        transformed = {}
        try:
            transformed['conversation_id'] = str(data.get('conversation_id', 'unknown'))
            transformed['user_id'] = str(data.get('user_id', 'unknown'))
            transformed['message'] = data.get('message', '')
            ts = data.get('timestamp')
            if ts:
                if isinstance(ts, str):
                    from dateutil import parser
                    dt = parser.isoparse(ts)
                    transformed['timestamp'] = int(dt.timestamp() * 1e9)
                elif isinstance(ts, (int, float)):
                    transformed['timestamp'] = int(float(ts) * 1e9)
                else:
                    transformed['timestamp'] = int(time.time() * 1e9)
            else:
                transformed['timestamp'] = int(time.time() * 1e9)
            logging.debug(f"Transformed data: {transformed}")
        except Exception as e:
            logging.error(f"Error in transform: {e} - Data: {data}")
        return transformed

    def load(self, data: Dict[str, Any]) -> bool:
        """
        Load data into QuestDB with retry logic.
        """
        for attempt in range(1, self.max_retries + 1):
            success = self.questdb_client.insert('claude_conversations', data)
            if success:
                logging.info(f"Data loaded successfully on attempt {attempt}")
                return True
            else:
                logging.warning(f"Retrying load attempt {attempt} for data: {data}")
                time.sleep(2 ** attempt)  # Exponential backoff
        logging.error(f"Failed to load data after {self.max_retries} attempts: {data}")
        return False

    def worker(self):
        """
        Worker thread to process queue items.
        """
        while not self.stop_event.is_set():
            try:
                raw_message = self.queue.get(timeout=1)
                extracted = self.extract(raw_message)
                if not extracted:
                    self.queue.task_done()
                    continue
                transformed = self.transform(extracted)
                if transformed:
                    self.load(transformed)
                self.queue.task_done()
            except queue.Empty:
                continue
            except Exception as e:
                logging.error(f"Unexpected error in worker: {e}")

    def start(self, num_workers: int = 4):
        """
        Start worker threads.
        """
        self.threads = []
        for i in range(num_workers):
            t = threading.Thread(target=self.worker, name=f"Worker-{i+1}", daemon=True)
            t.start()
            self.threads.append(t)
        logging.info(f"Started {num_workers} worker threads.")

    def stop(self):
        """
        Stop worker threads gracefully.
        """
        self.stop_event.set()
        for t in self.threads:
            t.join()
        logging.info("All worker threads stopped.")

    def enqueue(self, raw_message: str):
        """
        Add raw message to processing queue.
        """
        try:
            self.queue.put(raw_message, block=False)
            logging.debug("Enqueued raw message.")
        except queue.Full:
            logging.error("Queue is full. Dropping message.")
