# Please help me implement a remote access software application in Python that supports Windows, MacOS, iOS, Android, and Chromebook,

import socket
import threading
import ssl
import tkinter as tk
from tkinter import simpledialog, messagebox
import hashlib
import os

# Constants for server
SERVER_HOST = '0.0.0.0'
SERVER_PORT = 5000
BUFFER_SIZE = 4096

# Simple user database (username: hashed_password)
USER_DB = {
    "user1": hashlib.sha256("password1".encode()).hexdigest(),
    "user2": hashlib.sha256("password2".encode()).hexdigest(),
}

# SSL context for encryption
def create_ssl_context():
    context = ssl.create_default_context(ssl.Purpose.CLIENT_AUTH)
    # For demo purposes, generate self-signed cert or load existing cert/key files
    # Here we create a context without verification for simplicity
    context.check_hostname = False
    context.verify_mode = ssl.CERT_NONE
    return context

# Authentication function
def authenticate(conn):
    try:
        conn.send(b"Username: ")
        username = conn.recv(BUFFER_SIZE).decode().strip()
        conn.send(b"Password: ")
        password = conn.recv(BUFFER_SIZE).decode().strip()
        hashed_password = hashlib.sha256(password.encode()).hexdigest()
        if USER_DB.get(username) == hashed_password:
            conn.send(b"AUTH_SUCCESS")
            return True
        else:
            conn.send(b"AUTH_FAIL")
            return False
    except Exception:
        return False

# Handle client connection
def handle_client(conn, addr):
    try:
        if not authenticate(conn):
            conn.close()
            return
        conn.send(b"Welcome to Remote Access Server. Type commands or 'exit' to quit.\n")
        while True:
            conn.send(b"> ")
            data = conn.recv(BUFFER_SIZE).decode().strip()
            if not data or data.lower() == 'exit':
                break
            # For security, only allow limited commands or implement sandboxing
            # Here we just echo back the command for demo
            response = f"Executed: {data}\n"
            conn.send(response.encode())
    except Exception as e:
        pass
    finally:
        conn.close()

# Server main function
def start_server():
    context = create_ssl_context()
    server = socket.socket(socket.AF_INET, socket.SOCK_STREAM)
    server.bind((SERVER_HOST, SERVER_PORT))
    server.listen(5)
    print(f"Server listening on {SERVER_HOST}:{SERVER_PORT}")
    while True:
        client_sock, client_addr = server.accept()
        ssl_conn = context.wrap_socket(client_sock, server_side=True)
        threading.Thread(target=handle_client, args=(ssl_conn, client_addr), daemon=True).start()

class RemoteAccessClient(tk.Tk):
    def __init__(self):
        super().__init__()
        self.title("Remote Access Client")
        self.geometry("600x400")

        self.text_area = tk.Text(self, state='disabled')
        self.text_area.pack(expand=True, fill='both')

        self.entry = tk.Entry(self)
        self.entry.pack(fill='x')
        self.entry.bind("<Return>", self.send_command)

        self.sock = None
        self.ssl_sock = None

        self.connect_to_server()

    def connect_to_server(self):
        host = simpledialog.askstring("Connect", "Enter server IP:", parent=self)
        port = simpledialog.askinteger("Connect", "Enter server port:", parent=self, initialvalue=SERVER_PORT)
        username = simpledialog.askstring("Login", "Username:", parent=self)
        password = simpledialog.askstring("Login", "Password:", parent=self, show='*')

        try:
            self.sock = socket.socket(socket.AF_INET, socket.SOCK_STREAM)
            context = ssl.create_default_context()
            context.check_hostname = False
            context.verify_mode = ssl.CERT_NONE
            self.ssl_sock = context.wrap_socket(self.sock, server_hostname=host)
            self.ssl_sock.connect((host, port))

            # Receive Username prompt
            self.ssl_sock.recv(BUFFER_SIZE)
            self.ssl_sock.send((username + "\n").encode())
            # Receive Password prompt
            self.ssl_sock.recv(BUFFER_SIZE)
            self.ssl_sock.send((password + "\n").encode())
            auth_response = self.ssl_sock.recv(BUFFER_SIZE).decode()
            if auth_response != "AUTH_SUCCESS":
                messagebox.showerror("Authentication Failed", "Invalid username or password")
                self.destroy()
                return

            welcome = self.ssl_sock.recv(BUFFER_SIZE).decode()
            self.append_text(welcome)
            threading.Thread(target=self.receive_data, daemon=True).start()
        except Exception as e:
            messagebox.showerror("Connection Error", str(e))
            self.destroy()

    def append_text(self, text):
        self.text_area.config(state='normal')
        self.text_area.insert('end', text)
        self.text_area.see('end')
        self.text_area.config(state='disabled')

    def send_command(self, event=None):
        cmd = self.entry.get()
        if not cmd:
            return
        try:
            self.ssl_sock.send((cmd + "\n").encode())
            self.entry.delete(0, 'end')
        except Exception:
            self.append_text("\nConnection lost.\n")
            self.entry.config(state='disabled')

    def receive_data(self):
        try:
            while True:
                data = self.ssl_sock.recv(BUFFER_SIZE)
                if not data:
                    break
                self.append_text(data.decode())
        except Exception:
            pass
        finally:
            self.append_text("\nDisconnected from server.\n")
            self.entry.config(state='disabled')

def execute_command(command):
    # Only allow commands in whitelist
    if command in ALLOWED_COMMANDS:
        try:
            result = subprocess.run(ALLOWED_COMMANDS[command], capture_output=True, text=True, timeout=10, shell=False)
            return result.stdout + result.stderr
        except Exception as e:
            return f"Error executing command: {e}\n"
    else:
        return "Command not allowed.\n"

    try:
