@app.route('/webhook', methods=['POST'])
def webhook():
    # This endpoint receives WhatsApp webhook events (calls, messages)
    data = request.json

    # Example structure for missed call event (depends on WhatsApp API setup)
    # We assume a missed call event contains phone number and call status
    # For demo, we check if a call event is present and if it was missed
    # Adjust according to your WhatsApp webhook payload structure

    # Check for missed call event
    try:
        for entry in data.get('entry', []):
            for change in entry.get('changes', []):
                value = change.get('value', {})
                # Check if there is a call event
                calls = value.get('calls', [])
                for call in calls:
                    if call.get('status') == 'missed':
                        phone_number = call.get('from')
                        call_time_str = call.get('timestamp')
                        call_time = datetime.utcfromtimestamp(int(call_time_str)) if call_time_str else datetime.utcnow()

                        # Log the missed call
                        call_log = CallLog(phone_number=phone_number, call_time=call_time)
                        db.session.add(call_log)
                        db.session.commit()

                        # Send automated reply
                        reply_message = "Sorry we missed your call. How can we help you?"
                        if send_whatsapp_message(phone_number, reply_message):
                            call_log.message_sent = True
                            call_log.message_content = reply_message
                            call_log.message_time = datetime.utcnow()
                            db.session.commit()
        return jsonify({"status": "success"}), 200
    except Exception as e:
        return jsonify({"status": "error", "message": str(e)}), 400

@app.route('/schedule_message', methods=['POST'])
def schedule_message():
    data = request.json
    phone_number = data.get('phone_number')
    message_content = data.get('message')
    scheduled_time_str = data.get('scheduled_time')  # ISO format expected

    if not phone_number or not message_content or not scheduled_time_str:
        return jsonify({"error": "Missing required fields"}), 400

    try:
        scheduled_time = datetime.fromisoformat(scheduled_time_str)
    except ValueError:
        return jsonify({"error": "Invalid datetime format"}), 400

    scheduled_msg = ScheduledMessage(
        phone_number=phone_number,
        message_content=message_content,
        scheduled_time=scheduled_time
    )
    db.session.add(scheduled_msg)
    db.session.commit()
    return jsonify({"status": "scheduled"}), 200

@scheduler.task('interval', id='send_scheduled_messages', seconds=30, misfire_grace_time=10)
def send_scheduled_messages():
    now = datetime.utcnow()
    messages = ScheduledMessage.query.filter(
        ScheduledMessage.scheduled_time <= now,
        ScheduledMessage.sent == False
    ).all()
    for msg in messages:
        if send_whatsapp_message(msg.phone_number, msg.message_content):
            msg.sent = True
            db.session.commit()

@app.route('/')
def dashboard():
    # Show recent call logs and scheduled messages
    call_logs = CallLog.query.order_by(CallLog.call_time.desc()).limit(50).all()
    scheduled_msgs = ScheduledMessage.query.order_by(ScheduledMessage.scheduled_time.asc()).limit(50).all()
    return render_template('dashboard.html', call_logs=call_logs, scheduled_msgs=scheduled_msgs)

@app.route('/export_csv')
def export_csv():
    call_logs = CallLog.query.order_by(CallLog.call_time.desc()).all()

    output = io.StringIO()
    writer = csv.writer(output)
    writer.writerow(['ID', 'Phone Number', 'Call Time', 'Message Sent', 'Message Content', 'Message Time'])
    for log in call_logs:
        writer.writerow([
            log.id,
            log.phone_number,
            log.call_time.isoformat(),
            log.message_sent,
            log.message_content or '',
            log.message_time.isoformat() if log.message_time else ''
        ])

    output.seek(0)
    return send_file(
        io.BytesIO(output.getvalue().encode()),
        mimetype='text/csv',
        as_attachment=True,
        download_name='call_logs.csv'
    )
