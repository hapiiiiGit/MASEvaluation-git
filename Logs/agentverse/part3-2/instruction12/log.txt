import librosa
import numpy as np
from sklearn.svm import SVC
from sklearn.preprocessing import StandardScaler
from sklearn.pipeline import make_pipeline
import joblib
import os

# Pre-trained model path (you can train your own model and save it here)
MODEL_PATH = "gender_classifier_svm.joblib"

def extract_features(file_path, sr=22050, n_mfcc=13):
    y, sr = librosa.load(file_path, sr=sr)
    # Extract MFCC features
    mfccs = librosa.feature.mfcc(y=y, sr=sr, n_mfcc=n_mfcc)
    # Aggregate MFCCs by mean and std deviation
    mfccs_mean = np.mean(mfccs, axis=1)
    mfccs_std = np.std(mfccs, axis=1)
    features = np.hstack((mfccs_mean, mfccs_std))
    return features

def train_and_save_model(training_data_dir):
    """
    Train a gender classifier on a dataset organized as:
    training_data_dir/
        male/
        female/
    Each folder contains audio files of that gender.
    """
    X = []
    y = []
    for gender in ['male', 'female']:
        folder = os.path.join(training_data_dir, gender)
        for file in os.listdir(folder):
            if file.lower().endswith(('.wav', '.mp3')):
                file_path = os.path.join(folder, file)
                features = extract_features(file_path)
                X.append(features)
                y.append(gender)
    X = np.array(X)
    y = np.array(y)
    # Create a pipeline with scaler and SVM classifier
    clf = make_pipeline(StandardScaler(), SVC(probability=True, kernel='rbf', C=1, gamma='scale'))
    clf.fit(X, y)
    joblib.dump(clf, MODEL_PATH)
    print(f"Model trained and saved to {MODEL_PATH}")

def predict_gender(file_path):
    """
    Predict gender ('Male' or 'Female') from an audio file.
    Returns a tuple: (gender_str, confidence_score)
    """
    if not os.path.exists(MODEL_PATH):
        raise FileNotFoundError(f"Model file not found at {MODEL_PATH}. Train the model first using train_and_save_model().")
    clf = joblib.load(MODEL_PATH)
    features = extract_features(file_path).reshape(1, -1)
    probs = clf.predict_proba(features)[0]
    classes = clf.classes_
    max_idx = np.argmax(probs)
    gender = classes[max_idx].capitalize()
    confidence = probs[max_idx]
    return gender, confidence
