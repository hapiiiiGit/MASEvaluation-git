# Dataset class for COCO-like image-caption pairs
class ImageCaptionDataset(Dataset):
    def __init__(self, image_paths, captions, tokenizer, transform=None, max_length=50):
        self.image_paths = image_paths
        self.captions = captions
        self.tokenizer = tokenizer
        self.transform = transform
        self.max_length = max_length

    def __len__(self):
        return len(self.captions)

    def __getitem__(self, idx):
        # Load and preprocess image using OpenCV
        img_path = self.image_paths[idx]
        image = cv2.imread(img_path)
        image = cv2.cvtColor(image, cv2.COLOR_BGR2RGB)
        if self.transform:
            image = self.transform(image)
        else:
            # Default transform: resize and normalize
            image = cv2.resize(image, (224, 224))
            image = image.astype(np.float32) / 255.0
            image = torch.tensor(image).permute(2,0,1)  # HWC to CHW
            image = transforms.Normalize(mean=[0.485, 0.456, 0.406],
                                         std=[0.229, 0.224, 0.225])(image)
        # Tokenize caption
        caption = self.captions[idx]
        encoding = self.tokenizer(caption, max_length=self.max_length, padding='max_length', truncation=True, return_tensors="pt")
        input_ids = encoding.input_ids.squeeze(0)  # remove batch dim
        attention_mask = encoding.attention_mask.squeeze(0)
        return image, input_ids, attention_mask

# Model combining CNN image encoder and GPT-2 caption decoder
class ImageCaptioningModel(nn.Module):
    def __init__(self, cnn_model_name='resnet18', transformer_model_name='gpt2', embed_dim=512):
        super().__init__()
        # Load pretrained CNN (ResNet) and remove final classification layer
        cnn = models.resnet18(pretrained=True)
        modules = list(cnn.children())[:-1]  # remove last fc layer
        self.cnn_encoder = nn.Sequential(*modules)
        self.cnn_output_dim = cnn.fc.in_features  # 512 for resnet18

        # Linear layer to project CNN features to GPT-2 embedding size
        self.feature_proj = nn.Linear(self.cnn_output_dim, embed_dim)

        # Load pretrained GPT-2 model and tokenizer
        self.transformer = GPT2LMHeadModel.from_pretrained(transformer_model_name)
        self.transformer_embedding_size = self.transformer.config.n_embd

        # Freeze GPT-2 embeddings to keep stable
        # (optional, can be fine-tuned)
        # for param in self.transformer.transformer.wte.parameters():
        #     param.requires_grad = False

    def forward(self, images, input_ids, attention_mask):
        # Extract image features
        features = self.cnn_encoder(images)  # (batch, 512, 1, 1)
        features = features.view(features.size(0), -1)  # (batch, 512)
        features = self.feature_proj(features)  # (batch, embed_dim)

        # Prepare inputs for GPT-2
        # Prepend image features as a special token embedding to input embeddings
        inputs_embeds = self.transformer.transformer.wte(input_ids)  # (batch, seq_len, embed_dim)
        # Expand features to seq dimension 1
        features = features.unsqueeze(1)  # (batch, 1, embed_dim)
        # Concatenate image features at the start of the sequence
        inputs_embeds = torch.cat([features, inputs_embeds[:,1:,:]], dim=1)  # keep input length same

        # Adjust attention mask to include image feature token
        extended_attention_mask = torch.cat([torch.ones((attention_mask.size(0),1), dtype=attention_mask.dtype, device=attention_mask.device), attention_mask[:,1:]], dim=1)

        outputs = self.transformer(inputs_embeds=inputs_embeds, attention_mask=extended_attention_mask, labels=input_ids)
        return outputs

    def __init__(self, image_paths, captions, tokenizer, transform=None, max_length=50):
        self.image_paths = image_paths
        self.captions = captions
        self.tokenizer = tokenizer
        self.transform = transform
        self.max_length = max_length

    def __len__(self):
        return len(self.captions)

    def __getitem__(self, idx):
        # Load and preprocess image using OpenCV
        img_path = self.image_paths[idx]
        image = cv2.imread(img_path)
        image = cv2.cvtColor(image, cv2.COLOR_BGR2RGB)
        if self.transform:
            image = self.transform(image)
        else:
            # Default transform: resize and normalize
            image = cv2.resize(image, (224, 224))
            image = image.astype(np.float32) / 255.0
            image = torch.tensor(image).permute(2,0,1)  # HWC to CHW
            image = transforms.Normalize(mean=[0.485, 0.456, 0.406],
                                         std=[0.229, 0.224, 0.225])(image)
        # Tokenize caption
        caption = self.captions[idx]
        encoding = self.tokenizer(
            caption,
            max_length=self.max_length,
            padding='max_length',
            truncation=True,
            return_tensors="pt"
        )
        input_ids = encoding.input_ids.squeeze(0)  # remove batch dim
        attention_mask = encoding.attention_mask.squeeze(0)
        return image, input_ids, attention_mask

# Model combining CNN image encoder and GPT-2 caption decoder with learned image token
class ImageCaptioningModel(nn.Module):
    def __init__(self, cnn_model_name='resnet18', transformer_model_name='gpt2'):
        super().__init__()
        # Load pretrained CNN (ResNet) and remove final classification layer
        cnn = models.resnet18(pretrained=True)
        modules = list(cnn.children())[:-1]  # remove last fc layer
        self.cnn_encoder = nn.Sequential(*modules)
        self.cnn_output_dim = cnn.fc.in_features  # 512 for resnet18

        # Load pretrained GPT-2 model
        self.transformer = GPT2LMHeadModel.from_pretrained(transformer_model_name)
        self.transformer_embedding_size = self.transformer.config.n_embd  # typically 768 for gpt2

        # Linear layer to project CNN features to GPT-2 embedding size
        self.feature_proj = nn.Linear(self.cnn_output_dim, self.transformer_embedding_size)

        # Learned embedding token to represent image features prepended to text tokens
        self.img_token_embedding = nn.Parameter(torch.randn(1, 1, self.transformer_embedding_size))

    def forward(self, images, input_ids, attention_mask):
        batch_size = images.size(0)

        # Extract image features
        features = self.cnn_encoder(images)  # (batch, 512, 1, 1)
        features = features.view(batch_size, -1)  # (batch, 512)
        features = self.feature_proj(features)  # (batch, embed_dim)
        features = features.unsqueeze(1)  # (batch, 1, embed_dim)

        # Prepare text embeddings
        inputs_embeds = self.transformer.transformer.wte(input_ids)  # (batch, seq_len, embed_dim)

        # Expand learned image token embedding and add projected features
        img_token = self.img_token_embedding.expand(batch_size, -1, -1) + features  # (batch,1,embed_dim)

        # Concatenate image token embedding at the start of the sequence
        inputs_embeds = torch.cat([img_token, inputs_embeds], dim=1)  # (batch, seq_len+1, embed_dim)

        # Adjust attention mask to account for prepended image token
        extended_attention_mask = torch.cat(
            [torch.ones((batch_size,1), dtype=attention_mask.dtype, device=attention_mask.device), attention_mask],
            dim=1
        )  # (batch, seq_len+1)

        # Prepare labels: prepend -100 to ignore loss on image token
        labels = input_ids.clone()
        labels = torch.cat([torch.full((batch_size,1), -100, dtype=labels.dtype, device=labels.device), labels], dim=1)

        outputs = self.transformer(
            inputs_embeds=inputs_embeds,
            attention_mask=extended_attention_mask,
            labels=labels,
        )
        return outputs

# Example usage with dummy data for reproducibility
def main():
    device = torch.device("cuda" if torch.cuda.is_available() else "cpu")

    # Dummy data: 2 images and captions
    dummy_image_paths = ["dummy1.jpg", "dummy2.jpg"]
    dummy_captions = ["a cat sitting on a mat", "a dog playing with a ball"]

    # Create dummy images for testing (in-memory, no disk I/O)
    dummy_images = []
    for _ in dummy_image_paths:
        dummy_img = np.random.randint(0, 255, (256,256,3), dtype=np.uint8)
        dummy_images.append(dummy_img)

    tokenizer = GPT2Tokenizer.from_pretrained('gpt2')
    tokenizer.pad_token = tokenizer.eos_token  # GPT2 has no pad token by default

    transform = transforms.Compose([
        transforms.ToPILImage(),
        transforms.Resize((224,224)),
        transforms.ToTensor(),
        transforms.Normalize(mean=[0.485,0.456,0.406], std=[0.229,0.224,0.225])
    ])

    # Create dataset with in-memory images
    class InMemoryImageCaptionDataset(Dataset):
        def __init__(self, images, captions, tokenizer, transform=None, max_length=50):
            self.images = images
            self.captions = captions
            self.tokenizer = tokenizer
            self.transform = transform
            self.max_length = max_length

        def __len__(self):
            return len(self.captions)

        def __getitem__(self, idx):
            image = self.images[idx]
            image = cv2.cvtColor(image, cv2.COLOR_BGR2RGB)
            if self.transform:
                image = self.transform(image)
            else:
                image = cv2.resize(image, (224, 224))
                image = image.astype(np.float32) / 255.0
                image = torch.tensor(image).permute(2,0,1)
                image = transforms.Normalize(mean=[0.485, 0.456, 0.406],
                                             std=[0.229, 0.224, 0.225])(image)
            caption = self.captions[idx]
            encoding = self.tokenizer(
                caption,
                max_length=self.max_length,
                padding='max_length',
                truncation=True,
                return_tensors="pt"
            )
            input_ids = encoding.input_ids.squeeze(0)
            attention_mask = encoding.attention_mask.squeeze(0)
            return image, input_ids, attention_mask

    dataset = InMemoryImageCaptionDataset(dummy_images, dummy_captions, tokenizer, transform=transform)
    dataloader = DataLoader(dataset, batch_size=2, shuffle=True)

    model = ImageCaptioningModel().to(device)
    optimizer = optim.Adam(model.parameters(), lr=5e-5)

    model.train()
    for epoch in range(2):  # small number of epochs for prototype
        for images, input_ids, attention_mask in dataloader:
            images = images.to(device)
            input_ids = input_ids.to(device)
            attention_mask = attention_mask.to(device)

            outputs = model(images, input_ids, attention_mask)
            loss = outputs.loss

            optimizer.zero_grad()
            loss.backward()
            optimizer.step()

            print(f"Epoch {epoch+1}, Loss: {loss.item():.4f}")

if __name__ == "__main__":
    main()
