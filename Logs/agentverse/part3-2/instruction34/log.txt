def moving_average(series, window):

def relative_strength_index(series, window=14):
    delta = series.diff()
    gain = (delta.where(delta > 0, 0)).rolling(window=window).mean()
    loss = (-delta.where(delta < 0, 0)).rolling(window=window).mean()
    rs = gain / loss
    rsi = 100 - (100 / (1 + rs))
    return rsi

# Logic to apply on fetched data
def process_stock_data(raw_data):
    # Extract time series data from Alpha Vantage response
    try:
        time_series = raw_data['Time Series (1min)']
    except KeyError:
        return None  # or raise exception
    
    df = pd.DataFrame.from_dict(time_series, orient='index')
    df.index = pd.to_datetime(df.index)
    df = df.sort_index()
    df = df.astype(float)
    df.rename(columns={
        '1. open': 'open',
        '2. high': 'high',
        '3. low': 'low',
        '4. close': 'close',
        '5. volume': 'volume'
    }, inplace=True)
    
    df['MA_20'] = moving_average(df['close'], 20)
    df['RSI_14'] = relative_strength_index(df['close'], 14)
    
    # Example logic: signal when close crosses above MA_20 and RSI < 30 (oversold)
    df['signal'] = np.where(
        (df['close'] > df['MA_20']) & (df['close'].shift(1) <= df['MA_20'].shift(1)) & (df['RSI_14'] < 30),
        'buy',
        np.where(
            (df['close'] < df['MA_20']) & (df['close'].shift(1) >= df['MA_20'].shift(1)) & (df['RSI_14'] > 70),
            'sell',
            'hold'
        )
    )
    return df
