# Recruitment process implementation for a full-time software developer position

class Candidate:
    def __init__(self, name, experience, languages, problem_solving_score, teamwork_score, development_approach):
        self.name = name
        self.experience = experience  # years of relevant experience
        self.languages = languages  # list of programming languages known
        self.problem_solving_score = problem_solving_score  # score from problem-solving test (0-10)
        self.teamwork_score = teamwork_score  # score from teamwork assessment (0-10)
        self.development_approach = development_approach  # candidate's description of their development approach

    def is_qualified(self):
        required_languages = {'Python', 'Java', 'C++'}
        knows_all_languages = required_languages.issubset(set(self.languages))
        has_experience = self.experience >= 2  # minimum 2 years experience
        strong_problem_solving = self.problem_solving_score >= 7
        good_teamwork = self.teamwork_score >= 7
        return knows_all_languages and has_experience and strong_problem_solving and good_teamwork

    def submit_application(self):
        if self.is_qualified():
            print(f"Candidate {self.name} is qualified for the position.")
            print("Development Approach:")
            print(self.development_approach)
        else:
            print(f"Candidate {self.name} does not meet the qualification criteria.")

# Example usage:
