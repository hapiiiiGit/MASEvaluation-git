def require_auth(f):
    from functools import wraps
    @wraps(f)
    def decorated(*args, **kwargs):
        auth_header = request.headers.get('Authorization', '')
        if not auth_header.startswith('Bearer '):
            return jsonify({'error': 'Authorization header missing or malformed'}), 401
        token = auth_header.split(' ')[1]
        payload = verify_cognito_token(token)
        if not payload:
            return jsonify({'error': 'Invalid or expired token'}), 401
        request.user = payload
        return f(*args, **kwargs)
    return decorated

def verify_cognito_token(token):
    try:
        jwks = get_jwks()
        headers = jwt.get_unverified_header(token)
        kid = headers['kid']
        key = next((k for k in jwks['keys'] if k['kid'] == kid), None)
        if key is None:
            logger.warning("Public key not found in JWKS")
            return None
        public_key = jwt.algorithms.RSAAlgorithm.from_jwk(key)
        payload = jwt.decode(
            token,
            public_key,
            algorithms=['RS256'],
            audience=COGNITO_APP_CLIENT_ID,
            issuer=f"https://cognito-idp.{COGNITO_REGION}.amazonaws.com/{COGNITO_USERPOOL_ID}"
        )
        return payload
    except Exception as e:
        logger.warning(f"Token verification failed: {e}")
        return None

    """

    dynamodb_table = os.getenv('TERRAFORM_DYNAMODB_TABLE')
    aws_region = os.getenv('AWS_REGION', 'us-east-1')

    if not os.path.isdir(tf_dir):

        'bucket': backend_bucket,

        'dynamodb_table': dynamodb_table
    }

    init_cmd = ['terraform', 'init']
    for k, v in backend_config.items():
        init_cmd.extend(['-backend-config', f"{k}={v}"])

    logger.info(f"Running Terraform init for environment '{env}' with backend config: {backend_config}")
    subprocess.run(init_cmd, cwd=tf_dir, check=True)

def terraform_apply(env: str, var_file: str = None):

    Apply Terraform configuration for the given environment.

    """
