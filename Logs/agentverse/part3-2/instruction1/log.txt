                content = obj_data['Body'].read()
                logs.append(json.loads(content))
    export_data = json.dumps(logs, indent=2)
    return app.response_class(
        export_data,
        mimetype='application/json',
        headers={'Content-Disposition': f'attachment;filename=diver_logs_{current_user.id}.json'}
    )

@app.route('/sign_log', methods=['POST'])
@login_required
def sign_log():
    data = request.json
    log_data = data.get('log_data')
    if not log_data:
        return jsonify({'error': 'log_data is required'}), 400
    log_bytes = json.dumps(log_data, sort_keys=True).encode('utf-8')
    signature = sign_data(log_bytes)
    signature_b64 = base64.b64encode(signature).decode('utf-8')
    return jsonify({'signature': signature_b64}), 200
