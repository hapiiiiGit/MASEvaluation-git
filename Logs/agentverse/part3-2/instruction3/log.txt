    """
    Set environment variable persistently for user or system.

    try:

        # Notify system about environment variable change
        HWND_BROADCAST = 0xFFFF
        WM_SETTINGCHANGE = 0x1A

def get_reg_env_path(user=False):
    """
    Get the current PATH environment variable from registry for user or system.
    """
    if user:
        root = winreg.HKEY_CURRENT_USER
        path = r"Environment"
    else:
        root = winreg.HKEY_LOCAL_MACHINE
        path = r"SYSTEM\CurrentControlSet\Control\Session Manager\Environment"
    try:
        with winreg.OpenKey(root, path, 0, winreg.KEY_READ) as key:
            value, _ = winreg.QueryValueEx(key, "Path")
            return value
    except FileNotFoundError:
        return ""
