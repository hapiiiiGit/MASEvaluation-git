}

# GraphQL query to fetch token prices (simplified example)
GRAPHQL_QUERY = """
{
  tokens(first: 1000) {
    id
    symbol
    derivedETH
  }
  bundle(id: "1") {
    ethPrice
  }
}
"""

async def fetch_token_prices(session, dex_name, url):
    try:
        async with session.post(url, json={'query': GRAPHQL_QUERY}) as resp:
            if resp.status != 200:
                return {}
            data = await resp.json()
            eth_price = float(data['data']['bundle']['ethPrice'])
            tokens = data['data']['tokens']
            prices = {}
            for token in tokens:
                # price in USD = derivedETH * ethPrice
                price = float(token['derivedETH']) * eth_price
                prices[token['id']] = {
                    'symbol': token['symbol'],
                    'price': price,
                    'contract': token['id'],
                    'dex': dex_name
                }
            return prices
    except Exception:
        return {}

async def fetch_all_prices():
    async with aiohttp.ClientSession() as session:
        tasks = [fetch_token_prices(session, dex, url) for dex, url in DEX_APIS.items()]
        results = await asyncio.gather(*tasks)
        combined = defaultdict(list)
        for dex_prices in results:
            for contract, info in dex_prices.items():
                combined[contract].append(info)
        return combined

def analyze_tokens(combined):
    # Filter tokens available on at least two exchanges
    filtered = {k: v for k, v in combined.items() if len(v) >= 2}
    output = []
    for contract, infos in filtered.items():
        prices = [(info['dex'], info['price']) for info in infos]
        prices.sort(key=lambda x: x[1])
        lowest = prices[0]
        highest = prices[-1]
        diff_percent = (highest[1] - lowest[1]) / lowest[1] * 100
        if diff_percent > 1.8:
            output.append({

# GraphQL queries tailored per DEX to fetch tokens and base token price
GRAPHQL_QUERIES = {
    "Uniswap": """
    {
      tokens(first: 1000, orderBy: tradeVolumeUSD, orderDirection: desc) {
        id
        symbol
        derivedETH
      }
      bundle(id: "1") {
        ethPrice
      }
    }
    """,
    "Sushiswap": """
    {
      tokens(first: 1000, orderBy: tradeVolumeUSD, orderDirection: desc) {

        bnbPrice
      }

    """
}

async def fetch_token_prices(session, dex_name, dex_info):
    url = dex_info["url"]
    query = GRAPHQL_QUERIES[dex_name]
    base_token = dex_info["base_token"]
    try:
        async with session.post(url, json={'query': query}) as resp:
            if resp.status != 200:
                return {}
            data = await resp.json()
            if 'data' not in data:
                return {}
            tokens = data['data'].get('tokens', [])
            bundle = data['data'].get('bundle', {})
            if base_token == "ETH":
                base_price = float(bundle.get('ethPrice', 0))
            elif base_token == "BNB":
                base_price = float(bundle.get('bnbPrice', 0))
            else:
                base_price = 0
            if base_price == 0:
                return {}
            prices = {}
            for token in tokens:
                token_id = token.get('id', '').lower()
                symbol = token.get('symbol', '')
                if base_token == "ETH":
                    derived = token.get('derivedETH')
                elif base_token == "BNB":
                    derived = token.get('derivedBNB')
                else:
                    derived = None
                if derived is None or float(derived) == 0:
                    continue
                price = float(derived) * base_price
                prices[token_id] = {
                    'symbol': symbol,
                    'price': price,
                    'contract': token_id,
                    'dex': dex_name
                }
            return prices
    except Exception:
        return {}

async def fetch_all_prices():
    async with aiohttp.ClientSession() as session:
        tasks = [fetch_token_prices(session, dex, info) for dex, info in DEX_APIS.items()]
        results = await asyncio.gather(*tasks)
        combined = defaultdict(list)
        for dex_prices in results:
            for contract, info in dex_prices.items():
                combined[contract].append(info)
        return combined

def analyze_tokens(combined):
    filtered = {k: v for k, v in combined.items() if len(v) >= 2}
    output = []
    for contract, infos in filtered.items():
        prices = [(info['dex'], info['price']) for info in infos]
        prices.sort(key=lambda x: x[1])
        lowest = prices[0]
        highest = prices[-1]
        if lowest[1] == 0:
            continue
        diff_percent = (highest[1] - lowest[1]) / lowest[1] * 100
        if diff_percent > 1.8:
            output.append({
                'symbol': infos[0]['symbol'],
                'contract': contract,
                'lowest_dex': lowest[0],
                'lowest_price': lowest[1],
                'highest_dex': highest[0],
                'highest_price': highest[1],
                'diff_percent': diff_percent
            })
    return output
