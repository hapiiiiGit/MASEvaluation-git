import os

class Customer(db.Model):
    id = db.Column(db.Integer, primary_key=True)
    whatsapp_number = db.Column(db.String(20), unique=True, nullable=False)

class Booking(db.Model):
    id = db.Column(db.Integer, primary_key=True)
    customer_id = db.Column(db.Integer, db.ForeignKey('customer.id'), nullable=False)
    service = db.Column(db.String(100), nullable=False)

    status = db.Column(db.String(20), default='pending')

    data = {
        "messaging_product": "whatsapp",

        "text": {
            "body": message
        }
    }
    response = requests.post(url, headers=headers, json=data)
    return response.json()

def chatgpt_response(prompt):
    response = openai.ChatCompletion.create(
        model="gpt-4",
        messages=[{"role": "user", "content": prompt}],
        max_tokens=500,
        temperature=0.7,
    )
    return response.choices[0].message['content'].strip()

@app.route('/webhook', methods=['GET', 'POST'])
def webhook():
    if request.method == 'GET':
        # Verification challenge from Meta
        verify_token = os.getenv('META_VERIFY_TOKEN', 'verifytoken')
        mode = request.args.get('hub.mode')
        token = request.args.get('hub.verify_token')
        challenge = request.args.get('hub.challenge')
        if mode and token:
            if mode == 'subscribe' and token == verify_token:
                return challenge, 200
            else:
                return 'Verification token mismatch', 403
        return 'Hello World', 200

    if request.method == 'POST':
        data = request.get_json()
        # Parse incoming message
        try:
            entry = data['entry'][0]
            changes = entry['changes'][0]
            value = changes['value']
            messages = value.get('messages')
            if messages:
                message = messages[0]
                from_number = message['from']
                text = message.get('text', {}).get('body', '').strip()
                # Process message
                response_text = process_message(from_number, text)
                send_whatsapp_message(from_number, response_text)
        except Exception as e:
            print(f"Error processing webhook: {e}")
        return 'EVENT_RECEIVED', 200

    if last_booking and last_booking.status == 'pending' and last_booking.service and not last_booking.booking_date:
        # Try parse date from text

            booking_date = datetime.strptime(text, '%Y-%m-%d %H:%M')
            last_booking.booking_date = booking_date
            last_booking.status = 'confirmed'
            db.session.commit()
            notify_admin(f"New booking confirmed: {customer.name} - {last_booking.service} on {booking_date}")
            return f"Thank you {customer.name}, your booking for {last_booking.service} on {booking_date} is confirmed."
        except ValueError:
            return "Please enter the date and time in format YYYY-MM-DD HH:MM (24h)."

    # If user just sent service name, create booking and ask for date
    if text.lower() not in ["book service", "help"]:
        # Create new booking with service name
        booking = Booking(customer_id=customer.id, service=text, status='pending')
        db.session.add(booking)
        db.session.commit()
        return "Please enter the date and time for your booking in format YYYY-MM-DD HH:MM (24h)."

    # Help or fallback to ChatGPT
    if text.lower() == "help":
        return ("You can register by sending your name.\n"
                "Book a service by typing 'Book Service'.\n"
                "For other queries, just type your message.")

    # Use ChatGPT for intelligent conversation fallback
    prompt = f"Customer ({customer.name}): {text}\nChatbot:"
    answer = chatgpt_response(prompt)
    return answer

def notify_admin(message):
    # For simplicity, send notification to admin WhatsApp number
    admin_number = os.getenv('ADMIN_WHATSAPP_NUMBER')
    if admin_number:

def send_whatsapp_message(to, message):
    url = META_API_URL.format(phone_number_id=META_PHONE_NUMBER_ID)
    headers = {
        "Authorization": f"Bearer {META_ACCESS_TOKEN}",
        "Content-Type": "application/json"

        "to": to,

    }

def chatgpt_response(prompt):
    response = openai.ChatCompletion.create(
        model="gpt-4",
        messages=[{"role": "user", "content": prompt}],
        max_tokens=500,
        temperature=0.7,
    )
    return response.choices[0].message['content'].strip()

@app.route('/webhook', methods=['GET', 'POST'])
def webhook():
    if request.method == 'GET':
        # Verification challenge from Meta
        verify_token = os.getenv('META_VERIFY_TOKEN', 'verifytoken')
        mode = request.args.get('hub.mode')
        token = request.args.get('hub.verify_token')
        challenge = request.args.get('hub.challenge')
        if mode and token:
            if mode == 'subscribe' and token == verify_token:
                return challenge, 200
            else:
                return 'Verification token mismatch', 403
        return 'Hello World', 200

    if request.method == 'POST':
        data = request.get_json()
        try:
            entry = data['entry'][0]
            changes = entry['changes'][0]
            value = changes['value']
            messages = value.get('messages')
            if messages:
                message = messages[0]
                from_number = message['from']
                text = message.get('text', {}).get('body', '').strip()
                response_text = process_message(from_number, text)
                send_whatsapp_message(from_number, response_text)
        except Exception as e:
            print(f"Error processing webhook: {e}")
        return 'EVENT_RECEIVED', 200

def process_message(from_number, text):
    customer = Customer.query.filter_by(whatsapp_number=from_number).first()
    if not customer:
        customer = Customer(whatsapp_number=from_number, name="Customer")
        db.session.add(customer)
        db.session.commit()
        return ("Welcome! You have been registered.\n"
                "Please tell me your name to update your profile.")

    if customer.name == "Customer" and text.lower() != "book service":
        customer.name = text
        db.session.commit()
        return f"Thanks {customer.name}! You can now book a service by typing 'Book Service'."

    if text.lower() == "book service":
        # Start a new booking with no service or date yet
        booking = Booking(customer_id=customer.id, status='pending')
        db.session.add(booking)
        db.session.commit()
        return "Please enter the service you want to book (e.g., Haircut, Massage):"

    # Find the latest pending booking for this customer
    pending_booking = Booking.query.filter_by(customer_id=customer.id, status='pending').order_by(Booking.created_at.desc()).first()

    if pending_booking:
        if not pending_booking.service:
            # Treat text as service name
            pending_booking.service = text
            db.session.commit()
            return "Please enter the date and time for your booking in format YYYY-MM-DD HH:MM (24h)."
        elif not pending_booking.booking_date:
            # Try to parse date
            try:
                booking_date = datetime.strptime(text, '%Y-%m-%d %H:%M')
                pending_booking.booking_date = booking_date
                pending_booking.status = 'confirmed'
                db.session.commit()
                notify_admin(f"New booking confirmed: {customer.name} - {pending_booking.service} on {booking_date.strftime('%Y-%m-%d %H:%M')}")
                return f"Thank you {customer.name}, your booking for {pending_booking.service} on {booking_date.strftime('%Y-%m-%d %H:%M')} is confirmed."

                return "Please enter the date and time in format YYYY-MM-DD HH:MM (24h)."

    if text.lower() == "help":
