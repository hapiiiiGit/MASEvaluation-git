# Analyze user-provided .wav files to generate Synthesizer data text files
def analyze_wav_and_generate_synth_data(wav_path, output_txt_path):
    # Read wav file
    with wave.open(wav_path, 'rb') as wav_file:
        n_channels = wav_file.getnchannels()
        sample_width = wav_file.getsampwidth()
        framerate = wav_file.getframerate()
        n_frames = wav_file.getnframes()
        audio_data = wav_file.readframes(n_frames)

    # Convert audio data to numpy array
    dtype_map = {1: np.int8, 2: np.int16, 4: np.int32}
    dtype = dtype_map.get(sample_width, np.int16)
    audio_np = np.frombuffer(audio_data, dtype=dtype)
    if n_channels > 1:
        audio_np = audio_np.reshape(-1, n_channels)
        audio_np = audio_np.mean(axis=1)  # Convert to mono by averaging channels

    # Normalize audio
    audio_np = audio_np / np.max(np.abs(audio_np))

    # Extract features for synthesizer data
    # Example: Extract fundamental frequency using autocorrelation
    def estimate_f0(signal, fs, fmin=50, fmax=2000):
        corr = scipy.signal.correlate(signal, signal, mode='full')
        corr = corr[len(corr)//2:]
        d = np.diff(corr)
        start = np.where(d > 0)[0][0]
        peak = np.argmax(corr[start:]) + start
        f0 = fs / peak if peak != 0 else 0
        if f0 < fmin or f0 > fmax:
            f0 = 0
        return f0

    f0 = estimate_f0(audio_np, framerate)

    # Extract RMS amplitude
    rms = np.sqrt(np.mean(audio_np**2))

    # Extract spectral centroid
    spectrum = np.abs(np.fft.rfft(audio_np))
    freqs = np.fft.rfftfreq(len(audio_np), 1/framerate)
    spectral_centroid = np.sum(freqs * spectrum) / np.sum(spectrum)

    # Prepare synthesizer data dictionary
    synth_data = {
        "fundamental_frequency": f0,
        "rms_amplitude": rms,
        "spectral_centroid": spectral_centroid,
        "sample_rate": framerate,
        "duration_seconds": n_frames / framerate
    }

    # Write synthesizer data to text file (JSON format)
    with open(output_txt_path, 'w') as f:
        json.dump(synth_data, f, indent=4)

    print(f"Synthesizer data saved to {output_txt_path}")

        audio_np = audio_np.astype(float)

    # Extract features for synthesizer data
    # Example: Extract fundamental frequency using autocorrelation
    def estimate_f0(signal, fs, fmin=50, fmax=2000):
        corr = scipy.signal.correlate(signal, signal, mode='full')
        corr = corr[len(corr)//2:]
        d = np.diff(corr)
        try:
            start = np.where(d > 0)[0][0]
        except IndexError:
            return 0
        peak = np.argmax(corr[start:]) + start
        if peak == 0:
            return 0
        f0 = fs / peak
        if f0 < fmin or f0 > fmax:
            return 0
        return f0

    f0 = estimate_f0(audio_np, framerate)

    # Extract RMS amplitude
    rms = np.sqrt(np.mean(audio_np**2))

    # Extract spectral centroid
    spectrum = np.abs(np.fft.rfft(audio_np))
    freqs = np.fft.rfftfreq(len(audio_np), 1/framerate)
    if np.sum(spectrum) > 0:
        spectral_centroid = np.sum(freqs * spectrum) / np.sum(spectrum)
    else:
        spectral_centroid = 0

    # Prepare synthesizer data dictionary
    synth_data = {
        "fundamental_frequency": f0,
        "rms_amplitude": rms,
        "spectral_centroid": spectral_centroid,
        "sample_rate": framerate,
        "duration_seconds": n_frames / framerate
    }

    # Write synthesizer data to text file (JSON format)
    with open(output_txt_path, 'w') as f:
        json.dump(synth_data, f, indent=4)

    print(f"Synthesizer data saved to {output_txt_path}")
