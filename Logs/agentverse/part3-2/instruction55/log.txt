from tkinter import messagebox

import time
import pandas as pd

class IBApi(EWrapper, EClient):
    def __init__(self):
        EClient.__init__(self, self)
        self.data = []  # store historical data
        self.reqId = 1
        self.nextOrderId = None
        self.position = 0

    def nextValidId(self, orderId: int):
        self.nextOrderId = orderId

    def historicalData(self, reqId, bar):
        self.data.append([bar.date, bar.open, bar.high, bar.low, bar.close, bar.volume])

    def historicalDataEnd(self, reqId, start, end):
        print("Historical data received")

    def orderStatus(self, orderId, status, filled, remaining, avgFillPrice,
                    permId, parentId, lastFillPrice, clientId, whyHeld, mktCapPrice):
        print(f"OrderStatus. Id: {orderId}, Status: {status}, Filled: {filled}, Remaining: {remaining}, AvgFillPrice: {avgFillPrice}")

    def openOrder(self, orderId, contract, order, orderState):
        print(f"OpenOrder. ID: {orderId}, {contract.symbol}, {order.action}, {order.orderType}, {order.totalQuantity}, {orderState.status}")

    def execDetails(self, reqId, contract, execution):
        print(f"ExecDetails. {contract.symbol}, {execution.side}, {execution.shares}, {execution.price}")

class TradingBot:
    def __init__(self, symbol, quantity, momentum_window=5):
        self.app = IBApi()

        self.app.connect("127.0.0.1", 7497, clientId=0)
        thread = Thread(target=self.app.run, daemon=True)

        contract.currency = "USD"
        return contract

        self.app.data = []
        contract = self.get_contract()
        # Request 1 day of 1-minute bars for momentum calculation

        # Wait for data to be received
        time.sleep(5)

        df['date'] = pd.to_datetime(df['date'])
        self.data = df

        # Momentum = close price difference over momentum_window bars
        self.data['momentum'] = self.data['close'].diff(self.momentum_window)

import tkinter as tk
from tkinter import messagebox, scrolledtext
from ibapi.client import EClient
from ibapi.wrapper import EWrapper
from ibapi.contract import Contract
from ibapi.order import Order
from threading import Thread, Event
import time
import pandas as pd

class IBApi(EWrapper, EClient):
    def __init__(self):
        EClient.__init__(self, self)
        self.data = []  # store historical data
        self.reqId = 1
        self.nextOrderId = None
        self.position = 0
        self.historical_data_end_event = Event()

    def nextValidId(self, orderId: int):
        self.nextOrderId = orderId

    def historicalData(self, reqId, bar):
        self.data.append([bar.date, bar.open, bar.high, bar.low, bar.close, bar.volume])

    def historicalDataEnd(self, reqId, start, end):
        self.historical_data_end_event.set()

    def orderStatus(self, orderId, status, filled, remaining, avgFillPrice,
                    permId, parentId, lastFillPrice, clientId, whyHeld, mktCapPrice):
        print(f"OrderStatus. Id: {orderId}, Status: {status}, Filled: {filled}, Remaining: {remaining}, AvgFillPrice: {avgFillPrice}")

    def openOrder(self, orderId, contract, order, orderState):
        print(f"OpenOrder. ID: {orderId}, {contract.symbol}, {order.action}, {order.orderType}, {order.totalQuantity}, {orderState.status}")

    def execDetails(self, reqId, contract, execution):
        print(f"ExecDetails. {contract.symbol}, {execution.side}, {execution.shares}, {execution.price}")

class TradingBot:
    def __init__(self, symbol, quantity, momentum_window=5):
        self.app = IBApi()
        self.symbol = symbol
        self.quantity = quantity
        self.momentum_window = momentum_window
        self.data = None
        self.position = 0
        self.order_id = None

    def connect(self):
        self.app.connect("127.0.0.1", 7497, clientId=0)
        thread = Thread(target=self.app.run, daemon=True)
        thread.start()
        while self.app.nextOrderId is None:
            time.sleep(0.1)
        self.order_id = self.app.nextOrderId

    def disconnect(self):
        if self.app.isConnected():
            self.app.disconnect()

    def get_contract(self):
        contract = Contract()
        contract.symbol = self.symbol
        contract.secType = "STK"
        contract.exchange = "SMART"
        contract.currency = "USD"
        return contract

    def request_historical_data(self):
        self.app.data = []
        self.app.historical_data_end_event.clear()
        contract = self.get_contract()
        self.app.reqHistoricalData(self.app.reqId, contract, '', '1 D', '1 min', 'TRADES', 1, 1, False, [])
        if not self.app.historical_data_end_event.wait(timeout=10):
            raise TimeoutError("Timeout waiting for historical data")
        df = pd.DataFrame(self.app.data, columns=['date', 'open', 'high', 'low', 'close', 'volume'])
        df['date'] = pd.to_datetime(df['date'])
        self.data = df

    def calculate_momentum(self):
        if self.data is None or len(self.data) < self.momentum_window + 1:
            return 0
        self.data['momentum'] = self.data['close'].diff(self.momentum_window)
        momentum_value = self.data['momentum'].iloc[-1]
        if pd.isna(momentum_value):
            return 0
        return momentum_value

    def place_order(self, action):
        contract = self.get_contract()
        order = Order()
        order.action = action
        order.orderType = "MKT"
        order.totalQuantity = self.quantity
        self.app.placeOrder(self.order_id, contract, order)
        self.order_id += 1
        self.position = self.quantity if action == "BUY" else 0

    def run_strategy(self):
        self.connect()
        try:
            self.request_historical_data()
            momentum = self.calculate_momentum()
            print(f"Calculated momentum: {momentum}")
            if momentum > 0 and self.position == 0:
                print("Momentum positive, placing BUY order")
                self.place_order("BUY")
            elif momentum < 0 and self.position > 0:
                print("Momentum negative, placing SELL order")
                self.place_order("SELL")
            else:
                print("No trade signal")
        finally:
            self.disconnect()
