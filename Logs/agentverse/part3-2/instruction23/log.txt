def generate_test_input(client_id: str, prompt: str) -> pd.DataFrame:
    """
    Use GPT-4 API to generate test input data in JSON format, then convert to DataFrame.
    """
    response = openai.ChatCompletion.create(
        model="gpt-4",
        messages=[
            {"role": "system", "content": "You generate JSON data for automated test input Excel files."},
            {"role": "user", "content": prompt}
        ],
        temperature=0.7,
        max_tokens=1000
    )
    content = response['choices'][0]['message']['content']
    try:
        data = json.loads(content)

        return df
    except Exception as e:

def analyze_output(client_id: str, output_df: pd.DataFrame) -> Dict[str, Any]:
    """
    Analyze output DataFrame for errors and categorize them.
    """
    errors = []
    category = "No Errors"
    # Example error detection: any row with 'error' column not empty
    if 'error' in output_df.columns:
        error_rows = output_df[output_df['error'].notnull() & (output_df['error'] != "")]
        if not error_rows.empty:
            errors = error_rows['error'].tolist()
            category = "Errors Found"
    return {"errors": errors, "category": category}

def improve_prompt(client_id: str, previous_prompt: str, logs: List[Dict[str, Any]]) -> str:
    """
    Use GPT-4 to improve future input generation prompts based on past logs.
    """
    system_msg = "You are an assistant improving prompts for generating test input Excel files."
    user_msg = f"Previous prompt:\n{previous_prompt}\n\nLogs:\n{json.dumps(logs, indent=2)}\n\nSuggest an improved prompt."
    response = openai.ChatCompletion.create(
        model="gpt-4",
        messages=[
            {"role": "system", "content": system_msg},
            {"role": "user", "content": user_msg}
        ],
        temperature=0.5,
        max_tokens=500
    )
    improved_prompt = response['choices'][0]['message']['content']
    return improved_prompt

def save_excel(df: pd.DataFrame, filename: str):
    df.to_excel(filename, index=False)

    return pd.read_excel(filename)

class ClientSession:
    def __init__(self, client_id: str):
        self.client_id = client_id
        self.prompt = "Generate a JSON array of test input data with fields: id, name, value."
        self.logs = []

    def generate_and_save_input(self, filename: str):
        df = generate_test_input(self.client_id, self.prompt)
        save_excel(df, filename)
        return df

    def analyze_and_log_output(self, output_filename: str):
        output_df = load_excel(output_filename)
        analysis = analyze_output(self.client_id, output_df)
        errors_str = json.dumps(analysis["errors"])
        category = analysis["category"]
        # Log input and output data as JSON strings for simplicity
        log_to_db(self.client_id, self.prompt, output_df.to_json(orient='records'), errors_str, category)
        self.logs.append({
            "input": self.prompt,
            "output": output_df.to_dict(orient='records'),
            "errors": analysis["errors"],
            "category": category
        })
        return analysis

    def improve_prompt_based_on_logs(self):
        if not self.logs:
            return
        self.prompt = improve_prompt(self.client_id, self.prompt, self.logs)

def main():
    import argparse

    parser = argparse.ArgumentParser(description="Standalone .exe for GPT-4 test input generation and analysis")
    parser.add_argument("--client", required=True, help="Client ID")
    parser.add_argument("--generate", help="Filename to save generated input Excel")
    parser.add_argument("--analyze", help="Filename of output Excel to analyze")
    parser.add_argument("--improve", action="store_true", help="Improve prompt based on logs")
    args = parser.parse_args()

    init_db()
    session = ClientSession(args.client)

    if args.generate:
        df = session.generate_and_save_input(args.generate)
        print(f"Generated input Excel saved to {args.generate}")

    if args.analyze:
        analysis = session.analyze_and_log_output(args.analyze)
        print(f"Analysis: {analysis}")

    if args.improve:
        session.improve_prompt_based_on_logs()
        print(f"Improved prompt:\n{session.prompt}")

if __name__ == "__main__":
    main()
