import wx

class Shape:
    def __init__(self, pos, size, color):
        self.pos = pos  # (x, y)
        self.size = size  # (width, height)
        self.color = color
        self.selected = False

    def draw(self, dc):
        raise NotImplementedError

    def hit_test(self, point):
        x, y = self.pos
        w, h = self.size
        px, py = point
        return x <= px <= x + w and y <= py <= y + h

    def move(self, dx, dy):
        x, y = self.pos
        self.pos = (x + dx, y + dy)

class Rectangle(Shape):
    def draw(self, dc):
        dc.SetBrush(wx.Brush(self.color))
        dc.SetPen(wx.Pen(wx.BLACK, 2 if self.selected else 1))
        x, y = self.pos
        w, h = self.size
        dc.DrawRectangle(x, y, w, h)

class Ellipse(Shape):
    def draw(self, dc):
        dc.SetBrush(wx.Brush(self.color))
        dc.SetPen(wx.Pen(wx.BLACK, 2 if self.selected else 1))
        x, y = self.pos
        w, h = self.size
        dc.DrawEllipse(x, y, w, h)

class DrawingPanel(wx.Panel):
    def __init__(self, parent):
        super().__init__(parent)
        self.shapes = []
        self.current_shape_type = 'Rectangle'  # or 'Ellipse'
        self.current_color = wx.Colour(0, 0, 255)
        self.selected_shape = None
        self.dragging = False
        self.drag_start_pos = None
        self.shape_start_pos = None

        self.SetBackgroundStyle(wx.BG_STYLE_PAINT)
        self.Bind(wx.EVT_PAINT, self.on_paint)
        self.Bind(wx.EVT_LEFT_DOWN, self.on_left_down)
        self.Bind(wx.EVT_LEFT_UP, self.on_left_up)
        self.Bind(wx.EVT_MOTION, self.on_mouse_move)

    def on_paint(self, event):
        dc = wx.AutoBufferedPaintDC(self)
        dc.Clear()
        for shape in self.shapes:
            shape.draw(dc)

    def on_left_down(self, event):
        pos = event.GetPosition()
        # Check if clicking on a shape to select it
        for shape in reversed(self.shapes):
            if shape.hit_test(pos):
                if self.selected_shape:
                    self.selected_shape.selected = False
                self.selected_shape = shape
                shape.selected = True
                self.dragging = True
                self.drag_start_pos = pos
                self.shape_start_pos = shape.pos
                self.Refresh()
                return
        # If no shape selected, create a new shape
        if self.selected_shape:
            self.selected_shape.selected = False
            self.selected_shape = None
            self.Refresh()
        # Create new shape at pos with default size
        size = (80, 60)
        if self.current_shape_type == 'Rectangle':
            shape = Rectangle(pos, size, self.current_color)
        else:
            shape = Ellipse(pos, size, self.current_color)
        self.shapes.append(shape)
        if self.selected_shape:
            self.selected_shape.selected = False
        self.selected_shape = shape
        shape.selected = True
        self.dragging = True
        self.drag_start_pos = pos
        self.shape_start_pos = shape.pos
        self.Refresh()

    def on_left_up(self, event):
        self.dragging = False

    def on_mouse_move(self, event):
        if self.dragging and self.selected_shape and event.Dragging() and event.LeftIsDown():
            pos = event.GetPosition()
            dx = pos.x - self.drag_start_pos.x
            dy = pos.y - self.drag_start_pos.y
            new_x = self.shape_start_pos[0] + dx
            new_y = self.shape_start_pos[1] + dy
            self.selected_shape.pos = (new_x, new_y)
            self.Refresh()

class MainFrame(wx.Frame):
    def __init__(self):
        super().__init__(None, title="Shape Drawer", size=(600, 400))
        self.panel = DrawingPanel(self)

        # Toolbar for shape selection and color

        color_tool = toolbar.AddTool(wx.ID_ANY, "Color", wx.ArtProvider.GetBitmap(wx.ART_TIP, wx.ART_TOOLBAR, (16,16)))
        toolbar.Realize()

        self.Bind(wx.EVT_TOOL, self.on_select_rectangle, rect_tool)
        self.Bind(wx.EVT_TOOL, self.on_select_ellipse, ellipse_tool)
        self.Bind(wx.EVT_TOOL, self.on_select_color, color_tool)

        sizer = wx.BoxSizer(wx.VERTICAL)
        sizer.Add(toolbar, 0, wx.EXPAND)
        sizer.Add(self.panel, 1, wx.EXPAND)
        self.SetSizer(sizer)

    def on_select_rectangle(self, event):
        self.panel.current_shape_type = 'Rectangle'

    def on_select_ellipse(self, event):
        self.panel.current_shape_type = 'Ellipse'

    def on_select_color(self, event):
        dlg = wx.ColourDialog(self)

        dlg.Destroy()
