            "ivp_return_auth": return_url,

from frappe.utils.background_jobs import enqueue

from frappe import _
import requests

import hmac
import hashlib

class TelrPaymentGateway:
    def __init__(self, merchant_id, secret_key, test_mode=True):
        self.merchant_id = merchant_id
        self.secret_key = secret_key  # Secret key used for signing and webhook verification
        self.test_mode = test_mode
        self.api_url = "https://secure.telr.com/gateway/order.json"

    def create_payment(self, amount, currency, return_urls, customer_info, cart_id):
        # Telr expects a JSON payload with "order" key containing payment details
        # return_urls is a dict with keys: auth, can, decl
        payload = {
            "order": {
                "cartid": cart_id,
                "currency": currency,
                "amount": str(amount),
                "description": "Subscription Payment",
                "test": "1" if self.test_mode else "0",
                "return": {
                    "auth": return_urls.get("auth"),
                    "can": return_urls.get("can"),
                    "decl": return_urls.get("decl")
                },
                "billing": {
                    "fname": customer_info.get("first_name", ""),
                    "lname": customer_info.get("last_name", ""),
                    "email": customer_info.get("email", ""),
                }
            }
        }
        headers = {"Content-Type": "application/json"}
        try:
            response = requests.post(self.api_url, json=payload, headers=headers, timeout=30)
            response.raise_for_status()
            resp_json = response.json()
            order = resp_json.get("order", {})
            return {
                "order_ref": order.get("ref"),
                "payment_url": order.get("url"),
                "raw_response": resp_json
            }
        except requests.RequestException as e:
            frappe.log_error(f"Telr payment creation failed: {e}")
            frappe.throw(_("Payment gateway error, please try again later."))

    def verify_webhook_signature(self, payload_body, signature):
        # Telr webhook signature is base64-encoded HMAC SHA256 of payload_body using secret_key
        computed_hmac = hmac.new(self.secret_key.encode('utf-8'), payload_body, hashlib.sha256).digest()
        computed_signature = base64.b64encode(computed_hmac).decode()
        return hmac.compare_digest(computed_signature, signature)

    # Enqueue background job to provision the site asynchronously

    return site_name

        self.auth_key = auth_key
        self.secret_key = secret_key  # Secret key used for signing and webhook verification
        self.test_mode = test_mode
        self.api_url = "https://secure.telr.com/gateway/order.json"

    def create_payment(self, amount, currency, return_urls, customer_info, cart_id):

            "store": self.store_id,

            "order": {

                "description": "Subscription Payment",
                "test": "1" if self.test_mode else "0",
                "return": {
                    "auth": return_urls.get("auth"),
                    "can": return_urls.get("can"),

                },
                "billing": {

        }

        try:

            response.raise_for_status()
            resp_json = response.json()
            if resp_json.get("error"):
                frappe.log_error(f"Telr payment creation error: {resp_json['error']}")
                frappe.throw(_("Payment gateway error, please try again later."))
            order = resp_json.get("order", {})
            return {
                "order_ref": order.get("ref"),
                "payment_url": order.get("url"),
                "raw_response": resp_json
            }
        except requests.RequestException as e:
            frappe.log_error(f"Telr payment creation failed: {e}")
            frappe.throw(_("Payment gateway error, please try again later."))

    if payment_status == "A":  # Approved
        if payment_order_doc.status != "Paid":
            payment_order_doc.status = "Paid"
            payment_order_doc.save(ignore_permissions=True)
            enqueue("saas_app.api.provision_site", subscription_plan_name=payment_order_doc.subscription_plan, customer_email=payment_order_doc.customer_email, queue='long')
