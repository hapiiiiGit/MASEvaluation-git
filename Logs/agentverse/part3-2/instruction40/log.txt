class PythonLessonRoadmap:
    def __init__(self, learner_name):
        self.learner_name = learner_name
        self.lessons = []
        self.feedback_sessions = []
        self.follow_up_support = True

    def add_lesson(self, title, description, interactive_session=True, exercises=True):
        lesson = {
            'title': title,
            'description': description,
            'interactive_session': interactive_session,
            'exercises': exercises,
            'feedback': None
        }
        self.lessons.append(lesson)

    def provide_feedback(self, lesson_index, feedback):
        if 0 <= lesson_index < len(self.lessons):
            self.lessons[lesson_index]['feedback'] = feedback
            self.feedback_sessions.append({
                'lesson': self.lessons[lesson_index]['title'],
                'feedback': feedback
            })

    def get_roadmap(self):
        roadmap = f"Personalized Python Roadmap for {self.learner_name}:\n"
        for i, lesson in enumerate(self.lessons, 1):
            roadmap += f"\nLesson {i}: {lesson['title']}\n"
            roadmap += f"Description: {lesson['description']}\n"
            roadmap += f"Includes live interactive session: {'Yes' if lesson['interactive_session'] else 'No'}\n"
            roadmap += f"Includes practical exercises: {'Yes' if lesson['exercises'] else 'No'}\n"
            if lesson['feedback']:
                roadmap += f"Feedback: {lesson['feedback']}\n"
        if self.follow_up_support:
            roadmap += "\nSupport for follow-up questions is available.\n"
        return roadmap
