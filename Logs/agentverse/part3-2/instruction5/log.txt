from PIL import Image, ImageDraw, ImageFont

def create_logo():
    # Create a square logo 500x500 px
    logo_size = (500, 500)
    logo = Image.new('RGBA', logo_size, (255, 255, 255, 0))
    draw = ImageDraw.Draw(logo)

    # Background color
    bg_color = (30, 144, 255)  # Dodger Blue
    draw.rectangle([0, 0, logo_size[0], logo_size[1]], fill=bg_color)

    # Draw Python icon-like shape (simplified)
    # Draw two overlapping circles to represent Python's dual snakes
    circle_radius = 100
    circle_color1 = (255, 255, 255)
    circle_color2 = (255, 215, 0)  # Gold

    # Left circle
    left_circle_center = (logo_size[0]//3, logo_size[1]//2)
    draw.ellipse([left_circle_center[0]-circle_radius, left_circle_center[1]-circle_radius,
                  left_circle_center[0]+circle_radius, left_circle_center[1]+circle_radius], fill=circle_color1)

    # Right circle
    right_circle_center = (2*logo_size[0]//3, logo_size[1]//2)
    draw.ellipse([right_circle_center[0]-circle_radius, right_circle_center[1]-circle_radius,
                  right_circle_center[0]+circle_radius, right_circle_center[1]+circle_radius], fill=circle_color2)

    # Add text "PyScrape" in the center
    try:
        font = ImageFont.truetype("arial.ttf", 60)
    except IOError:
        font = ImageFont.load_default()

    text = "PyScrape"
    text_width, text_height = draw.textsize(text, font=font)
    text_position = ((logo_size[0] - text_width) // 2, logo_size[1] - text_height - 40)
    draw.text(text_position, text, font=font, fill=(255, 255, 255))

    logo.save("logo.png")
