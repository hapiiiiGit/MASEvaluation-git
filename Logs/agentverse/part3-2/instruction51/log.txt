import io
from PyPDF2 import PdfReader
import docx2txt

def extract_text_from_pdf(file):
    reader = PdfReader(file)
    text = []
    for page in reader.pages:
        text.append(page.extract_text())
    return "\n".join(text)

def extract_text_from_docx(file):
    # docx2txt requires a path or file-like object
    text = docx2txt.process(file)
    return text

def extract_text_from_txt(file):
    return file.read().decode('utf-8')

def extract_text(file):
    filename = file.name.lower()
    if filename.endswith('.pdf'):
        return extract_text_from_pdf(file)
    elif filename.endswith('.docx') or filename.endswith('.doc'):
        return extract_text_from_docx(file)
    elif filename.endswith('.txt'):
        return extract_text_from_txt(file)
    else:
        return ""

@login_required
def upload_documents(request):
    if request.method == 'POST':
        form = DocumentUploadForm(request.POST, request.FILES)
        files = request.FILES.getlist('files')
        if form.is_valid():
            for f in files:
                text = extract_text(f)
                summary = summarize_text(text)
                doc = Document.objects.create(user=request.user, file=f, summary=summary)
            messages.success(request, f'{len(files)} document(s) uploaded and summarized.')
            return redirect('upload_documents')
    else:
        form = DocumentUploadForm()
    documents = Document.objects.filter(user=request.user).order_by('-uploaded_at')
    return render(request, 'upload_documents.html', {'form': form, 'documents': documents})

@login_required
def formula_builder(request):
    if request.method == 'POST':
        form = FormulaForm(request.POST)
        if form.is_valid():
            formula = form.save(commit=False)
            formula.user = request.user
            formula.save()
            messages.success(request, 'Formula saved successfully.')
            return redirect('formula_builder')
    else:
        form = FormulaForm()
    formulas = Formula.objects.filter(user=request.user).order_by('-created_at')
    return render(request, 'formula_builder.html', {'form': form, 'formulas': formulas})

@login_required
def evaluate_formula(request, formula_id):
    formula = Formula.objects.get(id=formula_id, user=request.user)
    result = None
    error = None
    if request.method == 'POST':
        # Expecting variables as POST data
        variables = {}
        for key, value in request.POST.items():
            if key.startswith('var_'):
                var_name = key[4:]
                try:
                    variables[var_name] = float(value)
                except ValueError:
                    error = f"Invalid value for variable {var_name}"
                    break
        if not error:
            try:
                # Safe evaluation context
                allowed_names = {"__builtins__": {}}
                allowed_names.update(variables)
                result = eval(formula.expression, allowed_names)
            except Exception as e:
                error = str(e)
    else:
        # Extract variable names from expression (simple heuristic)
        import re
        vars_found = set(re.findall(r'\b[a-zA-Z_][a-zA-Z0-9_]*\b', formula.expression))
        # Remove Python keywords and builtins
        import keyword
        vars_filtered = [v for v in vars_found if not keyword.iskeyword(v) and v not in dir(__builtins__)]
        return render(request, 'evaluate_formula.html', {'formula': formula, 'variables': vars_filtered})
    return render(request, 'evaluate_formula.html', {'formula': formula, 'result': result, 'error': error})
