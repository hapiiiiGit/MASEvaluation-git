# Puppeteer PDF generation job
async def generate_pdf_with_puppeteer(html_content: str, watermark_text: str) -> str:
    import pyppeteer

    browser = await pyppeteer.launch(args=['--no-sandbox'])
    page = await browser.newPage()
    await page.setContent(html_content, waitUntil='networkidle0')

    # Add watermark via page.evaluate
    await page.evaluate(f'''
        () => {{
            const watermark = document.createElement('div');
            watermark.style.position = 'fixed';
            watermark.style.top = '50%';
            watermark.style.left = '50%';
            watermark.style.transform = 'translate(-50%, -50%) rotate(-45deg)';
            watermark.style.fontSize = '48px';
            watermark.style.color = 'rgba(200, 200, 200, 0.3)';
            watermark.style.zIndex = '10000';
            watermark.style.pointerEvents = 'none';
            watermark.style.userSelect = 'none';
            watermark.innerText = '{watermark_text}';
            document.body.appendChild(watermark);
        }}
    ''')

    # Generate PDF with index (bookmarks)
    pdf_bytes = await page.pdf({
        'format': 'A4',
        'printBackground': True,
        'displayHeaderFooter': False,
        'margin': {'top': '20mm', 'bottom': '20mm', 'left': '15mm', 'right': '15mm'},
        'preferCSSPageSize': True,
    })
    await browser.close()

    # Save to temp file
    tmp_file = tempfile.NamedTemporaryFile(delete=False, suffix=".pdf")
    tmp_file.write(pdf_bytes)
    tmp_file.close()
    return tmp_file.name

def process_dispute_evidence(data: dict):
    # Decrypt PII

    # Run Puppeteer PDF generation synchronously in event loop

    try:

            }

    except Exception as e:
        stripe_response = {"error": str(e)}

    # PayPal submission (simplified)
    try:
        # Get access token
        auth_resp = httpx.post(

            auth=(PAYPAL_CLIENT_ID, PAYPAL_CLIENT_SECRET),

        # Submit evidence (mocked, as PayPal API for dispute evidence is limited)
        paypal_response = httpx.post(

        paypal_result = paypal_response.json()
    except Exception as e:
        paypal_result = {"error": str(e)}

    # Cleanup PDF file
    os.remove(pdf_path)

    return {"stripe": stripe_response, "paypal": paypal_result}

# Puppeteer PDF generation job
async def generate_pdf_with_puppeteer(html_content: str, watermark_text: str) -> str:
    import pyppeteer

    browser = await pyppeteer.launch(args=['--no-sandbox'])
    page = await browser.newPage()
    await page.setContent(html_content, waitUntil='networkidle0')

    # Add watermark via page.evaluate
    await page.evaluate(f'''
        () => {{
            const watermark = document.createElement('div');
            watermark.style.position = 'fixed';
            watermark.style.top = '50%';
            watermark.style.left = '50%';
            watermark.style.transform = 'translate(-50%, -50%) rotate(-45deg)';
            watermark.style.fontSize = '48px';
            watermark.style.color = 'rgba(200, 200, 200, 0.3)';
            watermark.style.zIndex = '10000';
            watermark.style.pointerEvents = 'none';
            watermark.style.userSelect = 'none';
            watermark.innerText = '{watermark_text}';
            document.body.appendChild(watermark);
        }}
    ''')

    pdf_bytes = await page.pdf({
        'format': 'A4',
        'printBackground': True,
        'displayHeaderFooter': False,
        'margin': {'top': '20mm', 'bottom': '20mm', 'left': '15mm', 'right': '15mm'},
        'preferCSSPageSize': True,
    })
    await browser.close()

    tmp_file = tempfile.NamedTemporaryFile(delete=False, suffix=".pdf")
    tmp_file.write(pdf_bytes)
    tmp_file.close()
    return tmp_file.name

# Job function to process dispute evidence
def process_dispute_evidence(data: dict):
    customer_email = decrypt_pii(data["customer_email_enc"])
    order_id = data["order_id"]
    evidence_details = data["evidence_details"]

    html_content = f"""
    <html>
    <head><title>Dispute Evidence for Order {order_id}</title></head>
    <body>
        <h1>Dispute Evidence</h1>
        <p><strong>Order ID:</strong> {order_id}</p>
        <p><strong>Customer Email:</strong> {customer_email}</p>
        <h2>Details:</h2>
        <pre>{json.dumps(evidence_details, indent=2)}</pre>
    </body>
    </html>
    """

    pdf_path = generate_pdf_sync(html_content, watermark_text="CONFIDENTIAL")

    stripe_response = None
    try:
        with open(pdf_path, "rb") as f:
            stripe_file = stripe.File.create(
                purpose="dispute_evidence",
                file=f,
                file_link_data={"create": True},
            )
        stripe_response = stripe.Dispute.submit_evidence(
            dispute=data.get("stripe_dispute_id", ""),
            evidence={
                "customer_email_address": customer_email,
                "product_description": "Dispute evidence PDF attached.",
                "additional_evidence": "See attached PDF.",
                "file": stripe_file.id,
            }
        )
    except Exception as e:
        stripe_response = {"error": str(e)}

    paypal_result = None
    try:
        auth_resp = httpx.post(
            f"{PAYPAL_API_BASE}/v1/oauth2/token",
            auth=(PAYPAL_CLIENT_ID, PAYPAL_CLIENT_SECRET),
            data={"grant_type": "client_credentials"},
            timeout=10
        )
        auth_resp.raise_for_status()
        access_token = auth_resp.json()["access_token"]

        with open(pdf_path, "rb") as f:
            files = {"file": ("evidence.pdf", f, "application/pdf")}
            headers = {"Authorization": f"Bearer {access_token}"}
            paypal_resp = httpx.post(
                f"{PAYPAL_API_BASE}/v1/customer/disputes/{data.get('paypal_dispute_id', '')}/evidence",
                headers=headers,
                files=files,
                timeout=10
            )
        paypal_resp.raise_for_status()
        paypal_result = paypal_resp.json()
    except Exception as e:
        paypal_result = {"error": str(e)}

    try:
        os.remove(pdf_path)
    except Exception:
        pass

    return {"stripe": stripe_response, "paypal": paypal_result}

        browser = await p.chromium.launch(args=['--no-sandbox'])
        page = await browser.new_page()
        await page.set_content(html_content, wait_until='networkidle')

        # Inject watermark div
        await page.evaluate(f'''
            () => {{
                const watermark = document.createElement('div');
                watermark.style.position = 'fixed';
                watermark.style.top = '50%';
                watermark.style.left = '50%';
                watermark.style.transform = 'translate(-50%, -50%) rotate(-45deg)';
                watermark.style.fontSize = '48px';
                watermark.style.color = 'rgba(200, 200, 200, 0.3)';
                watermark.style.zIndex = '10000';
                watermark.style.pointerEvents = 'none';
                watermark.style.userSelect = 'none';
                watermark.innerText = '{watermark_text}';
                document.body.appendChild(watermark);
            }}
        ''')

        pdf_bytes = await page.pdf(
            format='A4',
            print_background=True,
            margin={'top': '20mm', 'bottom': '20mm', 'left': '15mm', 'right': '15mm'},
            prefer_css_page_size=True
        )

        await browser.close()

        tmp_file = tempfile.NamedTemporaryFile(delete=False, suffix=".pdf")
        tmp_file.write(pdf_bytes)
        tmp_file.close()
        return tmp_file.name

# Synchronous wrapper to run async PDF generation in RQ job
def generate_pdf_sync(html_content: str, watermark_text: str) -> str:
    return asyncio.run(generate_pdf_with_playwright(html_content, watermark_text))

# Job function to process dispute evidence
def process_dispute_evidence(data: dict):
    customer_email = decrypt_pii(data["customer_email_enc"])
    order_id = data["order_id"]
    evidence_details = data["evidence_details"]

    html_content = f"""
    <html>
    <head><title>Dispute Evidence for Order {order_id}</title></head>
    <body>
        <h1>Dispute Evidence</h1>
        <p><strong>Order ID:</strong> {order_id}</p>
        <p><strong>Customer Email:</strong> {customer_email}</p>
        <h2>Details:</h2>
        <pre>{json.dumps(evidence_details, indent=2)}</pre>
    </body>
    </html>
    """

    pdf_path = generate_pdf_sync(html_content, watermark_text="CONFIDENTIAL")

    stripe_response = None
    try:
        with open(pdf_path, "rb") as f:
            stripe_file = stripe.File.create(
                purpose="dispute_evidence",
                file=f,
                file_link_data={"create": True},
            )
        stripe_response = stripe.Dispute.submit_evidence(
            dispute=data.get("stripe_dispute_id", ""),
            evidence={
                "customer_email_address": customer_email,
                "product_description": "Dispute evidence PDF attached.",
                "additional_evidence": "See attached PDF.",
                "file": stripe_file.id,
            }
        )
    except Exception as e:
        stripe_response = {"error": str(e)}

    paypal_result = None
    try:
        auth_resp = httpx.post(
            f"{PAYPAL_API_BASE}/v1/oauth2/token",
            auth=(PAYPAL_CLIENT_ID, PAYPAL_CLIENT_SECRET),
            data={"grant_type": "client_credentials"},
            timeout=10
        )
        auth_resp.raise_for_status()
        access_token = auth_resp.json()["access_token"]

        with open(pdf_path, "rb") as f:
            files = {"file": ("evidence.pdf", f, "application/pdf")}
            headers = {"Authorization": f"Bearer {access_token}"}
            paypal_resp = httpx.post(
                f"{PAYPAL_API_BASE}/v1/customer/disputes/{data.get('paypal_dispute_id', '')}/evidence",
                headers=headers,
                files=files,
                timeout=10
            )
        paypal_resp.raise_for_status()
        paypal_result = paypal_resp.json()
    except Exception as e:
        paypal_result = {"error": str(e)}

    try:
        os.remove(pdf_path)
    except Exception:
        pass

    return {"stripe": stripe_response, "paypal": paypal_result}

@app.get("/job-status/{job_id}", dependencies=[Depends(verify_token)])
def get_job_status(job_id: str):
    job = queue.fetch_job(job_id)
    if not job:
        raise HTTPException(status_code=404, detail="Job not found")
    result = job.result if job.is_finished else None
    return {"job_id": job_id, "status": job.get_status(), "result": result}
