            print(f"Error fetching insights for account {account_id}: {insights_resp.text}")
            continue
        insights_data = insights_resp.json().get('data', [])
        all_metrics.extend(insights_data)
    return all_metrics

def map_metrics_to_hubspot(metrics):
    if not tokens['hubspot']:
        print("HubSpot token not available.")
        return
    headers = {
        'Authorization': f"Bearer {tokens['hubspot']}",
        'Content-Type': 'application/json'
    }
    for metric in metrics:
        # Example mapping: create or update a deal with campaign name and spend
        deal_data = {
            "properties": {
                "dealname": metric.get('campaign_name', 'Unnamed Campaign'),
                "amount": metric.get('spend', '0'),
                "dealstage": "appointmentscheduled",
                "pipeline": "default"
            }
        }
        url = "https://api.hubapi.com/crm/v3/objects/deals"
        resp = requests.post(url, headers=headers, json=deal_data)
        if resp.status_code not in (200, 201):
            print(f"Error creating/updating deal: {resp.text}")

# === Sync Routine ===
def sync_routine(interval_seconds=3600):
    while True:
        print("Starting sync routine...")
        metrics = fetch_facebook_ad_metrics()
        if metrics:
            map_metrics_to_hubspot(metrics)
            global ad_metrics_data
            ad_metrics_data = metrics
            print(f"Synced {len(metrics)} metrics.")
        else:
            print("No metrics fetched.")
        time.sleep(interval_seconds)
