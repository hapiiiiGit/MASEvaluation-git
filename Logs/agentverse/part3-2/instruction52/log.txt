# Please help me implement a Python program to convert a PHP Blood Bank Donor Management System into a Python application using a framework like Django, Flask, or FastAPI, while preserving the original UI and functionality and connecting to Firebase Realtime Database.

from flask import Flask, render_template, request, redirect, url_for, flash
import firebase_admin
from firebase_admin import credentials, db

app = Flask(__name__)
app.secret_key = 'your_secret_key'

# Initialize Firebase Admin SDK
cred = credentials.Certificate('path/to/your/firebase-adminsdk.json')
firebase_admin.initialize_app(cred, {
    'databaseURL': 'https://your-database-name.firebaseio.com/'
})

# Reference to donors in Firebase Realtime Database
donors_ref = db.reference('donors')

@app.route('/')
def index():
    donors = donors_ref.get()
    if donors is None:
        donors = {}
    return render_template('index.html', donors=donors)

@app.route('/add', methods=['GET', 'POST'])
def add_donor():
    if request.method == 'POST':
        name = request.form['name']
        blood_group = request.form['blood_group']
        age = request.form['age']
        phone = request.form['phone']
        address = request.form['address']

        if not name or not blood_group or not age or not phone or not address:
            flash('Please fill all the fields', 'error')
            return redirect(url_for('add_donor'))

        new_donor = {
            'name': name,
            'blood_group': blood_group,
            'age': age,
            'phone': phone,
            'address': address
        }
        donors_ref.push(new_donor)
        flash('Donor added successfully', 'success')
        return redirect(url_for('index'))
    return render_template('add_donor.html')

@app.route('/edit/<donor_id>', methods=['GET', 'POST'])
def edit_donor(donor_id):
    donor = donors_ref.child(donor_id).get()
    if donor is None:
        flash('Donor not found', 'error')
        return redirect(url_for('index'))

    if request.method == 'POST':
        name = request.form['name']
        blood_group = request.form['blood_group']
        age = request.form['age']
        phone = request.form['phone']
        address = request.form['address']

        if not name or not blood_group or not age or not phone or not address:
            flash('Please fill all the fields', 'error')
            return redirect(url_for('edit_donor', donor_id=donor_id))

        updated_donor = {
            'name': name,
            'blood_group': blood_group,
            'age': age,
            'phone': phone,
            'address': address
        }
        donors_ref.child(donor_id).update(updated_donor)
        flash('Donor updated successfully', 'success')
        return redirect(url_for('index'))

    return render_template('edit_donor.html', donor=donor, donor_id=donor_id)

@app.route('/delete/<donor_id>', methods=['POST'])
def delete_donor(donor_id):
    donor = donors_ref.child(donor_id).get()
    if donor:
        donors_ref.child(donor_id).delete()
        flash('Donor deleted successfully', 'success')
    else:
        flash('Donor not found', 'error')
    return redirect(url_for('index'))

if __name__ == '__main__':
    app.run(debug=True)
