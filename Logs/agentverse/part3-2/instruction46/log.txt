import time
import hmac
import hashlib
import requests
import json
import threading

class CoinDCXFuturesClient:
    BASE_URL = "https://api.coindcx.com/api/v1"

    def __init__(self, api_key, api_secret):
        self.api_key = api_key
        self.api_secret = api_secret

    def _get_headers(self, payload):
        signature = hmac.new(
            self.api_secret.encode(), 
            payload.encode(), 
            hashlib.sha256
        ).hexdigest()
        return {
            'X-AUTH-APIKEY': self.api_key,
            'X-AUTH-SIGNATURE': signature,
            'Content-Type': 'application/json'
        }

    def get_futures_ticker(self, symbol):
        url = f"{self.BASE_URL}/futures/ticker?market={symbol}"
        response = requests.get(url)
        return response.json()

    def get_futures_klines(self, symbol, interval='1m', limit=100):
        url = f"{self.BASE_URL}/futures/klines?market={symbol}&interval={interval}&limit={limit}"
        response = requests.get(url)
        return response.json()

    def place_order(self, symbol, side, quantity, price=None, order_type="LIMIT", time_in_force="GTC"):
        path = "/futures/orders"
        url = f"{self.BASE_URL}{path}"
        data = {
            "market": symbol,
            "side": side,
            "quantity": str(quantity),
            "type": order_type,
            "time_in_force": time_in_force
        }
        if price:
            data["price"] = str(price)
        payload = json.dumps(data)
        headers = self._get_headers(payload)
        response = requests.post(url, headers=headers, data=payload)
        return response.json()

    def get_open_positions(self, symbol):
        path = "/futures/positions"
        url = f"{self.BASE_URL}{path}?market={symbol}"
        payload = ""
        headers = self._get_headers(payload)
        response = requests.get(url, headers=headers)
        return response.json()

class TrendFollowingBot:
    def __init__(self, client, symbol, interval='1m', short_window=7, long_window=25, trade_quantity=0.01):
        self.client = client
        self.symbol = symbol
        self.interval = interval
        self.short_window = short_window
        self.long_window = long_window
        self.trade_quantity = trade_quantity
        self.position = None  # 'LONG', 'SHORT', or None
        self.running = False

    def calculate_sma(self, prices, window):
        if len(prices) < window:
            return None
        return sum(prices[-window:]) / window

    def fetch_prices(self):
        klines = self.client.get_futures_klines(self.symbol, self.interval, limit=self.long_window + 10)
        if not klines or 'error' in klines:
            return None
        # klines format: [ [timestamp, open, high, low, close, volume], ... ]
        closes = [float(k[4]) for k in klines]
        return closes

    def check_signal(self, closes):
        short_sma = self.calculate_sma(closes, self.short_window)
        long_sma = self.calculate_sma(closes, self.long_window)
        if short_sma is None or long_sma is None:
            return None
        if short_sma > long_sma:
            return 'BUY'
        elif short_sma < long_sma:
            return 'SELL'
        else:
            return None

    def execute_trade(self, signal):
        positions = self.client.get_open_positions(self.symbol)
        current_position = None
        if positions and isinstance(positions, list):
            for pos in positions:
                if float(pos.get('quantity', 0)) != 0:
                    current_position = 'LONG' if float(pos.get('quantity')) > 0 else 'SHORT'
                    break

        if signal == 'BUY' and current_position != 'LONG':
            # Close short if any
            if current_position == 'SHORT':
                self.client.place_order(self.symbol, 'BUY', self.trade_quantity, order_type="MARKET")
            # Open long
            self.client.place_order(self.symbol, 'BUY', self.trade_quantity, order_type="MARKET")
            self.position = 'LONG'
        elif signal == 'SELL' and current_position != 'SHORT':
            # Close long if any
            if current_position == 'LONG':
                self.client.place_order(self.symbol, 'SELL', self.trade_quantity, order_type="MARKET")
            # Open short
            self.client.place_order(self.symbol, 'SELL', self.trade_quantity, order_type="MARKET")
            self.position = 'SHORT'

    def run(self, poll_interval=60):
        self.running = True
        while self.running:
            closes = self.fetch_prices()
            if closes:
                signal = self.check_signal(closes)
                if signal:
                    self.execute_trade(signal)
            time.sleep(poll_interval)

        self.running = False

                    current_position = 'LONG' if qty > 0 else 'SHORT'
                    break

        if signal == 'BUY' and current_position != 'LONG':
            if current_position == 'SHORT':
                logging.info("Closing SHORT position")
                self.client.place_order(self.symbol, 'BUY', self.trade_quantity, order_type="MARKET")
            logging.info("Opening LONG position")
            self.client.place_order(self.symbol, 'BUY', self.trade_quantity, order_type="MARKET")
            self.position = 'LONG'
        elif signal == 'SELL' and current_position != 'SHORT':
            if current_position == 'LONG':
                logging.info("Closing LONG position")
                self.client.place_order(self.symbol, 'SELL', self.trade_quantity, order_type="MARKET")
            logging.info("Opening SHORT position")
            self.client.place_order(self.symbol, 'SELL', self.trade_quantity, order_type="MARKET")
            self.position = 'SHORT'
        else:
            logging.info("No trade executed. Current position matches signal or no signal.")

    def run(self, poll_interval=60):
        self.running = True
        logging.info("Starting TrendFollowingBot")
        while self.running:
            try:
                closes = self.fetch_prices()
                if closes:
                    signal = self.check_signal(closes)
                    if signal:
                        logging.info(f"Signal detected: {signal}")
                        self.execute_trade(signal)
                    else:
                        logging.info("No clear signal detected")
                else:
                    logging.warning("No price data fetched")
            except Exception as e:
                logging.error(f"Error in bot run loop: {e}")
            time.sleep(poll_interval)

    def stop(self):
        self.running = False
        logging.info("Stopping TrendFollowingBot")
