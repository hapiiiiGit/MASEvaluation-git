    return jsonify(data=real_time_data, predictions=preds)

@app.route('/api/predict', methods=['POST'])
def api_predict():
    global model, scaler, features
    content = request.json
    if not content:
        return jsonify({'error': 'No input data provided'}), 400
    try:
        input_data = [content.get(f, 0) for f in features]
        X_scaled = scaler.transform([input_data])
        pred = model.predict(X_scaled)[0]
        label = 0 if pred == 1 else 1
        return jsonify({'anomaly': bool(label)})
    except Exception as e:
        return jsonify({'error': str(e)}), 500

    model, scaler = train_model(labeled)
    # Start background data feed thread
    thread = threading.Thread(target=data_feed, daemon=True)
    thread.start()
    # Run Flask app
    app.run(debug=False, port=5000)
