class FormulaViewSet(viewsets.ModelViewSet):
    serializer_class = FormulaSerializer
    permission_classes = [IsAuthenticated]

    def get_queryset(self):
        return Formula.objects.filter(user=self.request.user)

    def perform_create(self, serializer):
        serializer.save(user=self.request.user)

    @action(detail=True, methods=['post'])
    def evaluate(self, request, pk=None):
        formula = self.get_object()
        variables = request.data.get('variables', {})
        expression = formula.expression

        # Safe evaluation of expression with variables
        # Only allow math operators and variables
        import ast
        import operator as op
        import math

        # Supported operators
        operators = {
            ast.Add: op.add,
            ast.Sub: op.sub,
            ast.Mult: op.mul,
            ast.Div: op.truediv,
            ast.Pow: op.pow,
            ast.USub: op.neg,
            ast.Mod: op.mod,
        }

        def eval_expr(expr):
            """
            Evaluate expression node safely.
            """
            if isinstance(expr, ast.Num):  # <number>
                return expr.n
            elif isinstance(expr, ast.BinOp):  # <left> <operator> <right>
                return operators[type(expr.op)](eval_expr(expr.left), eval_expr(expr.right))
            elif isinstance(expr, ast.UnaryOp):  # - <operand> e.g., -1
                return operators[type(expr.op)](eval_expr(expr.operand))
            elif isinstance(expr, ast.Name):
                if expr.id in variables:
                    return float(variables[expr.id])
                else:
                    raise ValueError(f"Variable '{expr.id}' not provided")
            else:
                raise TypeError(f"Unsupported expression: {expr}")

        try:
            expr_ast = ast.parse(expression, mode='eval').body
            result = eval_expr(expr_ast)
        except Exception as e:
            return Response({"error": str(e)}, status=status.HTTP_400_BAD_REQUEST)

        return Response({"result": result})

class FormulaViewSet(viewsets.ModelViewSet):
    serializer_class = FormulaSerializer
    permission_classes = [IsAuthenticated]

    def get_queryset(self):
        return Formula.objects.filter(user=self.request.user)

    def perform_create(self, serializer):
        serializer.save(user=self.request.user)

    @action(detail=True, methods=['post'])
    def evaluate(self, request, pk=None):
        formula = self.get_object()
        variables = request.data.get('variables', {})
        expression = formula.expression

        # Safe evaluation of expression with variables
        # Only allow math operators and variables
        import ast
        import operator as op

        # Supported operators
        operators = {
            ast.Add: op.add,
            ast.Sub: op.sub,
            ast.Mult: op.mul,
            ast.Div: op.truediv,
            ast.Pow: op.pow,
            ast.USub: op.neg,
            ast.Mod: op.mod,
        }

        def eval_expr(expr):
            """
            Evaluate expression node safely.
            """
            if isinstance(expr, ast.Constant):  # Python 3.8+
                if isinstance(expr.value, (int, float)):
                    return expr.value
                else:
                    raise TypeError(f"Unsupported constant type: {type(expr.value)}")
            elif isinstance(expr, ast.Num):  # For Python <3.8
                return expr.n
            elif isinstance(expr, ast.BinOp):  # <left> <operator> <right>
                if type(expr.op) not in operators:
                    raise TypeError(f"Unsupported operator: {type(expr.op)}")
                return operators[type(expr.op)](eval_expr(expr.left), eval_expr(expr.right))
            elif isinstance(expr, ast.UnaryOp):  # - <operand> e.g., -1
                if type(expr.op) not in operators:
                    raise TypeError(f"Unsupported unary operator: {type(expr.op)}")
                return operators[type(expr.op)](eval_expr(expr.operand))
            elif isinstance(expr, ast.Name):
                if expr.id in variables:
                    try:
                        return float(variables[expr.id])
                    except ValueError:
                        raise ValueError(f"Variable '{expr.id}' must be a number")
                else:
                    raise ValueError(f"Variable '{expr.id}' not provided")
            else:
                raise TypeError(f"Unsupported expression: {expr}")

        try:
            expr_ast = ast.parse(expression, mode='eval').body
            result = eval_expr(expr_ast)
        except Exception as e:
            return Response({"error": str(e)}, status=status.HTTP_400_BAD_REQUEST)

        return Response({"result": result})
