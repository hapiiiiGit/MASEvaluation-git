import numpy as np
from sklearn.feature_extraction.text import TfidfVectorizer
from sklearn.metrics.pairwise import cosine_similarity

def recommend_ads(user_ads, all_ads, top_n=5):
    """
    Recommend ads based on similarity of titles and descriptions.
    user_ads: list of Ad objects the user interacted with
    all_ads: list of all Ad objects
    """
    corpus = [ad.title + " " + ad.description for ad in all_ads]
    user_corpus = [ad.title + " " + ad.description for ad in user_ads]

    vectorizer = TfidfVectorizer(stop_words='english')
    tfidf_matrix = vectorizer.fit_transform(corpus)
    user_tfidf = vectorizer.transform(user_corpus)

    similarity = cosine_similarity(user_tfidf, tfidf_matrix)
    mean_similarity = similarity.mean(axis=0)

    recommended_indices = mean_similarity.argsort()[::-1]
    recommended_ads = []
    user_ad_ids = {ad.id for ad in user_ads}
    for idx in recommended_indices:
        ad = all_ads[idx]
        if ad.id not in user_ad_ids:
            recommended_ads.append(ad)
        if len(recommended_ads) >= top_n:
            break
    return recommended_ads

from sklearn.feature_extraction.text import TfidfVectorizer
from sklearn.metrics.pairwise import cosine_similarity

def recommend_ads(user_ads, all_ads, top_n=5):
    """
    Recommend ads based on similarity of titles and descriptions.
    user_ads: list of Ad objects the user interacted with
    all_ads: list of all Ad objects
    """
    corpus = [ad.title + " " + ad.description for ad in all_ads]
    user_corpus = [ad.title + " " + ad.description for ad in user_ads]

    vectorizer = TfidfVectorizer(stop_words='english')
    tfidf_matrix = vectorizer.fit_transform(corpus)
    user_tfidf = vectorizer.transform(user_corpus)

    similarity = cosine_similarity(user_tfidf, tfidf_matrix)
    mean_similarity = similarity.mean(axis=0)

    recommended_indices = mean_similarity.argsort()[::-1]
    recommended_ads = []
    user_ad_ids = {ad.id for ad in user_ads}
    for idx in recommended_indices:
        ad = all_ads[idx]
        if ad.id not in user_ad_ids:
            recommended_ads.append(ad)
        if len(recommended_ads) >= top_n:
            break
    return recommended_ads
