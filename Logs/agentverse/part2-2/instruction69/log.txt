def count_sequences(n):
    # The sum of first n bits can be from 0 to n
    # For each possible sum k, number of ways to choose k bits out of n is C(n, k)
    # The last n bits must also have sum k, so number of ways is also C(n, k)
    # Total sequences = sum over k of C(n, k)^2
    from math import comb
    return sum(comb(n, k)**2 for k in range(n+1))
