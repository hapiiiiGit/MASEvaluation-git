def min_swaps_to_convert(s1: str, s2: str) -> int:
    # If lengths differ, conversion is impossible
    if len(s1) != len(s2):
        return -1
    
    # Count mismatches where s1 has '0' and s2 has '1' and vice versa
    zero_to_one = 0
    one_to_zero = 0
    
    for a, b in zip(s1, s2):
        if a != b:
            if a == '0' and b == '1':
                zero_to_one += 1
            elif a == '1' and b == '0':
                one_to_zero += 1
    
    # If the number of mismatches of each type is not equal, 
    # it's impossible to convert by swapping
    if zero_to_one != one_to_zero:
        return -1
    
    # Minimum swaps required is the number of mismatches of one type
    return zero_to_one
