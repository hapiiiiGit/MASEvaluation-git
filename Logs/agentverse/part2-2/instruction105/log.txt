def max_product_pair(arr):
    if len(arr) < 2:
        return None  # Not enough elements for a pair
    
    # Initialize the two largest and two smallest values
    max1 = max2 = float('-inf')
    min1 = min2 = float('inf')
    
    for num in arr:
        # Update max values
        if num > max1:
            max2 = max1
            max1 = num
        elif num > max2:
            max2 = num
        
        # Update min values
        if num < min1:
            min2 = min1
            min1 = num
        elif num < min2:
            min2 = num
    
    # Compare product of two largest and two smallest (negative * negative)
    if max1 * max2 > min1 * min2:
        return (max1, max2)
    else:
        return (min1, min2)
