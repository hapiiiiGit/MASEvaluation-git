def sum_odd_length_subarrays(arr):
    n = len(arr)
    total_sum = 0
    for length in range(1, n+1, 2):  # only odd lengths
        for start in range(n - length + 1):
            subarray = arr[start:start+length]
            total_sum += sum(subarray)
    return total_sum
