import heapq

def k_largest_products(nums1, nums2, k):
    if not nums1 or not nums2 or k <= 0:
        return []

    nums1.sort(reverse=True)
    nums2.sort(reverse=True)

    max_heap = []
    visited = set()

    # Initialize heap with the largest product
    heapq.heappush(max_heap, (-(nums1[0] * nums2[0]), 0, 0))
    visited.add((0, 0))

    result = []
    while k > 0 and max_heap:
        product, i, j = heapq.heappop(max_heap)
        result.append(-product)
        k -= 1

        if i + 1 < len(nums1) and (i + 1, j) not in visited:
            heapq.heappush(max_heap, (-(nums1[i + 1] * nums2[j]), i + 1, j))
            visited.add((i + 1, j))

        if j + 1 < len(nums2) and (i, j + 1) not in visited:
            heapq.heappush(max_heap, (-(nums1[i] * nums2[j + 1]), i, j + 1))
            visited.add((i, j + 1))

    return result
