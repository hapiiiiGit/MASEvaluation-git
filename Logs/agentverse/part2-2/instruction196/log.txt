def polar_to_rectangular(r, theta):
    """
    Convert polar coordinates to rectangular coordinates.

    Parameters:
    r (float): The radius or distance from the origin.
    theta (float): The angle in radians.

    """
    x = r * math.cos(theta)

    return (x, y)
