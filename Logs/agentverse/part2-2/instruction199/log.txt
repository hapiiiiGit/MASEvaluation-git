def remove_leading_zeroes(ip):
    """
    Remove leading zeros from each octet of an IPv4 address.
    Assumes input is a string in IPv4 format.
    Raises ValueError if the IP is invalid.
    """
    octets = ip.split('.')
    if len(octets) != 4:
        raise ValueError("Invalid IP address format: should contain exactly 4 octets")
    cleaned_octets = []
    for octet in octets:
        if not octet.isdigit():
            raise ValueError(f"Invalid octet '{octet}': should contain only digits")
        # Convert to int to remove leading zeros, then back to string
        num = int(octet)
        if num < 0 or num > 255:
            raise ValueError(f"Invalid octet '{octet}': should be between 0 and 255")
        cleaned_octets.append(str(num))
    return '.'.join(cleaned_octets)
