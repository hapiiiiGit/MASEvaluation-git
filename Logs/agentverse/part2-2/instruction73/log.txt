def max_sum_increasing_subsequence_with_k(arr, k):
    n = len(arr)
    # dp[i] will store the maximum sum of increasing subsequence ending at index i
    dp = [0] * n
    for i in range(n):
        dp[i] = arr[i]
        for j in range(i):
            if arr[j] < arr[i] and dp[j] + arr[i] > dp[i]:
                dp[i] = dp[j] + arr[i]

    # Now find the maximum sum of increasing subsequence from prefix until i (i < k)
    # that can be extended by arr[k] (arr[i] < arr[k])
    max_sum = 0
    for i in range(k):
        if arr[i] < arr[k]:
            max_sum = max(max_sum, dp[i])

    # Include arr[k] itself
    max_sum += arr[k]

    return max_sum
