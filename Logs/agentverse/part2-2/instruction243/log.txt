def reverse_upto_position(arr, pos):
    """
    Reverses the elements of the array up to the given position (inclusive).
    
    Parameters:
    arr (list): The list to be modified.
    pos (int): The position up to which the array should be reversed.
    
    Returns:
    list: The modified list with elements reversed up to the given position.
    """
    if pos < 0 or pos >= len(arr):
        raise IndexError("Position out of range")
    arr[:pos+1] = arr[:pos+1][::-1]
    return arr
