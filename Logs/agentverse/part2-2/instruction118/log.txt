def find_directrix(vertex, focus):
    """
    Find the directrix of a parabola given its vertex and focus.

    Parameters:
    vertex (tuple): Coordinates of the vertex (h, k).
    focus (tuple): Coordinates of the focus (f_x, f_y).

    Returns:
    float or tuple: The equation of the directrix line.
                    If the parabola opens vertically, returns y = c.
                    If the parabola opens horizontally, returns x = c.
    """
    h, k = vertex
    f_x, f_y = focus

    # Determine the orientation of the parabola
    if h == f_x:
        # Parabola opens vertically
        # Directrix is horizontal line y = k - p, where p = distance from vertex to focus
        p = abs(f_y - k)
        directrix = k - p
        return f"y = {directrix}"
    elif k == f_y:
        # Parabola opens horizontally
        # Directrix is vertical line x = h - p, where p = distance from vertex to focus
        p = abs(f_x - h)
        directrix = h - p
        return f"x = {directrix}"
    else:
        # General case: parabola not aligned with axes
        # Directrix is the line perpendicular to the axis of symmetry passing through point symmetric to focus about vertex
        # Axis of symmetry vector
        axis_vec = (f_x - h, f_y - k)
        # Midpoint between vertex and focus
        mid_x = (h + f_x) / 2
        mid_y = (k + f_y) / 2
        # Slope of axis of symmetry
        if axis_vec[0] == 0:
            # Vertical axis of symmetry
            return f"y = {mid_y}"
        elif axis_vec[1] == 0:
            # Horizontal axis of symmetry
            return f"x = {mid_x}"
        else:
            slope = axis_vec[1] / axis_vec[0]
            # Slope of directrix is negative reciprocal
            directrix_slope = -1 / slope
            # Equation of directrix line passing through midpoint (mid_x, mid_y)
            # y - mid_y = directrix_slope * (x - mid_x)
            return f"y = {directrix_slope}*(x - {mid_x}) + {mid_y}"
