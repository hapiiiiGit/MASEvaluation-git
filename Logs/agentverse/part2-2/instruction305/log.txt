def find_right_insertion_point(arr, x):
    """
    Locate the right insertion point for x in sorted list arr to maintain sorted order.
    This is the index where x would be inserted to the right of any existing entries of x.
    """
    low, high = 0, len(arr)
    while low < high:
        mid = (low + high) // 2
        if x < arr[mid]:
            high = mid
        else:
            low = mid + 1
    return low
