def find_min_diff(arr):
    # Sort the array
    arr.sort()
    # Initialize min_diff to a large value
    min_diff = float('inf')
    # Iterate through the array and find the minimum difference between adjacent elements
    for i in range(len(arr) - 1):
        diff = arr[i+1] - arr[i]
        if diff < min_diff:
            min_diff = diff
    return min_diff
