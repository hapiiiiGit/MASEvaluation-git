def max_product_increasing_subsequence(arr):
    n = len(arr)
    if n == 0:
        return 0

    # dp[i] will store the maximum product of an increasing subsequence ending with arr[i]
    dp = arr.copy()

    for i in range(1, n):
        for j in range(i):
            if arr[j] < arr[i]:
                dp[i] = max(dp[i], dp[j] * arr[i])

    return max(dp)
