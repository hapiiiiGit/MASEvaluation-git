import math

def index_of_smallest_triangular_with_n_digits(n):
    # The smallest n-digit number is 10^(n-1)
    # Triangular number T_k = k(k+1)/2
    # We want the smallest k such that T_k >= 10^(n-1)
    # Solve k(k+1)/2 >= 10^(n-1)
    # k^2 + k - 2*10^(n-1) >= 0
    # Use quadratic formula: k = (-1 + sqrt(1 + 8*10^(n-1)))/2
    val = 10**(n-1)
    k = math.ceil((-1 + math.sqrt(1 + 8*val)) / 2)
    return k
