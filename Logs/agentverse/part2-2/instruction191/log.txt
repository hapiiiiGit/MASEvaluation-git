def find_n_most_expensive_items(data, n):
    """
    Find the n most expensive items in the given dataset.

    Parameters:
    data (list of dict): Dataset where each item is a dictionary with at least a 'price' key.
    n (int): Number of most expensive items to find.

    Returns:
    list of dict: The n most expensive items sorted by price in descending order.
    """
    # Sort the data by 'price' in descending order and return the top n items
    return sorted(data, key=lambda x: x['price'], reverse=True)[:n]
