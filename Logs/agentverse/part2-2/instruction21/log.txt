def max_sum_bitonic_subsequence(arr):
    n = len(arr)
    if n == 0:
        return 0

    # Initialize arrays to store the maximum sum of increasing subsequence ending at each index
    inc = arr[:]  # at least the element itself

    # Compute maximum sum increasing subsequence for each index
    for i in range(1, n):
        for j in range(i):
            if arr[j] < arr[i] and inc[i] < inc[j] + arr[i]:
                inc[i] = inc[j] + arr[i]

    # Initialize arrays to store the maximum sum of decreasing subsequence starting at each index
    dec = arr[:]  # at least the element itself

    # Compute maximum sum decreasing subsequence for each index
    for i in range(n-2, -1, -1):
        for j in range(n-1, i, -1):
            if arr[j] < arr[i] and dec[i] < dec[j] + arr[i]:
                dec[i] = dec[j] + arr[i]

    # Find the maximum sum of bitonic subsequence by combining inc and dec
    max_sum = 0
    for i in range(n):
        # inc[i] + dec[i] counts arr[i] twice, so subtract arr[i]
        max_sum = max(max_sum, inc[i] + dec[i] - arr[i])

    return max_sum
