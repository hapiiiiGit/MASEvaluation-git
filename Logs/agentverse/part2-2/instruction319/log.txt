def count_rotations(arr):
    n = len(arr)
    low = 0
    high = n - 1

    while low <= high:
        # If the subarray is already sorted, return the smallest element's index
        if arr[low] <= arr[high]:
            return low

        mid = (low + high) // 2
        next_idx = (mid + 1) % n
        prev_idx = (mid - 1 + n) % n

        # Check if mid element is the minimum
        if arr[mid] <= arr[next_idx] and arr[mid] <= arr[prev_idx]:
            return mid
        elif arr[mid] <= arr[high]:
            high = mid - 1
        else:
            low = mid + 1

    return 0
