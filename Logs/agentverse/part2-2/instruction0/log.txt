def sum_in_range(numbers, start, end):
    """
    Returns the sum of elements in the list 'numbers' from index 'start' to 'end' inclusive.
    
    Parameters:
    numbers (list of int/float): The list of numbers.
    start (int): The starting index of the range.
    end (int): The ending index of the range.
    
    Returns:
    int/float: The sum of the elements in the specified range.
    """
    if start < 0 or end >= len(numbers) or start > end:
        raise ValueError("Invalid start or end index")
    return sum(numbers[start:end+1])
