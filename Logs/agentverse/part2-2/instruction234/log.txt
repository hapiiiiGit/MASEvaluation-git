def move_zeroes_to_end(lst):
    """
    Move all zeroes in the list to the end while maintaining the order of non-zero elements.
    """
    non_zero_index = 0
    for i in range(len(lst)):
        if lst[i] != 0:
            lst[non_zero_index] = lst[i]
            non_zero_index += 1
    for i in range(non_zero_index, len(lst)):
        lst[i] = 0
    return lst
