from itertools import combinations

def find_combinations_with_sum(tuple_list, target_sum):
    result = []
    for r in range(1, len(tuple_list) + 1):
        for combo in combinations(tuple_list, r):
            if sum(x[0] for x in combo) == target_sum:
                result.append(combo)
    return result
