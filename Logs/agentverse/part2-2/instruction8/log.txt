def are_all_bits_unset(num: int, start: int, end: int) -> bool:
    """
    Check whether all bits are unset (0) in the given range [start, end] in the integer num.
    Bits are 0-indexed from the right (least significant bit).

    :param num: Integer number to check bits in.
    :param start: Starting bit position (inclusive).
    :param end: Ending bit position (inclusive).
    :return: True if all bits in the range are unset, False otherwise.
    """
    # Create a mask with bits set from start to end
    mask = ((1 << (end - start + 1)) - 1) << start
    # Check if any bit in the range is set
    return (num & mask) == 0
