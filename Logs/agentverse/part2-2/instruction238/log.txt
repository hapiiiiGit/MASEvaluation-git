def wind_chill_index(velocity_kmh, temperature_c):
    """
    Calculate the wind chill index (WCI) using the Canadian formula:
    WCI = 13.12 + 0.6215*T - 11.37*V^0.16 + 0.3965*T*V^0.16
    where T is temperature in Celsius and V is wind velocity in km/h.
    The formula is valid for temperatures <= 10°C and wind speeds > 4.8 km/h.
    The result is rounded up to the next integer.
    """
    if temperature_c > 10 or velocity_kmh <= 4.8:
        return math.ceil(temperature_c)
    wci = 13.12 + 0.6215 * temperature_c - 11.37 * (velocity_kmh ** 0.16) + 0.3965 * temperature_c * (velocity_kmh ** 0.16)
    return math.ceil(wci)
