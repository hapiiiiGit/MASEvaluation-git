def max_aggregate(tuples_list):
    """
    Calculate the maximum aggregate sum from a list of tuples.
    Each tuple contains numeric values, and the aggregate is the sum of the tuple elements.
    
    Args:
    tuples_list (list of tuples): List where each element is a tuple of numbers.
    
    Returns:
    int or float: The maximum aggregate sum found among the tuples.
    """
    if not tuples_list:
