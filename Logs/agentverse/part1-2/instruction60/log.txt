    precedence = {
        '+': (1, 'L'),

    stack = []
    i = 0

        token = tokens[i]
        if token.isdigit():
            output.append(int(token))
        elif token in precedence:
            while stack:
                top = stack[-1]
                if top in precedence:
                    p1, assoc1 = precedence[token]
                    p2, _ = precedence[top]
                    if (assoc1 == 'L' and p1 <= p2) or (assoc1 == 'R' and p1 < p2):
                        output.append(stack.pop())
                    else:
                        break
                else:
                    break
            stack.append(token)

        elif token in precedence:

                    p2, _ = precedence[top]

                        output.append(stack.pop())
                    else:
                        break
                else:
                    break
            stack.append(token)
        i += 1
    while stack:
        output.append(stack.pop())

    # Evaluate RPN

        if isinstance(token, int):
