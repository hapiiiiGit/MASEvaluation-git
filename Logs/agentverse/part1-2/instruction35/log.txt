def can_arrange(arr):

    is not greater than or equal to the element immediately preceding it. If
    no such element exists then return -1. The given array will not contain
    duplicate values.

    Examples:
    can_arrange([1,2,4,3,5]) = 3

    """
    idx = -1

        if arr[i] < arr[i-1]:

    return idx
