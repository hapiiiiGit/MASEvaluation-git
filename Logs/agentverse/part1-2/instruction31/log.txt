    """Given a positive integer n, return the product of the odd digits.
    Return 0 if all digits are even.
    For example:
    digits(1)  == 1
    digits(4)  == 0
    digits(235) == 15
    """
    product = 1
    found_odd = False
    for d in str(n):
        digit = int(d)
        if digit % 2 == 1:

            found_odd = True
    return product if found_odd else 0
