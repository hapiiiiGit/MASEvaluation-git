plate_cascade = cv2.CascadeClassifier('haarcascade_license_plate.xml')

def process_image(image_path):
    # Read the image
    image = cv2.imread(image_path)
    gray = cv2.cvtColor(image, cv2.COLOR_BGR2GRAY)

    cars = car_cascade.detectMultiScale(gray, 1.1, 1)

        # Region of interest for license plate detection

        roi_color = image[y:y + h, x:x + w]

        # Detect license plates

        for (px, py, pw, ph) in plates:
            cv2.rectangle(roi_color, (px, py), (px + pw, py + ph), (0, 255, 0), 2)

            # Extract the license plate

            text = pytesseract.image_to_string(plate_img, config='--psm 8')
