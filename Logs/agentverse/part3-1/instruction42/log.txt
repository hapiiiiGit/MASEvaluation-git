class ResumeWriter:
    def __init__(self, name, contact_info, skills, achievements, experience):
        self.name = name
        self.contact_info = contact_info
        self.skills = skills
        self.achievements = achievements
        self.experience = experience

    def create_resume(self):
        resume_content = f"Name: {self.name}\n"
        resume_content += f"Contact Info: {self.contact_info}\n\n"
        resume_content += "Skills:\n" + "\n".join(self.skills) + "\n\n"
        resume_content += "Achievements:\n" + "\n".join(self.achievements) + "\n\n"
        resume_content += "Experience:\n" + "\n".join(self.experience) + "\n"
        return resume_content

    def save_to_word(self, content):
        try:
            from docx import Document
            doc = Document()
            doc.add_paragraph(content)
            doc.save(f"{self.name}_resume.docx")
        except ImportError:
            print("Please install the 'python-docx' library to save as Word document.")
        except Exception as e:
            print(f"An error occurred while saving to Word: {e}")

    def save_to_pdf(self, content):
        try:
            from fpdf import FPDF
            pdf = FPDF()
            pdf.add_page()
            pdf.set_font("Arial", size=12)
            for line in content.split('\n'):
                pdf.cell(200, 10, txt=line, ln=True)
            pdf.output(f"{self.name}_resume.pdf")
        except ImportError:
            print("Please install the 'fpdf' library to save as PDF.")
        except Exception as e:
            print(f"An error occurred while saving to PDF: {e}")

        summary = f"{self.name} is a skilled Network/RF Engineer with expertise in Python automation. "
        summary += "With a proven track record of reducing network downtime by 30% and automating reporting processes, "
