from django.db import models
from django.contrib.auth.models import User
from django.urls import reverse
from django.core.validators import EmailValidator
from django.conf import settings

class Event(models.Model):
    user = models.ForeignKey(User, on_delete=models.CASCADE)
    title = models.CharField(max_length=200)
    description = models.TextField()
    date = models.DateTimeField()
    location = models.CharField(max_length=200)
    template = models.CharField(max_length=100)
    image = models.ImageField(upload_to='event_images/')  # Ensure cloud storage is configured in settings
    
    def get_absolute_url(self):
        return reverse('event_detail', kwargs={'pk': self.pk})

class RSVP(models.Model):
    event = models.ForeignKey(Event, on_delete=models.CASCADE)
    guest_name = models.CharField(max_length=100)
    guest_email = models.EmailField(validators=[EmailValidator()])  # Added email validation
    attending = models.BooleanField(default=False)

class Payment(models.Model):
    user = models.ForeignKey(User, on_delete=models.CASCADE)
    event = models.ForeignKey(Event, on_delete=models.CASCADE)
    amount = models.DecimalField(max_digits=10, decimal_places=2)
    payment_date = models.DateTimeField(auto_now_add=True)
    payment_status = models.CharField(max_length=20)

# Admin Panel
from django.contrib import admin

@admin.register(Event)
class EventAdmin(admin.ModelAdmin):
    list_display = ('title', 'date', 'user')
    search_fields = ('title', 'user__username')

@admin.register(RSVP)
class RSVPAdmin(admin.ModelAdmin):
    list_display = ('event', 'guest_name', 'attending')
    search_fields = ('guest_name',)

@admin.register(Payment)
class PaymentAdmin(admin.ModelAdmin):
    list_display = ('user', 'event', 'amount', 'payment_status')
    search_fields = ('user__username', 'event__title')
