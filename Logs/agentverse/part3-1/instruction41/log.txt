# Function to check stock conditions and send alerts
def check_stocks_and_alert(tickers, thresholds, to_email):
    for ticker, threshold in zip(tickers, thresholds):
        try:
            data = fetch_stock_data(ticker)
            current_price = data['Close'].iloc[-1]
            
            if current_price > threshold:
                subject = f"Alert: {ticker} price exceeded threshold"
                body = f"The current price of {ticker} is {current_price}, which exceeds the threshold of {threshold}."
                send_email_alert(subject, body, to_email)
        except Exception as e:
            print(f"Error fetching data for {ticker}: {e}")
