    try:
        url = "https://graph.microsoft.com/v1.0/communications/calls"
        headers = {
            "Authorization": "Bearer YOUR_ACCESS_TOKEN",
            "Content-Type": "application/json"
        }
        payload = {
            "callbackUri": "https://your.callback.url",
            "targets": [{"identity": {"phone": phone_number}}],
            "media": {
                "audio": {
                    "url": "https://your.audio.file/url"
                }
            }
        }
        response = requests.post(url, headers=headers, data=json.dumps(payload))
        response.raise_for_status()
        print(f"Call initiated to {phone_number}")
    except requests.exceptions.RequestException as e:
        print(f"Error making call to {phone_number}: {e}")

# Function to play audio
def play_audio(file_path):
    if not os.path.exists(file_path):
        print(f"Audio file does not exist: {file_path}")
        return
    try:
        winsound.PlaySound(file_path, winsound.SND_FILENAME)
    except Exception as e:
        print(f"Error playing audio file: {e}")

# Main function to automate calling
def automate_calls(file_path, audio_file):
    phone_numbers = load_phone_numbers(file_path)
    for number in phone_numbers:
        make_call(number)
        play_audio(audio_file)
        time.sleep(5)  # Wait before the next call
