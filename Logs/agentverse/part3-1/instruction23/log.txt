import cv2
import numpy as np
import onnxruntime as ort

# Load YOLO model
def load_model(model_path):
    session = ort.InferenceSession(model_path)
    return session

# Preprocess the input image
def preprocess_image(image, input_size):
    image = cv2.resize(image, input_size)
    image = image.astype(np.float32) / 255.0
    image = np.transpose(image, (2, 0, 1))  # Change data layout from HWC to CHW
    image = np.expand_dims(image, axis=0)  # Add batch dimension
    return image

# Perform object detection
def detect_objects(session, image):
    input_name = session.get_inputs()[0].name
    output_name = session.get_outputs()[0].name
    detections = session.run([output_name], {input_name: image})[0]
    return detections

# Draw bounding boxes on the image
def draw_boxes(image, detections, conf_threshold=0.5):
    height, width, _ = image.shape
    for detection in detections:
        confidence = detection[4]
        if confidence > conf_threshold:
            x1, y1, x2, y2 = (detection[0:4] * np.array([width, height, width, height])).astype(int)
            cv2.rectangle(image, (x1, y1), (x2, y2), (0, 255, 0), 2)
            cv2.putText(image, f'Conf: {confidence:.2f}', (x1, y1 - 10), cv2.FONT_HERSHEY_SIMPLEX, 0.5, (0, 255, 0), 2)
    return image

# Main function to run the object tracking
def main(video_source, model_path):
    session = load_model(model_path)
    cap = cv2.VideoCapture(video_source)
    input_size = (640, 640)  # YOLO input size

    while cap.isOpened():

        if not ret:
            break

import cv2
import numpy as np
import onnxruntime as ort

# Load YOLO model
def load_model(model_path):
    session = ort.InferenceSession(model_path)
    return session

# Preprocess the input image
def preprocess_image(image, input_size):
    image = cv2.resize(image, input_size)
    image = image.astype(np.float32) / 255.0
    image = np.transpose(image, (2, 0, 1))  # Change data layout from HWC to CHW
    image = np.expand_dims(image, axis=0)  # Add batch dimension
    return image

# Perform object detection
def detect_objects(session, image):
    input_name = session.get_inputs()[0].name
    output_name = session.get_outputs()[0].name
    detections = session.run([output_name], {input_name: image})[0]
    
    # Post-process detections to filter out low confidence detections
    # Assuming detections are in the format [x1, y1, x2, y2, confidence, class_id]
    return [detection for detection in detections if detection[4] > 0.5]

# Draw bounding boxes on the image
def draw_boxes(image, detections, conf_threshold=0.5):
    height, width, _ = image.shape
    for detection in detections:
        if detection[4] > conf_threshold:  # Check confidence
            x1, y1, x2, y2 = (detection[0:4] * np.array([width, height, width, height])).astype(int)
            cv2.rectangle(image, (x1, y1), (x2, y2), (0, 255, 0), 2)
            cv2.putText(image, f'Conf: {detection[4]:.2f}', (x1, y1 - 10), cv2.FONT_HERSHEY_SIMPLEX, 0.5, (0, 255, 0), 2)
    return image

# Main function to run the object tracking
def main(video_source, model_path):
    session = load_model(model_path)
    cap = cv2.VideoCapture(video_source)
    input_size = (640, 640)  # YOLO input size

    while cap.isOpened():
        ret, frame = cap.read()
        if not ret:
            break

        preprocessed_image = preprocess_image(frame, input_size)
        detections = detect_objects(session, preprocessed_image)
        output_frame = draw_boxes(frame, detections)

        cv2.imshow('Object Tracking', output_frame)
        if cv2.waitKey(1) & 0xFF == ord('q'):
            break

    cap.release()
    cv2.destroyAllWindows()
