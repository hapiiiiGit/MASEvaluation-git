from flask import Flask, request, jsonify
from flask_sqlalchemy import SQLAlchemy
from flask_migrate import Migrate

app = Flask(__name__)
app.config['SQLALCHEMY_DATABASE_URI'] = 'sqlite:///hiregenz.db'
app.config['SQLALCHEMY_TRACK_MODIFICATIONS'] = False
db = SQLAlchemy(app)
migrate = Migrate(app, db)

class Freelancer(db.Model):
    id = db.Column(db.Integer, primary_key=True)
    name = db.Column(db.String(100), nullable=False)
    skills = db.Column(db.String(200), nullable=False)
    bio = db.Column(db.Text, nullable=True)
    projects = db.relationship('Project', backref='freelancer', lazy=True)

class Project(db.Model):
    id = db.Column(db.Integer, primary_key=True)
    title = db.Column(db.String(100), nullable=False)
    description = db.Column(db.Text, nullable=False)
    freelancer_id = db.Column(db.Integer, db.ForeignKey('freelancer.id'), nullable=False)

@app.route('/freelancers', methods=['POST'])
def create_freelancer():
    data = request.json
    new_freelancer = Freelancer(name=data['name'], skills=data['skills'], bio=data.get('bio'))
    db.session.add(new_freelancer)
    db.session.commit()
    return jsonify({'message': 'Freelancer created successfully!'}), 201

@app.route('/projects', methods=['POST'])
def create_project():
    data = request.json
    new_project = Project(title=data['title'], description=data['description'], freelancer_id=data['freelancer_id'])
    db.session.add(new_project)
    db.session.commit()
    return jsonify({'message': 'Project created successfully!'}), 201

if __name__ == '__main__':
    app.run(debug=True)
