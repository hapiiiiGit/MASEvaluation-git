    annotation: str

@app.post("/annotate/")
async def annotate_text(file: UploadFile = File(...)):
    if not file.filename.endswith('.txt'):
        raise HTTPException(status_code=400, detail="Invalid file type. Please upload a .txt file.")
    
    content = await file.read()
    text = content.decode('utf-8')
    
    # Annotate the text using the model
    annotation = annotator(text)
    
    # Ensure annotation is a string
    annotation_str = str(annotation) if isinstance(annotation, list) else annotation
    
    # Save the annotation to the database
    with sqlite3.connect(DATABASE) as conn:
        cursor = conn.cursor()
        cursor.execute("INSERT INTO annotations (text, annotation) VALUES (?, ?)", (text, annotation_str))
        conn.commit()
    
    return {"text": text, "annotation": annotation_str}

@app.get("/annotations/")
def get_annotations():
    with sqlite3.connect(DATABASE) as conn:
        cursor = conn.cursor()
        cursor.execute("SELECT * FROM annotations")
        rows = cursor.fetchall()
    
    return [{"id": row[0], "text": row[1], "annotation": row[2]} for row in rows]
