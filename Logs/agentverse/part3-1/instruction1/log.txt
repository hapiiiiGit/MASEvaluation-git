import pandas as pd
import numpy as np
import requests
from datetime import datetime, timedelta
import xlsxwriter

def fetch_nse_data(symbols, start_date, end_date):
    data = {}
    for symbol in symbols:
        url = f'https://www.nseindia.com/api/option-chain-indices?symbol={symbol}'
        response = requests.get(url)
        if response.status_code == 200:
            json_data = response.json()
            # Assuming the data contains 'data' key with 'records' for historical prices
            historical_data = json_data['records']['data']
            data[symbol] = pd.DataFrame(historical_data)
            data[symbol]['date'] = pd.to_datetime(data[symbol]['date'])
            data[symbol].set_index('date', inplace=True)
    return data

def calculate_metrics(data):
    metrics = {}
    for symbol, df in data.items():
        df['daily_return'] = df['close'].pct_change()
        avg_return = df['daily_return'].mean()
        std_dev = df['daily_return'].std()
        momentum = df['close'].iloc[-1] / df['close'].iloc[-21] - 1  # 21-day momentum
        metrics[symbol] = {
            'average_return': avg_return,
            'standard_deviation': std_dev,
            'momentum': momentum
        }
    return metrics

def rank_stocks(metrics):
    ranked = pd.DataFrame(metrics).T
    ranked['rank_avg_return'] = ranked['average_return'].rank(ascending=False)
    ranked['rank_std_dev'] = ranked['standard_deviation'].rank(ascending=True)
    ranked['rank_momentum'] = ranked['momentum'].rank(ascending=False)
    ranked['final_rank'] = ranked[['rank_avg_return', 'rank_std_dev', 'rank_momentum']].mean(axis=1)
    return ranked.sort_values('final_rank')

import pandas as pd
import numpy as np
import requests
from datetime import datetime, timedelta
import xlsxwriter

def fetch_nse_data(symbols, start_date, end_date):
    data = {}
    for symbol in symbols:
        url = f'https://www.nseindia.com/api/historical/cm/equity?symbol={symbol}&series=[%22EQ%22]&from={start_date}&to={end_date}'
        headers = {'User-Agent': 'Mozilla/5.0'}
        response = requests.get(url, headers=headers)
        if response.status_code == 200:
            json_data = response.json()
            historical_data = json_data['data']
            df = pd.DataFrame(historical_data)
            df['date'] = pd.to_datetime(df['CH_TIMESTAMP'])
            df.set_index('date', inplace=True)
            data[symbol] = df
        else:
            print(f"Failed to fetch data for {symbol}")
    return data

def calculate_metrics(data):
    metrics = {}
    for symbol, df in data.items():
        if df.empty:
            continue
        df['daily_return'] = df['CH_CLOSE_PRICE'].pct_change()
        avg_return = df['daily_return'].mean()
        std_dev = df['daily_return'].std()
        if len(df) >= 21:
            momentum = df['CH_CLOSE_PRICE'].iloc[-1] / df['CH_CLOSE_PRICE'].iloc[-21] - 1
        else:
            momentum = np.nan
        metrics[symbol] = {
            'average_return': avg_return,
            'standard_deviation': std_dev,
            'momentum': momentum
        }
    return metrics

def rank_stocks(metrics):
    ranked = pd.DataFrame(metrics).T
    ranked['rank_avg_return'] = ranked['average_return'].rank(ascending=False)
    ranked['rank_std_dev'] = ranked['standard_deviation'].rank(ascending=True)
    ranked['rank_momentum'] = ranked['momentum'].rank(ascending=False)
