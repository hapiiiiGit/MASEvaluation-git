@login_required

    if request.user.profile.role.name == 'Admin':
        rooms = Room.objects.all()
        return render(request, 'room_management.html', {'rooms': rooms})
    return redirect('home')

@login_required
def make_reservation(request, room_id):
    room = get_object_or_404(Room, id=room_id)
    if request.method == 'POST':
        check_in = request.POST['check_in']
        check_out = request.POST['check_out']
        
        # Validate check-in and check-out dates
        if check_in >= check_out:
            return render(request, 'make_reservation.html', {'room_id': room_id, 'error': 'Check-out date must be after check-in date.'})
        
        # Check for overlapping reservations
        if Reservation.objects.filter(room=room, check_in_date__lt=check_out, check_out_date__gt=check_in).exists():
            return render(request, 'make_reservation.html', {'room_id': room_id, 'error': 'Room is already reserved for the selected dates.'})

        # Create reservation
        reservation = Reservation.objects.create(
            guest=request.user.profile,
            room=room,
            check_in_date=check_in,
            check_out_date=check_out
        )
        room.is_available = False  # Mark room as unavailable after reservation
        room.save()
        return redirect('reservation_success')
    
    return render(request, 'make_reservation.html', {'room_id': room_id})
