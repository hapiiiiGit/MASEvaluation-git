import argparse
import pandas as pd
from sklearn.feature_extraction.text import TfidfVectorizer
from sklearn.decomposition import LatentDirichletAllocation
from sklearn.pipeline import Pipeline
from sklearn.model_selection import train_test_split
from sklearn.metrics import classification_report
from nltk.sentiment import SentimentIntensityAnalyzer
import spacy

# Load Spacy model for key-phrase extraction
nlp = spacy.load("en_core_web_sm")

def preprocess_text(text):
    # Basic text preprocessing
    doc = nlp(text)
    return " ".join([token.lemma_ for token in doc if not token.is_stop and not token.is_punct])

def topic_modeling(texts, n_topics=5):
    vectorizer = TfidfVectorizer()
    lda = LatentDirichletAllocation(n_components=n_topics, random_state=42)
    pipeline = Pipeline([('vectorizer', vectorizer), ('lda', lda)])
    pipeline.fit(texts)
    return pipeline

def key_phrase_extraction(text):
    doc = nlp(text)
    return [chunk.text for chunk in doc.noun_chunks]

def sentiment_analysis(text):
    sia = SentimentIntensityAnalyzer()
    return sia.polarity_scores(text)

def analyze_texts(texts):
    preprocessed_texts = [preprocess_text(text) for text in texts]
    lda_model = topic_modeling(preprocessed_texts)
    
    topics = lda_model.named_steps['lda'].components_
    feature_names = lda_model.named_steps['vectorizer'].get_feature_names_out()
    
    topic_keywords = {}
    for idx, topic in enumerate(topics):
        topic_keywords[f'Topic {idx}'] = [feature_names[i] for i in topic.argsort()[-5:]]
    
    sentiments = [sentiment_analysis(text) for text in texts]
    
    return topic_keywords, sentiments

def main(input_file):

    texts = df['text'].tolist()
