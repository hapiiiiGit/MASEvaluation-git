def can_be_difference_of_two_squares(n):
    if n == 0:
        return True  # 0 = 1^2 - 1^2
    for y in range(1, abs(n)+1):
        if n % y != 0:
            continue
        x = n // y
        if x <= y:
            continue
        # x and y must have same parity
        if (x % 2) != (y % 2):
            continue
        a = (x + y) // 2
        b = (x - y) // 2
        if b >= 0:
            return True
    return False
