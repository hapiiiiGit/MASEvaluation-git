def sort_list(elements, ascending=True):
    """
    Sorts a list of elements in ascending or descending order.

    :param elements: List of elements to sort
    :param ascending: Boolean, True for ascending, False for descending
    :return: Sorted list
    """
    return sorted(elements, reverse=not ascending)
