def next_bigger_number(n: int) -> int:
    digits = list(str(n))
    length = len(digits)

    # Step 1: Find the pivot
    i = length - 2
    while i >= 0 and digits[i] >= digits[i + 1]:
        i -= 1

    if i == -1:
        # Already the largest permutation, return smallest
        return int(''.join(sorted(digits)))

    # Step 2: Find the smallest digit on right of pivot that is greater than pivot
    j = length - 1
    while digits[j] <= digits[i]:
        j -= 1

    # Step 3: Swap pivot and successor
    digits[i], digits[j] = digits[j], digits[i]

    # Step 4: Reverse the sublist after pivot
    digits[i + 1:] = reversed(digits[i + 1:])

    return int(''.join(digits))
