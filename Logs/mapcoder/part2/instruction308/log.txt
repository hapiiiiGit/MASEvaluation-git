def is_lucid(num):
    # Lucid numbers are numbers in the sequence of even-indexed Fibonacci numbers
    # i.e., numbers that appear as F(2k) for k >= 1
    # Generate Fibonacci numbers up to num and check if num is among them
    a, b = 0, 1
    lucid_set = set()
    idx = 0
    while a <= num:
        if idx % 2 == 0 and a != 0:
            lucid_set.add(a)
        a, b = b, a + b
        idx += 1
    return num in lucid_set

def get_lucid_numbers(n):
    result = []
    a, b = 0, 1
    idx = 0
    while a <= n:
        if idx % 2 == 0 and a != 0:
            result.append(a)
        a, b = b, a + b
        idx += 1
    return result
