def sum_per_digit_difference(a: int, b: int) -> int:
    sa = str(a)
    sb = str(b)
    max_len = max(len(sa), len(sb))
    sa = sa.zfill(max_len)
    sb = sb.zfill(max_len)
    total = 0
    for da, db in zip(sa, sb):
        total += abs(int(da) - int(db))
    return total
