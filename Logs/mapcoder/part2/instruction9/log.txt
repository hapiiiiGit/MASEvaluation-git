def triangular_prism_volume(base: float, height: float, length: float) -> float:
    """
    Calculate the volume of a triangular prism.

    Parameters:
    base (float): The base length of the triangle.
    height (float): The height of the triangle.
    length (float): The length of the prism.

    Returns:
    float: The volume of the triangular prism.
    """
    area_of_triangle = 0.5 * base * height
    volume = area_of_triangle * length
    return volume
