def set_leftmost_unset_bit(n):
    bit_len = n.bit_length() or 1
    for pos in reversed(range(bit_len)):
        if not (n & (1 << pos)):
            return n | (1 << pos)
    # If all bits in current bit length are set, set the next higher bit
    return n | (1 << bit_len)
