def remove_uppercase_substrings(s: str) -> str:
    result = []
    i = 0
    n = len(s)
    while i < n:
        if s[i].isupper():
            # Skip the entire uppercase substring
            while i < n and s[i].isupper():
                i += 1
        else:
            result.append(s[i])
            i += 1
    return ''.join(result)
