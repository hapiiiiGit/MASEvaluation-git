import re

def find_first_ly_adverb(s):
    # Use regex to find words ending with 'ly' (case-insensitive), handling punctuation
    pattern = re.compile(r'\b\w+ly\b', re.IGNORECASE)
    match = pattern.search(s)
    if match:
        word = match.group()
        start = match.start()
        end = match.end() - 1  # inclusive end position
        return (word, start, end)
    else:
        return None
