def count_unequal_pairs(arr):
    from collections import Counter
    freq = Counter(arr)
    n = len(arr)
    total_pairs = n * (n - 1) // 2
    equal_pairs = sum(f * (f - 1) // 2 for f in freq.values())
    return total_pairs - equal_pairs
