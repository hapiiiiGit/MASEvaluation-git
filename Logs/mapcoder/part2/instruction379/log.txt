def count_reverse_pairs(strings):
    from collections import defaultdict
    count_dict = defaultdict(int)
    pair_count = 0
    for s in strings:
        rev = s[::-1]
        pair_count += count_dict[rev]
        count_dict[s] += 1
    return pair_count
