def eulerian_number(n, m):
    if m < 0 or m >= n:
        return 0
    dp = [[0 for _ in range(n)] for _ in range(n+1)]
    dp[0][0] = 1
    for i in range(1, n+1):
        for j in range(i):
            val1 = (i - j) * dp[i-1][j-1] if j-1 >= 0 else 0
            val2 = (j + 1) * dp[i-1][j]
            dp[i][j] = val1 + val2
    return dp[n][m]
