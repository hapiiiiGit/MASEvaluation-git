def max_diff_tuple(tuple_list):
    max_diff = -1
    result_tuple = None
    for t in tuple_list:
        if len(t) != 2:
            continue  # skip invalid tuples
        diff = abs(t[0] - t[1])
        if diff > max_diff:
            max_diff = diff
            result_tuple = t
    return result_tuple
