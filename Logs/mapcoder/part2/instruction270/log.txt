def tuple_union_sorted(tuple1, tuple2):
    # Convert tuples to sets
    set1 = set(tuple1)
    set2 = set(tuple2)
    # Find the union
    union_set = set1 | set2
    # Sort the result
    sorted_list = sorted(union_set)
    return sorted_list
