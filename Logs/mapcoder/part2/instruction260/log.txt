def sum_of_min_and_max(arr):
    if not arr:
        raise ValueError("Array must not be empty")
    min_val = max_val = arr[0]
    for num in arr:
        if num < min_val:
            min_val = num
        if num > max_val:
            max_val = num
    return min_val + max_val
