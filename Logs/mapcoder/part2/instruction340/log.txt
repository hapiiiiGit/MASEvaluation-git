def remove_parenthesis_content(s: str) -> str:
    result = []
    depth = 0
    for c in s:
        if c == '(':
            depth += 1
        elif c == ')':
            if depth > 0:
                depth -= 1
        elif depth == 0:
            result.append(c)
    return ''.join(result)
