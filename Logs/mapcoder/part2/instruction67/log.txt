def convert_to_floats(data):
    result = []
    for sublist in data:
        new_sublist = []
        for elem in sublist:
            try:
                new_elem = float(elem)
            except (ValueError, TypeError):
                new_elem = elem
            new_sublist.append(new_elem)
        result.append(new_sublist)
    return result
