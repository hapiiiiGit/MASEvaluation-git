def can_be_sum_of_powers_of_2(n):
    if n <= 0:
        return False
    powers = []
    power = 1
    while power <= n:
        powers.append(power)
        power *= 2
    powers.reverse()
    remaining = n
    for p in powers:
        if p <= remaining:
            remaining -= p
    return remaining == 0
