import math

def smallest_triangular_index(n):
    # The smallest n-digit number
    min_val = 10 ** (n - 1)
    # Solve k^2 + k - 2*min_val >= 0
    k_est = math.ceil((-1 + math.sqrt(1 + 8 * min_val)) / 2)
    k = k_est
    while True:
        t_k = k * (k + 1) // 2
        if t_k >= min_val:
            return k
        k += 1
