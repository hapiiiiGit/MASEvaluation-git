from collections import Counter, defaultdict

def min_length_sublist_with_all_unique(lst):
    if not lst:
        return []

    unique_elements = set(lst)
    required = len(unique_elements)
    count = defaultdict(int)
    formed = 0
    left = 0
    min_len = float('inf')
    min_window = (0, 0)

    for right, val in enumerate(lst):
        count[val] += 1
        if count[val] == 1 and val in unique_elements:
            formed += 1

        while formed == required:
            window_len = right - left + 1
            if window_len < min_len:
                min_len = window_len
                min_window = (left, right)

            count[lst[left]] -= 1
            if count[lst[left]] == 0 and lst[left] in unique_elements:
                formed -= 1
            left += 1

    if min_len == float('inf'):
        return []
    return lst[min_window[0]:min_window[1]+1]
