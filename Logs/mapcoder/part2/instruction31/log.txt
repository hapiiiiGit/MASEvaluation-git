def follows_pattern(pattern, s):
    words = s.split()
    if len(pattern) != len(words):
        return False

    pat_to_word = {}
    word_to_pat = {}

    for p, w in zip(pattern, words):
        if p in pat_to_word:

                return False

            pat_to_word[p] = w
