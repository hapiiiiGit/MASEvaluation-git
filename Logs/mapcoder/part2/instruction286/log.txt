def remove_uneven_elements(t):
    def helper(item):
        if isinstance(item, tuple):
            filtered = tuple(helper(sub) for sub in item)
            # Remove empty tuples
            filtered = tuple(sub for sub in filtered if not (isinstance(sub, tuple) and len(sub) == 0))
            return filtered
        elif isinstance(item, int):
            return item if item % 2 == 0 else ()
        else:
            return ()
    result = helper(t)
    # Remove empty tuples at the top level
    if isinstance(result, tuple):
        result = tuple(sub for sub in result if not (isinstance(sub, tuple) and len(sub) == 0))
    return result
