def most_frequent_item(lst):
    freq = {}
    for item in lst:
        freq[item] = freq.get(item, 0) + 1
    max_count = -1
    max_item = None
    for item, count in freq.items():
        if count > max_count:
            max_count = count
            max_item = item
    return max_item
