def find_minimum_difference(arr):
    if len(arr) < 2:
        return None  # Not enough elements to form a pair

    arr.sort()
    min_diff = float('inf')
    for i in range(1, len(arr)):
        diff = arr[i] - arr[i - 1]
        if diff < min_diff:
            min_diff = diff
    return min_diff
