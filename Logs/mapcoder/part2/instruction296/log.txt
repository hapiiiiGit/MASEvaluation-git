import math

def polar_to_rectangular(r, theta):
    """
    Convert polar coordinates (r, theta) to rectangular coordinates (x, y).
    theta is in radians.
    """
    x = r * math.cos(theta)
    y = r * math.sin(theta)
    return (x, y)
