def sum_proper_divisors(n):
    if n < 2:
        return 0
    total = 1
    sqrt_n = int(n ** 0.5)
    for i in range(2, sqrt_n + 1):
        if n % i == 0:
            total += i
            if i != n // i:
                total += n // i
    return total

def sum_amicable_numbers(limit):
    amicable = set()
    for num in range(2, limit + 1):
        a = sum_proper_divisors(num)
        if a != num and a <= limit:
            b = sum_proper_divisors(a)
            if b == num and a != b:
                amicable.add(num)
                amicable.add(a)
    return sum([n for n in amicable if n <= limit])
