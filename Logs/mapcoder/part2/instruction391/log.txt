def is_balanced(expression):
    stack = []
    bracket_map = {')': '(', '}': '{', ']': '['}
    opening = set(bracket_map.values())

    for char in expression:
        if char in opening:
            stack.append(char)
        elif char in bracket_map:
            if not stack or stack[-1] != bracket_map[char]:
                return False
            stack.pop()
    return not stack
