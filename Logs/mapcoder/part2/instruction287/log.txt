import math

def surface_area_square_pyramid(base_edge, height):
    # Area of the base (square)
    base_area = base_edge ** 2
    # Slant height calculation
    slant_height = math.sqrt((base_edge / 2) ** 2 + height ** 2)
    # Area of one triangular face
    triangle_area = (base_edge * slant_height) / 2
    # Total lateral area (4 triangles)
    lateral_area = 4 * triangle_area
    # Total surface area
    total_surface_area = base_area + lateral_area
    return total_surface_area
