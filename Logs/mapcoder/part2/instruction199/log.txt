def min_tuple_product(tuple_list):
    if not tuple_list:
        return None  # or raise an exception if preferred
    min_product = None
    for t in tuple_list:
        product = 1
        for num in t:
            product *= num
        if min_product is None or product < min_product:
            min_product = product
    return min_product
