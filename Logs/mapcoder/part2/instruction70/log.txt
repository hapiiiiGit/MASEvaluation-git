def max_abs_product_pair(tuples_list):
    max_product = float('-inf')
    result = None
    n = len(tuples_list)
    for i in range(n):
        for j in range(i+1, n):
            tuple1 = tuples_list[i]
            tuple2 = tuples_list[j]
            for num1 in tuple1:
                for num2 in tuple2:
                    abs_product = abs(num1 * num2)
                    if abs_product > max_product:
                        max_product = abs_product
                        result = ((tuple1, tuple2), (num1, num2))
    return result
