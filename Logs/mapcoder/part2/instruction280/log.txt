from collections import Counter
from functools import reduce
import operator

def product_of_unique_numbers(lst):
    freq = Counter(lst)
    unique_numbers = [num for num, count in freq.items() if count == 1]
    if not unique_numbers:
        return 1  # or 0, depending on how you want to handle no unique numbers
    return reduce(operator.mul, unique_numbers, 1)
