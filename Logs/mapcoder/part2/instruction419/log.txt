def count_rotations(arr):
    n = len(arr)
    low, high = 0, n - 1

    while low <= high:
        # If the subarray is already sorted, return the smallest index
        if arr[low] <= arr[high]:
            return low

        mid = (low + high) // 2
        next_idx = (mid + 1) % n
        prev_idx = (mid - 1 + n) % n

        # Check if mid is the minimum element
        if arr[mid] <= arr[next_idx] and arr[mid] <= arr[prev_idx]:
            return mid

        # Decide which half to choose
        if arr[mid] >= arr[low]:
            low = mid + 1
        else:
            high = mid - 1

    return 0  # If not found, array is not rotated
