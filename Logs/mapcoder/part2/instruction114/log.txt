def set_even_bits(n):
    # Determine the number of bits needed to represent n
    num_bits = n.bit_length()
    # Construct mask: set all even bits (1-based positions 2,4,6,...)
    mask = 0
    for i in range(1, num_bits+1, 2):  # i: 1-based even positions
        mask |= (1 << i)
    return n | mask
