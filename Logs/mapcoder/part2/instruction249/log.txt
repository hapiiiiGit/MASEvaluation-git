def all_combinations(lst):
    result = []

    def backtrack(start, comb):
        result.append(comb[:])
        for i in range(start, len(lst)):
            comb.append(lst[i])
            backtrack(i + 1, comb)
            comb.pop()

    backtrack(0, [])
    return result
