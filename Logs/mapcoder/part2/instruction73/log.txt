def max_zero_one_diff(s: str) -> int:
    max_diff = float('-inf')
    curr_diff = 0
    for ch in s:
        val = 1 if ch == '0' else -1
        curr_diff = max(val, curr_diff + val)
        max_diff = max(max_diff, curr_diff)
    return max_diff
