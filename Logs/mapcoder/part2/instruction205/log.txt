def highest_product_pair(arr):
    if len(arr) < 2:
        return None  # Not enough elements for a pair

    arr.sort()
    prod1 = arr[0] * arr[1]
    prod2 = arr[-1] * arr[-2]

    if prod1 > prod2:
        return (arr[0], arr[1])
    else:
        return (arr[-2], arr[-1])
