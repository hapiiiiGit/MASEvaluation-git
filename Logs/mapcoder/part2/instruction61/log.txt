def merge_three_sorted_lists(list1, list2, list3):
    i, j, k = 0, 0, 0
    result = []
    n1, n2, n3 = len(list1), len(list2), len(list3)
    while i < n1 or j < n2 or k < n3:
        vals = []
        if i < n1:
            vals.append((list1[i], 'a'))
        if j < n2:
            vals.append((list2[j], 'b'))
        if k < n3:
            vals.append((list3[k], 'c'))
        if not vals:
            break
        val, which = min(vals)
        result.append(val)
        if which == 'a':
            i += 1
        elif which == 'b':
            j += 1
        else:
            k += 1
    return result
