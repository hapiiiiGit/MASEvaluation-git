def max_sum_increasing_subsequence(arr):
    n = len(arr)
    dp = arr.copy()

def max_sum_incr_subseq_with_k(arr, i, k):
    n = len(arr)
    if not (0 <= i < k < n):
        return 0  # invalid indices

    dp = [0] * (i + 1)
    for j in range(i + 1):
        dp[j] = arr[j]
        for l in range(j):
            if arr[l] < arr[j]:
                dp[j] = max(dp[j], dp[l] + arr[j])

    max_sum = 0
    for j in range(i + 1):
        if arr[j] < arr[k]:
            candidate_sum = dp[j] + arr[k]
            if candidate_sum > max_sum:
                max_sum = candidate_sum

    return max_sum
