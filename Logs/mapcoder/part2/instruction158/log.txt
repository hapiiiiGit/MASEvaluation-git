def max_subarray_sum_k_concat(arr, k):
    def kadane(nums):
        max_ending_here = max_so_far = nums[0]
        for x in nums[1:]:
            max_ending_here = max(x, max_ending_here + x)
            max_so_far = max(max_so_far, max_ending_here)
        return max_so_far

    n = len(arr)
    total_sum = sum(arr)

    # Max subarray sum in one array
    max_kadane = kadane(arr)

    # Max prefix sum
    max_prefix = curr = arr[0]
    for i in range(1, n):
        curr += arr[i]
        max_prefix = max(max_prefix, curr)

    # Max suffix sum
    max_suffix = curr = arr[-1]
    for i in range(n-2, -1, -1):
        curr += arr[i]
        max_suffix = max(max_suffix, curr)

    if k == 1:
        return max_kadane

    max_wrap = max_prefix + max_suffix
    if total_sum > 0 and k > 2:
        max_wrap += (k-2) * total_sum

    return max(max_kadane, max_wrap)
