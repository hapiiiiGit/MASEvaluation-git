def put_spaces_between_capital_words(s: str) -> str:
    if not s:
        return ""
    words = []
    start = 0
    for i in range(1, len(s)):
        if s[i].isupper():
            words.append(s[start:i])
            start = i
    words.append(s[start:])
    return ' '.join(words)
