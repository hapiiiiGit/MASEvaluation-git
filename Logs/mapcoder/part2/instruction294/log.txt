def difference_largest_smallest(lst):
    if not lst:
        raise ValueError("List must not be empty")
    max_value = min_value = lst[0]
    for num in lst:
        if num > max_value:
            max_value = num
        if num < min_value:
            min_value = num
    return max_value - min_value
