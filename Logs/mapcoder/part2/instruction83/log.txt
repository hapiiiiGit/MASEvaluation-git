def zero_nonzero_ratio(arr):
    total = len(arr)
    if total == 0:
        return (0.0, 0.0)
    zero_count = 0
    nonzero_count = 0
    for num in arr:
        if num == 0:
            zero_count += 1
        else:
            nonzero_count += 1
    zero_ratio = zero_count / total
    nonzero_ratio = nonzero_count / total
    return (zero_ratio, nonzero_ratio)
