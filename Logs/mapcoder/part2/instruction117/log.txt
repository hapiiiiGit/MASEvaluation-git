def count_sequences(m, n):
    from functools import lru_cache

    @lru_cache(maxsize=None)
    def dp(prev, k):
        if k == 0:
            return 1
        total = 0
        start = 1 if prev == 0 else 2 * prev
        for next_elem in range(start, m + 1):
            total += dp(next_elem, k - 1)
        return total

    return dp(0, n)
