def find_n_most_expensive_items(dataset, n):
    """
    dataset: list of dicts or tuples, each containing item name and price.
    n: number of most expensive items to return.
    Returns: list of (name, price) tuples of n most expensive items.
    """
    # Assume dataset is a list of dicts with 'name' and 'price' keys
    # or a list of (name, price) tuples
    # Normalize to (name, price) tuples
    items = []
    for item in dataset:
        if isinstance(item, dict):
            items.append((item['name'], item['price']))
        else:
            items.append((item[0], item[1]))
    # Sort by price descending
    items_sorted = sorted(items, key=lambda x: x[1], reverse=True)
    # Select top n
    return items_sorted[:n]
