def bell_number(n):
    # Precompute binomial coefficients using Pascal's triangle
    C = [[0 for _ in range(n+1)] for _ in range(n+1)]
    for i in range(n+1):
        C[i][0] = 1
        for j in range(1, i+1):
            C[i][j] = C[i-1][j-1] + C[i-1][j]

    bell = [0] * (n+1)
    bell[0] = 1

    for i in range(1, n+1):
        bell[i] = 0
        for k in range(i):
            bell[i] += C[i-1][k] * bell[k]
    return bell[n]
