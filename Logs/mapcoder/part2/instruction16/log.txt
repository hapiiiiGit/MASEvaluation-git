from functools import cmp_to_key

def largest_number(digits):
    # Convert all digits to strings
    digits_str = list(map(str, digits))
    
    # Custom comparator
    def compare(a, b):
        if a + b > b + a:
            return -1
        elif a + b < b + a:
            return 1
        else:
            return 0
    
    # Sort using the custom comparator
    digits_str.sort(key=cmp_to_key(compare))
    
    # Concatenate the sorted digits
    result = ''.join(digits_str)
    
    # Handle the case where all digits are zero
    if result[0] == '0':
        return '0'
    return result
