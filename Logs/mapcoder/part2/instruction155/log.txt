def check_digit_frequency(n: int) -> bool:
    s = str(abs(n))
    freq = {}
    for digit in s:
        freq[digit] = freq.get(digit, 0) + 1
    for digit, count in freq.items():
        if count > int(digit):
            return False
    return True
