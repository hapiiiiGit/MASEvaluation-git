import heapq

def k_largest_products(list1, list2, k):
    if not list1 or not list2 or k <= 0:
        return []
    n, m = len(list1), len(list2)
    list1.sort(reverse=True)
    list2.sort(reverse=True)
    heap = []
    visited = set()
    # Initial pair (0,0)
    heapq.heappush(heap, (-(list1[0] * list2[0]), 0, 0))
    visited.add((0, 0))
    result = []
    while heap and len(result) < k:
        neg_prod, i, j = heapq.heappop(heap)
        result.append(-neg_prod)
        if i + 1 < n and (i + 1, j) not in visited:
            heapq.heappush(heap, (-(list1[i + 1] * list2[j]), i + 1, j))
            visited.add((i + 1, j))
        if j + 1 < m and (i, j + 1) not in visited:
            heapq.heappush(heap, (-(list1[i] * list2[j + 1]), i, j + 1))
            visited.add((i, j + 1))
    return result
