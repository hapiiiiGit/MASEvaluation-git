def count_substrings_sum_equals_length(s):
    n = len(s)
    count = 0
    for i in range(n):
        sum_digits = 0
        for j in range(i, n):
            sum_digits += int(s[j])
            length = j - i + 1
            if sum_digits == length:
                count += 1
    return count
