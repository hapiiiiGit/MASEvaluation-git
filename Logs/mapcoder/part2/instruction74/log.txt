def count_common_divisors(a, b):
    import math
    def gcd(x, y):
        while y:
            x, y = y, x % y
        return x

    def count_divisors(n):
        count = 0
        for i in range(1, int(math.sqrt(n)) + 1):
            if n % i == 0:
                count += 1
                if i != n // i:
                    count += 1
        return count

    g = gcd(a, b)

import math

def sum_of_common_divisors(a, b):
    def gcd(x, y):
        while y:
            x, y = y, x % y
        return x

    g = gcd(a, b)
    total = 0
    for i in range(1, int(math.isqrt(g)) + 1):
        if g % i == 0:
            total += i
            if i != g // i:
                total += g // i
    return total
