import math

def is_prime(n):
    if n < 2:
        return False
    if n == 2 or n == 3:
        return True
    if n % 2 == 0:
        return False
    sqrt_n = int(math.isqrt(n))
    for i in range(3, sqrt_n + 1, 2):
        if n % i == 0:
            return False
    return True

def nth_nsw_prime(n):
    count = 0
    s0, s1 = 1, 1
    idx = 1  # S(1)
    while True:
        if is_prime(s1):
            count += 1
            if count == n:
                return s1
        idx += 1
        s2 = 2 * s1 + s0
        s0, s1 = s1, s2
