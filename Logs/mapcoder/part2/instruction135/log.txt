def elementwise_divide(tuple1, tuple2):
    min_len = min(len(tuple1), len(tuple2))
    result = []
    for i in range(min_len):
        if tuple2[i] == 0:
            raise ZeroDivisionError(f"Division by zero at index {i}")
        result.append(tuple1[i] / tuple2[i])
    return tuple(result)
