def max_sum_bitonic_subsequence(arr):
    n = len(arr)
    if n == 0:
        return 0

    inc = arr[:]  # inc[i]: max sum of increasing subsequence ending at i
    dec = arr[:]  # dec[i]: max sum of decreasing subsequence starting at i

    # Fill inc[]
    for i in range(n):
        for j in range(i):
            if arr[j] < arr[i]:
                inc[i] = max(inc[i], inc[j] + arr[i])

    # Fill dec[]
    for i in range(n-1, -1, -1):
        for j in range(n-1, i, -1):
            if arr[j] < arr[i]:
                dec[i] = max(dec[i], dec[j] + arr[i])

    # Find maximum sum of bitonic subsequence
    max_sum = 0
    for i in range(n):
        max_sum = max(max_sum, inc[i] + dec[i] - arr[i])

    return max_sum
