def count_equal_sum_sequences(n):
    # dp[pos][sum1][sum2]: number of ways to fill first 'pos' bits,
    # with sum1 in first n bits, sum2 in last n bits
    dp = [[[0] * (n+1) for _ in range(n+1)] for _ in range(2*n+1)]
    dp[0][0][0] = 1

    for pos in range(2*n):
        for sum1 in range(n+1):
            for sum2 in range(n+1):
                val = dp[pos][sum1][sum2]
                if val == 0:
                    continue
                if pos < n:
                    # First half: add 0 or 1 to sum1
                    dp[pos+1][sum1][sum2] += val
                    if sum1+1 <= n:
                        dp[pos+1][sum1+1][sum2] += val
                else:
                    # Second half: add 0 or 1 to sum2
                    dp[pos+1][sum1][sum2] += val
                    if sum2+1 <= n:
                        dp[pos+1][sum1][sum2+1] += val

    total = 0
    for k in range(n+1):
        total += dp[2*n][k][k]
    return total
