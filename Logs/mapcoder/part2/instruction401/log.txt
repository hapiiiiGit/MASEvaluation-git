def sum_even_length_subarrays(arr):
    n = len(arr)
    total_sum = 0
    for length in range(2, n+1, 2):  # Only even lengths

    max_sum = current_sum = arr[0]

def sumOddLengthSubarrays(arr):
    n = len(arr)
    total = 0
    for i in range(n):
        total_subarrays = (i + 1) * (n - i)
        odd_count = (total_subarrays + 1) // 2
        total += arr[i] * odd_count
    return total
