def power(a, b):
    if b == 0:
        return 1
    if b < 0:
        return 1 / power(a, -b)
    result = 1
    base = a
    exp = b
    while exp > 0:
        if exp % 2 == 1:
            result *= base
        base *= base
        exp //= 2
    return result
