def maxProduct(nums):
    if not nums:
        return 0
    max_product = min_product = result = nums[0]
    for num in nums[1:]:
        temp_max = max_product
        temp_min = min_product
        max_product = max(num, num * temp_max, num * temp_min)
        min_product = min(num, num * temp_max, num * temp_min)
        result = max(result, max_product)
    return result
