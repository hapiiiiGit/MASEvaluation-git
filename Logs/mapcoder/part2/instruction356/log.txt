def sum_of_product_of_all_sublists(arr):
    n = len(arr)
    if n == 0:
        return 0
    total = 0
    prev = 0
    for i in range(n):
        curr = arr[i] + arr[i] * prev
        total += curr
        prev = curr
    return total
