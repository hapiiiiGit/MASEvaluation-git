def remove_first_last_occurrence(s: str, ch: str) -> str:
    first = s.find(ch)
    last = s.rfind(ch)
    if first == -1:
        return s
    if first == last:
        return s[:first] + s[first+1:]
    # Remove last first to not affect index of first
    s = s[:last] + s[last+1:]
    s = s[:first] + s[first+1:]
    return s
