def tuple_intersection(list1, list2):
    # Normalize tuples by sorting their elements
    norm1 = set(tuple(sorted(t)) for t in list1)
    norm2 = set(tuple(sorted(t)) for t in list2)
    # Compute intersection
    result = list(norm1 & norm2)
    return result
