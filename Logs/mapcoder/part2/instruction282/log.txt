def is_sublist(main_list, sub_list):
    n, m = len(main_list), len(sub_list)
    if m == 0:
        return True
    if m > n:
        return False
    for i in range(n - m + 1):
        if main_list[i:i + m] == sub_list:
            return True
    return False
