def max_contiguous_sublist_sum(lst):
    if not lst:
        return 0  # or raise an error if empty list is not allowed
    max_sum = current_sum = lst[0]
    for num in lst[1:]:
        current_sum = max(num, current_sum + num)
        max_sum = max(max_sum, current_sum)
    return max_sum
