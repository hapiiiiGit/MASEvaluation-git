def min_swaps_to_convert(source: str, target: str) -> int:
    if sorted(source) != sorted(target):
        return -1

    source = list(source)
    n = len(source)
    swaps = 0

    for i in range(n):
        if source[i] == target[i]:
            continue
        # Find the nearest index j > i where source[j] == target[i]
        j = i + 1
        while j < n and source[j] != target[i]:
            j += 1
        if j == n:
            return -1  # Should not happen due to previous check
        # Swap adjacent elements to bring source[j] to position i
        while j > i:
            source[j], source[j-1] = source[j-1], source[j]
            swaps += 1
            j -= 1
    return swaps
