def babylonian_sqrt(N, tolerance=1e-10, max_iter=1000):
    if N < 0:
        raise ValueError("Cannot compute square root of negative number")
    if N == 0:
        return 0.0
    x = N / 2 if N >= 2 else 1.0
    for _ in range(max_iter):
        next_x = (x + N / x) / 2
        if abs(next_x - x) < tolerance:
            return next_x
        x = next_x
    return x
