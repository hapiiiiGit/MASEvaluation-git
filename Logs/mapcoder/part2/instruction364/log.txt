import math

def vector_subtract(a, b):
    return [a[i] - b[i] for i in range(3)]

def cross_product(u, v):
    return [
        u[1]*v[2] - u[2]*v[1],
        u[2]*v[0] - u[0]*v[2],
        u[0]*v[1] - u[1]*v[0]
    ]

def magnitude(v):
    return math.sqrt(sum(x*x for x in v))

def triangle_area(a, b, c):
    ab = vector_subtract(b, a)
    ac = vector_subtract(c, a)
    cross = cross_product(ab, ac)
    return 0.5 * magnitude(cross)

def tetrahedron_surface_area(vertices):
    # vertices: list of four 3D points [A, B, C, D]
    A, B, C, D = vertices
    faces = [
        (A, B, C),
        (A, B, D),
        (A, C, D),
        (B, C, D)
    ]
    area = 0.0
    for face in faces:
        area += triangle_area(*face)
    return area
