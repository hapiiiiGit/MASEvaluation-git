def toggle_bits_except_first_last(n):
    # Find the number of bits in n
    if n == 0:
        return 0
    num_bits = n.bit_length()
    if num_bits < 3:
        return n
    # Create mask: 1s except first and last bit
    mask = 0
    for i in range(1, num_bits - 1):
        mask |= (1 << i)
    # Toggle bits using XOR
    return n ^ mask
