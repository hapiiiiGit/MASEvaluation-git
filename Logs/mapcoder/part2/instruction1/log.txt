import math

def identify_non_primes(numbers):
    non_primes = []
    for n in numbers:
        if n < 2:
            non_primes.append(n)
            continue
        is_prime = True
        for i in range(2, int(math.isqrt(n)) + 1):
            if n % i == 0:
                is_prime = False
                break
        if not is_prime:
            non_primes.append(n)
    return non_primes
