def is_monotonic(arr):
    if not arr or len(arr) == 1:
        return True
    is_increasing = True
    is_decreasing = True
    for i in range(len(arr) - 1):
        if arr[i] < arr[i + 1]:
            is_decreasing = False
        if arr[i] > arr[i + 1]:
            is_increasing = False
    return is_increasing or is_decreasing
