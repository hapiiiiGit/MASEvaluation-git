import math

def find_perfect_squares(low, high):
    result = []
    start = math.ceil(math.sqrt(low))
    end = math.floor(math.sqrt(high))
    for i in range(start, end + 1):
        square = i * i
        if low <= square <= high:
            result.append(square)
    return result
