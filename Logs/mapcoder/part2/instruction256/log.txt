def contains_consecutive_numbers(lst):
    if not lst:
        return False
    unique_numbers = set(lst)
    min_num = min(unique_numbers)
    max_num = max(unique_numbers)
    expected_length = max_num - min_num + 1
    return len(unique_numbers) == expected_length
