def solve_linear_diophantine(a, b, n):
    def extended_gcd(a, b):
        if b == 0:
            return a, 1, 0
        g, x1, y1 = extended_gcd(b, a % b)
        x = y1
        y = x1 - (a // b) * y1
        return g, x, y

    g, x0, y0 = extended_gcd(a, b)
    if n % g != 0:
        return None
    factor = n // g
    x = x0 * factor
    y = y0 * factor
    return (x, y)
