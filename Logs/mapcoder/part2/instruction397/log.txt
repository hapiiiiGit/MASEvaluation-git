def pack_consecutive_duplicates(lst):
    if not lst:
        return []
    result = []
    current_group = [lst[0]]
    for elem in lst[1:]:
        if elem == current_group[-1]:
            current_group.append(elem)
        else:
            result.append(current_group)
            current_group = [elem]
    result.append(current_group)
