import math

def largest_inscribed_triangle_area(r):
    # The largest triangle inscribed in a semicircle is a right triangle
    # with its hypotenuse as the diameter of the semicircle.
    # The area is (1/2) * r * r = r^2 / 2
    return r * r / 2
