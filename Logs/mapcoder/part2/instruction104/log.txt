def is_majority_element(arr, n, target):
    # Binary search for first occurrence
    left, right = 0, n - 1
    first = -1
    while left <= right:
        mid = (left + right) // 2
        if arr[mid] == target:
            first = mid
            right = mid - 1
        elif arr[mid] < target:
            left = mid + 1
        else:
            right = mid - 1
    if first == -1:
        return False
    if first + n // 2 < n and arr[first + n // 2] == target:
        return True
    return False
