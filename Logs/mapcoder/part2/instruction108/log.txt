def are_all_bits_unset(num: int, left: int, right: int) -> bool:
    """
    Checks if all bits in num from position left to right (inclusive, 0-based from LSB) are unset (0).
    :param num: The integer to check.
    :param left: The starting bit position (inclusive).
    :param right: The ending bit position (inclusive).
    :return: True if all bits in the range are unset, False otherwise.
    """
    mask = ((1 << (right - left + 1)) - 1) << left
    return (num & mask) == 0
