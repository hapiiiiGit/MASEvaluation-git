import math

def find_third_side(side1, side2, sides_given):
    """
    Find the third side of a right-angled triangle.

    Parameters:
    - side1: float, first known side
    - side2: float, second known side
    - sides_given: str, either 'legs' (both legs given) or 'leg_hypotenuse' (one leg and hypotenuse given)

    Returns:
    - float: the length of the third side, or None if invalid input
    """
    if sides_given == 'legs':
        # Both legs are given, compute hypotenuse
        if side1 <= 0 or side2 <= 0:
            return None
        hypotenuse = math.sqrt(side1 ** 2 + side2 ** 2)
        return hypotenuse
    elif sides_given == 'leg_hypotenuse':
        # One leg and hypotenuse are given, compute other leg
        leg = min(side1, side2)
        hypotenuse = max(side1, side2)
        if leg <= 0 or hypotenuse <= 0 or hypotenuse <= leg:
            return None
        diff = hypotenuse ** 2 - leg ** 2
        if diff < 0:
            return None
        other_leg = math.sqrt(diff)
        return other_leg
    else:
        # Invalid specification
        return None
