def sum_perrin_numbers(n):
    if n < 0:
        return 0
    # Base cases
    perrin = [3, 0, 2]
    if n == 0:
        return 3
    if n == 1:
        return 3 + 0
    if n == 2:
        return 3 + 0 + 2

    sum_perrin = 3 + 0 + 2
    for i in range(3, n + 1):
        next_perrin = perrin[1] + perrin[0]
        sum_perrin += next_perrin
        perrin[0], perrin[1], perrin[2] = perrin[1], perrin[2], next_perrin
    return sum_perrin
