import math
from typing import List, Tuple

def complex_to_polar(complex_numbers: List[complex]) -> List[Tuple[float, float]]:
    polar_coords = []
    for z in complex_numbers:
        a = z.real
        b = z.imag
        r = abs(z)
        theta = math.atan2(b, a)
        polar_coords.append((r, theta))
    return polar_coords
