def find_common_first_elements(tuples_list):
    count_dict = {}
    for tup in tuples_list:
        first_elem = tup[0]
        count_dict[first_elem] = count_dict.get(first_elem, 0) + 1
    common_elements = [elem for elem, count in count_dict.items() if count > 1]
    return common_elements
