def tuple_str_to_int_tuple(s):
    s = s.strip()
    if s.startswith('(') and s.endswith(')'):
        s = s[1:-1].strip()
    if not s:
        return ()
    items = [item.strip() for item in s.split(',') if item.strip()]
    return tuple(int(item) for item in items)
