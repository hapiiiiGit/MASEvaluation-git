def is_second_tuple_smaller(first_tuple, second_tuple):
    if len(first_tuple) != len(second_tuple):
        raise ValueError("Tuples must be of equal length.")
    for a, b in zip(first_tuple, second_tuple):
        if b >= a:
            return False
    return True
