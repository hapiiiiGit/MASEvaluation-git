def find_kth_element(arr1, arr2, k):
    def kth(A, B, k):
        lenA, lenB = len(A), len(B)
        # Ensure A is the shorter array
        if lenA > lenB:
            return kth(B, A, k)
        if lenA == 0:
            return B[k-1]
        if k == 1:
            return min(A[0], B[0])
        i = min(lenA, k // 2)
        j = k - i
        if A[i-1] < B[j-1]:
            return kth(A[i:], B, k - i)
        else:
            return kth(A, B[j:], k - j)
    return kth(arr1, arr2, k)
