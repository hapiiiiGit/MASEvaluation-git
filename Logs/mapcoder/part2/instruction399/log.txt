def combinations_of_sum_with_tuples(tuple_list, target):
    result = []

    def backtrack(start, path, curr_sum):
        if curr_sum == target:
            result.append(path[:])
            return
        if curr_sum > target:
            return
        for i in range(start, len(tuple_list)):
            t = tuple_list[i]
            t_sum = sum(t)
            path.append(t)
            backtrack(i + 1, path, curr_sum + t_sum)
            path.pop()

    backtrack(0, [], 0)
    return result
