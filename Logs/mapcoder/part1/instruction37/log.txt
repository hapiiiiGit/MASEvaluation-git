def sort_even(l: list):
    even_indices = [l[i] for i in range(0, len(l), 2)]
    even_indices.sort()
    
    result = l[:]
    for i, idx in enumerate(range(0, len(l), 2)):
        result[idx] = even_indices[i]
    
    return result
