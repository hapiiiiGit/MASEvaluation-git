def parse_nested_parens(paren_string: str) -> List[int]:
    def find_deepest_level(s: str, level: int) -> int:
        max_level = level
        current_level = level
        for char in s:
            if char == '(':
                current_level += 1
                max_level = max(max_level, current_level)
            elif char == ')':
                current_level -= 1
        return max_level

    result = []
    groups = paren_string.split()
    for group in groups:
        result.append(find_deepest_level(group, 0))
    
    return result
