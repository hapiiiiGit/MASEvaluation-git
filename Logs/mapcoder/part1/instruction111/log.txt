def histogram(test):
    freq = {}
    max_freq = 0
    for char in test.split():
        freq[char] = freq.get(char, 0) + 1
        max_freq = max(max_freq, freq[char])
    
    result = {char: count for char, count in freq.items() if count == max_freq}
    return result
