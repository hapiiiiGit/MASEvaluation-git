def car_race_collision(n: int):
    collisions = 0
    left_cars = [i for i in range(1, n+1)]
    right_cars = [i for i in range(n+1, 2*n+1)]
    
    left_cars.sort()
    right_cars.sort(reverse=True)
    
    left_ptr = 0
    right_ptr = 0
    
    while left_ptr < n and right_ptr < n:
        if left_cars[left_ptr] < right_cars[right_ptr]:
            collisions += 1
            left_ptr += 1
        else:
            right_ptr += 1
    
    return collisions
