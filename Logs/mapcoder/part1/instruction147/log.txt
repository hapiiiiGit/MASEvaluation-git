def get_max_triples(n):
    a = [i * i - i + 1 for i in range(1, n + 1)]
    count = 0
    remainder_count = [0] * 3
    
    for num in a:
        remainder_count[num % 3] += 1
    
    # Count valid triples
    count += remainder_count[0] * (remainder_count[0] - 1) * (remainder_count[0] - 2) // 6  # all three from remainder 0
    count += remainder_count[1] * remainder_count[1] * remainder_count[0] // 2  # two from remainder 1, one from remainder 0
    count += remainder_count[2] * remainder_count[2] * remainder_count[0] // 2  # two from remainder 2, one from remainder 0
    count += remainder_count[1] * remainder_count[2] * remainder_count[1]  # one from remainder 1, one from remainder 2, one from remainder 1
    
    return count
