def correct_bracketing(brackets: str):
    stack = []
    mapping = {')': '('}
    
    for bracket in brackets:
        if bracket == '(':
            stack.append(bracket)
        elif bracket == ')':
            if not stack or stack[-1] != '(':
                return False
            stack.pop()
    
    return len(stack) == 0
