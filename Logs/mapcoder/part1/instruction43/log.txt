def pairs_sum_to_zero(l):
    l = sorted(l)
    left, right = 0, len(l) - 1
    
    while left < right:
        current_sum = l[left] + l[right]
        if current_sum == 0:
            return True
        elif current_sum > 0:
            right -= 1
        else:
            left += 1
            
    return False
