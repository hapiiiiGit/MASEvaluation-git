def longest_increasing_subsequence(nums: List[int]) -> int:
    if not nums:
        return 0
    
    dp = [1] * len(nums)
    
    for i in range(1, len(nums)):
        for j in range(i):
            if nums[i] > nums[j]:
                dp[i] = max(dp[i], dp[j] + 1)

def min_jumps(nums: List[int]) -> int:
    if len(nums) <= 1:
        return 0

    steps = nums[0]
    
    for i in range(1, len(nums)):
