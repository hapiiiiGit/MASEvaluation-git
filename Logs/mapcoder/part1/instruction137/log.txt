def compare_one(a, b):
    if isinstance(a, str):
        a = a.replace(',', '.')
        a = float(a)
    if isinstance(b, str):
        b = b.replace(',', '.')
        b = float(b)
    
    if a == b:
        return None
    elif a > b:
        return a
    else:
        return b
