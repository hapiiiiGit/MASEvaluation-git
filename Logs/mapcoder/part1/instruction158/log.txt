def find_max(words):
    def unique_count(word):
        return len(set(word))
    
    max_word = ""
    max_unique = -1
    
    for word in words:
        current_unique = unique_count(word)
        if (current_unique > max_unique) or (current_unique == max_unique and word < max_word):
            max_unique = current_unique
            max_word = word
            
    return max_word
