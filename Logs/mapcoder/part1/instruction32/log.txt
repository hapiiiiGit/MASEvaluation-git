import math

def poly(xs: list, x: float):
    return sum([coeff * math.pow(x, i) for i, coeff in enumerate(xs)])

def find_zero(xs: list):
    x = 1.0
    while True:
        fx = poly(xs, x)
        if abs(fx) < 0.0001:
            return x
        derivative = sum([i * coeff * math.pow(x, i-1) for i, coeff in enumerate(xs) if i > 0])
        x = x - fx / derivative
