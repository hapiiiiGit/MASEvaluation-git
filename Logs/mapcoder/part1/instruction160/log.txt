def do_algebra(operator, operand):
    if len(operator) == 0:
        return operand[0]
    
    precedence = {'**': 3, '*': 2, '//': 2, '+': 1, '-': 1}
    
    def evaluate_expression(operands, operators):
        if len(operators) == 0:
            return operands[0]
        
        min_precedence = min(operators, key=lambda x: precedence[x])
        idx = operators.index(min_precedence)
        
        left = evaluate_expression(operands[:idx], operators[:idx])
        right = evaluate_expression(operands[idx+1:], operators[idx+1:])
        
        if min_precedence == '+':
            return left + right
        elif min_precedence == '-':
            return left - right
        elif min_precedence == '*':
            return left * right
        elif min_precedence == '//':
            return left // right
        elif min_precedence == '**':
            return left ** right
