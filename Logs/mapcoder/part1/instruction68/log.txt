def longest_increasing_subsequence(arr):
    n = len(arr)
    lis_lengths = [1] * n

    for i in range(1, n):
        for j in range(i):
            if arr[i] > arr[j]:
                lis_lengths[i] = max(lis_lengths[i], lis_lengths[j] + 1)

def pluck(arr):
    smallest_even = float('inf')
    smallest_index = float('inf')
    
    for i in range(len(arr)):
        if arr[i] % 2 == 0 and arr[i] < smallest_even:
            smallest_even = arr[i]
            smallest_index = i
        elif arr[i] % 2 == 0 and arr[i] == smallest_even:
            smallest_index = min(smallest_index, i)
    
    if smallest_even == float('inf'):
        return []
    
    return [smallest_even, smallest_index]
