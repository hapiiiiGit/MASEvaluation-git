def match_parens(lst):
    stack = []
    for s in lst:
        for char in s:
            if char == '(':
                stack.append(char)
            else:
                if not stack or stack[-1] != '(':
                    return 'No'
                stack.pop()
    return 'Yes' if not stack else 'No'
