def specialFilter(nums):
    def is_odd_digit(num):
        num_str = str(abs(num))
        return int(num_str[0]) % 2 != 0 and int(num_str[-1]) % 2 != 0

    count = 0
    for num in nums:
        if num > 10 and is_odd_digit(num):
            count += 1

    return count
