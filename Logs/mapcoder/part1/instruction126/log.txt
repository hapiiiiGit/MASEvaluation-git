def is_sorted(lst):
    seen = {}
    for i in range(len(lst) - 1):
        if lst[i] > lst[i + 1]:
            return False
        if lst[i] in seen:
            seen[lst[i]] += 1
            if seen[lst[i]] > 1:
                return False
        else:
            seen[lst[i]] = 1
    if lst[-1] in seen:
        seen[lst[-1]] += 1
        if seen[lst[-1]] > 1:
            return False
    return True
