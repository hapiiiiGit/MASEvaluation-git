def chatbot_api(request):
    if request.method == 'POST':
        import json
        data = json.loads(request.body)
        message = data.get('message', '').lower()
        session_id = data.get('session_id', str(uuid.uuid4()))
        session, _ = ChatbotSession.objects.get_or_create(user=request.user, session_id=session_id)
        # Simple intent recognition
        if 'event' in message:
            response = "Here are some upcoming events: " + ", ".join(Event.objects.values_list('title', flat=True)[:5])
        elif 'ticket' in message:
            response = "You can purchase tickets from the event page."
        elif 'profile' in message:
            response = f"Your profile: {request.user.first_name} {request.user.last_name}, role: {request.user.role}"
        else:
            response = "I'm here to help! Ask me about events, tickets, or your profile."
        session.last_interaction = datetime.datetime.now()
        session.save()
        return JsonResponse({'response': response, 'session_id': session_id})
    return JsonResponse({'error': 'Invalid request'}, status=400)
