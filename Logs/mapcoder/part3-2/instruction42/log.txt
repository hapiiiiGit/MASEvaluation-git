METRICS_INTERVAL = 30  # seconds

# --- Logging Setup ---
logging.basicConfig(
    level=logging.INFO,
    format='%(asctime)s %(levelname)s %(threadName)s %(message)s',
    handlers=[
        logging.StreamHandler(),
        logging.FileHandler("etl_pipeline.log")
    ]
)
logger = logging.getLogger("ETL")

# --- Metrics ---
metrics = {
    'extracted': 0,
    'transformed': 0,
    'loaded': 0,
    'errors': 0,
    'duplicates': 0
}
metrics_lock = threading.Lock()

def log_metrics():
    while True:
        with metrics_lock:
            logger.info(f"METRICS: {metrics}")
        time.sleep(METRICS_INTERVAL)

# --- Extract ---
def extract_stream(data_queue, stop_event):
    session = requests.Session()
    headers = {"Authorization": "Bearer <YOUR_API_KEY>"}
    while not stop_event.is_set():
        try:
            with session.get(DATA_SOURCE_URL, headers=headers, stream=True, timeout=60) as resp:
                resp.raise_for_status()
                for line in resp.iter_lines():
                    if stop_event.is_set():
                        break
                    if line:
                        try:
                            data = parse_conversation_line(line)
                            data_queue.put(data)
                            with metrics_lock:
                                metrics['extracted'] += 1
                        except Exception as e:
                            logger.error(f"Extract parse error: {e}")
                            with metrics_lock:
                                metrics['errors'] += 1
        except Exception as e:
            logger.error(f"Extract connection error: {e}")
            with metrics_lock:
                metrics['errors'] += 1
            time.sleep(5)

def parse_conversation_line(line):
    import json
    obj = json.loads(line.decode('utf-8'))
    # Minimal schema validation
    assert 'id' in obj and 'timestamp' in obj and 'content' in obj
    return obj

# --- Transform ---
def transform_worker(data_queue, transformed_queue, stop_event):
    seen_ids = set()
    while not stop_event.is_set():
        try:
            data = data_queue.get(timeout=1)
        except queue.Empty:
            continue
        try:
            # Deduplication (idempotency)
            if data['id'] in seen_ids:
                with metrics_lock:
                    metrics['duplicates'] += 1
                continue
            seen_ids.add(data['id'])
            # Validation
            if not validate_conversation(data):
                logger.warning(f"Transform: Invalid data {data}")
                with metrics_lock:
                    metrics['errors'] += 1
                continue
            # Enrichment (add processing timestamp)
            data['etl_processed_at'] = int(time.time())
            transformed_queue.put(data)
            with metrics_lock:
                metrics['transformed'] += 1
        except Exception as e:
            logger.error(f"Transform error: {e}")
            with metrics_lock:
                metrics['errors'] += 1

def validate_conversation(data):
    # Example: filter out empty content or future timestamps
    if not data['content'] or data['timestamp'] > int(time.time()) + 60:
        return False
    return True

# --- Load ---
def load_worker(transformed_queue, stop_event):
    buffer = []
    conn = None
    retry_count = 0
    while not stop_event.is_set():
        try:
            if conn is None or conn.closed:
                conn = connect_questdb_with_retry()
            try:
                data = transformed_queue.get(timeout=1)
                buffer.append(data)
            except queue.Empty:
                pass
            if len(buffer) >= BATCH_SIZE or (buffer and stop_event.is_set()):
                insert_batch(conn, buffer)
                with metrics_lock:
                    metrics['loaded'] += len(buffer)
                buffer.clear()
        except Exception as e:
            logger.error(f"Load error: {e}")
            with metrics_lock:
                metrics['errors'] += 1
            retry_count += 1
            if retry_count > RETRY_LIMIT:
                logger.critical("Load: Retry limit exceeded, aborting.")
                stop_event.set()
            else:
                time.sleep(RETRY_BACKOFF_BASE ** retry_count)
                conn = None

def connect_questdb_with_retry():
    for attempt in range(RETRY_LIMIT):
        try:
            conn = psycopg2.connect(**QUESTDB_CONN_INFO)
            conn.autocommit = True
            logger.info("Connected to QuestDB.")
            return conn
        except Exception as e:
            logger.error(f"QuestDB connect error: {e}")
            time.sleep(RETRY_BACKOFF_BASE ** attempt)
    raise Exception("QuestDB connection failed after retries.")

def insert_batch(conn, batch):
    if not batch:
        return
    # Idempotent upsert using ON CONFLICT
    sql = """
    INSERT INTO claude_conversations (id, timestamp, content, etl_processed_at)
    VALUES %s
    ON CONFLICT (id) DO NOTHING;
    """
    values = [
        (d['id'], d['timestamp'], d['content'], d['etl_processed_at'])
        for d in batch
    ]
    with conn.cursor() as cur:
        execute_values(cur, sql, values)

# --- Health Check ---
def health_check(stop_event):
    while not stop_event.is_set():
        try:
            conn = psycopg2.connect(**QUESTDB_CONN_INFO)
            conn.close()
            logger.info("Health check: QuestDB OK")
        except Exception as e:
            logger.error(f"Health check failed: {e}")
            with metrics_lock:
                metrics['errors'] += 1
        time.sleep(60)

# --- Main ---
def main():
    data_queue = queue.Queue(maxsize=1000)
    transformed_queue = queue.Queue(maxsize=1000)
    stop_event = threading.Event()

    threads = [
        threading.Thread(target=extract_stream, args=(data_queue, stop_event), name="Extract"),
        threading.Thread(target=transform_worker, args=(data_queue, transformed_queue, stop_event), name="Transform"),
        threading.Thread(target=load_worker, args=(transformed_queue, stop_event), name="Load"),
        threading.Thread(target=log_metrics, name="Metrics"),
        threading.Thread(target=health_check, args=(stop_event,), name="HealthCheck")
    ]

    for t in threads:
        t.daemon = True
        t.start()

    try:
        while True:
            time.sleep(1)
    except KeyboardInterrupt:
        logger.info("Shutdown signal received.")
        stop_event.set()
        time.sleep(5)

if __name__ == "__main__":
    main()
