def hash_password(password):
    return hashlib.sha256(password.encode()).hexdigest()

def generate_self_signed_cert(certfile, keyfile):
    # Generate self-signed cert if not exists
    if not os.path.exists(certfile) or not os.path.exists(keyfile):
        from OpenSSL import crypto
        k = crypto.PKey()
        k.generate_key(crypto.TYPE_RSA, 2048)
        cert = crypto.X509()
        cert.get_subject().C = "US"
        cert.get_subject().ST = "State"
        cert.get_subject().L = "Locality"
        cert.get_subject().O = "Organization"
        cert.get_subject().OU = "OrgUnit"
        cert.get_subject().CN = "localhost"
        cert.set_serial_number(int(time.time()))
        cert.gmtime_adj_notBefore(0)
        cert.gmtime_adj_notAfter(10*365*24*60*60)
        cert.set_issuer(cert.get_subject())
        cert.set_pubkey(k)
        cert.sign(k, 'sha256')
        with open(certfile, "wb") as f:
            f.write(crypto.dump_certificate(crypto.FILETYPE_PEM, cert))
        with open(keyfile, "wb") as f:
            f.write(crypto.dump_privatekey(crypto.FILETYPE_PEM, k))

class RemoteServer:
    def __init__(self, host=SERVER_HOST, port=SERVER_PORT):
        self.host = host
        self.port = port
        self.clients = []
        self.running = False

    def start(self):
        generate_self_signed_cert(CERTFILE, KEYFILE)
        context = ssl.create_default_context(ssl.Purpose.CLIENT_AUTH)
        context.load_cert_chain(certfile=CERTFILE, keyfile=KEYFILE)
        bindsocket = socket.socket(socket.AF_INET, socket.SOCK_STREAM)
        bindsocket.setsockopt(socket.SOL_SOCKET, socket.SO_REUSEADDR, 1)
        bindsocket.bind((self.host, self.port))
        bindsocket.listen(5)
        self.running = True
        print(f"Server listening on {self.host}:{self.port}")
        threading.Thread(target=self.accept_clients, args=(bindsocket, context), daemon=True).start()

    def accept_clients(self, bindsocket, context):
        while self.running:
            try:
                newsocket, fromaddr = bindsocket.accept()
                conn = context.wrap_socket(newsocket, server_side=True)
                print(f"Connection from {fromaddr}")
                threading.Thread(target=self.handle_client, args=(conn,), daemon=True).start()
            except Exception as e:
                print("Accept error:", e)

    def handle_client(self, conn):

class RemoteClient(BoxLayout):
    def __init__(self, **kwargs):
        super().__init__(orientation='vertical', **kwargs)
        self.conn = None
        self.ssl_context = ssl.create_default_context()
        self.ssl_context.check_hostname = False

        self.screenshot_img = Image()
        self.add_widget(Label(text="Remote Host:"))
        self.host_input = TextInput(text='127.0.0.1', multiline=False)
        self.add_widget(self.host_input)
        self.add_widget(Label(text="Port:"))
        self.port_input = TextInput(text=str(SERVER_PORT), multiline=False)
        self.add_widget(self.port_input)
        self.add_widget(Label(text="Password:"))
        self.pw_input = TextInput(password=True, multiline=False)
        self.add_widget(self.pw_input)
        self.connect_btn = Button(text="Connect")
        self.connect_btn.bind(on_press=self.connect)
        self.add_widget(self.connect_btn)
        self.add_widget(self.screenshot_img)
        self.refresh_event = None

    def connect(self, instance):
        host = self.host_input.text.strip()
        port = int(self.port_input.text.strip())
        password = self.pw_input.text.strip()
        try:
            sock = socket.socket(socket.AF_INET, socket.SOCK_STREAM)
            self.conn = self.ssl_context.wrap_socket(sock)
            self.conn.connect((host, port))
            self.conn.sendall(pickle.dumps({'password': password}))
            resp = pickle.loads(self.conn.recv(4096))
            if resp.get('status') != 'ok':
                self.screenshot_img.source = ''
                self.screenshot_img.texture = None
                self.screenshot_img.reload()
                self.screenshot_img.add_widget(Label(text="Auth failed"))
                self.conn.close()
                self.conn = None
                return
            self.connected = True
            self.connect_btn.text = "Connected"
            self.connect_btn.disabled = True
            self.refresh_event = Clock.schedule_interval(self.get_screenshot, 1/2)
        except Exception as e:
            self.screenshot_img.add_widget(Label(text="Connection failed"))
            self.conn = None

    def get_screenshot(self, dt):
        if not self.conn:
            return
        try:
            self.conn.sendall(pickle.dumps({'cmd': 'screenshot'}))
            resp = pickle.loads(self.conn.recv(2**20))
            if resp.get('status') == 'ok':
                img_bytes = base64.b64decode(resp['img'])
                buf = io.BytesIO(img_bytes)
                from PIL import Image as PILImage
                pil_img = PILImage.open(buf)
                pil_img = pil_img.convert('RGB')
                w, h = pil_img.size
                tex = Texture.create(size=(w, h))
                tex.blit_buffer(pil_img.tobytes(), colorfmt='rgb', bufferfmt='ubyte')
                tex.flip_vertical()
                self.screenshot_img.texture = tex
        except Exception as e:
            if self.refresh_event:
                self.refresh_event.cancel()
            self.connected = False
            self.connect_btn.text = "Connect"
            self.connect_btn.disabled = False
            self.conn = None
