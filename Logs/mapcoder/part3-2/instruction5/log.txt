# Helper function to get a font
def get_font(size, bold=False):
    # Use DejaVuSans as a modern, clear font (bundled with Pillow)
    try:
        if bold:
            return ImageFont.truetype("DejaVuSans-Bold.ttf", size)
        else:
            return ImageFont.truetype("DejaVuSans.ttf", size)
    except:
        return ImageFont.load_default()

# Python color palette
PY_BLUE = (52, 102, 164)    # #3466A4
PY_YELLOW = (255, 213, 60)  # #FFD53C
PY_GREEN = (53, 114, 71)    # #357247
PY_WHITE = (255, 255, 255)
PY_DARK = (34, 40, 49)

# 1. Create Logo (500x500)
def create_logo(filename="logo.png"):
    size = 500
    img = Image.new("RGBA", (size, size), PY_WHITE)
    draw = ImageDraw.Draw(img)

    # Draw a stylized Python "snake" using two interlocking circles and a code bracket
    # Left circle (blue)
    draw.ellipse([70, 120, 270, 320], fill=PY_BLUE, outline=PY_DARK, width=6)
    # Right circle (yellow)
    draw.ellipse([230, 180, 430, 380], fill=PY_YELLOW, outline=PY_DARK, width=6)

    # Draw a code bracket (simulate with lines)
    bracket_color = PY_DARK
    # Left bracket
    draw.line([(60, 100), (60, 400)], fill=bracket_color, width=12)
    draw.arc([40, 90, 100, 150], 180, 270, fill=bracket_color, width=12)
    draw.arc([40, 350, 100, 410], 90, 180, fill=bracket_color, width=12)
    # Right bracket
    draw.line([(440, 100), (440, 400)], fill=bracket_color, width=12)
    draw.arc([400, 90, 460, 150], 270, 360, fill=bracket_color, width=12)
    draw.arc([400, 350, 460, 410], 0, 90, fill=bracket_color, width=12)

    # Add "PyScrape" text
    font = get_font(60, bold=True)
    text = "PyScrape"
    w, h = draw.textsize(text, font=font)
    draw.text(((size-w)//2, 410), text, font=font, fill=PY_DARK)

    img.save(filename)

# 2. Create Cover Image (1200x628)
def create_cover(filename="cover.png"):
    W, H = 1200, 628
    img = Image.new("RGBA", (W, H), PY_WHITE)
    draw = ImageDraw.Draw(img)

    # Background: subtle blue gradient
    for y in range(H):
        r = int(PY_BLUE[0] + (PY_WHITE[0]-PY_BLUE[0]) * y/H)
        g = int(PY_BLUE[1] + (PY_WHITE[1]-PY_BLUE[1]) * y/H)
        b = int(PY_BLUE[2] + (PY_WHITE[2]-PY_BLUE[2]) * y/H)
        draw.line([(0, y), (W, y)], fill=(r, g, b))

    # Place logo at left
    logo_size = 180
    logo_img = Image.open("logo.png").resize((logo_size, logo_size), Image.LANCZOS)
    img.paste(logo_img, (60, (H-logo_size)//2), logo_img)

    # Tagline (Promise)
    tagline = "Automate Your Data. Unlock Insights."
    font_tag = get_font(54, bold=True)
    tw, th = draw.textsize(tagline, font=font_tag)
    draw.text((270, 120), tagline, font=font_tag, fill=PY_DARK)

    # Subtext
    subtext = "Professional Web Scraping Services with Python"
    font_sub = get_font(32)
    sw, sh = draw.textsize(subtext, font=font_sub)
    draw.text((270, 190), subtext, font=font_sub, fill=PY_DARK)

    # Proof element: sample data table (drawn)
    table_x, table_y = 270, 270
    cell_w, cell_h = 120, 44
    cols = ["Site", "Data", "Result"]
    data = [
        ["example.com", "Prices", "$12.99"],
        ["news.com", "Headlines", "Top 5"],
        ["shop.com", "Stock", "In/Out"],
    ]
    font_table = get_font(26, bold=True)
    font_cell = get_font(24)

    # Draw table header
    for i, col in enumerate(cols):
        x0 = table_x + i*cell_w
        draw.rectangle([x0, table_y, x0+cell_w, table_y+cell_h], fill=PY_YELLOW, outline=PY_DARK, width=2)
        cw, ch = draw.textsize(col, font=font_table)
        draw.text((x0 + (cell_w-cw)//2, table_y + (cell_h-ch)//2), col, font=font_table, fill=PY_DARK)
    # Draw table rows
    for r, row in enumerate(data):
        for c, val in enumerate(row):
            x0 = table_x + c*cell_w
            y0 = table_y + (r+1)*cell_h
            draw.rectangle([x0, y0, x0+cell_w, y0+cell_h], fill=PY_WHITE, outline=PY_DARK, width=2)
            vw, vh = draw.textsize(val, font=font_cell)
            draw.text((x0 + (cell_w-vw)//2, y0 + (cell_h-vh)//2), val, font=font_cell, fill=PY_DARK)

    # Testimonial badge (proof)
    badge_x, badge_y = 900, 80
    badge_r = 70
    draw.ellipse([badge_x, badge_y, badge_x+badge_r*2, badge_y+badge_r*2], fill=PY_GREEN, outline=PY_DARK, width=4)
    font_badge = get_font(28, bold=True)
    badge_text = "100+\nClients"
    for i, line in enumerate(badge_text.split('\n')):
        lw, lh = draw.textsize(line, font=font_badge)
        draw.text((badge_x+badge_r - lw//2, badge_y+30+i*32), line, font=font_badge, fill=PY_WHITE)

    # Testimonial text
    testi = '"Fast, reliable, and accurate data every time!"'
    font_testi = get_font(28)
    draw.text((900, 180), testi, font=font_testi, fill=PY_DARK)

    img.save(filename)
