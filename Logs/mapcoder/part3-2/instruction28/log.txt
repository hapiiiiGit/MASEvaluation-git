# 4. Controller to handle signup and consent storage
class CustomerPortal(http.Controller):

    @http.route('/customer/signup', type='http', auth='public', website=True, csrf=True)
    def customer_signup(self, **post):

        tnc_accepted = post.get('tnc_accepted')
