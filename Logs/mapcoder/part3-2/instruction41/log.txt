# Step 1: Import libraries
import cv2
from ultralytics import YOLO
from deep_sort_realtime.deepsort_tracker import DeepSort
import numpy as np
import time

# Step 2: Load YOLOv8 model and DeepSORT tracker
model = YOLO('yolov8n.pt')
tracker = DeepSort(max_age=30)

# Step 3: Define video capture and virtual line
cap = cv2.VideoCapture('highway.mp4')
line_position = 400
vehicle_count = 0
track_times = {}

while True:
    ret, frame = cap.read()
    if not ret:
        break
    results = model(frame)
    detections = []
    for result in results:
        for box in result.boxes:
            x1, y1, x2, y2 = box.xyxy[0]
            conf = box.conf[0]
            cls = int(box.cls[0])
            if cls in [2, 3, 5, 7]:  # car, motorcycle, bus, truck
                detections.append(([x1, y1, x2-x1, y2-y1], conf, 'vehicle'))
    tracks = tracker.update_tracks(detections, frame=frame)
    for track in tracks:
        if not track.is_confirmed():
            continue
        track_id = track.track_id
        ltrb = track.to_ltrb()
        x1, y1, x2, y2 = map(int, ltrb)
        centroid = ((x1 + x2)//2, (y1 + y2)//2)
        if track_id not in track_times and centroid[1] > line_position:
            track_times[track_id] = time.time()
            vehicle_count += 1
        cv2.rectangle(frame, (x1, y1), (x2, y2), (0,255,0), 2)
        cv2.putText(frame, f'ID:{track_id}', (x1, y1-10), cv2.FONT_HERSHEY_SIMPLEX, 0.6, (255,255,255), 2)
    cv2.line(frame, (0, line_position), (frame.shape[1], line_position), (0,0,255), 2)
    cv2.imshow('Vehicle Counting', frame)
    if cv2.waitKey(1) & 0xFF == ord('q'):
        break

cap.release()
cv2.destroyAllWindows()
print(f"Total vehicles counted: {vehicle_count}")

import cv2

import numpy as np

# Step 3: Start video capture
cap = cv2.VideoCapture(0)
DIST_THRESHOLD = 100  # pixels, for demonstration

while True:
    ret, frame = cap.read()
    if not ret:
        break
    results = model(frame)
    person_boxes = []
    for result in results:
        for box in result.boxes:
            cls = int(box.cls[0])
            if cls == 0:  # person class
                x1, y1, x2, y2 = map(int, box.xyxy[0])

    # Compute pairwise distances
    for i in range(len(person_boxes)):
        for j in range(i+1, len(person_boxes)):
            dist = np.linalg.norm(np.array(person_boxes[i]) - np.array(person_boxes[j]))
            if dist < DIST_THRESHOLD:
                cv2.line(frame, person_boxes[i], person_boxes[j], (0,0,255), 2)
    cv2.imshow('Social Distancing Monitor', frame)

        break

cap.release()
cv2.destroyAllWindows()

from deep_sort_realtime.deepsort_tracker import DeepSort

# --- CONFIGURATION ---
YOLO_ENGINE_PATH = "yolov8n.trt"  # Path to TensorRT engine
YOLO_INPUT_SIZE = (640, 640)      # Input size for YOLOv8
YOLO_CLASSES = [
    # COCO classes (80)
    "person", "bicycle", "car", "motorcycle", "airplane", "bus", "train", "truck", "boat",
    "traffic light", "fire hydrant", "stop sign", "parking meter", "bench", "bird", "cat",

    "couch", "potted plant", "bed", "dining table", "toilet", "tv", "laptop", "mouse",

    "book", "clock", "vase", "scissors", "teddy bear", "hair drier", "toothbrush"
]
CONF_THRESH = 0.3
NMS_THRESH = 0.45
FRAME_SKIP = 0  # Set >0 to skip frames for speed

# --- TensorRT YOLOv8 Wrapper ---
class TRT_YOLOv8:
    def __init__(self, engine_path, input_size=(640, 640)):
        self.input_size = input_size
        self.logger = trt.Logger(trt.Logger.INFO)
        with open(engine_path, "rb") as f, trt.Runtime(self.logger) as runtime:
            self.engine = runtime.deserialize_cuda_engine(f.read())
        self.context = self.engine.create_execution_context()
        self.inputs, self.outputs, self.bindings, self.stream = self.allocate_buffers()

    def allocate_buffers(self):
        inputs = []
        outputs = []
        bindings = []
        stream = cuda.Stream()
        for binding in self.engine:
            size = trt.volume(self.engine.get_binding_shape(binding)) * self.engine.max_batch_size
            dtype = trt.nptype(self.engine.get_binding_dtype(binding))
            host_mem = cuda.pagelocked_empty(size, dtype)
            device_mem = cuda.mem_alloc(host_mem.nbytes)
            bindings.append(int(device_mem))
            if self.engine.binding_is_input(binding):
                inputs.append({'host': host_mem, 'device': device_mem})
            else:
                outputs.append({'host': host_mem, 'device': device_mem})
        return inputs, outputs, bindings, stream

    def preprocess(self, img):
        img = cv2.resize(img, self.input_size)
        img = cv2.cvtColor(img, cv2.COLOR_BGR2RGB)
        img = img.astype(np.float32) / 255.0
        img = np.transpose(img, (2, 0, 1))
        img = np.expand_dims(img, 0)
        return img

    def infer(self, img):
        img_input = self.preprocess(img)
        np.copyto(self.inputs[0]['host'], img_input.ravel())
        cuda.memcpy_htod_async(self.inputs[0]['device'], self.inputs[0]['host'], self.stream)
        self.context.execute_async_v2(bindings=self.bindings, stream_handle=self.stream.handle)
        for out in self.outputs:
            cuda.memcpy_dtoh_async(out['host'], out['device'], self.stream)
        self.stream.synchronize()
        return [out['host'] for out in self.outputs]

    def postprocess(self, outputs, orig_shape):
        # YOLOv8 ONNX/TensorRT output: (batch, num, 84) [x, y, w, h, conf, 80 classes]
        pred = outputs[0]
        pred = pred.reshape(-1, len(YOLO_CLASSES) + 5)
        boxes = []
        confidences = []
        class_ids = []
        for det in pred:
            scores = det[5:]
            class_id = np.argmax(scores)
            conf = det[4] * scores[class_id]
            if conf > CONF_THRESH:
                cx, cy, w, h = det[0], det[1], det[2], det[3]
                x1 = (cx - w / 2) * orig_shape[1] / self.input_size[0]
                y1 = (cy - h / 2) * orig_shape[0] / self.input_size[1]
                x2 = (cx + w / 2) * orig_shape[1] / self.input_size[0]
                y2 = (cy + h / 2) * orig_shape[0] / self.input_size[1]
                boxes.append([x1, y1, x2 - x1, y2 - y1])
                confidences.append(float(conf))
                class_ids.append(int(class_id))
        # NMS
        indices = cv2.dnn.NMSBoxes(boxes, confidences, CONF_THRESH, NMS_THRESH)
        result = []
        for i in indices:
            i = i[0] if isinstance(i, (list, np.ndarray)) else i
            result.append((boxes[i], confidences[i], class_ids[i]))
        return result

    def detect(self, img):
        orig_shape = img.shape[:2]
        outputs = self.infer(img)
        detections = self.postprocess(outputs, orig_shape)
        return detections

# --- Camera Thread ---
class CameraThread(threading.Thread):
    def __init__(self, src, frame_queue, width=1280, height=720):
        super().__init__()
        self.cap = cv2.VideoCapture(src)
        self.cap.set(cv2.CAP_PROP_FRAME_WIDTH, width)
        self.cap.set(cv2.CAP_PROP_FRAME_HEIGHT, height)
        self.frame_queue = frame_queue
        self.running = True

    def run(self):
        while self.running:
            ret, frame = self.cap.read()
            if not ret:
                continue
            if not self.frame_queue.full():
                self.frame_queue.put(frame)
        self.cap.release()

    def stop(self):
        self.running = False

# --- Main Pipeline ---
def main():
    # --- Initialize ---
    trt_yolo = TRT_YOLOv8(YOLO_ENGINE_PATH, YOLO_INPUT_SIZE)

    frame_id = 0
