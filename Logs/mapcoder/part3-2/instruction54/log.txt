import wx

# Step 2: Create a custom panel for drawing
class DrawPanel(wx.Panel):
    def __init__(self, parent):
        super().__init__(parent)
        self.shapes = []
        self.current_shape = None
        self.Bind(wx.EVT_LEFT_DOWN, self.on_left_down)
        self.Bind(wx.EVT_LEFT_UP, self.on_left_up)
        self.Bind(wx.EVT_MOTION, self.on_mouse_move)
        self.Bind(wx.EVT_PAINT, self.on_paint)
        self.drawing = False

    def on_left_down(self, event):
        self.drawing = True
        pos = event.GetPosition()
        self.current_shape = {'type': 'rect', 'start': pos, 'end': pos}
        self.Refresh()

    def on_mouse_move(self, event):
        if self.drawing and event.Dragging() and event.LeftIsDown():
            self.current_shape['end'] = event.GetPosition()
            self.Refresh()

    def on_left_up(self, event):
        if self.drawing:
            self.current_shape['end'] = event.GetPosition()
            self.shapes.append(self.current_shape)
            self.current_shape = None
            self.drawing = False
            self.Refresh()

    def on_paint(self, event):
        dc = wx.PaintDC(self)
        for shape in self.shapes:
            self.draw_shape(dc, shape)
        if self.current_shape:
            self.draw_shape(dc, self.current_shape)

    def draw_shape(self, dc, shape):
        if shape['type'] == 'rect':
            x1, y1 = shape['start']
            x2, y2 = shape['end']
            dc.DrawRectangle(min(x1, x2), min(y1, y2), abs(x2-x1), abs(y2-y1))

    def save_image(self, filename):
        size = self.GetSize()
        bmp = wx.Bitmap(size[0], size[1])
        memdc = wx.MemoryDC(bmp)
        memdc.SetBackground(wx.Brush(wx.WHITE))
        memdc.Clear()
        for shape in self.shapes:
            self.draw_shape(memdc, shape)
        memdc.SelectObject(wx.NullBitmap)
        bmp.SaveFile(filename, wx.BITMAP_TYPE_PNG)

# Step 3: Create the main frame
class MainFrame(wx.Frame):
    def __init__(self):
        super().__init__(None, title="Draw and Save Shapes", size=(600, 400))

import wx

class FreehandPanel(wx.Panel):
    def __init__(self, parent):
        super().__init__(parent)
        self.lines = []
        self.current_line = []
        self.current_color = wx.BLACK
        self.Bind(wx.EVT_LEFT_DOWN, self.on_left_down)
        self.Bind(wx.EVT_MOTION, self.on_mouse_move)
        self.Bind(wx.EVT_LEFT_UP, self.on_left_up)
        self.Bind(wx.EVT_PAINT, self.on_paint)

    def on_left_down(self, event):
        self.current_line = [(event.GetPosition(), self.current_color)]

    def on_mouse_move(self, event):
        if event.Dragging() and event.LeftIsDown():
            self.current_line.append((event.GetPosition(), self.current_color))
            self.Refresh()

    def on_left_up(self, event):
        if self.current_line:
            self.lines.append(self.current_line)
            self.current_line = []
            self.Refresh()

    def on_paint(self, event):
        dc = wx.PaintDC(self)
        for line in self.lines:
            self.draw_line(dc, line)
        if self.current_line:
            self.draw_line(dc, self.current_line)

    def draw_line(self, dc, line):
        if len(line) < 2:
            return
        dc.SetPen(wx.Pen(line[0][1], 2))
        for i in range(len(line)-1):
            dc.SetPen(wx.Pen(line[i][1], 2))
            dc.DrawLine(line[i][0][0], line[i][0][1], line[i+1][0][0], line[i+1][0][1])

    def set_color(self, color):
        self.current_color = color

# Step 3: Create the main frame
class MainFrame(wx.Frame):
    def __init__(self):
        super().__init__(None, title="Freehand Drawing with Color", size=(600, 400))
        self.panel = FreehandPanel(self)

        dlg.Destroy()

# Step 4: Run the application

# Step 1: Import wxPython modules
import wx

# Step 2: Create a custom panel for drawing polygons
class PolygonPanel(wx.Panel):
    def __init__(self, parent):
        super().__init__(parent)
        self.polygons = []
        self.current_points = []
        self.selected_index = None
        self.Bind(wx.EVT_LEFT_DOWN, self.on_left_down)
        self.Bind(wx.EVT_RIGHT_DOWN, self.on_right_down)
        self.Bind(wx.EVT_PAINT, self.on_paint)

    def on_left_down(self, event):
        pos = event.GetPosition()
        self.current_points.append(pos)
        self.Refresh()

    def on_right_down(self, event):
        # Finish polygon on right click
        if len(self.current_points) > 2:
            self.polygons.append(self.current_points[:])
            self.current_points = []
            self.Refresh()

    def on_paint(self, event):
        dc = wx.PaintDC(self)
        for idx, poly in enumerate(self.polygons):
            pen = wx.Pen(wx.RED if idx == self.selected_index else wx.BLACK, 2)
            dc.SetPen(pen)
            dc.DrawPolygon(poly)
        if self.current_points:
            dc.SetPen(wx.Pen(wx.BLUE, 2, wx.PENSTYLE_DOT))
            dc.DrawLines(self.current_points)

    def select_polygon(self, pos):
        for idx, poly in enumerate(self.polygons):
            if self.point_in_polygon(pos, poly):
                self.selected_index = idx
                self.Refresh()
                return
        self.selected_index = None
        self.Refresh()

    def point_in_polygon(self, point, poly):
        # Ray casting algorithm
        x, y = point
        n = len(poly)
        inside = False
        p1x, p1y = poly[0]
        for i in range(n+1):
            p2x, p2y = poly[i % n]
            if y > min(p1y, p2y):
                if y <= max(p1y, p2y):
                    if x <= max(p1x, p2x):
                        if p1y != p2y:
                            xinters = (y-p1y)*(p2x-p1x)/(p2y-p1y)+p1x
                        if p1x == p2x or x <= xinters:
                            inside = not inside
            p1x, p1y = p2x, p2y
        return inside

    def delete_selected(self):
        if self.selected_index is not None:
            del self.polygons[self.selected_index]
            self.selected_index = None
            self.Refresh()

# Step 3: Create the main frame
class MainFrame(wx.Frame):
    def __init__(self):
        super().__init__(None, title="Draw and Delete Polygons", size=(600, 400))
        self.panel = PolygonPanel(self)
        select_btn = wx.Button(self, label="Select Polygon")
        delete_btn = wx.Button(self, label="Delete Polygon")
        select_btn.Bind(wx.EVT_BUTTON, self.on_select)
        delete_btn.Bind(wx.EVT_BUTTON, self.on_delete)
        sizer = wx.BoxSizer(wx.VERTICAL)
        sizer.Add(self.panel, 1, wx.EXPAND)
        btn_sizer = wx.BoxSizer(wx.HORIZONTAL)
        btn_sizer.Add(select_btn, 0, wx.ALL, 5)
        btn_sizer.Add(delete_btn, 0, wx.ALL, 5)
        sizer.Add(btn_sizer, 0, wx.ALIGN_CENTER)
        self.SetSizer(sizer)

    def on_select(self, event):
        dlg = wx.TextEntryDialog(self, "Enter x,y to select polygon:", "Select Polygon")
        if dlg.ShowModal() == wx.ID_OK:
            try:
                x, y = map(int, dlg.GetValue().split(','))
                self.panel.select_polygon((x, y))
            except:
                pass
        dlg.Destroy()

    def on_delete(self, event):
        self.panel.delete_selected()

# Step 4: Run the application
if __name__ == "__main__":
    app = wx.App(False)
    frame = MainFrame()
    frame.Show()
    app.MainLoop()

# --- Shape Classes ---

class Shape:
    def __init__(self, pos, color=wx.Colour(0, 0, 0)):
        self.pos = pos  # (x, y)

        self.selected = False

    def draw(self, dc):

        return False

    def move(self, dx, dy):
        pass

class RectangleShape(Shape):
    def __init__(self, pos, size, color=wx.Colour(0, 0, 0)):
        super().__init__(pos, color)
        self.size = size  # (w, h)

    def draw(self, dc):
        pen = wx.Pen(wx.Colour(0, 120, 215) if self.selected else self.color, 2)
        dc.SetPen(pen)
        dc.SetBrush(wx.Brush(self.color, wx.BRUSHSTYLE_TRANSPARENT))
        dc.DrawRectangle(self.pos[0], self.pos[1], self.size[0], self.size[1])

    def hit_test(self, pt):
        x, y = self.pos
        w, h = self.size
        px, py = pt
        return x <= px <= x + w and y <= py <= y + h

    def move(self, dx, dy):
        self.pos = (self.pos[0] + dx, self.pos[1] + dy)

class CircleShape(Shape):
    def __init__(self, pos, radius, color=wx.Colour(0, 0, 0)):
        super().__init__(pos, color)
        self.radius = radius

    def draw(self, dc):
        pen = wx.Pen(wx.Colour(0, 120, 215) if self.selected else self.color, 2)
        dc.SetPen(pen)
        dc.SetBrush(wx.Brush(self.color, wx.BRUSHSTYLE_TRANSPARENT))
        dc.DrawCircle(self.pos[0], self.pos[1], self.radius)

    def hit_test(self, pt):
        px, py = pt
        cx, cy = self.pos
        return math.hypot(px - cx, py - cy) <= self.radius

    def move(self, dx, dy):
        self.pos = (self.pos[0] + dx, self.pos[1] + dy)
