import pandas as pd
import matplotlib.pyplot as plt
import seaborn as sns
import requests
from io import StringIO

# Step 1: Identify and access the data source
# We'll use Our World in Data's COVID-19 dataset (public, reputable)
DATA_URL = "https://covid.ourworldindata.org/data/owid-covid-data.csv"

# Step 2: Download and load the data into a DataFrame
response = requests.get(DATA_URL)
data = pd.read_csv(StringIO(response.text))

# Step 3: Inspect and clean the data
# Select relevant columns for analysis
columns_of_interest = [
    'location', 'date', 'total_cases', 'new_cases', 'total_deaths', 'new_deaths',
    'people_vaccinated', 'people_fully_vaccinated', 'population'
]
data = data[columns_of_interest]

# Remove rows with missing location or date
data = data.dropna(subset=['location', 'date'])

# Remove duplicates
