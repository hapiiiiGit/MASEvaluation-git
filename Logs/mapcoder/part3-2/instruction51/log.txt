def upload_csv(request):

            )

from io import BytesIO
import base64

def spending_chart(request):

        plt.savefig(buf, format='png')

    return render(request, 'chart.html', {'chart': chart})

# Step 3: View to handle audio upload and transcription
import speech_recognition as sr
from pydub import AudioSegment

def transcribe_audio(file_path):
    # Convert to wav if necessary
    if not file_path.endswith('.wav'):
        sound = AudioSegment.from_file(file_path)
        file_path_wav = file_path + '.wav'
        sound.export(file_path_wav, format='wav')
        file_path = file_path_wav
    recognizer = sr.Recognizer()
    with sr.AudioFile(file_path) as source:
        audio = recognizer.record(source)
    return recognizer.recognize_google(audio)

# Step 4: Summarize transcription using transformers
from transformers import pipeline

summarizer = pipeline('summarization')

def upload_audio(request):
    if request.method == 'POST':
        form = AudioUploadForm(request.POST, request.FILES)
        if form.is_valid():
            audio_instance = form.save()
            audio_path = audio_instance.audio.path
            transcription = transcribe_audio(audio_path)
            audio_instance.transcription = transcription
            summary = summarizer(transcription, max_length=50, min_length=25, do_sample=False)[0]['summary_text']
            audio_instance.summary = summary
            audio_instance.save()
            return redirect('audio_list')
    else:
        form = AudioUploadForm()
    return render(request, 'upload_audio.html', {'form': form})

import os
import PyPDF2
import docx
from transformers import pipeline

def extract_text_from_file(file_path):
    ext = os.path.splitext(file_path)[1].lower()
    if ext == '.pdf':
        with open(file_path, 'rb') as f:
            reader = PyPDF2.PdfReader(f)
            text = ''
            for page in reader.pages:
                text += page.extract_text() or ''
            return text
    elif ext in ['.docx', '.doc']:
        doc = docx.Document(file_path)
        return '\n'.join([para.text for para in doc.paragraphs])
    elif ext == '.txt':
        with open(file_path, 'r', encoding='utf-8', errors='ignore') as f:
            return f.read()
    else:
        return ''

summarizer = pipeline("summarization", model="sshleifer/distilbart-cnn-12-6")

def summarize_text(text):
    if not text.strip():
        return ''
    # HuggingFace models have a max token limit, so chunk if needed
    max_chunk = 1000
    text = text.replace('\n', ' ')
    chunks = [text[i:i+max_chunk] for i in range(0, len(text), max_chunk)]
    summary = ''
    for chunk in chunks:
        out = summarizer(chunk, max_length=130, min_length=30, do_sample=False)
        summary += out[0]['summary_text'] + ' '
    return summary.strip()

# Safe formula evaluation
import ast
import operator as op

SAFE_OPERATORS = {
    ast.Add: op.add, ast.Sub: op.sub, ast.Mult: op.mul, ast.Div: op.truediv,
    ast.Pow: op.pow, ast.USub: op.neg, ast.Mod: op.mod
}

def eval_expr(expr, variables):
    """
    Safely evaluate a mathematical expression with variables.
    """
    def _eval(node):
        if isinstance(node, ast.Num):  # <number>
            return node.n
        elif isinstance(node, ast.BinOp):  # <left> <operator> <right>
            return SAFE_OPERATORS[type(node.op)](_eval(node.left), _eval(node.right))
        elif isinstance(node, ast.UnaryOp):  # - <operand>
            return SAFE_OPERATORS[type(node.op)](_eval(node.operand))
        elif isinstance(node, ast.Name):
            if node.id in variables:
                return variables[node.id]
            else:
                raise ValueError(f"Unknown variable: {node.id}")
        else:
            raise TypeError(node)
    node = ast.parse(expr, mode='eval').body
    return _eval(node)

def formula_apply(request, pk):

    result = None
    error = None
    variables = {}
    # Extract variable names from formula
    import re
    var_names = set(re.findall(r'\b[a-zA-Z_][a-zA-Z0-9_]*\b', formula.formula))
    reserved = {'pow', 'abs', 'min', 'max', 'sum'}
    var_names = [v for v in var_names if v not in reserved]
    if request.method == 'POST':
        for v in var_names:
            try:
                variables[v] = float(request.POST.get(v, 0))
            except:
                variables[v] = 0
        try:
            result = eval_expr(formula.formula, variables)
        except Exception as e:
            error = str(e)
    return render(request, 'formula_apply.html', {
        'formula': formula,
        'variables': var_names,
        'result': result,
        'error': error
