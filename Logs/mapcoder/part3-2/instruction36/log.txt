def send_apdu(connection, apdu, data=[]):
    cmd = apdu + [len(data)] + data + [0x00]
    response, sw1, sw2 = connection.transmit(cmd)
    if sw1 != 0x90 or sw2 != 0x00:
        raise Exception(f"APDU failed: {toHexString(cmd)} SW={hex(sw1)} {hex(sw2)}")
    return response

def select_application(connection, aid):
    apdu = SELECT_APP + [len(aid)] + aid + [0x00]
    response, sw1, sw2 = connection.transmit(apdu)
    if sw1 != 0x90 or sw2 != 0x00:
        raise Exception("Select application failed")
    return response

def create_ef(connection, file_id, file_size):
    data = [file_id >> 8, file_id & 0xFF, file_size >> 8, file_size & 0xFF]
    return send_apdu(connection, CREATE_EF, data)

def set_access_rights(connection, file_id, access_rights):
    data = [file_id >> 8, file_id & 0xFF] + access_rights
    return send_apdu(connection, SET_ACCESS, data)
