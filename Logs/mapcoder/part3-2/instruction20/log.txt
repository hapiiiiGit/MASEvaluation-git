def aggregate_tokens(*token_lists):
    tokens_by_address = defaultdict(list)
    meta_by_address = {}
    for tokens in token_lists:
        for t in tokens:
            tokens_by_address[t["address"].lower()].append({
                "exchange": t["exchange"],
                "price": t["price"]
            })
            if t["address"].lower() not in meta_by_address:
                meta_by_address[t["address"].lower()] = {
                    "name": t["name"],
                    "symbol": t["symbol"],
                    "address": t["address"]
                }
    return tokens_by_address, meta_by_address

def analyze(tokens_by_address, meta_by_address, threshold=1.8):
    results = []
    for addr, prices in tokens_by_address.items():
        if len(prices) < 2:
            continue
        sorted_prices = sorted(prices, key=lambda x: x["price"])
        low = sorted_prices[0]
        high = sorted_prices[-1]
        if low["price"] == 0:
            continue
        diff_pct = ((high["price"] - low["price"]) / low["price"]) * 100
        if diff_pct > threshold:
            meta = meta_by_address[addr]
            results.append({
                "name": meta["name"],
                "symbol": meta["symbol"],
                "address": meta["address"],
                "low_exchange": low["exchange"],
                "low_price": low["price"],
                "high_exchange": high["exchange"],
                "high_price": high["price"],
                "diff_pct": diff_pct
            })
    results.sort(key=lambda x: -x["diff_pct"])
    return results
