def set_env_var(name, value, user=True):
    log_and_print(f"Setting environment variable {name}={value}")
    if user:
        scope = winreg.HKEY_CURRENT_USER
        reg_path = r"Environment"
    else:
        scope = winreg.HKEY_LOCAL_MACHINE
        reg_path = r"SYSTEM\CurrentControlSet\Control\Session Manager\Environment"
    try:
        with winreg.OpenKey(scope, reg_path, 0, winreg.KEY_SET_VALUE) as key:
            winreg.SetValueEx(key, name, 0, winreg.REG_EXPAND_SZ, value)
    except Exception as e:
        logging.error(f"Failed to set environment variable: {e}")

def update_path_env(python_dir):
    current_path = os.environ.get("PATH", "")
    python_path = str(python_dir)
    scripts_path = str(python_dir / "Scripts")
    paths = current_path.split(";")
    updated = False
    for p in [python_path, scripts_path]:
        if p not in paths:
            paths.insert(0, p)
            updated = True
    if updated:
        new_path = ";".join(paths)
        set_env_var("PATH", new_path)
        os.environ["PATH"] = new_path
        log_and_print("PATH environment variable updated.")

def verify_and_install_packages(python_exe, packages):

        try:

            log_and_print(f"Package {pkg} is already installed.")
        except Exception:
            log_and_print(f"Installing missing package: {pkg}")
            run_cmd([str(python_exe), "-m", "pip", "install", pkg], check=True)

def compile_script(python_exe, script_path, output_dir):
    log_and_print(f"Compiling {script_path} to .exe using PyInstaller")
    if not output_dir.exists():
        output_dir.mkdir(parents=True)
    run_cmd([
        str(python_exe), "-m", "PyInstaller",
        "--onefile",
        "--distpath", str(output_dir),
        "--workpath", str(output_dir / "build"),
        "--specpath", str(output_dir / "spec"),
        str(script_path)
    ])
    exe_name = script_path.stem + ".exe"
    exe_path = output_dir / exe_name
    if exe_path.exists():
        log_and_print(f"Executable created at {exe_path}")
        return exe_path
    else:
        raise FileNotFoundError(f"Executable {exe_path} not found after build.")
