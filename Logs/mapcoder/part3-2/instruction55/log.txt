# Step 1: Import necessary libraries
import tkinter as tk
from binance.client import Client
import matplotlib.pyplot as plt
from matplotlib.backends.backend_tkagg import FigureCanvasTkAgg
import pandas as pd
import threading
import time

# Step 2: Define function to fetch historical data
def fetch_data(symbol, interval, lookback):
    client = Client('API_KEY', 'API_SECRET')
    klines = client.get_historical_klines(symbol, interval, lookback)
    df = pd.DataFrame(klines, columns=['time','open','high','low','close','volume','close_time','quote_asset_volume','number_of_trades','taker_buy_base','taker_buy_quote','ignore'])
    df['close'] = df['close'].astype(float)
    return df

# Step 3: Define function to calculate SMA and generate signals
def sma_strategy(df, short_window=5, long_window=20):
    df['SMA_short'] = df['close'].rolling(window=short_window).mean()
    df['SMA_long'] = df['close'].rolling(window=long_window).mean()
    df['signal'] = 0
    df['signal'][short_window:] = (df['SMA_short'][short_window:] > df['SMA_long'][short_window:]).astype(int)
    df['positions'] = df['signal'].diff()
    return df

# Step 4: Plotting function
def plot_chart(df):
    fig, ax = plt.subplots()
    ax.plot(df['close'], label='Close Price')
    ax.plot(df['SMA_short'], label='SMA Short')
    ax.plot(df['SMA_long'], label='SMA Long')
    buy_signals = df[df['positions'] == 1]
    sell_signals = df[df['positions'] == -1]
    ax.plot(buy_signals.index, df['close'][buy_signals.index], '^', markersize=10, color='g', label='Buy')
    ax.plot(sell_signals.index, df['close'][sell_signals.index], 'v', markersize=10, color='r', label='Sell')
    ax.legend()
    return fig

# Step 5: GUI integration
def update_chart():
    df = fetch_data(symbol_entry.get(), '1m', '1 day ago UTC')
    df = sma_strategy(df)
    fig = plot_chart(df)
    canvas = FigureCanvasTkAgg(fig, master=root)
    canvas.get_tk_widget().pack()
    canvas.draw()

root = tk.Tk()
root.title("SMA Crossover Strategy")

tk.Label(root, text="Symbol (e.g., BTCUSDT):").pack()
symbol_entry = tk.Entry(root)
symbol_entry.pack()

import threading
import queue
import time
import tkinter as tk
from tkinter import ttk, scrolledtext, messagebox

# IBKR API imports
from ibapi.client import EClient
from ibapi.wrapper import EWrapper
from ibapi.contract import Contract
from ibapi.order import Order

# ========== IBKR API Wrapper ==========
class IBKRWrapper(EWrapper, EClient):
    def __init__(self, event_queue):
        EClient.__init__(self, self)
        self.event_queue = event_queue
        self.market_data = {}
        self.nextOrderId = None
        self.positions = {}
        self.pnl = 0.0
        self.trade_history = []

    def error(self, reqId, errorCode, errorString):
        self.event_queue.put(('log', f"Error {errorCode}: {errorString}"))

    def nextValidId(self, orderId):
        self.nextOrderId = orderId
        self.event_queue.put(('log', f"Next valid order id: {orderId}"))

    def tickPrice(self, reqId, tickType, price, attrib):
        if tickType == 4:  # Last price
            self.market_data[reqId] = price
            self.event_queue.put(('price', reqId, price))

    def orderStatus(self, orderId, status, filled, remaining, avgFillPrice, permId,
                    parentId, lastFillPrice, clientId, whyHeld, mktCapPrice):
        self.event_queue.put(('log', f"Order {orderId} status: {status}, filled: {filled}, avgFillPrice: {avgFillPrice}"))

    def execDetails(self, reqId, contract, execution):
        self.event_queue.put(('log', f"Execution: {execution.side} {execution.shares} @ {execution.price}"))
        self.trade_history.append((execution.time, execution.side, execution.shares, execution.price))

    def position(self, account, contract, position, avgCost):
        symbol = contract.symbol
        self.positions[symbol] = position
        self.event_queue.put(('position', symbol, position, avgCost))

    def pnl(self, reqId, dailyPnL, unrealizedPnL, realizedPnL):
        self.pnl = dailyPnL
        self.event_queue.put(('pnl', dailyPnL, unrealizedPnL, realizedPnL))

# ========== Momentum Strategy ==========
class MomentumStrategy:
    def __init__(self, symbol, window, threshold, ibkr, event_queue):
        self.symbol = symbol
        self.window = window
        self.threshold = threshold
        self.ibkr = ibkr
        self.event_queue = event_queue
        self.prices = []
        self.position = 0
        self.running = False
        self.contract = self.create_contract(symbol)
        self.reqId = 1

    def create_contract(self, symbol):
        contract = Contract()
        contract.symbol = symbol
        contract.secType = "STK"
        contract.exchange = "SMART"
        contract.currency = "USD"
        return contract

    def start(self):
        self.running = True
        self.event_queue.put(('log', f"Strategy started for {self.symbol}"))
        self.ibkr.reqMktData(self.reqId, self.contract, "", False, False, [])
        threading.Thread(target=self.run, daemon=True).start()

    def stop(self):
        self.running = False
        self.ibkr.cancelMktData(self.reqId)
        self.event_queue.put(('log', "Strategy stopped."))

    def run(self):
        while self.running:
            price = self.ibkr.market_data.get(self.reqId)
            if price is not None:
                self.prices.append(price)
                if len(self.prices) > self.window:
                    self.prices.pop(0)
                    momentum = self.prices[-1] - self.prices[0]
                    self.event_queue.put(('log', f"Momentum: {momentum:.2f}"))
                    if momentum > self.threshold and self.position <= 0:
                        self.place_order("BUY", 100)
                        self.position += 100
                    elif momentum < -self.threshold and self.position >= 0:
                        self.place_order("SELL", 100)
                        self.position -= 100
            time.sleep(1)

    def place_order(self, action, quantity):
        order = Order()
        order.action = action
        order.orderType = "MKT"
        order.totalQuantity = quantity
        order.transmit = True
        order_id = self.ibkr.nextOrderId
        if order_id is not None:
            self.ibkr.placeOrder(order_id, self.contract, order)
            self.ibkr.nextOrderId += 1
            self.event_queue.put(('log', f"Placed {action} order for {quantity} shares of {self.symbol}"))
        else:
            self.event_queue.put(('log', "Order ID not available yet."))

# ========== GUI ==========
class TradingBotGUI:
    def __init__(self, root):
        self.root = root
        self.root.title("Momentum Trading Bot")
        self.event_queue = queue.Queue()
        self.ibkr = IBKRWrapper(self.event_queue)
        self.strategy = None
        self.create_widgets()
        self.update_gui()
        threading.Thread(target=self.run_ibkr, daemon=True).start()

    def create_widgets(self):
        frame = ttk.Frame(self.root, padding="10")
        frame.grid(row=0, column=0, sticky="nsew")

        ttk.Label(frame, text="Symbol:").grid(row=0, column=0, sticky="w")
        self.symbol_entry = ttk.Entry(frame)
        self.symbol_entry.grid(row=0, column=1, sticky="ew")
        self.symbol_entry.insert(0, "AAPL")

        ttk.Label(frame, text="Window Size:").grid(row=1, column=0, sticky="w")
        self.window_entry = ttk.Entry(frame)
        self.window_entry.grid(row=1, column=1, sticky="ew")
        self.window_entry.insert(0, "5")

        ttk.Label(frame, text="Threshold:").grid(row=2, column=0, sticky="w")
        self.threshold_entry = ttk.Entry(frame)
        self.threshold_entry.grid(row=2, column=1, sticky="ew")
        self.threshold_entry.insert(0, "1.0")

        self.start_button = ttk.Button(frame, text="Start", command=self.start_strategy)
        self.start_button.grid(row=3, column=0, sticky="ew")
        self.stop_button = ttk.Button(frame, text="Stop", command=self.stop_strategy, state="disabled")
        self.stop_button.grid(row=3, column=1, sticky="ew")

        ttk.Label(frame, text="Logs:").grid(row=4, column=0, sticky="w")
        self.log_text = scrolledtext.ScrolledText(frame, width=60, height=10, state="disabled")
        self.log_text.grid(row=5, column=0, columnspan=2, sticky="ew")

        ttk.Label(frame, text="Positions:").grid(row=6, column=0, sticky="w")
        self.position_label = ttk.Label(frame, text="None")
        self.position_label.grid(row=6, column=1, sticky="w")

        ttk.Label(frame, text="PnL:").grid(row=7, column=0, sticky="w")
        self.pnl_label = ttk.Label(frame, text="0.0")
        self.pnl_label.grid(row=7, column=1, sticky="w")

        ttk.Label(frame, text="Trade History:").grid(row=8, column=0, sticky="w")
        self.trade_text = scrolledtext.ScrolledText(frame, width=60, height=5, state="disabled")
        self.trade_text.grid(row=9, column=0, columnspan=2, sticky="ew")

        frame.columnconfigure(1, weight=1)

    def run_ibkr(self):
        try:
            self.ibkr.connect("127.0.0.1", 7497, 0)
            self.ibkr.run()
        except Exception as e:
            self.event_queue.put(('log', f"IBKR connection error: {e}"))

    def start_strategy(self):
        symbol = self.symbol_entry.get().strip()
        try:
            window = int(self.window_entry.get())
            threshold = float(self.threshold_entry.get())
        except ValueError:
            messagebox.showerror("Error", "Invalid window or threshold value.")
            return
        self.strategy = MomentumStrategy(symbol, window, threshold, self.ibkr, self.event_queue)
        self.strategy.start()
        self.start_button.config(state="disabled")
        self.stop_button.config(state="normal")

    def stop_strategy(self):
        if self.strategy:
            self.strategy.stop()
        self.start_button.config(state="normal")
        self.stop_button.config(state="disabled")

    def update_gui(self):
        while not self.event_queue.empty():
            event = self.event_queue.get()
            if event[0] == 'log':
                self.log_text.config(state="normal")
                self.log_text.insert(tk.END, event[1] + "\n")
                self.log_text.see(tk.END)
                self.log_text.config(state="disabled")
            elif event[0] == 'price':
                pass  # Could display live price if desired
            elif event[0] == 'position':
                symbol, position, avgCost = event[1], event[2], event[3]
                self.position_label.config(text=f"{symbol}: {position} @ {avgCost:.2f}")
            elif event[0] == 'pnl':
                dailyPnL, unrealizedPnL, realizedPnL = event[1], event[2], event[3]
                self.pnl_label.config(text=f"{dailyPnL:.2f} (Unrealized: {unrealizedPnL:.2f}, Realized: {realizedPnL:.2f})")
            # Update trade history
            self.trade_text.config(state="normal")
            self.trade_text.delete(1.0, tk.END)
            for trade in self.ibkr.trade_history[-10:]:
                self.trade_text.insert(tk.END, f"{trade[0]} {trade[1]} {trade[2]} @ {trade[3]}\n")
            self.trade_text.config(state="disabled")
        self.root.after(500, self.update_gui)

# ========== Main ==========
def main():
    root = tk.Tk()
