import agenticai
import os

# Step 1: Set up connections and locate CSV files
POSTGRES_CONFIG = {
    'host': 'localhost',
    'port': 5432,
    'database': 'operations_db',
    'user': 'username',
    'password': 'password'
}
CSV_DIR = './csv_data'
csv_files = [os.path.join(CSV_DIR, f) for f in os.listdir(CSV_DIR) if f.endswith('.csv')]

# Step 2: Ingest data from PostgreSQL
pg_source = agenticai.data.PostgreSQLSource(
    host=POSTGRES_CONFIG['host'],
    port=POSTGRES_CONFIG['port'],
    database=POSTGRES_CONFIG['database'],
    user=POSTGRES_CONFIG['user'],
    password=POSTGRES_CONFIG['password']
)
pg_data = agenticai.ingest.from_postgresql(pg_source, table='operational_metrics')

# Step 3: Ingest data from CSV files
csv_data_frames = []
for csv_file in csv_files:
    csv_df = agenticai.ingest.from_csv(csv_file)
    csv_data_frames.append(csv_df)

# Step 4: Merge and clean data
all_data = agenticai.data.merge([pg_data] + csv_data_frames)
all_data = agenticai.data.clean(
    all_data,
    missing_strategy='mean_impute',
    dtype_consistency=True
)

# Step 5: Analyze data for efficiency insights
efficiency_metrics = agenticai.analysis.compute_metrics(
    all_data,
    metrics=['throughput', 'utilization', 'avg_processing_time']
)
bottlenecks = agenticai.analysis.detect_bottlenecks(
    all_data,
    group_by='process_step',
    metric='avg_processing_time'
)
efficiency_gaps = agenticai.analysis.detect_gaps(
    all_data,
    metric='utilization',
    threshold=0.7
)

# Step 6: Aggregate and summarize findings
summary = agenticai.analysis.summarize(
    metrics=efficiency_metrics,
    bottlenecks=bottlenecks,
    gaps=efficiency_gaps
)

# Step 7: Generate report
report = agenticai.report.generate(
    summary=summary,
    recommendations=agenticai.analysis.recommendations(
        bottlenecks=bottlenecks,
        gaps=efficiency_gaps
    ),
    visualizations=[
        agenticai.visualization.bar_chart(efficiency_metrics, title='Efficiency Metrics'),
        agenticai.visualization.table(bottlenecks, title='Detected Bottlenecks'),
        agenticai.visualization.table(efficiency_gaps, title='Efficiency Gaps')
    ]
)

# Step 8: Output report
agenticai.report.export(report, filename='efficiency_report.pdf')
print("Report generated: efficiency_report.pdf")
