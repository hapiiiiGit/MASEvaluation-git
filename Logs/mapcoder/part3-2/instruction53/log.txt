import pdfplumber
import docx
import spacy
import os

nlp = spacy.load("en_core_web_sm")

# Step 2: Extract text from PDF
def extract_text_pdf(path):
    with pdfplumber.open(path) as pdf:
        text = ''
        for page in pdf.pages:
            text += page.extract_text() + '\n'
    return text

# Step 3: Extract text from DOCX
def extract_text_docx(path):
    doc = docx.Document(path)
    return '\n'.join([para.text for para in doc.paragraphs])

# Step 4: Extract entities (name, skills, experience)
def extract_entities(text):
    doc = nlp(text)
    # For simplicity, just extract PERSON entities as names
    names = [ent.text for ent in doc.ents if ent.label_ == 'PERSON']
    # Skills and experience extraction would require custom patterns or ML models
    # Here, we just search for keywords as a placeholder
    skills = []
    for token in doc:
        if token.text.lower() in ['python', 'java', 'sql', 'excel']:
            skills.append(token.text)
    return {'names': names, 'skills': list(set(skills))}

# Step 5: Bulk process resumes
def process_resumes(folder):
    results = []
    for filename in os.listdir(folder):
        path = os.path.join(folder, filename)
        if filename.lower().endswith('.pdf'):
            text = extract_text_pdf(path)
        elif filename.lower().endswith('.docx'):
            text = extract_text_docx(path)
        else:
            continue
        entities = extract_entities(text)
        results.append({'filename': filename, **entities})
    return results

# Example usage:

def summarize_document(doc_id):
    doc = Document.objects.get(id=doc_id)

    try:

import re
import ast

def extract_financial_fields(text):
    # Dummy extractor: look for "amount: 123.45" and "tax_rate: 0.07"
    fields = {}
    amount = re.search(r'amount[:\s]+([\d\.]+)', text)
    tax_rate = re.search(r'tax[_\s]?rate[:\s]+([\d\.]+)', text)
    if amount:
        fields['amount'] = float(amount.group(1))
    if tax_rate:
        fields['tax_rate'] = float(tax_rate.group(1))
    return fields

def safe_eval(expr, context):
    allowed_names = set(context.keys())
    node = ast.parse(expr, mode='eval')
    for n in ast.walk(node):
        if isinstance(n, ast.Name):
            if n.id not in allowed_names:
                raise ValueError(f"Use of variable '{n.id}' not allowed.")
        elif isinstance(n, (ast.Call, ast.Import, ast.ImportFrom, ast.Attribute)):
            raise ValueError("Function calls and imports are not allowed.")
    code = compile(node, '<string>', 'eval')
    return eval(code, {"__builtins__": {}}, context)
