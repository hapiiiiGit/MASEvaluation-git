import logging

    format='%(asctime)s [%(levelname)s] %(message)s',
    handlers=[
        logging.StreamHandler(sys.stdout),
        logging.FileHandler('automation_script.log', mode='a')
    ]
)

def list_files(directory, recursive=False):
    """List all files in a directory (optionally recursively)."""
    files = []
    try:
        if recursive:
            for root, _, filenames in os.walk(directory):
                for f in filenames:
                    files.append(os.path.join(root, f))
        else:
            for f in os.listdir(directory):
                path = os.path.join(directory, f)
                if os.path.isfile(path):
                    files.append(path)
        logging.info(f"Listed {len(files)} files in '{directory}' (recursive={recursive})")
    except Exception as e:
        logging.error(f"Error listing files: {e}")
    return files

def copy_files(src_dir, dst_dir, recursive=False):
    """Copy all files from src_dir to dst_dir."""
    try:
        files = list_files(src_dir, recursive)
        os.makedirs(dst_dir, exist_ok=True)
        for f in files:
            rel_path = os.path.relpath(f, src_dir)
            dst_path = os.path.join(dst_dir, rel_path)
            os.makedirs(os.path.dirname(dst_path), exist_ok=True)
            shutil.copy2(f, dst_path)
            logging.info(f"Copied '{f}' to '{dst_path}'")
    except Exception as e:
        logging.error(f"Error copying files: {e}")

def web_scrape(url, selector):
    """Scrape text content from a web page using a CSS selector."""
    try:
        resp = requests.get(url, timeout=10)
        resp.raise_for_status()
        soup = BeautifulSoup(resp.text, 'html.parser')
        elements = soup.select(selector)
        results = [el.get_text(strip=True) for el in elements]
        logging.info(f"Scraped {len(results)} elements from '{url}' with selector '{selector}'")
        return results
    except Exception as e:
        logging.error(f"Error scraping '{url}': {e}")
        return []

def search_in_files(directory, keyword, recursive=False):
    """Search for a keyword in all files in a directory."""
    matches = []
    files = list_files(directory, recursive)
    for f in files:
        try:
            with open(f, 'r', encoding='utf-8', errors='ignore') as file:
                for i, line in enumerate(file, 1):
                    if keyword in line:
                        matches.append((f, i, line.strip()))
                        logging.info(f"Found '{keyword}' in '{f}' at line {i}")
        except Exception as e:
            logging.warning(f"Could not read '{f}': {e}")
    return matches
