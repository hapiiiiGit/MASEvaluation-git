import uuid
import threading
import time
from flask import Flask, request, jsonify
from enum import Enum

app = Flask(__name__)

# --- In-memory DB simulation ---
USERS = {}
TENANTS = {}
SUBSCRIPTION_PLANS = {
    "basic": {"price": 10, "modules": ["crm"], "workflows": ["basic_onboarding"]},
    "pro": {"price": 30, "modules": ["crm", "accounts"], "workflows": ["pro_onboarding"]},
}
PAYMENTS = {}
SITES = {}

# --- Enums ---
class SiteStatus(Enum):
    PENDING = "pending"
    PROVISIONING = "provisioning"
    ACTIVE = "active"
    FAILED = "failed"

# --- Utilities ---
def generate_id():
    return str(uuid.uuid4())

def log(msg):
    print(f"[LOG] {time.strftime('%Y-%m-%d %H:%M:%S')} - {msg}")

# --- Models ---
class User:
    def __init__(self, email, password):
        self.id = generate_id()
        self.email = email
        self.password = password
        self.tenant_id = None

class Tenant:
    def __init__(self, user_id, plan):
        self.id = generate_id()
        self.user_id = user_id
        self.plan = plan
        self.site_id = None

class Payment:
    def __init__(self, tenant_id, amount):
        self.id = generate_id()
        self.tenant_id = tenant_id
        self.amount = amount
        self.status = "pending"
        self.telr_ref = generate_id()

class Site:
    def __init__(self, tenant_id, plan):
        self.id = generate_id()
        self.tenant_id = tenant_id
        self.plan = plan
        self.status = SiteStatus.PENDING
        self.domain = f"{self.id}.saas.example.com"
        self.modules = []
        self.workflows = []

# --- API Endpoints ---

@app.route("/api/register", methods=["POST"])
def register():
    data = request.json
    email = data.get("email")
    password = data.get("password")
    plan = data.get("plan")
    if not email or not password or plan not in SUBSCRIPTION_PLANS:
        return jsonify({"error": "Invalid input"}), 400
    user = User(email, password)
    USERS[user.id] = user
    tenant = Tenant(user.id, plan)
    TENANTS[tenant.id] = tenant
    user.tenant_id = tenant.id
    log(f"User registered: {email}, Tenant: {tenant.id}, Plan: {plan}")
    return jsonify({"user_id": user.id, "tenant_id": tenant.id})

@app.route("/api/subscribe", methods=["POST"])
def subscribe():
    data = request.json
    tenant_id = data.get("tenant_id")
    plan = data.get("plan")
    if tenant_id not in TENANTS or plan not in SUBSCRIPTION_PLANS:
        return jsonify({"error": "Invalid tenant or plan"}), 400
    tenant = TENANTS[tenant_id]
    tenant.plan = plan
    amount = SUBSCRIPTION_PLANS[plan]["price"]
    payment = Payment(tenant_id, amount)
    PAYMENTS[payment.id] = payment
    # Simulate Telr payment URL
    telr_url = f"https://secure.telr.com/gateway/process?ref={payment.telr_ref}"
    log(f"Payment initiated for Tenant {tenant_id}, Amount: {amount}, TelrRef: {payment.telr_ref}")
    return jsonify({"payment_id": payment.id, "telr_url": telr_url})

@app.route("/api/payment/callback", methods=["POST"])
def payment_callback():
    data = request.json
    telr_ref = data.get("telr_ref")
    status = data.get("status")  # "paid" or "failed"
    payment = next((p for p in PAYMENTS.values() if p.telr_ref == telr_ref), None)
    if not payment:
        return jsonify({"error": "Invalid Telr reference"}), 400
    payment.status = status
    log(f"Payment callback: {telr_ref}, Status: {status}")
    if status == "paid":
        # Start site provisioning asynchronously
        threading.Thread(target=provision_site, args=(payment.tenant_id,)).start()
    return jsonify({"payment_id": payment.id, "status": status})

@app.route("/api/site/status", methods=["GET"])
def site_status():
    tenant_id = request.args.get("tenant_id")
    tenant = TENANTS.get(tenant_id)
    if not tenant or not tenant.site_id:
        return jsonify({"status": "not_provisioned"})
    site = SITES.get(tenant.site_id)
    return jsonify({
        "site_id": site.id,
        "status": site.status.value,
        "domain": site.domain if site.status == SiteStatus.ACTIVE else None,
        "modules": site.modules,
        "workflows": site.workflows
    })

@app.route("/api/trigger_workflow", methods=["POST"])
def trigger_workflow():
    data = request.json
    site_id = data.get("site_id")
    workflow = data.get("workflow")
    site = SITES.get(site_id)
    if not site or workflow not in SUBSCRIPTION_PLANS[site.plan]["workflows"]:
        return jsonify({"error": "Invalid site or workflow"}), 400
    # Simulate workflow execution
    log(f"Workflow '{workflow}' triggered for Site {site_id}")
    return jsonify({"status": "workflow_triggered", "workflow": workflow})

# --- Orchestration Logic ---

def provision_site(tenant_id):
    tenant = TENANTS.get(tenant_id)
    if not tenant:
        log(f"Provision failed: Tenant {tenant_id} not found")
        return
    plan = tenant.plan
    site = Site(tenant_id, plan)
    SITES[site.id] = site
    tenant.site_id = site.id
    site.status = SiteStatus.PROVISIONING
    log(f"Provisioning site {site.id} for Tenant {tenant_id} (Plan: {plan})")
    try:
        # Simulate resource allocation
        time.sleep(2)
        # Install custom modules
        modules = SUBSCRIPTION_PLANS[plan]["modules"]
        for module in modules:
            install_module(site, module)
        # Configure workflows
        workflows = SUBSCRIPTION_PLANS[plan]["workflows"]
        for wf in workflows:
            configure_workflow(site, wf)
        site.status = SiteStatus.ACTIVE
        log(f"Site {site.id} provisioned successfully. Domain: {site.domain}")
    except Exception as e:
        site.status = SiteStatus.FAILED
        log(f"Site provisioning failed for {site.id}: {str(e)}")
        # Rollback: remove site
        del SITES[site.id]
        tenant.site_id = None

def install_module(site, module):
    log(f"Installing module '{module}' on Site {site.id}")
    time.sleep(1)
    site.modules.append(module)

def configure_workflow(site, workflow):
    log(f"Configuring workflow '{workflow}' on Site {site.id}")
    time.sleep(1)
    site.workflows.append(workflow)

# --- Monitoring Endpoint ---
@app.route("/api/monitor", methods=["GET"])
def monitor():
    return jsonify({
        "users": len(USERS),
        "tenants": len(TENANTS),
        "sites": len(SITES),
        "payments": len(PAYMENTS),
        "active_sites": len([s for s in SITES.values() if s.status == SiteStatus.ACTIVE])
    })

# --- Main ---
if __name__ == "__main__":
    app.run(debug=True, port=5000)
