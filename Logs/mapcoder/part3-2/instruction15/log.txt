import pandas as pd
import numpy as np
import matplotlib.pyplot as plt
import seaborn as sns
import os

# Set display options for pandas
pd.set_option('display.max_columns', None)
pd.set_option('display.width', 1000)

# 1. Load the raw customer dataset
# For demonstration, let's assume the file is 'customer_data.csv'
DATA_PATH = 'customer_data.csv'
CLEANED_DATA_PATH = 'customer_data_cleaned.csv'
REPORT_PATH = 'descriptive_report.txt'
VIS_DIR = 'visualizations'

# Create visualizations directory if not exists
os.makedirs(VIS_DIR, exist_ok=True)

# Load data
df = pd.read_csv(DATA_PATH)

# 2. Inspect the dataset
with open(REPORT_PATH, 'w') as report:
    report.write("=== Initial Data Overview ===\n")
    report.write(f"Shape: {df.shape}\n")
    report.write(f"Columns: {list(df.columns)}\n")
    report.write(f"Data Types:\n{df.dtypes}\n\n")
    report.write("Missing Values per Column:\n")

# 3. Data Cleaning

# Remove duplicate records
df = df.drop_duplicates()

# Identify numerical and categorical columns
numerical_cols = df.select_dtypes(include=[np.number]).columns.tolist()
categorical_cols = df.select_dtypes(include=['object', 'category']).columns.tolist()

# Handle missing values
# For numerical columns: fill with median
for col in numerical_cols:
    if df[col].isnull().sum() > 0:
        median_val = df[col].median()
        df[col].fillna(median_val, inplace=True)

# For categorical columns: fill with mode
for col in categorical_cols:
    if df[col].isnull().sum() > 0:
        mode_val = df[col].mode()[0]
        df[col].fillna(mode_val, inplace=True)

# Convert columns to appropriate data types if possible
for col in categorical_cols:
    # Try to convert to category dtype
    df[col] = df[col].astype('category')

# Handle outliers in numerical columns (using IQR method)
for col in numerical_cols:
    Q1 = df[col].quantile(0.25)
    Q3 = df[col].quantile(0.75)
    IQR = Q3 - Q1
    lower = Q1 - 1.5 * IQR
    upper = Q3 + 1.5 * IQR
    # Cap outliers
    df[col] = np.where(df[col] < lower, lower, df[col])
    df[col] = np.where(df[col] > upper, upper, df[col])

# Save cleaned data
df.to_csv(CLEANED_DATA_PATH, index=False)

# 4. Descriptive Statistics
desc_stats = df[numerical_cols].describe().T
desc_stats['median'] = df[numerical_cols].median()
desc_stats = desc_stats[['count', 'mean', 'std', 'min', '25%', '50%', '75%', 'max', 'median']]

# Frequency tables for categorical columns
freq_tables = {}
for col in categorical_cols:
    freq_tables[col] = df[col].value_counts()

# Grouped analysis (if 'region' or 'segment' columns exist)
grouped_stats = {}
for group_col in ['region', 'segment']:
    if group_col in df.columns:
        grouped_stats[group_col] = df.groupby(group_col)[numerical_cols].agg(['count', 'mean', 'median', 'std'])

# 5. Visualizations

for col in numerical_cols:

    sns.histplot(df[col], kde=True, bins=30, color='skyblue')
    plt.title(f'Histogram of {col}')
    plt.xlabel(col)
    plt.ylabel('Frequency')
    plt.tight_layout()
    plt.savefig(f"{VIS_DIR}/hist_{col}.png")
    plt.close()

# Boxplots for numerical features
for col in numerical_cols:
    plt.figure(figsize=(7, 4))
    sns.boxplot(x=df[col], color='lightgreen')
    plt.title(f'Boxplot of {col}')
    plt.xlabel(col)
    plt.tight_layout()
    plt.savefig(f"{VIS_DIR}/box_{col}.png")

# Bar charts for categorical features
for col in categorical_cols:

    df[col].value_counts().plot(kind='bar', color='salmon')
    plt.title(f'Bar Chart of {col}')
    plt.xlabel(col)
    plt.ylabel('Count')
    plt.tight_layout()
    plt.savefig(f"{VIS_DIR}/bar_{col}.png")
    plt.close()

# Pie charts for categorical features (if few categories)
for col in categorical_cols:

        plt.figure(figsize=(6, 6))

        plt.title(f'Pie Chart of {col}')
        plt.ylabel('')
        plt.tight_layout()
        plt.savefig(f"{VIS_DIR}/pie_{col}.png")
        plt.close()

# 6. Summarize findings in a concise report
with open(REPORT_PATH, 'a') as report:

    report.write("=== Frequency Tables (Categorical Columns) ===\n")

    report.write(f"All plots saved in '{VIS_DIR}' directory.\n")
