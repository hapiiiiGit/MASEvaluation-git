# Email listener (simplified)
def check_emails():
    while True:
        mail = imaplib.IMAP4_SSL('imap.example.com')
        mail.login('user@example.com', 'password')
        mail.select('inbox')
        typ, data = mail.search(None, 'UNSEEN')
        for num in data[0].split():
            typ, msg_data = mail.fetch(num, '(RFC822)')
            msg = email.message_from_bytes(msg_data[0][1])
            sender = msg['From']
            subject = msg['Subject']
            received = time.strftime('%Y-%m-%d %H:%M:%S')
            # Log email
            c.execute("INSERT INTO logs (sender, subject, received) VALUES (?, ?, ?)", (sender, subject, received))
            conn.commit()
            # Send reply
            send_reply(sender)
        mail.logout()
        time.sleep(60)

def send_reply(to_addr):
    smtp = smtplib.SMTP('smtp.example.com', 587)
    smtp.starttls()
    smtp.login('user@example.com', 'password')
    msg = MIMEText('Thank you for your email. We will get back to you soon.')
    msg['Subject'] = 'Re: Your Email'
    msg['From'] = 'user@example.com'
    msg['To'] = to_addr
    smtp.sendmail('user@example.com', to_addr, msg.as_string())
    reply_sent = time.strftime('%Y-%m-%d %H:%M:%S')
    c.execute("UPDATE logs SET reply_sent=? WHERE sender=? AND reply_sent IS NULL", (reply_sent, to_addr))
    conn.commit()
    smtp.quit()

# Flask dashboard
app = Flask(__name__)

@app.route('/')
def dashboard():

    return send_file('logs.csv', as_attachment=True)

# Webhook Endpoint
@app.route('/webhook', methods=['POST'])
def webhook():
    data = request.get_json()
    # WhatsApp Business API webhook format
    try:
        for entry in data.get('entry', []):
            for change in entry.get('changes', []):
                value = change.get('value', {})
                for msg in value.get('messages', []):
                    wa_id = msg['from']
                    name = msg.get('profile', {}).get('name', '')
                    message = msg.get('text', {}).get('body', '')
                    timestamp = datetime.utcfromtimestamp(int(msg.get('timestamp', datetime.utcnow().timestamp())))
                    # Detect missed call event (e.g., keyword "missed call" or call event)
                    if 'missed call' in message.lower() or msg.get('type') == 'call':
                        log_event('missed_call', wa_id, name, message, timestamp)
                        send_whatsapp_message(wa_id, "Thank you for your missed call. We'll get back to you soon.")
                    else:
                        log_event('incoming_message', wa_id, name, message, timestamp)
        return jsonify({"status": "received"}), 200
    except Exception as e:
        return jsonify({"error": str(e)}), 400

def log_event(event_type, wa_id, name, message, timestamp=None, scheduled_for=None, status='sent'):
    log = EventLog(
        event_type=event_type,
        wa_id=wa_id,
        name=name,
        message=message,
        timestamp=timestamp or datetime.utcnow(),
        scheduled_for=scheduled_for,
        status=status
    )
    db.session.add(log)
    db.session.commit()
    return log

def send_whatsapp_message(wa_id, message):
    payload = {
        "messaging_product": "whatsapp",
        "to": wa_id,
        "type": "text",
        "text": {"body": message}
    }
    headers = {
        "Authorization": f"Bearer {WHATSAPP_TOKEN}",
        "Content-Type": "application/json"
    }
    try:
        resp = requests.post(WHATSAPP_API_URL, json=payload, headers=headers, timeout=10)
        status = 'sent' if resp.status_code == 200 else 'failed'
    except Exception:
        status = 'failed'
    log_event('outgoing_message', wa_id, '', message, status=status)

# Message Scheduling
def schedule_message_job(log_id, wa_id, message):
    send_whatsapp_message(wa_id, message)
    log = EventLog.query.get(log_id)
    if log:
        log.status = 'sent'
        db.session.commit()

@app.route('/schedule_message', methods=['POST'])
def schedule_message():
    if not check_auth():
        return redirect(url_for('login'))
    wa_id = request.form['wa_id']
    message = request.form['message']
    delay = int(request.form['delay'])
    scheduled_for = datetime.utcnow() + timedelta(minutes=delay)
    log = log_event('scheduled_message', wa_id, '', message, scheduled_for=scheduled_for, status='scheduled')
    scheduler.add_job(
        schedule_message_job,
        'date',
        run_date=scheduled_for,
        args=[log.id, wa_id, message],
        id=f"msg_{log.id}"
    )
    return redirect(url_for('dashboard'))

# Dashboard
@app.route('/')
def dashboard():
    if not check_auth():
        return redirect(url_for('login'))
    logs = EventLog.query.order_by(EventLog.timestamp.desc()).limit(200).all()
    return render_template_string(DASHBOARD_HTML, logs=logs)

# CSV Export
@app.route('/export_csv')
def export_csv():
    if not check_auth():
        return redirect(url_for('login'))
    logs = EventLog.query.order_by(EventLog.timestamp.desc()).all()
    filename = "missedcall_logs.csv"
    with open(filename, 'w', newline='', encoding='utf-8') as f:
        writer = csv.writer(f)
        writer.writerow(['ID', 'Type', 'WA ID', 'Name', 'Message', 'Timestamp', 'Scheduled For', 'Status'])
        for log in logs:
            writer.writerow([
                log.id, log.event_type, log.wa_id, log.name, log.message,
                log.timestamp, log.scheduled_for, log.status
            ])
    return send_file(filename, as_attachment=True)
