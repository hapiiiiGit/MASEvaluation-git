def create_job_listing():
    job_listing = {
        "title": "AI/ML Engineer",
        "level": "Senior",
        "specialization": "Natural Language Processing",
        "degree_requirement": "Master's degree",
        "experience_requirement": "Over 5 years of AI/ML experience",
        "communication_skills": "Excellent English communication skills",
        "technical_skills": [
            "Python",
            "R",
            "TensorFlow"
        ],
        "work_options": "Flexible engagement options",
        "salary_range": "$3K - $10K"
    }
    return job_listing

def display_job_listing(job_listing):
    print("Job Listing:")
    print(f"Title: {job_listing['title']}")
    print(f"Level: {job_listing['level']}")
    print(f"Specialization: {job_listing['specialization']}")
    print(f"Degree Requirement: {job_listing['degree_requirement']}")
    print(f"Experience Requirement: {job_listing['experience_requirement']}")
    print(f"Communication Skills: {job_listing['communication_skills']}")
    print("Technical Skills: " + ", ".join(job_listing['technical_skills']))
    print(f"Work Options: {job_listing['work_options']}")
    print(f"Salary Range: {job_listing['salary_range']}")

if __name__ == "__main__":
    job = create_job_listing()
    display_job_listing(job)
