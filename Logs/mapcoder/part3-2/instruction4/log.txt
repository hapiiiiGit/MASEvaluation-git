# Simple recommendation based on category
def recommend_products(user):
    cart = Cart.objects.get(user=user)

def ad_list(request):
    ads = Ad.objects.all()
    category = request.GET.get('category')
    min_price = request.GET.get('min_price')
    max_price = request.GET.get('max_price')
    location = request.GET.get('location')
    keywords = request.GET.get('keywords')

    q = Q()
    if category:
        q &= Q(category__name__icontains=category)
    if min_price:
        q &= Q(price__gte=min_price)
    if max_price:
        q &= Q(price__lte=max_price)
    if location:
        q &= Q(location__icontains=location)
    if keywords:
        q &= Q(title__icontains=keywords) | Q(description__icontains=keywords)

    ads = ads.filter(q).order_by('-created_at')

    paginator = Paginator(ads, 10)
    page = request.GET.get('page')
    ads_page = paginator.get_page(page)
    return render(request, 'ads/ad_list.html', {'ads': ads_page})

# AI Recommendation API
from django.views.decorators.csrf import csrf_exempt
@csrf_exempt
def recommend_ads(request):
    user_id = request.GET.get('user_id')
    user = get_object_or_404(User, pk=user_id)
    user_ads = Ad.objects.filter(user=user)
    all_ads = Ad.objects.exclude(user=user)

    # Content-based recommendation using TF-IDF and cosine similarity
    ad_texts = [ad.title + ' ' + ad.description for ad in all_ads]
    user_texts = [ad.title + ' ' + ad.description for ad in user_ads]
    if not user_texts or not ad_texts:
        return JsonResponse({'recommendations': []})

    vectorizer = TfidfVectorizer()
    tfidf_matrix = vectorizer.fit_transform(ad_texts + user_texts)
    user_vec = np.mean(tfidf_matrix[-len(user_texts):].toarray(), axis=0)
    ad_vecs = tfidf_matrix[:len(ad_texts)].toarray()
    similarities = cosine_similarity([user_vec], ad_vecs)[0]
    top_indices = np.argsort(similarities)[::-1][:5]
    recommended_ads = [all_ads[i] for i in top_indices]
    data = [{
        'id': ad.id,
        'title': ad.title,
        'description': ad.description,
        'price': str(ad.price),
        'category': ad.category.name,
        'location': ad.location,
    } for ad in recommended_ads]
    return JsonResponse({'recommendations': data})
