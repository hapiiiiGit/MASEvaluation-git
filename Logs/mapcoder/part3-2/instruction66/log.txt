import uuid
import hashlib

# In-memory data storage
USERS = {}
PROVIDERS = {}
ADMINS = {}
SERVICE_REQUESTS = {}
INVOICES = {}

def hash_password(password):
    return hashlib.sha256(password.encode()).hexdigest()

class User:
    def __init__(self, username, password, name, active=True):
        self.id = str(uuid.uuid4())
        self.username = username
        self.password_hash = hash_password(password)
        self.name = name
        self.active = active

    def check_password(self, password):
        return self.password_hash == hash_password(password)

class Customer(User):
    def __init__(self, username, password, name):
        super().__init__(username, password, name)
        USERS[self.id] = self

class Provider(User):
    def __init__(self, username, password, name, skills=None, availability=True):
        super().__init__(username, password, name)
        self.skills = skills if skills else []
        self.ratings = []
        self.availability = availability
        PROVIDERS[self.id] = self

    def update_profile(self, skills=None, availability=None):
        if skills is not None:
            self.skills = skills
        if availability is not None:
            self.availability = availability

    def add_rating(self, rating):
        self.ratings.append(rating)

    def get_average_rating(self):
        return round(sum(self.ratings) / len(self.ratings), 2) if self.ratings else None

class Admin(User):
    def __init__(self, username, password, name):
        super().__init__(username, password, name)
        ADMINS[self.id] = self

class ServiceRequest:
    STATUS_OPTIONS = ['requested', 'accepted', 'in progress', 'completed', 'cancelled']

    def __init__(self, customer_id, provider_id, description, price):
        self.id = str(uuid.uuid4())
        self.customer_id = customer_id
        self.provider_id = provider_id
        self.description = description
        self.price = price
        self.status = 'requested'
        self.status_history = [(self.status, 'Created')]
        SERVICE_REQUESTS[self.id] = self

    def update_status(self, new_status, note=''):
        if new_status in ServiceRequest.STATUS_OPTIONS:
            self.status = new_status
            self.status_history.append((self.status, note))

class Invoice:
    COMMISSION_RATE = 0.05

    def __init__(self, service_request_id):
        self.id = str(uuid.uuid4())
        self.service_request_id = service_request_id
        sr = SERVICE_REQUESTS[service_request_id]
        self.customer_id = sr.customer_id
        self.provider_id = sr.provider_id
        self.amount = sr.price
        self.commission = round(self.amount * Invoice.COMMISSION_RATE, 2)
        self.provider_earning = round(self.amount - self.commission, 2)
        INVOICES[self.id] = self

# Authentication and Session Management
SESSIONS = {}

def register_user(role, username, password, name, skills=None):
    if role == 'customer':
        for u in USERS.values():
            if u.username == username:
                return None, "Username already exists."
        user = Customer(username, password, name)
        return user, "Customer registered."
    elif role == 'provider':
        for p in PROVIDERS.values():
            if p.username == username:
                return None, "Username already exists."
        provider = Provider(username, password, name, skills)
        return provider, "Provider registered."
    elif role == 'admin':
        for a in ADMINS.values():
            if a.username == username:
                return None, "Username already exists."
        admin = Admin(username, password, name)
        return admin, "Admin registered."
    else:
        return None, "Invalid role."

def login(role, username, password):
    if role == 'customer':
        for u in USERS.values():
            if u.username == username and u.check_password(password) and u.active:
                SESSIONS[u.id] = u
                return u, "Login successful."
        return None, "Invalid credentials or inactive account."
    elif role == 'provider':
        for p in PROVIDERS.values():
            if p.username == username and p.check_password(password) and p.active:
                SESSIONS[p.id] = p
                return p, "Login successful."
        return None, "Invalid credentials or inactive account."
    elif role == 'admin':
        for a in ADMINS.values():
            if a.username == username and a.check_password(password) and a.active:
                SESSIONS[a.id] = a
                return a, "Login successful."
        return None, "Invalid credentials or inactive account."
    else:
        return None, "Invalid role."

def logout(user_id):
    if user_id in SESSIONS:
        del SESSIONS[user_id]
        return "Logged out."
    return "Session not found."
