LANGUAGE_PATTERNS = {

def inventory_files(root):
    files = []
    for dirpath, _, filenames in os.walk(root):
        for f in filenames:
            if f.startswith('.'):
                continue
            full_path = os.path.join(dirpath, f)
            if os.path.isfile(full_path):
                files.append(full_path)

def detect_language(filepath):
    ext = os.path.splitext(filepath)[1].lower()
    if ext == '.php' and filepath.endswith('.blade.php'):
        return 'laravel'
    if ext in EXTENSIONS:
        return EXTENSIONS[ext]
    try:

            content = f.read(2048)

                if pattern.search(content):
                    return lang
    except Exception:
        pass
    return None

def move_file(filepath, lang):
    target_dir = LANGUAGE_PATTERNS[lang][1]
    ensure_dir(target_dir)
    filename = os.path.basename(filepath)
    target_path = os.path.join(target_dir, filename)
    # Avoid overwriting
    if os.path.abspath(filepath) != os.path.abspath(target_path):

        return name + ext

        name = ''.join(word.capitalize() for word in re.split(r'[_\-\s]', name))
        return name + ext
    elif lang == 'html':

        return name + ext
    return filename
