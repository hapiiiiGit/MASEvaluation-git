def deliver_items(user_id, cart, payment_id):
    conn = get_db()
    c = conn.cursor()
    c.execute('SELECT * FROM users WHERE id = ?', (user_id,))

    roblox_username = user['roblox_username']

        # Call Roblox bot API
        payload = {

            "item_name": item['name'],

        }

            resp = requests.post(ROBLOX_BOT_API_URL, json=payload, timeout=10)
            if resp.status_code == 200:

        delivery_proof += f"{item['name']} x{qty}: {proof}\n"
        # Update order status and proof
        c.execute('''
