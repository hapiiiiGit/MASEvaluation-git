# --- API Rate Limit Management ---
class RateLimiter:
    def __init__(self, max_calls: int, period: float):
        self.max_calls = max_calls
        self.period = period
        self.lock = threading.Lock()
        self.calls = []
    
    def acquire(self):
        with self.lock:
            now = time.time()
            # Remove calls outside the period
            self.calls = [t for t in self.calls if now - t < self.period]
            if len(self.calls) >= self.max_calls:
                sleep_time = self.period - (now - self.calls[0])
                logging.info(f"Rate limit reached. Sleeping for {sleep_time:.2f} seconds.")
                time.sleep(sleep_time)
                now = time.time()
                self.calls = [t for t in self.calls if now - t < self.period]
            self.calls.append(time.time())

# Alpha Vantage: 5 requests per minute for free tier
rate_limiter = RateLimiter(max_calls=5, period=60)

def fetch_alpha_vantage(symbol: str, interval: str = "1min", outputsize: str = "compact") -> Optional[pd.DataFrame]:

# --- Technical Indicators ---
def moving_average(df: pd.DataFrame, window: int = 14, column: str = "close") -> pd.Series:
    return df[column].rolling(window=window, min_periods=1).mean()

def exponential_moving_average(df: pd.DataFrame, window: int = 14, column: str = "close") -> pd.Series:
    return df[column].ewm(span=window, adjust=False).mean()

def rsi(df: pd.DataFrame, window: int = 14, column: str = "close") -> pd.Series:
    delta = df[column].diff()
    gain = np.where(delta > 0, delta, 0)
    loss = np.where(delta < 0, -delta, 0)
    avg_gain = pd.Series(gain).rolling(window=window, min_periods=window).mean()
    avg_loss = pd.Series(loss).rolling(window=window, min_periods=window).mean()
    rs = avg_gain / (avg_loss + 1e-10)
    rsi = 100 - (100 / (1 + rs))
    rsi[:window] = np.nan
    return pd.Series(rsi, index=df.index)
