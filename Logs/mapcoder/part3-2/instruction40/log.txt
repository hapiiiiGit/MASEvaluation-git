def divide_into_modules(topics):
    modules = []
    for topic in topics:

            modules.append({

def generate_sessions(modules, user_availability, start_date, preferred_pace):
    # Greedy scheduling: assign next available slot per module
    sessions = []
    session_dates = []
    current_date = start_date
    weekday_map = {
        "Monday": 0,
        "Tuesday": 1,
        "Wednesday": 2,
        "Thursday": 3,
        "Friday": 4,
        "Saturday": 5,
        "Sunday": 6
    }
    slot_times = []
    for slot in user_availability:
        day, time = slot.split()
        slot_times.append((weekday_map[day], time))
    module_idx = 0
    while module_idx < len(modules):
        for slot in slot_times:
            # Find next date for this slot
            days_ahead = (slot[0] - current_date.weekday() + 7) % 7
            session_date = current_date + datetime.timedelta(days=days_ahead)
            session_datetime = datetime.datetime.combine(session_date, datetime.datetime.strptime(slot[1], "%H:%M").time())
            if session_datetime.date() < current_date:
                session_datetime += datetime.timedelta(days=7)
            if session_datetime not in session_dates:
                sessions.append({
                    "module": modules[module_idx],
                    "datetime": session_datetime
                })
                session_dates.append(session_datetime)
                module_idx += 1
                if module_idx >= len(modules):
                    break
        current_date += datetime.timedelta(days=7 // preferred_pace)
    return sorted(sessions, key=lambda x: x["datetime"])

def assign_exercises_and_feedback(sessions):

def schedule_support_sessions(sessions, user_availability):
    # Schedule support sessions every 2 modules
    support_sessions = []
    for idx in range(0, len(sessions), 2):
        session = sessions[idx]
        dt = session["datetime"] + datetime.timedelta(days=2)
        # Find next available slot after dt
        for slot in user_availability:
            day, time = slot.split()
            weekday = ["Monday","Tuesday","Wednesday","Thursday","Friday","Saturday","Sunday"].index(day)
            days_ahead = (weekday - dt.weekday() + 7) % 7
            support_date = dt + datetime.timedelta(days=days_ahead)
            support_datetime = datetime.datetime.combine(support_date.date(), datetime.datetime.strptime(time, "%H:%M").time())
            support_sessions.append({
                "for_module": session["module"],
                "datetime": support_datetime,
                "type": "Office Hour / Q&A"
            })
            break
    return support_sessions
