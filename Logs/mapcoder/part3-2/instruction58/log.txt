import csv
import threading
import queue
import requests
from random import choice

def read_urls(filename):
    with open(filename, newline='') as f:
        reader = csv.reader(f)
        return [row[0] for row in reader]

# Step 2: Define proxies and output queue
proxies = ['http://proxy1:port', 'http://proxy2:port', 'http://proxy3:port']
output_queue = queue.Queue()

# Step 3: Scraper function
def scrape_product(url):
    proxy = {'http': choice(proxies), 'https': choice(proxies)}
    try:
        resp = requests.get(url, proxies=proxy, timeout=10)
        # Dummy parsing logic
        product_name = "Extracted Name"
        price = "Extracted Price"
        output_queue.put([product_name, price, url])
    except Exception as e:
        output_queue.put(['ERROR', str(e), url])

# Step 4: Worker thread
def worker(urls):
    for url in urls:
        scrape_product(url)

# Step 5: Write results to CSV in real-time
def writer(output_filename):
    with open(output_filename, 'w', newline='') as f:
        writer = csv.writer(f)
        writer.writerow(['Product Name', 'Price', 'URL'])
        while True:
            row = output_queue.get()
            if row == 'DONE':
                break
            writer.writerow(row)
            f.flush()

# Step 6: Main execution
def main():
    urls = read_urls('input_urls.csv')
    n_threads = 10
    chunk_size = len(urls) // n_threads
    threads = []
    writer_thread = threading.Thread(target=writer, args=('output.csv',))
    writer_thread.start()
    for i in range(n_threads):
        chunk = urls[i*chunk_size:(i+1)*chunk_size]
        t = threading.Thread(target=worker, args=(chunk,))
        threads.append(t)
        t.start()
    for t in threads:
        t.join()
    output_queue.put('DONE')
    writer_thread.join()

Multithreading allows you to run multiple operations concurrently, making it ideal for I/O-bound tasks like web scraping. In Python, the `threading` module is commonly used for this purpose. You can create a pool of worker threads, each responsible for processing a subset of the input data (e.g., ZIP codes, URLs, cities).

### 2. Producer-Consumer Pattern
This pattern is used to decouple the production of data (scraping) from its consumption (writing to file). A thread-safe queue (such as `queue.Queue`) is used to pass data between producer threads (scrapers) and consumer threads (writers). Producers put results into the queue, and consumers take results out and write them to a file, enabling real-time streaming.

### 3. Proxy Rotation
To avoid IP bans and rate limits, you can rotate proxies for each request. Maintain a list of proxies and randomly select one for each HTTP request. This can be done by passing a proxy dictionary to the `requests` library.

import threading
import queue
import random
import requests
import time
import csv
import sys
import traceback
from openpyxl import load_workbook

# CONFIGURABLE PARAMETERS
NUM_WORKER_THREADS = 20
MAX_RETRIES = 3
RETRY_BACKOFF = 2  # seconds
OUTPUT_CSV = 'indeed_jobs.csv'
PROXY_LIST_FILE = 'proxies.txt'  # Each line: http://user:pass@host:port or http://host:port

# 1. Read ZIP codes from .xlsx file
def read_zip_codes(xlsx_file):
    wb = load_workbook(xlsx_file)
    ws = wb.active
    zip_codes = []
    for row in ws.iter_rows(min_row=2, values_only=True):  # Assuming header in row 1
        zip_code = str(row[0]).strip()
        if zip_code and zip_code.isdigit():
            zip_codes.append(zip_code)
    return zip_codes

# 2. Prepare proxy list
def load_proxies(proxy_file):
    proxies = []
    with open(proxy_file, 'r') as f:
        for line in f:
            proxy = line.strip()
            if proxy:
                proxies.append(proxy)
    return proxies

def get_random_proxy(proxies):
    proxy_url = random.choice(proxies)
    return {
        'http': proxy_url,
        'https': proxy_url
    }

# 3. Thread-safe queue for results
result_queue = queue.Queue()
zip_queue = queue.Queue()

# 4. Scraping function
def scrape_indeed_jobs(zip_code, proxies):
    # Example: Scrape first page of jobs for the ZIP code
    url = f'https://www.indeed.com/jobs?q=&l={zip_code}'
    headers = {
        'User-Agent': 'Mozilla/5.0 (Windows NT 10.0; Win64; x64) AppleWebKit/537.36 (KHTML, like Gecko) Chrome/114.0.0.0 Safari/537.36'
    }
    for attempt in range(MAX_RETRIES):
        proxy = get_random_proxy(proxies)
        try:
            resp = requests.get(url, headers=headers, proxies=proxy, timeout=10)
            if resp.status_code == 200:
                # Parse jobs from HTML (simple parsing for demo; use BeautifulSoup for real scraping)
                jobs = []
                html = resp.text
                # Naive parsing: look for job cards
                from bs4 import BeautifulSoup
                soup = BeautifulSoup(html, 'html.parser')
                for card in soup.select('div.job_seen_beacon'):
                    title = card.select_one('h2.jobTitle')
                    company = card.select_one('span.companyName')
                    location = card.select_one('div.companyLocation')
                    summary = card.select_one('div.job-snippet')
                    job_url = card.select_one('a')
                    jobs.append({
                        'zip_code': zip_code,
                        'title': title.get_text(strip=True) if title else '',
                        'company': company.get_text(strip=True) if company else '',
                        'location': location.get_text(strip=True) if location else '',
                        'summary': summary.get_text(strip=True) if summary else '',
                        'url': 'https://www.indeed.com' + job_url['href'] if job_url and job_url.has_attr('href') else ''
                    })
                return jobs
            else:
                time.sleep(RETRY_BACKOFF)
        except Exception as e:
            time.sleep(RETRY_BACKOFF)
    return []

def worker(proxies):
    while True:
        try:
            zip_code = zip_queue.get(timeout=3)
        except queue.Empty:
            break
        try:
            jobs = scrape_indeed_jobs(zip_code, proxies)
            for job in jobs:
                result_queue.put(job)
        except Exception as e:
            traceback.print_exc()
        finally:
            zip_queue.task_done()

def writer_thread_func(fieldnames):
    with open(OUTPUT_CSV, 'w', newline='', encoding='utf-8') as f:
        writer = csv.DictWriter(f, fieldnames=fieldnames)
        writer.writeheader()
        while True:
            job = result_queue.get()
            if job is None:
                break
            writer.writerow(job)
            result_queue.task_done()

def main():
    import argparse
    parser = argparse.ArgumentParser()
    parser.add_argument('--input', required=True, help='Input .xlsx file with ZIP codes')
    parser.add_argument('--proxies', default=PROXY_LIST_FILE, help='Proxy list file')
    parser.add_argument('--threads', type=int, default=NUM_WORKER_THREADS, help='Number of worker threads')
    args = parser.parse_args()

    zip_codes = read_zip_codes(args.input)
    proxies = load_proxies(args.proxies)
    if not proxies:
        print("No proxies loaded. Exiting.")
        sys.exit(1)

    # Fill zip_queue
    for z in zip_codes:
        zip_queue.put(z)

    # Fieldnames for CSV
    fieldnames = ['zip_code', 'title', 'company', 'location', 'summary', 'url']

    # Start writer thread
    writer_thread = threading.Thread(target=writer_thread_func, args=(fieldnames,), daemon=True)
    writer_thread.start()

    # Start worker threads
    threads = []
    for _ in range(args.threads):
        t = threading.Thread(target=worker, args=(proxies,))
        t.start()
        threads.append(t)

    # Wait for all ZIP codes to be processed
    zip_queue.join()
    # Signal writer to finish
    result_queue.put(None)
    writer_thread.join()
    # Wait for all worker threads
    for t in threads:
        t.join()

if __name__ == '__main__':
    main()
