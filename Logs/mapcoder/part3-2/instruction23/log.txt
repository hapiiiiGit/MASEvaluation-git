import openai
import pandas as pd
import sqlite3

def generate_test_scenarios(prompt, api_key):
    openai.api_key = api_key
    response = openai.ChatCompletion.create(
        model="gpt-4",
        messages=[{"role": "user", "content": prompt}]
    )
    scenarios = response['choices'][0]['message']['content']
    return scenarios.split('\n')

# Step 2: Save scenarios to Excel
def save_to_excel(scenarios, filename):
    df = pd.DataFrame({'Test Scenario': scenarios})
    df.to_excel(filename, index=False)

# Step 3: Analyze output files and categorize errors
def analyze_output(output_file):
    df = pd.read_excel(output_file)
    errors = df[df['Result'] == 'Fail']['Error']
    error_categories = errors.value_counts().to_dict()
    return error_categories

# Step 4: Log data into SQLite
def log_to_db(db_file, scenarios, error_categories):
    conn = sqlite3.connect(db_file)
    c = conn.cursor()
    c.execute('CREATE TABLE IF NOT EXISTS logs (scenario TEXT, error_category TEXT, count INTEGER)')
    for scenario in scenarios:
        for category, count in error_categories.items():
            c.execute('INSERT INTO logs VALUES (?, ?, ?)', (scenario, category, count))
    conn.commit()
    conn.close()

# Example usage
api_key = 'YOUR_API_KEY'
prompt = 'Generate 5 test scenarios for login functionality.'

def read_excel(path: str) -> List[Dict[str, Any]]:
    wb = openpyxl.load_workbook(path)
    ws = wb.active
    rows = list(ws.rows)
    if not rows:
        return []
    headers = [cell.value for cell in rows[0]]
    data = []
    for row in rows[1:]:
        data.append({headers[i]: row[i].value for i in range(len(headers))})
    return data
