import scipy.fftpack as fftpack
import logging

# CONFIGURATION PARAMETERS

WINDOW_SIZE = 2048           # Samples per analysis window
HOP_SIZE = 1024              # Overlap for sliding window
SAMPLE_RATE = 8000           # Hz, simulated
BANDPASS = (100, 2000)       # Hz, for noise reduction
N_FEATURES = 20              # Number of FFT bins to use as features
DETECTION_THRESHOLD = 0.7    # Probability threshold for detection
LOG_FILE = "detection_engine.log"
BENCHMARK_RUNS = 10          # Number of test runs for benchmarking

logging.basicConfig(

)

def simulate_sonar_stream(duration_sec=10, submarine_present_prob=0.3):
    """
    Simulate a raw sonar/hydrophone stream.
    Returns: (audio, labels) where audio is a 1D numpy array, labels is a list of (start, end, label)
    """
    n_samples = duration_sec * SAMPLE_RATE
    audio = np.random.normal(0, 0.1, n_samples)
    labels = []
    t = np.arange(n_samples) / SAMPLE_RATE

    # Simulate submarine events
    idx = 0
    while idx < n_samples:
        if random.random() < submarine_present_prob:
            # Submarine event: add low-frequency tone + harmonics
            event_len = random.randint(SAMPLE_RATE//2, SAMPLE_RATE*2)
            freq = random.uniform(150, 400)
            end_idx = min(idx + event_len, n_samples)
            audio[idx:end_idx] += 0.5 * np.sin(2 * np.pi * freq * t[idx:end_idx])
            audio[idx:end_idx] += 0.2 * np.sin(2 * np.pi * (freq*2) * t[idx:end_idx])
            labels.append((idx, end_idx, 1))
            idx = end_idx + random.randint(0, SAMPLE_RATE//2)
        else:
            # No submarine
            idle_len = random.randint(SAMPLE_RATE//2, SAMPLE_RATE*2)
            end_idx = min(idx + idle_len, n_samples)
            labels.append((idx, end_idx, 0))
            idx = end_idx
    return audio, labels

def bandpass_filter(data, lowcut, highcut, fs, order=4):
    sos = signal.butter(order, [lowcut, highcut], btype='band', fs=fs, output='sos')
    return signal.sosfilt(sos, data)

def extract_features(window, fs):
    # FFT features
    fft_vals = np.abs(fftpack.fft(window)[:len(window)//2])
    fft_freqs = np.fft.fftfreq(len(window), 1/fs)[:len(window)//2]
    # Select N_FEATURES bins in the bandpass range
    idx = np.where((fft_freqs >= BANDPASS[0]) & (fft_freqs <= BANDPASS[1]))[0]
    if len(idx) < N_FEATURES:
        idx = np.pad(idx, (0, N_FEATURES-len(idx)), 'wrap')
    selected = fft_vals[idx[:N_FEATURES]]
    # Energy
    energy = np.sum(window**2)
    # Zero-crossing rate
    zcr = ((window[:-1] * window[1:]) < 0).sum() / len(window)
    # Combine features
    features = np.concatenate([selected, [energy, zcr]])
    return features

def generate_dataset(n_samples=100):
    X = []
    y = []
    for _ in range(n_samples):
        audio, labels = simulate_sonar_stream(duration_sec=3)
        audio = bandpass_filter(audio, BANDPASS[0], BANDPASS[1], SAMPLE_RATE)
        for start, end, label in labels:
            for idx in range(start, end-WINDOW_SIZE, HOP_SIZE):
                window = audio[idx:idx+WINDOW_SIZE]
                if len(window) < WINDOW_SIZE:
                    continue
                features = extract_features(window, SAMPLE_RATE)
                X.append(features)
                y.append(label)
    return np.array(X), np.array(y)

class SubmarineDetector:
    def __init__(self):
        self.model = RandomForestClassifier(n_estimators=50, random_state=42)
        self.trained = False

    def train(self, X, y):
        self.model.fit(X, y)
        self.trained = True

    def predict_proba(self, features):
        if not self.trained:
            raise RuntimeError("Model not trained")
        return self.model.predict_proba([features])[0][1]

    def predict(self, features):
        return int(self.predict_proba(features) > DETECTION_THRESHOLD)

class CommunicationRelay:
    """
    State machine for switching between acoustic and RF links.
    """
    def __init__(self):
        self.state = "ACOUSTIC"  # or "RF"
        self.link_status = {"ACOUSTIC": True, "RF": True}
        self.last_switch_time = time.time()

    def update_link_status(self, acoustic_ok, rf_ok):
        self.link_status["ACOUSTIC"] = acoustic_ok
        self.link_status["RF"] = rf_ok

    def select_link(self):
        prev_state = self.state
        if self.link_status["RF"]:
            self.state = "RF"
        elif self.link_status["ACOUSTIC"]:
            self.state = "ACOUSTIC"
        else:
            self.state = "NONE"
        if self.state != prev_state:
            logging.info(f"Communication link switched from {prev_state} to {self.state}")
            self.last_switch_time = time.time()
        return self.state

class DetectionEngine:
    def __init__(self, detector, relay):
        self.detector = detector
        self.relay = relay
        self.log_queue = Queue()
        self.latencies = []
        self.detections = []
        self.running = False

    def process_stream(self, audio, labels=None, simulate_relay=True):
        """
        Process a stream of audio data in real-time.
        If labels are provided, compute accuracy.
        """
        audio = bandpass_filter(audio, BANDPASS[0], BANDPASS[1], SAMPLE_RATE)
        n_windows = (len(audio) - WINDOW_SIZE) // HOP_SIZE
        detection_results = []
        label_idx = 0
        for i in range(n_windows):
            start = i * HOP_SIZE
            window = audio[start:start+WINDOW_SIZE]
            features = extract_features(window, SAMPLE_RATE)
            t0 = time.time()
            proba = self.detector.predict_proba(features)
            detected = int(proba > DETECTION_THRESHOLD)
            latency = time.time() - t0
            self.latencies.append(latency)
            detection_results.append(detected)
            # Logging
            log_msg = f"Window {i}: Detected={detected} (proba={proba:.2f}), Latency={latency*1000:.2f}ms"
            logging.info(log_msg)
            self.log_queue.put(log_msg)
            # Simulate relay status changes
            if simulate_relay and i % 50 == 0:
                # Randomly drop links
                acoustic_ok = random.random() > 0.1
                rf_ok = random.random() > 0.2
                self.relay.update_link_status(acoustic_ok, rf_ok)
            link = self.relay.select_link()
            # Log relay state
            logging.info(f"Active link: {link}")
            # For benchmarking
            if labels is not None:
                # Find label for this window
                while label_idx < len(labels) and start >= labels[label_idx][1]:
                    label_idx += 1
                if label_idx < len(labels):
                    true_label = labels[label_idx][2]
                    self.detections.append((detected, true_label))
        return detection_results

    def get_accuracy(self):
        if not self.detections:
            return None
        y_pred, y_true = zip(*self.detections)
        return accuracy_score(y_true, y_pred)

    def get_avg_latency(self):
        if not self.latencies:
            return None
        return np.mean(self.latencies)
