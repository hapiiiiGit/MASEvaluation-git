def create_access_token(data: dict, expires_delta: Optional[timedelta] = None):
    to_encode = data.copy()
    expire = datetime.utcnow() + (expires_delta or timedelta(minutes=ACCESS_TOKEN_EXPIRE_MINUTES))
    to_encode.update({"exp": expire})
    return jwt.encode(to_encode, SECRET_KEY, algorithm=ALGORITHM)

def get_current_user(token: str = Depends(oauth2_scheme), db: Session = Depends(get_db)):
    try:
        payload = jwt.decode(token, SECRET_KEY, algorithms=[ALGORITHM])
        username: str = payload.get("sub")
        if username is None:

        user = db.query(User).filter(User.username == username).first()
        if user is None:
            raise HTTPException(status_code=401, detail="User not found")
        return user
    except jwt.PyJWTError:
        raise HTTPException(status_code=401, detail="Invalid token")

@app.get("/analytics/heatmap/{survey_id}")
def heatmap_data(survey_id: int, db: Session = Depends(get_db)):
    questions = db.query(Question).filter(Question.survey_id == survey_id, Question.type == "rating").all()
    responses = db.query(Response).filter(Response.survey_id == survey_id).all()
    times = [r.submitted_at.strftime("%Y-%m-%d") for r in responses]
    categories = list(set([q.category for q in questions if q.category]))
    matrix = {cat: {t: [] for t in times} for cat in categories}
    for r in responses:
        for q in questions:
            ans = db.query(Answer).filter(Answer.response_id == r.id, Answer.question_id == q.id).first()
            if ans and ans.value.replace('.', '', 1).isdigit():
                matrix[q.category][r.submitted_at.strftime("%Y-%m-%d")].append(float(ans.value))
    heatmap = []
    for cat in categories:
        row = []
        for t in times:
            vals = matrix[cat][t]
            row.append(np.mean(vals) if vals else 0)
        heatmap.append(row)
    return {"categories": categories, "times": times, "matrix": heatmap}
