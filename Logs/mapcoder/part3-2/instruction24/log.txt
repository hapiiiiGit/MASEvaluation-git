pdf_files = [f for f in os.listdir(pdf_folder) if f.endswith('.pdf')]
keywords = ['invoice', 'payment', 'due']

report = []

for pdf_file in pdf_files:

        found = {kw: text.lower().count(kw) for kw in keywords}

def authenticate(login_url, username, password, extra_payload=None):
    session = requests.Session()
    payload = {'username': username, 'password': password}
    if extra_payload:
        payload.update(extra_payload)
    resp = session.post(login_url, data=payload)
    if resp.ok and "logout" in resp.text.lower():

    else:

        raise Exception("Authentication failed.")

def fetch_and_parse(session, url):
    resp = session.get(url)
    resp.raise_for_status()
    return BeautifulSoup(resp.text, 'html.parser')

def extract_data_from_page(soup, fields):
    # fields: list of (label, css_selector)
    data = {}
    for label, selector in fields:
        el = soup.select_one(selector)
        data[label] = el.get_text(strip=True) if el else ""
    return data

def find_pdf_links(soup, base_url=None):
    pdf_links = []
    for a in soup.find_all('a', href=True):
        href = a['href']
        if href.lower().endswith('.pdf'):
            if base_url and not href.startswith('http'):
                href = requests.compat.urljoin(base_url, href)
            pdf_links.append(href)
    return pdf_links

def download_pdf(session, url, dest_folder):
    filename = get_filename_from_url(url)
    dest_path = os.path.join(dest_folder, filename)
    with session.get(url, stream=True) as r:
        r.raise_for_status()
        with open(dest_path, 'wb') as f:
            for chunk in r.iter_content(chunk_size=8192):
                f.write(chunk)
    return dest_path

def extract_pdf_text(pdf_path, max_pages=2):
    try:
        with pdfplumber.open(pdf_path) as pdf:
            text = ""
            for i, page in enumerate(pdf.pages):
                if i >= max_pages:
                    break
                text += page.extract_text() or ""
            return text.strip()
    except Exception as e:
        logging.error(f"Error reading PDF {pdf_path}: {e}")
        return ""
