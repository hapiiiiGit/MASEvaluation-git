def find_py_files(directory: str) -> List[str]:
    py_files = []
    for root, _, files in os.walk(directory):
        for f in files:
            if f.endswith('.py'):
                py_files.append(os.path.join(root, f))
    return py_files

def is_syntax_error_in_funcdef(error: SyntaxError) -> bool:
    # Check if the error is likely in a function definition
    msg = error.msg.lower()
    return ('invalid syntax' in msg or 'expected' in msg or 'unexpected' in msg) and (
        'def' in error.text if error.text else False
    )

def locate_faulty_funcdef(lines: List[str], error: SyntaxError) -> Tuple[int, str]:
    # Try to find the faulty function definition line
    lineno = error.lineno - 1 if error.lineno else 0
    for i in range(max(0, lineno-2), min(len(lines), lineno+3)):
        if re.match(r'^\s*def\s', lines[i]):
            return i, lines[i]
    # Fallback: return the error line
    return lineno, lines[lineno] if 0 <= lineno < len(lines) else ('', '')

def correct_funcdef_line(line: str) -> str:
    orig_line = line
    # Ensure 'def' starts the line (allow indentation)
    m = re.match(r'^(\s*)def\s+([a-zA-Z_][a-zA-Z0-9_]*)\s*(\(.*\))\s*(.*)', line)
    if m:
        indent, funcname, params, rest = m.groups()
        # Ensure colon at end
        if not rest.strip().endswith(':'):
            rest = rest.rstrip()
            if not rest.endswith(':'):
                rest += ':'
        # Fix parameter list: ensure balanced parentheses
        if params.count('(') != params.count(')'):
            # Try to balance parentheses
            diff = params.count('(') - params.count(')')
            if diff > 0:
                params += ')' * diff
            elif diff < 0:
                params = '(' * (-diff) + params
        # Remove invalid characters in parameter list
        params = re.sub(r'[^\(\),a-zA-Z0-9_=\s\*\[\]\.]', '', params)
        # Remove duplicate colons
        rest = re.sub(r':+', ':', rest)
        return f"{indent}def {funcname}{params} {rest}"
    # If missing colon at end
    if re.match(r'^\s*def\s+[a-zA-Z_][a-zA-Z0-9_]*\s*\(.*\)\s*$', line):
        return line.rstrip() + ':'
    # If missing parentheses
    m = re.match(r'^(\s*)def\s+([a-zA-Z_][a-zA-Z0-9_]*)\s*(:?)', line)
    if m:
        indent, funcname, colon = m.groups()
        return f"{indent}def {funcname}():{'' if colon else ''}"
    return line

def process_file(filepath: str, log_entries: List[str]) -> bool:
    with open(filepath, 'r', encoding='utf-8') as f:
        lines = f.readlines()
    code = ''.join(lines)
    ok, error = try_parse_code(code)
    if ok:
        return False  # No syntax error
    if not is_syntax_error_in_funcdef(error):
        return False  # Not a function definition error
    # Backup original file
    shutil.copy2(filepath, filepath + BACKUP_SUFFIX)
    changed = False
    for _ in range(3):  # Try up to 3 corrections
        idx, orig_line = locate_faulty_funcdef(lines, error)
        if not orig_line.strip():
            break
        corrected_line = correct_funcdef_line(orig_line)
        if corrected_line != orig_line:
            lines[idx] = corrected_line + '\n'
            changed = True
            log_entries.append(
                f"{filepath}: Line {idx+1}\n"
                f"  Original: {orig_line.rstrip()}\n"
                f"  Corrected: {corrected_line.rstrip()}\n"
            )
        code = ''.join(lines)
        ok, error = try_parse_code(code)
        if ok:
            break
        if not is_syntax_error_in_funcdef(error):
            break
    if changed:
        with open(filepath, 'w', encoding='utf-8') as f:
            f.writelines(lines)
    return changed

def run_script(filepath: str) -> Tuple[bool, str]:
    import subprocess
    try:
        result = subprocess.run([sys.executable, filepath], capture_output=True, timeout=10)
        if result.returncode == 0:
            return True, ''
        else:
            return False, result.stderr.decode()
    except Exception as e:
        return False, str(e)

def main():
    py_files = find_py_files(TARGET_DIR)
    log_entries = []
    summary = []
    for f in py_files:
        try:
            changed = process_file(f, log_entries)
            if changed:
                ok, err = run_script(f)
                summary.append(f"{f}: {'Fixed and runs OK' if ok else 'Fixed but error: ' + err.strip()}")
        except Exception as e:
            log_entries.append(f"{f}: Exception: {traceback.format_exc()}")
    # Write log
    with open(LOG_FILE, 'w', encoding='utf-8') as logf:
        logf.write('\n'.join(log_entries))
        logf.write('\n\nSummary:\n')
        logf.write('\n'.join(summary))
    print(f"Correction complete. See {LOG_FILE} for details.")
