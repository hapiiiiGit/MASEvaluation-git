def register_user(username, password):
    if username in users_db:

    salt = uuid.uuid4().hex

    users_db[username] = {'password': hashed_pw, 'salt': salt}

def login_user(username, password):
    user = users_db.get(username)
    if not user:

    if hashed_pw == user['password']:
        session_token = uuid.uuid4().hex
        sessions[session_token] = username
        return f"Login successful. Session token: {session_token}"
    return "Invalid credentials"

# Step 5: Portfolio CRUD
def add_stock(session_token, stock):
    username = sessions.get(session_token)
    if not username:
        return "Invalid session"

    return "Stock added"

def get_portfolio(session_token):
    username = sessions.get(session_token)
    if not username:
        return "Invalid session"
    return portfolios[username]

# Step 6: Logout
def logout_user(session_token):
    if session_token in sessions:

    return "Invalid session"

from flask import Flask, request, jsonify
import hashlib
import os
import uuid
from functools import wraps

app = Flask(__name__)

# In-memory stores
users = {}  # username: {salt, password_hash, api_key, funds}
sessions = {}  # session_token: username

# Helper functions
def hash_password(password, salt):
    return hashlib.sha256((password + salt).encode()).hexdigest()

def generate_salt():
    return os.urandom(16).hex()

def generate_api_key():
    return uuid.uuid4().hex

def generate_session_token():
    return uuid.uuid4().hex

def validate_api_key(username, api_key):
    return users.get(username, {}).get('api_key') == api_key

def validate_session_token(token):
    return token in sessions

def get_username_from_token(token):
    return sessions.get(token)

# Decorator for protected endpoints
def require_auth(f):
    @wraps(f)
    def decorated(*args, **kwargs):
        session_token = request.headers.get('Session-Token')
        api_key = request.headers.get('API-Key')
        if not session_token or not api_key:
            return jsonify({'error': 'Missing session token or API key'}), 401
        if not validate_session_token(session_token):
            return jsonify({'error': 'Invalid or expired session token'}), 401
        username = get_username_from_token(session_token)
        if not username or not validate_api_key(username, api_key):
            return jsonify({'error': 'Invalid API key'}), 401
        request.username = username
        return f(*args, **kwargs)
    return decorated

# Endpoints
