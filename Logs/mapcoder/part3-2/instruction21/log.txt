from PIL import Image
import json

def png_to_json(png_path, json_path):
    img = Image.open(png_path).convert('RGBA')
    pixels = list(img.getdata())
    width, height = img.size
    data = {
        'width': width,
        'height': height,
        'pixels': [list(pixel) for pixel in pixels]
    }
    with open(json_path, 'w') as f:
        json.dump(data, f)

def json_to_png(json_path, png_path):
    with open(json_path, 'r') as f:
        data = json.load(f)
    img = Image.new('RGBA', (data['width'], data['height']))
    img.putdata([tuple(pixel) for pixel in data['pixels']])
    img.save(png_path)

def parse_nba2k26_asset(filepath):
    with open(filepath, 'rb') as f:
        data = f.read()
    offset = 0

    # Header (magic, version)
    magic = data[offset:offset+4].decode('ascii')
    offset += 4
    version, = struct.unpack('<I', data[offset:offset+4])
    offset += 4

    # Vertices
    vertex_count, = struct.unpack('<I', data[offset:offset+4])
    offset += 4
    vertices = []
    for _ in range(vertex_count):
        x, y, z, nx, ny, nz, u, v = struct.unpack('<8f', data[offset:offset+32])
        offset += 32
        vertices.append({'pos': [x, y, z], 'normal': [nx, ny, nz], 'uv': [u, v]})

    # Faces
    face_count, = struct.unpack('<I', data[offset:offset+4])
    offset += 4
    faces = []
    for _ in range(face_count):
        a, b, c = struct.unpack('<3I', data[offset:offset+12])
        offset += 12
        faces.append([a, b, c])

    # Bones
    bone_count, = struct.unpack('<I', data[offset:offset+4])
    offset += 4
    bones = []
    for _ in range(bone_count):
        name = data[offset:offset+32].decode('utf-8').rstrip('\x00')
        offset += 32
        parent, = struct.unpack('<i', data[offset:offset+4])
        offset += 4
        tx, ty, tz = struct.unpack('<3f', data[offset:offset+12])
        offset += 12
        bones.append({'name': name, 'parent': parent, 'pos': [tx, ty, tz]})

    # Weights
    weight_count, = struct.unpack('<I', data[offset:offset+4])
    offset += 4
    weights = []
    for _ in range(weight_count):
        vertex_idx, bone_idx, weight = struct.unpack('<IIf', data[offset:offset+12])
        offset += 12
        weights.append({'vertex': vertex_idx, 'bone': bone_idx, 'weight': weight})

    # Texture (DDS)
    tex_size, = struct.unpack('<I', data[offset:offset+4])
    offset += 4
    texture_data = data[offset:offset+tex_size]
    offset += tex_size

    return {
        'header': {'magic': magic, 'version': version},
        'vertices': vertices,
        'faces': faces,
        'bones': bones,
        'weights': weights,
        'texture': texture_data
    }

def export_json(asset, out_json):
    json_data = {
        'header': asset['header'],

        'faces': asset['faces'],

def export_dds(texture_data, out_dds):

def import_json_dds(json_path, dds_path):
    with open(json_path, 'r') as f:
        data = json.load(f)

    # Create mesh
    mesh = bpy.data.meshes.new("NBA2k26Mesh")
    verts = [v['pos'] for v in data['vertices']]
    faces = data['faces']
    mesh.from_pydata(verts, [], faces)
    mesh.update()

    # Normals
    for i, v in enumerate(data['vertices']):
        mesh.vertices[i].normal = v['normal']

    # UVs
    if mesh.uv_layers.active is None:
        mesh.uv_layers.new()
    uv_layer = mesh.uv_layers.active.data
    for i, face in enumerate(mesh.polygons):
        for j, idx in enumerate(face.vertices):
            uv_layer[face.loop_start + j].uv = data['vertices'][idx]['uv']

    # Create object
    obj = bpy.data.objects.new("NBA2k26Player", mesh)
    bpy.context.collection.objects.link(obj)
    bpy.context.view_layer.objects.active = obj

    # Armature
    arm = bpy.data.armatures.new("NBA2k26Armature")
    arm_obj = bpy.data.objects.new("NBA2k26ArmatureObj", arm)
    bpy.context.collection.objects.link(arm_obj)
    bpy.context.view_layer.objects.active = arm_obj
    bpy.ops.object.mode_set(mode='EDIT')
    bone_map = {}
    for bone in data['bones']:
        b = arm.edit_bones.new(bone['name'])
        b.head = bone['pos']
        b.tail = [bone['pos'][0], bone['pos'][1], bone['pos'][2] + 0.1]
        bone_map[bone['name']] = b
    for i, bone in enumerate(data['bones']):
        if bone['parent'] != -1:
            parent_name = data['bones'][bone['parent']]['name']
            bone_map[bone['name']].parent = bone_map[parent_name]
    bpy.ops.object.mode_set(mode='OBJECT')

    # Parent mesh to armature
    obj.parent = arm_obj
    modifier = obj.modifiers.new(type='ARMATURE', name='Armature')
    modifier.object = arm_obj

    # Vertex Groups (weights)
    for bone in data['bones']:
        obj.vertex_groups.new(name=bone['name'])
    for w in data['weights']:
        bone_name = data['bones'][w['bone']]['name']
        obj.vertex_groups[bone_name].add([w['vertex']], w['weight'], 'ADD')

    # Material and Texture
    mat = bpy.data.materials.new(name="NBA2k26Material")
    mat.use_nodes = True
    bsdf = mat.node_tree.nodes.get("Principled BSDF")
    tex_image = mat.node_tree.nodes.new('ShaderNodeTexImage')
    tex_image.image = bpy.data.images.load(dds_path)
    mat.node_tree.links.new(bsdf.inputs['Base Color'], tex_image.outputs['Color'])
    obj.data.materials.append(mat)

def export_json_dds(obj, arm_obj, out_json, out_dds):
    mesh = obj.data
    vertices = []
    for v in mesh.vertices:
        uv = [0.0, 0.0]
        if mesh.uv_layers.active:
            for loop in mesh.loops:
                if loop.vertex_index == v.index:
                    uv = mesh.uv_layers.active.data[loop.index].uv[:]
                    break
        vertices.append({
            'pos': list(v.co),
            'normal': list(v.normal),
            'uv': list(uv)
        })
    faces = [list(p.vertices) for p in mesh.polygons]

    # Bones
    bones = []
    for i, bone in enumerate(arm_obj.data.bones):
        parent = -1
        if bone.parent:
            parent = [b.name for b in arm_obj.data.bones].index(bone.parent.name)
        bones.append({
            'name': bone.name,
            'parent': parent,
            'pos': list(bone.head_local)
        })

    # Weights
    weights = []
    for vg in obj.vertex_groups:
        bone_idx = [b['name'] for b in bones].index(vg.name)
        for v in mesh.vertices:
            for g in v.groups:
                if g.group == vg.index:
                    weights.append({'vertex': v.index, 'bone': bone_idx, 'weight': g.weight})

    # Texture
    mat = obj.data.materials[0]
    tex_image = None
    for node in mat.node_tree.nodes:
        if node.type == 'TEX_IMAGE':
            tex_image = node.image
            break
    texture_data = None
    if tex_image:
        tex_path = bpy.path.abspath(tex_image.filepath)
        with open(tex_path, 'rb') as f:
            texture_data = f.read()
        with open(out_dds, 'wb') as f:
            f.write(texture_data)

    json_data = {
        'header': {'magic': 'NBA2', 'version': 1},
        'vertices': vertices,
        'faces': faces,
        'bones': bones,
        'weights': weights
    }
    with open(out_json, 'w') as f:
        json.dump(json_data, f, indent=2)
