def get_headers(payload=None):
    nonce = str(int(time.time() * 1000))
    if payload:
        payload_str = json.dumps(payload, separators=(',', ':'))
        signature = base64.b64encode(
            hmac.new(
                API_SECRET.encode(),
                payload_str.encode(),
                hashlib.sha256
            ).digest()
        ).decode()
    else:
        signature = ''
    headers = {
        'Content-Type': 'application/json',
        'X-AUTH-APIKEY': API_KEY,
        'X-AUTH-SIGNATURE': signature,
        'X-AUTH-NONCE': nonce
    }
    return headers

def fetch_historical_ohlcv(pair, limit=500, interval='1m'):
    url = f"{BASE_URL}/exchange/v1/markets/ohlc"
    params = {
        "pair": pair,
        "interval": interval,
        "limit": limit
    }
    resp = requests.get(url, params=params)
    data = resp.json()
    df = pd.DataFrame(data['candles'], columns=['timestamp', 'open', 'high', 'low', 'close', 'volume'])
    df['close'] = df['close'].astype(float)
    df['open'] = df['open'].astype(float)
    df['high'] = df['high'].astype(float)
    df['low'] = df['low'].astype(float)
    df['volume'] = df['volume'].astype(float)
    df['timestamp'] = pd.to_datetime(df['timestamp'], unit='ms')
    return df

def fetch_latest_price(pair):
    url = f"{BASE_URL}/exchange/v1/markets_detail"
    resp = requests.get(url)
    data = resp.json()
    for item in data['markets']:
        if item['pair'] == pair:
            return float(item['last_price'])
    return None

def place_order(pair, side, size, order_type='market', price=None):
    url = f"{BASE_URL}/exchange/v1/orders/create"
    payload = {
        "order_type": order_type,
        "side": side,
        "market": pair,
        "quantity": str(size),
        "leverage": "1"
    }
    if order_type == 'limit' and price:
        payload['price_per_unit'] = str(price)

def calculate_ma(df, period, ema=True):
    if ema:
        return df['close'].ewm(span=period, adjust=False).mean()
    else:
        return df['close'].rolling(window=period).mean()

def generate_signal(df):
    df['short_ma'] = calculate_ma(df, SHORT_MA_PERIOD, USE_EMA)
    df['long_ma'] = calculate_ma(df, LONG_MA_PERIOD, USE_EMA)
    if df['short_ma'].iloc[-2] < df['long_ma'].iloc[-2] and df['short_ma'].iloc[-1] > df['long_ma'].iloc[-1]:
        return 'buy'
    elif df['short_ma'].iloc[-2] > df['long_ma'].iloc[-2] and df['short_ma'].iloc[-1] < df['long_ma'].iloc[-1]:
        return 'sell'
    else:
        return 'hold'

# ================== RISK MANAGEMENT ==================

def calculate_stop_loss(entry_price, side):
    if side == 'buy':
        return entry_price * (1 - STOP_LOSS_PCT)
    else:
        return entry_price * (1 + STOP_LOSS_PCT)

def calculate_take_profit(entry_price, side):
    if side == 'buy':
        return entry_price * (1 + TAKE_PROFIT_PCT)
    else:
        return entry_price * (1 - TAKE_PROFIT_PCT)

def backtest(df):
    df['short_ma'] = calculate_ma(df, SHORT_MA_PERIOD, USE_EMA)
    df['long_ma'] = calculate_ma(df, LONG_MA_PERIOD, USE_EMA)
    position = None
    entry_price = 0
    trades = []
    for i in range(1, len(df)):
        signal = None
        if df['short_ma'].iloc[i-1] < df['long_ma'].iloc[i-1] and df['short_ma'].iloc[i] > df['long_ma'].iloc[i]:
            signal = 'buy'
        elif df['short_ma'].iloc[i-1] > df['long_ma'].iloc[i-1] and df['short_ma'].iloc[i] < df['long_ma'].iloc[i]:
            signal = 'sell'
        else:
            signal = 'hold'
        price = df['close'].iloc[i]
        timestamp = df['timestamp'].iloc[i]
        if signal == 'buy' and not position:
            position = 'long'
            entry_price = price
            stop_loss = calculate_stop_loss(entry_price, 'buy')
            take_profit = calculate_take_profit(entry_price, 'buy')
            trades.append({'timestamp': str(timestamp), 'action': 'buy', 'price': price})
        elif signal == 'sell' and not position:
            position = 'short'
            entry_price = price
            stop_loss = calculate_stop_loss(entry_price, 'sell')
            take_profit = calculate_take_profit(entry_price, 'sell')
            trades.append({'timestamp': str(timestamp), 'action': 'sell', 'price': price})
        elif position == 'long':
            if price <= stop_loss or price >= take_profit or signal == 'sell':
                trades.append({'timestamp': str(timestamp), 'action': 'close_long', 'price': price})
                position = None
        elif position == 'short':
            if price >= stop_loss or price <= take_profit or signal == 'buy':
                trades.append({'timestamp': str(timestamp), 'action': 'close_short', 'price': price})
                position = None
    return trades
