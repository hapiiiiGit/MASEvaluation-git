class Course:
    def __init__(self, course_id, name, schedule, capacity):
        self.course_id = course_id
        self.name = name
        self.schedule = schedule  # e.g., {'Mon': '10-12', 'Wed': '10-12'}
        self.capacity = capacity
        self.registered_students = set()

class Student:
    def __init__(self, student_id, name):
        self.student_id = student_id
        self.name = name
        self.courses = set()

def has_conflict(student, new_course, courses_dict):
    for cid in student.courses:
        course = courses_dict[cid]
        for day, time in course.schedule.items():
            if day in new_course.schedule and new_course.schedule[day] == time:
                return True
    return False

def register_course(student, course, courses_dict):
    if len(course.registered_students) >= course.capacity:
        return "Course is full."
    if has_conflict(student, course, courses_dict):
        return "Schedule conflict."
    course.registered_students.add(student.student_id)
    student.courses.add(course.course_id)
    return "Registered successfully."

def drop_course(student, course):
    if course.course_id in student.courses:
        student.courses.remove(course.course_id)
        course.registered_students.remove(student.student_id)
        return "Dropped successfully."
    return "Not registered in this course."

# Example usage:
courses = {
    'CS101': Course('CS101', 'Intro to CS', {'Mon': '10-12', 'Wed': '10-12'}, 2),
    'MATH201': Course('MATH201', 'Calculus', {'Tue': '10-12', 'Thu': '10-12'}, 2)
}
student = Student('S001', 'Alice')
print(register_course(student, courses['CS101'], courses))
print(register_course(student, courses['MATH201'], courses))
print(drop_course(student, courses['CS101']))

from datetime import datetime, timedelta

class Book:
    def __init__(self, book_id, title, author):
        self.book_id = book_id
        self.title = title
        self.author = author
        self.is_borrowed = False

class User:
    def __init__(self, user_id, name, max_books=3):
        self.user_id = user_id
        self.name = name
        self.borrowed_books = {}  # book_id: due_date
        self.max_books = max_books

def search_books(books, query):
    return [book for book in books if query.lower() in book.title.lower() or query.lower() in book.author.lower()]

def borrow_book(user, book):
    if len(user.borrowed_books) >= user.max_books:
        return "Borrowing limit reached."
    if book.is_borrowed:
        return "Book is already borrowed."
    due_date = datetime.now() + timedelta(days=14)
    user.borrowed_books[book.book_id] = due_date
    book.is_borrowed = True
    return f"Book borrowed. Due date: {due_date.strftime('%Y-%m-%d')}"

def return_book(user, book):
    if book.book_id in user.borrowed_books:
        del user.borrowed_books[book.book_id]
        book.is_borrowed = False
        return "Book returned."
    return "Book not borrowed by user."

# Example usage:
books = [Book(1, "1984", "George Orwell"), Book(2, "Python Programming", "John Doe")]
user = User(1, "Bob")

print(return_book(user, books[1]))

from typing import List, Dict, Any

# Data model for a candidate
class Candidate:
    def __init__(self, name: str, email: str, languages: List[str],
                 problem_solving_experience: str, teamwork_experience: str,
                 development_approach: str):
        self.name = name
        self.email = email
        self.languages = [lang.strip().lower() for lang in languages]
        self.problem_solving_experience = problem_solving_experience
        self.teamwork_experience = teamwork_experience
        self.development_approach = development_approach
        self.score = 0
        self.passed_filter = False

    def meets_minimum_requirements(self) -> bool:
        required_languages = {'python', 'java', 'c++'}
        has_languages = required_languages.issubset(set(self.languages))
        has_problem_solving = len(self.problem_solving_experience.strip()) > 20
        has_teamwork = len(self.teamwork_experience.strip()) > 20
        return has_languages and has_problem_solving and has_teamwork

    def calculate_score(self):
        # Scoring rules
        required_languages = {'python', 'java', 'c++'}
        # 2 points for each required language known
        self.score += sum(2 for lang in required_languages if lang in self.languages)
        # 1 point for each additional language
        self.score += sum(1 for lang in self.languages if lang not in required_languages)
        # Problem-solving experience: up to 5 points
        self.score += min(5, len(self.problem_solving_experience.strip()) // 40)
        # Teamwork experience: up to 5 points
        self.score += min(5, len(self.teamwork_experience.strip()) // 40)
        # Development approach: up to 3 points for clarity and effectiveness
        self.score += min(3, len(self.development_approach.strip()) // 50)

    def to_dict(self) -> Dict[str, Any]:
        return {
            'name': self.name,
            'email': self.email,
            'score': self.score,
            'passed_filter': self.passed_filter
        }

class RecruitmentProcess:
    def __init__(self):
        self.candidates: List[Candidate] = []

    def add_candidate(self, candidate: Candidate):
        self.candidates.append(candidate)

    def filter_and_score(self):
        for candidate in self.candidates:
            candidate.passed_filter = candidate.meets_minimum_requirements()
            if candidate.passed_filter:
                candidate.calculate_score()

    def rank_candidates(self) -> List[Candidate]:
        # Only consider candidates who passed the filter
        filtered = [c for c in self.candidates if c.passed_filter]
        return sorted(filtered, key=lambda c: c.score, reverse=True)

    def shortlist(self, top_n: int = 5) -> List[Candidate]:
        ranked = self.rank_candidates()
        return ranked[:top_n]

    def reviewer_interface(self):
        print("\n=== Reviewer Interface ===")
        self.filter_and_score()
        ranked = self.rank_candidates()
