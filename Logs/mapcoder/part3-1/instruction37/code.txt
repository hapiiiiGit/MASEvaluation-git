class Freelancer:
    def __init__(self, name, skills, experience):
        self.name = name
        self.skills = skills
        self.experience = experience

    def update_skills(self, new_skills):
        self.skills = new_skills

    def update_experience(self, new_experience):
        self.experience = new_experience


class Profile:
    def __init__(self, freelancer, contact_info):
        self.freelancer = freelancer
        self.contact_info = contact_info
        self.services_offered = []

    def add_service(self, service):
        self.services_offered.append(service)

    def update_contact_info(self, new_contact_info):
        self.contact_info = new_contact_info


class Project:
    def __init__(self, title, description, client_details):
        self.title = title
        self.description = description
        self.client_details = client_details
        self.status = 'Pending'

    def update_status(self, new_status):
        self.status = new_status


class FreelancerManager:
    def __init__(self):
        self.freelancers = {}

    def create_freelancer(self, name, skills, experience, contact_info):
        freelancer = Freelancer(name, skills, experience)
        profile = Profile(freelancer, contact_info)
        self.freelancers[name] = profile

    def update_freelancer(self, name, skills=None, experience=None, contact_info=None):
        if name in self.freelancers:
            if skills:
                self.freelancers[name].freelancer.update_skills(skills)
            if experience:
                self.freelancers[name].freelancer.update_experience(experience)
            if contact_info:
                self.freelancers[name].update_contact_info(contact_info)

    def delete_freelancer(self, name):
        if name in self.freelancers:
            del self.freelancers[name]


class ProjectManager:
    def __init__(self):
        self.projects = {}

    def create_project(self, title, description, client_details):
        project = Project(title, description, client_details)
        self.projects[title] = project

    def update_project(self, title, new_status=None):
        if title in self.projects:
            if new_status:
                self.projects[title].update_status(new_status)

    def delete_project(self, title):
        if title in self.projects:
            del self.projects[title]


# Example usage
if __name__ == "__main__":
    freelancer_manager = FreelancerManager()
    project_manager = ProjectManager()

    # Create a freelancer
    freelancer_manager.create_freelancer("Alice", ["Web Development", "Python"], 5, "alice@example.com")

    # Add service to freelancer profile
    freelancer_manager.freelancers["Alice"].add_service("Full Stack Development")

    # Create a project
    project_manager.create_project("Website Redesign", "Redesign the company website", "Client A")

    # Update project status
    project_manager.update_project("Website Redesign", "In Progress")

    # Print freelancer and project details
    print(freelancer_manager.freelancers["Alice"].__dict__)
    print(project_manager.projects["Website Redesign"].__dict__)
