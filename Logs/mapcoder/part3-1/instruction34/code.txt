def first_non_repeating_character(s: str) -> str:
    # Step 2: Initialize an empty dictionary to store character counts
    char_count = {}
    
    # Step 3: Count character occurrences
    for char in s:
        char_count[char] = char_count.get(char, 0) + 1
    
    # Step 4: Find first non-repeating character
    for char in s:
        if char_count[char] == 1:
            return char
    
    # Step 5: Return None if not found
    return None

# Step 7: Testing the function with various test cases
if __name__ == "__main__":
    test_cases = [
        "",  # Edge case: empty string
        "aabbcc",  # All repeating characters
        "abcde",  # All non-repeating characters
        "abacabad",  # Multiple non-repeating characters
        "a",  # Single character
        "aabbc",  # One non-repeating character
        "aabbccddeeffg"  # One non-repeating character at the end
    ]
    
    for test in test_cases:
        result = first_non_repeating_character(test)
        print(f"Input: '{test}' => First non-repeating character: '{result}'")
