class MenuItem:
    def __init__(self, name, price, description):
        self.name = name
        self.price = price
        self.description = description

class Order:
    def __init__(self, customer, items):
        self.customer = customer
        self.items = items
        self.status = 'Pending'
        self.timestamp = None

    def update_status(self, status):
        self.status = status

class Customer:
    def __init__(self, name):
        self.name = name
        self.orders = []

    def view_menu(self, restaurant):
        return restaurant.menu

    def place_order(self, restaurant, items):
        order = Order(self, items)
        restaurant.add_order(order)
        self.orders.append(order)
        restaurant.notify("New order placed by {}".format(self.name))

class Seller:
    def __init__(self, name):
        self.name = name
        self.menu = []

    def add_menu_item(self, item):
        self.menu.append(item)

    def update_menu_item(self, item_name, new_item):
        for i, item in enumerate(self.menu):
            if item.name == item_name:
                self.menu[i] = new_item
                break

    def remove_menu_item(self, item_name):
        self.menu = [item for item in self.menu if item.name != item_name]

    def view_orders(self, restaurant):
        return restaurant.orders

class Driver:
    def __init__(self, name):
        self.name = name

    def update_order_status(self, order, status):
        order.update_status(status)
        order.customer.notify("Order status updated to {}".format(status))

class Restaurant:
    def __init__(self):
        self.menu = []
        self.orders = []

    def add_menu_item(self, item):
        self.menu.append(item)

    def add_order(self, order):
        self.orders.append(order)

    def notify(self, message):
        print(message)

# Example usage
if __name__ == "__main__":
    restaurant = Restaurant()
    seller = Seller("John's Diner")
    customer = Customer("Alice")
    driver = Driver("Bob")

    # Adding menu items
    item1 = MenuItem("Burger", 5.99, "Juicy beef burger")
    item2 = MenuItem("Fries", 2.99, "Crispy golden fries")
    seller.add_menu_item(item1)
    seller.add_menu_item(item2)
    restaurant.add_menu_item(item1)
    restaurant.add_menu_item(item2)

    # Customer places an order
    customer.place_order(restaurant, [item1, item2])

    # Driver updates order status
    driver.update_order_status(restaurant.orders[0], "Out for delivery")
