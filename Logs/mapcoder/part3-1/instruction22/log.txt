    def load_image(self):
        file_path = filedialog.askopenfilename()
        if file_path:
            image = cv2.imread(file_path)
            self.detect_objects(image)

    def start_video_stream(self):
        cap = cv2.VideoCapture(0)
        while True:
            ret, frame = cap.read()
            if not ret:
                break
            self.detect_objects(frame)
            if cv2.waitKey(1) & 0xFF == ord('q'):
                break
        cap.release()
        cv2.destroyAllWindows()

        input_image = cv2.resize(image, (224, 224))  # Adjust size as per model input
        input_image = np.expand_dims(input_image, axis=0) / 255.0

        # Perform object detection
        predictions = self.model.predict(input_image)

        # For example, draw bounding boxes on the image

            # Assuming pred contains bounding box coordinates and class label

            cv2.putText(image, label, (x, y - 10), cv2.FONT_HERSHEY_SIMPLEX, 0.5, (0, 255, 0), 2)
