from flask import Flask, jsonify, request
from flask_socketio import SocketIO, emit
import requests
import threading
import time

app = Flask(__name__)
app.config['SECRET_KEY'] = 'your_secret_key'
socketio = SocketIO(app)

# Replace with your actual API for futures prices
FUTURES_API_URL = 'https://api.example.com/futures/prices'
IP_API_URL = 'https://api.ipify.org?format=json'

# Function to fetch live futures prices
def fetch_prices():
    while True:
        response = requests.get(FUTURES_API_URL)
        if response.status_code == 200:
            prices = response.json()
            socketio.emit('price_update', prices)
        time.sleep(0.2)  # 200ms interval

# Start the price fetching thread
threading.Thread(target=fetch_prices, daemon=True).start()
