import requests
import time
import json
import pandas as pd
from telegram import Bot

class DeribitAPI:
    def __init__(self, client_id, client_secret):
        self.client_id = client_id
        self.client_secret = client_secret
        self.base_url = "https://www.deribit.com/api/v2/"
        self.access_token = self.authenticate()

    def authenticate(self):
        response = requests.post(f"{self.base_url}public/auth", data={
            'client_id': self.client_id,
            'client_secret': self.client_secret,
            'grant_type': 'client_credentials'
        })
        return response.json()['result']['access_token']

    def get_account_info(self):
        headers = {'Authorization': f'Bearer {self.access_token}'}
        response = requests.get(f"{self.base_url}private/get_account_summary", headers=headers)
        return response.json()

    def place_order(self, instrument_name, amount, order_type='limit', price=None):
        headers = {'Authorization': f'Bearer {self.access_token}'}
        data = {
            'instrument_name': instrument_name,
            'amount': amount,
            'type': order_type,
            'price': price
        }
        response = requests.post(f"{self.base_url}private/buy", headers=headers, data=data)
        return response.json()

    def get_market_data(self, instrument_name):
        response = requests.get(f"{self.base_url}public/get_order_book", params={'instrument_name': instrument_name})
        return response.json()
