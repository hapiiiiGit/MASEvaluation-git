import pandas as pd
import pyautogui
import time
import pyttsx3
import requests

# Function to read phone numbers from an Excel file
def read_phone_numbers(file_path):
    try:
        df = pd.read_excel(file_path)
        return df[['Name', 'Phone Number']].dropna().values.tolist()
    except Exception as e:
        print(f"Error reading Excel file: {e}")
        return []

# Function to initiate a call using Microsoft Teams
def initiate_call(phone_number):
    try:
        # Placeholder for Microsoft Teams API call or automation
        # Example: requests.post('TEAMS_API_URL', data={'phone': phone_number})
        print(f"Calling {phone_number} using Microsoft Teams...")
        time.sleep(2)  # Simulate call duration
        print(f"Call to {phone_number} completed.")
    except Exception as e:
        print(f"Error initiating call to {phone_number}: {e}")

# Function for audio playback
def play_audio(message):
    engine = pyttsx3.init()
    engine.say(message)
    engine.runAndWait()

# Main function to automate calling
def automate_calls(file_path):
    phone_numbers = read_phone_numbers(file_path)
    if not phone_numbers:
        print("No valid phone numbers found.")
        return

    for name, phone_number in phone_numbers:
        if validate_phone_number(phone_number):
            play_audio(f"Calling {name} at {phone_number}")
            initiate_call(phone_number)
        else:
            print(f"Invalid phone number for {name}: {phone_number}")

# Function to validate phone numbers (basic validation)
def validate_phone_number(phone_number):
    return isinstance(phone_number, str) and phone_number.isdigit() and len(phone_number) >= 10

# Example usage
if __name__ == "__main__":
    excel_file_path = 'clients.xlsx'  # Path to the Excel file
    automate_calls(excel_file_path)
